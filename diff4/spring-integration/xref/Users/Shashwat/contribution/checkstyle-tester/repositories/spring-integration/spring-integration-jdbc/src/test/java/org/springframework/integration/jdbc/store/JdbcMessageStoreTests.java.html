304 	@Test
305 	public void testExpireMessageGroupOnCreateOnly() throws Exception {
306 		final String groupId = "X";
307 		Message<String> message = MessageBuilder.withPayload("foo").setCorrelationId(groupId).build();
308 		messageStore.addMessagesToGroup(groupId, message);
309 		final CountDownLatch groupRemovalLatch = new CountDownLatch(1);
310 		messageStore.registerMessageGroupExpiryCallback((messageGroupStore, group) -> {
311 			messageGroupStore.removeMessageGroup(group.getGroupId());
312 			groupRemovalLatch.countDown();
313 		});
314 
315 		messageStore.expireMessageGroups(2000);
316 		MessageGroup group = messageStore.getMessageGroup(groupId);
317 		assertThat(group.size()).isEqualTo(1);
318 
319 		messageStore.addMessagesToGroup(groupId, MessageBuilder.withPayload("bar").setCorrelationId(groupId).build());
320 
321 		JdbcTemplate template = new JdbcTemplate(this.dataSource);
322 		template.afterPropertiesSet();
323 
324 		template.update("UPDATE INT_MESSAGE_GROUP set CREATED_DATE=? where GROUP_KEY=? and REGION=?",
325 				(PreparedStatementSetter) ps -> {
326 					ps.setTimestamp(1, new Timestamp(System.currentTimeMillis() - 10000));
327 					ps.setString(2, UUIDConverter.getUUID(groupId).toString());
328 					ps.setString(3, "DEFAULT");
329 				});
330 
331 		messageStore.expireMessageGroups(2000);
332 
333 		group = messageStore.getMessageGroup(groupId);
334 		assertThat(group.size()).isEqualTo(0);
335 		assertThat(groupRemovalLatch.await(10, TimeUnit.SECONDS)).isTrue();
336 	}

479 	@Test
480 	public void testCompletedNotExpiredGroupINT3037() throws Exception {
481 		/*
482 		 * based on the aggregator scenario as follows;
483 		 *
484 		 * send three messages in
485 		 * 1 of 2
486 		 * 2 of 2
487 		 * 2 of 2 (last again)
488 		 *
489 		 * expected behavior is that the LAST message (2 of 2 repeat) should be on the discard channel
490 		 * (discard behavior performed by the AbstractCorrelatingMessageHandler.handleMessageInternal)
491 		 */
492 		final JdbcMessageStore messageStore = new JdbcMessageStore(dataSource);
493 		//init
494 		String groupId = "group";
495 		//build the messages
496 		Message<?> oneOfTwo = MessageBuilder.withPayload("hello")
497 				.setSequenceNumber(1)
498 				.setSequenceSize(2)
499 				.setCorrelationId(groupId)
500 				.build();
501 		Message<?> twoOfTwo = MessageBuilder.withPayload("world")
502 				.setSequenceNumber(2)
503 				.setSequenceSize(2)
504 				.setCorrelationId(groupId)
505 				.build();
506 		//add to the messageStore
507 		messageStore.addMessagesToGroup(groupId, oneOfTwo, twoOfTwo);
508 		//check that 2 messages are there
509 		assertThat(messageStore.getMessageGroupCount() == 1).isTrue();
510 		assertThat(messageStore.getMessageCount() == 2).isTrue();
511 		//retrieve the group (like in the aggregator)
512 		MessageGroup messageGroup = messageStore.getMessageGroup(groupId);
513 		//'complete' the group
514 		messageStore.completeGroup(messageGroup.getGroupId());
515 		//now clear the messages
516 		for (Message<?> message : messageGroup.getMessages()) {
517 			messageStore.removeMessagesFromGroup(groupId, message);
518 		} //end for
519 		//'add' the other message --> emulated by getting the messageGroup
520 		messageGroup = messageStore.getMessageGroup(groupId);
521 		//should be marked 'complete' --> old behavior it would not
522 		assertThat(messageGroup.isComplete()).isTrue();
523 	}
