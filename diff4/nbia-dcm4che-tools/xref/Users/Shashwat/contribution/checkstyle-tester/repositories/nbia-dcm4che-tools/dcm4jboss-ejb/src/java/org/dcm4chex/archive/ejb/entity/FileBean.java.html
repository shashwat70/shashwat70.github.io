293     /**
294      * @ejb.interface-method
295      */
296     public FileDTO getFileDTO() {
297         FileSystemLocal fs = getFileSystem();
298         FileDTO retval = new FileDTO();
299         retval.setPk(getPk().longValue());
300         retval.setRetrieveAET(fs.getRetrieveAET());
301         retval.setFileSystemPk(fs.getPk().longValue());
302         retval.setFileSystemGroupID(fs.getGroupID());
303         retval.setDirectoryPath(fs.getDirectoryPath());
304         retval.setAvailability(fs.getAvailability());
305         retval.setUserInfo(fs.getUserInfo());
306         retval.setFilePath(getFilePath());
307         retval.setFileTsuid(getFileTsuid());
308         retval.setFileSize(getFileSize());
309         retval.setFileMd5(getFileMd5());
310         retval.setFileStatus(getFileStatus());
311         
312         InstanceLocal inst = getInstance();
313         if (inst != null) {
314             retval.setSopInstanceUID(inst.getSopIuid());
315             retval.setSopClassUID(inst.getSopCuid());
316         }
317         return retval;
318     }

375     /**    
376      * @ejb.home-method
377      */
378     public Collection ejbHomeSelectByStatusAndFileSystem(List dirPath, int status, 
379             Timestamp notBefore, Timestamp before, int limit) throws FinderException {
380         StringBuilder jbossQl = new StringBuilder()
381         .append("SELECT OBJECT(f) FROM File AS f WHERE f.fileStatus = ?1")
382         .append(" AND f.createdTime >= ?2 AND f.createdTime < ?3 AND f.fileSystem.directoryPath IN (");
383         Object[] args = new Object[dirPath.size()+4];
384         args[0] = status; args[1] = notBefore; args[2] = before;
385         int idx = 3;
386         for (int i = 1, len = dirPath.size(); i < len; i++) {
387             args[idx++] =dirPath.get(i);
388             jbossQl.append("?").append(idx).append(", ");
389         }
390         args[idx++] = dirPath.get(0);
391         jbossQl.append("?").append(idx).append(")");
392         args[idx++] = limit;
393         jbossQl.append(" LIMIT ?").append(idx);
394         return ejbSelectGeneric(jbossQl.toString(), args);
395     }

408         }
409         args[i] = status;
410         jbossQl.append(") AND f.fileStatus = ?").append(++i);
411         return ejbSelectGenericTime(jbossQl.toString(), args);
412     }
