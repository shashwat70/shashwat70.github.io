1   package com.codepoetics.protonpack;
2   
3   import org.junit.Test;
4   
5   import static org.hamcrest.MatcherAssert.assertThat;
6   import static org.hamcrest.Matchers.equalTo;
7   import static org.hamcrest.Matchers.not;
8   
9   public class SeqTest {
10  
11      @Test
12      public void emptySeqsAreEqual() {
13          assertThat(Seq.empty(), equalTo(Seq.empty()));
14      }
15  
16      @Test
17      public void seqsWithSameContentsAreEqual() {
18          assertThat(Seq.of(1, 2, 3), equalTo(Seq.of(1, 2, 3)));
19      }
20  
21      @Test
22      public void seqsWithDifferentContentsAreNotEqual() {
23          assertThat(Seq.of(1, 2, 4), not(equalTo(Seq.of(1, 2, 3))));
24      }
25  
26      @Test
27      public void seqsHaveAStringRepresentation() {
28          assertThat(Seq.of("Hello", "World").toString(), equalTo("(Hello,World)"));
29      }
30  
31      @Test
32      public void seqsAreAppendable() {
33          assertThat(Seq.of(1, 2, 3).append(Seq.of(4, 5, 6)), equalTo(Seq.of(1, 2, 3, 4, 5, 6)));
34      }
35  
36      @Test
37      public void valuesCanBeConsedOntoAStream() {
38          assertThat(Seq.of("Blankets").cons("In").cons("Pigs"), equalTo(Seq.of("Pigs", "In", "Blankets")));
39      }
40  }
