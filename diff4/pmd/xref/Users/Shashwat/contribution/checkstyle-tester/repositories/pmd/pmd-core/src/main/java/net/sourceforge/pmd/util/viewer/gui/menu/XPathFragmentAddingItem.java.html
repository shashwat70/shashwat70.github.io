5   package net.sourceforge.pmd.util.viewer.gui.menu;
6   
7   import java.awt.event.ActionEvent;
8   import java.awt.event.ActionListener;
9   import javax.swing.JMenuItem;
10  
11  import net.sourceforge.pmd.util.viewer.model.ViewerModel;
12  
13  /**
14   * adds the given path fragment to the XPath expression upon action
15   *
16   * @author Boris Gruschko ( boris at gruschko.org )
17   * @version $Id$
18   */
19  @Deprecated // to be removed with PMD 7.0.0
20  public class XPathFragmentAddingItem extends JMenuItem implements ActionListener {
21      private ViewerModel model;
22      private String fragment;
23  
24      /**
25       * constructs the item
26       *
27       * @param caption
28       *            menu item's caption
29       * @param model
30       *            model to refer to
31       * @param fragment
32       *            XPath expression fragment to be added upon action
33       */
34      public XPathFragmentAddingItem(String caption, ViewerModel model, String fragment) {
35          super(caption);
36          this.model = model;
37          this.fragment = fragment;
38          addActionListener(this);
39      }
40  
41      /**
42       * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
43       */
44      @Override
45      public void actionPerformed(ActionEvent e) {
46          model.appendToXPathExpression(fragment, this);
47      }
48  }
