5   package net.sourceforge.pmd.lang.rule;
6   
7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  public class XPathRuleTest {
11  
12      /**
13       * It's easy to forget the attribute "typeResolution=true" when
14       * defining XPath rules in xml. Therefore we by default enable
15       * typeresolution. For Java rules, type resolution was enabled by
16       * default long time ago.
17       * 
18       * @see <a href="https://github.com/pmd/pmd/issues/2048">#2048 [core] Enable type resolution by default for XPath rules</a>
19       */
20      @Test
21      public void typeResolutionShouldBeEnabledByDefault() {
22          XPathRule rule = new XPathRule();
23          Assert.assertTrue(rule.isTypeResolution());
24  
25          XPathRule rule2 = new XPathRule(".");
26          Assert.assertTrue(rule2.isTypeResolution());
27      }
28  
29  }
