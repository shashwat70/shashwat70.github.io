5   package net.sourceforge.pmd.lang.apex.ast;
6   
7   import java.util.List;
8   
9   import org.junit.Assert;
10  import org.junit.Test;
11  
12  import apex.jorje.semantic.ast.compilation.Compilation;
13  
14  public class ASTNewKeyValueObjectExpressionTest extends ApexParserTestBase {
15  
16      @Test
17      public void testParameterName() {
18          ApexNode<Compilation> node = parse("public class Foo { \n"
19                  + "    public void foo(String newName, String tempID) { \n"
20                  + "        if (Contact.sObjectType.getDescribe().isCreateable() && Contact.sObjectType.getDescribe().isUpdateable()) {\n"
21                  + "            upsert new Contact(FirstName = 'First', LastName = 'Last', Phone = '414-414-4414');\n"
22                  + "        }\n" + "    } \n" + "}");
23  
24          ASTNewKeyValueObjectExpression keyValueExpr = node.getFirstDescendantOfType(ASTNewKeyValueObjectExpression.class);
25          Assert.assertEquals(3, keyValueExpr.getParameterCount());
26  
27          List<ASTLiteralExpression> literals = keyValueExpr.findDescendantsOfType(ASTLiteralExpression.class);
28          Assert.assertEquals(3, literals.size());
29          Assert.assertEquals("FirstName", literals.get(0).getName());
30          Assert.assertEquals("LastName", literals.get(1).getName());
31          Assert.assertEquals("Phone", literals.get(2).getName());
32      }
