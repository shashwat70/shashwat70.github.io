559     /**
560      * &lt;x&gt; &lt;a&gt; &lt;b&gt; &lt;b&gt; &lt;/z&gt; &lt;/a&gt; &lt;/x&gt;
561      * An unmatched closing of 'z' appears randomly in the document. This should
562      * be disregarded and structure of children and parents should not be

632     /**
633      * will sort the AST element in list in alphabetical order and if tag name
634      * is the same it will sort against o1.getBeginColumn() +""+
635      * o1.getBeginLine(). so first criteria is the name, then the second is the
636      * column +""+line string.
637      *
638      * @param elements
639      * @return
640      */
641     private List<ASTElement> sortNodesByName(List<ASTElement> elements) {
642         Collections.sort(elements, new Comparator<ASTElement>() {
643             @Override
644             public int compare(ASTElement o1, ASTElement o2) {
645                 if (o1.getName() == null) {
646                     return Integer.MIN_VALUE;
647                 }
648                 if (o2.getName() == null) {
649                     return Integer.MAX_VALUE;
650                 }
651                 if (o1.getName().equals(o2.getName())) {
652                     String o1Value = o1.getBeginColumn() + "" + o1.getBeginLine();
653                     String o2Value = o2.getBeginColumn() + "" + o2.getBeginLine();
654                     return o1Value.compareTo(o2Value);
655                 }
656                 return o1.getName().compareTo(o2.getName());
657             }
658         });
659         return elements;
660     }
