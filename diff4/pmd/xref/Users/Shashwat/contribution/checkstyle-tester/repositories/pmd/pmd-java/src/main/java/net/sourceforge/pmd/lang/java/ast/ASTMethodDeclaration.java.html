5   package net.sourceforge.pmd.lang.java.ast;
6   
7   import net.sourceforge.pmd.annotation.InternalApi;
8   import net.sourceforge.pmd.lang.ast.Node;
9   import net.sourceforge.pmd.lang.dfa.DFAGraphMethod;
10  
11  
12  /**
13   * Method declaration node.
14   *
15   * <pre>
16   * MethodDeclaration := [ TypeParameters() ] (TypeAnnotation())* ResultType() MethodDeclarator() [ "throws" NameList() ] ( Block() | ";" )
17   * </pre>
18   */
19  public class ASTMethodDeclaration extends AbstractMethodOrConstructorDeclaration implements DFAGraphMethod {
20  
21  
22      @InternalApi
23      @Deprecated
24      public ASTMethodDeclaration(int id) {
25          super(id);
26      }
27  
28      @InternalApi
29      @Deprecated
30      public ASTMethodDeclaration(JavaParser p, int id) {
31          super(p, id);
32      }
33  
34      @Override
35      public Object jjtAccept(JavaParserVisitor visitor, Object data) {
36          return visitor.visit(this, data);
37      }
38  
39      /**
40       * Returns the simple name of the method.
41       *
42       * @deprecated Use {@link #getName()}
43       */
44      @Deprecated
45      public String getMethodName() {
46          return getName();
47      }
48  
49      /** Returns the simple name of the method. */
50      @Override
51      public String getName() {
52          return getFirstChildOfType(ASTMethodDeclarator.class).getImage();
53      }
54  
55  
56      /**
57       * Returns true if this method is explicitly modified by
58       * the {@code public} modifier.
59       */
60      public boolean isSyntacticallyPublic() {
61          return super.isPublic();
62      }
63  
64  
65      /**
66       * Returns true if this method is explicitly modified by
67       * the {@code abstract} modifier.
68       */
69      public boolean isSyntacticallyAbstract() {
70          return super.isAbstract();
71      }
72  
73  
74      /**
75       * Returns true if this method has public visibility.
76       * Non-private interface members are implicitly public,
77       * whether they declare the {@code public} modifier or
78       * not.
79       */
80      @Override
81      public boolean isPublic() {
82          // interface methods are public by default, but could be private since java9
83          return isInterfaceMember() && !isPrivate() || super.isPublic();
84      }
85  
86  
87      /**
88       * Returns true if this method is abstract, so doesn't
89       * declare a body. Interface members are
90       * implicitly abstract, whether they declare the
91       * {@code abstract} modifier or not. Default interface
92       * methods are not abstract though, consistently with the
93       * standard reflection API.
94       */
95      @Override
96      public boolean isAbstract() {
97          return isInterfaceMember() && !isDefault() || super.isAbstract();
98      }
99  
100 
101     /**
102      * Returns true if this method declaration is a member of an interface type.
103      */
104     public boolean isInterfaceMember() {
105         // for a real class/interface the 3rd parent is a ClassOrInterfaceDeclaration,
106         // for anonymous classes, the parent is e.g. a AllocationExpression
107         Node potentialTypeDeclaration = getNthParent(3);
108 
109         return potentialTypeDeclaration instanceof ASTClassOrInterfaceDeclaration
110             && ((ASTClassOrInterfaceDeclaration) potentialTypeDeclaration).isInterface();
111     }
112 
113 
114     /**
115      * Returns true if the result type of this method is {@code void}.
116      */
117     public boolean isVoid() {
118         return getResultType().isVoid();
119     }
120 
121 
122     /**
123      * Returns the result type node of the method.
124      */
125     public ASTResultType getResultType() {
126         return getFirstChildOfType(ASTResultType.class);
127     }
128 
129 
130     /**
131      * Returns the block defined by this method, or
132      * null if the method is abstract.
133      *
134      * @deprecated Use {@link #getBody()}
135      */
136     @Deprecated
137     public ASTBlock getBlock() {
138         return getBody();
139     }
140 
141     /**
142      * Returns the block defined by this method, or
143      * null if the method is abstract.
144      */
145     public ASTBlock getBody() {
146         return getFirstChildOfType(ASTBlock.class);
147     }
148 
149     /**
150      * Returns the number of formal parameters expected by this method
151      * (excluding any receiver parameter). A varargs parameter counts as one.
152      */
153     public int getArity() {
154         return getFormalParameters().getParameterCount();
155     }
156 
157 
158     /**
159      * Returns the exception names listed in the {@code throws} clause
160      * of this method declaration, or null if there are none.
161      */
162     public ASTNameList getThrows() {
163         return getFirstChildOfType(ASTNameList.class);
164     }
165 
166 
167     @Override
168     public MethodLikeKind getKind() {
169         return MethodLikeKind.METHOD;
170     }
171 
172     //@Override // enable this with PMD 7.0.0 - see interface ASTMethodOrConstructorDeclaration
173     public ASTFormalParameters getFormalParameters() {
174         return getFirstChildOfType(ASTMethodDeclarator.class).getFirstChildOfType(ASTFormalParameters.class);
175     }
176 
177 
178     /**
179      * Returns the method declarator. Never null.
180      *
181      * @deprecated Method declarator nodes will be removed with 7.0.0
182      */
183     @Deprecated
184     public ASTMethodDeclarator getMethodDeclarator() {
185         return getFirstChildOfType(ASTMethodDeclarator.class);
186     }
187 }
