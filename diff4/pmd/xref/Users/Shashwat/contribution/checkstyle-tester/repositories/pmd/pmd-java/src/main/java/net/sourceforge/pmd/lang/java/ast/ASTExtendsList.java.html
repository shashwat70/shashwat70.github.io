5   package net.sourceforge.pmd.lang.java.ast;

7   import java.util.Iterator;
8   
9   import net.sourceforge.pmd.annotation.InternalApi;
10  
11  
12  /**
13   * Represents the {@code extends} clause of a class or interface declaration.
14   * If the parent is an interface declaration, then these types are all interface
15   * types. Otherwise, then this list contains exactly one element.
16   *
17   * <pre>
18   *  ExtendsList ::= "extends" (TypeAnnotation)* ClassOrInterfaceType
19   *                ( "," (TypeAnnotation)* ClassOrInterfaceType )*
20   * </pre>
21   */
22  public class ASTExtendsList extends AbstractJavaNode implements Iterable<ASTClassOrInterfaceType> {
23  
24      @InternalApi
25      @Deprecated
26      public ASTExtendsList(int id) {
27          super(id);
28      }
29  
30      @InternalApi
31      @Deprecated
32      public ASTExtendsList(JavaParser p, int id) {
33          super(p, id);
34      }
35  
36      @Override
37      public Object jjtAccept(JavaParserVisitor visitor, Object data) {
38          return visitor.visit(this, data);
39      }
40  
41  
42      @Override    // TODO this doesn't preserve the annotations.
43      public Iterator<ASTClassOrInterfaceType> iterator() {
44          return new NodeChildrenIterator<>(this, ASTClassOrInterfaceType.class);
45      }
46  }
