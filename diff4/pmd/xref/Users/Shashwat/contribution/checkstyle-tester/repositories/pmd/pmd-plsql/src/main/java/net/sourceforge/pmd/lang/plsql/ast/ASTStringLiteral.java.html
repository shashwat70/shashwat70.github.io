6   package net.sourceforge.pmd.lang.plsql.ast;
7   
8   public final class ASTStringLiteral extends net.sourceforge.pmd.lang.plsql.ast.AbstractPLSQLNode {
9   
10  
11      ASTStringLiteral(int id) {
12          super(id);
13      }
14  
15  
16      ASTStringLiteral(PLSQLParser p, int id) {
17          super(p, id);
18      }
19  
20  
21      @Override
22      public Object jjtAccept(PLSQLParserVisitor visitor, Object data) {
23          return visitor.visit(this, data);
24      }
25  
26      /**
27       * Gets the plain string from the string literal.
28       * @return the plain string value from the string literal.
29       */
30      public String getString() {
31          String image = getImage();
32          if (image.charAt(0) == 'N' || image.charAt(0) == 'n') {
33              image = image.substring(1);
34          }
35  
36          if (image.charAt(0) == '\'') {
37              image = image.substring(1, image.length() - 1);
38          } else if (image.charAt(0) == 'Q' || image.charAt(0) == 'q') {
39              image = image.substring("q'x".length(), image.length() - 2);
40          }
41          return image;
42      }
43  }
