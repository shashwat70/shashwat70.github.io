7   import java.util.List;

9   import org.junit.Assert;
10  import org.junit.Test;
11  
12  import apex.jorje.semantic.ast.compilation.Compilation;
13  
14  public class ASTMethodTest extends ApexParserTestBase {
15  
16      @Test
17      public void testConstructorName() {
18          ApexNode<Compilation> node = parse("public class Foo { public Foo() {} public void bar() {} }");
19          Assert.assertSame(ASTUserClass.class, node.getClass());
20          List<ASTMethod> methods = node.findChildrenOfType(ASTMethod.class);
21          Assert.assertEquals("Foo", methods.get(0).getImage()); // constructor
22          Assert.assertEquals("bar", methods.get(1).getImage()); // normal method
23      }
