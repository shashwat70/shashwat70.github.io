5   package net.sourceforge.pmd.lang.java.ast;
6   
7   import static org.junit.Assert.assertEquals;
8   import static org.junit.Assert.assertTrue;
9   
10  import java.util.List;
11  
12  import org.junit.Test;
13  
14  public class ASTImportDeclarationTest extends BaseParserTest {
15  
16      @Test
17      public void testImportOnDemand() {
18          List<ASTImportDeclaration> ops = java.getNodes(ASTImportDeclaration.class, TEST1);
19          assertTrue(ops.get(0).isImportOnDemand());
20      }
21  
22      @Test
23      public void testGetImportedNameNode() {
24          ASTImportDeclaration i = java.getNodes(ASTImportDeclaration.class, TEST2).get(0);
25          assertEquals("foo.bar.Baz", i.getImportedName());
26      }
27  
28      @Test
29      public void testStaticImport() {
30          List<ASTImportDeclaration> ops = java.getNodes(ASTImportDeclaration.class, TEST3);
31          ASTImportDeclaration i = ops.get(0);
32          assertTrue(i.isStatic());
33      }
34  
35      @Test(expected = ParseException.class)
36      public void testStaticImportFailsWithJDK14() {
37          java.parse(TEST3, "1.4");
38      }
39  
40      private static final String TEST1 = "import foo.bar.*;\npublic class Foo {}";
41  
42      private static final String TEST2 = "import foo.bar.Baz;\npublic class Foo {}";
43  
44      private static final String TEST3 = "import static foo.bar.Baz;\npublic class Foo {}";
45  }
