31      @Test
32      public void testOverride() {
33          final StringProperty PROPERTY1_DESCRIPTOR = new StringProperty("property1", "Test property", null, 0f);
34          MockRule rule = new MockRule();
35          rule.definePropertyDescriptor(PROPERTY1_DESCRIPTOR);
36          rule.setLanguage(LanguageRegistry.getLanguage(Dummy2LanguageModule.NAME));
37          rule.setName("name1");
38          rule.setProperty(PROPERTY1_DESCRIPTOR, "value1");
39          rule.setMessage("message1");
40          rule.setDescription("description1");
41          rule.addExample("example1");
42          rule.setExternalInfoUrl("externalInfoUrl1");
43          rule.setPriority(RulePriority.HIGH);
44  
45          final StringProperty PROPERTY2_DESCRIPTOR = new StringProperty("property2", "Test property", null, 0f);
46          RuleReference ruleReference = new RuleReference();
47          ruleReference.setRule(rule);
48          ruleReference.definePropertyDescriptor(PROPERTY2_DESCRIPTOR);
49          ruleReference.setLanguage(LanguageRegistry.getLanguage(DummyLanguageModule.NAME));
50          ruleReference
51                  .setMinimumLanguageVersion(LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.3"));
52          ruleReference
53                  .setMaximumLanguageVersion(LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.7"));
54          ruleReference.setDeprecated(true);
55          ruleReference.setName("name2");
56          ruleReference.setProperty(PROPERTY1_DESCRIPTOR, "value2");
57          ruleReference.setProperty(PROPERTY2_DESCRIPTOR, "value3");
58          ruleReference.setMessage("message2");
59          ruleReference.setDescription("description2");
60          ruleReference.addExample("example2");
61          ruleReference.setExternalInfoUrl("externalInfoUrl2");
62          ruleReference.setPriority(RulePriority.MEDIUM_HIGH);
63  
64          validateOverridenValues(PROPERTY1_DESCRIPTOR, PROPERTY2_DESCRIPTOR, ruleReference);
65      }

103     private void validateOverridenValues(final StringProperty propertyDescriptor1,
104             final StringProperty propertyDescriptor2, RuleReference ruleReference) {
105         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME),
106                 ruleReference.getLanguage());
107         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME),
108                 ruleReference.getOverriddenLanguage());
109 
110         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.3"),
111                 ruleReference.getMinimumLanguageVersion());
112         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.3"),
113                 ruleReference.getOverriddenMinimumLanguageVersion());
114 
115         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.7"),
116                 ruleReference.getMaximumLanguageVersion());
117         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.7"),
118                 ruleReference.getOverriddenMaximumLanguageVersion());
119 
120         assertEquals("Override failed", false, ruleReference.getRule().isDeprecated());
121         assertEquals("Override failed", true, ruleReference.isDeprecated());
122         assertEquals("Override failed", true, ruleReference.isOverriddenDeprecated());
123 
124         assertEquals("Override failed", "name2", ruleReference.getName());
125         assertEquals("Override failed", "name2", ruleReference.getOverriddenName());
126 
127         assertEquals("Override failed", "value2", ruleReference.getProperty(propertyDescriptor1));
128         assertEquals("Override failed", "value3", ruleReference.getProperty(propertyDescriptor2));
129         assertTrue("Override failed", ruleReference.getPropertyDescriptors().contains(propertyDescriptor1));
130         assertTrue("Override failed", ruleReference.getPropertyDescriptors().contains(propertyDescriptor2));
131         assertFalse("Override failed", ruleReference.getOverriddenPropertyDescriptors().contains(propertyDescriptor1));
132         assertTrue("Override failed", ruleReference.getOverriddenPropertyDescriptors().contains(propertyDescriptor2));
133         assertTrue("Override failed",
134                 ruleReference.getPropertiesByPropertyDescriptor().containsKey(propertyDescriptor1));
135         assertTrue("Override failed",
136                 ruleReference.getPropertiesByPropertyDescriptor().containsKey(propertyDescriptor2));
137         assertTrue("Override failed",
138                 ruleReference.getOverriddenPropertiesByPropertyDescriptor().containsKey(propertyDescriptor1));
139         assertTrue("Override failed",
140                 ruleReference.getOverriddenPropertiesByPropertyDescriptor().containsKey(propertyDescriptor2));
141 
142         assertEquals("Override failed", "message2", ruleReference.getMessage());
143         assertEquals("Override failed", "message2", ruleReference.getOverriddenMessage());
144 
145         assertEquals("Override failed", "description2", ruleReference.getDescription());
146         assertEquals("Override failed", "description2", ruleReference.getOverriddenDescription());
147 
148         assertEquals("Override failed", 2, ruleReference.getExamples().size());
149         assertEquals("Override failed", "example1", ruleReference.getExamples().get(0));
150         assertEquals("Override failed", "example2", ruleReference.getExamples().get(1));
151         assertEquals("Override failed", "example2", ruleReference.getOverriddenExamples().get(0));
152 
153         assertEquals("Override failed", "externalInfoUrl2", ruleReference.getExternalInfoUrl());
154         assertEquals("Override failed", "externalInfoUrl2", ruleReference.getOverriddenExternalInfoUrl());
155 
156         assertEquals("Override failed", RulePriority.MEDIUM_HIGH, ruleReference.getPriority());
157         assertEquals("Override failed", RulePriority.MEDIUM_HIGH, ruleReference.getOverriddenPriority());
158     }

160     @Test
161     public void testNotOverride() {
162         final StringProperty PROPERTY1_DESCRIPTOR = new StringProperty("property1", "Test property", null, 0f);
163         MockRule rule = new MockRule();
164         rule.definePropertyDescriptor(PROPERTY1_DESCRIPTOR);
165         rule.setLanguage(LanguageRegistry.getLanguage(DummyLanguageModule.NAME));
166         rule.setMinimumLanguageVersion(LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.3"));
167         rule.setMaximumLanguageVersion(LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.7"));
168         rule.setName("name1");
169         rule.setProperty(PROPERTY1_DESCRIPTOR, "value1");
170         rule.setMessage("message1");
171         rule.setDescription("description1");
172         rule.addExample("example1");
173         rule.setExternalInfoUrl("externalInfoUrl1");
174         rule.setPriority(RulePriority.HIGH);
175 
176         RuleReference ruleReference = new RuleReference();
177         ruleReference.setRule(rule);
178         ruleReference.setLanguage(LanguageRegistry.getLanguage(DummyLanguageModule.NAME));
179         ruleReference
180                 .setMinimumLanguageVersion(LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.3"));
181         ruleReference
182                 .setMaximumLanguageVersion(LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.7"));
183         ruleReference.setDeprecated(false);
184         ruleReference.setName("name1");
185         ruleReference.setProperty(PROPERTY1_DESCRIPTOR, "value1");
186         ruleReference.setMessage("message1");
187         ruleReference.setDescription("description1");
188         ruleReference.addExample("example1");
189         ruleReference.setExternalInfoUrl("externalInfoUrl1");
190         ruleReference.setPriority(RulePriority.HIGH);
191 
192         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME),
193                 ruleReference.getLanguage());
194         assertNull("Override failed", ruleReference.getOverriddenLanguage());
195 
196         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.3"),
197                 ruleReference.getMinimumLanguageVersion());
198         assertNull("Override failed", ruleReference.getOverriddenMinimumLanguageVersion());
199 
200         assertEquals("Override failed", LanguageRegistry.getLanguage(DummyLanguageModule.NAME).getVersion("1.7"),

203 
204         assertEquals("Override failed", false, ruleReference.isDeprecated());
205         assertNull("Override failed", ruleReference.isOverriddenDeprecated());
206 
207         assertEquals("Override failed", "name1", ruleReference.getName());
208         assertNull("Override failed", ruleReference.getOverriddenName());
209 
210         assertEquals("Override failed", "value1", ruleReference.getProperty(PROPERTY1_DESCRIPTOR));
211 
212         assertEquals("Override failed", "message1", ruleReference.getMessage());
213         assertNull("Override failed", ruleReference.getOverriddenMessage());
214 
215         assertEquals("Override failed", "description1", ruleReference.getDescription());
216         assertNull("Override failed", ruleReference.getOverriddenDescription());
217 
218         assertEquals("Override failed", 1, ruleReference.getExamples().size());
219         assertEquals("Override failed", "example1", ruleReference.getExamples().get(0));
220         assertNull("Override failed", ruleReference.getOverriddenExamples());
221 
222         assertEquals("Override failed", "externalInfoUrl1", ruleReference.getExternalInfoUrl());
223         assertNull("Override failed", ruleReference.getOverriddenExternalInfoUrl());
224

226         assertNull("Override failed", ruleReference.getOverriddenPriority());
