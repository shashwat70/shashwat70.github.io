5   package net.sourceforge.pmd.lang.rule.xpath;
6   
7   import java.util.Collections;
8   import java.util.List;
9   
10  import org.junit.Assert;
11  import org.junit.Test;
12  
13  import net.sourceforge.pmd.RuleContext;
14  import net.sourceforge.pmd.lang.ast.Node;
15  import net.sourceforge.pmd.properties.PropertyDescriptor;
16  
17  public class SaxonXPathRuleQueryTest {
18  
19      @Test
20      public void testListAttribute() {
21          DummyNodeWithListAndEnum dummy = new DummyNodeWithListAndEnum(1);
22  
23          assertQuery(1, "//dummyNode[@List = \"A\"]", dummy);
24          assertQuery(1, "//dummyNode[@List = \"B\"]", dummy);
25          assertQuery(0, "//dummyNode[@List = \"C\"]", dummy);
26          assertQuery(1, "//dummyNode[@Enum = \"FOO\"]", dummy);
27          assertQuery(0, "//dummyNode[@Enum = \"BAR\"]", dummy);
28          assertQuery(1, "//dummyNode[@EnumList = \"FOO\"]", dummy);
29          assertQuery(1, "//dummyNode[@EnumList = \"BAR\"]", dummy);
30          assertQuery(1, "//dummyNode[@EnumList = (\"FOO\", \"BAR\")]", dummy);
31          assertQuery(0, "//dummyNode[@EmptyList = (\"A\")]", dummy);
32      }
33  
34      private static void assertQuery(int resultSize, String xpath, Node node) {
35          SaxonXPathRuleQuery query = createQuery(xpath);
36          List<Node> result = query.evaluate(node, new RuleContext());
37          Assert.assertEquals(resultSize, result.size());
38      }
39  
40      private static SaxonXPathRuleQuery createQuery(String xpath) {
41          SaxonXPathRuleQuery query = new SaxonXPathRuleQuery();
42          query.setVersion("2.0");
43          query.setProperties(Collections.<PropertyDescriptor<?>, Object>emptyMap());
44          query.setXPath(xpath);
45          return query;
46      }
47  }
