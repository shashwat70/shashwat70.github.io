5   package net.sourceforge.pmd.lang.java.ast;
6   
7   import java.util.Arrays;
8   import java.util.List;
9   
10  import net.sourceforge.pmd.Rule;
11  import net.sourceforge.pmd.annotation.InternalApi;
12  
13  
14  /**
15   * Represents an annotation. This node has three possible children,
16   * that correspond to specific syntactic variants.
17   *
18   * <pre>
19   *
20   * Annotation ::= {@linkplain ASTNormalAnnotation NormalAnnotation}
21   *              | {@linkplain ASTSingleMemberAnnotation SingleMemberAnnotation}
22   *              | {@linkplain ASTMarkerAnnotation MarkerAnnotation}
23   *
24   * </pre>
25   */
26  public class ASTAnnotation extends AbstractJavaTypeNode {
27  
28      private static final List<String> UNUSED_RULES
29          = Arrays.asList("UnusedPrivateField", "UnusedLocalVariable", "UnusedPrivateMethod", "UnusedFormalParameter");
30  
31      private static final List<String> SERIAL_RULES = Arrays.asList("BeanMembersShouldSerialize", "MissingSerialVersionUID");
32  
33      @InternalApi
34      @Deprecated
35      public ASTAnnotation(int id) {
36          super(id);
37      }
38  
39      @InternalApi
40      @Deprecated
41      public ASTAnnotation(JavaParser p, int id) {
42          super(p, id);
43      }
44  
45  
46      /**
47       * Returns the name of the annotation as it is used,
48       * eg {@code java.lang.Override} or {@code Override}.
49       */
50      public String getAnnotationName() {
51          return getChild(0).getChild(0).getImage();
52      }
53  
54      // @formatter:off
55      /**
56       * Returns true if this annotation suppresses the given rule.
57       * The suppression annotation is {@link SuppressWarnings}.
58       * This method returns true if this annotation is a SuppressWarnings,
59       * and if the set of suppressed warnings ({@link SuppressWarnings#value()})
60       * contains at least one of those:
61       * <ul>
62       *     <li>"PMD" (suppresses all rules);
63       *     <li>"PMD.rulename", where rulename is the name of the given rule;
64       *     <li>"all" (conventional value to suppress all warnings).
65       * </ul>
66       *
67       * <p>Additionnally, the following values suppress a specific set of rules:
68       * <ul>
69       *     <li>{@code "unused"}: suppresses rules like UnusedLocalVariable or UnusedPrivateField;
70       *     <li>{@code "serial"}: suppresses BeanMembersShouldSerialize and MissingSerialVersionUID;
71       * </ul>
72       *
73       * @param rule The rule for which to check for suppression
74       *
75       * @return True if this annotation suppresses the given rule
76       */
77      // @formatter:on
78      public boolean suppresses(Rule rule) {
79  
80          if (getChild(0) instanceof ASTMarkerAnnotation) {
81              return false;
82          }
83  
84          // if (SuppressWarnings.class.equals(getType())) { // typeres is not always on
85          if (isSuppressWarnings()) {
86              for (ASTLiteral element : findDescendantsOfType(ASTLiteral.class)) {
87                  if (element.hasImageEqualTo("\"PMD\"") || element.hasImageEqualTo("\"PMD." + rule.getName() + "\"")
88                      // Check for standard annotations values
89                      || element.hasImageEqualTo("\"all\"")
90                      || element.hasImageEqualTo("\"serial\"") && SERIAL_RULES.contains(rule.getName())
91                      || element.hasImageEqualTo("\"unused\"") && UNUSED_RULES.contains(rule.getName())) {
92                      return true;
93                  }
94              }
95          }
96  
97          return false;
98      }
99  
100 
101     private boolean isSuppressWarnings() {
102         return "SuppressWarnings".equals(getAnnotationName())
103             || "java.lang.SuppressWarnings".equals(getAnnotationName());
104     }
105 
106     @Override
107     public Object jjtAccept(JavaParserVisitor visitor, Object data) {
108         return visitor.visit(this, data);
109     }
110 }
