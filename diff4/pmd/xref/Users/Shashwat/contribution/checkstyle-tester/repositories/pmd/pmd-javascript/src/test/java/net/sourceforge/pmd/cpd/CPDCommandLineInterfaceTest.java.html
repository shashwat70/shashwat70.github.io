5   package net.sourceforge.pmd.cpd;
6   
7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  import net.sourceforge.pmd.cli.BaseCPDCLITest;
11  
12  public class CPDCommandLineInterfaceTest extends BaseCPDCLITest {
13      @Test
14      public void shouldFindDuplicatesWithDifferentFileExtensions() {
15          runCPD("--minimum-tokens", "5", "--language", "js", "--files",
16                  "src/test/resources/net/sourceforge/pmd/cpd/ts/File1.ts",
17                  "src/test/resources/net/sourceforge/pmd/cpd/ts/File2.ts");
18  
19          String out = getOutput();
20          Assert.assertTrue(out.contains("Found a 9 line (30 tokens) duplication in the following files"));
21      }
22  
23      @Test
24      public void shouldFindNoDuplicatesWithDifferentFileExtensions() {
25          runCPD("--minimum-tokens", "5", "--language", "js", "--files",
26                  "src/test/resources/net/sourceforge/pmd/cpd/ts/");
27  
28          String out = getOutput();
29          Assert.assertTrue(out.trim().isEmpty());
30      }
31  }
