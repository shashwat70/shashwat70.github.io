5   package net.sourceforge.pmd.lang.modelica.ast;
6   
7   import net.sourceforge.pmd.lang.modelica.resolver.ModelicaClassSpecialization;
8   
9   public class ASTClassDefinition extends AbstractModelicaNode {
10      private ASTClassPrefixes prefixes;
11      private ModelicaClassSpecialization specialization;
12      private ModelicaClassSpecifierNode specifier;
13  
14      ASTClassDefinition(int id) {
15          super(id);
16      }
17  
18      ASTClassDefinition(ModelicaParser p, int id) {
19          super(p, id);
20      }
21  
22      public boolean isPartial() {
23          return prefixes.getFirstChildOfType(ASTPartialClause.class) != null;
24      }
25  
26      public boolean isEncapsulated() {
27          return getFirstChildOfType(ASTEncapsulatedClause.class) != null;
28      }
29  
30      public ModelicaClassSpecialization getSpecialization() {
31          return specialization;
32      }
33  
34      public ModelicaClassSpecifierNode getClassSpecifier() {
35          return specifier;
36      }
37  
38      private void checkSpecialization(Class<? extends ModelicaNode> clauseClass, ModelicaClassSpecialization restriction) {
39          if (prefixes.getFirstChildOfType(clauseClass) != null) {
40              assert specialization == null;
41              specialization = restriction;
42          }
43      }
44  
45      private void detectSpecialization() {
46          checkSpecialization(ASTClassClause.class, ModelicaClassSpecialization.CLASS);
47          checkSpecialization(ASTModelClause.class, ModelicaClassSpecialization.MODEL);
48          checkSpecialization(ASTRecordClause.class, ModelicaClassSpecialization.RECORD);
49          checkSpecialization(ASTOperatorRecordClause.class, ModelicaClassSpecialization.OPERATOR_RECORD);
50          checkSpecialization(ASTBlockClause.class, ModelicaClassSpecialization.BLOCK);
51          checkSpecialization(ASTConnectorClause.class, ModelicaClassSpecialization.CONNECTOR);
52          checkSpecialization(ASTExpandableConnectorClause.class, ModelicaClassSpecialization.EXPANDABLE_CONNECTOR);
53          checkSpecialization(ASTTypeClause.class, ModelicaClassSpecialization.TYPE);
54          checkSpecialization(ASTPackageClause.class, ModelicaClassSpecialization.PACKAGE);
55          checkSpecialization(ASTOperatorClause.class, ModelicaClassSpecialization.OPERATOR);
56          ASTFunctionClause functionOrNull = prefixes.getFirstChildOfType(ASTFunctionClause.class);
57          if (functionOrNull != null) {
58              boolean isPure = functionOrNull.getFirstChildOfType(ASTPureClause.class) != null;
59              boolean isOperator = functionOrNull.getFirstChildOfType(ASTOperatorClause.class) != null;
60              assert specialization == null;
61              specialization = ModelicaClassSpecialization.getFunctionSpecialization(isPure, isOperator);
62          }
63          assert specialization != null;
64      }
65  
66      @Override
67      public void jjtClose() {
68          super.jjtClose();
69          prefixes = getFirstChildOfType(ASTClassPrefixes.class);
70          specifier = getFirstChildOfType(ASTClassSpecifier.class).getFirstChildOfType(ModelicaClassSpecifierNode.class);
71          detectSpecialization();
72      }
73  
74      @Override
75      public Object jjtAccept(ModelicaParserVisitor visitor, Object data) {
76          return visitor.visit(this, data);
77      }
78  }
