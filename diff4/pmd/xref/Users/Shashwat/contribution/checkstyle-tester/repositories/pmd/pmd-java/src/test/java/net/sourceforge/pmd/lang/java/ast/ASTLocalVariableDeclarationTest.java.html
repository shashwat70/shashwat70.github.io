7   import static org.junit.Assert.assertEquals;
8   
9   import org.junit.Test;
10  
11  public class ASTLocalVariableDeclarationTest extends BaseParserTest {
12  
13      @Test
14      public void testSingleDimArray() {
15          ASTLocalVariableDeclaration node = java.parse(TEST1).findDescendantsOfType(ASTLocalVariableDeclaration.class).get(0);
16          assertEquals(1, node.getArrayDepth());
17      }
18  
19      @Test
20      public void testMultDimArray() {
21          ASTLocalVariableDeclaration node = java.parse(TEST2).findDescendantsOfType(ASTLocalVariableDeclaration.class).get(0);
22          assertEquals(2, node.getArrayDepth());
23      }
24  
25      @Test
26      public void testMultDimArraySplitBraces() {
27          ASTLocalVariableDeclaration node = java.parse(TEST3).findDescendantsOfType(ASTLocalVariableDeclaration.class).get(0);
28          assertEquals(3, node.getArrayDepth());
29      }
30  
31      private static final String TEST1 = "class Foo {\n void bar() {int x[] = null;}\n}";
32  
33      private static final String TEST2 = "class Foo {\n void bar() {int x[][] = null;}\n}";
34  
35      private static final String TEST3 = "class Foo {\n void bar() {int[] x[][] = null;}\n}";
36  }
