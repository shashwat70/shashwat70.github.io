5   package net.sourceforge.pmd.lang.java.ast;
6   
7   import java.util.List;
8   
9   import net.sourceforge.pmd.annotation.InternalApi;
10  
11  
12  /**
13   * Represents a type bound on a {@linkplain ASTTypeParameter type parameter}.
14   * Type bounds specify the type of the type variable to which they apply as
15   * an <a href="https://docs.oracle.com/javase/specs/jls/se9/html/jls-4.html#jls-4.9">intersection type</a>.
16   * The first bound type is a class or interface type, while the additional
17   * bounds are necessarily interface types.
18   *
19   * <pre>
20   *
21   * TypeBound ::= "extends" {@linkplain ASTAnnotation Annotation}* {@linkplain ASTClassOrInterfaceType ClassOrInterfaceType} ( "&" {@linkplain ASTAnnotation Annotation}* {@linkplain ASTClassOrInterfaceType ClassOrInterfaceType} )*
22   *
23   * </pre>
24   */
25  public class ASTTypeBound extends AbstractJavaTypeNode {
26  
27      @InternalApi
28      @Deprecated
29      public ASTTypeBound(int id) {
30          super(id);
31      }
32  
33  
34      @InternalApi
35      @Deprecated
36      public ASTTypeBound(JavaParser p, int id) {
37          super(p, id);
38      }
39  
40  
41      /**
42       * Returns a list with the type bounds of this node.
43       * The returned list has at least one element.
44       */
45      public List<ASTClassOrInterfaceType> getBoundTypeNodes() {
46          return findChildrenOfType(ASTClassOrInterfaceType.class);
47      }
48  
49  
50      @Override
51      public Object jjtAccept(JavaParserVisitor visitor, Object data) {
52          return visitor.visit(this, data);
53      }
54  }
