18  /**
19   * Root interface for all AST nodes. This interface provides only the API
20   * shared by all AST implementations in PMD language modules. This includes for now:
21   * <ul>
22   * <li>Tree traversal methods, like {@link #getParent()} and {@link #getFirstChildOfType(Class)}
23   * <li>The API used to describe nodes in a form understandable by XPath expressions:
24   * {@link #getXPathNodeName()},  {@link #getXPathAttributesIterator()}
25   * <li>Location metadata: eg {@link #getBeginLine()}, {@link #getBeginColumn()}
26   * </ul>
27   *
28   * <p>Every language implementation must publish a sub-interface of Node
29   * which serves as a supertype for all nodes of that language (e.g.
30   * pmd-java provides JavaNode, pmd-apex provides ApexNode, etc.). It is
31   * assumed in many places that the {@link #getChild(int)} and {@link #getParent()}
32   * method return an instance of this sub-interface. For example,
33   * no JSP node should have a Java node as its child. Embedding nodes from
34   * different languages will not be done via these methods, and conforming
35   * implementations should ensure that every node returned by these methods
36   * are indeed of the same type. Possibly, a type parameter will be added to
37   * the Node interface in 7.0.0 to enforce it at compile-time.
38   *
39   * <p>A number of methods are deprecated and will be removed in 7.0.0.
40   * Most of them are implementation details that clutter this API and
41   * make implementation more difficult. Some methods prefixed with {@code jjt}
42   * have a more conventional counterpart (e.g. {@link #jjtGetParent()} and
43   * {@link #getParent()}) that should be preferred.
44   */
45  public interface Node {
46  
47      /**
48       * This method is called after the node has been made the current node. It
49       * indicates that child nodes can now be added to it.
50       *
51       * @deprecated This is JJTree-specific and will be removed from this interface
52       */
53      @Deprecated
54      void jjtOpen();
55  
56  
57      /**
58       * This method is called after all the child nodes have been added.
59       *
60       * @deprecated This is JJTree-specific and will be removed from this interface
61       */
62      @Deprecated
63      void jjtClose();
64  
65  
66      /**
67       * Sets the parent of this node.
68       *
69       * @param parent The parent
70       *
71       * @deprecated This is JJTree-specific and will be removed from this interface
72       */
73      @Deprecated
74      void jjtSetParent(Node parent);
75  
76  
77      /**
78       * Returns the parent of this node.
79       *
80       * @return The parent of the node
81       *
82       * @deprecated Use {@link #getParent()}
83       */
84      @Deprecated
85      Node jjtGetParent();
86  
87  
88      /**
89       * This method tells the node to add its argument to the node's list of
90       * children.
91       *
92       * @param child The child to add
93       * @param index The index to which the child will be added
94       *
95       * @deprecated This is JJTree-specific and will be removed from this interface
96       */
97      @Deprecated
98      void jjtAddChild(Node child, int index);
99  
100     /**
101      * Sets the index of this node from the perspective of its parent. This
102      * means: this.getParent().getChild(index) == this.
103      *
104      * @param index
105      *            the child index
106      *
107      * @deprecated This is JJTree-specific and will be removed from this interface
108      */
109     @Deprecated
110     void jjtSetChildIndex(int index);
111 
112 
113     /**
114      * Gets the index of this node in the children of its parent.
115      *
116      * @return The index of the node
117      *
118      * @deprecated Use {@link #getIndexInParent()}
119      */
120     @Deprecated
121     int jjtGetChildIndex();
122 
123 
124     /**
125      * This method returns a child node. The children are numbered from zero,
126      * left to right.
127      *
128      * @param index
129      *            the child index. Must be nonnegative and less than
130      *            {@link #jjtGetNumChildren}.
131      *
132      * @deprecated Use {@link #getChild(int)}
133      */
134     @Deprecated
135     Node jjtGetChild(int index);
136 
137 
138     /**
139      * Returns the number of children the node has.
140      *
141      * @deprecated Use {@link #getNumChildren()}
142      */
143     @Deprecated
144     int jjtGetNumChildren();
145 
146 
147     /**
148      * @deprecated This is JJTree-specific and will be removed from this interface.
149      */
150     @Deprecated
151     int jjtGetId();
152 
153 
154     /**
155      * Returns a string token, usually filled-in by the parser, which describes some textual
156      * characteristic of this node. This is usually an identifier, but you should check that
157      * using the Designer. On most nodes though, this method returns {@code null}.
158      */
159     String getImage();
160 
161 
162     /**
163      * @deprecated This is internal API, the image should never be set by developers.
164      */
165     @InternalApi
166     @Deprecated
167     void setImage(String image);
168 
169     /**
170      * Returns true if this node's image is equal to the given string.
171      *
172      * @param image The image to check
173      */
174     boolean hasImageEqualTo(String image);
175 
176     int getBeginLine();
177 
178     int getBeginColumn();
179 
180     int getEndLine();
181 
182     // FIXME should not be inclusive
183     int getEndColumn();
184 
185 
186     /**
187      * @deprecated This is Java-specific and will be removed from this interface
188      */
189     @Deprecated
190     DataFlowNode getDataFlowNode();
191 
192 
193     /**
194      * @deprecated This is Java-specific and will be removed from this interface
195      */
196     @Deprecated
197     void setDataFlowNode(DataFlowNode dataFlowNode);
198 
199 
200     /**
201      * Returns true if this node is considered a boundary by traversal methods.
202      * Traversal methods such as {@link #getFirstDescendantOfType(Class)} don't
203      * look past such boundaries by default, which is usually the expected thing
204      * to do. For example, in Java, lambdas and nested classes are considered
205      * find boundaries.
206      *
207      * <p>Note: This attribute is deprecated for XPath queries. It is not useful
208      * for XPath queries and will be removed with PMD 7.0.0.
209      */
210     @DeprecatedAttribute
211     boolean isFindBoundary();
212 
213 
214     /**
215      * Returns the n-th parent or null if there are less than {@code n} ancestors.
216      *
217      * @param n how many ancestors to iterate over.
218      *
219      * @return the n-th parent or null.
220      *
221      * @throws IllegalArgumentException if {@code n} is negative or zero.
222      */
223     Node getNthParent(int n);
224 
225 
226     /**
227      * Traverses up the tree to find the first parent instance of type
228      * parentType or one of its subclasses.
229      *
230      * @param parentType Class literal of the type you want to find
231      * @param <T>        The type you want to find
232      *
233      * @return Node of type parentType. Returns null if none found.
234      */
235     <T> T getFirstParentOfType(Class<T> parentType);
236 
237 
238     /**
239      * Traverses up the tree to find all of the parent instances of type
240      * parentType or one of its subclasses. The nodes are ordered
241      * deepest-first.
242      *
243      * @param parentType Class literal of the type you want to find
244      * @param <T>        The type you want to find
245      *
246      * @return List of parentType instances found.
247      */
248     <T> List<T> getParentsOfType(Class<T> parentType);
249 
250 
251     /**
252      * Gets the first parent that's an instance of any of the given types.
253      *
254      * @param parentTypes Types to look for
255      * @param <T>         Most specific common type of the parameters
256      *
257      * @return The first parent with a matching type. Returns null if there
258      * is no such parent
259      */
260     <T> T getFirstParentOfAnyType(Class<? extends T>... parentTypes);
261 
262     /**
263      * Traverses the children to find all the instances of type childType or
264      * one of its subclasses.
265      *
266      * @see #findDescendantsOfType(Class) if traversal of the entire tree is
267      *      needed.
268      *
269      * @param childType
270      *            class which you want to find.
271      * @return List of all children of type childType. Returns an empty list if
272      *         none found.
273      */
274     <T> List<T> findChildrenOfType(Class<T> childType);
275 
276     /**
277      * Traverses down the tree to find all the descendant instances of type
278      * descendantType without crossing find boundaries.
279      *
280      * @param targetType
281      *            class which you want to find.
282      * @return List of all children of type targetType. Returns an empty list if
283      *         none found.
284      */
285     <T> List<T> findDescendantsOfType(Class<T> targetType);
286 
287     /**
288      * Traverses down the tree to find all the descendant instances of type
289      * descendantType.
290      *
291      * @param targetType
292      *            class which you want to find.
293      * @param results
294      *            list to store the matching descendants
295      * @param crossFindBoundaries
296      *            if <code>false</code>, recursion stops for nodes for which
297      *            {@link #isFindBoundary()} is <code>true</code>
298      * @deprecated Use {@link #findDescendantsOfType(Class, boolean)} instead, which
299      * returns a result list.
300      */
301     @Deprecated
302     <T> void findDescendantsOfType(Class<T> targetType, List<T> results, boolean crossFindBoundaries);
303 
304     /**
305      * Traverses down the tree to find all the descendant instances of type
306      * descendantType.
307      *
308      * @param targetType
309      *            class which you want to find.
310      * @param crossFindBoundaries
311      *            if <code>false</code>, recursion stops for nodes for which
312      *            {@link #isFindBoundary()} is <code>true</code>
313      * @return List of all matching descendants
314      */
315     <T> List<T> findDescendantsOfType(Class<T> targetType, boolean crossFindBoundaries);
316 
317     /**
318      * Traverses the children to find the first instance of type childType.
319      *
320      * @see #getFirstDescendantOfType(Class) if traversal of the entire tree is
321      *      needed.
322      *
323      * @param childType
324      *            class which you want to find.
325      * @return Node of type childType. Returns <code>null</code> if none found.
326      */
327     <T> T getFirstChildOfType(Class<T> childType);
328 
329     /**
330      * Traverses down the tree to find the first descendant instance of type
331      * descendantType without crossing find boundaries.
332      *
333      * @param descendantType
334      *            class which you want to find.
335      * @return Node of type descendantType. Returns <code>null</code> if none
336      *         found.
337      */
338     <T> T getFirstDescendantOfType(Class<T> descendantType);
339 
340     /**
341      * Finds if this node contains a descendant of the given type without crossing find boundaries.
342      *
343      * @param type
344      *            the node type to search
345      * @return <code>true</code> if there is at least one descendant of the
346      *         given type
347      */
348     <T> boolean hasDescendantOfType(Class<T> type);
349 
350     /**
351      * Returns all the nodes matching the xpath expression.
352      *
353      * @param xpathString
354      *            the expression to check
355      * @return List of all matching nodes. Returns an empty list if none found.
356      * @throws JaxenException if the xpath is incorrect or fails altogether
357      *
358      * @deprecated This is very inefficient and should not be used in new code. PMD 7.0.0 will remove
359      *             support for this method.
360      */
361     @Deprecated
362     List<? extends Node> findChildNodesWithXPath(String xpathString) throws JaxenException;
363 
364     /**
365      * Checks whether at least one descendant matches the xpath expression.
366      *
367      * @param xpathString
368      *            the expression to check
369      * @return true if there is a match
370      *
371      * @deprecated This is very inefficient and should not be used in new code. PMD 7.0.0 will remove
372      *             support for this method.
373      */
374     @Deprecated
375     boolean hasDescendantMatchingXPath(String xpathString);
376 
377     /**
378      * Get a DOM Document which contains Elements and Attributes representative
379      * of this Node and it's children. Essentially a DOM tree representation of
380      * the Node AST, thereby allowing tools which can operate upon DOM to also
381      * indirectly operate on the AST.
382      */
383     Document getAsDocument();
384 
385     /**
386      * Get the user data associated with this node. By default there is no data,
387      * unless it has been set via {@link #setUserData(Object)}.
388      *
389      * @return The user data set on this node.
390      */
391     Object getUserData();
392 
393     /**
394      * Set the user data associated with this node.
395      *
396      * <p>PMD itself will never set user data onto a node. Nor should any Rule
397      * implementation, as the AST nodes are shared between concurrently
398      * executing Rules (i.e. it is <strong>not</strong> thread-safe).
399      *
400      * <p>This API is most useful for external applications looking to leverage
401      * PMD's robust support for AST structures, in which case application
402      * specific annotations on the AST nodes can be quite useful.
403      *
404      * @param userData
405      *            The data to set on this node.
406      */
407     void setUserData(Object userData);
408 
409     /**
410      * Remove the current node from its parent.
411      *
412      * @deprecated This is internal API and will be removed from this interface with 7.0.0
413      */
414     @Deprecated
415     @InternalApi
416     void remove();
417 
418     /**
419      * This method tells the node to remove the child node at the given index from the node's list of
420      * children, if any; if not, no changes are done.
421      *
422      * @param childIndex
423      *          The index of the child to be removed
424      *
425      * @deprecated This is internal API and will be removed from this interface with 7.0.0
426      */
427     @Deprecated
428     @InternalApi
429     void removeChildAtIndex(int childIndex);
430 
431 
432     /**
433      * Returns the parent of this node, or null if this is the {@linkplain RootNode root}
434      * of the tree.
435      *
436      * <p>This method should be preferred to {@link #jjtGetParent()}.
437      *
438      * @return The parent of this node
439      */
440     Node getParent();
441 
442     /**
443      * Returns the child of this node at the given index.
444      *
445      * @throws IndexOutOfBoundsException if the index is negative or greater than {@link #getNumChildren()}.
446      */
447     Node getChild(int index);
448 
449 
450     /**
451      * Returns the number of children of this node.
452      */
453     int getNumChildren();
454 
455     /**
456      * Returns the index of this node in its parent's children. If this
457      * node is a {@linkplain RootNode root node}, returns -1.
458      *
459      * <p>This method replaces {@link #jjtGetChildIndex()}, whose name was
460      * JJTree-specific.
461      *
462      * @return The index of this node in its parent's children
463      */
464     int getIndexInParent();
465 
466     /**
467      * Gets the name of the node that is used to match it with XPath queries.
468      *
469      * @return The XPath node name
470      */
471     String getXPathNodeName();
472 
473 
474     /**
475      * Returns an iterator enumerating all the attributes that are available
476      * from XPath for this node.
477      *
478      * @return An attribute iterator for this node
479      */
480     Iterator<Attribute> getXPathAttributesIterator();
481 
482     /**
483      * Returns an iterable enumerating the children of this node.
484      * Use it with a foreach loop:
485      * <pre>{@code
486      *      for (Node child : node.children()) {
487      *          // process child
488      *      }
489      * }</pre>
490      *
491      * <p>This method's return type will be changed to NodeStream
492      * in PMD 7, which is a more powerful kind of iterable. The
493      * change will be source compatible.
494      *
495      * @return A new iterable for the children of this node
496      */
497     Iterable<? extends Node> children();
