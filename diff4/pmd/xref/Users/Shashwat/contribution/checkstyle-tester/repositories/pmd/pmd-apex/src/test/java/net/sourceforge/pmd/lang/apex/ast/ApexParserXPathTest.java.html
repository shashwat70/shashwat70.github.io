7   import java.nio.charset.StandardCharsets;
8   import java.util.List;
9   
10  import org.apache.commons.io.IOUtils;
11  import org.junit.Assert;
12  import org.junit.Test;
13  
14  import net.sourceforge.pmd.lang.ast.Node;
15  
16  import apex.jorje.semantic.ast.compilation.Compilation;
17  
18  public class ApexParserXPathTest extends ApexParserTestBase {
19  
20      @Test
21      public void testBooleanExpressions() throws Exception {
22          ApexNode<Compilation> node = parse(IOUtils.toString(ApexParserXPathTest.class.getResourceAsStream("BooleanExpressions.cls"),
23                  StandardCharsets.UTF_8));
24          List<ASTBooleanExpression> booleanExpressions = node.findDescendantsOfType(ASTBooleanExpression.class);
25          Assert.assertEquals(2, booleanExpressions.size());
26          Assert.assertEquals("&&", booleanExpressions.get(0).getOperator().toString());
27          Assert.assertEquals("!=", booleanExpressions.get(1).getOperator().toString());
28  
29          List<? extends Node> xpathResult = node.findChildNodesWithXPath("//BooleanExpression[@Operator='&&']");
30          Assert.assertEquals(1, xpathResult.size());
31          Assert.assertSame(booleanExpressions.get(0), xpathResult.get(0));
32      }
