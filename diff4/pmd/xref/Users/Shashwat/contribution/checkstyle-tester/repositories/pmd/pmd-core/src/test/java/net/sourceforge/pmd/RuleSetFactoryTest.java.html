109         assertNotNull(ruleset4Rule1);

111         assertEquals(1, countRule(rs, "Ruleset4Rule1"));

118     private int countRule(RuleSets rs, String ruleName) {
119         int count = 0;
120         for (Rule r : rs.getAllRules()) {
121             if (ruleName.equals(r.getName())) {
122                 count++;
123             }
124         }
125         return count;
126     }
