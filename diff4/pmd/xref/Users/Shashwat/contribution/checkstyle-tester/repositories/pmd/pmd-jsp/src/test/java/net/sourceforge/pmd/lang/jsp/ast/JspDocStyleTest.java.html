44      public void testElementAttributeAndNamespace() {
45          ASTCompilationUnit root = jsp.parse(TEST_ELEMENT_AND_NAMESPACE);
46  
47          List<ASTElement> elementNodes = root.findDescendantsOfType(ASTElement.class);
48          assertEquals("One element node expected!", 1, elementNodes.size());

51          assertTrue("Has namespace prefix!", element.isHasNamespacePrefix());
52          assertTrue("Element is empty!", element.isEmpty());
53          assertEquals("Correct namespace prefix of element expected!", "h", element.getNamespacePrefix());
54          assertEquals("Correct local name of element expected!", "html", element.getLocalName());
55  
56          List<ASTAttribute> attributeNodes = root.findDescendantsOfType(ASTAttribute.class);
57          assertEquals("One attribute node expected!", 1, attributeNodes.size());
58          ASTAttribute attribute = attributeNodes.get(0);
59          assertEquals("Correct name expected!", "MyNsPrefix:MyAttr", attribute.getName());
60          assertTrue("Has namespace prefix!", attribute.isHasNamespacePrefix());
61          assertEquals("Correct namespace prefix of element expected!", "MyNsPrefix", attribute.getNamespacePrefix());
62          assertEquals("Correct local name of element expected!", "MyAttr", attribute.getLocalName());

251      * Make sure mixed expressions don't confuse the parser
252      */
253     @Test
254     public void mixedExpressions() {
255         List<ASTJspExpression> exprs = jsp.getNodes(ASTJspExpression.class, TEST_TAGS_WITH_MIXED_EXPRESSIONS);
256         assertEquals("One JSP expression expected!", 1, exprs.size());
257         assertEquals("Image of expression should be \"expr\"", "expr", exprs.iterator().next().getImage());
258         List<ASTElExpression> els = jsp.getNodes(ASTElExpression.class, TEST_TAGS_WITH_MIXED_EXPRESSIONS);
259         assertEquals("Two EL expression expected!", 2, els.size());
260         assertEquals("Image of el should be \"expr\"", "expr", els.iterator().next().getImage());
261 
262         List<ASTUnparsedText> unparsedtexts = jsp.getNodes(ASTUnparsedText.class, TEST_TAGS_WITH_MIXED_EXPRESSIONS);
263         List<ASTUnparsedText> sortedUnparsedTxts = sortByImage(unparsedtexts);
264         assertEquals("Two unparsed texts expected!", 2, sortedUnparsedTxts.size());
265         Iterator<ASTUnparsedText> iterator = sortedUnparsedTxts.iterator();
266         assertEquals("Image of text should be \"\\${expr}\"", " \\${expr} ", iterator.next().getImage());
267         assertEquals("Image of text should be \" aaa \"", " aaa ", iterator.next().getImage());
268 
269         // ASTText should contain the text between two tags.
270         List<ASTText> texts = jsp.getNodes(ASTText.class, TEST_TAGS_WITH_MIXED_EXPRESSIONS);
271         List<ASTText> sortedTxts = sortByImage(texts);
272         assertEquals("Two regular texts expected!", 2, sortedTxts.size());
273         Iterator<ASTText> iterator2 = sortedTxts.iterator();
274         assertEquals("Image of text should be \"\\${expr}\"", " \\${expr} ", iterator2.next().getImage());
275         assertEquals("Image of text should be all text between two nodes" + " \"  aaa ${expr}#{expr} \"",
276                 " aaa ${expr}#{expr}", iterator2.next().getImage());
277     }

777     /**
778      * will sort the AST element in list in alphabetical order and if tag name
779      * is the same it will sort against o1.getBeginColumn() +""+
780      * o1.getBeginLine(). so first criteria is the name, then the second is the
781      * column +""+line string.
782      *
783      * @param elements
784      * @return
785      */
786     private List<ASTElement> sortNodesByName(Collection<ASTElement> elements) {
787         List<ASTElement> list = new ArrayList<>();
788         list.addAll(elements);
789         Collections.sort(list, new Comparator<ASTElement>() {
790             public int compare(ASTElement o1, ASTElement o2) {
791                 if (o1.getName() == null) {
792                     return Integer.MIN_VALUE;
793                 }
794                 if (o2.getName() == null) {
795                     return Integer.MAX_VALUE;
796                 }
797                 if (o1.getName().equals(o2.getName())) {
798                     String o1Value = o1.getBeginColumn() + "" + o1.getBeginLine();
799                     String o2Value = o2.getBeginColumn() + "" + o2.getBeginLine();
800                     return o1Value.compareTo(o2Value);
801                 }
802                 return o1.getName().compareTo(o2.getName());
803             }
804         });
805         return list;
806     }

808     /**
809      * will sort the AST node by the image name.
810      *
811      * @param elements
812      * @return
813      */
814     private <T extends Node> List<T> sortByImage(Collection<T> elements) {
815         List<T> list = new ArrayList<>();
816         list.addAll(elements);
817         Collections.sort(list, new Comparator<Node>() {
818             public int compare(Node o1, Node o2) {
819                 if (o1.getImage() == null) {
820                     return Integer.MIN_VALUE;
821                 }
822                 if (o2.getImage() == null) {
823                     return Integer.MAX_VALUE;
824                 }
825                 if (o1.getImage().equals(o2.getImage())) {
826                     String o1Value = o1.getBeginColumn() + "" + o1.getBeginLine();
827                     String o2Value = o2.getBeginColumn() + "" + o2.getBeginLine();
828                     return o1Value.compareTo(o2Value);
829                 }
830                 return o1.getImage().compareTo(o2.getImage());
831             }
832         });
833         return list;
834     }
