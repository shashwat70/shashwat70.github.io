8   public class JXPathMetaModuleExample {
9   
10      // A null pointer error very similar to this was reported in
11      // http://issues.apache.org/bugzilla/show_bug.cgi?id=28646
12  
13      // TODO: generate a NPE warning for this bug
14      public Object[] getCollectionAsArray(Collection c) {
15          List values = null;
16          Iterator i = c.iterator();
17          if (i.hasNext()) {
18              values = new LinkedList();
19          }
20          while (i.hasNext()) {
21              values.add(i.next());
22          }
23          Object[] obj = values.toArray(); // guaranteed dereference here
24          return obj;
25      }
26  
27      // A variation, in which the NPE only occurs if value == null
28      // and !i.hasNext(). So don't report it.
29      public Object[] addCollectionToListDoNotReport(Collection c, List values) {
30          if (values == null)
31              System.out.println("Values shouldn't be null");
32          Iterator i = c.iterator();
33          if (i.hasNext()) {
34              values = new LinkedList();
35          }
36          while (i.hasNext()) {
37              values.add(i.next());
38          }
39          Object[] obj = values.toArray(); // guaranteed dereference here
40          return obj;
41      }
42  
43      // A variation,which we should report.
44      // if the test x == null succeeds, we are guaranteed to dereference the
45      // value in x
46      // But at the merge point after the if y >= 0, x is NCP.
47      // So if we don't track GuaranteedDereference backwards across
48      // value number mergers, we won't detect this one
49      public int variation(Object x, int y) {
50          int result = 2;
51          if (y >= 0) {
52              if (x == null)
53                  result = 1;
54              if (y > 0)
55                  result *= y;
56          } else
57              x = new Object();
58          result += x.hashCode();
59          return result;
60      }
