178     private static CertPath generateCertificatePath(String certStr)
179             throws CertificateException {
180         // generate certificate from cert strings
181         CertificateFactory cf = CertificateFactory.getInstance("X.509");
182 
183         ByteArrayInputStream is;
184 
185         is = new ByteArrayInputStream(certStr.getBytes());
186         Certificate cert = cf.generateCertificate(is);
187 
188         // generate certification path
189         List<Certificate> list = Arrays.asList(new Certificate[] {cert});
190 
191         return cf.generateCertPath(list);
192     }
193 
194     private static Set<TrustAnchor> generateTrustAnchors()
195             throws CertificateException {
196         // generate certificate from cert string
197         CertificateFactory cf = CertificateFactory.getInstance("X.509");
198         HashSet<TrustAnchor> anchors = new HashSet<TrustAnchor>();
199 
200         ByteArrayInputStream is =
201             new ByteArrayInputStream(trustAnchor_SHA1withRSA_1024.getBytes());
202         Certificate cert = cf.generateCertificate(is);
203         TrustAnchor anchor = new TrustAnchor((X509Certificate)cert, null);
204         anchors.add(anchor);
205 
206         is = new ByteArrayInputStream(trustAnchor_SHA1withRSA_512.getBytes());
207         cert = cf.generateCertificate(is);
208         anchor = new TrustAnchor((X509Certificate)cert, null);
209         anchors.add(anchor);
210 
211         return anchors;
212     }

214     public static void main(String args[]) throws Exception {
215         try {
216             validate(intermediate_SHA1withRSA_1024_1024);
217             validate(intermediate_SHA1withRSA_1024_512);
218             validate(intermediate_SHA1withRSA_512_1024);
219             validate(intermediate_SHA1withRSA_512_512);
220         } catch (CertPathValidatorException cpve) {
221             throw new Exception(
222                 "unexpect exception, it is valid cert", cpve);
223         }
224 
225         try {
226             validate(intermediate_MD2withRSA_1024_1024);
227             throw new Exception("expected algorithm disabled exception");
228         } catch (CertPathValidatorException cpve) {
229             // we may get ClassCastException here
230             BasicReason reason = (BasicReason)cpve.getReason();
231             if (reason != BasicReason.ALGORITHM_CONSTRAINED) {
232                 throw new Exception(
233                     "Expect to get ALGORITHM_CONSTRAINED CPVE", cpve);
234             }
235 
236             System.out.println("Get the expected exception " + cpve);
237         }
238 
239         try {
240             validate(intermediate_MD2withRSA_1024_512);
241             throw new Exception("expected algorithm disabled exception");
242         } catch (CertPathValidatorException cpve) {
243             // we may get ClassCastException here
244             BasicReason reason = (BasicReason)cpve.getReason();
245             if (reason != BasicReason.ALGORITHM_CONSTRAINED) {
246                 throw new Exception(
247                     "Expect to get ALGORITHM_CONSTRAINED CPVE", cpve);
248             }
249 
250             System.out.println("Get the expected exception " + cpve);
251         }
252     }
253 
254     private static void validate(String intermediate)
255             throws CertPathValidatorException, Exception {
256 
257         CertPath path = generateCertificatePath(intermediate);
258         Set<TrustAnchor> anchors = generateTrustAnchors();
259 
260         PKIXParameters params = new PKIXParameters(anchors);
261 
262         // disable certificate revocation checking
263         params.setRevocationEnabled(false);
264 
265         // set the validation time
266         params.setDate(new Date(109, 9, 1));   // 2009-09-01
267 
268         CertPathValidator validator = CertPathValidator.getInstance("PKIX");
269 
270         validator.validate(path, params);
271     }
