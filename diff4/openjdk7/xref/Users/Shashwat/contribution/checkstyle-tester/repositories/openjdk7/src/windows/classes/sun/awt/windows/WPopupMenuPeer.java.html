25  package sun.awt.windows;
26  
27  import java.awt.*;
28  import java.awt.peer.*;
29  import java.lang.reflect.Field;
30  
31  import sun.awt.SunToolkit;
32  import sun.awt.AWTAccessor;
33  
34  public class WPopupMenuPeer extends WMenuPeer implements PopupMenuPeer {
35      // We can't use target.getParent() for TrayIcon popup
36      // because this method should return null for the TrayIcon
37      // popup regardless of that whether it has parent or not.
38  
39      public WPopupMenuPeer(PopupMenu target) {
40          this.target = target;
41          MenuContainer parent = null;
42  
43          // We can't use target.getParent() for TrayIcon popup
44          // because this method should return null for the TrayIcon
45          // popup regardless of that whether it has parent or not.
46          boolean isTrayIconPopup = AWTAccessor.getPopupMenuAccessor().isTrayIconPopup(target);
47          if (isTrayIconPopup) {
48              parent = AWTAccessor.getMenuComponentAccessor().getParent(target);
49          } else {
50              parent = target.getParent();
51          }
52  
53          if (parent instanceof Component) {
54              WComponentPeer parentPeer = (WComponentPeer) WToolkit.targetToPeer(parent);
55              if (parentPeer == null) {
56                  // because the menu isn't a component (sigh) we first have to wait
57                  // for a failure to map the peer which should only happen for a
58                  // lightweight container, then find the actual native parent from
59                  // that component.
60                  parent = WToolkit.getNativeContainer((Component)parent);
61                  parentPeer = (WComponentPeer) WToolkit.targetToPeer(parent);
62              }
63              createMenu(parentPeer);
64              // fix for 5088782: check if menu object is created successfully
65              checkMenuCreation();
66          } else {
67              throw new IllegalArgumentException(
68                  "illegal popup menu container class");
69          }
70      }
71  
72      native void createMenu(WComponentPeer parent);
73  
74      public void show(Event e) {
75          Component origin = (Component)e.target;
76          WComponentPeer peer = (WComponentPeer) WToolkit.targetToPeer(origin);
77          if (peer == null) {
78              // A failure to map the peer should only happen for a
79              // lightweight component, then find the actual native parent from
80              // that component.  The event coorinates are going to have to be
81              // remapped as well.
82              Component nativeOrigin = WToolkit.getNativeContainer(origin);
83              e.target = nativeOrigin;
84  
85              // remove the event coordinates
86              for (Component c = origin; c != nativeOrigin; c = c.getParent()) {
87                  Point p = c.getLocation();
88                  e.x += p.x;
89                  e.y += p.y;
90              }
91          }
92          _show(e);
93      }
94  
95      /*
96       * This overloaded method is for TrayIcon.
97       * Its popup has special parent.
98       */
99      void show(Component origin, Point p) {
100         WComponentPeer peer = (WComponentPeer) WToolkit.targetToPeer(origin);
101         Event e = new Event(origin, 0, Event.MOUSE_DOWN, p.x, p.y, 0, 0);
102         if (peer == null) {
103             Component nativeOrigin = WToolkit.getNativeContainer(origin);
104             e.target = nativeOrigin;
105         }
106         e.x = p.x;
107         e.y = p.y;
108         _show(e);
109     }
110 
111     public native void _show(Event e);
112 }
