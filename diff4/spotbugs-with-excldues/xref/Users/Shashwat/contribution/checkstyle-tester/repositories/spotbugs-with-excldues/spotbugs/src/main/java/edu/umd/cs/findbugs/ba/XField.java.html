20  package edu.umd.cs.findbugs.ba;
21  
22  import edu.umd.cs.findbugs.classfile.FieldDescriptor;
23  import edu.umd.cs.findbugs.classfile.analysis.AnnotatedObject;
24  
25  /**
26   * Abstract representation of a field. Note that this is called "XField" to
27   * distinguish it from BCEL's Field class. Also, you can think of the "X" as
28   * expanding to "Instance" or "Static".
29   *
30   * <p>
31   * This interface and its implementations exist because Field objects in BCEL
32   * are awkward to deal with. They are not Comparable, it is difficult to find
33   * out what class they belong to, etc.
34   * </p>
35   *
36   * <p>
37   * If the resolved() method returns true, then any information queried from this
38   * object can be assumed to be accurate. If the resolved() method returns false,
39   * then FindBugs can't find the field and any information other than
40   * name/signature/etc. cannot be trusted.
41   * </p>
42   */
43  public interface XField extends ClassMember, AnnotatedObject, ComparableField {
44      /**
45       * Is the type of the field a reference type?
46       */
47      public boolean isReferenceType();
48  
49      /**
50       * Is this a volatile field?
51       */
52      public boolean isVolatile();
53  
54      /**
55       * @return true if this is a enum constant
56       */
57      public boolean isEnum();
58  
59      /**
60       * @return FieldDescriptor referring to this field
61       */
62      public FieldDescriptor getFieldDescriptor();
63  }
