923     /**
924      * Returns a string that includes non-ascii characters after expanding
925      * the non-ascii characters to their Java language \\uxxxx form.
926      * @param input an input string
927      * @return the encoded string.
928      */
929     private String convertNonAscii(String input) {
930         StringBuilder sb = new StringBuilder(input.length() * 6);
931         for (int i = 0; i < input.length(); i++) {
932             char ch = input.charAt(i);
933             if (ch < 255) {
934                 sb.append(ch);
935             } else {
936                 sb.append("\\u");
937                 sb.append(Integer.toHexString(ch));
938             }
939         }
940         return sb.toString();
941     }
