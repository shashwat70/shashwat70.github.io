19  package org.apache.hadoop.hbase.metrics;
20  
21  import javax.management.ObjectName;
22  
23  import org.apache.hadoop.hbase.classification.InterfaceAudience;
24  import org.apache.hadoop.metrics2.util.MBeans;
25  
26  /**
27   * Hadoop2 metrics2 implementation of an object that registers MBeans.
28   */
29  @InterfaceAudience.Private
30  public class MBeanSourceImpl implements MBeanSource {
31  
32    /**
33     * Register an mbean with the underlying metrics system
34     * @param serviceName Metrics service/system name
35     * @param metricsName name of the metrics obejct to expose
36     * @param theMbean the actual MBean
37     * @return ObjectName from jmx
38     */
39    @Override
40    public ObjectName register(String serviceName, String metricsName, Object theMbean) {
41      return MBeans.register(serviceName, metricsName, theMbean);
42    }
43  }
