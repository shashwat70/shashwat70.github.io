39  	@Test
40  	public void testInitEnvironmentDefault() throws Exception {
41  		final Launcher launcher = new Launcher();
42  		launcher.setArgs(new String[0]);
43  		launcher.processArguments();
44  
45  		final Environment environment = launcher.getEnvironment();
46  		// specify the default values
47  		assertFalse(environment.isAutoImports());
48  		assertFalse(environment.isUsingTabulations());
49  		assertFalse(environment.isPreserveLineNumbers());
50  		assertEquals(4, environment.getTabulationSize());
51  		assertTrue(environment.isCopyResources());
52  
53  		JavaOutputProcessor processor = (JavaOutputProcessor) environment.getDefaultFileGenerator();
54  		assertTrue(processor.getPrinter() instanceof DefaultJavaPrettyPrinter);
55  
56  		// now assertions on the model builder
57  		final SpoonModelBuilder builder = launcher.getModelBuilder();
58  		assertEquals(new File("spooned").getCanonicalFile(), builder.getSourceOutputDirectory());
59  		assertEquals(0, builder.getInputSources().size());
60  		assertEquals("UTF-8", environment.getEncoding().displayName());
61  	}

63  	@Test
64  	public void testInitEnvironment() throws Exception {
65  
66  		// Main class of Spoon who contain initEnvironment method.
67  		final Launcher launcher = new Launcher();
68  		launcher.setArgs("--tabs --tabsize 42 --compliance 5 --with-imports -r --lines -o spooned2 -i src/main/java --encoding UTF-16".split(" "));
69  		launcher.processArguments();
70  
71  		final Environment environment = launcher.getEnvironment();
72  
73  		// Verify if the environment is correct.
74  		assertTrue(environment.isAutoImports());
75  		assertTrue(environment.isUsingTabulations());
76  		assertTrue(environment.isPreserveLineNumbers());
77  		assertEquals(42, environment.getTabulationSize());
78  		assertEquals(5, environment.getComplianceLevel());
79  		assertFalse(environment.isCopyResources());
80  
81  		final SpoonModelBuilder builder = launcher.getModelBuilder();
82  		assertEquals(new File("spooned2").getCanonicalFile(), builder.getSourceOutputDirectory());
83  
84  		// the input directories
85  		List<File> inputSources = new ArrayList<>(builder.getInputSources());
86  		assertTrue(inputSources.get(0).getPath().replace('\\', '/').contains("src/main/java"));
87  		assertEquals("UTF-16", environment.getEncoding().displayName());
88  	}

90  	@Test
91  	public void testLauncherInEmptyWorkingDir() throws Exception {
92  
93  		// Contract: Spoon can be launched in an empty folder as a working directory
94  		// See: https://github.com/INRIA/spoon/pull/1208
95  		// This test does not fail (it's not enough to change user.dir we should launch process inside that dir) but it explains the problem
96  		final Launcher launcher = new Launcher();
97  		Path path = Files.createTempDirectory("emptydir");
98  
99  		String oldUserDir = System.getProperty("user.dir");
100 		System.setProperty("user.dir", path.toFile().getAbsolutePath());
101 
102 		// path should exist, otherwise it would crash on a filenotfoundexception before showing the bug
103 		launcher.addInputResource(oldUserDir + "/src/test/java/spoon/LauncherTest.java");
104 		try {
105 			launcher.buildModel();
106 		} finally {
107 			System.setProperty("user.dir", oldUserDir);
108 		}
109 	}
