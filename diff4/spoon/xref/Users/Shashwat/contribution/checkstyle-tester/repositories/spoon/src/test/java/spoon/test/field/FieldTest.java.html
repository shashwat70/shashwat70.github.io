52  	@Test
53  	public void testAddAFieldInAClassAtAPositionGiven() {
54  		final Factory factory = createFactory();
55  		final CtClass<Object> fieldClass = factory.Class().create("FieldClass");
56  
57  		final HashSet<ModifierKind> modifiers = new HashSet<>();
58  		modifiers.add(ModifierKind.STATIC);
59  		final CtField<Integer> first = createField(factory, modifiers, "FIELD");
60  		fieldClass.addField(first);
61  
62  		final CtField<Integer> second = createField(factory, modifiers, "FIELD_2");
63  		second.setDefaultExpression(factory.Code().createCodeSnippetExpression(first.getSimpleName() + " + 1"));
64  		fieldClass.addField(1, second);
65  
66  		final CtField<Integer> third = createField(factory, modifiers, "FIELD_3");
67  		third.setDefaultExpression(factory.Code().createCodeSnippetExpression(first.getSimpleName() + " + 1"));
68  		fieldClass.addField(1, third);
69  
70  		assertEquals(3, fieldClass.getFields().size());
71  		assertEquals(first, fieldClass.getFields().get(0));
72  		assertEquals(third, fieldClass.getFields().get(1));
73  		assertEquals(second, fieldClass.getFields().get(2));
74  	}

76  	@Test
77  	public void testgetDeclaredFields() throws Exception {
78  		// contract: get*Fields works for both references
79  		final CtClass<AddFieldAtTop> aClass = (CtClass<AddFieldAtTop>) buildClass(AddFieldAtTop.class);
80  
81  		assertEquals(1, aClass.getReference().getDeclaredFields().size());
82  		CtTypeReference<?> fileClass = aClass.getFactory().Type().get(File.class).getReference();
83  		assertEquals(13, fileClass.getDeclaredFields().size());
84  		assertEquals("pathSeparator", fileClass.getDeclaredField("pathSeparator").getSimpleName());
85  		assertEquals("pathSeparator", fileClass.getDeclaredOrInheritedField("pathSeparator").getSimpleName());
86  
87  		// double check that we can still go to the declaration
88  		assertEquals("pathSeparator", fileClass.getDeclaredField("pathSeparator").getFieldDeclaration().getSimpleName());
89  	}

91  	@Test
92  	public void testAddFieldsAtTop() throws Exception {
93  		// contract: When we use CtType#addFieldAtTop, field added should be printed at the top of the type.
94  		final CtClass<AddFieldAtTop> aClass = (CtClass<AddFieldAtTop>) buildClass(AddFieldAtTop.class);
95  
96  		assertEquals(1, aClass.getFields().size());
97  
98  		final CtField<String> generated = aClass.getFactory().Field().create(null, new HashSet<>(), aClass.getFactory().Type().STRING, "generated");
99  		aClass.addFieldAtTop(generated);
100 		final CtField<String> generated2 = aClass.getFactory().Field().create(null, new HashSet<>(), aClass.getFactory().Type().STRING, "generated2");
101 		aClass.addFieldAtTop(generated2);
102 
103 		assertEquals(3, aClass.getFields().size());
104 		// For now, DefaultJavaPrettyPrinter sorts elements according to their position.
105 		assertEquals(generated2, aClass.getTypeMembers().get(0));
106 		assertEquals(generated, aClass.getTypeMembers().get(1));
107 		assertEquals(aClass.getAnonymousExecutables().get(0), aClass.getTypeMembers().get(3));
108 	}

112 		// contract: no "." when target is implicit
113 		final CtClass<AddFieldAtTop> aClass = (CtClass<AddFieldAtTop>) buildClass(AddFieldAtTop.class);

115 		List<CtFieldRead> fieldReads = aClass.getElements(new TypeFilter<>(CtFieldRead.class));
116 		assertEquals(1, fieldReads.size());
117 		assertEquals("i", fieldReads.get(0).toString());
118 		fieldReads.get(0).getTarget().setImplicit(false);
119 		assertFalse(fieldReads.get(0).getTarget().isImplicit());
120 		assertEquals("this.i", fieldReads.get(0).toString());

123 	private CtField<Integer> createField(Factory factory, HashSet<ModifierKind> modifiers, String name) {
124 		final CtField<Integer> first = factory.Core().createField();
125 		first.setModifiers(modifiers);
126 		first.setType(factory.Type().INTEGER_PRIMITIVE);
127 		first.setSimpleName(name);
128 		return first;
129 	}

180 	@Test
181 	public void bugAfterRefactoringImports() {
182 		Launcher launcher = new Launcher();
183 		Factory factory = launcher.getFactory();
184 		final CtClass<?> klass = factory.createClass("foo.A");
185 
186 		final CtFieldRead<Object> fieldRead = factory.createFieldRead();
187 
188 		final CtField<Object> negative_infinity = (CtField<Object>) factory.Class().get(Double.class).getField("NEGATIVE_INFINITY");
189 		fieldRead.setVariable(negative_infinity.getReference());
190 
191 		launcher.getEnvironment().setAutoImports(false);
192 		assertEquals("java.lang.Double.NEGATIVE_INFINITY", fieldRead.toString());
193 
194 		launcher.getEnvironment().setAutoImports(true);
195 		assertEquals("Double.NEGATIVE_INFINITY", fieldRead.toString());
196 
197 		final CtField<Object> field = (CtField<Object>) factory.Class().get(File.class).getField("separator");
198 		fieldRead.setVariable(field.getReference());
199 		field.setDefaultExpression(fieldRead);
200 		klass.addField(field);
201 
202 		launcher.getEnvironment().setAutoImports(true);
203 		assertEquals("package foo;\n" +
204 				"import java.io.File;\n" +
205 				"class A {\n" +
206 				"    public static final String separator = File.separator;\n" +
207 				"}", klass.toStringWithImports());
