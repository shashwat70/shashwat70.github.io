17  package org.orekit.estimation.measurements;
18  
19  import java.util.List;
20  
21  import org.hipparchus.stat.descriptive.rank.Median;
22  import org.hipparchus.util.FastMath;
23  import org.junit.Assert;
24  import org.junit.Test;
25  import org.orekit.errors.OrekitException;
26  import org.orekit.estimation.Context;
27  import org.orekit.estimation.EstimationTestUtils;
28  import org.orekit.estimation.EstimationUtils;
29  import org.orekit.estimation.StateFunction;
30  import org.orekit.orbits.OrbitType;
31  import org.orekit.orbits.PositionAngle;
32  import org.orekit.propagation.Propagator;
33  import org.orekit.propagation.SpacecraftState;
34  import org.orekit.propagation.conversion.NumericalPropagatorBuilder;
35  import org.orekit.time.AbsoluteDate;
36  
37  public class PVTest {
38  
39      @Test
40      public void testStateDerivatives() throws OrekitException {
41  
42          Context context = EstimationTestUtils.eccentricContext();
43  
44          final NumericalPropagatorBuilder propagatorBuilder =
45                          context.createBuilder(OrbitType.KEPLERIAN, PositionAngle.TRUE, true,
46                                                1.0e-6, 60.0, 0.001);
47  
48          // create perfect range measurements
49          final Propagator propagator = EstimationTestUtils.createPropagator(context.initialOrbit,
50                                                                             propagatorBuilder);
51          final List<ObservedMeasurement<?>> measurements =
52                          EstimationTestUtils.createMeasurements(propagator,
53                                                                 new PVMeasurementCreator(),
54                                                                 1.0, 3.0, 300.0);
55          propagator.setSlaveMode();
56  
57          double[] errorsP = new double[3 * 6 * measurements.size()];
58          double[] errorsV = new double[3 * 6 * measurements.size()];
59          int indexP = 0;
60          int indexV = 0;
61          for (final ObservedMeasurement<?> measurement : measurements) {
62  
63              final AbsoluteDate    date      = measurement.getDate();
64              final SpacecraftState state     = propagator.propagate(date);
65              final double[][]      jacobian  = measurement.estimate(0, 0, state).getStateDerivatives();
66  
67              // compute a reference value using finite differences
68              final double[][] finiteDifferencesJacobian =
69                  EstimationUtils.differentiate(new StateFunction() {
70                      public double[] value(final SpacecraftState state) throws OrekitException {
71                          return measurement.estimate(0, 0, state).getEstimatedValue();
72                      }
73                                                    }, measurement.getDimension(), OrbitType.CARTESIAN,
74                                                    PositionAngle.TRUE, 1.0, 3).value(state);
75  
76              Assert.assertEquals(finiteDifferencesJacobian.length, jacobian.length);
77              Assert.assertEquals(finiteDifferencesJacobian[0].length, jacobian[0].length);
78              for (int i = 0; i < jacobian.length; ++i) {
79                  for (int j = 0; j < jacobian[i].length; ++j) {
80                      final double relativeError = FastMath.abs((finiteDifferencesJacobian[i][j] - jacobian[i][j]) /
81                                                                finiteDifferencesJacobian[i][j]);
82                      if (j < 3) {
83                          errorsP[indexP++] = relativeError;
84                      } else {
85                          errorsV[indexV++] = relativeError;
86                      }
87                  }
88              }
89  
90          }
91  
92          // median errors
93          Assert.assertEquals(0.0, new Median().evaluate(errorsP), 2.1e-10);
94          Assert.assertEquals(0.0, new Median().evaluate(errorsV), 2.1e-10);
95  
96      }
97  
98  }
