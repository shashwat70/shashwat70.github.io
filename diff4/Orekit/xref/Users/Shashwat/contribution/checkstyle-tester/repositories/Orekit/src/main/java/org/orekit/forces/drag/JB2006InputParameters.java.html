17  package org.orekit.forces.drag;
18  
19  import java.io.Serializable;
20  
21  import org.orekit.errors.OrekitException;
22  import org.orekit.time.AbsoluteDate;
23  
24  
25  /**  Interface for solar activity and magnetic activity data.
26   * @author Fabien Maussion
27   */
28  public interface JB2006InputParameters extends Serializable {
29  
30      /** Gets the available data range minimum date.
31       * @return the minimum date.
32       */
33      AbsoluteDate getMinDate();
34  
35      /** Gets the available data range maximum date.
36       * @return the maximum date.
37       */
38      AbsoluteDate getMaxDate();
39  
40      /** Get the value of the instantaneous solar flux index
41       * (1e<sup>-22</sup>*Watt/(m²*Hertz)).
42       * Tabular time 1.0 day earlier.
43       * @param date the current date
44       * @return the instantaneous F10.7 index
45       * @exception OrekitException if the date is out of range of available data
46       */
47      double getF10(AbsoluteDate date) throws OrekitException;
48  
49      /** Get the value of the mean solar flux.
50       * Averaged 81-day centered F10.7 B index on the input time.
51       * @param date the current date
52       * @return the mean solar flux F10.7B index
53       * @exception OrekitException if the date is out of range of available data
54       */
55      double getF10B(AbsoluteDate date) throws OrekitException;
56  
57      /** Get the EUV index (26-34 nm) scaled to F10.
58       * Tabular time 1 day earlier.
59       * @param date the current date
60       * @return the the EUV S10 index
61       * @exception OrekitException if the date is out of range of available data
62       */
63      double getS10(AbsoluteDate date) throws OrekitException;
64  
65      /** Get the EUV 81-day averaged centered index.
66       * @param date the current date
67       * @return the the mean EUV S10B index
68       * @exception OrekitException if the date is out of range of available data
69       */
70      double getS10B(AbsoluteDate date) throws OrekitException;
71  
72      /** Get the MG2 index scaled to F10.
73       * @param date the current date
74       * @return the the EUV S10 index
75       * @exception OrekitException if the date is out of range of available data
76       */
77      double getXM10(AbsoluteDate date) throws OrekitException;
78  
79      /** Get the MG2 81-day average centered index.
80       * Tabular time 5.0 days earlier.
81       * @param date the current date
82       * @return the the mean EUV S10B index
83       * @exception OrekitException if the date is out of range of available data
84       */
85      double getXM10B(AbsoluteDate date) throws OrekitException;
86  
87      /** Get the Geomagnetic planetary 3-hour index A<sub>p</sub>.
88       * Tabular time 6.7 hours earlier.
89       * @param date the current date
90       * @return the A<sub>p</sub> index
91       * @exception OrekitException if the date is out of range of available data
92       */
93      double getAp(AbsoluteDate date) throws OrekitException;
94  
95  }
