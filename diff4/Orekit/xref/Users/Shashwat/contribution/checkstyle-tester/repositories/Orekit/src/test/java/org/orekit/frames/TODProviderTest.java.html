176     @Test
177     public void testInterpolationAccuracyWithEOP() throws OrekitException, FileNotFoundException {
178 
179         // max interpolation error observed on a one month period with 60 seconds step
180         //
181         // number of sample points    time between sample points    max error
182         //        6                          86400s /  8 =  3h       19.56e-12 rad
183         //        6                          86400s / 12 =  2h       13.02e-12 rad
184         //        6                          86400s / 16 =  1h30      9.75e-12 rad
185         //        6                          86400s / 20 =  1h12      7.79e-12 rad
186         //        6                          86400s / 24 =  1h        6.48e-12 rad
187         //        8                          86400s /  8 =  3h       20.91e-12 rad
188         //        8                          86400s / 12 =  2h       13.91e-12 rad
189         //        8                          86400s / 16 =  1h30     10.42e-12 rad
190         //        8                          86400s / 20 =  1h12      8.32e-12 rad
191         //        8                          86400s / 24 =  1h        6.92e-12 rad
192         //       10                          86400s /  8 =  3h       21.65e-12 rad
193         //       10                          86400s / 12 =  2h       14.41e-12 rad
194         //       10                          86400s / 16 =  1h30     10.78e-12 rad
195         //       10                          86400s / 20 =  1h12      8.61e-12 rad
196         //       10                          86400s / 24 =  1h        7.16e-12 rad
197         //       12                          86400s /  8 =  3h       22.12e-12 rad
198         //       12                          86400s / 12 =  2h       14.72e-12 rad
199         //       12                          86400s / 16 =  1h30     11.02e-12 rad
200         //       12                          86400s / 20 =  1h12      8.80e-12 rad
201         //       12                          86400s / 24 =  1h        7.32e-12 rad
202         //
203         // looking at error behavior during along the sample show the max error is
204         // a peak at 00h00 each day for all curves, which matches the EOP samples
205         // points used for correction (eopHistoru is set to non null at construction here).
206         // So looking only at max error does not allow to select an interpolation
207         // setting as they all fall in a similar 6e-12 to 8e-12 range. Looking at
208         // the error behavior between these peaks however shows that there is still
209         // some signal if the time interval is between sample points is too large,
210         // in order to get only numerical noise, we have to go as far as 1h between
211         // the points.
212         // We finally select 6 interpolation points separated by 1 hour each
213         EOPHistory eopHistory = FramesFactory.getEOPHistory(IERSConventions.IERS_1996, false);
214         TransformProvider nonInterpolating = new TODProvider(IERSConventions.IERS_1996, eopHistory);
215         final TransformProvider interpolating =
216                 new InterpolatingTransformProvider(nonInterpolating,
217                                                    CartesianDerivativesFilter.USE_PVA,
218                                                    AngularDerivativesFilter.USE_R,
219                                                    AbsoluteDate.PAST_INFINITY, AbsoluteDate.FUTURE_INFINITY,
220                                                    6, Constants.JULIAN_DAY / 24,
221                                                    OrekitConfiguration.getCacheSlotsNumber(),
222                                                    Constants.JULIAN_YEAR, 30 * Constants.JULIAN_DAY);
223 
224         // the following time range is located around the maximal observed error
225         AbsoluteDate start = new AbsoluteDate(2002, 11, 11, 0, 0, 0.0, TimeScalesFactory.getTAI());
226         AbsoluteDate end   = new AbsoluteDate(2002, 11, 15, 6, 0, 0.0, TimeScalesFactory.getTAI());
227         double maxError = 0.0;
228         for (AbsoluteDate date = start; date.compareTo(end) < 0; date = date.shiftedBy(60)) {
229             final Transform transform =
230                     new Transform(date,
231                                   interpolating.getTransform(date),
232                                   nonInterpolating.getTransform(date).getInverse());
233             final double error = transform.getRotation().getAngle();
234             maxError = FastMath.max(maxError, error);
235         }
236 
237         Assert.assertTrue(maxError < 7e-12);
238 
239     }

241     @Test
242     public void testInterpolationAccuracyWithoutEOP() throws OrekitException, FileNotFoundException {
243 
244         // max interpolation error observed on a one month period with 60 seconds step
245         //
246         // number of sample points    time between sample points    max error
247         //        5                          86400s /  3 =  8h     3286.90e-15 rad
248         //        5                          86400s /  6 =  4h      103.90e-15 rad
249         //        5                          86400s /  8 =  3h       24.74e-15 rad
250         //        5                          86400s / 12 =  2h        4.00e-15 rad
251         //        6                          86400s /  3 =  8h      328.91e-15 rad
252         //        6                          86400s /  6 =  4h        5.92e-15 rad
253         //        6                          86400s /  8 =  3h        3.95e-15 rad
254         //        6                          86400s / 12 =  2h        3.94e-15 rad
255         //        8                          86400s /  3 =  8h        5.87e-15 rad
256         //        8                          86400s /  6 =  4h        4.73e-15 rad
257         //        8                          86400s /  8 =  3h        4.45e-15 rad
258         //        8                          86400s / 12 =  2h        3.87e-15 rad
259         //       10                          86400s /  3 =  8h        5.29e-15 rad
260         //       10                          86400s /  6 =  4h        5.36e-15 rad
261         //       10                          86400s /  8 =  3h        5.86e-15 rad
262         //       10                          86400s / 12 =  2h        5.76e-15 rad
263         //
264         // 
265         // We don't see anymore the peak at 00h00 so this confirms it is related to EOP
266         // sampling. All values between 3e-15 and 6e-15 are really equivalent: it is
267         // mostly numerical noise. The best settings are 6 or 8 points every 2 or 3 hours.
268         // We finally select 6 interpolation points separated by 3 hours each
269         TransformProvider nonInterpolating = new TODProvider(IERSConventions.IERS_1996, null);
270                 final TransformProvider interpolating =
271                         new InterpolatingTransformProvider(nonInterpolating,
272                                                            CartesianDerivativesFilter.USE_PVA,
273                                                            AngularDerivativesFilter.USE_R,
274                                                            AbsoluteDate.PAST_INFINITY, AbsoluteDate.FUTURE_INFINITY,
275                                                            6, Constants.JULIAN_DAY / 8,
276                                                            OrekitConfiguration.getCacheSlotsNumber(),
277                                                            Constants.JULIAN_YEAR, 30 * Constants.JULIAN_DAY);
278 
279                 // the following time range is located around the maximal observed error
280                 AbsoluteDate start = new AbsoluteDate(2002, 11, 11, 0, 0, 0.0, TimeScalesFactory.getTAI());
281                 AbsoluteDate end   = new AbsoluteDate(2002, 11, 15, 6, 0, 0.0, TimeScalesFactory.getTAI());
282                 double maxError = 0.0;
283                 for (AbsoluteDate date = start; date.compareTo(end) < 0; date = date.shiftedBy(60)) {
284                     final Transform transform =
285                             new Transform(date,
286                                           interpolating.getTransform(date),
287                                           nonInterpolating.getTransform(date).getInverse());
288                     final double error = transform.getRotation().getAngle();
289                     maxError = FastMath.max(maxError, error);
290                 }
291 
292                 Assert.assertTrue(maxError < 4.0e-15);
293 
294     }

381     @Test
382     public void testSerialization() throws OrekitException, IOException, ClassNotFoundException {
383         TODProvider provider = new TODProvider(IERSConventions.IERS_2010,
384                                                FramesFactory.getEOPHistory(IERSConventions.IERS_2010, true));
385         
386         ByteArrayOutputStream bos = new ByteArrayOutputStream();
387         ObjectOutputStream    oos = new ObjectOutputStream(bos);
388         oos.writeObject(provider);
389 
390         Assert.assertTrue(bos.size() > 280000);
391         Assert.assertTrue(bos.size() < 285000);
392 
393         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
394         ObjectInputStream     ois = new ObjectInputStream(bis);
395         TODProvider deserialized  = (TODProvider) ois.readObject();
396         for (int i = 0; i < FastMath.min(100, provider.getEOPHistory().getEntries().size()); ++i) {
397             AbsoluteDate date = provider.getEOPHistory().getEntries().get(i).getDate();
398             Transform expectedIdentity = new Transform(date,
399                                                        provider.getTransform(date).getInverse(),
400                                                        deserialized.getTransform(date));
401             Assert.assertEquals(0.0, expectedIdentity.getTranslation().getNorm(), 1.0e-15);
402             Assert.assertEquals(0.0, expectedIdentity.getRotation().getAngle(),   1.0e-15);
403         }

413                          double expectedPositionError, double expectedVelocityError) {
414 
415         Vector3D dP = result.getPosition().subtract(reference.getPosition());
416         Vector3D dV = result.getVelocity().subtract(reference.getVelocity());
417         Assert.assertEquals(expectedPositionError, dP.getNorm(), 0.01 * expectedPositionError);
418         Assert.assertEquals(expectedVelocityError, dV.getNorm(), 0.01 * expectedVelocityError);
419     }
