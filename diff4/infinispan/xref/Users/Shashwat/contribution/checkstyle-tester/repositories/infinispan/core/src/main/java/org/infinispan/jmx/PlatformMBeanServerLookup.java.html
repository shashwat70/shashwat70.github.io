1   package org.infinispan.jmx;
2   
3   import javax.management.MBeanServer;
4   import java.lang.management.ManagementFactory;
5   import java.util.Properties;
6   
7   /**
8    * Default implementation for {@link MBeanServerLookup}, will return the platform MBean server.
9    * <p/>
10   * Note: to enable platform MBeanServer the following system property should be passed to the Sun JVM:
11   * <b>-Dcom.sun.management.jmxremote</b>.
12   *
13   * @author Mircea.Markus@jboss.com
14   * @since 4.0
15   */
16  public class PlatformMBeanServerLookup implements MBeanServerLookup {
17  
18     @Override
19     public MBeanServer getMBeanServer(Properties properties) {
20        return ManagementFactory.getPlatformMBeanServer();
21     }
22  
23  }
