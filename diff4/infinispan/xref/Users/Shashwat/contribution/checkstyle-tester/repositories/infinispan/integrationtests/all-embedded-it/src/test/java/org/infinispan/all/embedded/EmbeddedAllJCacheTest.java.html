1   package org.infinispan.all.embedded;
2   
3   import javax.cache.Cache;
4   import javax.cache.CacheManager;
5   import javax.cache.Caching;
6   import javax.cache.configuration.MutableConfiguration;
7   import javax.cache.expiry.AccessedExpiryPolicy;
8   import javax.cache.spi.CachingProvider;
9   
10  import org.junit.Test;
11  
12  import static javax.cache.expiry.Duration.ONE_HOUR;
13  import static org.junit.Assert.assertEquals;
14  import static org.junit.Assert.assertNull;
15  
16  public class EmbeddedAllJCacheTest {
17     @Test
18     public void testAllEmbedded() {
19        CachingProvider cachingProvider = Caching.getCachingProvider();
20        CacheManager cacheManager = cachingProvider.getCacheManager();
21        //configure the cache
22        MutableConfiguration<String, Integer> config = new MutableConfiguration<String, Integer>();
23        config.setStoreByValue(true).setTypes(String.class, Integer.class)
24              .setExpiryPolicyFactory(AccessedExpiryPolicy.factoryOf(ONE_HOUR)).setStatisticsEnabled(true);
25        //create the cache
26        cacheManager.createCache("simpleCache", config);
27        //... and then later to get the cache
28        Cache<String, Integer> cache = Caching.getCache("simpleCache", String.class, Integer.class);
29        //use the cache
30        String key = "key";
31        Integer value1 = 1;
32        cache.put("key", value1);
33        Integer value2 = cache.get(key);
34        assertEquals(value1, value2);
35        cache.remove("key");
36        assertNull(cache.get("key"));
37        cacheManager.close();
38     }
39  }
