3   import org.infinispan.configuration.cache.PersistenceConfigurationBuilder;

9   
10  import java.io.File;
11  
12  public abstract class LevelDBStoreFunctionalTest extends BaseStoreFunctionalTest {
13     protected String tmpDirectory;
14  
15     @BeforeClass(alwaysRun = true)
16     protected void setUpTempDir() {
17        tmpDirectory = TestingUtil.tmpDirectory(this.getClass());
18     }
19  
20     @AfterClass(alwaysRun = true)
21     protected void clearTempDir() {
22        TestingUtil.recursiveFileRemove(tmpDirectory);
23        new File(tmpDirectory).mkdirs();
24     }
25  
26     LevelDBStoreConfigurationBuilder createStoreBuilder(PersistenceConfigurationBuilder loaders) {
27        return loaders.addStore(LevelDBStoreConfigurationBuilder.class).location(tmpDirectory + "/data").expiredLocation(tmpDirectory + "/expiry").clearThreshold(2);
28     }
29  }
