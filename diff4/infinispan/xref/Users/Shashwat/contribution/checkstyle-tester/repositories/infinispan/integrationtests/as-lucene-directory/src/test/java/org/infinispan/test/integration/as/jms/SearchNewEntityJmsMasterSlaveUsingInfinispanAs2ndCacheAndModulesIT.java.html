27     @Inject
28     StatisticsController stats;
29  
30     @Test
31     @InSequence(1005)
32     @OperateOnDeployment("slave-1")
33     public void secondLevelCacheShouldBeActive() throws Exception {
34        RegisteredMember cachedMember = memberRegistration.getNewMember();
35        cachedMember.setName("Johnny Cached");
36        memberRegistration.register();
37  
38        // Cache the result
39        memberRegistration.findById(cachedMember.getId());
40        memberRegistration.findById(cachedMember.getId());
41        memberRegistration.findById(cachedMember.getId());
42  
43        long secondLevelCacheMissCount = stats.getStatistics().getSecondLevelCacheMissCount();
44        assertTrue("Second level cache not enabled", secondLevelCacheMissCount > 0);
45     }
46
