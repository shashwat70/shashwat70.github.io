45     @Override
46     public void cacheManagerStarting(GlobalComponentRegistry gcr, GlobalConfiguration globalCfg) {
47        Map<Integer, AdvancedExternalizer<?>> externalizerMap = globalCfg.serialization().advancedExternalizers();
48        externalizerMap.put(ExternalizerIds.PROTOBUF_VALUE_WRAPPER, new ProtobufValueWrapper.Externalizer());
49        externalizerMap.put(ExternalizerIds.JPA_PROTOBUF_CACHE_EVENT_FILTER_CONVERTER, new JPAProtobufCacheEventFilterConverter.Externalizer());
50        externalizerMap.put(ExternalizerIds.JPA_PROTOBUF_FILTER_AND_CONVERTER, new JPAProtobufFilterAndConverter.Externalizer());
51     }

53     @Override
54     public void cacheManagerStarted(GlobalComponentRegistry gcr) {
55        EmbeddedCacheManager cacheManager = gcr.getComponent(EmbeddedCacheManager.class);
56        initProtobufMetadataManager((DefaultCacheManager) cacheManager, gcr);
57     }

59     private void initProtobufMetadataManager(DefaultCacheManager cacheManager, GlobalComponentRegistry gcr) {
60        ProtobufMetadataManager protobufMetadataManager = new ProtobufMetadataManager();
61        gcr.registerComponent(protobufMetadataManager, ProtobufMetadataManager.class);
62        registerProtobufMetadataManagerMBean(protobufMetadataManager, gcr, cacheManager.getName());
63     }
64  
65     private void registerProtobufMetadataManagerMBean(ProtobufMetadataManager protobufMetadataManager, GlobalComponentRegistry gcr, String cacheManagerName) {
66        GlobalConfiguration globalCfg = gcr.getGlobalConfiguration();
67        MBeanServer mBeanServer = JmxUtil.lookupMBeanServer(globalCfg);
68  
69        String groupName = "type=RemoteQuery,name=" + ObjectName.quote(cacheManagerName);
70        String jmxDomain = JmxUtil.buildJmxDomain(globalCfg, mBeanServer, groupName);
71        ComponentMetadataRepo metadataRepo = gcr.getComponentMetadataRepo();
72        ManageableComponentMetadata metadata = metadataRepo.findComponentMetadata(ProtobufMetadataManager.class)
73              .toManageableComponentMetadata();
74        try {
75           ResourceDMBean mBean = new ResourceDMBean(protobufMetadataManager, metadata);
76           ObjectName objName = new ObjectName(jmxDomain + ":" + groupName + ",component=" + metadata.getJmxObjectName());
77           protobufMetadataManager.setObjectName(objName);
78           JmxUtil.registerMBean(mBean, objName, mBeanServer);
79        } catch (Exception e) {
80           throw new CacheException("Unable to register ProtobufMetadataManager MBean", e);
81        }
82     }
83  
84     @Override
85     public void cacheManagerStopping(GlobalComponentRegistry gcr) {
86        unregisterProtobufMetadataManagerMBean(gcr);
87     }
88  
89     private void unregisterProtobufMetadataManagerMBean(GlobalComponentRegistry gcr) {
90        try {
91           ObjectName objName = gcr.getComponent(ProtobufMetadataManager.class).getObjectName();
92           MBeanServer mBeanServer = JmxUtil.lookupMBeanServer(gcr.getGlobalConfiguration());
93           JmxUtil.unregisterMBean(objName, mBeanServer);
94        } catch (Exception e) {
95           throw new CacheException("Unable to unregister ProtobufMetadataManager MBean", e);
96        }
97     }

99     /**
100     * Registers the remote value wrapper interceptor in the cache before it gets started.
101     */
102    @Override
103    public void cacheStarting(ComponentRegistry cr, Configuration cfg, String cacheName) {
104       if (!cacheName.equals(ProtobufMetadataManager.PROTOBUF_METADATA_CACHE_NAME)) {
105          ProtobufMetadataManager protobufMetadataManager = cr.getGlobalComponentRegistry().getComponent(ProtobufMetadataManager.class);
106 
107          // ensure the protobuf metadata cache is created
108          protobufMetadataManager.getCache();
109 
110          cr.registerComponent(new ProtobufMatcher(protobufMetadataManager.getSerializationContext()), ProtobufMatcher.class);
111 
112          if (cfg.compatibility().enabled()) {
113             cr.registerComponent(new CompatibilityReflectionMatcher(protobufMetadataManager.getSerializationContext()), CompatibilityReflectionMatcher.class);
114          }
115 
116          if (cfg.indexing().index().isEnabled() && !cfg.compatibility().enabled()) {
117             log.infof("Registering RemoteValueWrapperInterceptor for cache %s", cacheName);
118             createRemoteValueWrapperInterceptor(cr, cfg);
119          }
120       }
121    }
122 
123    private void createRemoteValueWrapperInterceptor(ComponentRegistry cr, Configuration cfg) {
124       RemoteValueWrapperInterceptor wrapperInterceptor = cr.getComponent(RemoteValueWrapperInterceptor.class);
125       if (wrapperInterceptor == null) {
126          wrapperInterceptor = new RemoteValueWrapperInterceptor();
127 
128          // Interceptor registration not needed, core configuration handling
129          // already does it for all custom interceptors - UNLESS the InterceptorChain already exists in the component registry!
130          InterceptorChain ic = cr.getComponent(InterceptorChain.class);
131 
132          ConfigurationBuilder builder = new ConfigurationBuilder().read(cfg);
133          InterceptorConfigurationBuilder interceptorBuilder = builder.customInterceptors().addInterceptor();
134          interceptorBuilder.interceptor(wrapperInterceptor);
135 
136          if (cfg.invocationBatching().enabled()) {
137             if (ic != null) ic.addInterceptorAfter(wrapperInterceptor, BatchingInterceptor.class);
138             interceptorBuilder.after(BatchingInterceptor.class);
139          } else {
140             if (ic != null) ic.addInterceptorAfter(wrapperInterceptor, InvocationContextInterceptor.class);
141             interceptorBuilder.after(InvocationContextInterceptor.class);
142          }
143          if (ic != null) {
144             cr.registerComponent(wrapperInterceptor, RemoteValueWrapperInterceptor.class);
145             cr.registerComponent(wrapperInterceptor, wrapperInterceptor.getClass().getName(), true);
146          }
147          cfg.customInterceptors().interceptors(builder.build().customInterceptors().interceptors());
148       }
149    }
150 
151    @Override
152    public void cacheStarted(ComponentRegistry cr, String cacheName) {
153       Configuration configuration = cr.getComponent(Configuration.class);
154       boolean remoteValueWrappingEnabled = configuration.indexing().index().isEnabled() && !configuration.compatibility().enabled();
155       if (!remoteValueWrappingEnabled) {
156          if (verifyChainContainsRemoteValueWrapperInterceptor(cr)) {
157             throw new IllegalStateException("It was NOT expected to find the RemoteValueWrapperInterceptor registered in the InterceptorChain as indexing was disabled, but it was found");
158          }
159          return;
160       }
161       if (!verifyChainContainsRemoteValueWrapperInterceptor(cr)) {
162          throw new IllegalStateException("It was expected to find the RemoteValueWrapperInterceptor registered in the InterceptorChain but it wasn't found");
163       }
164    }
165 
166    private boolean verifyChainContainsRemoteValueWrapperInterceptor(ComponentRegistry cr) {
167       InterceptorChain interceptorChain = cr.getComponent(InterceptorChain.class);
168       return interceptorChain.containsInterceptorType(RemoteValueWrapperInterceptor.class, true);
169    }
170 
171    @Override
172    public void cacheStopped(ComponentRegistry cr, String cacheName) {
173       Configuration cfg = cr.getComponent(Configuration.class);
174       removeRemoteIndexingInterceptorFromConfig(cfg);
175    }
176 
177    private void removeRemoteIndexingInterceptorFromConfig(Configuration cfg) {
178       ConfigurationBuilder builder = new ConfigurationBuilder();
179       CustomInterceptorsConfigurationBuilder customInterceptorsBuilder = builder.customInterceptors();
180 
181       for (InterceptorConfiguration interceptorConfig : cfg.customInterceptors().interceptors()) {
182          if (!(interceptorConfig.interceptor() instanceof RemoteValueWrapperInterceptor)) {
183             customInterceptorsBuilder.addInterceptor().read(interceptorConfig);
184          }
185       }
186 
187       cfg.customInterceptors().interceptors(builder.build().customInterceptors().interceptors());
188    }
