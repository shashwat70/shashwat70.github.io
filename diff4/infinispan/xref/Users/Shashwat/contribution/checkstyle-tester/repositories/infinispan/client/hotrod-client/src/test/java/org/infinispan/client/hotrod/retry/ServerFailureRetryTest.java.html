18  /**
19   * Test different server error situations and check how clients behave under
20   * those circumstances. Also verify whether failover is happening accordingly.
21   *
22   * @author Galder Zamarre√±o
23   * @since 4.2

25  @Test(groups = "functional", testName = "client.hotrod.retry.ServerFailureRetryTest")
26  public class ServerFailureRetryTest extends AbstractRetryTest {
27  
28     @Override
29     protected ConfigurationBuilder getCacheConfig() {
30        return hotRodCacheConfiguration(
31              getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, false));
32     }
33  
34     public void testRetryWithInfinispanSuspectException() {
35        retryExceptions(false);
36     }
37  
38     public void testRetryWithJGroupsSuspectedException() {
39        retryExceptions(true);
40     }
41  
42     private void retryExceptions(boolean throwJGroupsException) {
43        AdvancedCache<?, ?> nextCache = nextCacheToHit();
44        ErrorInducingInterceptor interceptor = new ErrorInducingInterceptor(throwJGroupsException);
45        nextCache.addInterceptor(interceptor, 1);
46        try {
47           remoteCache.put(1, "v1");
48           assertTrue(interceptor.suspectExceptionThrown);
49           assertEquals("v1", remoteCache.get(1));
50        } finally {
51           nextCache.removeInterceptor(ErrorInducingInterceptor.class);
52        }
53     }
54  
55     public void testRetryCacheStopped() {
56        // Put data in any of the cluster's default cache
57        remoteCache.put(1, "v1");
58        assertEquals("v1", remoteCache.get(1));
59        // Find out what next cluster cache to be hit and stop it
60        Cache<?, ?> cache = nextCacheToHit();
61        try {
62           cache.stop();
63           remoteCache.put(2, "v2");
64           assertEquals("v2", remoteCache.get(2));
65        } finally {
66           cache.start();
67        }
68     }
69  
70     // Listener callbacks can happen in the main thread or remote thread
71     // depending on primary owner considerations, even for replicated caches.
72     // Using an interceptor gives more flexibility by being able to put it
73     // at the top of the interceptor stack, before remote/local separation
74     public static class ErrorInducingInterceptor extends CommandInterceptor {
75        volatile boolean suspectExceptionThrown;
76        boolean throwJGroupsException;
77  
78        public ErrorInducingInterceptor(boolean throwJGroupsException) {
79           this.throwJGroupsException = throwJGroupsException;
80        }
81  
82        @Override
83        public Object visitPutKeyValueCommand(InvocationContext ctx, PutKeyValueCommand command) throws Throwable {
84           if (ctx.isOriginLocal()) {
85              suspectExceptionThrown = true;
86              throw throwJGroupsException ?
87                    new SuspectedException("Simulated suspicion")
88                    : new SuspectException("Simulated suspicion");
89           }
90           return super.visitPutKeyValueCommand(ctx, command);
91        }
92     }
