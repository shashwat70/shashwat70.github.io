14  import static org.junit.Assert.assertEquals;

18  /**
19   * Tests for keySet() method on a distributed remote cache that uses protobuf marshalling.
20   *
21   * @author anistor@redhat.com
22   */
23  @Category({ Queries.class })
24  @RunWith(Arquillian.class)
25  public class RemoteQueryKeySetIT extends RemoteQueryBaseIT {
26  
27     @InfinispanResource("remote-query")
28     protected RemoteInfinispanServer server;
29  
30     public RemoteQueryKeySetIT() {
31        super("clustered", "disttestcache");
32     }
33  
34     @Override
35     protected RemoteInfinispanServer getServer() {
36        return server;
37     }
38  
39     @Test
40     public void testDistributedKeySet() throws Exception {
41        remoteCache.put(1, createUser(1));
42        remoteCache.put(2, createUser(2));
43  
44        Set<Integer> keys = remoteCache.keySet();
45        assertNotNull(keys);
46        assertEquals(2, keys.size());
47        assertTrue(keys.contains(1));
48        assertTrue(keys.contains(2));
49     }
50  
51     private User createUser(int id) {
52        User user = new User();
53        user.setId(id);
54        user.setName("John " + id);
55        user.setSurname("Doe");
56        user.setGender(User.Gender.MALE);
57        return user;
58     }
59  }
