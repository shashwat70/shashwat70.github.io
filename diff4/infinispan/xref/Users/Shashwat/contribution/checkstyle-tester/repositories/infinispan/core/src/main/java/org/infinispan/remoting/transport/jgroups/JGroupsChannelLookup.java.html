1   package org.infinispan.remoting.transport.jgroups;
2   
3   import org.jgroups.Channel;
4   
5   import java.util.Properties;
6   
7   /**
8    * A hook to pass in a JGroups channel.  Implementations need to provide a public no-arg constructor as instances are
9    * created via reflection.
10   *
11   * @author Manik Surtani
12   * @since 4.0
13   */
14  public interface JGroupsChannelLookup {
15     /**
16      * Retrieves a JGroups channel.  Passes in all of the properties used to configure the channel.
17      * @param p properties
18      * @return a JGroups channel
19      */
20     Channel getJGroupsChannel(Properties p);
21  
22     /**
23      * @return true if the JGroupsTransport should connect the channel before using it; false if the transport
24      * should assume that the channel is connected.
25      */
26     boolean shouldConnect();
27  
28     /**
29      * @return true if the JGroupsTransport should disconnect the channel once it is done with it; false if
30      * the channel is to be left connected.
31      */
32     boolean shouldDisconnect();
33  
34     /**
35      * @return true if the JGroupsTransport should close the channel once it is done with it; false if
36      * the channel is to be left open.
37      */
38     boolean shouldClose();
39  }
