1   package org.infinispan.server.test.util;
2   
3   import org.infinispan.arquillian.core.RemoteInfinispanServer;
4   import org.infinispan.arquillian.core.RemoteInfinispanServers;
5   import org.infinispan.arquillian.model.RemoteInfinispanCache;
6   import org.infinispan.arquillian.model.RemoteInfinispanCacheManager;
7   
8   /**
9    * A tuple-style object holder containing references to Remote interfaces for JMX statistics access.
10   * 
11   * @author Michal Linhard (mlinhard@redhat.com)
12   * 
13   */
14  public class RemoteInfinispanMBeans {
15      public String serverName;
16      public String cacheName;
17      public String managerName;
18      public RemoteInfinispanServer server;
19      public RemoteInfinispanCache cache;
20      public RemoteInfinispanCacheManager manager;
21  
22      public static RemoteInfinispanMBeans create(RemoteInfinispanServers servers, String serverName, String cacheName,
23          String managerName) {
24          return create(servers.getServer(serverName), serverName, cacheName, managerName);
25      }
26  
27      public static RemoteInfinispanMBeans create(RemoteInfinispanServer server, String serverName, String cacheName,
28                                                  String managerName) {
29          RemoteInfinispanMBeans r = new RemoteInfinispanMBeans();
30          r.serverName = serverName;
31          r.cacheName = cacheName;
32          r.managerName = managerName;
33          r.server = server;
34          r.manager = r.server.getCacheManager(managerName);
35          r.cache = r.manager.getCache(cacheName);
36          return r;
37      }
38  }
