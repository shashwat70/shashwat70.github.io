80      @Before
81      public void setUp() throws IOException {
82          if (provider == null) { // initialize just once
83              provider = new MBeanServerConnectionProvider(server1.getHotrodEndpoint().getInetAddress().getHostName(), SERVER1_MGMT_PORT);
84              provider2 = new MBeanServerConnectionProvider(server2.getHotrodEndpoint().getInetAddress().getHostName(), SERVER2_MGMT_PORT);
85              Configuration conf = new ConfigurationBuilder().addServer().host(server1.getHotrodEndpoint().getInetAddress().getHostName()).port(server1
86                      .getHotrodEndpoint().getPort()).build();
87              manager = new RemoteCacheManager(conf);
88              distCache = manager.getCache();
89              mc = new MemcachedClient("UTF-8", server1.getMemcachedEndpoint().getInetAddress()
90                      .getHostName(), server1.getMemcachedEndpoint().getPort(), 10000);
91          }
92          distCache.clear();
93      }

95      @Test
96      public void testHotRodConnectionCount() throws Exception {
97  
98          // get number of current local/global connections
99          int initialLocal = Integer.parseInt(getAttribute(provider, hotRodServerMBean, "numberOfLocalConnections"));
100         int initialGlobal = Integer.parseInt(getAttribute(provider, hotRodServerMBean, "numberOfGlobalConnections"));
101         assertEquals("Number of global connections obtained from node1 and node2 is not the same", initialGlobal,
102                 Integer.parseInt(getAttribute(provider2, hotRodServerMBean, "numberOfGlobalConnections")));
103         // create another RCM and use it
104         Configuration conf = new ConfigurationBuilder().addServer().host(server1.getHotrodEndpoint().getInetAddress().getHostName()).port(server1
105                 .getHotrodEndpoint().getPort()).build();
106         RemoteCacheManager manager2 = new RemoteCacheManager(conf);
107         manager2.getCache().put("key", "value");
108 
109         // local connections increase by 1, global (in both nodes) by 2 (because we have distributed cache with 2 nodes, both nodes are accessed)
110         assertEquals(initialLocal + 1, Integer.parseInt(getAttribute(provider, hotRodServerMBean, "numberOfLocalConnections")));
111         assertEquals(initialGlobal + 2, Integer.parseInt(getAttribute(provider, hotRodServerMBean, "numberOfGlobalConnections")));
112         assertEquals(initialGlobal + 2, Integer.parseInt(getAttribute(provider2, hotRodServerMBean, "numberOfGlobalConnections")));
113     }

235         assertNotEquals(0, Integer.parseInt(getAttribute(provider, memcachedCacheStatisticsMBean, "TimeSinceReset")));
236         assertNotEquals(0, Integer.parseInt(getAttribute(provider, memcachedCacheStatisticsMBean, "ElapsedTime")));
237         assertEquals(0, Integer.parseInt(getAttribute(provider, memcachedCacheStatisticsMBean, "Misses")));
238         assertNotEquals(0, Integer.parseInt(getAttribute(provider, memcachedCacheStatisticsMBean, "RemoveHits")));
239         assertNotEquals(null, getAttribute(provider, memcachedCacheStatisticsMBean, "AverageWriteTime"));
240         assertNotEquals(0, Integer.parseInt(getAttribute(provider, memcachedCacheStatisticsMBean, "Stores")));
241         assertTrue(1.0 == Double.parseDouble(getAttribute(provider, memcachedCacheStatisticsMBean, "HitRatio")));
242         assertNotEquals(null, getAttribute(provider, memcachedCacheStatisticsMBean, "AverageReadTime"));
243     }
