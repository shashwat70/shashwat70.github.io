7   package org.hibernate.dialect;
8   
9   import org.hibernate.dialect.identity.DB2IdentityColumnSupport;
10  import org.hibernate.dialect.identity.IdentityColumnSupport;
11  import org.hibernate.dialect.unique.DefaultUniqueDelegate;
12  import org.hibernate.dialect.unique.UniqueDelegate;
13  
14  /**
15   * An SQL dialect for i. This class provides support for DB2 Universal Database for i V7R1 and
16   * later, also known as DB2/400.
17   *
18   * @author Pierrick Rouxel (pierrickrouxel)
19   */
20  public class DB2400V7R3Dialect extends DB2400Dialect {
21  
22  	private final UniqueDelegate uniqueDelegate;
23  
24  	public DB2400V7R3Dialect() {
25  		super();
26  
27  		uniqueDelegate = new DefaultUniqueDelegate(this);
28  	}
29  
30  	@Override
31  	public UniqueDelegate getUniqueDelegate() {
32  		return uniqueDelegate;
33  	}
34  
35  	@Override
36  	public boolean supportsSequences() {
37  		return true;
38  	}
39  
40  	@Override
41  	public String getQuerySequencesString() {
42  		return "select distinct sequence_name from qsys2.syssequences " +
43  				"where ( current_schema = '*LIBL' and sequence_schema in ( select schema_name from qsys2.library_list_info ) ) " +
44  				"or sequence_schema = current_schema";
45  	}
46  
47  	@Override
48  	@SuppressWarnings("deprecation")
49  	public String getLimitString(String sql, int offset, int limit) {
50  		if ( offset == 0 ) {
51  			return sql + " fetch first " + limit + " rows only";
52  		}
53  		//nest the main query in an outer select
54  		return "select * from ( select inner2_.*, rownumber() over(order by order of inner2_) as rownumber_ from ( "
55  				+ sql + " fetch first " + limit + " rows only ) as inner2_ ) as inner1_ where rownumber_ > "
56  				+ offset + " order by rownumber_";
57  	}
58  
59  	@Override
60  	public IdentityColumnSupport getIdentityColumnSupport() {
61  		return new DB2IdentityColumnSupport();
62  	}
63  }
