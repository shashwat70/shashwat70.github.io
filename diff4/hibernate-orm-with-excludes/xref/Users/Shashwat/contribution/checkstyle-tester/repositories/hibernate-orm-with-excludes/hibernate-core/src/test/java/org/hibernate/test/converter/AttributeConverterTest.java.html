232 		cfg.addAnnotatedClass( Tester4.class );
233 		cfg.setProperty( AvailableSettings.HBM2DDL_AUTO, "create-drop" );
234 		cfg.setProperty( AvailableSettings.GENERATE_STATISTICS, "true" );
235 
236 		SessionFactory sf = cfg.buildSessionFactory();
237 
238 		try {
239 			Session session = sf.openSession();
240 			session.beginTransaction();
241 			session.save( new Tester4( 1L, "steve", 200 ) );
242 			session.getTransaction().commit();
243 			session.close();
244 
245 			sf.getStatistics().clear();
246 			session = sf.openSession();
247 			session.beginTransaction();
248 			session.get( Tester4.class, 1L );
249 			session.getTransaction().commit();
250 			session.close();
251 			assertEquals( 0, sf.getStatistics().getEntityUpdateCount() );
252 
253 			session = sf.openSession();
254 			session.beginTransaction();
255 			Tester4 t4 = (Tester4) session.get( Tester4.class, 1L );
256 			t4.code = 300;
257 			session.getTransaction().commit();
258 			session.close();
259 
260 			session = sf.openSession();
261 			session.beginTransaction();

310 	@TestForIssue(jiraKey = "HHH-8866")
311 	public void testEnumConverter() {
312 		final StandardServiceRegistry ssr = new StandardServiceRegistryBuilder()
313 				.applySetting( AvailableSettings.HBM2DDL_AUTO, "create-drop" )
314 				.build();
315 
316 		try {
317 			MetadataImplementor metadata = (MetadataImplementor) new MetadataSources( ssr )
318 					.addAnnotatedClass( EntityWithConvertibleField.class )
319 					.getMetadataBuilder()
320 					.applyAttributeConverter( ConvertibleEnumConverter.class, true )
321 					.build();
322 
323 			// first lets validate that the converter was applied...
324 			PersistentClass tester = metadata.getEntityBinding( EntityWithConvertibleField.class.getName() );
325 			Property nameProp = tester.getProperty( "convertibleEnum" );
326 			SimpleValue nameValue = (SimpleValue) nameProp.getValue();
327 			Type type = nameValue.getType();
328 			assertNotNull( type );
329 			assertTyping( BasicType.class, type );
330 			if ( !AttributeConverterTypeAdapter.class.isInstance( type ) ) {
331 				fail( "AttributeConverter not applied" );
332 			}
333 			AbstractStandardBasicType basicType = assertTyping( AbstractStandardBasicType.class, type );
334 			assertTyping( EnumJavaTypeDescriptor.class, basicType.getJavaTypeDescriptor() );
335 			assertEquals( Types.VARCHAR, basicType.getSqlTypeDescriptor().getSqlType() );
336 
337 			// then lets build the SF and verify its use...
338 			final SessionFactory sf = metadata.buildSessionFactory();
339 			try {
340 				Session s = sf.openSession();
341 				s.getTransaction().begin();
342 				EntityWithConvertibleField entity = new EntityWithConvertibleField();
343 				entity.setId( "ID" );
344 				entity.setConvertibleEnum( ConvertibleEnum.VALUE );
345 				String entityID = entity.getId();
346 				s.persist( entity );
347 				s.getTransaction().commit();
348 				s.close();
349 
350 				s = sf.openSession();
351 				s.beginTransaction();
352 				entity = (EntityWithConvertibleField) s.load( EntityWithConvertibleField.class, entityID );
353 				assertEquals( ConvertibleEnum.VALUE, entity.getConvertibleEnum() );
354 				s.getTransaction().commit();
355 				s.close();
356 
357 				JavaConstantNode javaConstantNode = new JavaConstantNode();
358 				javaConstantNode.setExpectedType( type );
359 				javaConstantNode.setSessionFactory( (SessionFactoryImplementor) sf );
360 				javaConstantNode.setText( "org.hibernate.test.converter.AttributeConverterTest$ConvertibleEnum.VALUE" );
361 				final String outcome = javaConstantNode.getRenderText( (SessionFactoryImplementor) sf );
362 				assertEquals( "'VALUE'", outcome );
363 
364 				s = sf.openSession();
365 				s.beginTransaction();
366 				s.createQuery( "FROM EntityWithConvertibleField e where e.convertibleEnum = org.hibernate.test.converter.AttributeConverterTest$ConvertibleEnum.VALUE" )
367 						.list();
368 				s.getTransaction().commit();
369 				s.close();
370 
371 				s = sf.openSession();
372 				s.beginTransaction();
373 				s.delete( entity );
374 				s.getTransaction().commit();
375 				s.close();
376 			}
377 			finally {
378 				try {
379 					sf.close();
380 				}
381 				catch (Exception ignore) {
382 				}
383 			}
384 		}
385 		finally {
386 			StandardServiceRegistryBuilder.destroy( ssr );
387 		}
388 	}

581 	public static class IntegerToVarcharConverter implements AttributeConverter<Integer,String> {
582 		@Override

584 			return attribute == null ? null : attribute.toString();
585 		}
