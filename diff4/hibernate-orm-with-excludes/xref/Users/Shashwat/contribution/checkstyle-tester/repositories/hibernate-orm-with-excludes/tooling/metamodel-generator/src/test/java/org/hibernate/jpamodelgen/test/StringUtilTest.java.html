7   package org.hibernate.jpamodelgen.test;
8   
9   import org.hibernate.jpamodelgen.test.util.TestForIssue;
10  import org.hibernate.jpamodelgen.util.StringUtil;
11  import org.junit.Test;
12  
13  import static org.junit.Assert.assertEquals;
14  import static org.junit.Assert.assertFalse;
15  import static org.junit.Assert.assertTrue;
16  
17  /**
18   * @author Hardy Ferentschik
19   */
20  public class StringUtilTest {
21  	@Test
22  	public void testIsPropertyName() {
23  		assertTrue( StringUtil.isProperty( "getFoo", "java.lang.Object" ) );
24  		assertTrue( StringUtil.isProperty( "isFoo", "Boolean" ) );
25  		assertTrue( StringUtil.isProperty( "hasFoo", "java.lang.Boolean" ) );
26  
27  		assertFalse( StringUtil.isProperty( "isfoo", "void" ) );
28  		assertFalse( StringUtil.isProperty( "hasfoo", "java.lang.Object" ) );
29  
30  		assertFalse( StringUtil.isProperty( "", "java.lang.Object" ) );
31  		assertFalse( StringUtil.isProperty( null, "java.lang.Object" ) );
32  	}
33  
34  	@Test
35  	@TestForIssue(jiraKey = "METAGEN-76")
36  	public void testHashCodeNotAProperty() {
37  		assertFalse( StringUtil.isProperty( "hashCode", "Integer" ) );
38  	}
39  
40  	@Test
41  	public void testGetUpperUnderscoreCaseFromLowerCamelCase(){
42  		assertEquals("USER_PARENT_NAME", StringUtil.getUpperUnderscoreCaseFromLowerCamelCase("userParentName"));
43  	}
44  }
