19  package org.apache.struts2.views.jsp;
20  
21  import javax.servlet.http.HttpServletRequest;
22  import javax.servlet.http.HttpServletResponse;
23  
24  import org.apache.struts2.components.Component;
25  import org.apache.struts2.components.URL;
26  
27  import com.opensymphony.xwork2.util.ValueStack;
28  
29  /**
30   * @see URL
31   */
32  public class URLTag extends ContextBeanTag {
33  
34      private static final long serialVersionUID = 1722460444125206226L;
35  
36      protected String includeParams;
37      protected String scheme;
38      protected String value;
39      protected String action;
40      protected String namespace;
41      protected String method;
42      protected String encode;
43      protected String includeContext;
44      protected String escapeAmp;
45      protected String portletMode;
46      protected String windowState;
47      protected String portletUrlType;
48      protected String anchor;
49      protected String forceAddSchemeHostAndPort;
50  
51      public Component getBean(ValueStack stack, HttpServletRequest req, HttpServletResponse res) {
52          return new URL(stack, req, res);
53      }
54  
55      protected void populateParams() {
56          super.populateParams();
57  
58          URL url = (URL) component;
59          url.setIncludeParams(includeParams);
60          url.setScheme(scheme);
61          url.setValue(value);
62          url.setMethod(method);
63          url.setNamespace(namespace);
64          url.setAction(action);
65          url.setPortletMode(portletMode);
66          url.setPortletUrlType(portletUrlType);
67          url.setWindowState(windowState);
68          url.setAnchor(anchor);
69  
70          if (encode != null) {
71              url.setEncode(Boolean.valueOf(encode).booleanValue());
72          }
73          if (includeContext != null) {
74              url.setIncludeContext(Boolean.valueOf(includeContext).booleanValue());
75          }
76          if (escapeAmp != null) {
77              url.setEscapeAmp(Boolean.valueOf(escapeAmp).booleanValue());
78          }
79  	    if (forceAddSchemeHostAndPort != null) {
80              url.setForceAddSchemeHostAndPort(Boolean.valueOf(forceAddSchemeHostAndPort).booleanValue());
81          }
82      }
83  
84      public void setEncode(String encode) {
85          this.encode = encode;
86      }
87  
88      public void setIncludeContext(String includeContext) {
89          this.includeContext = includeContext;
90      }
91      
92      public void setEscapeAmp(String escapeAmp) {
93          this.escapeAmp = escapeAmp;
94      }
95  
96      public void setIncludeParams(String name) {
97          includeParams = name;
98      }
99  
100     public void setAction(String action) {
101         this.action = action;
102     }
103 
104     public void setNamespace(String namespace) {
105         this.namespace = namespace;
106     }
107 
108     public void setMethod(String method) {
109         this.method = method;
110     }
111 
112     public void setScheme(String scheme) {
113         this.scheme = scheme;
114     }
115 
116     public void setValue(String value) {
117         this.value = value;
118     }
119 
120     public void setPortletMode(String portletMode) {
121         this.portletMode = portletMode;
122     }
123 
124     public void setPortletUrlType(String portletUrlType) {
125         this.portletUrlType = portletUrlType;
126     }
127 
128     public void setWindowState(String windowState) {
129         this.windowState = windowState;
130     }
131 
132     public void setAnchor(String anchor) {
133         this.anchor = anchor;
134     }
135 
136     public void setForceAddSchemeHostAndPort(String forceAddSchemeHostAndPort) {
137         this.forceAddSchemeHostAndPort = forceAddSchemeHostAndPort;
138     }
139 }
