38  	@Test
39  	public void testEqualsFileSystemFile() {
40  		// two files with equivalent paths are equals
41  		String entry = "src/test/resources/spoon/test/api/Foo.java";
42  		assertTrue(new FileSystemFile(new File(entry)).equals(new FileSystemFile(new File("./"+entry))));
43  	}
44  
45  	@Test
46  	public void testFileSystemFolder() {
47  		String dir = "src/test/resources/spoon/test/api/";
48  		FileSystemFolder fileSystemFolder = new FileSystemFolder(new File(dir));
49  
50  		// there is one file in api
51  		assertEquals(2, fileSystemFolder.getAllFiles().size());
52  		assertEquals(2, fileSystemFolder.getAllJavaFiles().size());
53  
54  		String entry = "src/test/resources/spoon/test/api/Foo.java";
55  		FileSystemFile file = new FileSystemFile(new File(entry));
56  
57  		String entry1 = "src/test/resources/spoon/test/api/CommentedClass.java";
58  		FileSystemFile file1 = new FileSystemFile(new File(entry1));
59  
60  		assertThat(fileSystemFolder.getAllFiles().contains(file), is(true));
61  		assertThat(fileSystemFolder.getAllFiles().contains(file1), is(true));
62  	}
63  
64  	@Test
65  	public void testVirtualFolder() {
66  		String dir = "src/test/resources/spoon/test/api/";
67  		FileSystemFolder fileSystemFolder = new FileSystemFolder(new File(dir));
68  
69  		String dir2 = "src/test/resources/spoon/test/exceptions/";
70  		FileSystemFolder fileSystemFolder2 = new FileSystemFolder(new File(dir2));
71  
72  		SpoonFolder folder = new VirtualFolder();
73  		folder.addFolder(fileSystemFolder);
74  		folder.addFolder(fileSystemFolder2);
75  
76  		assertEquals(4, folder.getAllFiles().size());
77  
78  		// the README is not a Java file
79  		assertEquals(3, folder.getAllJavaFiles().size());
80  	}
81  
82  	@Test
83  	public void testFilteringFolder() {
84  		// contract: one can filter input files
85  		// the core of this test is the removeIfMatches at the end
86  
87  		// all files
88  		SpoonModelBuilder mb = new Launcher().getModelBuilder();
89  		FilteringFolder resources = new FilteringFolder();
90  		resources.addFolder(new FileSystemFolder("src/test/java/spoon/test/visibility/"));
91  		mb.addInputSource(resources);
92  		mb.build();
93  		int nbAll = mb.getFactory().getModel().getAllTypes().size();
94  		assertEquals(12, nbAll);
95  
96  		SpoonModelBuilder mb2 = new Launcher().getModelBuilder();
97  		FilteringFolder resources2 = new FilteringFolder();
98  		resources2.addFolder(new FileSystemFolder("src/test/java/spoon/test/visibility/packageprotected/"));
99  		mb2.addInputSource(resources2);
100 		mb2.build();
101 		int nbPackageProtected = mb2.getFactory().getModel().getAllTypes().size();
102 		assertEquals(2, nbPackageProtected);
103 
104 		// now the core of this test
105 		SpoonModelBuilder mb3 = new Launcher().getModelBuilder();
106 		FilteringFolder resources3 = new FilteringFolder();
107 		resources3.addFolder(new FileSystemFolder("src/test/java/spoon/test/visibility/"));
108 		// we remove a number of input resources
109 		resources3.removeAllThatMatch(".*packageprotected.*");
110 		mb3.addInputSource(resources3);
111 		mb3.build();
112 		assertEquals(nbAll - nbPackageProtected, mb3.getFactory().getModel().getAllTypes().size());
113 	}
