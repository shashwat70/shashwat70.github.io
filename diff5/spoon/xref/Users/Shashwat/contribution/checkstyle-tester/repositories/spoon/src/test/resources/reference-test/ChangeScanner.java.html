15      /**
16       * Finds the entry in neu best fitting the specified entry in old. If no entries get a score
17       * above zero, an entry is still returned.
18       * @param old EntrySorter
19       * @param neu EntrySorter
20       * @param index int
21       * @return BibtexEntry
22       */
23      private BibtexEntry bestFit(EntrySorter old, EntrySorter neu, int index) {
24          double comp = -1;
25          int found = 0;
26          loop: for (int i=0; i<neu.getEntryCount(); i++) {
27              double res = Util.compareEntriesStrictly(old.getEntryAt(index),
28              neu.getEntryAt(i));
29              if (res > comp) {
30                  comp = res;
31                  found = i;
32              }
33              if (comp > 1)
34                  break loop;
35          }
36          return neu.getEntryAt(found);
37      }
