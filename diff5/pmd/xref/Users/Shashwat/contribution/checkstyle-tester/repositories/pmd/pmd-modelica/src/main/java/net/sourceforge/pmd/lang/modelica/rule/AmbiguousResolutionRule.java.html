5   package net.sourceforge.pmd.lang.modelica.rule;
6   
7   import net.sourceforge.pmd.lang.modelica.ast.ASTName;
8   import net.sourceforge.pmd.lang.modelica.resolver.ResolutionResult;
9   import net.sourceforge.pmd.lang.modelica.resolver.ResolvableEntity;
10  
11  public class AmbiguousResolutionRule extends AbstractModelicaRule {
12      @Override
13      public Object visit(ASTName node, Object data) {
14          ResolutionResult<ResolvableEntity> candidates = node.getResolutionCandidates();
15          if (candidates.isClashed()) {
16              StringBuilder sb = new StringBuilder();
17              sb.append("Candidate resolutions: ");
18              for (ResolvableEntity candidate: candidates.getBestCandidates()) {
19                  sb.append(candidate.getDescriptiveName());
20                  sb.append(", ");
21              }
22              addViolation(data, node, sb.substring(0, sb.length() - 2));
23          }
24          return super.visit(node, data);
25      }
26  }
