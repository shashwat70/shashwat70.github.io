20  package edu.umd.cs.findbugs.detect;
21  
22  import java.util.Collections;
23  
24  import org.apache.bcel.Const;
25  import org.apache.bcel.classfile.Method;
26  
27  import edu.umd.cs.findbugs.BugInstance;
28  import edu.umd.cs.findbugs.BugReporter;
29  import edu.umd.cs.findbugs.OpcodeStack;
30  import edu.umd.cs.findbugs.Priorities;
31  import edu.umd.cs.findbugs.ba.ClassContext;
32  import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
33  
34  public class AppendingToAnObjectOutputStream extends OpcodeStackDetector {
35  
36      BugReporter bugReporter;
37  
38      public AppendingToAnObjectOutputStream(BugReporter bugReporter) {
39          this.bugReporter = bugReporter;
40      }
41  
42      @Override
43      public void visitClassContext(ClassContext classContext) {
44          if(hasInterestingClass(classContext.getJavaClass().getConstantPool(), Collections.singleton("java/io/ObjectOutputStream"))) {
45              super.visitClassContext(classContext);
46          }
47      }
48  
49      boolean sawOpenInAppendMode;
50  
51      @Override
52      public void visit(Method obj) {
53          sawOpenInAppendMode = false;
54      }
55  
56      /*
57       * (non-Javadoc)
58       *
59       * @see edu.umd.cs.findbugs.bcel.OpcodeStackDetector#sawOpcode(int)
60       */
61      @Override
62      public void sawOpcode(int seen) {
63          if (seen != Const.INVOKESPECIAL) {
64              sawOpenInAppendMode = false;
65              return;
66          }
67          String calledClassName = getClassConstantOperand();
68          String calledMethodName = getNameConstantOperand();
69          String calledMethodSig = getSigConstantOperand();
70          if (!sawOpenInAppendMode) {
71              if ("java/io/ObjectOutputStream".equals(calledClassName) && Const.CONSTRUCTOR_NAME.equals(calledMethodName)
72                      && "(Ljava/io/OutputStream;)V".equals(calledMethodSig)
73                      && stack.getStackItem(0).getSpecialKind() == OpcodeStack.Item.FILE_OPENED_IN_APPEND_MODE) {
74                  bugReporter.reportBug(new BugInstance(this, "IO_APPENDING_TO_OBJECT_OUTPUT_STREAM", Priorities.HIGH_PRIORITY)
75                  .addClassAndMethod(this).addSourceLine(this));
76              }
77              return;
78          }
79          if ("java/io/FileOutputStream".equals(calledClassName) && Const.CONSTRUCTOR_NAME.equals(calledMethodName)
80                  && ("(Ljava/io/File;Z)V".equals(calledMethodSig) || "(Ljava/lang/String;Z)V".equals(calledMethodSig))) {
81              OpcodeStack.Item item = stack.getStackItem(0);
82              Object value = item.getConstant();
83              sawOpenInAppendMode = value instanceof Integer && ((Integer) value).intValue() == 1;
84          } else if ("java/io/BufferedOutputStream".equals(calledClassName) && Const.CONSTRUCTOR_NAME.equals(calledMethodName)
85                  && "(Ljava/io/OutputStream;)V".equals(calledMethodSig)) {
86              // do nothing
87  
88          } else if ("java/io/ObjectOutputStream".equals(calledClassName) && Const.CONSTRUCTOR_NAME.equals(calledMethodName)
89                  && "(Ljava/io/OutputStream;)V".equals(calledMethodSig)) {
90              bugReporter.reportBug(new BugInstance(this, "IO_APPENDING_TO_OBJECT_OUTPUT_STREAM", Priorities.HIGH_PRIORITY)
91                  .addClassAndMethod(this).addSourceLine(this));
92              sawOpenInAppendMode = false;
93          } else {
94              sawOpenInAppendMode = false;
95          }
96  
97      }
98  
99  }
