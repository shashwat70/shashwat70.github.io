55      /**
56       * Check {@link MarshallSolarActivityFutureEstimation#get24HoursKp(AbsoluteDate)} and
57       * {@link MarshallSolarActivityFutureEstimation#getThreeHourlyKP(AbsoluteDate)} are
58       * continuous.
59       *
60       * @throws OrekitException on error.
61       */
62      @Test
63      public void testGetKp() throws OrekitException {
64          //setup
65          DTM2000InputParameters flux = getFlux();
66          final AbsoluteDate july = new AbsoluteDate(2008, 7, 1, utc);
67          final AbsoluteDate august = new AbsoluteDate(2008, 8, 1, utc);
68          final AbsoluteDate middle = july.shiftedBy(august.durationFrom(july) / 2.0);
69          final double minute = 60;
70          final AbsoluteDate before = middle.shiftedBy(-minute);
71          final AbsoluteDate after = middle.shiftedBy(+minute);
72  
73          // action + verify
74          // non-chaotic i.e. small change in input produces small change in output.
75          double kpHourlyDifference =
76                  flux.getThreeHourlyKP(before) - flux.getThreeHourlyKP(after);
77          assertThat(kpHourlyDifference, closeTo(0.0, 1e-4));
78          double kpDailyDifference = flux.get24HoursKp(before) - flux.get24HoursKp(after);
79          assertThat(kpDailyDifference, closeTo(0.0, 1e-4));
80          assertThat(flux.getThreeHourlyKP(middle), closeTo(2.18, 0.3));
81          assertThat(flux.get24HoursKp(middle), closeTo(2.18, 0.3));
82      }

84      /**
85       * Check integration error is small when integrating the same equations over the same
86       * interval.
87       *
88       * @throws OrekitException on error.
89       */
90      @Test
91      public void testWithPropagator() throws OrekitException {
92          CelestialBody sun = CelestialBodyFactory.getSun();
93          final Frame eci = FramesFactory.getGCRF();
94          final Frame ecef = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
95          AbsoluteDate date = new AbsoluteDate(2004, 1, 1, utc);
96          OneAxisEllipsoid earth = new OneAxisEllipsoid(
97                  Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
98                  Constants.WGS84_EARTH_FLATTENING,
99                  ecef);
100         Orbit orbit = new KeplerianOrbit(
101                 6378137 + 400e3, 1e-3, FastMath.toRadians(50), 0, 0, 0,
102                 PositionAngle.TRUE, eci, date, Constants.EIGEN5C_EARTH_MU);
103         final SpacecraftState ic = new SpacecraftState(orbit);
104 
105         final AbsoluteDate end = date.shiftedBy(5 * Constants.JULIAN_DAY);
106         final AbsoluteDate resetDate = date.shiftedBy(0.8 * Constants.JULIAN_DAY + 0.1);
107 
108         final SpacecraftState[] lastState = new SpacecraftState[1];
109         final OrekitStepHandler stepSaver = (interpolator, isLast) -> {
110             final AbsoluteDate start = interpolator.getPreviousState().getDate();
111             if (start.compareTo(resetDate) < 0) {
112                 lastState[0] = interpolator.getPreviousState();
113             }
114         };
115 
116         // propagate with state rest to take slightly different path
117         NumericalPropagator propagator = getNumericalPropagator(sun, earth, ic);
118         propagator.setMasterMode(stepSaver);
119         propagator.propagate(resetDate);
120         propagator.resetInitialState(lastState[0]);
121         propagator.setSlaveMode();
122         final SpacecraftState actual = propagator.propagate(end);
123 
124         // propagate straight through
125         propagator = getNumericalPropagator(sun, earth, ic);
126         propagator.resetInitialState(ic);
127         propagator.setSlaveMode();
128         final SpacecraftState expected = propagator.propagate(end);
129 
130         assertThat(actual.getPVCoordinates(), pvCloseTo(expected.getPVCoordinates(), 1.0));
131     }

133     /**
134      * Configure a numerical propagator.
135      *
136      * @param sun   Sun.
137      * @param earth Earth.
138      * @param ic    initial condition.
139      * @return a propagator.
140      * @throws OrekitException on error.
141      */
142     private NumericalPropagator getNumericalPropagator(CelestialBody sun,
143                                                        OneAxisEllipsoid earth,
144                                                        SpacecraftState ic)
145             throws OrekitException {
146         // some non-integer step size to induce truncation error in flux interpolation
147         final ODEIntegrator integrator = new ClassicalRungeKuttaIntegrator(120 + 0.1);
148         NumericalPropagator propagator = new NumericalPropagator(integrator);
149         DTM2000InputParameters flux = getFlux();
150         final Atmosphere atmosphere = new DTM2000(flux, sun, earth);
151         final IsotropicDrag satellite = new IsotropicDrag(1, 3.2);
152         propagator.addForceModel(new DragForce(atmosphere, satellite));
153 
154         propagator.setInitialState(ic);
155         propagator.setOrbitType(OrbitType.CARTESIAN);
156 
157         return propagator;
158     }

187                             msafe.getFileDate(new AbsoluteDate("2030-01-01", utc)));

258     private MarshallSolarActivityFutureEstimation loadMsafe(MarshallSolarActivityFutureEstimation.StrengthLevel strength)
259         throws OrekitException {
260 
261         MarshallSolarActivityFutureEstimation msafe =
262             new MarshallSolarActivityFutureEstimation("(?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\\p{Digit}\\p{Digit}\\p{Digit}\\p{Digit}F10\\.(?:txt|TXT)",
263                                                       strength);
264         DataProvidersManager manager = DataProvidersManager.getInstance();
265         manager.feed(msafe.getSupportedNames(), msafe);
266         return msafe;
267     }
