54      @Test
55      public void testStandard() throws OrekitException {
56  
57          final HarrisPriester hp = new HarrisPriester(sun, earth);
58  
59          // Position at 500 km height
60          final GeodeticPoint point = new GeodeticPoint(0, 0, 500000.);
61          final Vector3D pos = earth.transform(point);
62  
63          // COMPUTE DENSITY KG/M3 RHO
64          final double rho = hp.getDensity(date, pos, earthFrame);
65  
66          Assert.assertEquals(3.9237E-13, rho, 1.0e-17);
67  
68      }

70      @Test
71      public void testParameterN() throws OrekitException {
72  
73          final HarrisPriester hp = new HarrisPriester(sun, earth);
74  
75          // Position at 500 km height
76          final GeodeticPoint point = new GeodeticPoint(0, 0, 500000.);
77          final Vector3D pos = earth.transform(point);
78  
79          // COMPUTE DENSITY KG/M3 RHO
80          final double rho4 = hp.getDensity(date, pos, earthFrame);
81  
82  
83          final HarrisPriester hp2 = new HarrisPriester(sun, earth, 2);
84  
85          // COMPUTE DENSITY KG/M3 RHO
86          final double rho2 = hp2.getDensity(date, pos, earthFrame);
87  
88          final HarrisPriester hp6 = new HarrisPriester(sun, earth, 6);
89  
90          // COMPUTE DENSITY KG/M3 RHO
91          final double rho6 = hp6.getDensity(date, pos, earthFrame);
92  
93          final double c2Psi2 = 0.02163787;
94  
95          Assert.assertEquals(c2Psi2, (rho6-rho2)/(rho4-rho2) - 1., 1.e-8);
96  
97      }
