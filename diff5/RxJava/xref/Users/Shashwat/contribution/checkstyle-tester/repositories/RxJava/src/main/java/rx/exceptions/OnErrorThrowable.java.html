16  package rx.exceptions;
17  
18  import rx.plugins.RxJavaErrorHandler;
19  import rx.plugins.RxJavaPlugins;
20  
21  /**
22   * Represents a {@code Throwable} that an {@code Observable} might notify its subscribers of, but that then can
23   * be handled by an operator that is designed to recover from or react appropriately to such an error. You can
24   * recover more information from an {@code OnErrorThrowable} than is found in a typical {@code Throwable}, such
25   * as the item the {@code Observable} was trying to emit at the time the error was encountered.
26   */
27  public final class OnErrorThrowable extends RuntimeException {
28  
29      private static final long serialVersionUID = -569558213262703934L;
30  
31      private final boolean hasValue;
32      private final Object value;
33  
34      private OnErrorThrowable(Throwable exception) {
35          super(exception);
36          hasValue = false;
37          this.value = null;
38      }
39  
40      private OnErrorThrowable(Throwable exception, Object value) {
41          super(exception);
42          hasValue = true;
43          this.value = value;
44      }
45  
46      /**
47       * Get the value associated with this {@code OnErrorThrowable}
48       *
49       * @return the value associated with this {@code OnErrorThrowable} (or {@code null} if there is none)
50       */
51      public Object getValue() {
52          return value;
53      }
54  
55      /**
56       * Indicates whether or not there is a value associated with this {@code OnErrorThrowable}
57       *
58       * @return {@code true} if there is a value associated with this {@code OnErrorThrowable}, otherwise
59       *         {@code false}
60       */
61      public boolean isValueNull() {
62          return hasValue;
63      }
64  
65      /**
66       * Converts a {@link Throwable} into an {@link OnErrorThrowable}.
67       *
68       * @param t
69       *          the {@code Throwable} to convert
70       * @return an {@code OnErrorThrowable} representation of {@code t}
71       */
72      public static OnErrorThrowable from(Throwable t) {
73          Throwable cause = Exceptions.getFinalCause(t);
74          if (cause instanceof OnErrorThrowable.OnNextValue) {
75              return new OnErrorThrowable(t, ((OnNextValue) cause).getValue());
76          } else {
77              return new OnErrorThrowable(t);
78          }
79      }
80  
81      /**
82       * Adds the given item as the final cause of the given {@code Throwable}, wrapped in {@code OnNextValue}
83       * (which extends {@code RuntimeException}).
84       * 
85       * @param e
86       *          the {@link Throwable} to which you want to add a cause
87       * @param value
88       *          the item you want to add to {@code e} as the cause of the {@code Throwable}
89       * @return the same {@code Throwable} ({@code e}) that was passed in, with {@code value} added to it as a
90       *         cause
91       */
92      public static Throwable addValueAsLastCause(Throwable e, Object value) {
93          Throwable lastCause = Exceptions.getFinalCause(e);
94          if (lastCause != null && lastCause instanceof OnNextValue) {
95              // purposefully using == for object reference check
96              if (((OnNextValue) lastCause).getValue() == value) {
97                  // don't add another
98                  return e;
99              }
100         }
101         Exceptions.addCause(e, new OnNextValue(value));
102         return e;
103     }
104 
105     /**
106      * Represents an exception that was encountered while trying to emit an item from an Observable, and
107      * tries to preserve that item for future use and/or reporting.
108      */
109     public static class OnNextValue extends RuntimeException {
110 
111         private static final long serialVersionUID = -3454462756050397899L;
112 
113         private final Object value;
114 
115         /**
116          * Create an {@code OnNextValue} exception and include in its error message a string representation of
117          * the item that was intended to be emitted at the time the exception was handled.
118          *
119          * @param value
120          *         the item that the Observable was trying to emit at the time of the exception
121          */
122         public OnNextValue(Object value) {
123             super("OnError while emitting onNext value: " + renderValue(value));
124             this.value = value;
125         }
126 
127         /**
128          * Retrieve the item that the Observable was trying to emit at the time this exception occurred.
129          *
130          * @return the item that the Observable was trying to emit at the time of the exception
131          */
132         public Object getValue() {
133             return value;
134         }
135 
136         /**
137          * Render the object if it is a basic type. This avoids the library making potentially expensive
138          * or calls to toString() which may throw exceptions.
139          *
140          * If a specific behavior has been defined in the {@link RxJavaErrorHandler} plugin, some types
141          * may also have a specific rendering. Non-primitive types not managed by the plugin are rendered
142          * as the classname of the object.
143          * <p>
144          * See PR #1401 and Issue #2468 for details.
145          *
146          * @param value
147          *        the item that the Observable was trying to emit at the time of the exception
148          * @return a string version of the object if primitive or managed through error plugin,
149          *        otherwise the classname of the object
150          */
151         private static String renderValue(Object value){
152             if (value == null) {
153                 return "null";
154             }
155             if (value.getClass().isPrimitive()) {
156                 return value.toString();
157             }
158             if (value instanceof String) {
159                 return (String) value;
160             }
161             if (value instanceof Enum) {
162                 return ((Enum<?>) value).name();
163             }
164 
165             String pluggedRendering = RxJavaPlugins.getInstance().getErrorHandler().handleOnNextValueRendering(value);
166             if (pluggedRendering != null) {
167                 return pluggedRendering;
168             }
169 
170             return value.getClass().getName() + ".class";
171         }
172     }
