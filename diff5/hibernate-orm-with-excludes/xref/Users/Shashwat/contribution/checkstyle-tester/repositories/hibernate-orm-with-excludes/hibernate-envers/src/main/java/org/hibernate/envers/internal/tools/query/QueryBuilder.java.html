40  /**
41   * A class for incrementally building a HQL query.
42   *
43   * @author Adam Warski (adam at warski dot org)
44   */
45  public class QueryBuilder {
46  	private final String entityName;
47  	private final String alias;
48  
49  	/**
50  	 * For use by alias generator (in case an alias is not provided by the user).
51  	 */
52  	private final MutableInteger aliasCounter;
53  	/**
54  	 * For use by parameter generator, in {@link Parameters}. This counter must be
55  	 * the same in all parameters and sub-queries of this query.
56  	 */
57  	private final MutableInteger paramCounter;
58  	/**
59  	 * "where" parameters for this query. Each parameter element of the list for one alias from the "from" part.
60  	 */
61  	private final List<Parameters> parameters = new ArrayList<>();
62  
63  	/**
64  	 * A list of triples (from entity name, alias name, whether to select the entity).
65  	 */
66  	private final List<JoinParameter> froms;
67  	/**
68  	 * A list of triples (alias, property name, order ascending?).
69  	 */
70  	private final List<Triple<String, String, Boolean>> orders;
71  	/**
72  	 * A list of complete projection definitions: either a sole property name, or a function(property name).
73  	 */
74  	private final List<String> projections;
75  
76  	private final List<Pair<String, String>> orderFragments;
77  
78  	private final SessionFactoryImplementor sessionFactory;
79  
80  	/**
81  	 * @param entityName Main entity which should be selected.
82  	 * @param alias Alias of the entity
83  	 * @param sessionFactory Session factory
84  	 */
85  	public QueryBuilder(String entityName, String alias, SessionFactoryImplementor sessionFactory) {
86  		this( entityName, alias, new MutableInteger(), new MutableInteger(), sessionFactory );
87  	}
88  
89  	private QueryBuilder(
90  			String entityName,
91  			String alias,
92  			MutableInteger aliasCounter,
93  			MutableInteger paramCounter,
94  			SessionFactoryImplementor sessionFactory) {
95  		this.entityName = entityName;
96  		this.alias = alias;
97  		this.aliasCounter = aliasCounter;
98  		this.paramCounter = paramCounter;
99  		this.sessionFactory = sessionFactory;
100 
101 		final Parameters rootParameters = new Parameters( alias, "and", paramCounter );
102 		parameters.add( rootParameters );
103 
104 		froms = new ArrayList<>();
105 		orders = new ArrayList<>();
106 		projections = new ArrayList<>();
107 		orderFragments = new ArrayList<>();
108 
109 		addFrom( entityName, alias, true );
110 	}
111 
112 	// Only for deep copy purpose.
113 	private QueryBuilder(QueryBuilder other) {
114 		this.entityName = other.entityName;
115 		this.alias = other.alias;
116 		this.sessionFactory = other.sessionFactory;
117 		this.aliasCounter = other.aliasCounter.deepCopy();
118 		this.paramCounter = other.paramCounter.deepCopy();
119 		for (final Parameters params : other.parameters) {
120 			this.parameters.add( params.deepCopy() );
121 		}
122 
123 		froms = new ArrayList<>( other.froms );
124 		orders = new ArrayList<>( other.orders );
125 		projections = new ArrayList<>( other.projections );
126 		orderFragments = new ArrayList<>( other.orderFragments );
127 	}
128 
129 	public QueryBuilder deepCopy() {
130 		return new QueryBuilder( this );
131 	}

137 		return alias;
138 	}

140 	/**
141 	 * Add an entity from which to select.
142 	 *
143 	 * @param entityName Name of the entity from which to select.
144 	 * @param alias Alias of the entity. Should be different than all other aliases.
145 	 * @param select whether the entity should be selected
146 	 */
147 	public void addFrom(String entityName, String alias, boolean select) {
148 		CrossJoinParameter joinParameter = new CrossJoinParameter( entityName, alias, select );
149 		froms.add( joinParameter );
150 	}
151 
152 	public Parameters addJoin(JoinType joinType, String entityName, String alias, boolean select) {
153 		Parameters joinConditionParameters = new Parameters( alias, Parameters.AND, paramCounter );
154 		InnerOuterJoinParameter joinParameter = new InnerOuterJoinParameter(
155 				joinType,
156 				entityName,
157 				alias,
158 				select,
159 				joinConditionParameters
160 		);
161 		froms.add( joinParameter );
162 		return joinConditionParameters;
163 	}

165 	public String generateAlias() {
166 		return "_e" + aliasCounter.getAndIncrease();
167 	}

170 	 * @param entityName Entity name, which will be the main entity for the sub-query.

184 	public Parameters addParameters(final String alias) {
185 		final Parameters result = new Parameters( alias, Parameters.AND, paramCounter);
186 		parameters.add( result );
187 		return result;
188 	}

190 	public void addOrder(String alias, String propertyName, boolean ascending) {
191 		orders.add( Triple.make( alias, propertyName, ascending ) );
192 	}
193 
194 	public void addOrderFragment(String alias, String fragment) {
195 		orderFragments.add( Pair.make( alias, fragment ) );
196 	}
197 
198 	public void addProjection(String function, String alias, String propertyName, boolean distinct) {
199 		final String effectivePropertyName = propertyName == null ? "" : ".".concat( propertyName );
200 		if ( function == null ) {
201 			projections.add( (distinct ? "distinct " : "") + alias + effectivePropertyName );
202 		}
203 		else {
204 			projections.add(
205 					function + "(" + (distinct ? "distinct " : "") + alias +
206 					effectivePropertyName + ")"
207 			);
208 		}
209 	}

211 	/**
212 	 * Builds the given query, appending results to the given string buffer, and adding all query parameter values
213 	 * that are used to the map provided.
214 	 *
215 	 * @param sb String builder to which the query will be appended.
216 	 * @param queryParamValues Map to which name and values of parameters used in the query should be added.
217 	 */
218 	public void build(StringBuilder sb, Map<String, Object> queryParamValues) {
219 		sb.append( "select " );
220 		if ( projections.size() > 0 ) {
221 			// all projections separated with commas
222 			StringTools.append( sb, projections.iterator(), ", " );
223 		}
224 		else {
225 			// all aliases separated with commas
226 			StringTools.append( sb, getSelectAliasList().iterator(), ", " );
227 		}
228 		sb.append( " from " );
229 		// all from entities with aliases
230 		boolean first = true;
231 		for (final JoinParameter joinParameter : froms) {
232 			joinParameter.appendJoin( first, sb, queryParamValues );
233 			first = false;
234 		}
235 		// where part - rootParameters
236 		first = true;
237 		for (final Parameters params : parameters) {
238 			if (!params.isEmpty()) {
239 				if (first) {
240 					sb.append( " where " );
241 					first = false;
242 				}
243 				else {
244 					sb.append( " and " );
245 				}
246 				params.build( sb, queryParamValues );
247 			}
248 		}
249 		// orders
250 		if ( !orders.isEmpty() ) {
251 			sb.append( " order by " );
252 			StringTools.append( sb, getOrderList().iterator(), ", " );
253 		}
254 		else if ( !orderFragments.isEmpty() ) {
255 			sb.append( " order by " );
256 
257 			final Iterator<Pair<String, String>> fragmentIterator = orderFragments.iterator();
258 			while( fragmentIterator.hasNext() ) {
259 				final Pair<String, String> fragment = fragmentIterator.next();
260 				final OrderByTranslation orderByFragmentTranslation = Template.translateOrderBy(
261 						fragment.getSecond(),
262 						new ColumnMapper() {
263 							@Override
264 							public SqlValueReference[] map(String reference) throws HibernateException {
265 								return new SqlValueReference[ 0 ];
266 							}
267 						},
268 						sessionFactory,
269 						sessionFactory.getJdbcServices().getDialect(),
270 						sessionFactory.getSqlFunctionRegistry()
271 				);
272 
273 				sb.append( orderByFragmentTranslation.injectAliases( new QueryOrderByAliasResolver( fragment.getFirst() ) ) );
274 				if ( fragmentIterator.hasNext() ) {
275 					sb.append( ", " );
276 				}
277 			}
278 		}
279 	}

281 	private class QueryOrderByAliasResolver implements OrderByAliasResolver {
282 		private String alias;
283 		public QueryOrderByAliasResolver(String alias) {
284 			this.alias = alias;
285 		}
286 
287 		@Override
288 		public String resolveTableAlias(String columnReference) {
289 			return alias;
290 		}

293 	private List<String> getSelectAliasList() {
294 		final List<String> aliasList = new ArrayList<>();
295 		for ( JoinParameter from : froms ) {
296 			if ( from.isSelect() ) {
297 				aliasList.add( from.getAlias() );
298 			}
299 		}
300 
301 		return aliasList;
302 	}
303 
304 	public String getRootAlias() {
305 		return alias;
306 	}
307 
308 	private List<String> getOrderList() {
309 		final List<String> orderList = new ArrayList<>();
310 		for ( Triple<String, String, Boolean> order : orders ) {
311 			orderList.add( order.getFirst() + "." + order.getSecond() + " " + (order.getThird() ? "asc" : "desc") );
312 		}
313 
314 		return orderList;
315 	}

317 	public Query toQuery(Session session) {
318 		final StringBuilder querySb = new StringBuilder();
319 		final Map<String, Object> queryParamValues = new HashMap<>();
320 
321 		build( querySb, queryParamValues );
322 
323 		final Query query = session.createQuery( querySb.toString() );
324 		for ( Map.Entry<String, Object> paramValue : queryParamValues.entrySet() ) {
325 			if ( paramValue.getValue() instanceof RevisionType ) {
326 				// this is needed when the ClassicQueryTranslatorFactory is used
327 				query.setParameter(
328 						paramValue.getKey(),
329 						paramValue.getValue(),
330 						new CustomType( new RevisionTypeType() )
331 				);
332 			}
333 			else {
334 				query.setParameter( paramValue.getKey(), paramValue.getValue() );
335 			}
336 		}
337 		return query;
338 	}

340 	private abstract static class JoinParameter {
341 
342 		private final String alias;
343 		private final boolean select;
344 
345 		protected JoinParameter(String alias, boolean select) {
346 			this.alias = alias;
347 			this.select = select;
348 		}
349 
350 		public String getAlias() {
351 			return alias;
352 		}
353 
354 		public boolean isSelect() {
355 			return select;
356 		}
357 
358 		public abstract void appendJoin(boolean firstFromElement, StringBuilder builder, Map<String, Object> queryParamValues);
359 
360 	}
361 
362 	private static class CrossJoinParameter extends JoinParameter {
363 
364 		private final String entityName;
365 
366 		public CrossJoinParameter(String entityName, String alias, boolean select) {
367 			super( alias, select );
368 			this.entityName = entityName;
369 		}
370 
371 		@Override
372 		public void appendJoin(boolean firstFromElement, StringBuilder builder, Map<String, Object> queryParamValues) {
373 			if ( !firstFromElement ) {
374 				builder.append( ", " );
375 			}
376 			builder.append( entityName ).append( ' ' ).append( getAlias() );
377 		}
378 
379 	}
380 
381 	private static class InnerOuterJoinParameter extends JoinParameter {
382 
383 		private final JoinType joinType;
384 		private final String entityName;
385 		private final Parameters joinConditionParameters;
386 
387 		public InnerOuterJoinParameter(
388 				JoinType joinType,
389 				String entityName,
390 				String alias,
391 				boolean select,
392 				Parameters joinConditionParameters) {
393 			super(alias, select);
394 			this.joinType = joinType;
395 			this.entityName = entityName;
396 			this.joinConditionParameters = joinConditionParameters;
397 		}
398 
399 		@Override
400 		public void appendJoin(boolean firstFromElement, StringBuilder builder, Map<String, Object> queryParamValues) {
401 			if (firstFromElement) {
402 				throw new IllegalArgumentException( "An inner/outer join cannot come as first 'from element'" );
403 			}
404 			builder.append( ' ' ).append( joinType.name()
405 					.toLowerCase( Locale.US ) ).append( " join " )
406 					.append( entityName ).append( ' ' )
407 					.append( getAlias() ).append( " on " );
408 			joinConditionParameters.build( builder, queryParamValues );
409 		}
