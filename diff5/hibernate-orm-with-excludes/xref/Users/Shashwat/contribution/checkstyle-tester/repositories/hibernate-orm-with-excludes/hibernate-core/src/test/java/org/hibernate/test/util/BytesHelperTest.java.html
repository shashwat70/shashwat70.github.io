7   package org.hibernate.test.util;
8   
9   import static org.junit.Assert.assertArrayEquals;
10  import static org.junit.Assert.assertEquals;
11  
12  import org.hibernate.internal.util.BytesHelper;
13  import org.hibernate.testing.junit4.BaseUnitTestCase;
14  import org.junit.Test;
15  
16  /**
17   * @author Benoit W
18   */
19  public class BytesHelperTest extends BaseUnitTestCase {
20  
21  	@Test
22  	public void testAsLongNullArray() {
23  		assertEquals(0, BytesHelper.asLong(null, 0));
24  	}
25  	
26  	@Test(expected=IllegalArgumentException.class)
27  	public void testAsLongArrayTooSmall() {
28  		byte[] src = new byte[16];
29  		assertEquals(0, BytesHelper.asLong(src, 9));
30  	}
31  	
32  	@Test
33  	public void testAsLong() {
34  		byte[] src = new byte[] {-92, -120, -59, -64, 97, 55, -41, -55, 64, -43, 20, 109, -7, -95, 77, -115};
35  		assertEquals(-6590800624601278519L, BytesHelper.asLong(src, 0));
36  		assertEquals(4671662651038846349L, BytesHelper.asLong(src, 8));
37  	}
38  	
39  	@Test
40  	public void testfromLong() {
41  		byte[] expected = new byte[] {-92, -120, -59, -64, 97, 55, -41, -55, 64, -43, 20, 109, -7, -95, 77, -115};
42  		byte[] dest = new byte[16];
43  		BytesHelper.fromLong(-6590800624601278519L, dest, 0);
44  		BytesHelper.fromLong(4671662651038846349L, dest, 8);
45  		assertArrayEquals(expected, dest);
46  	}
47  }
