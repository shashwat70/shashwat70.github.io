7   package org.hibernate.jpa.test.graphs;
8   
9   import java.util.Set;
10  
11  import javax.persistence.CascadeType;
12  import javax.persistence.Entity;
13  import javax.persistence.GeneratedValue;
14  import javax.persistence.Id;
15  import javax.persistence.JoinColumn;
16  import javax.persistence.JoinTable;
17  import javax.persistence.ManyToMany;
18  import javax.persistence.NamedAttributeNode;
19  import javax.persistence.NamedEntityGraph;
20  import javax.persistence.NamedEntityGraphs;
21  import javax.persistence.NamedQueries;
22  import javax.persistence.NamedQuery;
23  import javax.persistence.UniqueConstraint;
24  
25  import org.hibernate.annotations.Cache;
26  import org.hibernate.annotations.CacheConcurrencyStrategy;
27  
28  @Entity
29  @Cache(usage=CacheConcurrencyStrategy.READ_WRITE)
30  @NamedEntityGraphs({
31  	@NamedEntityGraph(
32  		name = "Student.Full",
33  		attributeNodes = {
34  			@NamedAttributeNode(value = Student_.COURSES)
35  		}
36  	)
37  })
38  @NamedQueries({
39  	@NamedQuery(name="LIST_OF_STD", query="select std from Student std")
40  })
41  public class Student {
42  	@Id
43  	private int id;
44  	
45  	private String name;
46  	
47  	@ManyToMany(cascade=CascadeType.PERSIST)
48  	@JoinTable(
49  		name="STUDENT_COURSES",
50  		joinColumns=@JoinColumn(referencedColumnName="ID", name="STUDENT_ID"),
51  		inverseJoinColumns=@JoinColumn(referencedColumnName="ID", name="COURSE_ID"),
52  		uniqueConstraints={@UniqueConstraint(columnNames={"STUDENT_ID", "COURSE_ID"})}
53  	)
54  	private Set<Course> courses;
55  
56  	public int getId() {
57  		return id;
58  	}
59  
60  	public void setId(int id) {
61  		this.id = id;
62  	}
63  
64  	public Set<Course> getCourses() {
65  		return courses;
66  	}
67  
68  	public void setCourses(Set<Course> courses) {
69  		this.courses = courses;
70  	}
71  
72  	public String getName() {
73  		return name;
74  	}
75  
76  	public void setName(String name) {
77  		this.name = name;
78  	}
79  
80  	@Override
81  	public String toString() {
82  		return "Student [name=" + name + "]";
83  	}
84  }
