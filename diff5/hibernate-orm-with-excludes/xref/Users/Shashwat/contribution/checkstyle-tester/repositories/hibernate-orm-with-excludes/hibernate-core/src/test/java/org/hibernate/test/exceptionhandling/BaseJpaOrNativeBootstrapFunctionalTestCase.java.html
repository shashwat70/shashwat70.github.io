103 	public void buildSessionOrEntityManagerFactory() {
104 		switch ( bootstrapMethod ) {
105 			case JPA:
106 				buildEntityManagerFactory();

108 			case NATIVE:
109 				buildSessionFactory();
110 				break;
111 		}
112 
113 		afterSessionOrEntityManagerFactoryBuilt();

116 	private void buildEntityManagerFactory() {
117 		log.trace( "Building EntityManagerFactory" );
118 
119 		Properties properties = buildProperties();
120 		ArrayList<Class> classes = new ArrayList<Class>();
121 
122 		classes.addAll( Arrays.asList( getAnnotatedClasses() ) );
123 		properties.put( AvailableSettings.LOADED_CLASSES, classes );
124 
125 		sessionFactory =  Bootstrap.getEntityManagerFactoryBuilder(
126 				buildPersistenceUnitDescriptor(),
127 				properties
128 		).build().unwrap( SessionFactoryImplementor.class );
129 
130 		serviceRegistry = (StandardServiceRegistryImpl) sessionFactory.getServiceRegistry()
131 				.getParentServiceRegistry();
132 	}
133 
134 	private void buildSessionFactory() {
135 		// for now, build the configuration to get all the property settings
136 		Configuration configuration = new Configuration();
137 		configuration.setProperties( buildProperties() );
138 
139 		Class<?>[] annotatedClasses = getAnnotatedClasses();
140 		if ( annotatedClasses != null ) {
141 			for ( Class<?> annotatedClass : annotatedClasses ) {
142 				configuration.addAnnotatedClass( annotatedClass );
143 			}
144 		}
145 
146 		BootstrapServiceRegistry bootRegistry = buildBootstrapServiceRegistry();
147 		serviceRegistry = buildServiceRegistry( bootRegistry, configuration );
148 		sessionFactory = ( SessionFactoryImplementor ) configuration.buildSessionFactory( serviceRegistry );
149 
150 		afterSessionOrEntityManagerFactoryBuilt();
151 	}

260 	private StandardServiceRegistryImpl buildServiceRegistry(BootstrapServiceRegistry bootRegistry, Configuration configuration) {
261 		Properties properties = new Properties();
262 		properties.putAll( configuration.getProperties() );
263 		ConfigurationHelper.resolvePlaceHolders( properties );
264 
265 		StandardServiceRegistryBuilder cfgRegistryBuilder = configuration.getStandardServiceRegistryBuilder();
266 
267 		StandardServiceRegistryBuilder registryBuilder = new StandardServiceRegistryBuilder( bootRegistry, cfgRegistryBuilder.getAggregatedCfgXml() )
268 				.applySettings( properties );
269 
270 		return (StandardServiceRegistryImpl) registryBuilder.build();
271 	}
272 
273 	private Properties buildProperties() {
274 		Properties properties = Environment.getProperties();
275 
276 		properties.put( org.hibernate.cfg.AvailableSettings.CACHE_REGION_FACTORY, CachingRegionFactory.class.getName() );
277 		for ( Map.Entry<Class, String> entry : getCachedClasses().entrySet() ) {
278 			properties.put( AvailableSettings.CLASS_CACHE_PREFIX + "." + entry.getKey().getName(), entry.getValue() );
279 		}
280 		for ( Map.Entry<String, String> entry : getCachedCollections().entrySet() ) {
281 			properties.put( AvailableSettings.COLLECTION_CACHE_PREFIX + "." + entry.getKey(), entry.getValue() );
282 		}
283 
284 		configure( properties );
285 
286 		if ( createSchema() ) {
287 			properties.put( org.hibernate.cfg.AvailableSettings.HBM2DDL_AUTO, "create-drop" );
288 		}
289 		properties.put( org.hibernate.cfg.AvailableSettings.USE_NEW_ID_GENERATOR_MAPPINGS, "true" );
290 		properties.put( org.hibernate.cfg.AvailableSettings.DIALECT, getDialect().getClass().getName() );
291 
292 		return properties;
293 	}

327 	@AfterClassOnce
328 	@SuppressWarnings( {"UnusedDeclaration"})
329 	protected void releaseSessionFactory() {
330 		if ( sessionFactory == null ) {
331 			return;
332 		}
333 		sessionFactory.close();
334 		sessionFactory = null;
335 		if ( serviceRegistry != null ) {
336 			if ( serviceRegistry.isActive() ) {
337 				try {
338 					serviceRegistry.destroy();
339 				}
340 				catch (Exception ignore) {
341 				}
342 				fail( "StandardServiceRegistry was not closed down as expected" );
343 			}
344 		}
345 		serviceRegistry=null;
346 	}
347 
348 	private void completeStrayTransaction() {
349 		if ( session == null ) {
350 			// nothing to do
351 			return;
352 		}
353 
354 		if ( ( (SessionImplementor) session ).isClosed() ) {
355 			// nothing to do
356 			return;
357 		}
358 
359 		if ( !session.isConnected() ) {
360 			// nothing to do
361 			return;
362 		}
363 
364 		final TransactionCoordinator.TransactionDriver tdc =
365 				( (SessionImplementor) session ).getTransactionCoordinator().getTransactionDriverControl();
366 
367 		if ( tdc.getStatus().canRollback() ) {
368 			session.getTransaction().rollback();
369 		}
370 		session.close();
371 	}
372 
373 	private void cleanupSession() {
374 		if ( session != null && ! ( (SessionImplementor) session ).isClosed() ) {
375 			session.close();
376 		}
377 		session = null;
378 	}
