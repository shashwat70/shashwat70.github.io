116             ServiceRegistry standardRegistry =
117                     new StandardServiceRegistryBuilder().build();

119             MetadataSources sources = new MetadataSources( standardRegistry );

124             // above is preferred
125             // MetadataSources sources = new MetadataSources();
126 
127             // add a class using JPA/Hibernate annotations for mapping
128             sources.addAnnotatedClass( MyEntity.class );
129 
130             // add the name of a class using JPA/Hibernate annotations for mapping.
131             // differs from above in that accessing the Class is deferred which is
132             // important if using runtime bytecode-enhancement
133             sources.addAnnotatedClassName( "org.hibernate.example.Customer" );
134 
135             // Read package-level metadata.
136             sources.addPackage( "hibernate.example" );
137 
138             // Read package-level metadata.
139             sources.addPackage( MyEntity.class.getPackage() );
140 
141             // Adds the named hbm.xml resource as a source: which performs the
142             // classpath lookup and parses the XML
143             sources.addResource( "org/hibernate/example/Order.hbm.xml" );
144 
145             // Adds the named JPA orm.xml resource as a source: which performs the
146             // classpath lookup and parses the XML
147             sources.addResource( "org/hibernate/example/Product.orm.xml" );
148 
149             // Read all mapping documents from a directory tree.
150             // Assumes that any file named *.hbm.xml is a mapping document.
151             sources.addDirectory( new File( ".") );
152 
153             // Read mappings from a particular XML file
154             sources.addFile( new File( "./mapping.xml") );
155 
156             // Read all mappings from a jar file.
157             // Assumes that any file named *.hbm.xml is a mapping document.
158             sources.addJar( new File( "./entities.jar") );
159 
160             // Read a mapping as an application resource using the convention that a class named foo.bar.MyEntity is
161             // mapped by a file named foo/bar/MyEntity.hbm.xml which can be resolved as a classpath resource.
162             sources.addClass( MyEntity.class );
163             //end::bootstrap-bootstrap-native-registry-MetadataSources-example[]

338     //tag::bootstrap-event-listener-registration-example[]
339     public class MyIntegrator implements org.hibernate.integrator.spi.Integrator {
340 
341         @Override
342         public void integrate(
343                 Metadata metadata,
344                 SessionFactoryImplementor sessionFactory,
345                 SessionFactoryServiceRegistry serviceRegistry) {
346 
347             // As you might expect, an EventListenerRegistry is the thing with which event
348             // listeners are registered
349             // It is a service so we look it up using the service registry
350             final EventListenerRegistry eventListenerRegistry =
351                 serviceRegistry.getService( EventListenerRegistry.class );
352 
353             // If you wish to have custom determination and handling of "duplicate" listeners,
354             // you would have to add an implementation of the
355             // org.hibernate.event.service.spi.DuplicationStrategy contract like this
356             eventListenerRegistry.addDuplicationStrategy( new CustomDuplicationStrategy() );
357 
358             // EventListenerRegistry defines 3 ways to register listeners:
359 
360             // 1) This form overrides any existing registrations with
361             eventListenerRegistry.setListeners( EventType.AUTO_FLUSH,
362                                                 DefaultAutoFlushEventListener.class );
363 
364             // 2) This form adds the specified listener(s) to the beginning of the listener chain
365             eventListenerRegistry.prependListeners( EventType.PERSIST,
366                                                     DefaultPersistEventListener.class );
367 
368             // 3) This form adds the specified listener(s) to the end of the listener chain
369             eventListenerRegistry.appendListeners( EventType.MERGE,
370                                                    DefaultMergeEventListener.class );
371         }
372 
373         @Override
374         public void disintegrate(
375                 SessionFactoryImplementor sessionFactory,
376                 SessionFactoryServiceRegistry serviceRegistry) {
377 
378         }
379     }
