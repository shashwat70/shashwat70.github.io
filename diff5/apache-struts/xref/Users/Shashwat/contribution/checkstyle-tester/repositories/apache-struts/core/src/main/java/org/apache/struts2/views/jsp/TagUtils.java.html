42  public class TagUtils {
43  
44      public static ValueStack getStack(PageContext pageContext) {
45          HttpServletRequest req = (HttpServletRequest) pageContext.getRequest();
46          ValueStack stack = (ValueStack) req.getAttribute(ServletActionContext.STRUTS_VALUESTACK_KEY);
47  
48          if (stack == null) {
49  
50              HttpServletResponse res = (HttpServletResponse) pageContext.getResponse();
51              Dispatcher du = Dispatcher.getInstance();
52              if (du == null) {
53                  throw new ConfigurationException("The Struts dispatcher cannot be found.  This is usually caused by "+
54                          "using Struts tags without the associated filter. Struts tags are only usable when the request "+
55                          "has passed through its servlet filter, which initializes the Struts dispatcher needed for this tag.");
56              }
57              stack = du.getContainer().getInstance(ValueStackFactory.class).createValueStack();
58  
59              HttpParameters params = HttpParameters.create(req.getParameterMap()).build();
60  
61              Map<String, Object> extraContext = du.createContextMap(new RequestMap(req),
62                      params,
63                      new SessionMap(req),
64                      new ApplicationMap(pageContext.getServletContext()),
65                      req,
66                      res);
67              extraContext.put(ServletActionContext.PAGE_CONTEXT, pageContext);
68              stack.getContext().putAll(extraContext);
69              req.setAttribute(ServletActionContext.STRUTS_VALUESTACK_KEY, stack);
70  
71              // also tie this stack/context to the ThreadLocal
72              ActionContext.setContext(new ActionContext(stack.getContext()));
73          } else {
74              // let's make sure that the current page context is in the action context
75              Map<String, Object> context = stack.getContext();
76              context.put(ServletActionContext.PAGE_CONTEXT, pageContext);
77  
78              AttributeMap attrMap = new AttributeMap(context);
79              context.put("attr", attrMap);
80          }
81  
82          return stack;
83      }
84  
85      public static String buildNamespace(ActionMapper mapper, ValueStack stack, HttpServletRequest request) {
86          ActionContext context = new ActionContext(stack.getContext());
87          ActionInvocation invocation = context.getActionInvocation();
88  
89          if (invocation == null) {
90              ActionMapping mapping = mapper.getMapping(request,
91                      Dispatcher.getInstance().getConfigurationManager());
92  
93              if (mapping != null) {
94                  return mapping.getNamespace();
95              } else {
96                  // well, if the ActionMapper can't tell us, and there is no existing action invocation,
97                  // let's just go with a default guess that the namespace is the last the path minus the
98                  // last part (/foo/bar/baz.xyz -> /foo/bar)
99  
100                 String path = RequestUtils.getServletPath(request);
101                 return path.substring(0, path.lastIndexOf('/'));
102             }
103         } else {
104             return invocation.getProxy().getNamespace();
105         }
106     }
