1   package org.infinispan.query.impl.externalizers;
2   
3   import java.io.IOException;
4   import java.io.ObjectInput;
5   import java.io.ObjectOutput;
6   import java.util.Set;
7   
8   import org.apache.lucene.search.Sort;
9   import org.apache.lucene.search.SortField;
10  import org.infinispan.commons.io.UnsignedNumeric;
11  import org.infinispan.commons.marshall.AbstractExternalizer;
12  import org.infinispan.commons.util.Util;
13  
14  
15  public class LuceneSortExternalizer extends AbstractExternalizer<Sort> {
16  
17     @Override
18     public Set<Class<? extends Sort>> getTypeClasses() {
19        return Util.<Class<? extends Sort>>asSet(Sort.class);
20     }
21  
22     @Override
23     public Sort readObject(final ObjectInput input) throws IOException, ClassNotFoundException {
24        final int count = UnsignedNumeric.readUnsignedInt(input);
25        SortField[] sortfields = new SortField[count];
26        for (int i=0; i<count; i++) {
27           sortfields[i] = LuceneSortFieldExternalizer.readObjectStatic(input);
28        }
29        Sort sort = new Sort();
30        sort.setSort(sortfields);
31        return sort;
32     }
33  
34     @Override
35     public void writeObject(final ObjectOutput output, final Sort sort) throws IOException {
36        final SortField[] sortFields = sort.getSort();
37        final int count = sortFields.length;
38        UnsignedNumeric.writeUnsignedInt(output, count);
39        for (int i=0; i<count; i++) {
40           LuceneSortFieldExternalizer.writeObjectStatic(output, sortFields[i]);
41        }
42     }
43  
44     @Override
45     public Integer getId() {
46        return ExternalizerIds.LUCENE_SORT;
47     }
48  
49  }
