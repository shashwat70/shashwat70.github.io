13     public void testSessionExpiration() throws Exception {
14        Interpreter interpreter = new Interpreter();
15        interpreter.initialize(this.cacheManager, TIME_SERVICE);
16        interpreter.setSessionTimeout(500);
17        interpreter.setSessionReaperWakeupInterval(1000);
18        interpreter.start();
19  
20        try {
21           String sessionId = interpreter.createSessionId(null);
22           Thread.sleep(1500);
23           Map<String, String> response = interpreter.execute(sessionId, "");
24           assert response.containsKey("ERROR");
25        } finally {
26           interpreter.stop();
27        }
28  
29     }
30
