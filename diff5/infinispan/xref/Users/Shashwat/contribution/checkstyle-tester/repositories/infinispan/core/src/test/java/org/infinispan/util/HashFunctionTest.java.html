1   package org.infinispan.util;
2   
3   import org.infinispan.commons.hash.Hash;
4   import org.infinispan.commons.hash.MurmurHash2;
5   import org.infinispan.commons.hash.MurmurHash2Compat;
6   import org.infinispan.commons.hash.MurmurHash3;
7   import org.infinispan.test.AbstractInfinispanTest;
8   import org.testng.annotations.Test;
9   
10  @Test(testName = "util.HashFunctionTest", groups = "unit")
11  public class HashFunctionTest extends AbstractInfinispanTest {
12  
13     public void testMurmurHash2Consistency() {
14        testHashConsistency(new MurmurHash2());
15     }
16  
17     public void testMurmurHash2CompatConsistency() {
18        testHashConsistency(new MurmurHash2Compat());
19     }
20  
21     public void testMurmurHash3Consistency() {
22        testHashConsistency(MurmurHash3.getInstance());
23     }
24  
25     private void testHashConsistency(Hash hash) {
26        Object o = new Object();
27        int i1 = hash.hash(o);
28        int i2 = hash.hash(o);
29        int i3 = hash.hash(o);
30  
31        assert i1 == i2: "i1 and i2 are not the same: " + i1 + ", " + i2;
32        assert i3 == i2: "i3 and i2 are not the same: " + i2 + ", " + i3;
33        assert i1 == i3: "i1 and i3 are not the same: " + i1 + ", " + i3;
34     }
35  
36  }
