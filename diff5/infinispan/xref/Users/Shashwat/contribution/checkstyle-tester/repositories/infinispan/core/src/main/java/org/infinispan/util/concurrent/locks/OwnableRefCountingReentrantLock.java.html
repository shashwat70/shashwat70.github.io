1   package org.infinispan.util.concurrent.locks;
2   
3   import org.infinispan.util.concurrent.locks.containers.OwnableReentrantPerEntryLockContainer;
4   
5   import java.util.concurrent.atomic.AtomicInteger;
6   
7   /**
8    * A version of {@link OwnableReentrantLock} that has a reference counter, and implements {@link RefCountingLock}.
9    * Used with a lock-per-entry container, in this case the {@link OwnableReentrantPerEntryLockContainer}.
10   *
11   * @author Manik Surtani
12   * @since 5.2
13   * @see OwnableReentrantPerEntryLockContainer
14   */
15  public class OwnableRefCountingReentrantLock extends OwnableReentrantLock implements RefCountingLock {
16     private final AtomicInteger references = new AtomicInteger(1);
17  
18     @Override
19     public AtomicInteger getReferenceCounter() {
20        return references;
21     }
22  
23     @Override
24     public String toString() {
25        return super.toString() + "[References: "+references.toString()+"]";
26     }
27  }
