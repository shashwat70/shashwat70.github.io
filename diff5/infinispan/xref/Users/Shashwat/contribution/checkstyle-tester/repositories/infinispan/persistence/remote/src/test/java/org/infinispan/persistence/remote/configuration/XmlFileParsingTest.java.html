7   import org.infinispan.configuration.cache.StoreConfiguration;

16  @Test(groups = "unit", testName = "persistence.remote.configuration.XmlFileParsingTest")
17  public class XmlFileParsingTest extends AbstractInfinispanTest {
18  
19     private EmbeddedCacheManager cacheManager;
20  
21     @AfterMethod
22     public void cleanup() {
23        TestingUtil.killCacheManagers(cacheManager);
24     }
25  
26     public void testRemoteCacheStore() throws Exception {
27        String config = InfinispanStartTag.LATEST +
28              "<cache-container default-cache=\"default\">" +
29              "   <local-cache name=\"default\">\n" +
30              "     <persistence>\n" +
31              "       <remote-store xmlns=\"urn:infinispan:config:store:remote:"+ InfinispanStartTag.LATEST.majorMinor()+"\" >\n" +
32              "         <remote-server host=\"one\" />\n" +
33              "         <remote-server host=\"two\" />\n" +
34              "         <connection-pool max-active=\"10\" exhausted-action=\"CREATE_NEW\" />\n" +
35              "         <async-executor>\n" +
36              "             <property name=\"maxThreads\">4</property>" +
37              "         </async-executor>\n" +
38              "         <write-behind/>\n" +
39              "       </remote-store>\n" +
40              "     </persistence>\n" +
41              "   </local-cache>\n" +
42              "</cache-container>" +
43              TestingUtil.INFINISPAN_END_TAG;
44  
45        RemoteStoreConfiguration store = (RemoteStoreConfiguration) buildCacheManagerWithCacheStore(config);
46        assert store.servers().size() == 2;
47        assert store.connectionPool().exhaustedAction() == ExhaustedAction.CREATE_NEW;
48        assert store.asyncExecutorFactory().properties().getIntProperty("maxThreads", 0) == 4;
49        assert store.async().enabled();
50     }
51  
52     private StoreConfiguration buildCacheManagerWithCacheStore(final String config) throws IOException {
53        InputStream is = new ByteArrayInputStream(config.getBytes());
54        cacheManager = TestCacheManagerFactory.fromStream(is);
55        assert cacheManager.getDefaultCacheConfiguration().persistence().stores().size() == 1;
56        return cacheManager.getDefaultCacheConfiguration().persistence().stores().get(0);
57     }
