1   package org.infinispan.checkstyle.filters;
2   
3   import java.io.File;
4   import java.util.List;
5   
6   import com.puppycrawl.tools.checkstyle.api.AbstractFileSetCheck;
7   
8   /**
9    * Use a simple CheckStyle rule to make sure no copyright templates are being used:
10   * Infinispan uses a single copyright file which can be found in the root of the project.
11   *
12   * @author Sanne Grinovero
13   */
14  public class HeadersNoCopyrightCheck extends AbstractFileSetCheck {
15  
16     @Override
17     protected void processFiltered(File aFile, List<String> aLines) {
18        final String fileName = aFile.getName();
19        if (fileName != null && !fileName.endsWith(".java")) {
20           //Not a Java source file, skip it.
21           return;
22        }
23        else if ("package-info.java".equals(fileName)) {
24           //package-info files don't necessarily start with "package"
25           return;
26        }
27        else if (!aLines.isEmpty()) {
28           final String firstLine = aLines.get(0);
29           if (firstLine!=null && !firstLine.startsWith("package ")) {
30              log(1, "Java files should start with \''package \''. Infinispan doesn\''t use bulky copyright headers!");
31           }
32        }
33     }
34  
35  }
