1   package org.infinispan.cli;
2   
3   import org.infinispan.cli.commands.ProcessedCommand;
4   import org.testng.annotations.Test;
5   
6   @Test(groups="functional", testName="cli.shell.ProcessedCommandTest")
7   public class ProcessedCommandTest {
8   
9      public void testArgumentParsing() {
10        ProcessedCommand pc = new ProcessedCommand("cmd abc");
11        assert "cmd".equals(pc.getCommand());
12  
13        assert pc.getArguments().size()==1;
14  
15        assert "abc".equals(pc.getArguments().get(0).getValue());
16     }
17  
18     public void testQuotedArgumentParsing() {
19        ProcessedCommand pc = new ProcessedCommand("cmd \"abc\" \"def\"");
20        assert "cmd".equals(pc.getCommand());
21  
22        assert pc.getArguments().size()==2;
23  
24        assert "abc".equals(pc.getArguments().get(0).getValue());
25        assert "def".equals(pc.getArguments().get(1).getValue());
26     }
27  
28     public void testMixedArgumentParsing() {
29        ProcessedCommand pc = new ProcessedCommand("cmd \"abc\" 'def' ghi");
30        assert "cmd".equals(pc.getCommand());
31  
32        assert pc.getArguments().size()==3;
33  
34        assert "abc".equals(pc.getArguments().get(0).getValue());
35        assert "def".equals(pc.getArguments().get(1).getValue());
36        assert "ghi".equals(pc.getArguments().get(2).getValue());
37     }
38  
39     public void testNoArguments() {
40        ProcessedCommand pc = new ProcessedCommand("cmd ");
41        assert "cmd".equals(pc.getCommand());
42  
43        assert pc.getArguments().size()==0;
44  
45        assert pc.isCommandComplete();
46     }
47  }
