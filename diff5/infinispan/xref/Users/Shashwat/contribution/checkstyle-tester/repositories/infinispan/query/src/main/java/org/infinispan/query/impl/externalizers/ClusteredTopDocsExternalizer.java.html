1   package org.infinispan.query.impl.externalizers;
2   
3   import java.io.IOException;
4   import java.io.ObjectInput;
5   import java.io.ObjectOutput;
6   import java.util.Set;
7   
8   import org.apache.lucene.search.TopDocs;
9   import org.infinispan.commons.io.UnsignedNumeric;
10  import org.infinispan.commons.marshall.AbstractExternalizer;
11  import org.infinispan.commons.util.Util;
12  import org.infinispan.query.clustered.ISPNEagerTopDocs;
13  
14  public class ClusteredTopDocsExternalizer extends AbstractExternalizer<ISPNEagerTopDocs> {
15  
16     @Override
17     public Set<Class<? extends ISPNEagerTopDocs>> getTypeClasses() {
18        return Util.<Class<? extends ISPNEagerTopDocs>>asSet(ISPNEagerTopDocs.class);
19     }
20  
21     @Override
22     public ISPNEagerTopDocs readObject(final ObjectInput input) throws IOException, ClassNotFoundException {
23        final int keysNumber = UnsignedNumeric.readUnsignedInt(input);
24        final Object[] keys = new Object[keysNumber];
25        for (int i=0; i<keysNumber; i++) {
26           keys[i] = input.readObject();
27        }
28        final TopDocs innerTopDocs = LuceneTopDocsExternalizer.readObjectStatic(input);
29        return new ISPNEagerTopDocs(innerTopDocs, keys);
30     }
31  
32     @Override
33     public void writeObject(final ObjectOutput output, final ISPNEagerTopDocs topDocs) throws IOException {
34        final Object[] keys = topDocs.keys;
35        UnsignedNumeric.writeUnsignedInt(output, keys.length);
36        for (Object o : keys) {
37           output.writeObject(o);
38        }
39        LuceneTopDocsExternalizer.writeObjectStatic(output, topDocs);
40     }
41  
42     @Override
43     public Integer getId() {
44        return ExternalizerIds.CLUSTERED_QUERY_TOPDOCS;
45     }
46  }
