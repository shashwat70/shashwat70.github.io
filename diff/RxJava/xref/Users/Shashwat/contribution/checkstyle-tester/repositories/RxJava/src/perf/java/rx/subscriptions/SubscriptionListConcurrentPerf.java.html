17  package rx.subscriptions;
18  
19  import java.util.concurrent.TimeUnit;
20  
21  import org.openjdk.jmh.annotations.Benchmark;
22  import org.openjdk.jmh.annotations.BenchmarkMode;
23  import org.openjdk.jmh.annotations.Group;
24  import org.openjdk.jmh.annotations.GroupThreads;
25  import org.openjdk.jmh.annotations.Mode;
26  import org.openjdk.jmh.annotations.OutputTimeUnit;
27  import org.openjdk.jmh.annotations.Param;
28  import org.openjdk.jmh.annotations.Scope;
29  import org.openjdk.jmh.annotations.Setup;
30  import org.openjdk.jmh.annotations.State;
31  import org.openjdk.jmh.annotations.Threads;
32  
33  import rx.Subscription;
34  import rx.internal.util.SubscriptionList;
35  
36  /**
37   * Benchmark typical subscription list concurrent behavior.
38   * <p>
39   * gradlew benchmarks "-Pjmh=-f 1 -tu s -bm thrpt -wi 5 -i 5 -r 1 .*CompositeSubscriptionConcurrentPerf.*"
40   * <p>
41   * gradlew benchmarks "-Pjmh=-f 1 -tu ns -bm avgt -wi 5 -i 5 -r 1 .*CompositeSubscriptionConcurrentPerf.*"
42   */
43  @BenchmarkMode(Mode.Throughput)
44  @OutputTimeUnit(TimeUnit.SECONDS)
45  @Threads(2)
46  @State(Scope.Group)
47  public class SubscriptionListConcurrentPerf {
48      @Param({ "1", "1000", "100000" })
49      public int loop;
50      
51      public final SubscriptionList csub = new SubscriptionList();
52      @Param({ "1", "5", "10", "20" })
53      public int count;
54      
55      public Subscription[] values;
56      @Setup
57      public void setup() {
58          values = new Subscription[count * 2];
59          for (int i = 0; i < count * 2; i++) {
60              values[i] = new Subscription() {
61                  @Override
62                  public boolean isUnsubscribed() {
63                      return false;
64                  }
65                  @Override
66                  public void unsubscribe() {
67                      
68                  }
69              };
70          }
71      }
72  
73      @Group("g1")
74      @GroupThreads(1)
75      @Benchmark
76      public void addClearT1() {
77          SubscriptionList csub = this.csub;
78          Subscription[] values = this.values;
79          
80          for (int i = loop; i > 0; i--) {
81              for (int j = values.length - 1; j >= 0; j--) {
82                  csub.add(values[j]);
83              }
84              csub.clear();
85          }
86      }
87      @Group("g1")
88      @GroupThreads(1)
89      @Benchmark
90      public void addClearT2() {
91          SubscriptionList csub = this.csub;
92          Subscription[] values = this.values;
93          
94          for (int i = loop; i > 0; i--) {
95              for (int j = values.length - 1; j >= 0; j--) {
96                  csub.add(values[j]);
97              }
98              csub.clear();
99          }
100     }
101     @Group("g2")
102     @GroupThreads(1)
103     @Benchmark
104     public void addClearHalfT1() {
105         SubscriptionList csub = this.csub;
106         Subscription[] values = this.values;
107         int n = values.length;
108         
109         for (int i = loop; i > 0; i--) {
110             for (int j = n / 2 - 1; j >= 0; j--) {
111                 csub.add(values[j]);
112             }
113             csub.clear();
114         }
115     }
116     @Group("g2")
117     @GroupThreads(1)
118     @Benchmark
119     public void addRemoveHalfT2() {
120         SubscriptionList csub = this.csub;
121         Subscription[] values = this.values;
122         int n = values.length;
123         
124         for (int i = loop; i > 0; i--) {
125             for (int j = n - 1; j >= n / 2; j--) {
126                 csub.add(values[j]);
127             }
128             csub.clear();
129         }
130     }
131 }
