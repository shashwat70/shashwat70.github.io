79      BufferedImage def;
80      BufferedImage raw;
81  
82      ImageTypeSpecifier raw_type;
83  
84      public void doTest() throws IOException {
85          /*
86           * This test verifies that png images with color type RGB or RGBA
87           * are decoded as buffered  image of some standard type.
88           *
89           * So we need to be sure that image provided by
90           * user has required color type - RGB or RGBA
91           */
92          if (!checkImageType()) {
93              System.out.println("Test IGNORED!");
94              return;
95          }
96  
97          def = reader.read(0);
98          System.out.println("Default image type: " + def.getType());
99          if (def == null || def.getType() == BufferedImage.TYPE_CUSTOM) {
100             throw new RuntimeException("Test FAILED!");
101         }
102 
103         raw_type = reader.getRawImageType(0);
104         ImageReadParam param = reader.getDefaultReadParam();
105         param.setDestinationType(raw_type);
106 
107         System.out.println("Reading with raw image type...");
108         raw = reader.read(0, param);
109         System.out.println("Type of raw image is " + raw.getType());
110 
111         compare(def, raw);
112 
113         Iterator<ImageTypeSpecifier> types = reader.getImageTypes(0);
114         while (types.hasNext()) {
115             ImageTypeSpecifier t = types.next();
116             System.out.println("Test type: " + t);
117             param.setDestinationType(t);
118 
119             BufferedImage img = reader.read(0, param);
120             System.out.println("Result type: " + img.getType());
121 
122             compare(def, img);
123             System.out.println("Done.\n");
124         }
125 
126         System.out.println("Test PASSED.");
127     }

129     private boolean checkImageType() throws IOException {
130         IIOMetadata md  = null;
131         try {
132             md = reader.getImageMetadata(0);
133         } catch (IOException e) {
134             return false;
135         }
136 
137         String format = md.getNativeMetadataFormatName();
138 
139         Node root = md.getAsTree(format);
140 
141         Node ihdr = getNode(root, "IHDR");
142         if (ihdr == null) {
143             throw new RuntimeException("No ihdr node: invalid png image!");
144         }
145 
146         String colorType = getAttributeValue(ihdr, "colorType");
147         System.out.println("ColorType: " + colorType);
148         if ("RGB".equals(colorType) || "RGBAlpha".equals(colorType)) {
149             // we shuld chek bitDepth
150             System.out.println("Good color type!");
151             String bitDepthStr = getAttributeValue(ihdr, "bitDepth");
152             System.out.println("bitDepth: " + bitDepthStr);
153             int bitDepth = -1;
154             try {
155                 bitDepth = Integer.parseInt(bitDepthStr);
156             } catch (NumberFormatException e) {
157                 throw new RuntimeException("Invalid bitDepth!");
158             }
159             if (bitDepth == 8) {
160                 /*
161                  * This image is RGB or RGBA color type and
162                  * 8 bit tepth. so it can be used for test
163                  */
164                 return true;
165             }
166         }
167 
168         return false;
169     }

171     private String getAttributeValue(Node n, String attrname) {
172         NamedNodeMap attrs = n.getAttributes();
173         if (attrs == null) {
174             return null;
175         } else {
176             Node a = attrs.getNamedItem(attrname);
177             if (a == null) {
178                 return null;
179             } else {
180                 return a.getNodeValue();
181             }
182         }
183     }
184 
185     private Node getNode(Node root, String name) {
186         Node n = root;
187         return lookupNode(n, name);
188     }
189 
190     private Node lookupNode(Node n, String name) {
191         if (n == null) {
192             return null;
193         }
194         if (name.equals(n.getNodeName())) {
195             return n;
196         } else {
197             // may be next on this level?
198             Node res = lookupNode(n.getNextSibling(), name);
199 
200             if (res != null) {
201                 return res;
202             } else {
203                 /// try children then
204                 return lookupNode(n.getFirstChild(), name);
205             }
206         }
207     }

209     private static void compare(BufferedImage a, BufferedImage b) {
210         int w = a.getWidth();
211         int h = a.getHeight();
212 
213         if (w != b.getWidth() || h != b.getHeight()) {
214             throw new RuntimeException("Test FAILED!");
215         }
216 
217         for (int y = 0; y < h; y++) {
218             for (int x = 0; x < w; x++) {
219                 if (a.getRGB(x, y) != b.getRGB(x, y)) {
220                     throw new RuntimeException("Test FAILED!");
221                 }
222             }
223         }
224     }

226     static Color[] colors = new Color[] { Color.red, Color.green, Color.blue };
227 
228     private static ImageInputStream createTestImage(int type) throws IOException  {
229         int w = 100;
230         int h = 100;
231 
232         BufferedImage img = new BufferedImage(w, h, type);
233 
234         int dx = w / colors.length;
235 
236         for (int i = 0; i < colors.length; i++) {
237             for (int x = i *dx; (x < (i + 1) * dx) && (x < w) ; x++) {
238                 for (int y = 0; y < h; y++) {
239                     img.setRGB(x, y, colors[i].getRGB());
240                 }
241             }
242         }
243 
244         File pwd = new File(".");
245         File out = File.createTempFile("rgba_", ".png", pwd);
246         System.out.println("Create file: " + out.getAbsolutePath());
247         ImageIO.write(img, "PNG", out);
248         return ImageIO.createImageInputStream(out);
249     }
