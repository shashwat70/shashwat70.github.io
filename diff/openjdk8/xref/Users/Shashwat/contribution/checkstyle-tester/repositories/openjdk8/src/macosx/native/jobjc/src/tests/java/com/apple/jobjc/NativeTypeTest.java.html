25  package com.apple.jobjc;
26  
27  import com.apple.internal.jobjc.generator.model.types.NType;
28  import com.apple.internal.jobjc.generator.utils.NTypeMerger;
29  import com.apple.internal.jobjc.generator.utils.NTypeParser;
30  
31  public class NativeTypeTest extends PooledTestCase{
32  
33      private NType doParse(String type){
34          NType nt = NTypeParser.parseFrom(type);
35          String printed = nt.toString();
36          System.out.println("Original: " + type);
37          System.out.println("Printed.: " + printed);
38          assertEquals(type, printed);
39          return nt;
40      }
41  
42      // {_NSRect=
43      //   "origin"{_NSPoint="x"f"y"f}
44      //   "size"{_NSSize="width"f"height"f}}
45      public void testStruct(){
46          doParse("{_NSRect=\"origin\"{_NSPoint=\"x\"f\"y\"f}\"size\"{_NSSize=\"width\"f\"height\"f}}");
47      }
48  
49      // {IOBluetoothL2CAPChannelEvent=
50      //   "eventType"i
51      //   "u"(?=
52      //     "data"{IOBluetoothL2CAPChannelDataBlock=
53      //       "dataPtr"^v
54      //       "dataSize"I}
55      //     "writeRefCon"^v
56      //     "padding"[32C])
57      //   "status"i}
58      public void testUnion(){
59          doParse("{IOBluetoothL2CAPChannelEvent=\"eventType\"i\"u\"(?=\"data\"{IOBluetoothL2CAPChannelDataBlock=\"dataPtr\"^v\"dataSize\"I}\"writeRefCon\"^v\"padding\"[32C])\"status\"i}");
60      }
61  
62      public void testUnknown(){
63          doParse("{_CFSocketContext=\"version\"i\"info\"^v\"retain\"^?\"release\"^?\"copyDescription\"^?}");
64      }
65  
66      public void testEmptyStruct(){
67          doParse("{_CFSocketSignature=\"protocolFamily\"i\"socketType\"i\"protocol\"i\"address\"^{__CFData}}");
68      }
69  
70      public void testCharPtr(){
71          doParse("^*");
72      }
73  
74      public void doEquals(final String s){
75          assertEquals(doParse(s), doParse(s));
76      }
77  
78      public void testEquals(){
79          doEquals("{_NSRect=\"origin\"{_NSPoint=\"x\"f\"y\"f}\"size\"{_NSSize=\"width\"f\"height\"f}}");
80          doEquals("{IOBluetoothL2CAPChannelEvent=\"eventType\"i\"u\"(?=\"data\"{IOBluetoothL2CAPChannelDataBlock=\"dataPtr\"^v\"dataSize\"I}\"writeRefCon\"^v\"padding\"[32C])\"status\"i}");
81          doEquals("{_CFSocketContext=\"version\"i\"info\"^v\"retain\"^?\"release\"^?\"copyDescription\"^?}");
82          doEquals("{_CFSocketSignature=\"protocolFamily\"i\"socketType\"i\"protocol\"i\"address\"^{__CFData}}");
83      }
84  
85      public void testMerge(){
86          NType a = doParse("{_NSRect={_NSPoint=\"x\"f\"y\"f}\"size\"{_NSSize=ff}}");
87          NType b = doParse("{_NSRect=\"origin\"{_NSPoint=ff}{_NSSize=\"width\"f\"height\"f}}");
88          NType c = NTypeMerger.inst().merge(a, b);
89          NType expected = doParse("{_NSRect=\"origin\"{_NSPoint=\"x\"f\"y\"f}\"size\"{_NSSize=\"width\"f\"height\"f}}");
90          System.out.println("Merge results:");
91          System.out.println("\ta: " + a.toString());
92          System.out.println("\tb: " + b.toString());
93          System.out.println("\tc: " + c.toString());
94          System.out.println("\tx: " + expected.toString());
95          assertEquals(expected, c);
96      }
97  }
