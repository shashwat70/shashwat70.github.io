105     @Test
106     public void test_parseDefaulting_entireDate() {
107         DateTimeFormatter f = builder
108             .parseDefaulting(YEAR, 2012).parseDefaulting(MONTH_OF_YEAR, 6)
109             .parseDefaulting(DAY_OF_MONTH, 30).toFormatter();
110         LocalDate parsed = f.parse("", LocalDate::from);  // blank string can be parsed
111         assertEquals(parsed, LocalDate.of(2012, 6, 30));
112     }

114     @Test
115     public void test_parseDefaulting_yearOptionalMonthOptionalDay() {
116         DateTimeFormatter f = builder
117                 .appendValue(YEAR)
118                 .optionalStart().appendLiteral('-').appendValue(MONTH_OF_YEAR)
119                 .optionalStart().appendLiteral('-').appendValue(DAY_OF_MONTH)
120                 .optionalEnd().optionalEnd()
121                 .parseDefaulting(MONTH_OF_YEAR, 1)
122                 .parseDefaulting(DAY_OF_MONTH, 1).toFormatter();
123         assertEquals(f.parse("2012", LocalDate::from), LocalDate.of(2012, 1, 1));
124         assertEquals(f.parse("2012-6", LocalDate::from), LocalDate.of(2012, 6, 1));
125         assertEquals(f.parse("2012-6-30", LocalDate::from), LocalDate.of(2012, 6, 30));
126     }
