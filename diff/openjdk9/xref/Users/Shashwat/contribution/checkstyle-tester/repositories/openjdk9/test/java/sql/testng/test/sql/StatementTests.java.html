23  package test.sql;
24  
25  import java.sql.SQLException;
26  import static org.testng.Assert.assertEquals;
27  import org.testng.annotations.BeforeClass;
28  import org.testng.annotations.BeforeMethod;
29  import org.testng.annotations.DataProvider;
30  import org.testng.annotations.Test;
31  import util.BaseTest;
32  import util.StubStatement;
33  
34  public class StatementTests extends BaseTest {
35  
36      protected StubStatement stmt;
37      protected static String maxIdentifier;
38  
39      @BeforeMethod
40      public void setUpMethod() throws Exception {
41          stmt = new StubStatement();
42      }
43  
44      @BeforeClass
45      public static void setUpClass() throws Exception {
46          int maxLen = 128;
47          StringBuilder s = new StringBuilder(maxLen);
48          for (int i = 0; i < maxLen; i++) {
49              s.append('a');
50          }
51          maxIdentifier = s.toString();
52      }
53      /*
54       * Verify that enquoteLiteral creates a  valid literal and converts every
55       * single quote to two single quotes
56       */
57  
58      @Test(dataProvider = "validEnquotedLiteralValues")
59      public void test00(String s, String expected) throws SQLException {
60          assertEquals(stmt.enquoteLiteral(s), expected);
61      }
62  
63      /*
64       * Validate a NullPointerException is thrown if the string passed to
65       * enquoteLiteral is null
66       */
67      @Test(expectedExceptions = NullPointerException.class)
68      public void test01() throws SQLException {
69          stmt.enquoteLiteral(null);
70  
71      }
72  
73      /*
74       * Validate that enquoteIdentifier returns the expected value
75       */
76      @Test(dataProvider = "validIdentifierValues")
77      public void test02(String s, boolean alwaysQuote, String expected) throws SQLException {
78          assertEquals(stmt.enquoteIdentifier(s, alwaysQuote), expected);
79  
80      }
81  
82      /*
83       * Validate that a SQLException is thrown for values that are not valid
84       * for a SQL identifier
85       */
86      @Test(dataProvider = "invalidIdentifierValues",
87              expectedExceptions = SQLException.class)
88      public void test03(String s, boolean alwaysQuote) throws SQLException {
89          stmt.enquoteIdentifier(s, alwaysQuote);
90  
91      }
92  
93      /*
94       * Validate a NullPointerException is thrown is the string passed to
95       * enquoteIdentiifer is null
96       */
97      @Test(dataProvider = "trueFalse",
98              expectedExceptions = NullPointerException.class)
99      public void test04(boolean alwaysQuote) throws SQLException {
100         stmt.enquoteIdentifier(null, alwaysQuote);
101 
102     }
103 
104     /*
105      * Validate that isSimpleIdentifier returns the expected value
106      */
107     @Test(dataProvider = "simpleIdentifierValues")
108     public void test05(String s, boolean expected) throws SQLException {
109         assertEquals(stmt.isSimpleIdentifier(s), expected);
110     }
111 
112     /*
113      * Validate a NullPointerException is thrown if the string passed to
114      * isSimpleIdentifier is null
115      */
116     @Test(expectedExceptions = NullPointerException.class)
117     public void test06() throws SQLException {
118         stmt.isSimpleIdentifier(null);
119 
120     }
121 
122     /*
123      * Verify that enquoteLiteral creates a  valid literal and converts every
124      * single quote to two single quotes
125      */
126     @Test(dataProvider = "validEnquotedNCharLiteralValues")
127     public void test07(String s, String expected) throws SQLException {
128         assertEquals(stmt.enquoteNCharLiteral(s), expected);
129     }
130 
131     /*
132      * Validate a NullPointerException is thrown if the string passed to
133      * enquoteNCharLiteral is null
134      */
135     @Test(expectedExceptions = NullPointerException.class)
136     public void test08() throws SQLException {
137         stmt.enquoteNCharLiteral(null);
138     }
139 
140     /*
141      * DataProvider used to provide strings that will be used to validate
142      * that enquoteLiteral converts a string to a literal and every instance of
143      * a single quote will be converted into two single quotes in the literal.
144      */
145     @DataProvider(name = "validEnquotedLiteralValues")
146     protected Object[][] validEnquotedLiteralValues() {
147         return new Object[][]{
148             {"Hello", "'Hello'"},
149             {"G'Day", "'G''Day'"},
150             {"'G''Day'", "'''G''''Day'''"},
151             {"I'''M", "'I''''''M'"},
152             {"The Dark Knight", "'The Dark Knight'"}
153 
154         };
155     }
156 
157     /*
158      * DataProvider used to provide strings that will be used to validate
159      * that enqouteIdentifier returns a simple SQL Identifier or a double
160      * quoted identifier
161      */
162     @DataProvider(name = "validIdentifierValues")
163     protected Object[][] validEnquotedIdentifierValues() {
164         return new Object[][]{
165             {"b", false, "b"},
166             {"b", true, "\"b\""},
167             {maxIdentifier, false, maxIdentifier},
168             {maxIdentifier, true, "\"" + maxIdentifier + "\""},
169             {"Hello", false, "Hello"},
170             {"Hello", true, "\"Hello\""},
171             {"G'Day", false, "\"G'Day\""},
172             {"G'Day", true, "\"G'Day\""},
173             {"Bruce Wayne", false, "\"Bruce Wayne\""},
174             {"Bruce Wayne", true, "\"Bruce Wayne\""},
175             {"GoodDay$", false, "\"GoodDay$\""},
176             {"GoodDay$", true, "\"GoodDay$\""},};
177     }
178 
179     /*
180      * DataProvider used to provide strings are invalid for enquoteIdentifier
181      * resulting in a SQLException being thrown
182      */
183     @DataProvider(name = "invalidIdentifierValues")
184     protected Object[][] invalidEnquotedIdentifierValues() {
185         return new Object[][]{
186             {"Hel\"lo", false},
187             {"\"Hel\"lo\"", true},
188             {"Hello" + '\0', false},
189             {"", false},
190             {maxIdentifier + 'a', false},};
191     }
192 
193     /*
194      * DataProvider used to provide strings that will be used to validate
195      * that isSimpleIdentifier returns the correct value based on the
196      * identifier specified.
197      */
198     @DataProvider(name = "simpleIdentifierValues")
199     protected Object[][] simpleIdentifierValues() {
200         return new Object[][]{
201             {"b", true},
202             {"Hello", true},
203             {"\"Gotham\"", false},
204             {"G'Day", false},
205             {"Bruce Wayne", false},
206             {"GoodDay$", false},
207             {"Dick_Grayson", true},
208             {"Batmobile1966", true},
209             {maxIdentifier, true},
210             {maxIdentifier + 'a', false},
211             {"", false},};
212     }
213 
214     /*
215      * DataProvider used to provide strings that will be used to validate
216      * that enquoteNCharLiteral converts a string to a National Character
217      * literal and every instance of
218      * a single quote will be converted into two single quotes in the literal.
219      */
220     @DataProvider(name = "validEnquotedNCharLiteralValues")
221     protected Object[][] validEnquotedNCharLiteralValues() {
222         return new Object[][]{
223             {"Hello", "N'Hello'"},
224             {"G'Day", "N'G''Day'"},
225             {"'G''Day'", "N'''G''''Day'''"},
226             {"I'''M", "N'I''''''M'"},
227             {"N'Hello'", "N'N''Hello'''"},
228             {"The Dark Knight", "N'The Dark Knight'"}
229 
230         };
231     }
