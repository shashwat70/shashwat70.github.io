24  /* @test
25   * @bug 6253145
26   * @key intermittent
27   * @summary Test FileChannel.transferTo with file positions up to 8GB
28   * @run testng/timeout=300 TransferTo6GBFile
29   */
30  
31  import java.io.File;
32  import java.io.IOException;
33  import java.io.PrintStream;
34  import java.io.RandomAccessFile;
35  import java.net.InetAddress;
36  import java.net.InetSocketAddress;
37  import java.nio.ByteBuffer;
38  import java.nio.channels.FileChannel;
39  import java.nio.channels.ServerSocketChannel;
40  import java.nio.channels.SocketChannel;
41  import java.util.concurrent.TimeUnit;
42  
43  import org.testng.annotations.Test;
44  
45  public class TransferTo6GBFile {
46  
47      private static PrintStream err = System.err;
48      private static PrintStream out = System.out;
49  
50      // Test transferTo with file positions larger than 2 and 4GB
51      @Test
52      public void xferTest08() throws Exception { // for bug 6253145
53          final long G = 1024L * 1024L * 1024L;
54  
55          // Create 6GB file
56  
57          File file = File.createTempFile("source", null);
58          file.deleteOnExit();
59  
60          RandomAccessFile raf = new RandomAccessFile(file, "rw");
61          FileChannel fc = raf.getChannel();
62  
63          out.println("  Writing large file...");
64          long t0 = System.nanoTime();
65          try {
66              fc.write(ByteBuffer.wrap("0123456789012345".getBytes("UTF-8")), 6*G);
67              long t1 = System.nanoTime();
68              out.printf("  Wrote large file in %d ns (%d ms) %n",
69              t1 - t0, TimeUnit.NANOSECONDS.toMillis(t1 - t0));
70          } catch (IOException x) {
71              err.println("  Unable to create test file:" + x);
72              fc.close();
73              return;
74          }
75  
76          // Setup looback connection and echo server
77  
78          ServerSocketChannel ssc = ServerSocketChannel.open();
79          ssc.socket().bind(new InetSocketAddress(0));
80  
81          InetAddress lh = InetAddress.getLocalHost();
82          InetSocketAddress isa = new InetSocketAddress(lh, ssc.socket().getLocalPort());
83          SocketChannel source = SocketChannel.open(isa);
84          SocketChannel sink = ssc.accept();
85  
86          Thread thr = new Thread(new EchoServer(sink));
87          thr.start();
88  
89          // Test data is array of positions and counts
90  
91          long testdata[][] = {
92              { 2*G-1,    1 },
93              { 2*G-1,    10 },       // across 2GB boundary
94              { 2*G,      1 },
95              { 2*G,      10 },
96              { 2*G+1,    1 },
97              { 4*G-1,    1 },
98              { 4*G-1,    10 },       // across 4GB boundary
99              { 4*G,      1 },
100             { 4*G,      10 },
101             { 4*G+1,    1 },
102             { 5*G-1,    1 },
103             { 5*G-1,    10 },
104             { 5*G,      1 },
105             { 5*G,      10 },
106             { 5*G+1,    1 },
107             { 6*G,      1 },
108         };
109 
110         ByteBuffer sendbuf = ByteBuffer.allocateDirect(100);
111         ByteBuffer readbuf = ByteBuffer.allocateDirect(100);
112 
113         try {
114             byte value = 0;
115             for (int i=0; i<testdata.length; i++) {
116                 long position = testdata[(int)i][0];
117                 long count = testdata[(int)i][1];
118 
119                 // generate bytes
120                 for (long j=0; j<count; j++) {
121                     sendbuf.put(++value);
122                 }
123                 sendbuf.flip();
124 
125                 // write to file and transfer to echo server
126                 fc.write(sendbuf, position);
127                 t0 = System.nanoTime();
128                 fc.transferTo(position, count, source);
129                 out.printf("  transferTo(%d, %2d, source): %d ns%n",
130                     position, count, System.nanoTime() - t0);
131 
132                 // read from echo server
133                 long nread = 0;
134                 while (nread < count) {
135                     int n = source.read(readbuf);
136                     if (n < 0)
137                         throw new RuntimeException("Premature EOF!");
138                     nread += n;
139                 }
140 
141                 // check reply from echo server
142                 readbuf.flip();
143                 sendbuf.flip();
144                 if (!readbuf.equals(sendbuf))
145                     throw new RuntimeException("Echoed bytes do not match!");
146                 readbuf.clear();
147                 sendbuf.clear();
148             }
149         } finally {
150             source.close();
151             ssc.close();
152             fc.close();
153             file.delete();
154         }
155     }
156 
157     /**
158      * Simple in-process server to echo bytes read by a given socket channel
159      */
160     static class EchoServer implements Runnable {
161         private SocketChannel sc;
162 
163         public EchoServer(SocketChannel sc) {
164             this.sc = sc;
165         }
166 
167         public void run() {
168             ByteBuffer bb = ByteBuffer.allocateDirect(1024);
169             try {
170                 for (;;) {
171                     int n = sc.read(bb);
172                     if (n < 0)
173                         break;
174 
175                     bb.flip();
176                     while (bb.remaining() > 0) {
177                         sc.write(bb);
178                     }
179                     bb.clear();
180                 }
181             } catch (IOException x) {
182                 x.printStackTrace();
183             } finally {
184                 try {
185                     sc.close();
186                 } catch (IOException ignore) { }
187             }
188         }
189     }
