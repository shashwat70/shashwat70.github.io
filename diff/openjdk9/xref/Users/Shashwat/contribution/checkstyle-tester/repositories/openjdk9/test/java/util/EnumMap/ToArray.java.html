31  import java.util.*;
32  
33  public class ToArray {
34      enum Country { FRENCH, POLISH }
35      public static void main(String[] args) throws Throwable {
36          Map<Country, String> m = new EnumMap<Country, String>(Country.class);
37          m.put(Country.FRENCH, "connection");
38          m.put(Country.POLISH, "sausage");
39  
40          Object[] z = m.entrySet().toArray();
41          System.out.println(Arrays.toString(z));
42          if (! (z.getClass() == Object[].class &&
43                 z.length == 2 &&
44                 ((Map.Entry)z[0]).getKey() == Country.FRENCH &&
45                 ((Map.Entry)z[1]).getKey() == Country.POLISH))
46              throw new AssertionError();
47  
48          Map.Entry[] x1 = new Map.Entry[3];
49          x1[2] = m.entrySet().iterator().next();
50          Map.Entry[] x2 = m.entrySet().toArray(x1);
51          System.out.println(Arrays.toString(x2));
52          if (! (x1 == x2 &&
53                 x2[0].getKey() == Country.FRENCH &&
54                 x2[1].getKey() == Country.POLISH &&
55                 x2[2] == null))
56              throw new AssertionError();
57  
58          Map.Entry[] y1 = new Map.Entry[1];
59          Map.Entry[] y2 = m.entrySet().toArray(y1);
60          System.out.println(Arrays.toString(y2));
61          if (! (y1 != y2 &&
62                 y2.length == 2 &&
63                 y2[0].getKey() == Country.FRENCH &&
64                 y2[1].getKey() == Country.POLISH))
65              throw new AssertionError();
66      }
67  }
