24  package org.hibernate.metamodel.binding;
25  
26  import org.hibernate.mapping.PropertyGeneration;
27  import org.hibernate.metamodel.relational.Value;
28  
29  /**
30   * Specialized binding contract for singular (non-collection) attributes
31   *
32   * @author Steve Ebersole
33   */
34  public interface SingularAttributeBinding extends AttributeBinding {
35  	/**
36  	 * Obtain the value bound here.  This could potentially be a {@link org.hibernate.metamodel.relational.Tuple}
37  	 * indicating multiple database values are bound, in which case access to the individual values can be achieved by
38  	 * either casting this return to {@link org.hibernate.metamodel.relational.Tuple} and using its
39  	 * {@link org.hibernate.metamodel.relational.Tuple#values()} method or using the {@link #getSimpleValueBindings()}
40  	 * method here and accessing each bindings {@link SimpleValueBinding#getSimpleValue simple value}
41  	 *
42  	 * @return The bound value
43  	 */
44  	public Value getValue();
45  
46  	/**
47  	 * Returns the number of {@link SimpleValueBinding} objects that will be returned by
48  	 * {@link #getSimpleValueBindings()}
49  	 *
50  	 * @return the number of {@link SimpleValueBinding simple value bindings}
51  	 *
52  	 * @see #getSimpleValueBindings()
53  	 */
54  	public int getSimpleValueSpan();
55  
56  	public Iterable<SimpleValueBinding> getSimpleValueBindings();
57  
58  	public void setSimpleValueBindings(Iterable<SimpleValueBinding> simpleValueBindings);
59  
60  	/**
61  	 * Convenience method to determine if any {@link SimpleValueBinding simple value bindings} are derived values
62  	 * (formula mappings).
63  	 *
64  	 * @return {@code true} indicates that the binding contains a derived value; {@code false} indicates it does not.
65  	 */
66  	public boolean hasDerivedValue();
67  
68  	/**
69  	 * Convenience method to determine if all {@link SimpleValueBinding simple value bindings} allow nulls.
70  	 *
71  	 * @return {@code true} indicates that all values allow {@code null}; {@code false} indicates one or more do not
72  	 */
73  	public boolean isNullable();
74  
75  	/**
76  	 * Obtain the generation strategy for this attribute/value.
77  	 *
78  	 * @return The generation strategy
79  	 */
80  	public PropertyGeneration getGeneration();
81  }
