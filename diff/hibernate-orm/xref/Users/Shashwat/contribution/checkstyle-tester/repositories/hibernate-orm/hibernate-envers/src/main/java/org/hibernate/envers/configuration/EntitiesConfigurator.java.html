57  public class EntitiesConfigurator {
58      public EntitiesConfigurations configure(Configuration cfg, ReflectionManager reflectionManager,
59                                              GlobalConfiguration globalCfg, AuditEntitiesConfiguration verEntCfg,
60                                              AuditStrategy auditStrategy,
61                                              Document revisionInfoXmlMapping, Element revisionInfoRelationMapping) {
62          // Creating a name register to capture all audit entity names created.
63          AuditEntityNameRegister auditEntityNameRegister = new AuditEntityNameRegister();
64          DOMWriter writer = new DOMWriter();
65  
66          // Sorting the persistent class topologically - superclass always before subclass
67          Iterator<PersistentClass> classes = GraphTopologicalSort.sort(new PersistentClassGraphDefiner(cfg)).iterator();
68  
69          ClassesAuditingData classesAuditingData = new ClassesAuditingData();
70          Map<PersistentClass, EntityXmlMappingData> xmlMappings = new HashMap<PersistentClass, EntityXmlMappingData>();
71  
72          // Reading metadata from annotations
73          while (classes.hasNext()) {
74              PersistentClass pc = classes.next();
75  
76              // Collecting information from annotations on the persistent class pc
77              AnnotationsMetadataReader annotationsMetadataReader =
78                      new AnnotationsMetadataReader(globalCfg, reflectionManager, pc);
79              ClassAuditingData auditData = annotationsMetadataReader.getAuditData();
80  
81              classesAuditingData.addClassAuditingData(pc, auditData);
82          }
83  
84          // Now that all information is read we can update the calculated fields.
85          classesAuditingData.updateCalculatedFields();
86  
87          AuditMetadataGenerator auditMetaGen = new AuditMetadataGenerator(cfg, globalCfg, verEntCfg, auditStrategy,
88                  revisionInfoRelationMapping, auditEntityNameRegister);
89  
90          // First pass
91          for (Map.Entry<PersistentClass, ClassAuditingData> pcDatasEntry : classesAuditingData.getAllClassAuditedData()) {
92              PersistentClass pc = pcDatasEntry.getKey();
93              ClassAuditingData auditData = pcDatasEntry.getValue();
94  
95              EntityXmlMappingData xmlMappingData = new EntityXmlMappingData();
96              if (auditData.isAudited()) {
97                  if (!StringTools.isEmpty(auditData.getAuditTable().value())) {
98                      verEntCfg.addCustomAuditTableName(pc.getEntityName(), auditData.getAuditTable().value());
99                  }
100 
101                 auditMetaGen.generateFirstPass(pc, auditData, xmlMappingData, true);
102 			} else {
103 				auditMetaGen.generateFirstPass(pc, auditData, xmlMappingData, false);
104 			}
105 
106             xmlMappings.put(pc, xmlMappingData);
107         }
108 
109         // Second pass
110         for (Map.Entry<PersistentClass, ClassAuditingData> pcDatasEntry : classesAuditingData.getAllClassAuditedData()) {
111             EntityXmlMappingData xmlMappingData = xmlMappings.get(pcDatasEntry.getKey());
112 
113             if (pcDatasEntry.getValue().isAudited()) {
114                 auditMetaGen.generateSecondPass(pcDatasEntry.getKey(), pcDatasEntry.getValue(), xmlMappingData);
115                 try {
116                     cfg.addDocument(writer.write(xmlMappingData.getMainXmlMapping()));
117                     //writeDocument(xmlMappingData.getMainXmlMapping());
118 
119                     for (Document additionalMapping : xmlMappingData.getAdditionalXmlMappings()) {
120                         cfg.addDocument(writer.write(additionalMapping));
121                         //writeDocument(additionalMapping);
122                     }
123                 } catch (DocumentException e) {
124                     throw new MappingException(e);
125                 }
126             }
127         }
128 
129         // Only if there are any versioned classes
130         if (auditMetaGen.getEntitiesConfigurations().size() > 0) {
131             try {
132                 if (revisionInfoXmlMapping !=  null) {
133                     //writeDocument(revisionInfoXmlMapping);
134                     cfg.addDocument(writer.write(revisionInfoXmlMapping));
135                 }
136             } catch (DocumentException e) {
137                 throw new MappingException(e);
138             }
139         }
140 
141 		return new EntitiesConfigurations(auditMetaGen.getEntitiesConfigurations(),
142 				auditMetaGen.getNotAuditedEntitiesConfigurations());
143     }
144 
145     @SuppressWarnings({"UnusedDeclaration"})
146     private void writeDocument(Document e) {
147         ByteArrayOutputStream baos = new ByteArrayOutputStream();
148         Writer w = new PrintWriter(baos);
149 
150         try {
151             XMLWriter xw = new XMLWriter(w, new OutputFormat(" ", true));
152             xw.write(e);
153             w.flush();
154         } catch (IOException e1) {
155             e1.printStackTrace();
156         }
157 
158         System.out.println("-----------");
159         System.out.println(baos.toString());
160         System.out.println("-----------");
161     }
