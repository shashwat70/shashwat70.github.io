1   package org.hibernate.envers.revisioninfo;
2   
3   import java.io.Serializable;
4   import java.util.HashSet;
5   import java.util.Set;
6   
7   import org.hibernate.envers.DefaultTrackingModifiedEntitiesRevisionEntity;
8   import org.hibernate.envers.ModifiedEntityNames;
9   import org.hibernate.envers.RevisionListener;
10  import org.hibernate.envers.RevisionType;
11  import org.hibernate.envers.entities.PropertyData;
12  import org.hibernate.envers.tools.reflection.ReflectionTools;
13  import org.hibernate.property.Getter;
14  import org.hibernate.property.Setter;
15  
16  /**
17   * Automatically adds entity names, that have been changed during current revision, to revision entity.
18   * @see ModifiedEntityNames
19   * @see DefaultTrackingModifiedEntitiesRevisionEntity
20   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
21   */
22  public class DefaultTrackingModifiedEntitiesRevisionInfoGenerator extends DefaultRevisionInfoGenerator {
23      private final Setter modifiedEntityNamesSetter;
24      private final Getter modifiedEntityNamesGetter;
25  
26      public DefaultTrackingModifiedEntitiesRevisionInfoGenerator(String revisionInfoEntityName, Class<?> revisionInfoClass,
27                                                                  Class<? extends RevisionListener> listenerClass,
28                                                                  PropertyData revisionInfoTimestampData, boolean timestampAsDate,
29                                                                  PropertyData modifiedEntityNamesData) {
30          super(revisionInfoEntityName, revisionInfoClass, listenerClass, revisionInfoTimestampData, timestampAsDate);
31          modifiedEntityNamesSetter = ReflectionTools.getSetter(revisionInfoClass, modifiedEntityNamesData);
32          modifiedEntityNamesGetter = ReflectionTools.getGetter(revisionInfoClass, modifiedEntityNamesData);
33      }
34  
35      @Override
36      @SuppressWarnings({"unchecked"})
37      public void entityChanged(Class entityClass, String entityName, Serializable entityId, RevisionType revisionType,
38                                Object revisionEntity) {
39          super.entityChanged(entityClass, entityName, entityId, revisionType, revisionEntity);
40          Set<String> modifiedEntityNames = (Set<String>) modifiedEntityNamesGetter.get(revisionEntity);
41          if (modifiedEntityNames == null) {
42              modifiedEntityNames = new HashSet<String>();
43              modifiedEntityNamesSetter.set(revisionEntity, modifiedEntityNames, null);
44          }
45          modifiedEntityNames.add(entityName);
46      }
47  }
