24  package org.hibernate.cfg.naming;
25  
26  /**
27   * A
28   * @author Gail Badner
29   */
30  public interface NamingStrategyDelegate {
31  
32  	/**
33  	 * Determine the name of a entity's primary table when a name is not explicitly configured.
34  	 *
35  	 * @param entityName The fully qualified entity name
36  	 * @param jpaEntityName The entity name provided by the {@link javax.persistence.Entity}
37  	 *                      {@code name} attribute; or, if not mapped in this way, then the
38  	 *                      unqualified entity name.
39  	 *
40  	 * @return The implicit table name.
41  	 */
42  	public String determineImplicitPrimaryTableName(String entityName, String jpaEntityName);
43  
44  	/**
45  	 * Determine the name of a property's column when a name is not explicitly configured.
46  	 *
47  	 * @param propertyPath The property path (not qualified by the entity name)
48  	 * @return The implicit column name.
49  	 */
50  	public String determineImplicitPropertyColumnName(String propertyPath);
51  
52  	/**
53  	 * Determine the name of a collection table for a collection of (non-entity) values
54  	 * when a name is not explicitly configured.
55  	 *
56  	 * @param ownerEntityName The fully qualified entity name for the entity that owns the collection.
57  	 * @param ownerJpaEntityName The entity name provided by the {@link javax.persistence.Entity}
58  	 *                      {@code name} attribute for the entity that owns the collection;
59  	 *                      or, if not mapped in this way, then the unqualified owner entity name.
60  	 * @param ownerEntityTable The owner entity's physical primary table name;
61  	 * @param propertyPath The property path (not qualified by the entity name),
62  	 * @return The implicit table name.
63  	 */
64  	public String determineImplicitElementCollectionTableName(
65  			String ownerEntityName,
66  			String ownerJpaEntityName,
67  			String ownerEntityTable,
68  			String propertyPath);
69  
70  	/**
71  	 * Determine the name of the join column in a collection table for
72  	 * a collection of (non-entity) values when a name is not explicitly configured.
73  	 *
74  	 * @param ownerEntityName The fully qualified name of the entity that owns the collection.
75  	 * @param ownerJpaEntityName The entity name provided by the {@link javax.persistence.Entity}
76  	 *                      {@code name} attribute for the entity that owns the collection;
77  	 *                      or, if not mapped in this way, then the unqualified entity name.
78  	 * @param ownerEntityTable The owner entity's physical primary table name;
79  	 * @param referencedColumnName The physical name of the column that the join column references.
80  	 * @param propertyPath The property path (not qualified by the entity name),
81  	 * @return The implicit column name.
82  	 */
83  	public String determineImplicitElementCollectionJoinColumnName(
84  			String ownerEntityName,
85  			String ownerJpaEntityName,
86  			String ownerEntityTable,
87  			String referencedColumnName,
88  			String propertyPath);
89  
90  	/**
91  	 * Determine the name of the join table for an entity (singular or plural) association when
92  	 * a name is not explicitly configured.
93  	 *
94  	 * @param ownerEntityName The fully qualified name of the entity that owns the association;.
95  	 * @param ownerJpaEntityName The entity name provided by the {@link javax.persistence.Entity}
96  	 *                      {@code name} attribute for the entity that owns the association;
97  	 *                      or, if not mapped in this way, then the unqualified owner entity name.
98  	 * @param ownerEntityTable The owner entity's physical primary table name;
99  	 * @param associatedEntityName The fully qualified name of the associated entity.
100 	 * @param associatedJpaEntityName The entity name provided by the {@link javax.persistence.Entity}
101 	 *                      {@code name} attribute for the associated entity;
102 	 *                      or, if not mapped in this way, then the unqualified associated entity name.
103 	 * @param associatedEntityTable The associated entity's physical primary table name;
104 	 * @param propertyPath The property path (not qualified by the entity name),
105 	 * @return The implicit table name.
106 	 */
107 	public String determineImplicitEntityAssociationJoinTableName(
108 			String ownerEntityName,
109 			String ownerJpaEntityName,
110 			String ownerEntityTable,
111 			String associatedEntityName,
112 			String associatedJpaEntityName,
113 			String associatedEntityTable,
114 			String propertyPath);
115 
116 	/**
117 	 * Determine the name of join column for an entity (singular or plural) association when
118 	 * a name is not explicitly configured.
119 	 *
120 	 * @param propertyEntityName The fully qualified name of the entity that contains the association;
121 	 * @param propertyJpaEntityName The entity name provided by the {@link javax.persistence.Entity}
122 	 *                      {@code name} attribute for the entity that contains the association;
123 	 *                      or, if not mapped in this way, then the unqualified entity name.
124 	 * @param propertyTableName The physical primary table name for the entity that contains the association.
125 	 * @param referencedColumnName  The physical name of the column that the join column references.
126 	 * @param propertyPath The property path (not qualified by the entity name),
127 	 * @return The implicit table name.
128 	 */
129 	public String determineImplicitEntityAssociationJoinColumnName(
130 			String propertyEntityName,
131 			String propertyJpaEntityName,
132 			String propertyTableName,
133 			String referencedColumnName,
134 			String propertyPath);
135 
136 	public String toPhysicalJoinKeyColumnName(String joinedColumn, String joinedTable);
137 
138 	public String determineLogicalColumnName(String columnName, String propertyName);
139 
140 	public String determineLogicalElementCollectionTableName(
141 			String tableName,
142 			String ownerEntityName,
143 			String ownerJpaEntityName,
144 			String ownerEntityTable,
145 			String propertyName);
146 
147 	public String determineLogicalEntityAssociationJoinTableName(
148 			String tableName,
149 			String ownerEntityName,
150 			String ownerJpaEntityName,
151 			String ownerEntityTable,
152 			String associatedEntityName,
153 			String associatedJpaEntityName,
154 			String associatedEntityTable,
155 			String propertyName);
156 
157 	public String determineLogicalCollectionColumnName(String columnName, String propertyName, String referencedColumn);
158 
159 	/**
160 	 * Alter the table name given in the mapping document
161 	 * @param tableName a table name
162 	 * @return a table name
163 	 */
164 	public String toPhysicalTableName(String tableName);
165 
166 	/**
167 	 * Alter the column name given in the mapping document
168 	 * @param columnName a column name
169 	 * @return a column name
170 	 */
171 	public String toPhysicalColumnName(String columnName);
172 
173 }
