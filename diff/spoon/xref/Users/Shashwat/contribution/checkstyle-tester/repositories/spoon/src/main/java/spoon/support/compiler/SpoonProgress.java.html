6   package spoon.support.compiler;
7   
8   /**
9    * is the interface to follow the progress of the creation of the Spoon model.
10   */
11  public interface SpoonProgress {
12  
13  	enum Process {
14  		COMPILE,
15  		COMMENT,
16  		MODEL,
17  		IMPORT,
18  		COMMENT_LINKING,
19  		PROCESS,
20  		PRINT
21  	}
22  
23  	/**
24  	 * is called when a new process is started
25  	 * @param process the started process
26  	 */
27  	void start(Process process);
28  
29  	/**
30  	 * is called when a step in the precess is started
31  	 * @param process the current process
32  	 * @param task the task that has been processed
33  	 * @param taskId the task id
34  	 * @param nbTask the number of task in the process
35  	 */
36  	void step(Process process, String task, int taskId, int nbTask);
37  
38  	/**
39  	 * is called when a step in the precess is started
40  	 * @param process the current process
41  	 * @param task the task that has been processed
42  	 */
43  	void step(Process process, String task);
44  
45  	/**
46  	 * is called when a new process is started
47  	 * @param process the finished process
48  	 */
49  	void end(Process process);
50  
51  }
