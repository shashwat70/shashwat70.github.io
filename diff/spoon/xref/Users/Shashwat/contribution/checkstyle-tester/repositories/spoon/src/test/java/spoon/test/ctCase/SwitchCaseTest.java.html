17  package spoon.test.ctCase;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertSame;
21  import static spoon.testing.utils.ModelUtils.build;
22  
23  import java.util.List;
24  
25  import org.junit.Test;
26  
27  import spoon.reflect.code.CtCase;
28  import spoon.reflect.code.CtStatement;
29  import spoon.reflect.declaration.CtElement;
30  import spoon.reflect.factory.Factory;
31  import spoon.reflect.visitor.Query;
32  import spoon.reflect.visitor.filter.TypeFilter;
33  
34  @SuppressWarnings({"unchecked", "rawtypes"})
35  public class SwitchCaseTest {
36  
37  	@Test
38  	public void insertAfterStatementInSwitchCaseWithoutException() throws Exception {
39  		String packageName = "spoon.test.ctCase.testclasses";
40  		String className = "ClassWithSwitchExample";
41  		Factory factory = factoryFor(packageName, className);
42  		List<CtCase> elements = elementsOfType(CtCase.class, factory);
43  		assertEquals(3, elements.size());
44  		CtCase firstCase = elements.get(0);
45  		List<CtStatement> statements = firstCase.getStatements();
46  		assertEquals(2, statements.size());
47  		CtStatement newStatement = factory.Code().createCodeSnippetStatement("result = 0");
48  		statements.get(0).insertAfter(newStatement);
49  		statements = firstCase.getStatements();
50  		assertEquals(3, statements.size());
51  		assertSame(statements.get(1), newStatement);
52  	}
53  
54  	@Test
55  	public void insertBeforeStatementInSwitchCaseWithoutException() throws Exception {
56  		String packageName = "spoon.test.ctCase.testclasses";
57  		String className = "ClassWithSwitchExample";
58  		Factory factory = factoryFor(packageName, className);
59  		List<CtCase> elements = elementsOfType(CtCase.class, factory);
60  		assertEquals(3, elements.size());
61  		CtCase firstCase = elements.get(0);
62  		List<CtStatement> statements = firstCase.getStatements();
63  		assertEquals(2, statements.size());
64  		CtStatement newStatement = factory.Code().createCodeSnippetStatement("result = 0");
65  		statements.get(0).insertBefore(newStatement);
66  		statements = firstCase.getStatements();
67  		assertEquals(3, statements.size());
68  		assertSame(statements.get(0), newStatement);
69  	}
70  
71  	private <T extends CtElement> List<T> elementsOfType(Class<T> type, Factory factory) {
72  		return Query.getElements(factory, new TypeFilter<>(type));
73  	}
74  
75  	private Factory factoryFor(String packageName, String className) throws Exception {
76  		return build(packageName, className).getFactory();
77  	}
78  }
