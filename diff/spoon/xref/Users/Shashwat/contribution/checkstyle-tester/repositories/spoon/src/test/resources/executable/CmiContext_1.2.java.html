28  package org.objectweb.carol.jndi.spi;
29  
30  import java.io.Serializable;
31  import java.rmi.Remote;
32  
33  import javax.naming.Context;
34  import javax.naming.Name;
35  import javax.naming.NamingException;
36  import javax.naming.Reference;
37  import javax.naming.Referenceable;
38  import javax.rmi.CORBA.PortableRemoteObjectDelegate;
39  
40  import org.objectweb.carol.jndi.wrapping.JNDIResourceWrapper;
41  import org.objectweb.carol.jndi.wrapping.UnicastJNDIReferenceWrapper;
42  import org.objectweb.carol.rmi.exception.NamingExceptionHelper;
43  import org.objectweb.carol.util.configuration.ConfigurationRepository;
44  
45  /**
46   * @author Florent Benoit
47   * @author Benoit Pelletier
48   */
49  public class CmiContext extends AbsContext implements Context {
50  
51      /**
52       * Constructs an CMI Wrapper context
53       * @param cmiContext the inital CMI context
54       */
55      public CmiContext(Context cmiContext) {
56          super(cmiContext);
57      }
58  
59  
60      /**
61       * If this object is a reference wrapper return the reference If this object
62       * is a resource wrapper return the resource
63       * @param o the object to resolve
64       * @param name name of the object to unwrap
65       * @return the unwrapped object
66       * @throws NamingException if the object cannot be unwraped
67       */
68      protected Object unwrapObject(Object o, Name name) throws NamingException {
69          return super.defaultUnwrapObject(o, name);
70      }
71  
72      /**
73       * Wrap an Object : If the object is a reference wrap it into a Reference
74       * Wrapper Object here the good way is to contact the carol configuration to
75       * get the portable remote object
76       * @param o the object to encode
77       * @param name of the object
78       * @param replace if the object need to be replaced
79       * @return a <code>Remote JNDIRemoteReference Object</code> if o is a
80       *         resource o if else
81       * @throws NamingException if object cannot be wrapped
82       */
83      protected Object wrapObject(Object o, Name name, boolean replace) throws NamingException {
84              try {
85                  // Add wrapper for the two first cases. Then it will use PortableRemoteObject instead of UnicastRemoteObject
86                  // and when fixing JRMP exported objects port, it use JRMPProdelegate which is OK.
87                  if ((!(o instanceof Remote)) && (o instanceof Referenceable)) {
88                      return new UnicastJNDIReferenceWrapper(((Referenceable) o).getReference(), getObjectPort());
89                  } else if ((!(o instanceof Remote)) && (o instanceof Reference)) {
90                      return new UnicastJNDIReferenceWrapper((Reference) o, getObjectPort());
91                  } else if ((!(o instanceof Remote)) && (!(o instanceof Referenceable)) && (!(o instanceof Reference))
92                          && (o instanceof Serializable)) {
93                      // Only Serializable (not implementing Remote or Referenceable or
94                      // Reference)
95                      JNDIResourceWrapper irw = new JNDIResourceWrapper((Serializable) o);
96                      PortableRemoteObjectDelegate proDelegate = ConfigurationRepository.getCurrentConfiguration().getProtocol().getPortableRemoteObject();
97                      proDelegate.exportObject(irw);
98  
99                      Remote oldObj = (Remote) addToExported(name, irw);
100                     if (oldObj != null) {
101                         if (replace) {
102                             proDelegate.unexportObject(oldObj);
103                         } else {
104                             proDelegate.unexportObject(irw);
105                             addToExported(name, oldObj);
106                             throw new NamingException("Object '" + o + "' with name '" + name + "' is already bind");
107                         }
108                     }
109                     return irw;
110                 } else {
111                     return o;
112                 }
113             } catch (Exception e) {
114                 throw NamingExceptionHelper.create("Cannot wrap object '" + o + "' with name '" + name + "' : "
115                         + e.getMessage(), e);
116             }
117     }
118 }
