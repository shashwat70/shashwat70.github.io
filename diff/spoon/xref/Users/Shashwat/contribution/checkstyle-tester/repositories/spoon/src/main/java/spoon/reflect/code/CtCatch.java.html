6   package spoon.reflect.code;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  
11  import static spoon.reflect.path.CtRole.BODY;
12  import static spoon.reflect.path.CtRole.PARAMETER;
13  
14  
15  /**
16   * This code element defines a <code>catch</code> of a <code>try</code>.
17   *
18   * @see spoon.reflect.code.CtTry
19   */
20  public interface CtCatch extends CtCodeElement, CtBodyHolder {
21  
22  	/**
23  	 * Gets the catch's parameter (a throwable).
24  	 */
25  	@PropertyGetter(role = PARAMETER)
26  	CtCatchVariable<? extends Throwable> getParameter();
27  
28  	/**
29  	 * Sets the catch's parameter (a throwable).
30  	 */
31  	@PropertySetter(role = PARAMETER)
32  	<T extends CtCatch> T setParameter(CtCatchVariable<? extends Throwable> parameter);
33  
34  	/**
35  	 * Gets the catch's body.
36  	 */
37  	@Override
38  	@PropertyGetter(role = BODY)
39  	CtBlock<?> getBody();
40  
41  	@Override
42  	CtCatch clone();
43  }
