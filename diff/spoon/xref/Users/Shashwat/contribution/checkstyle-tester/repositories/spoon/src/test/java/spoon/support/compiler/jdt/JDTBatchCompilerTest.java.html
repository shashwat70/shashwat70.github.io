17  package spoon.support.compiler.jdt;
18  
19  import org.junit.Test;
20  
21  import spoon.Launcher;
22  import java.io.File;
23  
24  import static org.junit.Assert.assertTrue;
25  
26  public class JDTBatchCompilerTest {
27  
28  	@Test
29  	public void testCompileGeneratedJavaFile() {
30  		final Launcher launcher = new Launcher();
31  		launcher.setArgs(new String[] {"--output-type", "nooutput" });
32  		launcher.addInputResource("./src/test/java/spoon/support/compiler/jdt/testclasses/Foo.java");
33  		launcher.setBinaryOutputDirectory("./target/binaries");
34  		launcher.getEnvironment().setShouldCompile(true);
35  		launcher.buildModel();
36  
37  		launcher.getFactory().Class().create("spoon.Test");
38  		assertTrue(launcher.getModelBuilder().compile());
39  		assertTrue(new File("./target/binaries/spoon/Test.class").exists());
40  	}
41  }
