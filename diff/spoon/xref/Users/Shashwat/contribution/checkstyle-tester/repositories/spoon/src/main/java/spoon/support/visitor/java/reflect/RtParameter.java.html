6   package spoon.support.visitor.java.reflect;
7   
8   import spoon.SpoonException;
9   
10  import java.lang.annotation.Annotation;
11  import java.lang.reflect.Constructor;
12  import java.lang.reflect.Type;
13  
14  /**
15   * To be compatible with Java 6, RtParameter has been created from
16   * the Parameter class in Java 8.
17   */
18  public class RtParameter {
19  	private final String name;
20  	private final Class<?> type;
21  	private final Type genericType;
22  	private final RtMethod method;
23  	private final Constructor constructor;
24  	private final int index;
25  
26  	public RtParameter(String name, Class<?> type, Type genericType, RtMethod method, Constructor constructor, int index) {
27  		this.name = name;
28  		this.type = type;
29  		this.genericType = genericType;
30  		this.method = method;
31  		this.constructor = constructor;
32  		this.index = index;
33  	}
34  
35  	/**
36  	 * Returns the name of the parameter.
37  	 *
38  	 * @return The name of the parameter, either provided by the class
39  	 * file or synthesized if the class file does not provide
40  	 * a name.
41  	 */
42  	public String getName() {
43  		// Note: empty strings as parameter names are now outlawed.
44  		// The .equals("") is for compatibility with current JVM
45  		// behavior.  It may be removed at some point.
46  		if (name == null || name.isEmpty()) {
47  			return "arg" + index;
48  		} else {
49  			return name;
50  		}
51  	}
52  
53  	/**
54  	 * Returns the {@code Class} which is the type of the parameter.
55  	 *
56  	 * @return The type of the parameter.
57  	 */
58  	public Class<?> getType() {
59  		return type;
60  	}
61  
62  	public Type getGenericType() {
63  		return genericType;
64  	}
65  
66  	/**
67  	 * Return annotations which declared on the parameter.
68  	 *
69  	 * @return Annotations declared on the parameter.
70  	 */
71  	public Annotation[] getDeclaredAnnotations() {
72  		if (method == null) {
73  			/*
74  			 * According to oracle sources (for jdk 8) for java.lang.reflect.Executable#getParameterAnnotations(),
75  			 * the length of the returned array may vary at the discretion of the compiler.
76  			 * It seems that eclipse and javac do not produce the same results.
77  			 * In our case the value of index is based on getGenericParameterTypes(), so if
78  			 * getParameterAnnotations() returns a smaller array, we can assume that
79  			 * a synthetic parameter pointing to the outer class has been added at the begining of
80  			 * getGenericParameterTypes() and not in getParameterAnnotations().
81  			 * The actual index is then shifted by the difference.
82  			 */
83  			int diff = constructor.getGenericParameterTypes().length - constructor.getParameterAnnotations().length;
84  			return constructor.getParameterAnnotations()[index - diff];
85  		}
86  		return method.getParameterAnnotations()[index];
87  	}
88  
89  	/**
90  	 * Returns {@code true} if this parameter represents a variable
91  	 * argument list; returns {@code false} otherwise.
92  	 *
93  	 * @return {@code true} if an only if this parameter represents a
94  	 * variable argument list.
95  	 */
96  	public boolean isVarArgs() {
97  		if (method == null) {
98  			return constructor.isVarArgs() && index == constructor.getParameterTypes().length - 1;
99  		}
100 		return method.isVarArgs() && index == method.getParameterTypes().length - 1;
101 	}
102 
103 	/**
104 	 * Compares based on the executable and the index.
105 	 *
106 	 * @param obj
107 	 * 		The object to compare.
108 	 * @return Whether or not this is equal to the argument.
109 	 */
110 	public boolean equals(Object obj) {
111 		if (obj instanceof RtParameter) {
112 			RtParameter other = (RtParameter) obj;
113 			if (method == null) {
114 				return (other.constructor.equals(constructor) && other.index == index);
115 			}
116 			return (other.method.equals(method) && other.index == index);
117 		}
118 		return false;
119 	}
120 
121 	/**
122 	 * Returns a hash code based on the executable's hash code and the
123 	 * index.
124 	 *
125 	 * @return A hash code based on the executable's hash code.
126 	 */
127 	public int hashCode() {
128 		if (method == null) {
129 			return constructor.hashCode() ^ index;
130 		}
131 		return method.hashCode() ^ index;
132 	}
133 
134 	/**
135 	 * Get parameters in a method to spoon parameters.
136 	 *
137 	 * @param method
138 	 * 		Parent executable of parameters.
139 	 * @return Parameters of the executable.
140 	 */
141 	public static RtParameter[] parametersOf(RtMethod method) {
142 		RtParameter[] parameters = new RtParameter[method.getParameterTypes().length];
143 		for (int index = 0; index < method.getParameterTypes().length; index++) {
144 			parameters[index] = new RtParameter(null, method.getParameterTypes()[index], method.getGenericParameterTypes()[index], method, null, index);
145 		}
146 		return parameters;
147 	}
148 
149 	/**
150 	 * Get parameters in a constructor to spoon parameters.
151 	 *
152 	 * @param constructor
153 	 * 		Parent executable of parameters.
154 	 * @return Parameters of the executable.
155 	 */
156 	public static RtParameter[] parametersOf(Constructor constructor) {
157 		RtParameter[] parameters;
158 		// Apparently getGenericParameterTypes and getParameterTypes could have different length
159 		// if the first parameter is implicit: a private non-static inner class will have an implicit parameter for its superclass
160 		// but it won't be present in the result of getGenericParameterTypes (e.g. ArrayList$SubList)
161 		// moreover if it's an enum, there will be 2 implicit parameters (String and int) we won't consider them in the model.
162 		int lengthGenericParameterTypes = constructor.getGenericParameterTypes().length;
163 		int lengthParameterTypes = constructor.getParameterTypes().length;
164 
165 		int offset;
166 		if (lengthGenericParameterTypes == lengthParameterTypes) {
167 			parameters = new RtParameter[lengthParameterTypes];
168 			offset = 0;
169 		} else if (lengthGenericParameterTypes == lengthParameterTypes - 1) {
170 			parameters = new RtParameter[lengthGenericParameterTypes];
171 			offset = 1;
172 		} else if (constructor.getDeclaringClass().isEnum() && lengthGenericParameterTypes == lengthParameterTypes - 2) {
173 			parameters = new RtParameter[lengthGenericParameterTypes];
174 			offset = 2;
175 		} else {
176 			throw new SpoonException("Error while analyzing parameters of constructor: " + constructor + ". # of parameters: " + lengthParameterTypes + " - # of generic parameter types: " + lengthGenericParameterTypes);
177 		}
178 
179 		for (int index = 0; index < constructor.getGenericParameterTypes().length; index++) {
180 			parameters[index] = new RtParameter(null, constructor.getParameterTypes()[index + offset], constructor.getGenericParameterTypes()[index], null, constructor, index);
181 		}
182 		return parameters;
183 	}
184 }
