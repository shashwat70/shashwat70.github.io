6   package spoon.reflect.visitor.filter;
7   
8   import spoon.reflect.code.CtCFlowBreak;
9   import spoon.reflect.code.CtReturn;
10  import spoon.reflect.code.CtThrow;
11  import spoon.reflect.visitor.Filter;
12  
13  /**
14   * This simple filter matches all the occurrences of a return or a throw
15   * statement (end of execution flow).
16   */
17  public class ReturnOrThrowFilter implements Filter<CtCFlowBreak> {
18  
19  	@Override
20  	public boolean matches(CtCFlowBreak cflowBreak) {
21  		return (cflowBreak instanceof CtReturn)
22  				|| (cflowBreak instanceof CtThrow);
23  	}
24  }
