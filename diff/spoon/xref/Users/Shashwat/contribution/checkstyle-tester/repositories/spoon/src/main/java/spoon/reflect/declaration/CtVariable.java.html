6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.code.CtExpression;
9   import spoon.reflect.reference.CtVariableReference;
10  import spoon.support.DerivedProperty;
11  import spoon.reflect.annotations.PropertyGetter;
12  import spoon.reflect.annotations.PropertySetter;
13  
14  import static spoon.reflect.path.CtRole.DEFAULT_EXPRESSION;
15  
16  /**
17   * This abstract element defines a variable declaration.
18   */
19  public interface CtVariable<T> extends CtNamedElement, CtTypedElement<T>, CtModifiable {
20  	/**
21  	 * Gets the initialization expression assigned to the variable (also known
22  	 * as the initializer), when declared.
23  	 */
24  	@PropertyGetter(role = DEFAULT_EXPRESSION)
25  	CtExpression<T> getDefaultExpression();
26  
27  	/*
28  	 * (non-Javadoc)
29  	 *
30  	 * @see spoon.reflect.declaration.CtNamedElement#getReference()
31  	 */
32  	@Override
33  	@DerivedProperty
34  	CtVariableReference<T> getReference();
35  
36  	/**
37  	 * Sets the initialization expression assigned to the variable, when
38  	 * declared.
39  	 */
40  	@PropertySetter(role = DEFAULT_EXPRESSION)
41  	<C extends CtVariable<T>> C setDefaultExpression(CtExpression<T> assignedExpression);
42  }
