6   package spoon.template;
7   
8   import spoon.SpoonException;
9   
10  /**
11   * This runtime exception can be throws when something wrong occurs in template
12   * operations such as loading, substitution, and matching.
13   *
14   * @see spoon.template.Substitution
15   * @see spoon.template.TemplateMatcher
16   */
17  public class TemplateException extends SpoonException {
18  
19  	private static final long serialVersionUID = 1L;
20  
21  	/**
22  	 * Empty exception.
23  	 */
24  	public TemplateException() {
25  	}
26  
27  	/**
28  	 * Exception with a message.
29  	 */
30  	public TemplateException(String message) {
31  		super(message);
32  	}
33  
34  	/**
35  	 * Exception with a cause.
36  	 */
37  	public TemplateException(Throwable cause) {
38  		super(cause);
39  	}
40  
41  }
