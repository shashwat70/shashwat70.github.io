17  package spoon.test.imports;
18  
19  import org.junit.Test;
20  import spoon.Launcher;
21  import spoon.reflect.declaration.CtType;
22  import spoon.reflect.reference.CtTypeReference;
23  
24  import java.util.Collection;
25  
26  import static org.junit.Assert.assertEquals;
27  
28  public class ImportAndExtendWithPackageNameTest {
29  
30      private static final String inputResource =
31              "./src/test/resources/import-resources/ImportAndExtendWithPackageName.java";
32  
33      @Test
34      public void testBuildModel() {
35          final Launcher runLaunch = new Launcher();
36          runLaunch.getEnvironment().setNoClasspath(true);
37          runLaunch.addInputResource(inputResource);
38          runLaunch.buildModel();
39  
40          final Collection<CtType<?>> types = runLaunch.getModel().getAllTypes();
41          assertEquals(1, types.size());
42  
43          final CtType type = types.iterator().next();
44          assertEquals("ImportAndExtendWithPackageName", type.getSimpleName());
45  
46          final CtTypeReference superClass = type.getSuperclass();
47          assertEquals("LLkParser", superClass.getSimpleName());
48      }
49  }
