6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.reference.CtTypeReference;
9   import spoon.support.DerivedProperty;
10  import spoon.reflect.annotations.PropertyGetter;
11  import spoon.reflect.annotations.PropertySetter;
12  
13  import java.util.List;
14  import java.util.Set;
15  
16  import static spoon.reflect.path.CtRole.METHOD;
17  import static spoon.reflect.path.CtRole.FIELD;
18  import static spoon.reflect.path.CtRole.INTERFACE;
19  import static spoon.reflect.path.CtRole.NAME;
20  import static spoon.reflect.path.CtRole.NESTED_TYPE;
21  import static spoon.reflect.path.CtRole.SUPER_TYPE;
22  import static spoon.reflect.path.CtRole.TYPE_MEMBER;
23  
24  /**
25   * This abstract element defines a super-type for classes and interfaces, which
26   * can declare methods.
27   *
28   * The type parameter T refers to the actual class that this type represents.
29   */
30  public interface CtType<T> extends CtNamedElement, CtTypeInformation, CtTypeMember, CtFormalTypeDeclarer, CtShadowable {
31  	/**
32  	 * The string separator in a Java innertype qualified name.
33  	 */
34  	String INNERTTYPE_SEPARATOR = "$";
35  	/**
36  	 * Used in no classpath when we don't have any information to build the name of the type.
37  	 */
38  	String NAME_UNKNOWN = "<unknown>";
39  
40  	/**
41  	 * Returns the simple (unqualified) name of this element.
42  	 * Following the compilation convention, if the type is a local type,
43  	 * the name starts with a numeric prefix (e.g. local class Foo has simple name 1Foo).
44  	 */
45  	@Override
46  	@PropertyGetter(role = NAME)
47  	String getSimpleName();
48  
49  	/**
50  	 * Returns the types used by this type.
51  	 *
52  	 * @param includeSamePackage
53  	 * 		set to true if the method should return also the types located
54  	 * 		in the same package as the current type
55  	 */
56  	Set<CtTypeReference<?>> getUsedTypes(boolean includeSamePackage);
57  
58  	/**
59  	 *
60  	 * NEVER USE THIS.
61  	 *
62  	 * See {@link CtTypeReference#getActualClass()}.
63  	 *
64  	 * @deprecated (since Spoon 7.0.0) this will be removed from the public API
65  	 */
66  	@DerivedProperty
67  	Class<T> getActualClass();
68  
69  	/**
70  	 * Gets a field from its name.
71  	 *
72  	 * @return null if does not exit
73  	 */
74  	@PropertyGetter(role = FIELD)
75  	CtField<?> getField(String name);
76  
77  	/**
78  	 * Returns the fields that are directly declared by this class or interface.
79  	 * Includes enum constants.
80  	 *
81  	 * Derived from {@link #getTypeMembers()}
82  	 */
83  	@DerivedProperty
84  	@PropertyGetter(role = FIELD)
85  	List<CtField<?>> getFields();
86  
87  	/**
88  	 * Gets a nested type from its name.
89  	 */
90  	@PropertyGetter(role = NESTED_TYPE)
91  	<N extends CtType<?>> N getNestedType(String name);
92  
93  	/**
94  	 * Returns the declarations of the nested classes and interfaces that are
95  	 * directly declared by this class or interface.
96  	 */
97  	@DerivedProperty
98  	@PropertyGetter(role = NESTED_TYPE)
99  	Set<CtType<?>> getNestedTypes();
100 
101 	/**
102 	 * Gets the package where this type is declared.
103 	 */
104 	@DerivedProperty
105 	CtPackage getPackage();
106 
107 	/**
108 	 * Gets the entire class code with package and imports.
109 	 */
110 	@DerivedProperty
111 	String toStringWithImports();
112 
113 	/**
114 	 * Returns the corresponding type reference.
115 	 *
116 	 * Overrides the return type.
117 	 */
118 	@Override
119 	@DerivedProperty
120 	CtTypeReference<T> getReference();
121 
122 	/**
123 	 * Returns true if this type is top-level (declared as the main type in a
124 	 * file).
125 	 */
126 	boolean isTopLevel();
127 
128 	/**
129 	 * Adds a field at the top of the type (before static block).
130 	 * Note that the position of these field will be negative to be written at the top of the type.
131 	 *
132 	 * @param field
133 	 * @return <tt>true</tt> if the field is added.
134 	 */
135 	@PropertySetter(role = FIELD)
136 	<F, C extends CtType<T>> C addFieldAtTop(CtField<F> field);
137 
138 	/**
139 	 * add a field at the end of the field list.
140 	 *
141 	 * @param field
142 	 * @return <tt>true</tt> if this element changed as a result of the call
143 	 */
144 	@PropertySetter(role = FIELD)
145 	<F, C extends CtType<T>> C addField(CtField<F> field);
146 
147 	/**
148 	 * add a field at a given position.
149 	 *
150 	 * @param field
151 	 * @return <tt>true</tt> if this element changed as a result of the call
152 	 */
153 	@PropertySetter(role = FIELD)
154 	<F, C extends CtType<T>> C addField(int index, CtField<F> field);
155 
156 	/**
157 	 * Sets all fields in the type.
158 	 */
159 	@PropertySetter(role = FIELD)
160 	<C extends CtType<T>> C setFields(List<CtField<?>> fields);
161 
162 	/**
163 	 * remove a Field
164 	 *
165 	 * @param field
166 	 * @return <tt>true</tt> if this element changed as a result of the call
167 	 */
168 	@PropertySetter(role = FIELD)
169 	<F> boolean removeField(CtField<F> field);
170 
171 	/**
172 	 * Add a nested type.
173 	 *
174 	 * @param nestedType
175 	 * @return <tt>true</tt> if this element changed as a result of the call
176 	 */
177 	@PropertySetter(role = NESTED_TYPE)
178 	<N, C extends CtType<T>> C addNestedType(CtType<N> nestedType);
179 
180 	/**
181 	 * Remove a nested type.
182 	 *
183 	 * @param nestedType
184 	 * @return <tt>true</tt> if this element changed as a result of the call
185 	 */
186 	@PropertySetter(role = NESTED_TYPE)
187 	<N> boolean removeNestedType(CtType<N> nestedType);
188 
189 	/**
190 	 * Sets all nested types.
191 	 */
192 	@PropertySetter(role = NESTED_TYPE)
193 	<C extends CtType<T>> C setNestedTypes(Set<CtType<?>> nestedTypes);
194 
195 	/**
196 	 * Replace all the code snippets that are found in this type by the corresponding Spoon AST.
197 	 *
198 	 * @see CtCodeSnippet
199 	 * @see spoon.reflect.code.CtCodeSnippetExpression
200 	 * @see spoon.reflect.code.CtCodeSnippetStatement
201 	 */
202 	void compileAndReplaceSnippets();
203 
204 	/**
205 	 * Return all the methods that can be called on an instance of this type.
206 	 * It recursively collects all methods (both concrete and abstract) from all super-classes and all super-interfaces.
207 	 * It deduplicates methods with the same signature, which are defined several times somewhere in the type hierarchy.
208 	 *
209 	 * Warning: this method can be really slow due to signature deduplication.
210 	 *
211 	 * It includes all methods: the methods of types whose source code is in the Spoon model,
212 	 * the methods of types from the JDK and from libraries present in the classpath,
213 	 * the methods of java.lang.Object (for all CtClass objects).
214 	 * However, in noclasspath mode, it does not include methods from unknown types.
215 	 * If methods are overridden somewhere in the type hierarchy, it returns only top methods (ie method definitions).
216 	 */
217 	@DerivedProperty
218 	Set<CtMethod<?>> getAllMethods();
219 
220 	/**
221 	 * Gets a method from its return type, name, and parameter types.
222 	 *
223 	 * @return null if does not exit
224 	 */
225 	@PropertyGetter(role = METHOD)
226 	<R> CtMethod<R> getMethod(CtTypeReference<R> returnType, String name, CtTypeReference<?>... parameterTypes);
227 
228 	/**
229 	 * Gets a method from its name and parameter types.
230 	 *
231 	 * @return null if does not exit
232 	 */
233 	@PropertyGetter(role = METHOD)
234 	<R> CtMethod<R> getMethod(String name, CtTypeReference<?>... parameterTypes);
235 
236 	/**
237 	 * Returns the methods that are directly declared by this class or
238 	 * interface.
239 	 *
240 	 * Derived from {@link #getTypeMembers()}
241 	 *
242 	 */
243 	@DerivedProperty
244 	@PropertyGetter(role = METHOD)
245 	Set<CtMethod<?>> getMethods();
246 
247 	/**
248 	 * Returns the methods that are directly declared by this class or
249 	 * interface and annotated with one of the given annotations.
250 	 */
251 	@PropertyGetter(role = METHOD)
252 	Set<CtMethod<?>> getMethodsAnnotatedWith(CtTypeReference<?>... annotationTypes);
253 
254 	/**
255 	 * Returns the methods that are directly declared by this class or
256 	 * interface and that have the given name.
257 	 */
258 	@PropertyGetter(role = METHOD)
259 	List<CtMethod<?>> getMethodsByName(String name);
260 
261 	/**
262 	 * Searches in the type for the given method.
263 	 * Super classes and implemented interfaces are considered.
264 	 * The matching criterion is that the signatures are identical.
265 	 * @param method The method to search for in the class.
266 	 * @return True: the type has the given method. False otherwise.
267 	 */
268 	boolean hasMethod(CtMethod<?> method);
269 
270 	/**
271 	 * Sets the methods of this type.
272 	 */
273 	@PropertySetter(role = METHOD)
274 	<C extends CtType<T>> C setMethods(Set<CtMethod<?>> methods);
275 
276 	/**
277 	 * Adds a method to this type.
278 	 */
279 	@PropertySetter(role = METHOD)
280 	<M, C extends CtType<T>> C addMethod(CtMethod<M> method);
281 
282 	/**
283 	 * Removes a method from this type.
284 	 */
285 	@PropertySetter(role = METHOD)
286 	<M> boolean removeMethod(CtMethod<M> method);
287 
288 	/**
289 	 * Sets the superclass type.
290 	 */
291 	@PropertySetter(role = SUPER_TYPE)
292 	<C extends CtType<T>> C setSuperclass(CtTypeReference<?> superClass);
293 
294 	/**
295 	 * Sets the super interfaces of this type.
296 	 */
297 	@PropertySetter(role = INTERFACE)
298 	<C extends CtType<T>> C setSuperInterfaces(Set<CtTypeReference<?>> interfaces);
299 
300 	/**
301 	 * @param interfac
302 	 * @return <tt>true</tt> if this element changed as a result of the call
303 	 */
304 	@PropertySetter(role = INTERFACE)
305 	<S, C extends CtType<T>> C addSuperInterface(CtTypeReference<S> interfac);
306 
307 	/**
308 	 * @param interfac
309 	 * @return <tt>true</tt> if this element changed as a result of the call
310 	 */
311 	@PropertySetter(role = INTERFACE)
312 	<S> boolean removeSuperInterface(CtTypeReference<S> interfac);
313 
314 	/**
315 	 * Gets all type members of the type like fields, methods, anonymous block, etc.
316 	 */
317 	@PropertyGetter(role = TYPE_MEMBER)
318 	List<CtTypeMember> getTypeMembers();
319 
320 	/**
321 	 * Adds a type member at the end of all type member of the type.
322 	 */
323 	@PropertySetter(role = TYPE_MEMBER)
324 	<C extends CtType<T>> C addTypeMember(CtTypeMember member);
325 
326 	/**
327 	 * Adds a type member at a given position. Think to use this method if the order is
328 	 * important for you.
329 	 */
330 	@PropertySetter(role = TYPE_MEMBER)
331 	<C extends CtType<T>> C addTypeMemberAt(int position, CtTypeMember member);
332 
333 	/**
334 	 * Removes the type member.
335 	 */
336 	@PropertySetter(role = TYPE_MEMBER)
337 	boolean removeTypeMember(CtTypeMember member);
338 
339 	/**
340 	 * Removes all types members with these new members.
341 	 */
342 	@PropertySetter(role = TYPE_MEMBER)
343 	<C extends CtType<T>> C setTypeMembers(List<CtTypeMember> members);
344 
345 	@Override
346 	CtType<T> clone();
347 
348 	/**
349 	 * Copy the type, where copy means cloning + porting all the references in the clone from the old type to the new type.
350 	 *
351 	 * The copied type is added to the same package (and this to the factory as well).
352 	 *
353 	 * A new unique method name is given for each copy, and this method can be called several times.
354 	 */
355 	CtType<?> copyType();
356 }
