6   package spoon.support.util;
7   
8   import java.util.Map;
9   
10  /**
11   *
12   * An immutable map.
13   * (eg unmodifiable storage of parameter name-value pairs).
14   * The values may be primitive values or List, Set, Map of values.
15   * All internal containers are unmodifiable too.
16   *
17   * Internal class only, not in the public API.
18   */
19  public interface ImmutableMap {
20  
21  	/**
22  	 * @param parameterName to be checked parameter name
23  	 * @return true if there is defined some value for the parameter. null can be a value too
24  	 */
25  	boolean hasValue(String parameterName);
26  
27  	/**
28  	 * @param parameterName the name of the parameter
29  	 * @return a value of the parameter under the name `parameterNamer
30  	 */
31  	Object getValue(String parameterName);
32  
33  	/**
34  	 * @param parameterName to be set parameter name
35  	 * @param value the new value
36  	 * @return copies this {@link ImmutableMap}, sets the new value there and returns that copy
37  	 */
38  	ImmutableMap putValue(String parameterName, Object value);
39  
40  	/**
41  	 * @return underlying unmodifiable Map&lt;String, Object&gt;
42  	 */
43  	Map<String, Object> asMap();
44  
45  	/**
46  	 * @return a new instance of {@link ImmutableMap}, which inherits all values from this {@link ImmutableMap}
47  	 * Any call of {@link #putValue(String, Object)} is remembered in local Map of parameters.
48  	 * At the end of process the {@link #getModifiedValues()} can be used to return all the parameters which were changed
49  	 * after local {@link ImmutableMap} was created
50  	 */
51  	ImmutableMap checkpoint();
52  
53  	/**
54  	 * @return the modified parameters since last call to {@link #checkpoint()}
55  	 */
56  	Map<String, Object> getModifiedValues();
57  }
