6   package spoon.reflect.visitor.chain;
7   
8   import spoon.reflect.declaration.CtElement;
9   import spoon.reflect.path.CtRole;
10  
11  /**
12   * Responsible for performing an action when a scanner enters/exits a node while scanning the AST.
13   */
14  public interface CtScannerListener {
15  	/**
16  	 * Called before the scanner enters an element
17  	 *
18  	 * @param element the element about to be scanned.
19  	 * @return a {@link ScanningMode} that drives how the scanner processes this element and its children.
20  	 * For instance, returning {@link ScanningMode#SKIP_ALL} causes that element and all children to be skipped and {@link #exit(CtElement)} are be NOT called for that element.
21  	 */
22  	default ScanningMode enter(CtElement element) {
23  		return ScanningMode.NORMAL;
24  	}
25  
26  	/**
27  	 * Called before the scanner enters an element
28  	 *
29  	 * @param role the {@link CtRole}, which `element` has in its parent. It is null for the first scanned element
30  	 * @param element the element about to be scanned.
31  	 * @return a {@link ScanningMode} that drives how the scanner processes this element and its children.
32  	 * For instance, returning {@link ScanningMode#SKIP_ALL} causes that element and all children to be skipped and {@link #exit(CtElement)} are be NOT called for that element.
33  	 */
34  	default ScanningMode enter(CtRole role, CtElement element) {
35  		return enter(element);
36  	}
37  
38  	/**
39  	 * This method is called after the element and all its children have been visited.
40  	 * This method is NOT called if an exception is thrown in {@link #enter(CtElement)} or during the scanning of the element or any of its children element.
41  	 * This method is NOT called for an element for which {@link #enter(CtElement)} returned {@link ScanningMode#SKIP_ALL}.
42  	 *
43  	 * @param element the element that has just been scanned.
44  	 */
45  	default void exit(CtElement element) {
46  	}
47  	/**
48  	 * This method is called after the element and all its children have been visited.
49  	 * This method is NOT called if an exception is thrown in {@link #enter(CtElement)} or during the scanning of the element or any of its children element.
50  	 * This method is NOT called for an element for which {@link #enter(CtElement)} returned {@link ScanningMode#SKIP_ALL}.
51  	 *
52  	 * @param role the {@link CtRole}, which `element` has in its parent. It is null for the first scanned element
53  	 * @param element the element that has just been scanned.
54  	 */
55  	default void exit(CtRole role, CtElement element) {
56  		exit(element);
57  	}
58  }
