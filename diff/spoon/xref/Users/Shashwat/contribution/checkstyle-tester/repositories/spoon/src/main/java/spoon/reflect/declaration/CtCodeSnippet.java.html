6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  
11  import static spoon.reflect.path.CtRole.SNIPPET;
12  
13  /**
14   * This interface represents snippets of source code that can be used in the AST
15   * to represent complex code without having to build the corresponding program
16   * model structure. It is mainly provided on simplification purpose in order to
17   * avoid having to build the program's model. Code snippets should be compiled
18   * to validate their contents and the result of the compilation should be used
19   * to replace the code snippet in the final AST.
20   *
21   * @see CtType#compileAndReplaceSnippets()
22   */
23  public interface CtCodeSnippet {
24  
25  	/**
26  	 * Sets the textual value of the code.
27  	 */
28  	@PropertySetter(role = SNIPPET)
29  	<C extends CtCodeSnippet> C setValue(String value);
30  
31  	/**
32  	 * Gets the textual value of the code.
33  	 */
34  	@PropertyGetter(role = SNIPPET)
35  	String getValue();
36  
37  }
