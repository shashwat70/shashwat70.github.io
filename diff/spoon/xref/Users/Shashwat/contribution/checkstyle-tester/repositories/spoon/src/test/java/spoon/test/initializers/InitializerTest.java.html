38  	@Test
39  	public void testModelBuildingStaticInitializer() throws Exception {
40  		CtClass<?> type = build("spoon.test.initializers.testclasses",
41  				"InternalClassStaticFieldInit");
42  		assertEquals("InternalClassStaticFieldInit", type.getSimpleName());
43  
44  		CtClass<?> InternalClass = type.getNestedType("InternalClass");
45  		assertTrue(InternalClass.getModifiers().contains(ModifierKind.STATIC));
46  		CtAnonymousExecutable staticBlock = type.getElements(new TypeFilter<>(CtAnonymousExecutable.class)).get(0);
47  		assertTrue(staticBlock.getModifiers().contains(ModifierKind.STATIC));
48  		assertEquals(1, staticBlock.getBody().getStatements().size());
49  
50  		// this fails: regression or known bug?
51  		// RP: this look OK. Spoon adds the full path
52  		// assertEquals("InternalClass.tmp = \"nop\"",
53  		// staticBlock.getBody().getStatements().get(0).toString());
54  	}
55  
56  	@Test
57  	public void testModelBuildingInitializer() throws Exception {
58  		CtClass<?> type = build("spoon.test.initializers.testclasses",
59  				"InstanceInitializers");
60  		assertEquals("InstanceInitializers", type.getSimpleName());
61  
62  		CtField<?> k = type.getElements(new NamedElementFilter<>(CtField.class,"k")).get(0);
63  		assertTrue(k.getDefaultExpression() instanceof CtConstructorCall);
64  
65  		CtField<?> l = type.getElements(new NamedElementFilter<>(CtField.class,"l")).get(0);
66  		assertTrue(l.getDefaultExpression() instanceof CtConstructorCall);
67  
68  		CtField<?> x = type.getElements(new NamedElementFilter<>(CtField.class,"x")).get(0);
69  		assertNull(x.getDefaultExpression());
70  
71  		CtField<?> y = type.getElements(new NamedElementFilter<>(CtField.class,"y")).get(0);
72  		assertTrue(y.getDefaultExpression() instanceof CtLiteral);
73  
74  		CtField<?> z = type.getElements(new NamedElementFilter<>(CtField.class,"z")).get(0);
75  		assertTrue("5".equals(z.getDefaultExpression().toString()));
76  
77  		// static initializer
78  		CtAnonymousExecutable ex = type.getElements(new TypeFilter<>(CtAnonymousExecutable.class)).get(0);
79  		assertEquals("x = 3", ex.getBody().getStatements().get(0).toString());
80  	}
81  
82  	@Test
83  	public void testModelBuildingInitializerNoclasspath() {
84  		Launcher launcher = new Launcher();
85  		launcher.addInputResource("./src/test/resources/noclasspath/initializer/Utf8HttpResponse.java");
86  		launcher.getEnvironment().setNoClasspath(true);
87  		launcher.getEnvironment().setAutoImports(true);
88  		CtModel model = launcher.buildModel();
89  		CtClass<?> ctClass = model.getElements(new NamedElementFilter<>(CtClass.class, "Utf8HttpResponse")).get(0);
90  
91  		CtAnonymousExecutable ex = ctClass.getElements(new TypeFilter<>(CtAnonymousExecutable.class)).get(0);
92  		// we are indeed in autoimport
93  		assertEquals("UnicodeUtil.UTF8Result temp = new UnicodeUtil.UTF8Result()",
94  				ex.getBody().getStatements().get(0).toString());
95  		assertEquals("temp.result = new byte[0]",
96  				ex.getBody().getStatements().get(1).toString());
97  		assertTrue(ImportTest.printByPrinter(ctClass).contains("UnicodeUtil.UTF8Result temp = new UnicodeUtil.UTF8Result()"));
98  	}
