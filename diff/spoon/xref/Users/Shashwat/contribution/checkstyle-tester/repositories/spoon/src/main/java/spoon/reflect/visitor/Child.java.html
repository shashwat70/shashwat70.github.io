6   package spoon.reflect.visitor;
7   
8   import java.lang.annotation.ElementType;
9   import java.lang.annotation.Target;
10  
11  /**
12   * This annotation defines a method that contains children of an element of the
13   * meta model. It is used for the automatic generation of visitors of spoon
14   * metamodel elements.
15   */
16  @Target({ ElementType.METHOD })
17  public @interface Child {
18  	/**
19  	 * The ordering of the child relatively to the other children of the parent
20  	 * node. Children will be scanned in the declared order.
21  	 */
22  	int order() default 0;
23  }
