17  package spoon.test.enums;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertTrue;
21  
22  import java.util.List;
23  
24  import org.junit.Test;
25  
26  import spoon.Launcher;
27  import spoon.compiler.SpoonResource;
28  import spoon.compiler.SpoonResourceHelper;
29  import spoon.reflect.code.CtAssignment;
30  import spoon.reflect.declaration.CtEnum;
31  import spoon.reflect.declaration.CtType;
32  import spoon.reflect.factory.Factory;
33  import spoon.reflect.reference.CtTypeReference;
34  import spoon.reflect.visitor.Query;
35  import spoon.reflect.visitor.filter.TypeFilter;
36  
37  public class EnumsTypeTest {
38  
39  	@Test
40  	public void testEnumsType() throws Exception {
41  		// contract: shadow enum should still be considered as an enum
42  		Launcher launcher = new Launcher();
43  		launcher.addInputResource("./src/test/resources/reference-test/EnumsRef.java");
44  
45  		Factory factory = launcher.getFactory();
46  		List<SpoonResource> classpath = SpoonResourceHelper.resources("./src/test/resources/reference-test/EnumJar.jar");
47  		String[] dependencyClasspath = { classpath.get(0).getPath() };
48  		factory.getEnvironment().setSourceClasspath(dependencyClasspath);
49  		assertEquals(1, classpath.size());
50  
51  		launcher.buildModel();
52  
53  		List<CtAssignment> assignments = Query.getElements(factory, new TypeFilter<>(CtAssignment.class));
54  
55  		CtTypeReference typeRefFromSource = assignments.get(0).getType();
56  		CtType typeFromSource = typeRefFromSource.getTypeDeclaration();
57  		assertTrue(typeRefFromSource.isEnum());
58  		assertTrue(typeFromSource.isEnum());
59  		assertTrue(typeFromSource instanceof CtEnum);
60  
61  		CtTypeReference typeRefFromJar = assignments.get(1).getType();
62  		CtType typeFromJar = typeRefFromJar.getTypeDeclaration();
63  		assertTrue(typeRefFromJar.isEnum());
64  		assertTrue(typeFromJar.isEnum());
65  		assertTrue(typeFromJar instanceof CtEnum);
66  	}
67  
68  	@Test
69  	public void testEnumsFromInterface() throws Exception {
70  		// contract: shadow enum from an interface should still be considered as an enum
71  		Launcher launcher = new Launcher();
72  		launcher.addInputResource("./src/test/resources/reference-test/InterfaceWithEnum.java");
73  		launcher.addInputResource("./src/test/resources/reference-test/InterfaceEnumRef.java");
74  
75  		Factory factory = launcher.getFactory();
76  		List<SpoonResource> classpath = SpoonResourceHelper.resources("./src/test/resources/reference-test/InterfaceWithEnumJar.jar");
77  		String[] dependencyClasspath = { classpath.get(0).getPath() };
78  		factory.getEnvironment().setSourceClasspath(dependencyClasspath);
79  		assertEquals(1, classpath.size());
80  
81  		launcher.buildModel();
82  
83  		List<CtAssignment> assignments = Query.getElements(factory, new TypeFilter<>(CtAssignment.class));
84  
85  		CtTypeReference typeRefFromSource = assignments.get(0).getType();
86  		CtType typeFromSource = typeRefFromSource.getTypeDeclaration();
87  		assertTrue(typeRefFromSource.isEnum());
88  		assertTrue(typeFromSource.isEnum());
89  		assertTrue(typeFromSource instanceof CtEnum);
90  
91  		CtTypeReference typeRefFromJar = assignments.get(1).getType();
92  		CtType typeFromJar = typeRefFromJar.getTypeDeclaration();
93  		assertTrue(typeRefFromJar.isEnum()); // fail
94  		assertTrue(typeFromJar.isEnum()); // fail
95  		assertTrue(typeFromJar instanceof CtEnum); // fail
96  	}
97  }
