17  package spoon.test.annotation;
18  
19  import org.junit.Test;
20  import spoon.reflect.code.CtFor;
21  import spoon.reflect.code.CtLocalVariable;
22  import spoon.reflect.declaration.CtType;
23  import spoon.reflect.visitor.filter.TypeFilter;
24  import spoon.test.annotation.testclasses.Pozole;
25  import spoon.testing.utils.ModelUtils;
26  
27  import static org.junit.Assert.assertEquals;
28  import static org.junit.Assert.assertSame;
29  
30  public class AnnotationLoopTest {
31  
32  	@Test
33  	public void testAnnotationDeclaredInForInit() throws Exception {
34  		final CtType<Pozole> aPozole = ModelUtils.buildClass(Pozole.class);
35  
36  		final CtFor aLoop = aPozole.getMethod("cook").getElements(new TypeFilter<>(CtFor.class)).get(0);
37  		assertEquals(3, aLoop.getForInit().size());
38  		assertSame(SuppressWarnings.class, aLoop.getForInit().get(0).getAnnotations().get(0).getAnnotationType().getActualClass());
39  		assertSame(SuppressWarnings.class, aLoop.getForInit().get(1).getAnnotations().get(0).getAnnotationType().getActualClass());
40  		assertSame(SuppressWarnings.class, aLoop.getForInit().get(2).getAnnotations().get(0).getAnnotationType().getActualClass());
41  
42  		assertEquals("u", ((CtLocalVariable) aLoop.getForInit().get(0)).getSimpleName());
43  		assertEquals("p", ((CtLocalVariable) aLoop.getForInit().get(1)).getSimpleName());
44  		assertEquals("e", ((CtLocalVariable) aLoop.getForInit().get(2)).getSimpleName());
45  
46  		assertEquals(aPozole.getFactory().Type().STRING, ((CtLocalVariable) aLoop.getForInit().get(0)).getType());
47  		assertEquals(aPozole.getFactory().Type().STRING, ((CtLocalVariable) aLoop.getForInit().get(1)).getType());
48  		assertEquals(aPozole.getFactory().Type().STRING, ((CtLocalVariable) aLoop.getForInit().get(2)).getType());
49  
50  		final String nl = System.lineSeparator();
51  		final String expected = "for (@java.lang.SuppressWarnings(\"rawtypes\")" + nl + "java.lang.String u = \"\", p = \"\", e = \"\"; u != e; u = p , p = \"\") {" + nl + "}";
52  		assertEquals(expected, aLoop.toString());
53  	}
54  }
