6   package spoon.support.reflect.reference;
7   
8   import spoon.reflect.code.CtLocalVariable;
9   import spoon.reflect.declaration.CtVariable;
10  import spoon.reflect.declaration.ParentNotInitializedException;
11  import spoon.reflect.factory.Factory;
12  import spoon.reflect.reference.CtLocalVariableReference;
13  import spoon.reflect.visitor.CtVisitor;
14  import spoon.reflect.visitor.filter.PotentialVariableDeclarationFunction;
15  
16  /**
17   * An implementation for {@link CtLocalVariableReference}.
18   */
19  public class CtLocalVariableReferenceImpl<T>
20  		extends CtVariableReferenceImpl<T> implements CtLocalVariableReference<T> {
21  
22  	/**
23  	 * Id for serialization.
24  	 */
25  	private static final long serialVersionUID = 1L;
26  
27  	/**
28  	 * Default constructor.
29  	 */
30  	public CtLocalVariableReferenceImpl() {
31  	}
32  
33  	@Override
34  	public void accept(CtVisitor visitor) {
35  		visitor.visitCtLocalVariableReference(this);
36  	}
37  
38  	@SuppressWarnings("unchecked")
39  	@Override
40  	public CtLocalVariable<T> getDeclaration() {
41  		// without a factory, we are not able to filter for local variables
42  		final Factory factory = getFactory();
43  		if (factory == null) {
44  			return null;
45  		}
46  
47  		final String simpleName = getSimpleName();
48  
49  		//handle the CtLocalVariableReference which were created by CtLocalVariable#getReference() and which are not yet part of model, so we cannot found them using standard rules
50  		if (parent instanceof CtLocalVariable) {
51  			CtLocalVariable<T> var = (CtLocalVariable<T>) parent;
52  			if (simpleName.equals(var.getSimpleName())) {
53  				return var;
54  			}
55  		}
56  		try {
57  			// successively iterate through all parents of this reference and
58  			// return first result (which must be the closest declaration
59  			// respecting visible scope)
60  			CtVariable<?> var = map(new PotentialVariableDeclarationFunction(simpleName)).first();
61  			if (var instanceof CtLocalVariable) {
62  				return (CtLocalVariable<T>) var;
63  			}
64  			if (var != null) {
65  				//we have found another variable declaration with same simple name, which hides declaration of this local variable reference
66  				//handle it as not found
67  				return null;
68  			}
69  		} catch (ParentNotInitializedException e) {
70  			// handle this case as 'not found'
71  		}
72  		return null;
73  	}
74  
75  	@Override
76  	public CtLocalVariableReference<T> clone() {
77  		return (CtLocalVariableReference<T>) super.clone();
78  	}
79  }
