6   package spoon.testing;
7   
8   import spoon.reflect.declaration.CtPackage;
9   import spoon.reflect.declaration.CtType;
10  
11  import java.util.ArrayList;
12  import java.util.Collections;
13  import java.util.Comparator;
14  import java.util.List;
15  
16  import static spoon.testing.Assert.assertThat;
17  import static spoon.testing.utils.Check.assertNotNull;
18  import static spoon.testing.utils.ProcessorUtils.process;
19  
20  public abstract class AbstractCtPackageAssert<T extends AbstractCtPackageAssert<T>> extends AbstractAssert<T, CtPackage> {
21  	protected AbstractCtPackageAssert(CtPackage actual, Class<?> selfType) {
22  		super(actual, selfType);
23  	}
24  
25  	/**
26  	 * Verifies that the actual value is equal to the given one.
27  	 *
28  	 * @param expected
29  	 * 		The expected package.
30  	 * @return {@code this} assertion object.
31  	 */
32  	public T isEqualTo(CtPackage expected) {
33  		assertNotNull(expected);
34  
35  		if (!actual.getSimpleName().equals(expected.getSimpleName())) {
36  			throw new AssertionError(String.format("The actual package named %1$s isn't equals to the expected package named %2$s", actual.getSimpleName(), expected.getSimpleName()));
37  		}
38  
39  		if (processors != null && !processors.isEmpty()) {
40  			process(actual.getFactory(), processors);
41  		}
42  
43  		class TypeComparator implements Comparator<CtType<?>> {
44  			@Override
45  			public int compare(CtType<?> o1, CtType<?> o2) {
46  				return o1.getSimpleName().compareTo(o2.getSimpleName());
47  			}
48  		}
49  
50  		final List<CtType<?>> actualTypes = new ArrayList<>(actual.getTypes());
51  		Collections.sort(actualTypes, new TypeComparator());
52  		final List<CtType<?>> expectedTypes = new ArrayList<>(expected.getTypes());
53  		Collections.sort(expectedTypes, new TypeComparator());
54  		for (int i = 0; i < actual.getTypes().size(); i++) {
55  			final CtType<?> actualType = actualTypes.get(i);
56  			final CtType<?> expectedType = expectedTypes.get(i);
57  			if (!actualType.toString().equals(expectedType.toString())) {
58  				throw new AssertionError(String.format("%1$s and %2$s aren't equals.", actualType.getShortRepresentation(), expectedType.getShortRepresentation()));
59  			}
60  		}
61  
62  		class PackageComparator implements Comparator<CtPackage> {
63  			@Override
64  			public int compare(CtPackage o1, CtPackage o2) {
65  				return o1.getSimpleName().compareTo(o2.getSimpleName());
66  			}
67  		}
68  
69  		final List<CtPackage> actualPackages = new ArrayList<>(actual.getPackages());
70  		Collections.sort(actualPackages, new PackageComparator());
71  		final List<CtPackage> expectedPackages = new ArrayList<>(expected.getPackages());
72  		Collections.sort(expectedPackages, new PackageComparator());
73  
74  		for (int i = 0; i < actualPackages.size(); i++) {
75  			final CtPackage actualPackage = actualPackages.get(i);
76  			final CtPackage expectedPackage = expectedPackages.get(i);
77  			assertThat(actualPackage).isEqualTo(expectedPackage);
78  		}
79  
80  		return this.myself;
81  	}
82  }
