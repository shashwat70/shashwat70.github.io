6   package spoon.support.compiler.jdt;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import org.eclipse.jdt.internal.compiler.batch.CompilationUnit;
12  
13  import spoon.SpoonModelBuilder;
14  import spoon.reflect.declaration.CtType;
15  
16  public class FactoryCompilerConfig implements SpoonModelBuilder.InputType {
17  
18  	public static final SpoonModelBuilder.InputType INSTANCE = new FactoryCompilerConfig();
19  
20  	//avoid direct instantiation. But somebody can inherit
21  	protected FactoryCompilerConfig() {
22  	}
23  
24  	/**
25  	 * returns the compilation units corresponding to the types in the factory.
26  	 */
27  	@Override
28  	public void initializeCompiler(JDTBatchCompiler compiler) {
29  		JDTBasedSpoonCompiler jdtCompiler = compiler.getJdtCompiler();
30  		List<CompilationUnit> unitList = new ArrayList<>();
31  		for (CtType<?> ctType : jdtCompiler.getFactory().Type().getAll()) {
32  			if (ctType.isTopLevel()) {
33  				unitList.add(new CompilationUnitWrapper(ctType));
34  			}
35  		}
36  		compiler.setCompilationUnits(unitList.toArray(new CompilationUnit[0]));
37  	}
38  }
