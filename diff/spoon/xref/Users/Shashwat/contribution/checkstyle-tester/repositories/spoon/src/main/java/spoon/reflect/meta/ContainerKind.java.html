6   package spoon.reflect.meta;
7   
8   /**
9    * Represents a kind of data, which is used in an field or in a role in the Spoon metamodel.
10   * See also {@link spoon.metamodel.MetamodelProperty}.
11   */
12  public enum ContainerKind {
13  	/**
14  	 * It is a single value field
15  	 * Example: {@link spoon.support.reflect.declaration.CtClassImpl#simpleName}
16  	 */
17  	SINGLE,
18  
19  
20  	/**
21  	 * It is a list of values
22  	 * Example: {@link spoon.support.reflect.declaration.CtClassImpl#typeMembers}
23  	 */
24  	LIST,
25  
26  
27  	/**
28  	 * It is a set of values
29  	 * Example: {@link spoon.support.reflect.declaration.CtPackageImpl#types}
30  	 */
31  	SET,
32  
33  
34  	/**
35  	 * It is a map&lt;String, T&gt; of values
36  	 * Example: {@link spoon.support.reflect.declaration.CtAnnotationImpl#elementValues}
37  	 */
38  	MAP;
39  }
