6   package spoon.processing;
7   
8   /**
9    * This enumeration defines the traversal strategies available for a processor.
10   *
11   * @see spoon.processing.Processor#getTraversalStrategy()
12   */
13  public enum TraversalStrategy {
14  
15  	/**
16  	 * When this strategy is selected, the processor will traverse the parent
17  	 * elements before the children. This implies that if the processor adds new
18  	 * elements to a given model sub-tree, these new elements will be processed
19  	 * too. This strategy can easily lead to complex processing behavior, so it
20  	 * is to be used with care.
21  	 */
22  	PRE_ORDER,
23  
24  	/**
25  	 * When this strategy is selected, the processor will traverse the child
26  	 * elements before the parents. This implies that child elements that are
27  	 * removed by a processor will still be processed before they are removed.
28  	 */
29  	POST_ORDER
30  
31  }
