6   package spoon.support.comparator;
7   
8   import spoon.reflect.declaration.CtElement;
9   import spoon.support.visitor.SignaturePrinter;
10  
11  import java.io.Serializable;
12  import java.util.Comparator;
13  
14  /**
15   * Compares executables (method, executable-references) based on a signature.
16   */
17  public class SignatureComparator implements Comparator<CtElement>, Serializable {
18  
19  	private static final long serialVersionUID = 1L;
20  
21  	@Override
22  	public int compare(CtElement o1, CtElement o2) {
23  		SignaturePrinter signaturePrinter1 = new SignaturePrinter();
24  		SignaturePrinter signaturePrinter2 = new SignaturePrinter();
25  		signaturePrinter1.scan(o1);
26  		signaturePrinter2.scan(o2);
27  		return signaturePrinter1.getSignature().compareTo(signaturePrinter2.getSignature());
28  	}
29  
30  }
