6   package spoon.reflect;
7   
8   import spoon.SpoonModelBuilder;
9   import spoon.processing.Processor;
10  import spoon.reflect.declaration.CtElement;
11  import spoon.reflect.declaration.CtModule;
12  import spoon.reflect.declaration.CtPackage;
13  import spoon.reflect.declaration.CtType;
14  import spoon.reflect.visitor.Filter;
15  import spoon.reflect.visitor.chain.CtQueryable;
16  
17  import java.io.Serializable;
18  import java.util.Collection;
19  import java.util.List;
20  
21  /** represents a Java program, modeled by a set of compile-time (Ct) objects
22   * where each object is a program element (for instance, a CtClass represents a class).
23   */
24  public interface CtModel extends Serializable, CtQueryable {
25  
26  	/** returns the root package of the unnamed module */
27  	CtPackage getRootPackage();
28  
29  	/** returns all top-level types of the model */
30  	Collection<CtType<?>> getAllTypes();
31  
32  	/** returns all packages of the model */
33  	Collection<CtPackage> getAllPackages();
34  
35  	/**
36  	 * Returns the unnamed module.
37  	 */
38  	CtModule getUnnamedModule();
39  
40  	/**
41  	 * returns all modules of the model
42  	 */
43  	Collection<CtModule> getAllModules();
44  
45  	/** process this model with the given processor */
46  	void processWith(Processor<?> processor);
47  
48  	/** Returns all the model elements matching the filter. */
49  	<E extends CtElement> List<E> getElements(Filter<E> filter);
50  
51  	/**
52  	 * If true, the first build model has been finished.
53  	 * This value should be false at the beginning of {@link SpoonModelBuilder#build()} and true at this end.
54  	 */
55  	boolean isBuildModelFinished();
56  
57  	/**
58  	 * Set to true to flag that a build model has been finished.
59  	 * By default, this method is called with a true value at the end of {@link SpoonModelBuilder#build()}
60  	 */
61  	<T extends CtModel> T setBuildModelIsFinished(boolean buildModelFinished);
62  
63  }
