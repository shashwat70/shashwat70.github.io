6   package spoon.reflect.visitor.chain;
7   
8   import spoon.reflect.declaration.CtElement;
9   import spoon.reflect.visitor.Filter;
10  
11  /**
12   * Represents an object on which one can make queries.
13   * It is implemented
14   * <ol>
15   * <li> by {@link CtElement} to allow creation of a new query on
16   * children of an element.
17   * <li> by {@link CtQuery} to allow reusable queries and chaining query steps.
18   * </ol>
19   *
20   * The main methods are documented in CtQuery
21   */
22  public interface CtQueryable {
23  
24  	/**
25  	 * @see CtQuery#filterChildren(Filter)
26  	 */
27  	<R extends CtElement> CtQuery filterChildren(Filter<R> filter);
28  
29  	/**
30  	 * @see CtQuery#map(CtFunction)
31  	 */
32  	<I, R> CtQuery map(CtFunction<I, R> function);
33  
34  	/**
35  	 * @see CtQuery#map(CtConsumableFunction)
36  	 */
37  	<I> CtQuery map(CtConsumableFunction<I> queryStep);
38  
39  }
