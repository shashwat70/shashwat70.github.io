6   package spoon.support.sniper.internal;

8   import java.util.List;
9   
10  import spoon.reflect.declaration.CtElement;
11  import spoon.reflect.visitor.DefaultJavaPrettyPrinter;
12  
13  /**
14   * Handles printing of changes of the unordered Set of elements.
15   * E.g. set of modifiers and annotations or set of super interfaces
16   * Such sets must be printed in same order like they were in origin source code
17   *
18   * If anything is modified (add/remove/modify element) in such collection,
19   * then collection is printed in the order defined by {@link DefaultJavaPrettyPrinter}.
20   * The not modified items of collection are printed using origin sources
21   */
22  public class SourceFragmentContextSet extends AbstractSourceFragmentContextCollection {
23  	/**
24  	 * @param mutableTokenWriter {@link MutableTokenWriter}, which is used for printing
25  	 * @param element the {@link CtElement} whose list attribute is handled
26  	 */
27  	public SourceFragmentContextSet(MutableTokenWriter mutableTokenWriter, CtElement element, List<SourceFragment> fragments, ChangeResolver changeResolver) {
28  		super(mutableTokenWriter, fragments, changeResolver);
29  	}
30  }
