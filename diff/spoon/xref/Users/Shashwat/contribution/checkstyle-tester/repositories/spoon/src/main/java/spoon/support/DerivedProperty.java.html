8   import spoon.reflect.declaration.CtType;

10  import java.lang.annotation.ElementType;
11  import java.lang.annotation.Retention;
12  import java.lang.annotation.RetentionPolicy;
13  import java.lang.annotation.Target;
14  
15  /**
16   * Tells that a metamodel property is derived, ie computed from the value of another property.
17   *
18   * For instance {@link CtType#getFields()}  is derived from {@link CtType#getTypeMembers()}
19   *
20   * It can be put on getter and setters.
21   *
22   * Contracts:
23   * - A setter with @DerivedProperty only triggers one single model intercession event,
24   *   on the element primarily responsible for handling the state from which this dervied property is computed.
25   * - A getter with @DerivedProperty is never called in CtScanner and derived classes (clone, replace)
26   */
27  @Retention(RetentionPolicy.RUNTIME)
28  @Target({ ElementType.METHOD })
29  public @interface DerivedProperty {
30  }
