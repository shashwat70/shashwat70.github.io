6   package spoon.support.modelobs.context;
7   
8   import spoon.reflect.declaration.CtElement;
9   import spoon.reflect.path.CtRole;
10  
11  import java.util.Map;
12  
13  /**
14   * defines the map context
15   * @param <K>
16   * @param <V>
17   */
18  public class MapContext<K, V> extends Context {
19  	private final Map<K, V> map;
20  	private  K key;
21  
22  	public MapContext(CtElement element, CtRole role, Map<K, V> map) {
23  		super(element, role);
24  		this.map = map;
25  	}
26  
27  	public MapContext(CtElement element, CtRole role, Map<K, V> map, K key) {
28  		this(element, role, map);
29  		this.key = key;
30  	}
31  
32  	/**
33  	 * get the changed key
34  	 * @return the changed key
35  	 */
36  	public K getKey() {
37  		return key;
38  	}
39  
40  	/**
41  	 * the changed map
42  	 * @return the changed map
43  	 */
44  	public Map<K, V> getMap() {
45  		return map;
46  	}
47  }
