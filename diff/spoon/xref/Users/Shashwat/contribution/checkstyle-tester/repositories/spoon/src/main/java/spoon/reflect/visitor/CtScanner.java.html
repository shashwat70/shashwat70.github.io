102 /**
103  * This visitor implements a deep-search scan on the model.
104  *
105  * Ensures that all children nodes are visited once, a visit means three method
106  * calls, one call to "enter", one call to "exit" and one call to scan.
107  *
108  * Is used by the processing and filtering engine.
109  */
110 public abstract class CtScanner implements CtVisitor {
111 	/**
112 	 * Default constructor.
113 	 */
114 	public CtScanner() {
115 	}
116 
117 	/**
118 	 * This method is upcalled by the scanner when entering a scanned element.
119 	 * To be overridden to implement specific scanners.
120 	 */
121 	protected void enter(CtElement e) {
122 	}
123 
124 	/**
125 	 * This method is upcalled by the scanner when exiting a scanned element. To
126 	 * be overridden to implement specific scanners.
127 	 */
128 	protected void exit(CtElement e) {
129 	}
130 
131 	/**
132 	 * Generically scans a collection of meta-model elements.
133 	 */
134 	public void scan(CtRole role, Collection<? extends CtElement> elements) {
135 		if (elements != null) {
136 			// we use defensive copy so as to be able to change the class while scanning
137 			// otherwise one gets a ConcurrentModificationException
138 			for (CtElement e : new ArrayList<>(elements)) {
139 				scan(role, e);
140 			}
141 		}
142 	}
143 	/**
144 	 * Generically scans a Map of meta-model elements.
145 	 */
146 	public void scan(CtRole role, Map<String, ? extends CtElement> elements) {
147 		if (elements != null) {
148 			for (CtElement obj : elements.values()) {
149 				scan(role, obj);
150 			}
151 		}
152 	}
153 
154 	/**
155 	 * Generically scans a collection of meta-model elements.
156 	 */
157 	public void scan(Collection<? extends CtElement> elements) {
158 		scan(null, elements);
159 	}
160 
161 	/**
162 	 * Generically scans a meta-model element.
163 	 */
164 	public void scan(CtRole role, CtElement element) {
165 		scan(element);
166 	}
167 
168 	/**
169 	 * Generically scans a meta-model element.
170 	 */
171 	public void scan(CtElement element) {
172 		if (element != null) {
173 			element.accept(this);
174 		}
175 	}

177 	public <A extends Annotation> void visitCtAnnotation(final CtAnnotation<A> annotation) {
178 		enter(annotation);
179 		scan(CtRole.TYPE, annotation.getType());
180 		scan(CtRole.COMMENT, annotation.getComments());
181 		scan(CtRole.ANNOTATION_TYPE, annotation.getAnnotationType());
182 		scan(CtRole.ANNOTATION, annotation.getAnnotations());
183 		scan(CtRole.VALUE, annotation.getValues());
184 		exit(annotation);
185 	}
186 
187 	/**
188 	 * Generically scans an object that can be an element, a reference, or a
189 	 * collection of those.
190 	 */
191 	public void scan(Object o) {
192 		scan(null, o);
193 	}
194 	/**
195 	 * Generically scans an object that can be an element, a reference, or a
196 	 * collection of those.
197 	 */
198 	public void scan(CtRole role, Object o) {
199 		if (o instanceof CtElement) {
200 			scan(role, ((CtElement) (o)));
201 		}
202 		if (o instanceof Collection<?>) {
203 			scan(role, (Collection<? extends CtElement>) o);
204 		}
205 		if (o instanceof Map<?, ?>) {
206 			scan(role, (Map<String, ? extends CtElement>) o);
207 		}
208 	}
