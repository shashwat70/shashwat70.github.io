6   package spoon.reflect.visitor.filter;
7   
8   import java.util.HashSet;
9   import java.util.Set;
10  
11  import spoon.reflect.code.CtLambda;
12  import spoon.reflect.declaration.CtInterface;
13  import spoon.reflect.declaration.CtType;
14  import spoon.reflect.declaration.CtTypeInformation;
15  import spoon.reflect.reference.CtTypeReference;
16  import spoon.reflect.visitor.Filter;
17  
18  /**
19   * This filter matches all the {@link CtLambda} elements, which implements defined interface(s)
20   */
21  public class LambdaFilter implements Filter<CtLambda<?>> {
22  
23  	private Set<String> qualifiedNamesOfInterfaces = new HashSet<>();
24  
25  	/**
26  	 * Use {@link #addImplementingInterface(CtTypeInformation)} to define set of interfaces whose lambdas it is search for
27  	 */
28  	public LambdaFilter() {
29  	}
30  
31  	/**
32  	 * Matches all lambdas implementing `iface`
33  	 * Use {@link #addImplementingInterface(CtTypeInformation)} to define set of interfaces whose lambdas it is search for
34  	 */
35  	public LambdaFilter(CtInterface<?> iface) {
36  		addImplementingInterface(iface);
37  	}
38  
39  	/**
40  	 * Matches all lambdas implementing `iface`
41  	 * Use {@link #addImplementingInterface(CtTypeInformation)} to define set of interfaces whose lambdas it is search for
42  	 */
43  	public LambdaFilter(CtTypeReference<?> iface) {
44  		addImplementingInterface(iface);
45  	}
46  
47  	/**
48  	 * Allows to search for lambdas implemented by different interfaces.
49  	 * @param typeInfo interface whose lambda implementations it is searching for
50  	 */
51  	public LambdaFilter addImplementingInterface(CtTypeInformation typeInfo) {
52  		if (typeInfo instanceof CtType) {
53  			if (typeInfo instanceof CtInterface) {
54  				qualifiedNamesOfInterfaces.add(typeInfo.getQualifiedName());
55  			} //else ignore that request, because lambda can implement only interfaces
56  		} else {
57  			//do not check if it is interface or not. That check needs CtType in model and it might be not available in some modes
58  			//it is OK to search for non interface types. It simply founds no lambda implementing that
59  			qualifiedNamesOfInterfaces.add(typeInfo.getQualifiedName());
60  		}
61  		return this;
62  	}
63  
64  	@Override
65  	public boolean matches(CtLambda<?> lambda) {
66  		return qualifiedNamesOfInterfaces.contains(lambda.getType().getQualifiedName());
67  	}
68  }
