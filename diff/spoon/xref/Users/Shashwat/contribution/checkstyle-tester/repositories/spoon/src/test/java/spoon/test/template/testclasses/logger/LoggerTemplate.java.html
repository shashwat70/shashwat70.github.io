18  package spoon.test.template.testclasses.logger;
19  
20  import spoon.reflect.code.CtBlock;
21  import spoon.template.BlockTemplate;
22  import spoon.template.Local;
23  import spoon.template.Parameter;
24  
25  public class LoggerTemplate extends BlockTemplate {
26  	@Parameter
27  	private String _classname_;
28  	@Parameter
29  	private String _methodName_;
30  	@Parameter
31  	private CtBlock<?> _block_;
32  
33  	@Local
34  	public LoggerTemplate(String _classname_, String _methodName_, CtBlock<?> _block_) {
35  		this._classname_ = _classname_;
36  		this._methodName_ = _methodName_;
37  		this._block_ = _block_;
38  	}
39  
40  	@Override
41  	public void block() throws Throwable {
42  		try {
43  			Logger.enter("_classname_", "_methodName_");
44  			_block_.S();
45  		} finally {
46  			Logger.exit("_methodName_");
47  		}
48  	}
49  }
