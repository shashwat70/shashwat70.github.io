6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.code.CtBlock;
9   import spoon.reflect.code.CtBodyHolder;
10  import spoon.reflect.reference.CtExecutableReference;
11  import spoon.reflect.reference.CtTypeReference;
12  import spoon.support.DerivedProperty;
13  import spoon.reflect.annotations.PropertyGetter;
14  import spoon.reflect.annotations.PropertySetter;
15  
16  import java.util.List;
17  import java.util.Set;
18  
19  import static spoon.reflect.path.CtRole.PARAMETER;
20  import static spoon.reflect.path.CtRole.THROWN;
21  
22  /**
23   * This element represents an executable element such as a method, a
24   * constructor, or an anonymous block.
25   */
26  public interface CtExecutable<R> extends CtNamedElement, CtTypedElement<R>, CtBodyHolder {
27  
28  	/**
29  	 * The separator for a string representation of an executable.
30  	 */
31  	String EXECUTABLE_SEPARATOR = "#";
32  
33  	/*
34  	 * (non-Javadoc)
35  	 *
36  	 * @see spoon.reflect.declaration.CtNamedElement#getReference()
37  	 */
38  	@Override
39  	@DerivedProperty
40  	CtExecutableReference<R> getReference();
41  
42  	/**
43  	 * Gets the body expression.
44  	 */
45  	@Override
46  	CtBlock<R> getBody();
47  
48  	/**
49  	 * Gets the parameters list.
50  	 */
51  	@PropertyGetter(role = PARAMETER)
52  	List<CtParameter<?>> getParameters();
53  
54  	/**
55  	 * Sets the parameters.
56  	 */
57  	@PropertySetter(role = PARAMETER)
58  	<T extends CtExecutable<R>> T setParameters(List<CtParameter<?>> parameters);
59  
60  	/**
61  	 * Add a parameter for this executable
62  	 *
63  	 * @param parameter
64  	 * @return <tt>true</tt> if this element changed as a result of the call
65  	 */
66  	@PropertySetter(role = PARAMETER)
67  	<T extends CtExecutable<R>> T addParameter(CtParameter<?> parameter);
68  
69  	/**
70  	 * Remove a parameter for this executable
71  	 *
72  	 * @param parameter
73  	 * @return <tt>true</tt> if this element changed as a result of the call
74  	 */
75  	boolean removeParameter(CtParameter<?> parameter);
76  
77  	/**
78  	 * Returns the exceptions and other throwables listed in this method or
79  	 * constructor's <tt>throws</tt> clause.
80  	 */
81  	@PropertyGetter(role = THROWN)
82  	Set<CtTypeReference<? extends Throwable>> getThrownTypes();
83  
84  	/**
85  	 * Sets the thrown types.
86  	 */
87  	@PropertySetter(role = THROWN)
88  	<T extends CtExecutable<R>> T setThrownTypes(Set<CtTypeReference<? extends Throwable>> thrownTypes);
89  
90  	/**
91  	 * add a thrown type.
92  	 *
93  	 * @param throwType
94  	 * @return <tt>true</tt> if this element changed as a result of the call
95  	 */
96  	@PropertySetter(role = THROWN)
97  	<T extends CtExecutable<R>> T addThrownType(CtTypeReference<? extends Throwable> throwType);
98  
99  	/**
100 	 * remove a thrown type.
101 	 *
102 	 * @param throwType
103 	 * @return <tt>true</tt> if this element changed as a result of the call
104 	 */
105 	@PropertySetter(role = THROWN)
106 	boolean removeThrownType(CtTypeReference<? extends Throwable> throwType);
107 
108 	/**
109 	 * Gets the signature of this method or constructor.
110 	 * The signature is composed of the method name and the parameter types, all fully-qualified, eg "int foo(java.lang.String)".
111 	 * The core contract is that in a type, there cannot be two methods with the same signature.
112 	 *
113 	 * Note that the concept of method signature in Java is not well defined (see chapter "8.4.2 Method Signature" of the Java specification, which defines what relations between signatures but not what a signature is exactly).
114 	 *
115 	 * Note also that the signature of a method reference is the same as the signature of the corresponding method if and only if the method parameters does not involve generics in their types. Otherwise, one has eg m(String) (reference) and m(T) (declaration)
116 	 *
117 	 * Reference: "In the Java programming language, a method signature is the method name and the number and type of its parameters. Return types and thrown exceptions are not considered to be a part of the method signature."
118 	 * see https://stackoverflow.com/questions/16149285/does-a-methods-signature-in-java-include-its-return-type
119 	 * see https://en.wikipedia.org/wiki/Type_signature
120 	 */
121 	String getSignature();
122 
123 	@Override
124 	CtExecutable<R> clone();
125 }
