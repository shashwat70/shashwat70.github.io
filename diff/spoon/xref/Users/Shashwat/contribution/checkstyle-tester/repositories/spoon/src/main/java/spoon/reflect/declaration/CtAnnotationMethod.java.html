6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.code.CtBodyHolder;
9   import spoon.reflect.code.CtExpression;
10  import spoon.reflect.code.CtStatement;
11  import spoon.reflect.reference.CtTypeReference;
12  import spoon.reflect.annotations.PropertyGetter;
13  import spoon.reflect.annotations.PropertySetter;
14  import spoon.support.UnsettableProperty;
15  
16  import java.util.List;
17  import java.util.Set;
18  
19  import static spoon.reflect.path.CtRole.DEFAULT_EXPRESSION;
20  
21  /**
22   * This element defines an annotation method declared in an annotation type.
23   */
24  public interface CtAnnotationMethod<T> extends CtMethod<T> {
25  	/**
26  	 * Gets the default expression assigned to the annotation method.
27  	 */
28  	@PropertyGetter(role = DEFAULT_EXPRESSION)
29  	CtExpression<T> getDefaultExpression();
30  
31  	/**
32  	 * Sets the default expression assigned to the annotation method.
33  	 */
34  	@PropertySetter(role = DEFAULT_EXPRESSION)
35  	<C extends CtAnnotationMethod<T>> C setDefaultExpression(CtExpression<T> assignedExpression);
36  
37  	@Override
38  	CtAnnotationMethod<T> clone();
39  
40  	@Override
41  	@UnsettableProperty
42  	<T1 extends CtBodyHolder> T1 setBody(CtStatement body);
43  
44  	@Override
45  	@UnsettableProperty
46  	<T1 extends CtExecutable<T>> T1 setThrownTypes(Set<CtTypeReference<? extends Throwable>> thrownTypes);
47  
48  	@Override
49  	@UnsettableProperty
50  	<T extends CtFormalTypeDeclarer> T setFormalCtTypeParameters(List<CtTypeParameter> formalTypeParameters);
51  
52  	@Override
53  	@UnsettableProperty
54  	<T1 extends CtExecutable<T>> T1 setParameters(List<CtParameter<?>> parameters);
55  }
