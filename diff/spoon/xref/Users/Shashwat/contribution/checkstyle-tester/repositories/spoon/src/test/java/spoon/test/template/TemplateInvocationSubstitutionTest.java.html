17  package spoon.test.template;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertFalse;
21  
22  import org.junit.Test;
23  
24  import spoon.Launcher;
25  import spoon.reflect.code.CtBlock;
26  import spoon.reflect.code.CtStatement;
27  import spoon.reflect.declaration.CtClass;
28  import spoon.reflect.factory.Factory;
29  import spoon.support.compiler.FileSystemFile;
30  import spoon.test.template.testclasses.InvocationSubstitutionByExpressionTemplate;
31  import spoon.test.template.testclasses.InvocationSubstitutionByStatementTemplate;
32  import spoon.test.template.testclasses.SubstitutionByExpressionTemplate;
33  
34  public class TemplateInvocationSubstitutionTest {
35  
36  	@Test
37  	public void testInvocationSubstitutionByStatement() {
38  		//contract: the template engine supports substitution of any method invocation
39  		Launcher spoon = new Launcher();
40  		spoon.addTemplateResource(new FileSystemFile("./src/test/java/spoon/test/template/testclasses/InvocationSubstitutionByStatementTemplate.java"));
41  
42  		spoon.buildModel();
43  		Factory factory = spoon.getFactory();
44  
45  		CtBlock<?> templateArg = factory.Class().get(InvocationSubstitutionByStatementTemplate.class).getMethod("sample").getBody();
46  		
47  		CtClass<?> resultKlass = factory.Class().create("Result");
48  		CtStatement result = new InvocationSubstitutionByStatementTemplate(templateArg).apply(resultKlass);
49  		assertEquals("throw new java.lang.RuntimeException(\"Failed\")", result.toString());
50  	}
51  
52  	@Test
53  	public void testInvocationSubstitutionByExpression() {
54  		//contract: the template engine supports substitution of any method invocation
55  		Launcher spoon = new Launcher();
56  		spoon.addTemplateResource(new FileSystemFile("./src/test/java/spoon/test/template/testclasses/InvocationSubstitutionByExpressionTemplate.java"));
57  
58  		spoon.buildModel();
59  		Factory factory = spoon.getFactory();
60  
61  		CtClass<?> resultKlass = factory.Class().create("Result");
62  		CtBlock<?> result = new InvocationSubstitutionByExpressionTemplate(factory.createLiteral("abc")).apply(resultKlass);
63  
64  		assertEquals("java.lang.System.out.println(\"abc\".substring(1))", result.getStatement(0).toString());
65  		assertEquals("java.lang.System.out.println(\"abc\".substring(1))", result.getStatement(1).toString());
66  
67  		// contract: the result of the template has no parent, and can be put anywhere in an AST
68  		assertFalse(result.isParentInitialized());
69  	}
70  
71  	@Test
72  	public void testSubstitutionByExpression() {
73  		//contract: the template engine understands fields whose type extends from TemplateParameter as template parameter automatically. No need for extra annotation
74  		Launcher spoon = new Launcher();
75  		spoon.addTemplateResource(new FileSystemFile("./src/test/java/spoon/test/template/testclasses/SubstitutionByExpressionTemplate.java"));
76  
77  		spoon.buildModel();
78  		Factory factory = spoon.getFactory();
79  
80  		CtClass<?> resultKlass = factory.Class().create("Result");
81  		CtBlock<?> result = new SubstitutionByExpressionTemplate(factory.createLiteral("abc")).apply(resultKlass);
82  		assertEquals("java.lang.System.out.println(\"abc\".substring(1))", result.getStatement(0).toString());
83  	}
