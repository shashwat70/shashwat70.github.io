17  package spoon.test.constructorcallnewclass;
18  
19  import org.junit.Before;
20  import org.junit.Test;
21  import spoon.Launcher;
22  import spoon.reflect.code.CtConstructorCall;
23  import spoon.reflect.declaration.CtClass;
24  import spoon.reflect.declaration.CtType;
25  import spoon.reflect.factory.Factory;
26  import spoon.reflect.reference.CtArrayTypeReference;
27  import spoon.reflect.reference.CtTypeReference;
28  import spoon.reflect.visitor.filter.AbstractFilter;
29  import spoon.reflect.visitor.filter.TypeFilter;
30  import spoon.support.comparator.DeepRepresentationComparator;
31  import spoon.test.constructorcallnewclass.testclasses.Foo;
32  import spoon.test.constructorcallnewclass.testclasses.Panini;
33  
34  import java.util.ArrayList;
35  import java.util.List;
36  import java.util.TreeSet;
37  
38  import static org.junit.Assert.assertEquals;
39  import static org.junit.Assert.assertSame;
40  import static org.junit.Assert.assertTrue;
41  
42  public class ConstructorCallTest {
43  	private List<CtConstructorCall<?>> constructorCalls;
44  	private List<CtConstructorCall<?>> constructorCallsPanini;
45  
46  	@Before
47  	public void setUp() {
48  		final Launcher launcher = new Launcher();
49  		launcher.addInputResource("./src/test/java/" + Foo.class.getCanonicalName().replace(".", "/") + ".java");
50  		launcher.addInputResource("./src/test/java/" + Panini.class.getCanonicalName().replace(".", "/") + ".java");
51  		launcher.setSourceOutputDirectory("./target/spooned");
52  		launcher.run();
53  		final Factory factory = launcher.getFactory();
54  		final CtClass<?> foo = (CtClass<?>) factory.Type().get(Foo.class);
55  		TreeSet ts = new TreeSet(new DeepRepresentationComparator());
56  		ts.addAll(foo.getElements(new AbstractFilter<CtConstructorCall<?>>(CtConstructorCall.class) {
57  			@Override
58  			public boolean matches(CtConstructorCall<?> element) {
59  				return true;
60  			}
61  		}));
62  		constructorCalls = new ArrayList(ts);
63  		final CtType<Panini> panini = factory.Type().get(Panini.class);
64  		constructorCallsPanini = panini.getElements(new TypeFilter<>(CtConstructorCall.class));
65  	}
66  
67  	@Test
68  	public void testConstructorCallStringWithoutParameters() {
69  		final CtConstructorCall<?> constructorCall = constructorCalls.get(2);
70  		assertConstructorCallWithType(String.class, constructorCall);
71  		assertIsConstructor(constructorCall);
72  		assertHasParameters(0, constructorCall);
73  	}
74  
75  	@Test
76  	public void testConstructorCallStringWithParameters() {
77  		final CtConstructorCall<?> constructorCall = constructorCalls.get(1);
78  		assertConstructorCallWithType(String.class, constructorCall);
79  		assertIsConstructor(constructorCall);
80  		assertHasParameters(1, constructorCall);
81  	}
82  
83  	@Test
84  	public void testConstructorCallObjectWithoutParameters() {
85  		final CtConstructorCall<?> constructorCall = constructorCalls.get(3);
86  		assertConstructorCallWithType(Foo.class, constructorCall);
87  		assertIsConstructor(constructorCall);
88  		assertHasParameters(0, constructorCall);
89  	}
90  
91  	@Test
92  	public void testConstructorCallObjectWithParameters() {
93  		final CtConstructorCall<?> constructorCall = constructorCalls.get(4);
94  		assertConstructorCallWithType(Foo.class, constructorCall);
95  		assertIsConstructor(constructorCall);
96  		assertHasParameters(1, constructorCall);
97  	}
98  
99  	@Test
100 	public void testConstructorCallWithGenericArray() {
101 		final CtConstructorCall<?> ctConstructorCall = constructorCallsPanini.get(0);
102 
103 		assertEquals(1, ctConstructorCall.getType().getActualTypeArguments().size());
104 		final CtTypeReference<?> implicitArray = ctConstructorCall.getType().getActualTypeArguments().get(0);
105 		assertTrue(implicitArray.isImplicit());
106 		final CtArrayTypeReference implicitArrayTyped = (CtArrayTypeReference) implicitArray;
107 		assertEquals("", implicitArrayTyped.toString());
108 		assertEquals("AtomicLong[]", implicitArrayTyped.getSimpleName());
109 		assertTrue(implicitArrayTyped.getComponentType().isImplicit());
110 		assertEquals("", implicitArrayTyped.getComponentType().prettyprint());
111 		assertEquals("AtomicLong", implicitArrayTyped.getComponentType().getSimpleName());
112 	}
113 
114 	private void assertHasParameters(int sizeExpected, CtConstructorCall<?> constructorCall) {
115 		if (sizeExpected == 0) {
116 			assertEquals("Constructor call without parameter", sizeExpected, constructorCall.getArguments().size());
117 		} else {
118 			assertEquals("Constructor call with parameters", sizeExpected, constructorCall.getArguments().size());
119 		}
120 	}
121 
122 	private void assertIsConstructor(CtConstructorCall<?> constructorCall) {
123 		assertTrue("Method must be a constructor", constructorCall.getExecutable().isConstructor());
124 	}
125 
126 	private void assertConstructorCallWithType(Class<?> typeExpected, CtConstructorCall<?> constructorCall) {
127 		assertSame("Constructor call is typed by the class of the constructor", typeExpected, constructorCall.getType().getActualClass());
128 	}
129 
130 	@Test
131 	public void testCoreConstructorCall() {
132 		Launcher spoon = new Launcher();
133 		// the minimum is setType()
134 		CtConstructorCall call = spoon.getFactory().Core().createConstructorCall();
135 		call.setType(spoon.getFactory().Core().createTypeReference().setSimpleName("Foo"));
136 		assertEquals("new Foo()", call.toString());
137 
138 		// now with Code factory
139 		CtConstructorCall call2 = spoon.getFactory().Code().createConstructorCall(spoon.getFactory().Core().createTypeReference().setSimpleName("Bar"));
140 		assertEquals("new Bar()", call2.toString());
141 	}
