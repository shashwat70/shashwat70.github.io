84  	class SimpleSignature extends CtScanner {
85  		String signature = "";
86  		@Override
87  		public <T> void visitCtParameter(CtParameter<T> parameter) {
88  			signature += parameter.getType().getQualifiedName() + ", ";
89  			super.visitCtParameter(parameter);
90  		}
91  
92  		@Override
93  		public <T> void visitCtMethod(CtMethod<T> m) {
94  			signature += m.getSimpleName() + "(";
95  			super.visitCtMethod(m);
96  			signature += ")";
97  		}
98  	}

101 	class SimpleSignatureComparator implements Comparator<CtMethod<?>> {
102 		@Override
103 		public int compare(CtMethod<?> o1, CtMethod<?> o2) {
104 			return computeSimpleSignature(o1).compareTo(computeSimpleSignature(o2));
105 		}
106 	}
107 
108 	private String computeSimpleSignature(CtMethod<?> m) {
109 		SimpleSignature sc1 = new SimpleSignature();
110 		sc1.visitCtMethod(m);
111 		return sc1.signature;
112 	}

114 	@Test
115 	public void testScannerCallsAllProperties() {
116 		// contract: CtScanner must visit all metamodel properties and use correct CtRole!
117 		final Launcher launcher = new Launcher();
118 		launcher.addInputResource("./src/main/java/spoon/reflect/");
119 		launcher.run();
120 
121 		CtTypeReference<?> ctElementRef = launcher.getFactory().createCtTypeReference(CtElement.class);
122 		CtTypeReference<?> ctRefRef = launcher.getFactory().createCtTypeReference(CtReference.class);
123 
124 		CtClass<?> scannerCtClass = (CtClass<?>) launcher.getFactory().Type().get(CtScanner.class);
125 
126 		List<String> problems = new ArrayList<>();
127 		Set<String> ignoredInvocations = new HashSet(Arrays.asList("scan", "enter", "exit"));
128 
129 		Metamodel metaModel = Metamodel.getInstance();
130 
131 		//collect all scanner visit methods, to check if all were checked
132 		Map<String, CtMethod<?>> scannerVisitMethodsByName = new HashMap<>();
133 		scannerCtClass.getAllMethods().forEach(m -> {
134 			if (m.getSimpleName().startsWith("visit")) {
135 				scannerVisitMethodsByName.put(m.getSimpleName(), m);
136 			}
137 		});
138 
139 		class Counter  { int nbChecks = 0; }
140 		Counter c = new Counter();
141 		for (MetamodelConcept leafConcept : metaModel.getConcepts()) {
142 
143 			// we only consider leaf, actual classes of the metamodel (eg CtInvocation) and not abstract ones (eg CtModifiable)
144 			if (leafConcept.getKind() != ConceptKind.LEAF) {
145 				continue;
146 			}
147 
148 			CtMethod<?> visitMethod = scannerVisitMethodsByName.remove("visit" + leafConcept.getName());
149 			assertNotNull("CtScanner#" + "visit" + leafConcept.getName() + "(...) not found", visitMethod);
150 			Set<String> calledMethods = new HashSet<>();
151 			Set<String> checkedMethods = new HashSet<>();
152 
153 			// go over the roles and the corresponding fields of this type
154 			leafConcept.getRoleToProperty().forEach((role, mmField) -> {
155 
156 				if (mmField.isDerived() || mmField.isUnsettable()) {
157 					//ignore derived or unsettable fields
158 					return; // return of the lambda
159 				}
160 
161 				// ignore fields, which doesn't return CtElement
162 				if (mmField.getTypeofItems().isSubtypeOf(ctElementRef) == false) {
163 					return; // return of the lambda
164 				}
165 
166 				MMMethod getter = mmField.getMethod(MMMethodKind.GET);
167 				checkedMethods.add(getter.getSignature());
168 				//System.out.println("checking "+m.getSignature() +" in "+visitMethod.getSignature());
169 
170 				// now, we collect at least one invocation to this getter in the visit method
171 				CtInvocation invocation = visitMethod.filterChildren(new TypeFilter<CtInvocation>(CtInvocation.class) {
172 					@Override
173 					public boolean matches(CtInvocation element) {
174 						if (ignoredInvocations.contains(element.getExecutable().getSimpleName())) {
175 							return false;
176 						}
177 						calledMethods.add(element.getExecutable().getSignature());
178 						return super.matches(element) && element.getExecutable().getSimpleName().equals(getter.getName());
179 					}
180 				}).first();
181 
182 				if ("getComments".equals(getter.getName()) && leafConcept.getMetamodelInterface().isSubtypeOf(ctRefRef)) {
183 					//one cannot set comments on references see the @UnsettableProperty of CtReference#setComments
184 					return;
185 				}
186 
187 				// contract: there ia at least one invocation to all non-derived, role-based getters in the visit method of the Scanner
188 				if (invocation == null) {
189 					problems.add("no " + getter.getSignature() + " in " + visitMethod);
190 				} else {
191 					c.nbChecks++;
192 					//System.out.println(invocation.toString());
193 
194 					// contract: the scan method is called with the same role as the one set on field / property
195 					CtRole expectedRole = metaModel.getRoleOfMethod((CtMethod<?>) invocation.getExecutable().getDeclaration());
196 					CtInvocation<?> scanInvocation = invocation.getParent(CtInvocation.class);
197 					String realRoleName = ((CtFieldRead<?>) scanInvocation.getArguments().get(0)).getVariable().getSimpleName();
198 					if (expectedRole.name().equals(realRoleName) == false) {
199 						problems.add("Wrong role " + realRoleName + " used in " + scanInvocation.getPosition());
200 					}
201 				}
202 
203 			});
204 			calledMethods.removeAll(checkedMethods);
205 
206 			// contract: CtScanner only calls methods that have a role and the associated getter
207 			if (!calledMethods.isEmpty()) {
208 				problems.add("CtScanner " + visitMethod.getPosition() + " calls unexpected methods: " + calledMethods);
209 			}
210 		}
211 
212 		// contract: all visit* methods in CtScanner have been checked
213 		if (scannerVisitMethodsByName.isEmpty() == false) {
214 			problems.add("These CtScanner visit methods were not checked: " + scannerVisitMethodsByName.keySet());
215 		}
216 		if (!problems.isEmpty()) {
217 			fail(String.join("\n", problems));
218 		}
219 		assertTrue("not enough checks " + c.nbChecks, c.nbChecks >= 200);
220 	}

222 	@Test
223 	public void testScan() {
224 		// contract: all AST nodes are visited through method "scan"
225 		Launcher launcher;
226 		launcher = new Launcher();
227 		launcher.getEnvironment().setNoClasspath(true);
228 		launcher.addInputResource("src/test/resources/noclasspath/draw2d");
229 		launcher.buildModel();
230 		class Counter {
231 			int nEnter = 0;
232 			int nExit = 0;
233 			int nObject = 0;
234 			int nElement = 0;
235 			Deque<CollectionContext> contexts = new ArrayDeque<>();
236 		}
237 		Counter counter = new Counter();
238 		launcher.getModel().getRootPackage().accept(new CtScanner() {
239 			@Override
240 			public void scan(Object o) {
241 				counter.nObject++;
242 				super.scan(o);
243 			}
244 			@Override
245 			public void scan(CtElement o) {
246 				counter.nElement++;
247 				super.scan(o);
248 			}
249 			@Override
250 			public void enter(CtElement o) {
251 				counter.nEnter++;
252 				super.enter(o);
253 			}
254 			@Override
255 			public void exit(CtElement o) {
256 				counter.nExit++;
257 				super.exit(o);
258 			}
259 		});
260 		
261 		//top comment of file belongs to compilation unit which is not visited by standard scanning
262 		//so count comments of these compilation units
263 		int countOfCommentsInCompilationUnits = 0;
264 		for (CompilationUnit cu : launcher.getFactory().CompilationUnit().getMap().values()) {
265 			countOfCommentsInCompilationUnits += cu.getComments().size();
266 		}
267 
268 		// interesting, this is never called because of covariance, only CtElement or Collection is called
269 		assertEquals(0, counter.nObject);
270 		// this is a coarse-grain check to see if the scanner changes
271 		// no more exec ref in paramref
272 		// also takes into account the comments
273 		assertEquals(3655, counter.nElement + countOfCommentsInCompilationUnits);
274 		assertEquals(2435, counter.nEnter + countOfCommentsInCompilationUnits);
275 		assertEquals(2435, counter.nExit + countOfCommentsInCompilationUnits);
276 
277 		// contract: all AST nodes which are part of Collection or Map are visited first by method "scan(Collection|Map)" and then by method "scan(CtElement)"
278 		Counter counter2 = new Counter();
279 		launcher.getModel().getRootPackage().accept(new CtScanner() {
280 			@Override
281 			public void scan(Object o) {
282 				counter2.nObject++;
283 				super.scan(o);
284 			}
285 			@Override
286 			public void scan(CtRole role, CtElement o) {
287 				if (o == null) {
288 					//there is no collection involved in scanning of this single value NULL attribute
289 					assertNull(counter2.contexts.peek().col);
290 
291 				} else {
292 					RoleHandler rh = RoleHandlerHelper.getRoleHandler(o.getParent().getClass(), role);
293 					if (rh.getContainerKind() == ContainerKind.SINGLE) {
294 						//there is no collection involved in scanning of this single value attribute
295 						assertNull(counter2.contexts.peek().col);
296 					} else {
297 						counter2.contexts.peek().assertRemoveSame(o);
298 					}
299 				}
300 				counter2.nElement++;
301 				super.scan(o);
302 			}
303 			@Override
304 			public void scan(CtRole role, Collection<? extends CtElement> elements) {
305 				//contract: before processed collection is finished before it starts with next collection
306 				counter2.contexts.peek().initCollection(elements);
307 				super.scan(role, elements);
308 				//contract: all elements of collection are processed in previous super.scan call
309 				counter2.contexts.peek().assertCollectionIsEmpty();
310 			}
311 			@Override
312 			public void scan(CtRole role, Map<String, ? extends CtElement> elements) {
313 				//contract: before processed collection is finished before it starts with next collection
314 				counter2.contexts.peek().initCollection(elements.values());
315 				super.scan(role, elements);
316 				//contract: all elements of collection are processed in previous super.scan call
317 				counter2.contexts.peek().assertCollectionIsEmpty();
318 			}
319 			@Override
320 			public void enter(CtElement o) {
321 				counter2.nEnter++;
322 				counter2.contexts.push(new CollectionContext());
323 			}
324 			@Override
325 			public void exit(CtElement o) {
326 				counter2.nExit++;
327 				counter2.contexts.peek().assertCollectionIsEmpty();
328 				counter2.contexts.pop();
329 			}
330 		});
331 		assertEquals(counter.nObject, counter2.nObject);
332 		assertEquals(counter.nElement, counter2.nElement);
333 		assertEquals(counter.nEnter, counter2.nEnter);
334 		assertEquals(counter.nExit, counter2.nExit);
335 	}
336 	private static class CollectionContext {
337 		Collection<CtElement> col;
338 		void assertCollectionIsEmpty() {
339 			assertTrue(col == null || col.isEmpty());
340 			col = null;
341 		}
342 		public void initCollection(Collection<? extends CtElement> elements) {
343 			assertCollectionIsEmpty();
344 			col = new ArrayList<>(elements);
345 			assertFalse(col.contains(null));
346 		}
347 		public void assertRemoveSame(CtElement o) {
348 			assertNotNull(col);
349 			for (Iterator iter = col.iterator(); iter.hasNext();) {
350 				CtElement ctElement = (CtElement) iter.next();
351 				if (o == ctElement) {
352 					iter.remove();
353 					return;
354 				}
355 			}
356 			fail();
357 		}
