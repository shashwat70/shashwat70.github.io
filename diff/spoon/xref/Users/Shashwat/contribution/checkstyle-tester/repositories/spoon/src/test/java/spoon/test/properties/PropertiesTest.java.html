17  package spoon.test.properties;
18  
19  import org.junit.Test;
20  import spoon.Launcher;
21  import spoon.SpoonModelBuilder;
22  import spoon.compiler.SpoonResourceHelper;
23  import spoon.reflect.factory.Factory;
24  
25  import java.io.File;
26  import java.util.Arrays;
27  
28  import static org.junit.Assert.assertEquals;
29  
30  public class PropertiesTest {
31  
32  	@Test
33  	public void testNonExistingDirectory() throws Exception {
34  		File tempFile = File.createTempFile("SPOON", "SPOON");
35  		tempFile.delete();
36  
37  		Launcher spoon = new Launcher();
38  		Factory factory = spoon.createFactory();
39  		SpoonModelBuilder compiler = spoon.createCompiler(
40  				factory,
41  				SpoonResourceHelper
42  						.resources(
43  								"./src/test/java/spoon/test/properties/testclasses/Sample.java"
44  						));
45  		compiler.build();
46  
47  		compiler.instantiateAndProcess(Arrays.asList(SimpleProcessor.class.getName()));
48  		assertEquals(0, factory.getEnvironment().getErrorCount());
49  	}
50  
51  }
