49  	@Test
50  	public void test() {
51  		// do we still have a correct model when the complete classpath is not given as input?
52  		Launcher spoon = new Launcher();
53  		spoon.getEnvironment().setNoClasspath(true);
54  		spoon.getEnvironment().setLevel("OFF");
55  		spoon.getEnvironment().setCommentEnabled(false); // avoid getting the comments for the equals
56  		spoon.addInputResource("./src/test/resources/spoon/test/noclasspath/fields");
57  		spoon.getEnvironment().setSourceOutputDirectory(new File("target/spooned/apitest"));
58  		spoon.run();
59  		Factory factory = spoon.getFactory();
60  		CtClass<Object> clazz = factory.Class().get("Foo");
61  
62  		assertEquals("Foo", clazz.getSimpleName());
63  		CtTypeReference<?> superclass = clazz.getSuperclass();
64  		// "Unknown" is not in the classpath at all
65  		assertEquals("Unknown", superclass.getSimpleName());
66  		try {
67  			superclass.getActualClass();
68  			fail();
69  		} catch (SpoonClassNotFoundException e) {
70  			// expected
71  		}
72  		assertNull(superclass.getDeclaration());
73  
74  		// should be empty as in noClasspath the actual class cannot be retrieved
75  		assertTrue(superclass.getAllFields().isEmpty());
76  
77  		// now we really make sure we don't have the class in the classpath
78  		try {
79  			superclass.getActualClass();
80  			fail();
81  		} catch (SpoonClassNotFoundException e) {
82  			// expected
83  		}
84  
85  		{
86  			CtMethod<?> method = clazz.getMethod("method", new CtTypeReference[0]);
87  			assertNotNull(method);
88  			List<CtInvocation<?>> invocations = method.getElements(new TypeFilter<>(CtInvocation.class));
89  			assertEquals(1, invocations.size());
90  			CtInvocation<?> c = invocations.get(0);
91  			assertEquals("method", c.getExecutable().getSimpleName());
92  			assertEquals("x.method()", method.getBody().getStatement(1).toString());
93  		}
94  
95  		{
96  			CtMethod<?> method = clazz.getMethod("m2", new CtTypeReference[0]);
97  			assertNotNull(method);
98  			List<CtInvocation<?>> invocations = method.getElements(new TypeFilter<>(CtInvocation.class));
99  			assertEquals(3, invocations.size());
100 			CtInvocation<?> c = invocations.get(1);
101 			assertEquals("second", c.getExecutable().getSimpleName());
102 			assertEquals("x.first().second().third()", method.getBody().getStatement(1).toString());
103 		}
104 
105 		{
106 			CtMethod<?> method = clazz.getMethod("m1", new CtTypeReference[0]);
107 			assertNotNull(method);
108 			List<CtInvocation<?>> invocations = method.getElements(new TypeFilter<>(CtInvocation.class));
109 			assertEquals(1, invocations.size());
110 			invocations.get(0);
111 			assertEquals("x.y.z.method()", method.getBody().getStatement(0).toString());
112 		}
113 
114 		{
115 			CtMethod<?> method = clazz.getMethod("m3", new CtTypeReference[0]);
116 			assertNotNull(method);
117 			List<CtInvocation<?>> invocations = method.getElements(new TypeFilter<>(CtInvocation.class));
118 			assertEquals(1, invocations.size());
119 			invocations.get(0);
120 			CtLocalVariable<?> statement = method.getBody().getStatement(0);
121 			CtFieldAccess<?>  fa = (CtFieldAccess<?>) statement.getDefaultExpression();
122 			assertTrue(fa.getTarget() instanceof CtInvocation);
123 			assertEquals("field", fa.getVariable().getSimpleName());
124 			assertEquals("int x = first().field", statement.toString());
125 		}
126 	}

171 	@Test
172 	public void testInheritanceInNoClassPathWithClasses() {
173 		// contract: when using noclasspath in combination with a source classpath
174 		// spoon is able to resolve the inheritance between classes contained in source cp
175 		String sourceInputDirPath = "./src/test/resources/spoon/test/inheritance";
176 		String targetBinPath = "./target/spoon-nocp-bin";
177 
178 		Launcher spoon = new Launcher();
179 		spoon.getEnvironment().setShouldCompile(true);
180 		spoon.addInputResource(sourceInputDirPath);
181 		spoon.setBinaryOutputDirectory(targetBinPath);
182 		spoon.run();
183 
184 		spoon = new Launcher();
185 		spoon.getEnvironment().setNoClasspath(true);
186 		spoon.getEnvironment().setSourceClasspath(new String[] { targetBinPath });
187 		spoon.addInputResource(sourceInputDirPath + "/AnotherClass.java");
188 		spoon.buildModel();
189 
190 		CtType anotherclass = spoon.getFactory().Type().get("org.acme.AnotherClass");
191 		assertEquals(1, anotherclass.getFields().size());
192 
193 		CtField field = (CtField) anotherclass.getFields().get(0);
194 
195 		CtTypeReference myClassReference = spoon.getFactory().Type().createReference("fr.acme.MyClass");
196 		assertEquals(myClassReference, field.getType());
197 		assertNotNull(myClassReference.getActualClass());
198 
199 		CtTypeReference myInterfaceReference = spoon.getFactory().Type().createReference("org.myorganization.MyInterface");
200 		assertTrue(myClassReference.isSubtypeOf(myInterfaceReference));
201 		assertTrue(field.getType().isSubtypeOf(myInterfaceReference));
202 	}
