17  package spoon.support.compiler.jdt;
18  
19  import static org.junit.Assert.assertEquals;
20  
21  import org.eclipse.jdt.internal.compiler.ast.CompilationUnitDeclaration;
22  import org.eclipse.jdt.internal.compiler.env.INameEnvironment;
23  import org.eclipse.jdt.internal.compiler.impl.CompilerOptions;
24  import org.junit.Test;
25  
26  import spoon.Launcher;
27  import spoon.SpoonModelBuilder;
28  import spoon.compiler.SpoonResourceHelper;
29  import spoon.reflect.code.CtExpression;
30  import spoon.reflect.declaration.CtClass;
31  import spoon.reflect.declaration.CtField;
32  import spoon.support.compiler.jdt.testclasses.ExtendedStringLiteralTestClass;
33  
34  public class ExtendedStringLiteralTest {
35  
36  	@Test
37  	public void testExtendedStringLiteral() throws Exception {
38  		Launcher launcher = new Launcher() {
39  			@Override
40  			public SpoonModelBuilder createCompiler() {
41  				return new JDTBasedSpoonCompiler(getFactory()) {
42  					@Override
43  					protected JDTBatchCompiler createBatchCompiler() {
44  						return new JDTBatchCompiler(this) {
45  							@Override
46  							public CompilationUnitDeclaration[] getUnits() {
47  								startTime = System.currentTimeMillis();
48  								INameEnvironment environment = this.jdtCompiler.environment;
49  								if (environment == null) {
50  									environment = getLibraryAccess();
51  								}
52  								CompilerOptions compilerOptions = new CompilerOptions(this.options);
53  
54  								// set to true to force executing visit(ExtendedStringLiteral,BlockScope)
55  								// which is not the case when set to false which is the default in JDTBatchCompiler
56  								// the test not succeeds since this was not the case before the pull request
57  								// and since visit(ExtendedStringLiteral,BlockScope) was throwing a RuntimeException
58  								compilerOptions.parseLiteralExpressionsAsConstants = true;
59  
60  								TreeBuilderCompiler treeBuilderCompiler = new TreeBuilderCompiler(
61  										environment, getHandlingPolicy(), compilerOptions,
62  										this.jdtCompiler.requestor, getProblemFactory(), this.out,
63  										null);
64  								return treeBuilderCompiler.buildUnits(getCompilationUnits());
65  							}
66  						};
67  					}
68  				};
69  			}
70  		};
71  		SpoonModelBuilder comp = launcher.createCompiler();
72  		comp.addInputSources(SpoonResourceHelper.resources(
73  				"./src/test/java/" + ExtendedStringLiteralTestClass.class.getCanonicalName().replace('.', '/') + ".java"));
74  		comp.build();
75  
76  		CtClass<?> cl =
77  			comp.getFactory().Package().get("spoon.support.compiler.jdt.testclasses").
78  			getType("ExtendedStringLiteralTestClass");
79  		CtField<?> f = cl.getField("extendedStringLiteral");
80  		CtExpression<?> de = f.getDefaultExpression();
81  		assertEquals("\"hello world!\"", de.toString());
82  	}
83  }
