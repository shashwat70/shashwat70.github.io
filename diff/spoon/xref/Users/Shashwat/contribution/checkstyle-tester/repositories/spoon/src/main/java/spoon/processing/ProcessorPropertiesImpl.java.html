6   package spoon.processing;
7   
8   import org.apache.commons.lang3.ClassUtils;
9   
10  import java.util.HashMap;
11  import java.util.Map;
12  
13  public class ProcessorPropertiesImpl implements ProcessorProperties {
14  	private final Map<String, Object> _properties = new HashMap<>();
15  
16  	@Override
17  	public <T> T get(Class<T> type, String name) {
18  		if (type.isPrimitive()) {
19  			type = (Class<T>) ClassUtils.primitiveToWrapper(type);
20  		}
21  		T result = (T) _properties.get(name);
22  		if (result == null) {
23  			return null;
24  		} else {
25  			return (type.isAssignableFrom(result.getClass())) ? result : null;
26  		}
27  	}
28  
29  	@Override
30  	public void set(String name, Object o) {
31  		_properties.put(name, o);
32  	}
33  
34  	/**
35  	 * Gets the corresponding processor name.
36  	 */
37  	@Override
38  	public String getProcessorName() {
39  		return (String) _properties.get("__NAME__");
40  	}
41  
42  }
