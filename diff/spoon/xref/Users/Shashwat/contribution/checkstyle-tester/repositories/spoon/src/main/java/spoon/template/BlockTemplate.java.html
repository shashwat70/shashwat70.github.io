6   package spoon.template;
7   
8   import spoon.reflect.code.CtBlock;
9   import spoon.reflect.declaration.CtClass;
10  import spoon.reflect.declaration.CtType;
11  
12  /**
13   * This class represents a template parameter that defines a void block
14   * statement directly expressed in Java (no returns).
15   *
16   *
17   * <p>
18   * To define a new block template parameter, you must subclass this class and
19   * implement the {@link #block()} method, which actually defines the Java block.
20   * It corresponds to a {@link spoon.reflect.code.CtBlock}.
21   */
22  public abstract class BlockTemplate extends AbstractTemplate<CtBlock<?>> {
23  
24  	/**
25  	 * Returns the block.
26  	 */
27  	public static CtBlock<?> getBlock(CtClass<? extends BlockTemplate> p) {
28  		return p.getMethod("block").getBody();
29  	}
30  
31  	/**
32  	 * Creates a new block template parameter.
33  	 */
34  	public BlockTemplate() {
35  	}
36  
37  	@Override
38  	public CtBlock<?> apply(CtType<?> targetType) {
39  		CtClass<? extends BlockTemplate> c = Substitution.getTemplateCtClass(targetType, this);
40  		return TemplateBuilder.createPattern(getBlock(c), this).setAddGeneratedBy(isAddGeneratedBy()).substituteSingle(targetType, CtBlock.class);
41  	}
42  
43  	public Void S() {
44  		return null;
45  	}
46  
47  	/**
48  	 * This method must be implemented to define the template block.
49  	 */
50  	public abstract void block() throws Throwable;
51  }
