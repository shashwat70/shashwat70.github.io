32  import static org.junit.Assert.assertEquals;
33  import static org.junit.Assert.assertSame;
34  import static org.junit.Assert.assertTrue;
35  import static spoon.testing.utils.ModelUtils.build;
36  import static spoon.testing.utils.ModelUtils.createFactory;
37  
38  public class SwitchCaseTest {
39  
40  	@Test
41  	public void testIterationStatements() {
42  		Factory factory = createFactory();
43  		CtClass<?> clazz = factory
44  				.Code()
45  				.createCodeSnippetStatement(
46  						"" + "class X {" + "public void foo() {" + " int x=0;" +
47  								"switch(x) {"
48  								+ "case 0: x=x+1;break;"
49  								+ "case 1: x=0;"
50  								+ "default: x=-1;"
51  								+ "}"
52  								+ "}};")
53  				.compile();
54  		CtMethod<?> foo = (CtMethod<?>) clazz.getMethods().toArray()[0];
55  
56  		CtSwitch<?> sw = foo.getElements(
57  				new TypeFilter<CtSwitch<?>>(CtSwitch.class)).get(0);
58  
59  		assertEquals(3, sw.getCases().size());
60  
61  		CtCase<?> c = sw.getCases().get(0);
62  
63  		assertEquals(0, ((CtLiteral<?>) c.getCaseExpression()).getValue());
64  		assertEquals(2, c.getStatements().size());
65  
66  		List<CtStatement> l = new ArrayList<>();
67  
68  		// this compiles (thanks to the new CtCase extends CtStatementList)
69  		for (CtStatement s : c) {
70  			l.add(s);
71  		}
72  		assertTrue(c.getStatements().equals(l));
73  	}
74  
75  	@Test
76  	public void testSwitchStatementOnAString() throws Exception {
77  		CtClass<?> clazz = build("spoon.test.model.testclasses", "SwitchStringClass");
78  
79  		CtMethod<?> method = (CtMethod<?>) clazz.getMethods().toArray()[0];
80  		CtSwitch<?> ctSwitch = method
81  				.getElements(new TypeFilter<CtSwitch<?>>(CtSwitch.class))
82  				.get(0);
83  
84  		// Checks the selector is a string.
85  		assertSame(String.class,
86  				ctSwitch.getSelector().getType().getActualClass());
87  
88  		// Checks all cases are strings.
89  		for (CtCase<?> aCase : ctSwitch.getCases()) {
90  			if (aCase.getCaseExpression() == null) {
91  				// default case
92  				continue;
93  			}
94  			assertSame(String.class,
95  					aCase.getCaseExpression().getType().getActualClass());
96  		}
97  	}
98  }
