6   package spoon.reflect.visitor;
7   
8   import spoon.reflect.declaration.CtElement;
9   import spoon.reflect.declaration.CtNamedElement;
10  import spoon.support.Experimental;
11  
12  import java.util.function.Function;
13  
14  /**
15   * Represents that a lexical scope in the language
16   *
17   * Note that scopes are changing after variable declaration. For example:
18   *
19   * void draw() {
20   *  //scope1
21   * int a;
22   * //scope2
23   * int b;
24   * //scope3
25   * }
26   *
27   * See https://en.wikipedia.org/wiki/Scope_(computer_science)#Lexical_scoping
28   */
29  @Experimental
30  public interface LexicalScope {
31  	/** adds an element to the scope */
32  	LexicalScope addNamedElement(CtNamedElement element);
33  
34  	/**
35  	 * @return the {@link CtElement} which represents the current scope
36  	 */
37  	CtElement getScopeElement();
38  
39  	/**
40  	 * @param name to be searched simple name
41  	 * @param fnc is called for each named element with same simple name, which is defined in this or parent {@link LexicalScope}.
42  	 * 	Function `fnc` is called as long as there are some matching elements and `fnc` returns null.
43  	 * 	If `fnc` returns not null value then searching is stopped and that value is a returned
44  	 * @return the value returned by `fnc` or null
45  	 */
46  	<T> T forEachElementByName(String name, Function<? super CtNamedElement, T> fnc);
47  
48  }
