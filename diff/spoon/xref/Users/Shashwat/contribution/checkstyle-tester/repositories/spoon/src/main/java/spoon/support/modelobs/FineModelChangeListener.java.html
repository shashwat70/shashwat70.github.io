6   package spoon.support.modelobs;
7   
8   import spoon.reflect.declaration.CtElement;
9   import spoon.reflect.declaration.ModifierKind;
10  import spoon.reflect.path.CtRole;
11  
12  import java.util.Collection;
13  import java.util.List;
14  import java.util.Map;
15  import java.util.Set;
16  
17  
18  /** Can be subclassed by clients who want to be notified on all changes in AST nodes */
19  public interface FineModelChangeListener {
20  	/** a field corresponding to the role is being set in the AST node */
21  	void onObjectUpdate(CtElement currentElement, CtRole role, CtElement newValue, CtElement oldValue);
22  
23  	/** a field corresponding to the role is being set in the AST node */
24  	void onObjectUpdate(CtElement currentElement, CtRole role, Object newValue, Object oldValue);
25  
26  	/** a field corresponding to the role is being set to null */
27  	void onObjectDelete(CtElement currentElement, CtRole role, CtElement oldValue);
28  
29  	/** a newValue is appended to the list corresponding to the role in the AST node */
30  	void onListAdd(CtElement currentElement, CtRole role, List field, CtElement newValue);
31  
32  	/** a newValue is appended to the list corresponding to the role in the AST node */
33  	void onListAdd(CtElement currentElement, CtRole role, List field, int index, CtElement newValue);
34  
35  	/** an oldValue is deleted in the list corresponding to the role in the AST node */
36  	void onListDelete(CtElement currentElement, CtRole role, List field, Collection<? extends CtElement> oldValue);
37  
38  	/** an oldValue is deleted in the list corresponding to the role in the AST node */
39  	void onListDelete(CtElement currentElement, CtRole role, List field, int index, CtElement oldValue);
40  
41  	/** a list corresponding to the role in the AST node is emptied */
42  	void onListDeleteAll(CtElement currentElement, CtRole role, List field, List oldValue);
43  
44  	/** a newValue is appended to the map corresponding to the role in the AST node */
45  	<K, V> void onMapAdd(CtElement currentElement, CtRole role, Map<K, V> field, K key, CtElement newValue);
46  
47  	/** a map corresponding to the role in the AST node is emptied */
48  	<K, V> void onMapDeleteAll(CtElement currentElement, CtRole role, Map<K, V> field, Map<K, V> oldValue);
49  
50  	/** a newValue is appended to the set corresponding to the role in the AST node */
51  	void onSetAdd(CtElement currentElement, CtRole role, Set field, CtElement newValue);
52  
53  	/** a newValue is appended to the set corresponding to the role in the AST node */
54  	<T extends Enum> void onSetAdd(CtElement currentElement, CtRole role, Set field, T newValue);
55  
56  	/** an oldValue is deleted in the set corresponding to the role in the AST node */
57  	void onSetDelete(CtElement currentElement, CtRole role, Set field, CtElement oldValue);
58  
59  	/** an oldValue is deleted in the set corresponding to the role in the AST node */
60  	void onSetDelete(CtElement currentElement, CtRole role, Set field, Collection<ModifierKind> oldValue);
61  
62  	/** an oldValue is deleted in the set corresponding to the role in the AST node */
63  	void onSetDelete(CtElement currentElement, CtRole role, Set field, ModifierKind oldValue);
64  
65  	/** a set corresponding to the role in the AST node is emptied */
66  	void onSetDeleteAll(CtElement currentElement, CtRole role, Set field, Set oldValue);
67  }
