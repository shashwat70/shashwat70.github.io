6   package spoon.reflect.code;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  import spoon.support.DerivedProperty;
11  
12  import static spoon.reflect.path.CtRole.TARGET_LABEL;
13  
14  /**
15   * This abstract code element represents all the statements that break the
16   * control flow of the program and which can support a label.
17   */
18  public interface CtLabelledFlowBreak extends CtCFlowBreak {
19  	/**
20  	 * Gets the label from which the control flow breaks (null if no label
21  	 * defined).
22  	 */
23  	@PropertyGetter(role = TARGET_LABEL)
24  	String getTargetLabel();
25  
26  	/**
27  	 * Sets the label from which the control flow breaks (null if no label
28  	 * defined).
29  	 */
30  	@PropertySetter(role = TARGET_LABEL)
31  	<T extends CtLabelledFlowBreak> T setTargetLabel(String targetLabel);
32  
33  	@DerivedProperty
34  	CtStatement getLabelledStatement();
35  }
