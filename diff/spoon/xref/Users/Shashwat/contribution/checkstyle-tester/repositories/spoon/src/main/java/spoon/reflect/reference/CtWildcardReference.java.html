6   package spoon.reflect.reference;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  import spoon.reflect.path.CtRole;
11  import spoon.support.UnsettableProperty;
12  
13  /**
14   * Represents a wildcard in generic type annotations, i.e. the "?" (e.g. the "?" in Collection&lt;?&gt; or Collection&lt;? extends List&gt;).
15   */
16  public interface CtWildcardReference extends CtTypeParameterReference {
17  	@Override
18  	CtWildcardReference clone();
19  
20  	@Override
21  	@UnsettableProperty
22  	<C extends CtReference> C setSimpleName(String simpleName);
23  
24  	/**
25  	 * Returns {@code true} if the bounds are in <code>extends</code> clause.
26  	 * {@code false} means a <code>super</code> clause.
27  	 */
28  	@PropertyGetter(role = CtRole.IS_UPPER)
29  	boolean isUpper();
30  
31  	/**
32  	 * Set to {@code true} to write <code>extends</code> clause for bounds types.
33  	 */
34  	@PropertySetter(role = CtRole.IS_UPPER)
35  	<T extends CtWildcardReference> T setUpper(boolean upper);
36  
37  	/**
38  	 * A type parameter can have an <code>extends</code> clause which declare
39  	 * one ({@link CtTypeReference} or more ({@link CtIntersectionTypeReference} references.
40  	 * <pre>
41  	 *     // Extends with generics.
42  	 *     T extends Interface1
43  	 *     // Intersection type with generics.
44  	 *     T extends Interface1 &amp; Interface2
45  	 * </pre>
46  	 */
47  	@PropertyGetter(role = CtRole.BOUNDING_TYPE)
48  	@Override
49  	CtTypeReference<?> getBoundingType();
50  
51  	/**
52  	 * Sets the <code>extends</code> clause of the type parameter.
53  	 */
54  	@PropertySetter(role = CtRole.BOUNDING_TYPE)
55  	<T extends CtWildcardReference> T setBoundingType(CtTypeReference<?> superType);
56  }
