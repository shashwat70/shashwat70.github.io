6   package spoon.reflect.code;
7   
8   import spoon.template.TemplateParameter;
9   
10  /**
11   * This code element represents a block of code, that is to say a list of
12   * statements enclosed in curly brackets.
13   *
14   * Example: <pre>
15   *  { // &lt;-- block start
16   *   System.out.println("foo");
17   *  }
18   *	</pre>
19   *
20   * When the context calls for a return
21   * value, the block should contain a return statement as a lastly reachable
22   * statement. The returned type if any is given by R.
23   */
24  public interface CtBlock<R> extends CtStatement, CtStatementList, TemplateParameter<R> {
25  
26  	@Override
27  	CtBlock<R> clone();
28  }
