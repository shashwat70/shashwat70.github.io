6   package spoon.reflect.code;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  
11  import static spoon.reflect.path.CtRole.TARGET;
12  
13  
14  /**
15   * This abstract code element defines an expression which contains a target
16   * expression. In Java, it is generally of the form:
17   * <code>targetExpr.targetedExpr</code>.
18   *
19   * @param <T>
20   * 		"Return" type of this expression
21   * @param <E>
22   * 		Type of the target
23   */
24  public interface CtTargetedExpression<T, E extends CtExpression<?>> extends CtExpression<T> {
25  	/**
26  	 * Gets the target expression. The target is a `CtTypeAccess` for static methods and a sub type of `CtExpression` for everything else.
27  	 */
28  	@PropertyGetter(role = TARGET)
29  	E getTarget();
30  
31  	/**
32  	 * Sets the target expression.
33  	 */
34  	@PropertySetter(role = TARGET)
35  	<C extends CtTargetedExpression<T, E>> C setTarget(E target);
36  
37  	@Override
38  	CtTargetedExpression<T, E> clone();
39  }
