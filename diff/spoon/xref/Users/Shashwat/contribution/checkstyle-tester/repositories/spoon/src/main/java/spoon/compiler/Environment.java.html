29  /**
30   * This interface represents the environment in which Spoon is launched -
31   * accessible through {@link spoon.reflect.factory.Factory#getEnvironment()}. Its
32   * primary use is to report messages, warnings, and errors.
33   */
34  public interface Environment {
35  
36  	/**
37  	 * Gets the Java version compliance level.
38  	 */
39  	int getComplianceLevel();
40  
41  	/**
42  	 * Sets the Java version compliance level.
43  	 */
44  	void setComplianceLevel(int level);
45  
46  	/**
47  	 * @return the kind of pretty-printing expected.
48  	 * most robust: {@link PRETTY_PRINTING_MODE#DEBUG}
49  	 * most sophisticated: {@link PRETTY_PRINTING_MODE#AUTOIMPORT}
50  	 */
51  	PRETTY_PRINTING_MODE getPrettyPrintingMode();
52  
53  	void setPrettyPrintingMode(PRETTY_PRINTING_MODE prettyPrintingMode);
54  
55  	/**
56  	 * This method should be called to print out a message with a source
57  	 * position link during the processing.
58  	 */
59  	void debugMessage(String message);
60  
61  	/**
62  	 * Returns the default file generator for this environment (gives the
63  	 * default output directory for the created files).
64  	 */
65  	FileGenerator<? extends CtElement> getDefaultFileGenerator();
66  
67  	/**
68  	 * Gets the processing manager.
69  	 */
70  	ProcessingManager getManager();
71  
72  	/**
73  	 * Returns the properties for a given processor.
74  	 */
75  	ProcessorProperties getProcessorProperties(String processorName);
76  
77  	/**
78  	 * Sets the properties for a given processor.
79  	 */
80  	void setProcessorProperties(String processorName, ProcessorProperties prop);
81  
82  	/**
83  	 * Returns true is we let Spoon handle imports
84  	 */
85  	boolean isAutoImports();
86  
87  	/**
88  	 * Tells if the processing is stopped, generally because one of the
89  	 * processors called {@link #setProcessingStopped(boolean)} after reporting
90  	 * an error.
91  	 */
92  	boolean isProcessingStopped();
93  
94  	/**
95  	 * Helper method called by a processor to report an error, warning or
96  	 * message as dictated by the severity parameter. Note that this does not
97  	 * stop the processing or any remaining task. To do so, use
98  	 * {@link #setProcessingStopped(boolean)}.
99  	 *
100 	 * @param processor
101 	 *            The processor that report this message. Can be null.
102 	 * @param level
103 	 *            The level of the report
104 	 * @param element
105 	 *            The CtElement to which the report is associated
106 	 * @param message
107 	 *            The message to report
108 	 */
109 	void report(Processor<?> processor, Level level, CtElement element, String message);
110 
111 	/**
112 	 * Helper method called by a processor to report an error, warning or
113 	 * message as dictated by the severity parameter. Note that this does not
114 	 * stop the processing or any remaining task. To do so, use
115 	 * {@link #setProcessingStopped(boolean)}.
116 	 *
117 	 * @param processor
118 	 *            The processor that report this message. Can be null.
119 	 * @param level
120 	 *            The level of the report
121 	 * @param element
122 	 *            The CtElement to which the report is associated
123 	 * @param message
124 	 *            The message to report
125 	 * @param fixes
126 	 *            The problem fixer(s) to correct this problem
127 	 */
128 	void report(Processor<?> processor, Level level,
129 				CtElement element, String message, ProblemFixer<?>... fixes);
130 
131 	/**
132 	 * This method should be called to print out a message during the
133 	 * processing.
134 	 *
135 	 * @param processor
136 	 *            The processor that report this message. Can be null.
137 	 * @param level
138 	 *            The level of the report
139 	 * @param message
140 	 *            The message to report
141 	 */
142 	void report(Processor<?> processor, Level level, String message);
143 
144 	/**
145 	 * This method should be called to report the end of the processing.
146 	 */
147 	void reportEnd();
148 
149 	/**
150 	 * This method should be called to print out a progress message during the
151 	 * processing. On contrary to regular messages, progress messages are not
152 	 * meant to remain in the message logs and just indicate to the user some
153 	 * task progression information.
154 	 */
155 	void reportProgressMessage(String message);
156 
157 	/**
158 	 * Sets the default file generator for this environment.
159 	 */
160 	void setDefaultFileGenerator(FileGenerator<? extends CtElement> generator);
161 
162 	/**
163 	 * Sets the processing manager of this environment.
164 	 */
165 	void setManager(ProcessingManager manager);
166 
167 	/**
168 	 * This method can be called to stop the processing and all the remaining
169 	 * tasks. In general, a processor calls it after reporting a fatal error.
170 	 */
171 	void setProcessingStopped(boolean processingStopped);
172 
173 	/**
174 	 * Gets the size of the tabulations in the generated source code.
175 	 */
176 	int getTabulationSize();
177 
178 	/**
179 	 * Sets the size of the tabulations in the generated source code.
180 	 */
181 	void setTabulationSize(int size);
182 
183 	/**
184 	 * Tells if Spoon uses tabulations in the source code.
185 	 */
186 	boolean isUsingTabulations();
187 
188 	/**
189 	 * Sets Spoon to use tabulations in the source code.
190 	 */
191 	void useTabulations(boolean b);
192 
193 	/**
194 	 * Tell to the Java printer to automatically generate imports and use simple
195 	 * names instead of fully-qualified name.
196 	 */
197 	void setAutoImports(boolean autoImports);
198 
199 	/**
200 	 * Gets the error count from building, processing, and compiling within this
201 	 * environment.
202 	 */
203 	int getErrorCount();
204 
205 	/**
206 	 * Gets the warning count from building, processing, and compiling within
207 	 * this environment.
208 	 */
209 	int getWarningCount();
210 
211 	/**
212 	 * Returns the {@code ClassLoader} which is used by JDT and to resolve classes from references.
213 	 *
214 	 * By default, returns a class loader able to load classes from the
215 	 * Spoon-specific class path set with  {@link #setSourceClasspath(String[])}
216 	 */
217 	ClassLoader getInputClassLoader();
218 
219 	/**
220 	 * Sets a specific classloader for JDT and reference resolution
221 	 */
222 	void setInputClassLoader(ClassLoader classLoader);
223 
224 	/**
225 	 * When set, the generated source code will try to generate code that
226 	 * preserves the line numbers of the original source code. This option may
227 	 * lead to difficult-to-read indentation and formatting.
228 	 */
229 	void setPreserveLineNumbers(boolean preserveLineNumbers);
230 
231 	/**
232 	 * Tells if the source generator will try to preserve the original line numbers.
233 	 */
234 	boolean isPreserveLineNumbers();
235 
236 	/**
237 	 * Returns the source class path of the Spoon model.
238 	 * This class path is used when the SpoonCompiler is building the model and also
239 	 * to find external classes, referenced from within the model.
240 	 */
241 	String[] getSourceClasspath();
242 
243 	/**
244 	 * Sets the source class path of the Spoon model.
245 	 * After the class path is set, it can be retrieved by
246 	 * {@link #getSourceClasspath()}. Only .jar files or directories with *.class files are accepted.
247 	 * The *.jar or *.java files contained in given directories are ignored.
248 	 *
249 	 * @throws InvalidClassPathException if a given classpath does not exists or
250 	 * does not have the right format (.jar file or directory)
251 	 */
252 	void setSourceClasspath(String[] sourceClasspath);
253 
254 	/**
255 	 * Sets the option "noclasspath", use with caution (see explanation below).
256 	 *
257 	 * With this option, Spoon does not require the full classpath to build the
258 	 * model. In this case, all references to classes that are not in the
259 	 * classpath are handled with the reference mechanism. The "simplename" of
260 	 * the reference object refers to the unbound identifier.
261 	 *
262 	 * This option facilitates the use of Spoon when is is hard to have the
263 	 * complete and correct classpath, for example for mining software
264 	 * repositories.
265 	 *
266 	 * For writing analyses, this option works well if you don't cross the
267 	 * reference by a call to getDeclaration() (if you really want to do so,
268 	 * then check for nullness of the result before).
269 	 *
270 	 * In normal mode, compilation errors are signaled as exception, with this
271 	 * option enabled they are signaled as message only. The reason is that in
272 	 * most cases, there are necessarily errors related to the missing classpath
273 	 * elements.
274 	 *
275 	 */
276 	void setNoClasspath(boolean option);
277 
278 	/** Returns the value ot the option noclasspath */
279 	boolean getNoClasspath();
280 
281 	/**
282 	 * Returns the value of the option copy-resources.
283 	 */
284 	boolean isCopyResources();
285 
286 	/**
287 	 * Sets the option copy-resources to copy all resources in a project on the folder destination.
288 	 */
289 	void setCopyResources(boolean copyResources);
290 
291 	/**
292 	 * Returns the value of the option enable-comments.
293 	 */
294 	boolean isCommentsEnabled();
295 
296 	/**
297 	 * Sets the option enable-comments to parse comments of the target project.
298 	 */
299 	void setCommentEnabled(boolean commentEnabled);
300 
301 	/**
302 	 * Gets the level of loggers asked by the user.
303 	 */
304 	Level getLevel();
305 
306 	/**
307 	 * Sets the level of loggers asked by the user.
308 	 */
309 	void setLevel(String level);
310 
311 	/**
312 	 * Checks if we want compile the target source code and get their binary.
313 	 */
314 	boolean shouldCompile();
315 
316 	/**
317 	 * Sets the compile argument.
318 	 */
319 	void setShouldCompile(boolean shouldCompile);
320 
321 	/**
322 	 * Tells whether Spoon does no checks at all.
323 	 * - parents are consistent (see {@link spoon.reflect.visitor.AstParentConsistencyChecker})
324 	 * - hashcode violation (see {@link spoon.support.reflect.declaration.CtElementImpl#equals(Object)})
325 	 * - method violation (see {@link spoon.reflect.declaration.CtType#addMethod(CtMethod)})
326 	 * are active or not.
327 	 *
328 	 * By default all checks are enabled and {@link #checksAreSkipped()} return false.
329 	 */
330 	boolean checksAreSkipped();
331 
332 	/**
333 	 * Disable all consistency checks on the AST. Dangerous! The only valid usage of this is to keep
334 	 * full backward-compatibility.
335 	 */
336 	void disableConsistencyChecks();
337 
338 
339 	/** Return the directory where binary .class files are created */
340 	void setBinaryOutputDirectory(String directory);
341 
342 	/** Set the directory where binary .class files are created */
343 	String getBinaryOutputDirectory();
344 
345 	/**
346 	 * Sets the directory where source files are written
347 	 */
348 	void setSourceOutputDirectory(File directory);
349 
350 	/**
351 	 * Returns the directory where source files are written
352 	 */
353 	File getSourceOutputDirectory();
354 
355 	/**
356 	 * Set the output destination that handles where source files are written
357 	 */
358 	void setOutputDestinationHandler(OutputDestinationHandler outputDestinationHandler);
359 
360 	/**
361 	 * Returns the output destination that handles where source files are written
362 	 */
363 	OutputDestinationHandler getOutputDestinationHandler();
364 
365 	/**
366 	 * get the model change listener that is used to follow the change of the AST.
367 	 */
368 	FineModelChangeListener getModelChangeListener();
369 
370 	/**
371 	 * set the model change listener
372 	 */
373 	void setModelChangeListener(FineModelChangeListener modelChangeListener);
374 
375 	/**
376 	 * Get the encoding used inside the project
377 	 */
378 	Charset getEncoding();
379 
380 	/**
381 	 * Get encoding provider, which is used to detect encoding for each file separately
382 	 */
383 	EncodingProvider getEncodingProvider();
384 
385 	/**
386 	 * Set the encoding to use for parsing source code
387 	 */
388 	void setEncoding(Charset encoding);
389 
390 	/**
391 	 * Set encoding provider, which is used to detect encoding for each file separately
392 	 */
393 	void setEncodingProvider(EncodingProvider encodingProvider);
394 
395 	/**
396 	 * Set the output type used for processing files
397 	 */
398 	void setOutputType(OutputType outputType);
399 
400 	/**
401 	 * Get the output type
402 	 */
403 	OutputType getOutputType();
404 
405 	SpoonProgress getSpoonProgress();
406 
407 	void setSpoonProgress(SpoonProgress spoonProgress);
408 
409 	/**
410 	 * Get the type of serialization to be used by default
411 	 */
412 	CompressionType getCompressionType();
413 
414 	/**
415 	 * Set the type of serialization to be used by default
416 	 */
417 	void setCompressionType(CompressionType serializationType);
418 
419 	/**
420 	 * @return new instance of {@link PrettyPrinter} which is configured for this environment
421 	 */
422 	PrettyPrinter createPrettyPrinter();
423 
424 	/**
425 	 * @return new instance of {@link PrettyPrinter} which prints nice code
426 	 */
427 	PrettyPrinter createPrettyPrinterAutoImport();
428 
429 	/**
430 	 * @param creator a {@link Supplier}, which creates new instance of pretty printer.
431 	 * Can be used to create a {@link SniperJavaPrettyPrinter} for enabling the sniper mode.
432 	 *
433 	 */
434 	void setPrettyPrinterCreator(Supplier<PrettyPrinter> creator);
435 
436 
437 	/**
438 	 * @return true if spoon is allowed to create a model of a project that contains multiple times the same class
439 	 */
440 	boolean isIgnoreDuplicateDeclarations();
441 
442 	/**
443 	 * @param ignoreDuplicateDeclarations (default false)  set to true to allow spoon to create a model of a project that
444 	 *                                 contains multiple times the same class
445 	 */
446 	void setIgnoreDuplicateDeclarations(boolean ignoreDuplicateDeclarations);
447 
448 	/** Drives how the model is pretty-printed to disk, or when {@link CtElement#prettyprint()} is called */
449 	enum PRETTY_PRINTING_MODE {
450 		/** direct in {@link spoon.reflect.visitor.DefaultJavaPrettyPrinter}, no preprocessors are applied to the model before pretty-printing }. */
451 		DEBUG,
452 
453 		/** autoimport mode, adds as many imports as possible */
454 		AUTOIMPORT,
455 
456 		/** force everything to be fully-qualified */
457 		FULLYQUALIFIED
458 	}
