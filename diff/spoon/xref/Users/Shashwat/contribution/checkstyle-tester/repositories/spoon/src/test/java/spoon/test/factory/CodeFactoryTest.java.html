17  package spoon.test.factory;
18  
19  import org.junit.Test;
20  import spoon.reflect.code.CtAssignment;
21  import spoon.reflect.code.CtThisAccess;
22  import spoon.reflect.code.CtTypeAccess;
23  import spoon.reflect.code.CtVariableWrite;
24  import spoon.reflect.declaration.CtField;
25  import spoon.reflect.declaration.CtType;
26  import spoon.reflect.declaration.ModifierKind;
27  import spoon.reflect.factory.Factory;
28  import spoon.reflect.reference.CtTypeReference;
29  
30  import static org.junit.Assert.assertEquals;
31  import static org.junit.Assert.assertNotNull;
32  import static org.junit.Assert.assertTrue;
33  import static spoon.testing.utils.ModelUtils.createFactory;
34  
35  public class CodeFactoryTest {
36  	int i;
37  
38  	@Test
39  	public void testThisAccess() {
40  		final Factory factory = createFactory();
41  		final CtTypeReference<Object> type = factory.Type().createReference("fr.inria.Test");
42  		final CtThisAccess<Object> thisAccess = factory.Code().createThisAccess(type);
43  
44  		assertNotNull(thisAccess.getTarget());
45  		assertTrue(thisAccess.getTarget() instanceof CtTypeAccess);
46  		assertEquals(type, ((CtTypeAccess) thisAccess.getTarget()).getAccessedType());
47  	}
48  
49  	@Test
50  	public void testCreateVariableAssignement() {
51  		Factory factory = createFactory();
52  		CtType t = factory.Class().create("my.MyClass");
53  		CtField f = factory.createCtField("f", factory.Type().booleanPrimitiveType(), "false", ModifierKind.PRIVATE);
54  		t.addField(f);
55  		CtAssignment va = factory.Code().createVariableAssignment(f.getReference(),false,factory.createLiteral(true));
56  
57  		//Variable assignment's assignee is a CtVariableWrite that point toward the right variable.
58  		assertTrue(va.getAssigned() instanceof CtVariableWrite);
59  		assertEquals(f.getReference(), ((CtVariableWrite) va.getAssigned()).getVariable());
60  	}
61  }
