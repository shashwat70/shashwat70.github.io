6   package spoon.reflect.code;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  
11  import static spoon.reflect.path.CtRole.CONDITION;
12  import static spoon.reflect.path.CtRole.EXPRESSION;
13  
14  
15  /**
16   * This code element defines an assert clause.
17   * Example: <pre>assert 1+1==2</pre>
18   */
19  public interface CtAssert<T> extends CtStatement {
20  	/**
21  	 * Gets the assert expression.
22  	 */
23  	@PropertyGetter(role = CONDITION)
24  	CtExpression<Boolean> getAssertExpression();
25  
26  	/**
27  	 * Sets the assert expression.
28  	 */
29  	@PropertySetter(role = CONDITION)
30  	<A extends CtAssert<T>> A setAssertExpression(CtExpression<Boolean> asserted);
31  
32  	/**
33  	 * Gets the expression of the assertion if defined.
34  	 */
35  	@PropertyGetter(role = EXPRESSION)
36  	CtExpression<T> getExpression();
37  
38  	/**
39  	 * Sets the expression of the assertion.
40  	 *
41  	 * For instance assert.setExpression(factory.createCodeSnippetExpression("param != null"))
42  	 */
43  	@PropertySetter(role = EXPRESSION)
44  	<A extends CtAssert<T>> A setExpression(CtExpression<T> expression);
45  
46  	@Override
47  	CtAssert<T> clone();
48  }
