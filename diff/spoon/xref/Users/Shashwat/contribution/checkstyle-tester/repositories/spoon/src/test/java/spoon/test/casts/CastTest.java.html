38  public class CastTest {
39  	Factory factory = ModelUtils.createFactory();
40  	@Test
41  	public void testCast1() {
42  		CtClass<?> clazz = factory
43  				.Code()
44  				.createCodeSnippetStatement(
45  						"" + "class X {" + "public void foo() {"
46  								+ " String x=(String) new Object();" + "}"
47  								+ "};").compile();
48  		CtMethod<?> foo = (CtMethod<?>) clazz.getMethods().toArray()[0];
49  
50  		assertEquals(
51  				"java.lang.String x = ((java.lang.String) (new java.lang.Object()))",
52  				foo.getBody().getStatements().get(0).toString());
53  	}
54  
55  	@Test
56  	public void testCast2() {
57  		CtClass<?> clazz = factory
58  				.Code()
59  				.createCodeSnippetStatement(
60  						""
61  								+ "class X {"
62  								+ "public void foo() {"
63  								+ " Class<String> x=(Class<String>) new Object();"
64  								+ "}" + "};").compile();
65  		CtMethod<?> foo = (CtMethod<?>) clazz.getMethods().toArray()[0];
66  		assertEquals(
67  				"java.lang.Class<java.lang.String> x = ((java.lang.Class<java.lang.String>) (new java.lang.Object()))",
68  				foo.getBody().getStatements().get(0).toString());
69  	}
70  
71  	@Test
72  	public void testCast3() {
73  		CtClass<?> clazz = factory
74  				.Code()
75  				.createCodeSnippetStatement(
76  						""
77  								+ "class X<A> {"
78  								+ "void addConsumedAnnotationType(Class<? extends A> annotationType) {}\n"
79  								+ "public void foo() {" + " Class<?> x = null;"
80  								+ " addConsumedAnnotationType((Class<A>) x);"
81  								+ "}" + "};").compile();
82  		CtMethod<?> foo = clazz.getElements(new NamedElementFilter<>(CtMethod.class, "foo"))
83  				.get(0);
84  		CtVariableRead<?> a = (CtVariableRead<?>) clazz.getElements(
85  				new TypeFilter<>(CtVariableRead.class)).get(0);
86  		assertEquals(1, a.getTypeCasts().size());
87  		assertEquals("addConsumedAnnotationType(((java.lang.Class<A>) (x)))",
88  				foo.getBody().getStatements().get(1).toString());
89  	}
90  
91  	@Test
92  	public void testCase4() throws Exception {
93  		// contract: If the invocation is to a method where the returned type is a generic type,
94  		// getType returns the actual type bound to this particular invocation.
95  		CtType<?> type = build("spoon.test.casts.testclasses", "Castings");
96  
97  		final CtMethod<?> getValueMethod = type.getMethodsByName("getValue").get(0);
98  		final CtInvocation<?> getValueInvocation = Query.getElements(type, new TypeFilter<CtInvocation<?>>(CtInvocation.class) {
99  			@Override
100 			public boolean matches(CtInvocation element) {
101 				return "getValue".equals(element.getExecutable().getSimpleName()) && super.matches(element);
102 			}
103 		}).get(0);
104 
105 		assertEquals("T", getValueMethod.getType().getSimpleName());
106 		assertEquals("T", getValueMethod.getParameters().get(0).getType().getActualTypeArguments().get(0).toString());
107 
108 		assertEquals(type.getFactory().Class().INTEGER, getValueInvocation.getType());
109 	}

111 	@Test
112 	public void testTypeAnnotationOnCast() throws Exception {
113 		final CtType<Castings> aCastings = buildClass(Castings.class);
114 		final CtLocalVariable local = aCastings.getMethod("bar").getElements(new TypeFilter<>(CtLocalVariable.class)).get(0);
115 
116 		assertEquals(1, ((CtTypeReference) local.getDefaultExpression().getTypeCasts().get(0)).getAnnotations().size());
117 		assertEquals("((java.lang.@spoon.test.casts.testclasses.Castings.TypeAnnotation(integer = 1)" + System.lineSeparator() + "String) (\"\"))", local.getDefaultExpression().toString());
118 	}
