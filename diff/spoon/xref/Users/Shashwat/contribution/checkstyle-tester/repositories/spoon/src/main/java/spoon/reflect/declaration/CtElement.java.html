33  import static spoon.reflect.path.CtRole.ANNOTATION;
34  import static spoon.reflect.path.CtRole.COMMENT;
35  import static spoon.reflect.path.CtRole.IS_IMPLICIT;
36  import static spoon.reflect.path.CtRole.POSITION;
37  
38  /**
39   * This interface is the root interface for the metamodel elements (any program
40   * element).
41   */
42  @Root
43  public interface CtElement extends FactoryAccessor, CtVisitable, Cloneable, CtQueryable, Serializable, SourcePositionHolder {
44  
45  	/**
46  	 * Searches for an annotation of the given class that annotates the
47  	 * current element.
48  	 *
49  	 * When used with a shadow element, this method might return an empty list even on an annotated element
50  	 * because annotations without a RUNTIME retention policy are lost after compilation.
51  	 *
52  	 * WARNING: this method uses a class loader proxy, which is costly.
53  	 * Use {@link #getAnnotation(CtTypeReference)} preferably.
54  	 *
55  	 * <p>
56  	 * NOTE: before using an annotation proxy, you have to make sure that all
57  	 * the types referenced by the annotation have been compiled and are in the
58  	 * classpath so that accessed values can be converted into the actual types.
59  	 *
60  	 * @param <A>
61  	 * 		the annotation's type
62  	 * @param annotationType
63  	 * 		the annotation's class
64  	 * @return if found, returns a proxy for this annotation
65  	 */
66  	@PropertyGetter(role = ANNOTATION)
67  	<A extends Annotation> A getAnnotation(Class<A> annotationType);
68  
69  	/**
70  	 * Gets the annotation element for a given annotation type.
71  	 *
72  	 * When used with a shadow element, this method might return an empty list even on an annotated element
73  	 * because annotations without a RUNTIME retention policy are lost after compilation.
74  	 *
75  	 * @param annotationType
76  	 * 		the annotation type
77  	 * @return the annotation if this element is annotated by one annotation of
78  	 * the given type
79  	 */
80  	@PropertyGetter(role = ANNOTATION)
81  	<A extends Annotation> CtAnnotation<A> getAnnotation(
82  			CtTypeReference<A> annotationType);
83  
84  	/**
85  	 * @return  true if the element is annotated by the given annotation type.
86  	 *
87  	 * @param annotationType
88  	 * 		the annotation type
89  	 */
90  	@DerivedProperty
91  	<A extends Annotation> boolean hasAnnotation(Class<A> annotationType);
92  
93  
94  	/**
95  	 * Returns the annotations that are present on this element.
96  	 *
97  	 * For sake of encapsulation, the returned list is unmodifiable.
98  	 */
99  	@PropertyGetter(role = ANNOTATION)
100 	List<CtAnnotation<? extends Annotation>> getAnnotations();
101 
102 	/**
103 	 * Returns the text of the documentation ("javadoc") comment of this
104 	 * element. It contains the text of Javadoc together with the tags.
105 	 *
106 	 * If one only wants only the text without the tag, one can call `getComments().get(0).getContent()`
107 	 *
108 	 * If one wants to analyze the tags, one can call `getComments().get(0).asJavaDoc().getTags()`
109 	 *
110 	 * See also {@link #getComments()}.and {@link spoon.reflect.code.CtJavaDoc}
111 	 */
112 	@DerivedProperty
113 	String getDocComment();
114 
115 	/**
116 	 * Build a short representation of any element.
117 	 */
118 	@DerivedProperty
119 	String getShortRepresentation();
120 
121 	/**
122 	 * Gets the position of this element in input source files
123 	 *
124 	 * @return Source file and line number of this element.
125 	 * It never returns null. Use {@link SourcePosition#isValidPosition()}
126 	 * to detect whether return instance contains start/end indexes.
127 	 */
128 	@PropertyGetter(role = POSITION)
129 	@Override
130 	SourcePosition getPosition();
131 
132 	/**
133 	 * Replaces this element by another one.
134 	 */
135 	void replace(CtElement element);
136 
137 	/**
138 	 * Replaces this element by several elements.
139 	 * If `elements` contains one single element, it is equivalent to {@link #replace(CtElement)}.
140 	 * If `elements` is empty, it is equivalent to {@link #delete()}.
141 	 */
142 	<E extends CtElement> void replace(Collection<E> elements);
143 
144 	/**
145 	 * Add an annotation for this element
146 	 *
147 	 * @param annotation
148 	 * @return <tt>true</tt> if this element changed as a result of the call
149 	 */
150 	@PropertySetter(role = ANNOTATION)
151 	<E extends CtElement> E addAnnotation(CtAnnotation<? extends Annotation> annotation);
152 
153 	/**
154 	 * Remove an annotation for this element
155 	 *
156 	 * @param annotation
157 	 * @return <tt>true</tt> if this element changed as a result of the call
158 	 */
159 	@PropertySetter(role = ANNOTATION)
160 	boolean removeAnnotation(CtAnnotation<? extends Annotation> annotation);
161 
162 	/**
163 	 * Sets the text of the documentation ("javadoc") comment of this
164 	 * declaration. This API will set the content of the first javadoc
165 	 * {@link CtComment} or create a new  javadoc {@link CtComment} if
166 	 * no javadoc {@link CtComment} is available on this object.
167 	 */
168 	<E extends CtElement> E setDocComment(String docComment);
169 
170 	/**
171 	 * Sets the position in the Java source file. Note that this information is
172 	 * used to feed the line numbers in the generated bytecode if any (which is
173 	 * useful for debugging).
174 	 *
175 	 * @param position
176 	 * 		of this element in the input source files
177 	 */
178 	@PropertySetter(role = POSITION)
179 	<E extends CtElement> E setPosition(SourcePosition position);
180 
181 	/**
182 	 * Gets the child elements annotated with the given annotation type's
183 	 * instances.
184 	 *
185 	 * @param <E>
186 	 * 		the element's type
187 	 * @param annotationType
188 	 * 		the annotation type
189 	 * @return all the child elements annotated with an instance of the given
190 	 * annotation type
191 	 */
192 	<E extends CtElement> List<E> getAnnotatedChildren(
193 			Class<? extends Annotation> annotationType);
194 
195 	/**
196 	 * Returns true if this element is not present in the code (automatically added by the
197 	 * Java compiler or inferred when the model is built).
198 	 * Consequently, implicit elements are not pretty-printed and have no position.
199 	 */
200 	@PropertyGetter(role = IS_IMPLICIT)
201 	boolean isImplicit();
202 
203 	/**
204 	 * Sets this element to be implicit.
205 	 */
206 	@PropertySetter(role = IS_IMPLICIT)
207 	<E extends CtElement> E setImplicit(boolean b);
208 
209 	/**
210 	 * Calculates and returns the set of all the types referenced by this
211 	 * element (and sub-elements in the AST).
212 	 */
213 	@DerivedProperty
214 	Set<CtTypeReference<?>> getReferencedTypes();
215 
216 	/**
217 	 * Returns all the children elements recursively matching the filter.
218 	 * If the receiver (this) matches the filter, it is also returned
219 	 */
220 	<E extends CtElement> List<E> getElements(Filter<E> filter);
221 
222 	/**
223 	 * Sets the position of this element and all its children element. Note that
224 	 * this information is used to feed the line numbers in the generated
225 	 * bytecode if any (which is useful for debugging).
226 	 *
227 	 * @param position
228 	 * 		of this element and all children in the input source file
229 	 */
230 	<E extends CtElement> E setPositions(SourcePosition position);
231 
232 	/**
233 	 * Sets the annotations for this element.
234 	 */
235 	@PropertySetter(role = ANNOTATION)
236 	<E extends CtElement> E setAnnotations(List<CtAnnotation<? extends Annotation>> annotation);
237 
238 	/**
239 	 * Gets the parent of current reference.
240 	 *
241 	 * @throws ParentNotInitializedException
242 	 * 		when the parent of this element is not initialized
243 	 */
244 	@DerivedProperty
245 	CtElement getParent() throws ParentNotInitializedException;
246 
247 	/**
248 	 * Gets the first parent that matches the given type.
249 	 */
250 	<P extends CtElement> P getParent(Class<P> parentType) throws ParentNotInitializedException;
251 
252 	/**
253 	 * Gets the first parent that matches the filter.
254 	 * If the receiver (this) matches the filter, it is also returned
255 	 */
256 	<E extends CtElement> E getParent(Filter<E> filter) throws ParentNotInitializedException;
257 
258 	/**
259 	 * Manually sets the parent element of the current element.
260 	 *
261 	 * @param parent
262 	 * 		parent reference.
263 	 */
264 	<E extends CtElement> E setParent(E parent);
265 
266 	/**
267 	 * Tells if this parent has been initialized.
268 	 */
269 	boolean isParentInitialized();
270 
271 	/**
272 	 * Tells if the given element is a direct or indirect parent.
273 	 */
274 	boolean hasParent(CtElement candidate);
275 
276 	/**
277 	 * Calculates and sets all the parents below this element. This function can
278 	 * be called to check and fix parents after manipulating the model.
279 	 */
280 	void updateAllParentsBelow();
281 
282 	/**
283 	 * @return the {@link CtRole} of the parent's attribute where this element is used.
284 	 * It returns the primary role. For example ((CtMethod) method).getRoleInParent() returns {@link CtRole#TYPE_MEMBER}.
285 	 * <br>
286 	 * Returns null if parent doesn't contain this element as direct children or if this element has no parent.
287 	 */
288 	CtRole getRoleInParent();
289 
290 	/**
291 	 * Deletes the element. For instance, delete a statement from its containing block. Warning: it may result in an incorrect AST, use at your own risk.
292 	 */
293 	void delete();
294 
295 	/**
296 	 * Saves a bunch of metadata inside an Element
297 	 */
298 	<E extends CtElement> E setAllMetadata(Map<String, Object> metadata);
299 
300 	/**
301 	 * Saves metadata inside an Element.
302 	 */
303 	<E extends CtElement> E putMetadata(String key, Object val);
304 
305 	/**
306 	 * Retrieves metadata stored in an element. Returns null if it does not exist.
307 	 */
308 	Object getMetadata(String key);
309 
310 	/**
311 	 * Retrieves all metadata stored in an element.
312 	 */
313 	Map<String, Object> getAllMetadata();
314 
315 	/**
316 	 * Returns the metadata keys stored in an element.
317 	 */
318 	Set<String> getMetadataKeys();
319 
320 	/**
321 	 * Set the comment list
322 	 */
323 	@PropertySetter(role = COMMENT)
324 	<E extends CtElement> E setComments(List<CtComment> comments);
325 
326 	/**
327 	 * The list of comments
328 	 * @return the list of comment
329 	 */
330 	@PropertyGetter(role = COMMENT)
331 	List<CtComment> getComments();
332 
333 	/**
334 	 * Add a comment to the current element
335 	 * <code>element.addComment(element.getFactory().Code().createComment("comment", CtComment.CommentType.INLINE)</code>
336 	 * @param comment the comment
337 	 */
338 	@PropertySetter(role = COMMENT)
339 	<E extends CtElement> E addComment(CtComment comment);
340 
341 	/**
342 	 * Remove a comment
343 	 * @param comment the comment to remove
344 	 */
345 	@PropertySetter(role = COMMENT)
346 	<E extends CtElement> E removeComment(CtComment comment);
347 
348 	/**
349 	 * Clone the element which calls this method in a new object.
350 	 *
351 	 * Note that that references are kept as is, and thus, so if you clone whole classes
352 	 * or methods, some parts of the cloned element (eg executable references) may still point to the initial element.
353 	 * In this case, consider using methods {@link spoon.refactoring.Refactoring#copyType(CtType)} and {@link spoon.refactoring.Refactoring#copyMethod(CtMethod)} instead which does additional work beyond cloning.
354 	 */
355 	CtElement clone();
356 
357 	/**
358 	 * @return a a single value (eg a CtElement), List, Set or Map depending on this `element` and `role`. Returned collections are read-only.
359 	 * @param role the role of the returned attribute with respect to this element.
360 	 *
361 	 * For instance, "klass.getValueByRole(CtRole.METHOD)" returns a list of methods.
362 	 *
363 	 * See {@link spoon.reflect.meta.impl.RoleHandlerHelper} for more advanced methods.
364 	 */
365 	<T> T getValueByRole(CtRole role);
366 
367 	/**
368 	 * Sets a field according to a role.
369 	 * @param role the role of the field to be set
370 	 * @param value to be assigned to this field.
371 	 */
372 	<E extends CtElement, T> E  setValueByRole(CtRole role, T value);
373 
374 	/**
375 	 * Return the path from the model root to this CtElement, eg `.spoon.test.path.Foo.foo#body#statement[index=0]`
376 	 */
377 	CtPath getPath();
378 
379 	/**
380 	 * Returns an iterator over this CtElement's descendants.
381 	 * @return An iterator over this CtElement's descendants.
382 	 */
383 	Iterator<CtElement> descendantIterator();
384 
385 	/**
386 	 * Returns an Iterable instance of this CtElement, allowing for dfs traversal of its descendants.
387 	 * @return an Iterable object that allows iterating through this CtElement's descendants.
388 	 */
389 	Iterable<CtElement> asIterable();
390 
391 	/**
392 	 * @return a list of CtElement containing the element's direct children.
393 	 */
394 	List<CtElement> getDirectChildren();
395 
396 	/**
397 	 * @return the source code of this element according to the setting of {@link Environment#getPrettyPrintingMode()}.
398 	 *
399 	 */
400 	String toString();
401 
402 	/**
403 	 * @return the most straightforward and explicit version of this element.
404 	 */
405 	String toStringDebug();
406 
407 	/**
408 	 * @return the source code of this element with the pretty-printing rules of Spoon
409 	 * Warning: this is not side-effect free, this triggers some {@link spoon.reflect.visitor.ImportAnalyzer} which would change the model: add/remove imports, change the value `implicit` of some model elements, etc.
410 	 */
411 	@Experimental
412 	String prettyprint();
413 
414 }
