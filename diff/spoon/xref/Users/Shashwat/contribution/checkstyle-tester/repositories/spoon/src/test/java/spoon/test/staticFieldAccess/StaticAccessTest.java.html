17  package spoon.test.staticFieldAccess;
18  
19  import org.junit.Before;
20  import org.junit.Test;
21  import spoon.Launcher;
22  import spoon.OutputType;
23  import spoon.SpoonModelBuilder;
24  import spoon.compiler.SpoonResourceHelper;
25  import spoon.reflect.code.CtBlock;
26  import spoon.reflect.declaration.CtType;
27  import spoon.reflect.factory.Factory;
28  import spoon.test.staticFieldAccess.processors.InsertBlockProcessor;
29  
30  import java.io.File;
31  import java.util.Arrays;
32  
33  import static org.junit.Assert.assertTrue;
34  
35  
36  public class StaticAccessTest {
37  
38      Launcher spoon;
39      Factory factory;
40      SpoonModelBuilder compiler;
41  
42      @Before
43      public void setUp()  throws Exception {
44            spoon = new Launcher();
45            factory = spoon.createFactory();
46            compiler = spoon.createCompiler(
47                  factory,
48                  SpoonResourceHelper
49                          .resources(
50                                  "./src/test/java/spoon/test/staticFieldAccess/internal/",
51                                  "./src/test/java/spoon/test/staticFieldAccess/StaticAccessBug.java"
52                          ));
53          compiler.build();
54      }
55  
56      @Test
57      public void testReferences() {
58          CtType<?> type = factory.Type().get("spoon.test.staticFieldAccess.StaticAccessBug");
59          CtBlock<?> block = type.getMethod("references").getBody();
60          assertTrue(block.getStatement(0).toString().contains("Extends.MY_STATIC_VALUE"));
61          assertTrue(block.getStatement(1).toString().contains("Extends.MY_OTHER_STATIC_VALUE"));
62      }
63  
64  
65      @Test
66      public void testProcessAndCompile() throws Exception{
67          compiler.instantiateAndProcess(Arrays.asList(InsertBlockProcessor.class.getName()));
68  
69          // generate files
70          File tmpdir = new File("target/spooned/staticFieldAccess");
71          tmpdir.mkdirs();
72          //    tmpdir.deleteOnExit();
73          factory.getEnvironment().setSourceOutputDirectory(tmpdir);
74          compiler.generateProcessedSourceFiles(OutputType.COMPILATION_UNITS);
75  
76          // try to reload generated datas
77          spoon = new Launcher();
78          compiler = spoon.createCompiler(
79                  SpoonResourceHelper
80                          .resources(tmpdir.getAbsolutePath()));
81          assertTrue(compiler.build());
82      }
83  
84  }
