6   package spoon.reflect.code;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  
11  import java.util.List;
12  
13  import static spoon.reflect.path.CtRole.CASE;
14  import static spoon.reflect.path.CtRole.EXPRESSION;
15  
16  /**
17   * This code element defines a switch statement.
18   *
19   * Example: <pre>
20   * int x = 0;
21   * switch(x) { // &lt;-- switch statement
22   *     case 1:
23   *       System.out.println("foo");
24   * }</pre>
25  
26   * @param <S>
27   * 		the type of the selector expression (it would be better to be able
28   * 		to define an upper bound, but it is not possible because of Java's
29   * 		type hierarchy, especially since the enums that make things even
30   * 		worse!)
31   */
32  public interface CtSwitch<S> extends CtStatement {
33  	/**
34  	 * Gets the selector. The type of the Expression must be <code>char</code>,
35  	 * <code>byte</code>, <code>short</code>, <code>int</code>,
36  	 * <code>Character</code>, <code>Byte</code>, <code>Short</code>,
37  	 * <code>Integer</code>, or an <code>enum</code> type
38  	 */
39  	@PropertyGetter(role = EXPRESSION)
40  	CtExpression<S> getSelector();
41  
42  	/**
43  	 * Sets the selector. The type of the Expression must be <code>char</code>,
44  	 * <code>byte</code>, <code>short</code>, <code>int</code>,
45  	 * <code>Character</code>, <code>Byte</code>, <code>Short</code>,
46  	 * <code>Integer</code>, or an <code>enum</code> type
47  	 */
48  	@PropertySetter(role = EXPRESSION)
49  	<T extends CtSwitch<S>> T setSelector(CtExpression<S> selector);
50  
51  	/**
52  	 * Gets the list of cases defined for this switch.
53  	 */
54  	@PropertyGetter(role = CASE)
55  	List<CtCase<? super S>> getCases();
56  
57  	/**
58  	 * Sets the list of cases defined for this switch.
59  	 */
60  	@PropertySetter(role = CASE)
61  	<T extends CtSwitch<S>> T setCases(List<CtCase<? super S>> cases);
62  
63  	/**
64  	 * Adds a case;
65  	 */
66  	@PropertySetter(role = CASE)
67  	<T extends CtSwitch<S>> T addCase(CtCase<? super S> c);
68  
69  	/**
70  	 * Removes a case;
71  	 */
72  	@PropertySetter(role = CASE)
73  	boolean removeCase(CtCase<? super S> c);
74  
75  	@Override
76  	CtSwitch<S> clone();
77  }
