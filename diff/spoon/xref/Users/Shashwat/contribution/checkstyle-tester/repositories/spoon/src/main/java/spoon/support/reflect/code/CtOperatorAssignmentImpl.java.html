6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.BinaryOperatorKind;
10  import spoon.reflect.code.CtOperatorAssignment;
11  import spoon.reflect.visitor.CtVisitor;
12  
13  import static spoon.reflect.path.CtRole.OPERATOR_KIND;
14  
15  public class CtOperatorAssignmentImpl<T, A extends T> extends CtAssignmentImpl<T, A> implements CtOperatorAssignment<T, A> {
16  	private static final long serialVersionUID = 1L;
17  
18  	@MetamodelPropertyField(role = OPERATOR_KIND)
19  	BinaryOperatorKind kind;
20  
21  	@Override
22  	public void accept(CtVisitor visitor) {
23  		visitor.visitCtOperatorAssignment(this);
24  	}
25  
26  	@Override
27  	public BinaryOperatorKind getKind() {
28  		return kind;
29  	}
30  
31  	@Override
32  	public <C extends CtOperatorAssignment<T, A>> C setKind(BinaryOperatorKind kind) {
33  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, OPERATOR_KIND, kind, this.kind);
34  		this.kind = kind;
35  		return (C) this;
36  	}
37  
38  	@Override
39  	public CtOperatorAssignment<T, A> clone() {
40  		return (CtOperatorAssignment<T, A>) super.clone();
41  	}
42  }
