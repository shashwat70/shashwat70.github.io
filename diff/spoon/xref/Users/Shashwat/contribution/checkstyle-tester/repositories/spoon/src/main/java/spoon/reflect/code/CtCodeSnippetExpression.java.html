6   package spoon.reflect.code;
7   
8   import spoon.reflect.declaration.CtCodeSnippet;
9   import spoon.support.compiler.SnippetCompilationError;
10  
11  /**
12   * This element is a code snippet that must represent an expression and can thus
13   * be inserted in the program's model as is. Code snippets should be avoided
14   * since no controls can be performed on them.
15   */
16  public interface CtCodeSnippetExpression<T> extends CtExpression<T>, CtCodeSnippet {
17  
18  	/**
19  	 * Compiles this expression snippet to produce the corresponding AST expression.
20  	 *
21  	 * @throws SnippetCompilationError
22  	 * 		when the current snippet is not valid Java code expression
23  	 */
24  	<E extends CtExpression<T>> E compile() throws SnippetCompilationError;
25  
26  	@Override
27  	CtCodeSnippetExpression<T> clone();
28  
29  }
