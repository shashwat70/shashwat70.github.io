6   package spoon.reflect.code;
7   
8   import spoon.reflect.declaration.CtCodeSnippet;
9   import spoon.reflect.reference.CtTypeReference;
10  import spoon.support.compiler.SnippetCompilationError;
11  
12  /**
13   * This element is a code snippet that must represent a statement and can thus
14   * be inserted in the program's model as is. Code snippets should be avoided
15   * since no controls can be performed on them.
16   */
17  
18  public interface CtCodeSnippetStatement extends CtStatement, CtCodeSnippet {
19  
20  	/**
21  	 * Compiles this statement code snippet to produce the corresponding AST
22  	 * statement.
23  	 *
24  	 * If you want to compile a non-void return or a snippet that uses a non-void  return,
25  	 * use, {@link spoon.support.compiler.SnippetCompilationHelper#compileStatement(CtCodeSnippetStatement, CtTypeReference)}
26  	 *
27  	 * @return a statement
28  	 * @throws SnippetCompilationError
29  	 * 		when the current snippet is not valid Java code
30  	 */
31  	<S extends CtStatement> S compile() throws SnippetCompilationError;
32  
33  	@Override
34  	CtCodeSnippetStatement clone();
35  }
