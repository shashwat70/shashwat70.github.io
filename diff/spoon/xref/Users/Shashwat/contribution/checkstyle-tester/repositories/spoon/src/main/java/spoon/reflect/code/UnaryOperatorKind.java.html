6   package spoon.reflect.code;
7   
8   /**
9    * This enumeration defines all the kinds of unary operators.
10   */
11  public enum UnaryOperatorKind {
12  	/**
13  	 * Positivation.
14  	 */
15  	POS, // +
16  	/**
17  	 * Negation.
18  	 */
19  	NEG, // -
20  	/**
21  	 * Logical inversion.
22  	 */
23  	NOT, // !
24  	/**
25  	 * Binary complement.
26  	 */
27  	COMPL, // ~
28  	/**
29  	 * Incrementation pre assignment.
30  	 */
31  	PREINC, // ++ _
32  	/**
33  	 * Decrementation pre assignment.
34  	 */
35  	PREDEC, // -- _
36  	/**
37  	 * Incrementation post assignment.
38  	 */
39  	POSTINC, // _ ++
40  	/**
41  	 * Decrementation post assignment.
42  	 */
43  	POSTDEC
44  	// _ --
45  }
