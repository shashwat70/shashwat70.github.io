6   package spoon.pattern.internal.node;
7   
8   import spoon.pattern.Generator;
9   import spoon.pattern.internal.DefaultGenerator;
10  import spoon.pattern.internal.ResultHolder;
11  import spoon.support.util.ImmutableMap;
12  
13  /**
14   * Represents a kind of {@link RootNode},
15   * whose AST statements are understood as pattern statements.
16   * For example CtForEach statement is handled as repeated generation of pattern
17   * Or CtIf statement is handled as optionally generated pattern
18   */
19  public interface InlineNode extends RootNode {
20  	/**
21  	 * Generates inline statements of this inline {@link RootNode}.
22  	 * This method is used when sources of pattern have to be printed
23  	 * @param generator a to be used {@link Generator}
24  	 * @param result holder of the result
25  	 * @param parameters a {@link ImmutableMap} with current parameters
26  	 */
27  	<T> void generateInlineTargets(DefaultGenerator generator, ResultHolder<T> result, ImmutableMap parameters);
28  }
