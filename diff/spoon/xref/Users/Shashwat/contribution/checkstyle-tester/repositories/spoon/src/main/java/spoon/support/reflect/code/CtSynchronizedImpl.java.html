6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtBlock;
10  import spoon.reflect.code.CtExpression;
11  import spoon.reflect.code.CtSynchronized;
12  import spoon.reflect.visitor.CtVisitor;
13  
14  import static spoon.reflect.path.CtRole.BODY;
15  import static spoon.reflect.path.CtRole.EXPRESSION;
16  
17  public class CtSynchronizedImpl extends CtStatementImpl implements CtSynchronized {
18  	private static final long serialVersionUID = 1L;
19  
20  	@MetamodelPropertyField(role = BODY)
21  	CtBlock<?> block;
22  
23  	@MetamodelPropertyField(role = EXPRESSION)
24  	CtExpression<?> expression;
25  
26  	@Override
27  	public void accept(CtVisitor visitor) {
28  		visitor.visitCtSynchronized(this);
29  	}
30  
31  	@Override
32  	public CtBlock<?> getBlock() {
33  		return block;
34  	}
35  
36  	@Override
37  	public CtExpression<?> getExpression() {
38  		return expression;
39  	}
40  
41  	@Override
42  	public <T extends CtSynchronized> T setBlock(CtBlock<?> block) {
43  		if (block != null) {
44  			block.setParent(this);
45  		}
46  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, BODY, block, this.block);
47  		this.block = block;
48  		return (T) this;
49  	}
50  
51  	@Override
52  	public <T extends CtSynchronized> T setExpression(CtExpression<?> expression) {
53  		if (expression != null) {
54  			expression.setParent(this);
55  		}
56  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, EXPRESSION, expression, this.expression);
57  		this.expression = expression;
58  		return (T) this;
59  	}
60  
61  	@Override
62  	public CtSynchronized clone() {
63  		return (CtSynchronized) super.clone();
64  	}
65  }
