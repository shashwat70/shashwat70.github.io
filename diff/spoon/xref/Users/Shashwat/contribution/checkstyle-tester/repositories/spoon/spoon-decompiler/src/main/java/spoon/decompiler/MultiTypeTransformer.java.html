17  package spoon.decompiler;
18  
19  import spoon.reflect.declaration.CtType;
20  
21  import java.util.Collection;
22  import java.util.LinkedHashSet;
23  
24  public class MultiTypeTransformer implements TypeTransformer {
25  
26  	protected LinkedHashSet<TypeTransformer> transformers;
27  
28  	public MultiTypeTransformer() {
29  		transformers = new LinkedHashSet<>();
30  	}
31  
32  	public void addTransformer(TypeTransformer transformer) {
33  		transformers.add(transformer);
34  	}
35  
36  	public void addTransformers(Collection<TypeTransformer> transformers) {
37  		this.transformers.addAll(transformers);
38  	}
39  
40  	@Override
41  	public void transform(CtType type) {
42  		for (TypeTransformer transformer: transformers) {
43  			if (transformer.accept(type)) {
44  				transformer.transform(type);
45  			}
46  		}
47  	}
48  
49  	@Override
50  	public boolean accept(CtType type) {
51  		for (TypeTransformer transformer: transformers) {
52  			if (transformer.accept(type)) {
53  				return true;
54  			}
55  		}
56  		return false;
57  	}
58  }
