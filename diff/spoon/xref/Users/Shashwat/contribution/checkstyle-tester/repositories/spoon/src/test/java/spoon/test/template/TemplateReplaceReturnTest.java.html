17  package spoon.test.template;
18  
19  import static org.junit.Assert.assertEquals;
20  import static spoon.testing.utils.ModelUtils.getOptimizedString;
21  
22  import java.io.File;
23  
24  import org.junit.Test;
25  
26  import spoon.Launcher;
27  import spoon.OutputType;
28  import spoon.reflect.code.CtBlock;
29  import spoon.reflect.code.CtExpression;
30  import spoon.reflect.declaration.CtClass;
31  import spoon.reflect.factory.Factory;
32  import spoon.support.compiler.FileSystemFile;
33  import spoon.test.template.testclasses.ReturnReplaceTemplate;
34  import spoon.testing.utils.ModelUtils;
35  
36  public class TemplateReplaceReturnTest {
37  
38  	@Test
39  	public void testReturnReplaceTemplate() {
40  		//contract: the template engine supports replace of `return _param_.S()` by `<CtBlock>`
41  		Launcher launcher = new Launcher();
42  		launcher.addTemplateResource(new FileSystemFile("./src/test/java/spoon/test/template/testclasses/ReturnReplaceTemplate.java"));
43  
44  		launcher.buildModel();
45  		Factory factory = launcher.getFactory();
46  
47  		CtBlock<String> model = (CtBlock) factory.Class().get(ReturnReplaceTemplate.class).getMethod("sample").getBody();
48  		
49  		CtClass<?> resultKlass = factory.Class().create(factory.Package().getOrCreate("spoon.test.template"), "ReturnReplaceResult");
50  		new ReturnReplaceTemplate(model).apply(resultKlass);
51  		assertEquals("{ if ((java.lang.System.currentTimeMillis() % 2L) == 0) { return \"Panna\"; } else { return \"Orel\"; }}", getOptimizedString(resultKlass.getMethod("method").getBody()));
52  		launcher.setSourceOutputDirectory(new File("./target/spooned/"));
53  		launcher.getModelBuilder().generateProcessedSourceFiles(OutputType.CLASSES);
54  		ModelUtils.canBeBuilt(new File("./target/spooned/spoon/test/template/ReturnReplaceResult.java"), 8);
55  	}
56  
57  	@Test
58  	public void testNoReturnReplaceTemplate() {
59  		//contract: the template engine supports replace of return expression by `<CtExpression>`
60  		Launcher launcher = new Launcher();
61  		launcher.addTemplateResource(new FileSystemFile("./src/test/java/spoon/test/template/testclasses/ReturnReplaceTemplate.java"));
62  
63  		launcher.buildModel();
64  		Factory factory = launcher.getFactory();
65  
66  		CtExpression<String> model = factory.createLiteral("AStringLiteral");
67  		
68  		CtClass<?> resultKlass = factory.Class().create(factory.Package().getOrCreate("spoon.test.template"), "ReturnReplaceResult");
69  		new ReturnReplaceTemplate(model).apply(resultKlass);
70  		assertEquals("{ return \"AStringLiteral\";}", getOptimizedString(resultKlass.getMethod("method").getBody()));
71  		launcher.setSourceOutputDirectory(new File("./target/spooned/"));
72  		launcher.getModelBuilder().generateProcessedSourceFiles(OutputType.CLASSES);
73  		ModelUtils.canBeBuilt(new File("./target/spooned/spoon/test/template/ReturnReplaceResult.java"), 8);
74  	}

76  }
