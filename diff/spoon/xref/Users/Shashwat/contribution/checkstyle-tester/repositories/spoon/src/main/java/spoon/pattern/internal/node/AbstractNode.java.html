6   package spoon.pattern.internal.node;
7   
8   import spoon.pattern.internal.PatternPrinter;
9   
10  /**
11   * Represents a parameterized Pattern ValueResolver, which can be used
12   * <ul>
13   * <li>to generate a zero, one or more copies of model using provided parameters</li>
14   * <li>to match zero, one or more instances of model and deliver a matching parameters</li>
15   * </ul>
16   */
17  public abstract class AbstractNode implements RootNode {
18  	private boolean simplifyGenerated = false;
19  	@Override
20  	public String toString() {
21  		return new PatternPrinter().printNode(this);
22  	}
23  	@Override
24  	public boolean isSimplifyGenerated() {
25  		return simplifyGenerated;
26  	}
27  	@Override
28  	public void setSimplifyGenerated(boolean simplifyGenerated) {
29  		this.simplifyGenerated = simplifyGenerated;
30  	}
31  }
