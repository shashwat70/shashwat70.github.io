6   package spoon.reflect.reference;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  import spoon.reflect.code.CtComment;
11  import spoon.reflect.declaration.CtElement;
12  import spoon.reflect.path.CtRole;
13  import spoon.support.DerivedProperty;
14  import spoon.support.UnsettableProperty;
15  
16  import java.util.List;
17  
18  
19  /**
20   * This is the root interface for named program element references. References
21   * can point to program element built in the model or not. In the latter case,
22   * introspection methods fall back on runtime reflection (
23   * <code>java.lang.reflect</code>) to access the program information, as long as
24   * available in the classpath.
25   *
26   * @see spoon.reflect.declaration.CtElement
27   */
28  public interface CtReference extends CtElement {
29  
30  	/**
31  	 * Gets the simple name of referenced element.
32  	 */
33  	@PropertyGetter(role = CtRole.NAME)
34  	String getSimpleName();
35  
36  	/**
37  	 * Sets the name of referenced element.
38  	 */
39  	@PropertySetter(role = CtRole.NAME)
40  	<T extends CtReference> T setSimpleName(String simpleName);
41  
42  	/**
43  	 * Returns the declaration that corresponds to the referenced element only
44  	 * if the declaration is in the analyzed source files.
45  	 * It is strongly advised to use the more robust {@link CtTypeReference#getTypeDeclaration()} and {@link CtExecutableReference#getExecutableDeclaration()} that never return null.
46  	 * @return referenced element or null if element is not in the source path (aka input resource).
47  	 */
48  	@DerivedProperty
49  	CtElement getDeclaration();
50  
51  	@Override
52  	CtReference clone();
53  
54  	/** comments are not possible for references */
55  	@Override
56  	@UnsettableProperty
57  	<E extends CtElement> E setComments(List<CtComment> comments);
58  }
