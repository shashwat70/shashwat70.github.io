6   package spoon.compiler.builder;
7   
8   import org.apache.commons.io.IOUtils;
9   import spoon.compiler.SpoonFile;
10  
11  import java.io.File;
12  import java.io.FileOutputStream;
13  import java.io.IOException;
14  import java.util.Arrays;
15  import java.util.List;
16  
17  public class SourceOptions<T extends SourceOptions<T>> extends Options<T> {
18  	public SourceOptions() {
19  		super(SourceOptions.class);
20  	}
21  
22  	/** adds the given paths as concatenated string with File.pathSeparator as sources */
23  	public T sources(String sources) {
24  		if (sources == null || sources.isEmpty()) {
25  			return myself;
26  		}
27  		return sources(sources.split(File.pathSeparator));
28  	}
29  
30  	/** adds the given paths as sources */
31  	public T sources(String... sources) {
32  		if (sources == null || sources.length == 0) {
33  			args.add(".");
34  			return myself;
35  		}
36  		args.addAll(Arrays.asList(sources));
37  		return myself;
38  	}
39  
40  	/** adds the given {@link spoon.compiler.SpoonFile} as sources */
41  	public T sources(List<SpoonFile> sources) {
42  		if (sources == null || sources.isEmpty()) {
43  			args.add(".");
44  			return myself;
45  		}
46  		for (SpoonFile source : sources) {
47  			if (source.isActualFile()) {
48  				args.add(source.toString());
49  			} else {
50  				try {
51  					File file = File.createTempFile(source.getName(), ".java");
52  					file.deleteOnExit();
53  					try (FileOutputStream fileOutputStream = new FileOutputStream(file)) {
54  						IOUtils.copy(source.getContent(), fileOutputStream);
55  					}
56  					args.add(file.toString());
57  				} catch (IOException e) {
58  					throw new RuntimeException(e.getMessage(), e);
59  				}
60  			}
61  		}
62  		return myself;
63  	}
64  }
