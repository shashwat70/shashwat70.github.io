6   package spoon.processing;
7   
8   import java.io.File;
9   import java.util.List;
10  
11  import spoon.reflect.declaration.CtElement;
12  
13  /**
14   * This interface should be implemented by processing tasks that generate new
15   * files as processing results. This interface is typically implemented by
16   * processors that generate files during processing. For a given processing
17   * environment, the default file generator is set to the default output directory
18   * that is retrieved by using
19   * {@link spoon.compiler.Environment#getDefaultFileGenerator()}.
20   */
21  public interface FileGenerator<T extends CtElement> extends Processor<T> {
22  	/**
23  	 * Gets the root directory where files are created.
24  	 */
25  	File getOutputDirectory();
26  
27  	/**
28  	 * Gets the created files.
29  	 */
30  	List<File> getCreatedFiles();
31  
32  }
