6   package spoon.reflect.cu;
7   
8   import spoon.support.sniper.internal.ElementSourceFragment;
9   import spoon.support.Experimental;
10  
11  /**
12   * This interface represents an element which knows its position in a source file.
13   */
14  public interface SourcePositionHolder {
15  	/** If the element comes from a Java source file (hence has not created during transformation), returns the position in the original source file */
16  	SourcePosition getPosition();
17  
18  	/**
19  	 * Returns the original source code (maybe different from toString() if a transformation has been applied.
20  	 * Or {@link ElementSourceFragment#NO_SOURCE_FRAGMENT} if this element has no original source fragment.
21  	 *
22  	 * Warning: this is a advanced method which cannot be considered as part of the stable API
23  	 *
24  	 */
25  	@Experimental
26  	default ElementSourceFragment getOriginalSourceFragment() {
27  		return ElementSourceFragment.NO_SOURCE_FRAGMENT;
28  	}
29  }
