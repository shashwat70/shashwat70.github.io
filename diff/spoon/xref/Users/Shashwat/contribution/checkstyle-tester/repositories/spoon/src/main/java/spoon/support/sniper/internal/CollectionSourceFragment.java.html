6   package spoon.support.sniper.internal;
7   
8   import java.util.List;
9   
10  import spoon.reflect.meta.ContainerKind;
11  import spoon.reflect.path.CtRole;
12  import spoon.support.Experimental;
13  
14  /**
15   * {@link SourceFragment} of List or Set of {@link ElementSourceFragment}s which belong to collection role.
16   * For example list of Type members or list of parameters, etc.
17   * Or set of modifiers and annotations
18   */
19  @Experimental
20  public class CollectionSourceFragment implements SourceFragment {
21  
22  	private final List<SourceFragment> items;
23  
24  	public CollectionSourceFragment(List<SourceFragment> items) {
25  		this.items = items;
26  	}
27  
28  	@Override
29  	public String getSourceCode() {
30  		StringBuilder sb = new StringBuilder();
31  		for (SourceFragment childSourceFragment : items) {
32  			sb.append(childSourceFragment.getSourceCode());
33  		}
34  		return sb.toString();
35  	}
36  
37  	/**
38  	 * @return child source fragments of this collection
39  	 */
40  	public List<SourceFragment> getItems() {
41  		return items;
42  	}
43  
44  	@Override
45  	public String toString() {
46  		return items.toString();
47  	}
48  
49  	/**
50  	 * @return true if collection contains only children of one role handler with container kind LIST
51  	 */
52  	public boolean isOrdered() {
53  		CtRole role = null;
54  		for (SourceFragment childSourceFragment : items) {
55  			if (childSourceFragment instanceof ElementSourceFragment) {
56  				ElementSourceFragment esf = (ElementSourceFragment) childSourceFragment;
57  				if (role == null) {
58  					role = esf.getRoleInParent();
59  					ContainerKind kind = esf.getContainerKindInParent();
60  					if (kind != ContainerKind.LIST) {
61  						return false;
62  					}
63  				} else {
64  					if (role != esf.getRoleInParent()) {
65  						//the collection contains elements of different roles. It cannot be ordered
66  						return false;
67  					}
68  					//else there is another element of the same role - ok
69  				}
70  			}
71  		}
72  		//there are only elements of one role of container kind LIST
73  		return true;
74  	}
75  }
