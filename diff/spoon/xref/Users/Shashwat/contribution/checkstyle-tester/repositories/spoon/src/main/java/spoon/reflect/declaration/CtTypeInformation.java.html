6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.reference.CtExecutableReference;
9   import spoon.reflect.reference.CtFieldReference;
10  import spoon.reflect.reference.CtTypeReference;
11  import spoon.support.DerivedProperty;
12  import spoon.reflect.annotations.PropertyGetter;
13  
14  import java.util.Collection;
15  import java.util.Set;
16  
17  import static spoon.reflect.path.CtRole.INTERFACE;
18  import static spoon.reflect.path.CtRole.MODIFIER;
19  import static spoon.reflect.path.CtRole.SUPER_TYPE;
20  
21  /**
22   * Returns information that can be obtained both at compile-time and run-time
23   *
24   * For CtElement, the compile-time information is given
25   *
26   * For CtTypeReference, the runtime information is given (using the Reflection API)
27   *
28   */
29  public interface CtTypeInformation {
30  	/**
31  	 * Returns the interface types directly implemented by this class or
32  	 * extended by this interface.
33  	 */
34  	@PropertyGetter(role = INTERFACE)
35  	Set<CtTypeReference<?>> getSuperInterfaces();
36  
37  	/**
38  	 * Returns the fully qualified name of this type declaration.
39  	 */
40  	@DerivedProperty
41  	String getQualifiedName();
42  
43  	/**
44  	 * Gets modifiers of this type.
45  	 */
46  	@PropertyGetter(role = MODIFIER)
47  	Set<ModifierKind> getModifiers();
48  
49  	/**
50  	 * Return {@code true} if the referenced type is a primitive type (int,
51  	 * double, boolean...).
52  	 */
53  	boolean isPrimitive();
54  
55  	/**
56  	 * Return {@code true} if the referenced type is a anonymous type
57  	 */
58  	boolean isAnonymous();
59  
60  	/**
61  	 * Return {@code true} if the referenced type is declared in an executable.
62  	 * e.g. a type declared in a method or a lambda.
63  	 *
64  	 * This corresponds to <code>isLocalClass</code> of <code>java.lang.Class</code>.
65  	 *
66  	 * <pre>
67  	 *     // Type declared in a method.
68  	 *     public void make() {
69  	 *         class Cook {
70  	 *         }
71  	 *     }
72  	 *     // Type declared in a lambda.
73  	 *     s -&gt; {
74  	 *         class Cook {
75  	 *         }
76  	 *     }
77  	 * </pre>
78  	 */
79  	boolean isLocalType();
80  
81  	/**
82  	 * Returns true if this type is a class. Returns false for others (enum, interface, generics, annotation).
83  	 */
84  	boolean isClass();
85  
86  	/**
87  	 * Returns true if this type is an interface.
88  	 */
89  	boolean isInterface();
90  
91  	/**
92  	 * Returns true if this type is an enum.
93  	 */
94  	boolean isEnum();
95  
96  	/**
97  	 * Returns true if this type is an annotation type.
98  	 */
99  	boolean isAnnotationType();
100 
101 	/**
102 	 * Returns true if it is not a concrete, resolvable class, it if refers to a type parameter directly or indirectly.
103 	 * Direct: "T foo" isGenerics returns true.
104 	 * Indirect: List&lt;T&gt;, or Set&lt;List&lt;T&gt;&gt; isGenerics returns true
105 	 */
106 	@DerivedProperty
107 	boolean isGenerics();
108 
109 	/**
110 	 * Returns true if it has any type parameter (generic or not).
111 	 */
112 	@DerivedProperty
113 	boolean isParameterized();
114 
115 	/**
116 	 * Returns true if the referenced type is a sub-type of the given type.
117 	 * Returns true is type is self, it means: typeX.isSubtypeOf(typeX) is true too
118 	 */
119 	boolean isSubtypeOf(CtTypeReference<?> type);
120 
121 	/**
122 	 * Returns the class type directly extended by this class.
123 	 *
124 	 * getSuperClass().getDeclaration()/getTypeDeclaration() returns the corresponding CtType (if in the source folder of Spoon).
125 	 *
126 	 * @return the class type directly extended by this class, or null if there
127 	 *         is none or if the super class is not in the classpath (in noclasspath mode)
128 	 */
129 	@PropertyGetter(role = SUPER_TYPE)
130 	CtTypeReference<?> getSuperclass();
131 
132 	/**
133 	 * Gets the fields declared by this type.
134 	 */
135 	@DerivedProperty
136 	Collection<CtFieldReference<?>> getDeclaredFields();
137 
138 	/**
139 	 * Gets the fields declared by this type and by all its supertypes if
140 	 * applicable.
141 	 */
142 	@DerivedProperty
143 	Collection<CtFieldReference<?>> getAllFields();
144 
145 	/**
146 	 * Gets a field from its name.
147 	 *
148 	 * @return null if does not exit
149 	 */
150 	CtFieldReference<?> getDeclaredField(String name);
151 
152 	/**
153 	 * Gets a field from this type or any super type or any implemented interface by field name.
154 	 *
155 	 * @return null if does not exit
156 	 */
157 	CtFieldReference<?> getDeclaredOrInheritedField(String fieldName);
158 
159 	/**
160 	 * Gets the executables declared by this type if applicable.
161 	 */
162 	@DerivedProperty
163 	Collection<CtExecutableReference<?>> getDeclaredExecutables();
164 
165 	/**
166 	 * Gets the executables declared by this type and by all its supertypes (static/instance methods, constructors, anonymous static blocks) if
167 	 * applicable. This method returns:
168 	 *
169 	 * <ul>
170 	 *     <li>static, instance and default methods</li>
171 	 *     <li>constructors</li>
172 	 * </ul>
173 	 *
174 	 * If a method is overridden twice in the hierarchy, it counts for two different elements.
175 	 * The method can be abstract.
176 	 */
177 	@DerivedProperty
178 	Collection<CtExecutableReference<?>> getAllExecutables();
179 
180 	/**
181 	 * @return the type erasure, which is computed by the java compiler to ensure that no new classes are created for parametrized types so that generics incur no runtime overhead.
182 	 * See https://docs.oracle.com/javase/tutorial/java/generics/erasure.html
183 	 */
184 	@DerivedProperty
185 	CtTypeReference<?> getTypeErasure();
186 
187 
188 	/**
189 	 * @return true if this represents an array e.g. Object[] or int[]
190 	 */
191 	boolean isArray();
192 }
