72  	@Test
73  	public void testSourceFragmentAddChild() {
74  		//contract: check build of the tree of SourceFragments
75  		ElementSourceFragment rootFragment = createFragment(10, 20);
76  		ElementSourceFragment f;
77  		//add child
78  		assertSame(rootFragment, rootFragment.add(f = createFragment(10, 15)));
79  		assertSame(rootFragment.getFirstChild(), f);
80  		
81  		//add child which is next sibling of first child
82  		assertSame(rootFragment, rootFragment.add(f = createFragment(15, 20)));
83  		assertSame(rootFragment.getFirstChild().getNextSibling(), f);
84  		
85  		//add another child of same start/end, which has to be child of last child
86  		assertSame(rootFragment, rootFragment.add(f = createFragment(15, 20)));
87  		assertSame(rootFragment.getFirstChild().getNextSibling().getFirstChild(), f);
88  
89  		//add another child of smaller start/end, which has to be child of last child
90  		assertSame(rootFragment, rootFragment.add(f = createFragment(16, 20)));
91  		assertSame(rootFragment.getFirstChild().getNextSibling().getFirstChild().getFirstChild(), f);
92  
93  		//add next sibling of root element
94  		assertSame(rootFragment, rootFragment.add(f = createFragment(20, 100)));
95  		assertSame(rootFragment.getNextSibling(), f);
96  		
97  		//add prev sibling of root element. We should get new root
98  		f = createFragment(5, 10);
99  		assertSame(f, rootFragment.add(f));
100 		assertSame(f.getNextSibling(), rootFragment);
101 	}

128 	@Test
129 	public void testSourceFragmentWrapChildrenAndSiblings() {
130 		//contract: the two SourceFragment trees merge correctly together 
131 		ElementSourceFragment rootFragment = createFragment(0, 100);
132 		ElementSourceFragment child = createFragment(50, 60);
133 		rootFragment.add(child);
134 		
135 		ElementSourceFragment childWrapper = createFragment(40, 70);
136 		ElementSourceFragment childA = createFragment(40, 50);
137 		ElementSourceFragment childB = createFragment(50, 55);
138 		ElementSourceFragment childC = createFragment(60, 65);
139 		ElementSourceFragment childD = createFragment(65, 70);
140 		//add all children and check the root is still childWrapper
141 		assertSame(childWrapper, childWrapper.add(childA).add(childB).add(childC).add(childD));
142 		//add childWrapper which has to merge with before added child, because childWrapper is parent of child
143 		rootFragment.add(childWrapper);
144 		assertSame(rootFragment.getFirstChild(), childWrapper);
145 		assertSame(childA, childWrapper.getFirstChild());
146 		assertSame(child, childA.getNextSibling());
147 		assertSame(childB, child.getFirstChild());
148 		assertSame(childC, child.getNextSibling());
149 		assertSame(childD, childC.getNextSibling());
150 	}

214 		//contract: SourceFragments of compilation unit children like, package declaration, imports, types
215 		final Launcher launcher = new Launcher();
216 		launcher.getEnvironment().setNoClasspath(false);
217 		launcher.getEnvironment().setCommentEnabled(true);
218 		SpoonModelBuilder comp = launcher.createCompiler();
219 		comp.addInputSources(SpoonResourceHelper.resources("./src/test/java/" + FooSourceFragments.class.getName().replace('.', '/') + ".java"));
220 		comp.build();
221 		Factory f = comp.getFactory();
222 		
223 		final CtType<?> foo = f.Type().get(FooSourceFragments.class);
224 		CtCompilationUnit compilationUnit = foo.getPosition().getCompilationUnit();
225 		
226 		ElementSourceFragment fragment = compilationUnit.getOriginalSourceFragment();
227 		List<SourceFragment> children = fragment.getChildrenFragments();
228 		assertEquals(11, children.size());
229 		assertEquals("/**\n" + 
230 				" * Javadoc at top of file\n" + 
231 				" */", children.get(0).getSourceCode());
232 		assertEquals("\n", children.get(1).getSourceCode());
233 		assertEquals("/* comment before package declaration*/\n" + 
234 				"package spoon.test.position.testclasses;", children.get(2).getSourceCode());
235 		assertEquals("\n\n", children.get(3).getSourceCode());
236 		assertEquals("/*\n" + 
237 				" * Comment before import\n" + 
238 				" */\n" + 
239 				"import java.lang.Deprecated;", children.get(4).getSourceCode());
240 		assertEquals("\n\n", children.get(5).getSourceCode());
241 		assertEquals("import java.lang.Class;", children.get(6).getSourceCode());
242 		assertEquals("\n\n", children.get(7).getSourceCode());
243 		assertTrue(((ElementSourceFragment) children.get(8)).getElement() instanceof CtClass);
244 		assertStartsWith("/*\n" + 
245 				" * Comment before type\n" + 
246 				" */\n" + 
247 				"public class FooSourceFragments", children.get(8).getSourceCode());
248 		assertEndsWith("//after last type member\n}", children.get(8).getSourceCode());
249 		assertEquals("\n\n", children.get(9).getSourceCode());
250 		assertEquals("//comment at the end of file", children.get(10).getSourceCode());
251 	}

362 	private void checkElementFragments(CtElement ele, Object... expectedFragments) {
363 		ElementSourceFragment fragment = ele.getOriginalSourceFragment();
364 		List<SourceFragment> children = fragment.getChildrenFragments();
365 
366 		// calls getSourceCode on on elements of children
367 		assertEquals(expandGroup(new ArrayList<>(), expectedFragments), toCodeStrings(children));
368 
369 		assertGroupsEqual(expectedFragments, fragment.getGroupedChildrenFragments());
370 	}

372 	private String[] group(String ...str) {
373 		return str;
374 	}
375 
376 	private List<String> expandGroup(List<String> result, Object[] items) {
377 		for (Object object : items) {
378 			if (object instanceof String[]) {
379 				String[] strings = (String[]) object;
380 				expandGroup(result, strings);
381 			} else {
382 				result.add((String) object);
383 			}
384 		}
385 		return result;
386 	}
387 
388 	private static void assertGroupsEqual(Object[] expectedFragments, List<SourceFragment> groupedChildrenFragments) {
389 		assertEquals(Arrays.stream(expectedFragments).map(item->{
390 			if (item instanceof String[]) {
391 				return "group("+Arrays.asList((String[]) item).toString() + ")";
392 			}
393 			return "\"" + item.toString() + "\"";
394 		}).collect(Collectors.joining(";")), groupedChildrenFragments.stream().map(item -> {
395 			if (item instanceof CollectionSourceFragment) {
396 				CollectionSourceFragment csf = (CollectionSourceFragment) item;
397 				return "group("+ toCodeStrings(csf.getItems()).toString() + ")";
398 			}
399 			return "\"" + item.getSourceCode() + "\"";
400 		}).collect(Collectors.joining(";")));
401 	}
402 	
403 	private static List<String> toCodeStrings(List<SourceFragment> csf) {
404 		return csf.stream().map(SourceFragment::getSourceCode).collect(Collectors.toList());
405 	}
