8   import java.lang.annotation.ElementType;
9   import java.lang.annotation.Retention;
10  import java.lang.annotation.RetentionPolicy;
11  import java.lang.annotation.Target;

13  /**
14   * Tells that a metamodel property (ie a getter in metamodel interface) is unsettable because it does not make sense for a certain metamodel element.
15   * This is used when the Java static type system cannot express this contract.
16   * For instance, a catch variable cannot have a default expression.
17   */
18  @Retention(RetentionPolicy.RUNTIME)
19  @Target({ ElementType.METHOD })
20  public @interface UnsettableProperty {
21  }
