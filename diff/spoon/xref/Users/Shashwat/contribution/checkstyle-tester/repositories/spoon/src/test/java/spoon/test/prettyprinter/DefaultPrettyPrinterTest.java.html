199 	public void autoImportUsesFullyQualifiedNameWhenImportedNameAlreadyPresent() {
200 		final Launcher launcher = new Launcher();
201 		final Factory factory = launcher.getFactory();
202 		factory.getEnvironment().setAutoImports(true);
203 		final SpoonModelBuilder compiler = launcher.createCompiler();
204 		compiler.addInputSource(new File("./src/test/java/spoon/test/prettyprinter/testclasses/sub/TypeIdentifierCollision.java"));
205 		compiler.addInputSource(new File("./src/test/java/spoon/test/prettyprinter/testclasses/TypeIdentifierCollision.java"));
206 		compiler.build();
207 
208 		final CtClass<?> aClass = (CtClass<?>) factory.Type().get(spoon.test.prettyprinter.testclasses.TypeIdentifierCollision.class);
209 
210 		String expected =
211 			"public void setFieldUsingExternallyDefinedEnumWithSameNameAsLocal() {" + nl
212 			+ "    localField = TypeIdentifierCollision.ENUM.E1.ordinal();" + nl
213 			+ "}";
214 
215 		String computed = aClass.getMethodsByName("setFieldUsingExternallyDefinedEnumWithSameNameAsLocal").get(0).toString();
216 		assertEquals("We use FQN for E1", expected, computed);
217 
218 		expected = 
219 			"public void setFieldUsingLocallyDefinedEnum() {" + nl
220 			+ "    localField = ENUM.E1.ordinal();" + nl
221 			+ "}";
222 
223 		computed = aClass.getMethodsByName("setFieldUsingLocallyDefinedEnum").get(0).prettyprint();
224 		assertEquals(expected, computed);
225 
226 		expected =
227 			"public void setFieldOfClassWithSameNameAsTheCompilationUnitClass() {" + nl
228 			+ "    TypeIdentifierCollision.globalField = localField;" + nl
229 			+ "}";
230 
231 		computed = aClass.getMethodsByName("setFieldOfClassWithSameNameAsTheCompilationUnitClass").get(0).toString();
232 		assertEquals("The static field of an external type with the same identifier as the compilation unit is printed with FQN", expected, computed);
233 
234 		expected =
235 			"public void referToTwoInnerClassesWithTheSameName() {" + nl
236 			+ "    ClassA.VAR0 = ClassA.getNum();" + nl
237 			+ "    Class1.ClassA.VAR1 = Class1.ClassA.getNum();" + nl
238 			+ "}";
239 
240 		//Ensure the ClassA of Class0 takes precedence over an import statement for ClassA in Class1, and its identifier can be the short version.
241 
242 		computed = aClass.getMethodsByName("referToTwoInnerClassesWithTheSameName").get(0).prettyprint();
243 		assertEquals("where inner types have the same identifier only one may be shortened and the other should be fully qualified", expected, computed);
244 
245 		expected =
246 			"public enum ENUM {" + nl + nl
247 			+ "    E1(TypeIdentifierCollision.globalField, TypeIdentifierCollision.ENUM.E1);" + nl
248 			+ "    final int NUM;" + nl + nl
249 			+ "    final Enum<?> e;" + nl + nl
250 			+ "    private ENUM(int num, Enum<?> e) {" + nl
251 			+ "        NUM = num;" + nl
252 			+ "        this.e = e;" + nl
253 			+ "    }" + nl
254 			+ "}";
255 
256 		computed = aClass.getNestedType("ENUM").toString();
257 		assertEquals(expected, computed);
258 	}

275 	public void printClassCreatedWithSpoon() throws Exception {
276 
277 		/* test that spoon is able to print a class that it created without setting manually the output (default configuration) */
278 
279 		final String nl = System.getProperty("line.separator");
280 
281 		Launcher launcher = new Launcher();
282 		launcher.getEnvironment().setNoClasspath(true);
283 		launcher.getEnvironment().setSourceOutputDirectory(File.createTempFile("foo", "").getParentFile());
284 		launcher.buildModel();
285 		Factory factory = launcher.getFactory();
286 		CtClass<Object> ctClass = factory.Class().create("foo.Bar");
287 		JavaOutputProcessor jop = launcher.createOutputWriter();
288 		jop.setFactory(factory);
289 
290 		jop.createJavaFile(ctClass); //JavaOutputProcessor is able to create the file even if we do not set the cu manually
291 
292 		String pathname = System.getProperty("java.io.tmpdir") + "/foo/Bar.java";
293 		File javaFile = new File(pathname);
294 		assertTrue(javaFile.exists());
295 
296 		assertEquals("package foo;" + nl + "class Bar {}",
297 				IOUtils.toString(new FileInputStream(javaFile), "UTF-8"));

373 	public void testThisConstructorCall2() throws Exception {
374 		// contract: the this(...) call of another constructor is printed well
375 		Launcher launcher = new Launcher();
376 		launcher.getEnvironment().setNoClasspath(true);
377 		launcher.addInputResource("./src/test/resources/noclasspath/ArrayRealVector.java");
378 		CtModel model = launcher.buildModel();
379 		Factory f = launcher.getFactory();
380 
381 		CtClass<?> type = (CtClass) f.Class().get("org.apache.commons.math4.linear.ArrayRealVector");
382 		{
383 			CtConstructor<?> constr = type.getConstructors().stream()
384 					.filter(c -> 
385 						c.getParameters().size() == 1 
386 						&& "ArrayRealVector".equals(c.getParameters().get(0).getType().getSimpleName())
387 					).findFirst().get();
388 			assertEquals("this(v, true)", constr.getBody().getStatement(0).toString());
389 		}
390 		{
391 			String printed = type.toString();
392 			Pattern re = Pattern.compile("public ArrayRealVector\\(org.apache.commons.math4.linear.ArrayRealVector v\\)[^\\{]*\\{\\s*([^;]*)");
393 			Matcher m = re.matcher(printed);
394 			assertTrue(m.find());
395 			assertEquals("this(v, true)", m.group(1));
396 		}
397 	}
