6   package spoon.reflect.visitor.chain;
7   
8   /**
9    * Defines how a {@link CtScannerListener} drives the scanning of {@link spoon.reflect.visitor.EarlyTerminatingScanner}
10   */
11  public enum ScanningMode {
12  	/**
13  	 * Continue with scanning in a normal way, the current element and all children are visited.
14  	 */
15  	NORMAL(true, true),
16  	/**
17  	 * Skip the current element and skip all its children.
18  	 */
19  	SKIP_ALL(false, false),
20  	/**
21  	 * Visit current element but skips all its children.
22  	 */
23  	SKIP_CHILDREN(true, false);
24  
25  	public final boolean visitElement;
26  	public final boolean visitChildren;
27  
28  	ScanningMode(boolean visitElement, boolean visitChildren) {
29  		this.visitElement = visitElement;
30  		this.visitChildren = visitChildren;
31  	}
32  }
