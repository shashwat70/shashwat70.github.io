157 	static class Request {
158 		private static Request myself = new Request();
159 		private static CtElement element;
160 
161 		public static Request on(CtElement ctElement) {
162 			assertNotNull(ctElement);
163 			element = ctElement;
164 			return myself;
165 		}
166 
167 		public <A extends Annotation> CtAnnotation<? extends Annotation> giveMeAnnotation(Class<A> annotation) {
168 			for (CtAnnotation<? extends Annotation> ctAnnotation : element.getAnnotations()) {
169 				if (ctAnnotation.getActualAnnotation().annotationType().equals(annotation)) {
170 					return ctAnnotation;
171 				}
172 			}
173 			fail("Annotation isn't present on the current element.");
174 			return null;
175 		}
176 
177 		public Request giveMeAnnotationValue(String key) {
178 			assertTrue("Element given in the method on should be an CtAnnotation.", element instanceof CtAnnotation);
179 			CtAnnotation<?> ctAnnotation = (CtAnnotation<?>) element;
180 			CtExpression value = null;
181 			try {
182 				value = ctAnnotation.getValue(key);
183 			} catch (ClassCastException e) {
184 				fail("Value of the given key can't be cast to an expression.");
185 			}
186 			assertNotNull(value);
187 			element = value;
188 			return myself;
189 		}
190 
191 		public <T extends CtElement> Request isTypedBy(Class<T> expectedType) {
192 			try {
193 				expectedType.cast(element);
194 			} catch (ClassCastException e) {
195 				fail("The given element can't be cast by the given type.");
196 			}
197 			return myself;
198 		}
199 
200 		public Request isNotAnnotated() {
201 			assertEquals(0, element.getAnnotations().size());
202 			return myself;
203 		}
204 	}
