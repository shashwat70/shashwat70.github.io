6   package spoon.reflect.visitor.filter;
7   
8   /**
9    * This enumeration defines the possible composition operators for filters. It
10   * is used in {@link spoon.reflect.visitor.filter.CompositeFilter}.
11   *
12   * @see spoon.reflect.visitor.Filter
13   */
14  public enum FilteringOperator {
15  
16  	/**
17  	 * Defines the union of several filters: it matches if one of the filters
18  	 * matches.
19  	 */
20  	UNION,
21  	/**
22  	 * Defines the intersection of several filters: it matches if all the
23  	 * filters match.
24  	 */
25  	INTERSECTION,
26  	/**
27  	 * Defines the substraction of several filters to one filter: it matches if
28  	 * the first filter matches and all the others do not match.
29  	 */
30  	SUBSTRACTION
31  
32  }
