13  package spoon.test.loop.testclasses;
14  
15  import java.util.ArrayList;
16  import java.util.Collection;
17  
18  import static java.util.Collections.unmodifiableCollection;
19  
20  
21  /**
22   * Join of two or more Condition.
23   * @param <T> the type of object this condition accepts.
24   *
25   * @author Yvonne Wang
26   * @author Mikhail Mazursky
27   */
28  public abstract class Join<T> extends Condition<T> {
29  
30    final Collection<Condition<? super T>> conditions;
31  
32    /**
33     * Creates a new Join.
34     * @param conditions the conditions to join.
35     * @throws NullPointerException if the given array is {@code null}.
36     * @throws NullPointerException if any of the elements in the given array is {@code null}.
37     */
38    @SafeVarargs
39    protected Join(Condition<? super T>... conditions) {
40      if (conditions == null) throw conditionsIsNull();
41      this.conditions = new ArrayList<>();
42      for (Condition<? super T> condition : conditions)
43        this.conditions.add(notNull(condition));
44    }
45  
46    private static NullPointerException conditionsIsNull() {
47      return new NullPointerException("The given conditions should not be null");
48    }
49  
50    private static <T> Condition<T> notNull(Condition<T> condition) {
51      return null;
52    }
53  
54    /**
55     * Returns the conditions to join.
56     * @return the conditions to join.
57     */
58    protected final Collection<Condition<? super T>> conditions() {
59      return unmodifiableCollection(conditions);
60    }
61  }
