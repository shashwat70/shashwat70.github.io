8   /**
9    * Represents a function, as {@link CtFunction}. However, the main difference is that
10   * while a {@link CtFunction} returns something with a standard Java return keyword,
11   * a {@link CtConsumableFunction} returns something by passing the returned object
12   * as parameter to the given outpuConsumer#accept. This enables to write efficient and concise code in certain situations.
13   * It also enables one to emulate several returns, by simply calling several times accept, while not paying
14   * the code or performance price of creating a list or an iterable object.
15   *
16   * It is typically used as parameter of {@link CtQueryable#map(CtConsumableFunction)}, can be written as one-liners
17   * with Java8 lambdas:.`cls.map((CtClass&lt;?&gt; c, CtConsumer&lt;Object&gt; out)-&gt;out.accept(c.getParent()))`
18   *
19   * @param <T> the type of the input to the function
20   */
21  public interface CtConsumableFunction<T> {
22  	/**
23  	 * Evaluates the function on the given input.
24  	 * @param input the input of the function
25  	 * @param outputConsumer the consumer which accepts the results of this function.
26  	 */
27  	void apply(T input, CtConsumer<Object> outputConsumer);
28  }
