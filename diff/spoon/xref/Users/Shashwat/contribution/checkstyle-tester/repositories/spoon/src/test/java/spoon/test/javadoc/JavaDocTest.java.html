49  	@Test
50  	public void testJavaDocReprint() {
51  		SpoonAPI launcher = new Launcher();
52  		launcher.getEnvironment().setAutoImports(true);
53  		launcher.getEnvironment().setNoClasspath(true);
54  		launcher.getEnvironment().setCommentEnabled(true);
55  		launcher.getEnvironment().setCopyResources(false);
56  		launcher.addInputResource("./src/test/java/spoon/test/javadoc/testclasses/");
57  		launcher.setSourceOutputDirectory("./target/spooned/");
58  		launcher.run();
59  		Factory factory = launcher.getFactory();
60  		CtClass<?> aClass = factory.Class().get(Bar.class);
61  
62  		assertEquals("public class Bar {" + System.lineSeparator()
63  				+ "    /**" + System.lineSeparator()
64  				+ "     * Creates an annotation type." + System.lineSeparator()
65  				+ "     *" + System.lineSeparator()
66  				+ "     * @param owner" + System.lineSeparator()
67  				+ "     * \t\tthe package of the annotation type" + System.lineSeparator()
68  				+ "     * @param simpleName" + System.lineSeparator()
69  				+ "     * \t\tthe name of annotation" + System.lineSeparator()
70  				+ "     */" + System.lineSeparator()
71  				+ "    public <T> CtAnnotationType<?> create(CtPackage owner, String simpleName) {" + System.lineSeparator()
72  				+ "        return null;" + System.lineSeparator()
73  				+ "    }" + System.lineSeparator()
74  				+ "}", ImportTest.printByPrinter(aClass));
75  
76  		// contract: getDocComment never returns null, it returns an empty string if no comment
77  		assertEquals("", aClass.getDocComment());
78  
79  		// contract: getDocComment returns the comment content together with the tag content
80  		CtMethod<?> method = aClass.getMethodsByName("create").get(0);
81  		assertEquals("Creates an annotation type." + System.lineSeparator() + System.lineSeparator()
82  				+ "@param owner" + System.lineSeparator()
83  				+ "\t\tthe package of the annotation type" + System.lineSeparator()
84  				+ "@param simpleName" + System.lineSeparator()
85  				+ "\t\tthe name of annotation" + System.lineSeparator()
86  				, method.getDocComment());
87  
88  		CtJavaDoc ctJavaDoc = method.getComments().get(0).asJavaDoc();
89  		assertEquals(2, ctJavaDoc.getTags().size());
90  
91  		assertEquals(2, ctJavaDoc.clone().getTags().size());
92  		assertCtElementEquals(ctJavaDoc, ctJavaDoc.clone());
93  
94  		// contract: one can parse the content of the javadoc
95  		CtClass<?> bar = factory.Class().get("spoon.test.javadoc.testclasses.Foo");
96  		CtJavaDoc javadoc = (CtJavaDoc) bar.getComments().get(0);
97  		assertEquals(2, javadoc.getJavadocElements().size());
98  		assertEquals("This is a ", javadoc.getJavadocElements().get(0).toText());
99  		JavadocInlineTag link = (JavadocInlineTag) javadoc.getJavadocElements().get(1);
100 		assertEquals("{@link Exception}", link.toText());
101 		assertEquals("link", link.getName());
102 		assertEquals("Exception", link.getContent());
103 
104 
105 	}

143 	public void testTagsParameters() {
144 		// contract: @throws and @exception should have proper params
145 		Launcher launcher = new Launcher();
146 		launcher.addInputResource("./src/test/java/spoon/test/javadoc/testclasses/A.java");
147 		launcher.getEnvironment().setCommentEnabled(true);
148 		launcher.getEnvironment().setNoClasspath(true);
149 		CtModel model = launcher.buildModel();
150 
151 		List<CtJavaDoc> javadocs = model.getElements(new TypeFilter<>(CtJavaDoc.class));
152 
153 		CtJavaDocTag throwsTag = javadocs.get(0).getTags().get(0);
154 		CtJavaDocTag exceptionTag = javadocs.get(1).getTags().get(0);
155 
156 		assertEquals("IllegalArgumentException", throwsTag.getParam());
157 		assertEquals("FileNotFoundException", exceptionTag.getParam());
158 	}

160 	@Test
161 	public void testJavadocTagNames() {
162 		//contract: we should handle all possible javadoc tags properly
163 		Launcher launcher = new Launcher();
164 		launcher.addInputResource("./src/test/java/spoon/test/javadoc/testclasses/B.java");
165 		launcher.getEnvironment().setCommentEnabled(true);
166 		launcher.getEnvironment().setNoClasspath(true);
167 		CtModel model = launcher.buildModel();
168 
169 		CtType<?> type = model.getAllTypes().stream().findFirst().get();
170 		assertEquals(TagType.VERSION, type.getElements(new TypeFilter<>(CtEnum.class)).get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
171 		assertEquals(TagType.AUTHOR, type.getMethodsByName("m1").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
172 		assertEquals(TagType.DEPRECATED, type.getMethodsByName("m2").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
173 		assertEquals(TagType.EXCEPTION, type.getMethodsByName("m3").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
174 		assertEquals(TagType.PARAM, type.getMethodsByName("m4").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
175 		assertEquals(TagType.RETURN, type.getMethodsByName("m5").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
176 		assertEquals(TagType.SEE, type.getMethodsByName("m6").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
177 		assertEquals(TagType.SERIAL, type.getField("m7").getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
178 		assertEquals(TagType.SERIAL_DATA, type.getMethodsByName("m8").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
179 		assertEquals(TagType.SERIAL_FIELD, type.getField("m9").getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
180 		assertEquals(TagType.SINCE, type.getMethodsByName("m10").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
181 		assertEquals(TagType.THROWS, type.getMethodsByName("m11").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
182 		assertEquals(TagType.UNKNOWN, type.getMethodsByName("m12").get(0).getElements(new TypeFilter<>(CtJavaDoc.class)).get(0).getTags().get(0).getType());
183 	}
