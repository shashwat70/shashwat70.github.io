6   package spoon.reflect.path;
7   
8   import spoon.reflect.declaration.CtElement;
9   
10  import java.util.List;
11  
12  /**
13   * A CtPath allows to define the path to a CtElement in the Spoon model, eg ".spoon.test.path.Foo.foo#body#statement[index=0]"
14   */
15  public interface CtPath {
16  
17  	/**
18  	 * Search for elements matching this CtPatch from start nodes given as parameters.
19  	 */
20  	<T extends CtElement> List<T> evaluateOn(CtElement... startNode);
21  
22  	/**
23  	 *
24  	 * Returns the path that is relative to the given element (subpath from it to the end of the path).
25  	 * This is used to have relative paths, instead of absolute path from the root package.
26  	 *
27  	 * For example,
28  	 * "#typeMember[index=2]#body#statement[index=2]#else"
29  	 * is a relative path to the class of absolute path
30  	 * "#subPackage[name=spoon]#subPackage[name=test]#subPackage[name=path]#subPackage[name=testclasses]#containedType[name=Foo]#typeMember[index=2]#body#statement[index=2]#else"
31  	 */
32  	CtPath relativePath(CtElement parent);
33  
34  }
