6   package spoon.reflect.code;
7   
8   import spoon.reflect.reference.CtExecutableReference;
9   import spoon.reflect.annotations.PropertyGetter;
10  import spoon.reflect.annotations.PropertySetter;
11  
12  import static spoon.reflect.path.CtRole.EXECUTABLE_REF;
13  
14  /**
15   * This abstract code element defines an expression which represents an executable reference.
16   *
17   * * Example:
18   * <pre>
19   *     java.util.function.Supplier p =
20   *       Object::new;
21   * </pre>
22   *
23   * @param <T>
24   * 		Each executable references are typed by an interface with one method. This generic type
25   * 		correspond to this concept.
26   * @param <E>
27   * 		Correspond of <code>Type</code> in <code>Type::method</code>.
28   */
29  public interface CtExecutableReferenceExpression<T, E extends CtExpression<?>> extends CtTargetedExpression<T, E> {
30  	/**
31  	 * Gets the executable referenced by the expression.
32  	 */
33  	@PropertyGetter(role = EXECUTABLE_REF)
34  	CtExecutableReference<T> getExecutable();
35  
36  	/**
37  	 * Sets the executable will be referenced by the expression.
38  	 */
39  	@PropertySetter(role = EXECUTABLE_REF)
40  	<C extends CtExecutableReferenceExpression<T, E>> C setExecutable(CtExecutableReference<T> executable);
41  
42  	@Override
43  	CtExecutableReferenceExpression<T, E> clone();
44  }
