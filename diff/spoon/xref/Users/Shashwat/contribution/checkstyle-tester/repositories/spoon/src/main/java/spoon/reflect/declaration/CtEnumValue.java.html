6   package spoon.reflect.declaration;
7   
8   import spoon.reflect.code.CtExpression;
9   import spoon.reflect.code.CtRHSReceiver;
10  import spoon.support.UnsettableProperty;
11  
12  /**
13   * Corresponds to one enum value specified in an enumeration.
14   * If the enum value implicitly calls a constructor (see example below),
15   * it is stored in the default expression of the field as CtConstructorCall,
16   *
17   * <pre>
18   *     class enum {
19   *         ENUM_VALUE("default expression.");
20   *     }
21   * </pre>
22   *
23   * @param <T>
24   * 		the type of the enum, hence equal to the type of getParent().
25   */
26  public interface CtEnumValue<T> extends CtField<T> {
27  	@Override
28  	CtEnumValue clone();
29  
30  	@Override
31  	@UnsettableProperty
32  	<U extends CtRHSReceiver<T>> U setAssignment(CtExpression<T> assignment);
33  }
