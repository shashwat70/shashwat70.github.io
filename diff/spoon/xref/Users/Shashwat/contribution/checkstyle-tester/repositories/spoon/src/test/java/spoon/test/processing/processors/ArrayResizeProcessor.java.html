17  package spoon.test.processing.processors;
18  
19  import spoon.processing.AbstractProcessor;
20  import spoon.reflect.declaration.CtClass;
21  import spoon.reflect.declaration.CtField;
22  import spoon.reflect.reference.CtArrayTypeReference;
23  import spoon.template.Template;
24  import spoon.test.template.ArrayResizeTemplate;
25  
26  public class ArrayResizeProcessor extends AbstractProcessor<CtField<?>> {
27  
28  	public void process(CtField<?> field) {
29  		if ((field.getDeclaringType() instanceof CtClass)
30  				&& field.getType() instanceof CtArrayTypeReference) {
31  			Template<?> t = new ArrayResizeTemplate(field, 10);
32  			t.apply(field.getDeclaringType());
33  		}
34  	}
35  
36  }
