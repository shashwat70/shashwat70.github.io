65  	@Test
66  	public void testVisibilityOfClassesNamedByClassesInJavaLangPackage() {
67  		final File sourceOutputDir = new File("target/spooned/spoon/test/visibility_package/testclasses");
68  		final Launcher launcher = new Launcher();
69  		launcher.getEnvironment().setAutoImports(true);
70  		launcher.getEnvironment().setSourceOutputDirectory(sourceOutputDir);
71  		final Factory factory = launcher.getFactory();
72  		final SpoonModelBuilder compiler = launcher.createCompiler();
73  		compiler.addInputSource(new File("./src/test/java/spoon/test/visibility/testclasses/"));
74  		compiler.build();
75  		compiler.generateProcessedSourceFiles(OutputType.CLASSES);
76  
77  		// Class must be imported.
78  		final CtClass<?> aDouble = (CtClass<?>) factory.Type().get(spoon.test.visibility.testclasses.internal.Double.class);
79  		assertNotNull(aDouble);
80  		assertSame(spoon.test.visibility.testclasses.internal.Double.class, aDouble.getActualClass());
81  
82  		// Class mustn't be imported.
83  		final CtClass<?> aFloat = (CtClass<?>) factory.Type().get(spoon.test.visibility.testclasses.Float.class);
84  		assertNotNull(aFloat);
85  		assertSame(spoon.test.visibility.testclasses.Float.class, aFloat.getActualClass());
86  
87  		canBeBuilt(new File("./target/spooned/spoon/test/visibility_package/testclasses/"), 7);
88  	}

90  	@Test
91  	public void testFullyQualifiedNameOfTypeReferenceWithGeneric() {
92  		// contract: Generics are written when there are specified in the return type of a method.
93  		final String target = "./target/spooned/spoon/test/visibility_generics/testclasses/";
94  		final SpoonAPI launcher = new Launcher();
95  		launcher.addInputResource("./src/test/java/spoon/test/visibility/testclasses/A.java");
96  		launcher.addInputResource("./src/test/java/spoon/test/visibility/testclasses/A2.java");
97  		launcher.addInputResource("./src/test/java/spoon/test/visibility/testclasses/Foo.java");
98  		launcher.setSourceOutputDirectory(target);
99  		launcher.run();
100 
101 		final CtClass<A> aClass = launcher.getFactory().Class().get(A.class);
102 		CtType<?> nestedB = aClass.getNestedType("B");
103 		List<CtFieldAccess> elements = nestedB.getElements(new TypeFilter<>(CtFieldAccess.class));
104 		assertEquals(1, elements.size());
105 		assertEquals("spoon.test.visibility.testclasses.A.B.i", elements.get(0).toString());
106 
107 		CtMethod<?> instanceOf = aClass.getMethodsByName("instanceOf").get(0);
108 		List<CtBinaryOperator> elements1 = instanceOf.getElements(new TypeFilter<>(CtBinaryOperator.class));
109 		assertEquals(1, elements1.size());
110 		assertEquals("spoon.test.visibility.testclasses.A.B", elements1.get(0).getRightHandOperand().toString());
111 
112 		CtMethod<?> returnType = aClass.getMethodsByName("returnType").get(0);
113 		assertEquals("spoon.test.visibility.testclasses.A<T>.C<T>", returnType.getType().toString());
114 
115 		final CtClass<A2> secondClass = launcher.getFactory().Class().get(A2.class);
116 		nestedB = secondClass.getNestedType("B");
117 		elements = nestedB.getElements(new TypeFilter<>(CtFieldAccess.class));
118 		assertEquals(1, elements.size());
119 		assertEquals("spoon.test.visibility.testclasses.A2.B.i", elements.get(0).toString());
120 
121 		instanceOf = secondClass.getMethodsByName("instanceOf").get(0);
122 		elements1 = instanceOf.getElements(new TypeFilter<>(CtBinaryOperator.class));
123 		assertEquals(1, elements1.size());
124 		assertEquals("spoon.test.visibility.testclasses.A2.B", elements1.get(0).getRightHandOperand().toString());
125 
126 		returnType = secondClass.getMethodsByName("returnType").get(0);
127 		assertEquals("spoon.test.visibility.testclasses.A2.C<java.lang.String>", returnType.getType().toString());
128 
129 		returnType = secondClass.getMethodsByName("returnType2").get(0);
130 		assertEquals("spoon.test.visibility.testclasses.Foo<java.lang.String>.Bar<java.lang.String>", returnType.getType().toString());
131 
132 		canBeBuilt(target, 8);
133 	}

135 	@Test
136 	public void testName() {
137 		final SpoonAPI launcher = new Launcher();
138 		launcher.run(new String[] {
139 				"-i", "./src/test/java/spoon/test/visibility/testclasses/Tacos.java",
140 				"-o", "./target/spooned/visibility"
141 		});
142 
143 		final List<CtFieldReference<?>> references = Query.getElements(launcher.getFactory(), new AbstractReferenceFilter<CtFieldReference<?>>(CtFieldReference.class) {
144 			@Override
145 			public boolean matches(CtFieldReference<?> reference) {
146 				return "x".equals(reference.getSimpleName());
147 			}
148 		});
149 		assertEquals(1, references.size());
150 		final CtFieldReference<?> field = references.get(0);
151 		assertNotNull(field.getDeclaration());
152 		final CtClass<?> tacos = launcher.getFactory().Class().get("spoon.test.visibility.testclasses.Tacos");
153 		assertEquals(tacos, field.getDeclaringType().getDeclaration());
154 		assertEquals(tacos.getFields().get(0), field.getDeclaration());
155 	}

157 	@Test
158 	public void testInvocationVisibilityInFieldDeclaration() {
159 		final Launcher launcher = new Launcher();
160 		launcher.getEnvironment().setNoClasspath(true);
161 		launcher.addInputResource("./src/test/resources/noclasspath/Solver.java");
162 		launcher.setSourceOutputDirectory("./target/spooned");
163 		launcher.buildModel();
164 
165 		final CtType<Object> aSolver = launcher.getFactory().Type().get("org.sat4j.minisat.core.Solver");
166 		final CtField<?> lbdTimerField = aSolver.getField("lbdTimer");
167 		final CtInvocation<?> ctInvocation = lbdTimerField.getElements(new TypeFilter<CtInvocation<?>>(CtInvocation.class) {
168 			@Override
169 			public boolean matches(CtInvocation<?> element) {
170 				return "bound".equals(element.getExecutable().getSimpleName()) && super.matches(element);
171 			}
172 		}).get(0);
173 		assertNotNull(ctInvocation.getTarget());
174 		assertTrue(ctInvocation.getTarget().isImplicit());
175 		assertEquals("bound()", ctInvocation.prettyprint());
176 	}
