167 	public void testTypeAccessImplicitIsDerived() throws Exception {
168 		// contract: A CtTypeAccess#implicit is derived
169 		CtType<?> aPozole = ModelUtils.buildClass(Pozole.class);
170 		final CtMethod<?> season = aPozole.getMethodsByName("season").get(0);
171 
172 		CtTypeAccess<?> typeAccesses = season.getElements(new TypeFilter<>(CtTypeAccess.class)).get(0);
173 		assertFalse(typeAccesses.isImplicit());
174 		assertFalse(typeAccesses.getAccessedType().isImplicit());
175 		//contract: setting the value on accessed type influences value on type access too
176 		typeAccesses.getAccessedType().setImplicit(true);
177 		assertTrue(typeAccesses.isImplicit());
178 		assertTrue(typeAccesses.getAccessedType().isImplicit());
179 		//contract: setting the value on type access influences value on type too
180 		typeAccesses.setImplicit(false);
181 		assertFalse(typeAccesses.isImplicit());
182 		assertFalse(typeAccesses.getAccessedType().isImplicit());
183 	}

201 	public void testIntersectionTypeReferenceInGenericsAndCasts() {
202 		final String target = "./target/type";
203 		final Launcher launcher = new Launcher();
204 		launcher.addInputResource("./src/test/java/spoon/test/type/testclasses");
205 		launcher.setSourceOutputDirectory(target);
206 		launcher.getEnvironment().setNoClasspath(true);
207 		launcher.run();
208 
209 		final CtClass<Pozole> aPozole = launcher.getFactory().Class().get(Pozole.class);
210 		final CtMethod<?> prepare = aPozole.getMethodsByName("prepare").get(0);
211 
212 		// Intersection type in generic types.
213 		final List<CtClass> localTypes = prepare.getElements(new TypeFilter<>(CtClass.class));
214 		assertEquals(1, localTypes.size());
215 
216 		// New type parameter declaration.
217 		final CtTypeParameter typeParameter = localTypes.get(0).getFormalCtTypeParameters().get(0);
218 		assertNotNull(typeParameter);
219 		assertEquals("T", typeParameter.getSimpleName());
220 		assertIntersectionTypeForPozolePrepareMethod(aPozole, typeParameter.getSuperclass());
221 
222 		// Intersection type in casts.
223 		final List<CtLambda<?>> lambdas = prepare.getElements(new TypeFilter<>(CtLambda.class));
224 		assertEquals(1, lambdas.size());
225 
226 		assertEquals(1, lambdas.get(0).getTypeCasts().size());
227 		assertTrue(lambdas.get(0).getTypeCasts().get(0) instanceof CtIntersectionTypeReference);
228 		final CtIntersectionTypeReference<?> intersectionType = lambdas.get(0).getTypeCasts().get(0).asCtIntersectionTypeReference();
229 		assertTrue(intersectionType.toString().contains("java.lang.Runnable")
230 			&& intersectionType.toString().contains("java.io.Serializable"));
231 		CtTypeReference refRunnable = aPozole.getFactory().Type().createReference(Runnable.class);
232 		CtTypeReference refSerializable = aPozole.getFactory().Type().createReference(Serializable.class);
233 		CtTypeReference ref0 = intersectionType.getBounds().stream().collect(Collectors.toList()).get(0);
234 		CtTypeReference ref1 = intersectionType.getBounds().stream().collect(Collectors.toList()).get(1);
235 		assertTrue((ref0.equals(refRunnable) || ref0.equals(refSerializable))
236 			&& (ref1.equals(refRunnable) || ref1.equals(refSerializable)));
237 
238 		canBeBuilt(target, 8, true);
239 	}

247 		assertEquals(aPozole.getFactory().Type().createReference(Serializable.class), superType.getBounds().stream().collect(Collectors.toList()).get(1));
