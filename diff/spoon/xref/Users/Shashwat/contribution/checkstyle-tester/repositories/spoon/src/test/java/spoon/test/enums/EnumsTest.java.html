110 	@Test
111 	public void testNestedPrivateEnumValues() throws Exception {
112 		// contract: enum values have correct modifiers
113 		CtType<?> ctClass = ModelUtils.buildClass(NestedEnums.class);
114 		{
115 			CtEnum<?> ctEnum = ctClass.getNestedType("PrivateENUM");
116 			assertEquals(asSet(ModifierKind.PRIVATE), ctEnum.getModifiers());
117 			assertEquals(asSet(ModifierKind.PRIVATE, ModifierKind.STATIC, ModifierKind.FINAL), ctEnum.getField("VALUE").getModifiers());
118 		}
119 		{
120 			CtEnum<?> ctEnum = ctClass.getNestedType("PublicENUM");
121 			assertEquals(asSet(ModifierKind.PUBLIC), ctEnum.getModifiers());
122 			assertEquals(asSet(ModifierKind.PUBLIC, ModifierKind.STATIC, ModifierKind.FINAL), ctEnum.getField("VALUE").getModifiers());
123 		}
124 		{
125 			CtEnum<?> ctEnum = ctClass.getNestedType("ProtectedENUM");
126 			assertEquals(asSet(ModifierKind.PROTECTED), ctEnum.getModifiers());
127 			assertEquals(asSet(ModifierKind.PROTECTED, ModifierKind.STATIC, ModifierKind.FINAL), ctEnum.getField("VALUE").getModifiers());
128 		}
129 		{
130 			CtEnum<?> ctEnum = ctClass.getNestedType("PackageProtectedENUM");
131 			assertEquals(asSet(), ctEnum.getModifiers());
132 			assertEquals(asSet(ModifierKind.STATIC, ModifierKind.FINAL), ctEnum.getField("VALUE").getModifiers());
133 		}

140 	@Test
141 	public void testPrintEnumValues() throws IOException {
142 		// contract: enum values constructor calls are correctly interpreted as implicit or not
143 		Launcher launcher = new Launcher();
144 		launcher.addInputResource("./src/test/java/spoon/test/comment/testclasses/EnumClass.java");
145 		launcher.setSourceOutputDirectory("./target/test-enum");
146 		launcher.getEnvironment().setCommentEnabled(true);
147 		launcher.run();
148 
149 		File file = new File("target/test-enum/spoon/test/comment/testclasses/EnumClass.java");
150 		assertTrue(file.exists());
151 
152 		String content = StringUtils.join(Files.readAllLines(file.toPath()), "\n");
153 
154 		assertTrue(content.contains("FAIL,"));
155 		assertTrue(content.contains("KEEP_OLD_NODE(),"));
156 
157 		assertTrue(content.contains("/**\n"
158 				+ "     * Add new {@link RootNode} after existing nodes\n"
159 				+ "     */\n"
160 				+ "    APPEND"));
161 
162 		assertTrue(content.contains("/**\n"
163 				+ "     * Keep old {@link RootNode} and ignore requests to add new {@link RootNode}\n"
164 				+ "     */\n"
165 				+ "    KEEP_OLD_NODE(),"));
166 	}

168 	@Test
169 	public void testEnumValue() {
170 		// contract: constructorCall on enum values should be implicit if they're not declared
171 
172 		Launcher launcher = new Launcher();
173 		launcher.addInputResource("./src/test/java/spoon/test/comment/testclasses/EnumClass.java");
174 		CtModel model = launcher.buildModel();
175 
176 		List<CtEnumValue> enumValues = model.getElements(new TypeFilter<>(CtEnumValue.class));
177 
178 		assertEquals(4, enumValues.size());
179 
180 		for (int i = 0; i < 3; i++) {
181 			CtEnumValue ctEnumValue = enumValues.get(i);
182 			CtExpression defaultExpression = ctEnumValue.getDefaultExpression();
183 
184 			if (i != 2) {
185 				assertTrue(defaultExpression.isImplicit());
186 			} else {
187 				assertFalse(defaultExpression.isImplicit());
188 			}
189 		}
190 	}

192 	@Test
193 	public void testEnumMembersModifiers() throws Exception {
194 		// contract: enum members should have correct modifiers
195 		final Factory factory = build(EnumWithMembers.class);
196 		CtModel model = factory.getModel();
197 
198 		CtField lenField = model.getElements(new TypeFilter<>(CtField.class)).stream()
199 				.filter(p -> "len".equals(p.getSimpleName()))
200 				.findFirst().get();
201 
202 		assertTrue(lenField.isPrivate());
203 		assertTrue(lenField.isStatic());
204 		assertFalse(lenField.isFinal());
205 		assertFalse(lenField.isPublic());
206 		assertFalse(lenField.isProtected());
207 	}
