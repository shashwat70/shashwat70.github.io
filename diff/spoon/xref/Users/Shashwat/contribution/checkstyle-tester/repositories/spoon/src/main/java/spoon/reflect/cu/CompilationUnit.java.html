6   package spoon.reflect.cu;
7   
8   import spoon.reflect.declaration.CtCompilationUnit;
9   
10  /**
11   * Defines a compilation unit. In Java, a compilation unit can contain only one
12   * public type declaration and other secondary types declarations (not public).
13   */
14  public interface CompilationUnit extends CtCompilationUnit {
15  
16  
17  	/**
18  	 * Helper method to get the begin index of the line that corresponds to the
19  	 * given index.
20  	 *
21  	 * @param index
22  	 * 		an arbitrary index in the source code
23  	 * @return the index where the line starts
24  	 */
25  	@Deprecated
26  	int beginOfLineIndex(int index);
27  
28  	/**
29  	 * Helper method to get the begin index of the line that corresponds to the
30  	 * next line of the given index.
31  	 *
32  	 * @param index
33  	 * 		an arbitrary index in the source code
34  	 * @return the index where the next line starts
35  	 */
36  	@Deprecated
37  	int nextLineIndex(int index);
38  
39  	/**
40  	 * Gets the number of tabulations for a given line.
41  	 *
42  	 * @param index
43  	 * 		the index where the line starts in the source code
44  	 * @return the number of tabs for this line
45  	 */
46  	@Deprecated
47  	int getTabCount(int index);
48  }
