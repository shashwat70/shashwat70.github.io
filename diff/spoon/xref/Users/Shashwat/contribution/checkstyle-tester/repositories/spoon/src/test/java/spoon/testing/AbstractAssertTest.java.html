17  package spoon.testing;
18  
19  import org.junit.Test;
20  import spoon.processing.AbstractProcessor;
21  import spoon.reflect.declaration.CtField;
22  import spoon.reflect.declaration.CtType;
23  import spoon.testing.processors.FooToBarProcessor;
24  
25  import static spoon.testing.Assert.assertThat;
26  import static spoon.testing.utils.ModelUtils.buildNoClasspath;
27  
28  public class AbstractAssertTest {
29  	public static final String PATH = "./src/test/java/spoon/testing/testclasses/";
30  
31  	@Test
32  	public void testTransformationWithProcessorInstantiated() {
33  		assertThat(PATH + "Foo.java").withProcessor(new FooToBarProcessor()).isEqualTo(PATH + "Bar.java");
34  	}
35  
36  	@Test
37  	public void testTransformationWithProcessorClass() {
38  		assertThat(PATH + "Foo.java").withProcessor(FooToBarProcessor.class).isEqualTo(PATH + "Bar.java");
39  	}
40  
41  	@Test
42  	public void testTransformationWithProcessorName() {
43  		assertThat(PATH + "Foo.java").withProcessor(FooToBarProcessor.class.getName()).isEqualTo(PATH + "Bar.java");
44  	}
45  
46  	@Test
47  	public void testTransformationFromCtElementWithProcessor() throws Exception {
48  		class MyProcessor extends AbstractProcessor<CtField<?>> {
49  			@Override
50  			public void process(CtField<?> element) {
51  				element.setSimpleName("j");
52  			}
53  		}
54  		final CtType<CtElementAssertTest> type = buildNoClasspath(CtElementAssertTest.class).Type().get(CtElementAssertTest.class);
55  		assertThat(type.getField("i")).withProcessor(new MyProcessor()).isEqualTo("public int j;");
56  	}
57  }
