6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtLiteral;
10  import spoon.reflect.code.LiteralBase;
11  import spoon.reflect.declaration.CtElement;
12  import spoon.reflect.path.CtRole;
13  import spoon.reflect.visitor.CtVisitor;
14  
15  import static spoon.reflect.path.CtRole.EXPRESSION;
16  import static spoon.reflect.path.CtRole.LITERAL_BASE;
17  
18  public class CtLiteralImpl<T> extends CtExpressionImpl<T> implements CtLiteral<T> {
19  	private static final long serialVersionUID = 1L;
20  
21  	@MetamodelPropertyField(role = CtRole.VALUE)
22  	T value;
23  
24  	@MetamodelPropertyField(role = CtRole.LITERAL_BASE)
25  	LiteralBase base;
26  
27  	@Override
28  	public void accept(CtVisitor visitor) {
29  		visitor.visitCtLiteral(this);
30  	}
31  
32  	@Override
33  	public T getValue() {
34  		return value;
35  	}
36  
37  	@Override
38  	public <C extends CtLiteral<T>> C setValue(T value) {
39  		if (this.value instanceof CtElement) {
40  			((CtElement) this.value).setParent(this);
41  		}
42  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, EXPRESSION, value, this.value);
43  		this.value = value;
44  		return (C) this;
45  	}
46  
47  	@Override
48  	public LiteralBase getBase() {
49  		return base;
50  	}
51  
52  	@Override
53  	public <C extends CtLiteral<T>> C setBase(LiteralBase base) {
54  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, LITERAL_BASE, base, this.base);
55  		this.base = base;
56  		return (C) this;
57  	}
58  
59  	@Override
60  	public CtLiteral<T> clone() {
61  		return (CtLiteral<T>) super.clone();
62  	}
63  }
