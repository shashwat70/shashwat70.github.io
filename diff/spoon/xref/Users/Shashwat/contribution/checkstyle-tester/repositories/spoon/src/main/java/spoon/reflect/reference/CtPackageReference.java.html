6   package spoon.reflect.reference;
7   
8   import spoon.reflect.declaration.CtPackage;
9   import spoon.support.DerivedProperty;
10  
11  /**
12   * This interface defines a reference to a
13   * {@link spoon.reflect.declaration.CtPackage}.
14   */
15  public interface CtPackageReference extends CtReference {
16  	/**
17  	 * Gets the package element when available in the source code.
18  	 */
19  	@Override
20  	@DerivedProperty
21  	CtPackage getDeclaration();
22  
23  	/**
24  	 * Gets the package element when available in the class path.
25  	 */
26  	Package getActualPackage();
27  
28  	/**
29  	 * Returns {@code true} if this is a reference to an <i>unnamed</i>
30  	 * Java package. See JLS ยง7.4.2. Unnamed Packages.
31  	 */
32  	boolean isUnnamedPackage();
33  
34  	@Override
35  	CtPackageReference clone();
36  
37  	/**
38  	 * The simple name of a CtPackageReference is always the fully qualified name of its referenced package. (see {@link spoon.reflect.factory.PackageFactory})
39  	 * @return The fully qualified name of its referenced package
40  	 */
41  	@Override
42  	String getSimpleName();
43  
44  	/**
45  	 * The qualified name of a CtPackageReference is directly given by its simple name (see {@link CtPackageReference#getSimpleName})
46  	 * @return the fully qualified name of its referenced package
47  	 */
48  	String getQualifiedName();
49  }
