6   package spoon.processing;
7   
8   import spoon.compiler.Environment;
9   import spoon.reflect.declaration.CtElement;
10  
11  import java.util.Set;
12  
13  /**
14   * This interface defines a generic code processor. To define a new processor,
15   * the user should subclass {@link spoon.processing.AbstractProcessor}, the
16   * abstract default implementation of this interface.
17   *
18   * If a processor contains fields annotated with @{@link Property}, they can be set using a {@link ProcessorProperties}
19   */
20  public interface Processor<E extends CtElement> extends FactoryAccessor {
21  
22  	/**
23  	 * Gets the model's traversal strategy for this processor (default is
24  	 * {@link TraversalStrategy#POST_ORDER}). Programmers should override this
25  	 * method to return another strategy if needed.
26  	 */
27  	TraversalStrategy getTraversalStrategy();
28  
29  	/**
30  	 * Gets the environment of this processor.
31  	 */
32  	Environment getEnvironment();
33  
34  	/**
35  	 * Tells if this element is to be processed (returns <code>true</code> in
36  	 * the default implementation).
37  	 *
38  	 * @param candidate
39  	 * 		the candidate
40  	 * @return true if the candidate is to be processed by the
41  	 * {@link #process(CtElement)}
42  	 */
43  	boolean isToBeProcessed(E candidate);
44  
45  	/**
46  	 * A callback method upcalled by the meta-model scanner to perform a
47  	 * dedicated job on the currently scanned element. The way Spoon upcalls
48  	 * this method depends on the processed element types (
49  	 * {@link #getProcessedElementTypes()}), the traversal strategy (
50  	 * {@link #getTraversalStrategy()}), and the used processing manager (
51  	 * {@link Environment#getManager()}. Also, this method is upcalled only if
52  	 * the method {@link #isToBeProcessed(CtElement)} returns true for a given
53  	 * scanned element. In order to manually scan the meta-model, one can define
54  	 * the {@link #process()} method instead.
55  	 *
56  	 * @param element
57  	 * 		the element that is currently being scanned
58  	 */
59  	void process(E element);
60  
61  	/**
62  	 * A callback method upcalled by the manager so that this processor can
63  	 * manually implement a processing job. On contrary to
64  	 * {@link #process(CtElement)}, this method does not rely on a built-in
65  	 * meta-model scanner and has to implement its own traversal strategy on the
66  	 * meta-model, which is stored in the factory (
67  	 * {@link FactoryAccessor#getFactory}). Note that if a processor implements
68  	 * both process methods, this one is upcalled first. This method does
69  	 * nothing in default implementations (
70  	 * {@link spoon.processing.AbstractProcessor}).
71  	 */
72  	void process();
73  
74  	/**
75  	 * Do the processing job for a given element. This method is upcalled on an
76  	 * element if the method {@link #isToBeProcessed(CtElement)} returns true.
77  	 *
78  	 * @param element
79  	 *            the element that holds the processed annotations
80  	 */
81  
82  	/**
83  	 * Gets all the element types than need to be processed.
84  	 */
85  	Set<Class<? extends CtElement>> getProcessedElementTypes();
86  
87  	/**
88  	 * This method is upcalled by the {@link ProcessingManager} when this
89  	 * processor has finished a full processing round on the program's model. It
90  	 * is convenient to override this method to tune the application's strategy
91  	 * of a set of processors, for instance by dynamically adding processors to
92  	 * the processing manager when a processing round ends (see
93  	 * {@link ProcessingManager#addProcessor(Class)}). Does nothing by default.
94  	 */
95  	void processingDone();
96  
97  	/**
98  	 * This method is upcalled to initialize the processor before each
99  	 * processing round. It is convenient to override this method rather than
100 	 * using a default constructor to initialize the processor, since the
101 	 * factory is not initialized at construction time. When overriding, do not
102 	 * forget to call super.init() first so that all the initializations
103 	 * performed by superclasses are also applied.
104 	 */
105 	void init();
106 
107 	/**
108 	 * Initializes the properties defined by this processor by using the
109 	 * environment.
110 	 *
111 	 * @see Environment#getProcessorProperties(String)
112 	 */
113 	void initProperties(ProcessorProperties properties);
114 
115 	/**
116 	 * Interrupts the processing of this processor but changes on your AST are kept
117 	 * and the invocation of this method doesn't interrupt the processing of all
118 	 * processors specified in the {@link ProcessingManager}.
119 	 */
120 	void interrupt();
121 }
