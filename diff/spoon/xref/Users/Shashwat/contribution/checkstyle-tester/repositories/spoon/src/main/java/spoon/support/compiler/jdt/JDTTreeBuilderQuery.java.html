32  /** Helper class for JDTTreeBuilder. Package visible to reduce API surface. */
33  class JDTTreeBuilderQuery {
34  	private JDTTreeBuilderQuery() { }
35  	/**
36  	 * Searches a type from an entry-point according to a simple name.
37  	 *
38  	 * @param type
39  	 * 		Entry-point to search.
40  	 * @param simpleName
41  	 * 		Expected type name.
42  	 * @return type binding.
43  	 */
44  	static TypeBinding searchTypeBinding(ReferenceBinding type, String simpleName) {
45  		if (simpleName == null || type == null) {
46  			return null;
47  		}
48  
49  		if (type.memberTypes() != null) {
50  			for (ReferenceBinding memberType : type.memberTypes()) {
51  				if (simpleName.equals(CharOperation.charToString(memberType.sourceName()))) {
52  					return memberType;
53  				}
54  			}
55  		}
56  
57  		return searchTypeBinding(type.superclass(), simpleName);
58  	}
59  
60  	/**
61  	 * Searches a type used in units declared in a compilation unit.
62  	 *
63  	 * @param qualifiedName
64  	 * 		Qualified name of the expected type.
65  	 * @param unitsToProcess
66  	 * 		Search the type expected in units.
67  	 * @return type binding.
68  	 */
69  	static TypeBinding searchTypeBinding(String qualifiedName, CompilationUnitDeclaration[] unitsToProcess) {
70  		if (qualifiedName == null) {
71  			return null;
72  		}
73  		for (CompilationUnitDeclaration unitToProcess : unitsToProcess) {
74  			if (unitToProcess.types != null) {
75  				for (TypeDeclaration type : unitToProcess.types) {
76  					if (qualifiedName.equals(CharOperation.toString(type.binding.compoundName))) {
77  						return type.binding;
78  					}
79  					if (type.memberTypes != null) {
80  						for (TypeDeclaration memberType : type.memberTypes) {
81  							if (qualifiedName.equals(CharOperation.toString(memberType.binding.compoundName))) {
82  								return type.binding;
83  							}
84  						}
85  					}
86  				}
87  			}
88  		}
89  		return null;
90  	}

92  	/**
93  	 * Searches a type declared in imports.
94  	 *
95  	 * @param typeName
96  	 * 		Expected type name.
97  	 * @param imports
98  	 * 		Search the type in imports.
99  	 * @return qualified name of the expected type.
100 	 */
101 	static String searchType(String typeName, ImportReference[] imports) {
102 		if (typeName == null || imports == null) {
103 			return null;
104 		}
105 		for (ImportReference anImport : imports) {
106 			final String importType = CharOperation.charToString(anImport.getImportName()[anImport.getImportName().length - 1]);
107 			if (typeName.equals(importType)) {
108 				return CharOperation.toString(anImport.getImportName());
109 			}
110 		}
111 		return null;
112 	}

114 	/**
115 	 * Searches a package used in units declared in a compilation unit.
116 	 *
117 	 * @param packageName
118 	 * 		Package name.
119 	 * @param unitsToProcess
120 	 * 		Search the package expected in units.
121 	 * @return import reference which correspond to the package expected.
122 	 */
123 	static ImportReference searchPackage(char[][] packageName, CompilationUnitDeclaration[] unitsToProcess) {
124 		for (CompilationUnitDeclaration unit : unitsToProcess) {
125 			final ImportReference currentPackage = unit.currentPackage;
126 			if (currentPackage == null) {
127 				continue;
128 			}
129 			final char[][] tokens = currentPackage.tokens;
130 			if (packageName.length > tokens.length) {
131 				continue;
132 			}
133 			boolean isFound = true;
134 			for (int i = 0; i < packageName.length; i++) {
135 				char[] chars = packageName[i];
136 				if (!CharOperation.equals(chars, tokens[i])) {
137 					isFound = false;
138 					break;
139 				}
140 			}
141 			if (isFound) {
142 				return currentPackage;
143 			}
144 		}
145 		return null;
146 	}

148 	/**
149 	 * Checks in an annotation if a given type is present.
150 	 *
151 	 * @param a
152 	 * 		An annotation.
153 	 * @param elementType
154 	 * 		Expected element type of the annotation.
155 	 * @return true if the annotation is compatible with the given element type.
156 	 */
157 	static boolean hasAnnotationWithType(Annotation a, CtAnnotatedElementType elementType) {
158 		if (a.resolvedType == null) {
159 			return false;
160 		}
161 
162 		// JLS says:
163 		//  "If an annotation of type java.lang.annotation.Target is not present on the declaration of an annotation type T,
164 		// then T is applicable in all declaration contexts except type parameter declarations, and in no type contexts."
165 		boolean shouldTargetAnnotationExists = (elementType == CtAnnotatedElementType.TYPE_USE || elementType == CtAnnotatedElementType.TYPE_PARAMETER);
166 		boolean targetAnnotationExists = false;
167 
168 		for (AnnotationBinding annotation : a.resolvedType.getAnnotations()) {
169 			if (!"Target".equals(CharOperation.charToString(annotation.getAnnotationType().sourceName()))) {
170 				continue;
171 			}
172 			targetAnnotationExists = true;
173 			Object value = annotation.getElementValuePairs()[0].value;
174 			if (value == null) {
175 				continue;
176 			}
177 			if (value instanceof FieldBinding && elementType.name().equals(CharOperation.charToString(((FieldBinding) value).name))) {
178 				return true;
179 			}
180 			if (value.getClass().isArray()) {
181 				Object[] fields = (Object[]) value;
182 				for (Object field : fields) {
183 					if (field instanceof FieldBinding && elementType.name().equals(CharOperation.charToString(((FieldBinding) field).name))) {
184 						return true;
185 					}
186 				}
187 			}
188 		}
189 
190 		// true here means that the target annotation is not mandatory and we have not found it
191 		return !shouldTargetAnnotationExists && !targetAnnotationExists;
192 	}

194 	/**
195 	 * Checks if the qualified name reference is a problem field binding and have a valid field.
196 	 *
197 	 * @param qualifiedNameReference
198 	 * 		Reference which should contains a problem field binding.
199 	 * @return true if the qualified name reference is a valid problem field binding.
200 	 */
201 	static boolean isValidProblemBindingField(QualifiedNameReference qualifiedNameReference) {
202 		return qualifiedNameReference.binding instanceof ProblemFieldBinding && !((FieldBinding) qualifiedNameReference.binding).declaringClass.isAnonymousType()
203 				&& qualifiedNameReference.tokens.length - 1 == ((FieldBinding) qualifiedNameReference.binding).declaringClass.compoundName.length && CharOperation
204 				.equals(CharOperation.subarray(qualifiedNameReference.tokens, 0, qualifiedNameReference.tokens.length - 1),
205 						((FieldBinding) qualifiedNameReference.binding).declaringClass.compoundName);
206 	}

208 	/**
209 	 * Checks if the last node in the stack in the context is an assignment and have a lhs equals to the given expression.
210 	 *
211 	 * @param context
212 	 * 		Context of the {@link JDTTreeBuilder}.
213 	 * @param lhs
214 	 * 		Potential lhs of the assignment.
215 	 * @return true if the lhs is equals to the given expression.
216 	 */
217 	static boolean isLhsAssignment(ContextBuilder context, Expression lhs) {
218 		return context.stack.peek().node instanceof Assignment && ((Assignment) context.stack.peek().node).lhs.equals(lhs);
219 	}

221 	/**
222 	 * Converts the unary operator from JDT to Spoon.
223 	 *
224 	 * @param operator
225 	 * 		Identifier of the unary operator.
226 	 * @return enum value of {@link UnaryOperatorKind}.
227 	 */
228 	static UnaryOperatorKind getUnaryOperator(int operator) {
229 		switch (operator) {
230 		case OperatorIds.PLUS:
231 			return UnaryOperatorKind.POS;
232 		case OperatorIds.MINUS:
233 			return UnaryOperatorKind.NEG;
234 		case OperatorIds.NOT:
235 			return UnaryOperatorKind.NOT;
236 		case OperatorIds.TWIDDLE:
237 			return UnaryOperatorKind.COMPL;
238 		}
239 		return null;
240 	}
241 
242 	/**
243 	 * Converts the binary operator from JDT to Spoon.
244 	 *
245 	 * @param operator
246 	 * 		Identifier of the binary operator.
247 	 * @return enum value of {@link BinaryOperatorKind}.
248 	 */
249 	static BinaryOperatorKind getBinaryOperatorKind(int operator) {
250 		switch (operator) {
251 		case OperatorIds.EQUAL_EQUAL:
252 			return BinaryOperatorKind.EQ;
253 		case OperatorIds.LESS_EQUAL:
254 			return BinaryOperatorKind.LE;
255 		case OperatorIds.GREATER_EQUAL:
256 			return BinaryOperatorKind.GE;
257 		case OperatorIds.NOT_EQUAL:
258 			return BinaryOperatorKind.NE;
259 		case OperatorIds.LEFT_SHIFT:
260 			return BinaryOperatorKind.SL;
261 		case OperatorIds.RIGHT_SHIFT:
262 			return BinaryOperatorKind.SR;
263 		case OperatorIds.UNSIGNED_RIGHT_SHIFT:
264 			return BinaryOperatorKind.USR;
265 		case OperatorIds.OR_OR:
266 			return BinaryOperatorKind.OR;
267 		case OperatorIds.AND_AND:
268 			return BinaryOperatorKind.AND;
269 		case OperatorIds.PLUS:
270 			return BinaryOperatorKind.PLUS;
271 		case OperatorIds.MINUS:
272 			return BinaryOperatorKind.MINUS;
273 		case OperatorIds.NOT:
274 			return BinaryOperatorKind.NE;
275 		case OperatorIds.REMAINDER:
276 			return BinaryOperatorKind.MOD;
277 		case OperatorIds.XOR:
278 			return BinaryOperatorKind.BITXOR;
279 		case OperatorIds.AND:
280 			return BinaryOperatorKind.BITAND;
281 		case OperatorIds.MULTIPLY:
282 			return BinaryOperatorKind.MUL;
283 		case OperatorIds.OR:
284 			return BinaryOperatorKind.BITOR;
285 		case OperatorIds.DIVIDE:
286 			return BinaryOperatorKind.DIV;
287 		case OperatorIds.GREATER:
288 			return BinaryOperatorKind.GT;
289 		case OperatorIds.LESS:
290 			return BinaryOperatorKind.LT;
291 		case OperatorIds.QUESTIONCOLON:
292 			throw new RuntimeException("Unknown operator");
293 		case OperatorIds.EQUAL:
294 			return BinaryOperatorKind.EQ;
295 		}
296 		return null;
297 	}

299 	/**
300 	 * Converts the modifier from JDT to Spoon.
301 	 *
302 	 * @param modifier
303 	 * 		Identifier of the modifier.
304 	 * @param implicit True if the modifier is not explicit in the source code (e.g. a missing 'public' in an interface)
305 	 * @return Set of enum value of {@link CtExtendedModifier}.
306 	 */
307 	static Set<CtExtendedModifier> getModifiers(int modifier, boolean implicit, boolean isMethod) {
308 		Set<CtExtendedModifier> modifiers = new HashSet<>();
309 		if ((modifier & ClassFileConstants.AccPublic) != 0) {
310 			modifiers.add(new CtExtendedModifier(ModifierKind.PUBLIC, implicit));
311 		}
312 		if ((modifier & ClassFileConstants.AccPrivate) != 0) {
313 			modifiers.add(new CtExtendedModifier(ModifierKind.PRIVATE, implicit));
314 		}
315 		if ((modifier & ClassFileConstants.AccProtected) != 0) {
316 			modifiers.add(new CtExtendedModifier(ModifierKind.PROTECTED, implicit));
317 		}
318 		if ((modifier & ClassFileConstants.AccStatic) != 0) {
319 			modifiers.add(new CtExtendedModifier(ModifierKind.STATIC, implicit));
320 		}
321 		if ((modifier & ClassFileConstants.AccFinal) != 0) {
322 			modifiers.add(new CtExtendedModifier(ModifierKind.FINAL, implicit));
323 		}
324 		if ((modifier & ClassFileConstants.AccSynchronized) != 0) {
325 			modifiers.add(new CtExtendedModifier(ModifierKind.SYNCHRONIZED, implicit));
326 		}
327 		if ((modifier & ClassFileConstants.AccVolatile) != 0) {
328 			modifiers.add(new CtExtendedModifier(ModifierKind.VOLATILE, implicit));
329 		}
330 		// a method can never be transient, but it can have the flag because of varArgs.
331 		// source: https://stackoverflow.com/questions/16233910/can-transient-keywords-mark-a-method
332 		if (!isMethod && (modifier & ClassFileConstants.AccTransient) != 0) {
333 			modifiers.add(new CtExtendedModifier(ModifierKind.TRANSIENT, implicit));
334 		}
335 		if ((modifier & ClassFileConstants.AccAbstract) != 0) {
336 			modifiers.add(new CtExtendedModifier(ModifierKind.ABSTRACT, implicit));
337 		}
338 		if ((modifier & ClassFileConstants.AccStrictfp) != 0) {
339 			modifiers.add(new CtExtendedModifier(ModifierKind.STRICTFP, implicit));
340 		}
341 		if ((modifier & ClassFileConstants.AccNative) != 0) {
342 			modifiers.add(new CtExtendedModifier(ModifierKind.NATIVE, implicit));
343 		}
344 		return modifiers;
345 	}
