6   package spoon.support.modelobs;
7   
8   import spoon.reflect.cu.CompilationUnit;
9   import spoon.reflect.cu.position.NoSourcePosition;
10  import spoon.reflect.declaration.CtElement;
11  import spoon.reflect.path.CtRole;
12  import spoon.support.sniper.internal.ElementSourceFragment;
13  
14  /**
15   * A {@link ChangeCollector}, which builds a tree of {@link ElementSourceFragment}s of {@link CompilationUnit} of the modified element
16   * lazily just before the element is changed
17   */
18  public class SourceFragmentCreator extends ChangeCollector {
19  	@Override
20  	protected void onChange(CtElement currentElement, CtRole role) {
21  		if (!currentElement.isParentInitialized()) {
22  			//parent is not initialized. It is just creation of a temporary element
23  			//ignore such "change"
24  			return;
25  		}
26  		CompilationUnit cu = currentElement.getPosition().getCompilationUnit();
27  		if (!(cu instanceof NoSourcePosition.NullCompilationUnit)) {
28  			//getOriginalSourceFragment is not only a getter, it actually
29  			//builds a tree of SourceFragments of compilation unit of the modified element
30  			cu.getOriginalSourceFragment();
31  		}
32  		super.onChange(currentElement, role);
33  	}
34  }
