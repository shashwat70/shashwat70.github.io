17  package spoon.test.template;
18  
19  import static org.junit.Assert.assertEquals;
20  
21  import java.io.File;
22  import java.lang.annotation.ElementType;
23  
24  import org.junit.Test;
25  
26  import spoon.Launcher;
27  import spoon.OutputType;
28  import spoon.reflect.declaration.CtClass;
29  import spoon.reflect.factory.Factory;
30  import spoon.support.compiler.FileSystemFile;
31  import spoon.test.template.testclasses.EnumAccessTemplate;
32  import spoon.testing.utils.ModelUtils;
33  
34  public class TemplateEnumAccessTest {
35  
36  	@Test
37  	public void testEnumAccessTest() {
38  		//contract: the template engine supports enum value access substitution
39  		Launcher launcher = new Launcher();
40  		launcher.addTemplateResource(new FileSystemFile("./src/test/java/spoon/test/template/testclasses/EnumAccessTemplate.java"));
41  
42  		launcher.buildModel();
43  		Factory factory = launcher.getFactory();
44  
45  		CtClass<?> resultKlass = factory.Class().create(factory.Package().getOrCreate("spoon.test.template"), "EnumAccessResult");
46  		new EnumAccessTemplate(ElementType.FIELD, launcher.getFactory()).apply(resultKlass);
47  		assertEquals("java.lang.annotation.ElementType.FIELD.name()", resultKlass.getMethod("method").getBody().getStatement(0).toString());
48  		launcher.setSourceOutputDirectory(new File("./target/spooned/"));
49  		launcher.getModelBuilder().generateProcessedSourceFiles(OutputType.CLASSES);
50  		ModelUtils.canBeBuilt(new File("./target/spooned/spoon/test/template/EnumAccessResult.java"), 8);
51  	}
52  
53  }
