6   package spoon.reflect.code;
7   
8   import spoon.reflect.annotations.PropertyGetter;
9   import spoon.reflect.annotations.PropertySetter;
10  
11  import static spoon.reflect.path.CtRole.ASSIGNED;
12  
13  
14  /**
15   * This code element defines an assignment.
16   *
17   * Example:
18   * <pre>
19   *     int x;
20   *     x = 4; // &lt;-- an assignment
21   * </pre>
22   * @param <T>
23   * 		type of assigned expression
24   * @param <A>
25   * 		type of expression to assign, it should extends &lt;T&gt;
26   */
27  public interface CtAssignment<T, A extends T> extends CtStatement, CtExpression<T>, CtRHSReceiver<A> {
28  	/**
29  	 * Returns the assigned expression on the left-hand side (where the value is stored,
30  	 * e.g. in a variable, in an array, in a field ...).
31  	 */
32  	@PropertyGetter(role = ASSIGNED)
33  	CtExpression<T> getAssigned();
34  
35  	/**
36  	 * Sets the assigned expression (left hand side - LHS).
37  	 */
38  	@PropertySetter(role = ASSIGNED)
39  	<C extends CtAssignment<T, A>> C setAssigned(CtExpression<T> assigned);
40  
41  	@Override
42  	CtAssignment<T, A> clone();
43  }
