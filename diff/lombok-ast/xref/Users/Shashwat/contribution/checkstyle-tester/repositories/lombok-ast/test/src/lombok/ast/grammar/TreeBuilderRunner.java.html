22  package lombok.ast.grammar;
23  
24  import static org.junit.Assert.*;
25  import static lombok.ast.grammar.RunForEachFileInDirRunner.fixLineEndings;
26  
27  import java.io.File;
28  import java.util.Arrays;
29  import java.util.Collection;
30  
31  import lombok.ast.grammar.RunForEachFileInDirRunner.DirDescriptor;
32  
33  abstract class TreeBuilderRunner<N> extends RunForEachFileInDirRunner.SourceFileBasedTester {
34  	private final boolean lombokIsActual;
35  	
36  	protected TreeBuilderRunner(boolean lombokIsActual) {
37  		this.lombokIsActual = lombokIsActual;
38  		
39  	}
40  	
41  	@Override
42  	protected Collection<DirDescriptor> getDirDescriptors() {
43  		return Arrays.asList(
44  				DirDescriptor.of(new File("test/resources/idempotency"), true),
45  				DirDescriptor.of(new File("test/resources/alias"), true),
46  				DirDescriptor.of(new File("test/resources/special"), true));
47  	}
48  	
49  	protected boolean testCompiler(Source source) throws Exception {
50  		N parsedWithTargetCompiler = parseWithTargetCompiler(source);
51  		if (parsedWithTargetCompiler == null) {
52  			// Skip test if target compiler can't compile it.
53  			// A separate test that checks if test samples compile
54  			// at all will do error reporting.
55  			return false;
56  		}
57  		String targetString = convertToString(parsedWithTargetCompiler);
58  		
59  		if (checkForLombokAstParseFailure()) {
60  			source.parseCompilationUnit();
61  			if (!source.getProblems().isEmpty()) {
62  				StringBuilder message = new StringBuilder();
63  				for (ParseProblem p : source.getProblems()) {
64  					message.append(p.toString());
65  					message.append("\n");
66  				}
67  				printDebugInformation(source, targetString, null);
68  				fail(message.toString());
69  			}
70  		}
71  		
72  		String lombokString;
73  		try {
74  			lombokString = fixLineEndings(convertToString(parseWithLombok(source)));
75  		} catch (Exception e) {
76  			printDebugInformation(source, targetString, null);
77  			throw e;
78  		} catch (Error e) {
79  			printDebugInformation(source, targetString, null);
80  			throw e;
81  		}
82  		
83  		try {
84  			if (lombokIsActual) {
85  				assertEquals(targetString, lombokString);
86  			} else {
87  				assertEquals(lombokString, targetString);
88  			}
89  		} catch (AssertionError e) {
90  			if (lombokIsActual) {
91  				printDebugInformation(source, targetString, lombokString);
92  			} else {
93  				printDebugInformation(source, lombokString, targetString);
94  			}
95  			throw e;
96  		}
97  		
98  		return true;
99  	}
100 	
101 	protected boolean checkForLombokAstParseFailure() {
102 		return true;
103 	}
104 	
105 	protected void printDebugInformation(Source source, String targetString, String lombokString) {
106 		String name = source.getName();
107 		System.out.printf("==== Processing %s ====\n", name);
108 		System.out.println(fixLineEndings(source.getRawInput()));
109 		System.out.println("=========== Expected ============");
110 		System.out.println(targetString);
111 		if (lombokString != null) {
112 			System.out.println("============ Actual =============");
113 			System.out.println(lombokString);
114 		}
115 		System.out.printf("======= End of %s =======\n", name);
116 	}
117 	
118 	protected abstract String convertToString(N tree) throws Exception;
119 	
120 	protected abstract N parseWithLombok(Source source) throws Exception;
121 	
122 	protected abstract N parseWithTargetCompiler(Source source) throws Exception;
123 }
