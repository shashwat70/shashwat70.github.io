22  package lombok.ast.grammar;
23  
24  import java.util.List;
25  
26  import lombok.ast.DanglingNodes;
27  import lombok.ast.Identifier;
28  import lombok.ast.Node;
29  import lombok.ast.Position;
30  import lombok.ast.TypeReference;
31  import lombok.ast.TypeReferencePart;
32  import lombok.ast.TypeVariable;
33  import lombok.ast.WildcardKind;
34  
35  public class TypesActions extends SourceActions {
36  	public TypesActions(Source source) {
37  		super(source);
38  	}
39  	
40  	public Node createPrimitiveType(String text) {
41  		Identifier identifier = posify(new Identifier().astValue(text));
42  		TypeReferencePart typeReferencePart = posify(new TypeReferencePart()
43  				.astIdentifier(identifier));
44  		return posify(new TypeReference().rawParts().addToStart(typeReferencePart));
45  	}
46  	
47  	public Node createTypeReferencePart(org.parboiled.Node<Node> identifier, Node typeArguments) {
48  		TypeReferencePart result = new TypeReferencePart().astIdentifier(createIdentifierIfNeeded(identifier.getValue(), currentPos()));
49  		
50  		if (typeArguments instanceof TemporaryNode.TypeArguments) {
51  			for (Node arg : ((TemporaryNode.TypeArguments)typeArguments).arguments) {
52  				result.rawTypeArguments().addToEnd(arg);
53  			}
54  		}
55  		
56  		posify(result); //We only care about the end position here.
57  		return result.setPosition(new Position(identifier.getStartIndex(), result.getPosition().getEnd()));
58  	}
59  	
60  	public Node createWildcardedType(org.parboiled.Node<Node> qmark, org.parboiled.Node<Node> boundType, String extendsOrSuper, Node type) {
61  		if (extendsOrSuper != null) extendsOrSuper = extendsOrSuper.trim();
62  		WildcardKind wildcard = WildcardKind.UNBOUND;
63  		if ("extends".equalsIgnoreCase(extendsOrSuper)) wildcard = WildcardKind.EXTENDS;
64  		if ("super".equalsIgnoreCase(extendsOrSuper)) wildcard = WildcardKind.SUPER;
65  		
66  		TypeReference ref;
67  		
68  		if (!(type instanceof TypeReference)) {
69  			ref = new TypeReference();
70  			DanglingNodes.addDanglingNode(ref, type);
71  		} else {
72  			ref = (TypeReference)type;
73  		}
74  		
75  		ref.astWildcard(wildcard);
76  		source.registerStructure(ref, qmark);
77  		for (org.parboiled.Node<Node> childPNode : boundType.getChildren()) {
78  			if (childPNode != null) source.registerStructure(ref, childPNode);
79  		}
80  		return posify(ref);
81  	}
82  	
83  	public Node createUnboundedWildcardType(org.parboiled.Node<Node> qmark) {
84  		TypeReference ref = new TypeReference().astWildcard(WildcardKind.UNBOUND);
85  		source.registerStructure(ref, qmark);
86  		return posify(ref);
87  	}
88  	
89  	public Node createTypeArguments(Node head, List<Node> tail) {
90  		TemporaryNode.TypeArguments ta = new TemporaryNode.TypeArguments();
91  		if (head != null) ta.arguments.add(head);
92  		if (tail != null) for (Node n : tail) {
93  			if (n != null) ta.arguments.add(n);
94  		}
95  		
96  		return posify(ta);
97  	}
98  	
99  	public Node createReferenceType(Node head, List<Node> tail) {
100 		TypeReference t = new TypeReference();
101 		if (head != null) t.rawParts().addToEnd(head);
102 		if (tail != null) for (Node n : tail) {
103 			if (n != null) t.rawParts().addToEnd(n);
104 		}
105 		
106 		return posify(t);
107 	}
108 	
109 	public Node setArrayDimensionsOfType(Node value, List<String> bracketPairs) {
110 		//TODO test a public int foo() [] {} method declaration.
111 		int arrayDims = bracketPairs == null ? 0 : bracketPairs.size();
112 		if (arrayDims == 0) return value;
113 		
114 		TypeReference ref = new TypeReference().astArrayDimensions(arrayDims);
115 		if (value instanceof TypeReference) {
116 			TypeReference orig = (TypeReference)value;
117 			ref.astWildcard(orig.astWildcard());
118 			ref.rawParts().migrateAllFrom(orig.rawParts());
119 		}
120 		return posify(ref);
121 	}
122 	
123 	public Node createTypeVariable(Node name, Node head, List<Node> tail) {
124 		TypeVariable tv = new TypeVariable().astName(createIdentifierIfNeeded(name, currentPos()));
125 		
126 		if (head != null) tv.rawExtending().addToEnd(head);
127 		if (tail != null) for (Node t : tail) if (t != null) tv.rawExtending().addToEnd(t);
128 		return posify(tv);
129 	}
130 	
131 	public Node createTypeVariables(Node head, List<Node> tail) {
132 		TemporaryNode.OrphanedTypeVariables otv = new TemporaryNode.OrphanedTypeVariables();
133 		if (head != null) otv.variables.add(head);
134 		if (tail != null) for (Node t : tail) otv.variables.add(t);
135 		return posify(otv);
136 	}
