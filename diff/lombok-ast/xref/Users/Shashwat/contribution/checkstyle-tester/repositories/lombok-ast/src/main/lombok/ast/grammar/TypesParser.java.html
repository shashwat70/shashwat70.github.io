22  package lombok.ast.grammar;
23  
24  import lombok.ast.Node;
25  
26  import org.parboiled.BaseParser;
27  import org.parboiled.Rule;
28  import org.parboiled.annotations.SuppressSubnodes;
29  
30  public class TypesParser extends BaseParser<Node> {
31  	final ParserGroup group;
32  	final TypesActions actions;
33  	
34  	public TypesParser(ParserGroup group) {
35  		actions = new TypesActions(group.getSource());
36  		this.group = group;
37  	}
38  	
39  	public Rule nonArrayType() {
40  		return FirstOf(primitiveType(), referenceType());
41  	}
42  	
43  	/**
44  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#4.2">JLS section 4.2</a>
45  	 */
46  	public Rule type() {
47  		return Sequence(
48  				nonArrayType(),
49  				set(),
50  				ZeroOrMore(Sequence(
51  						Ch('['), group.basics.optWS(), Ch(']'), group.basics.optWS())),
52  				set(actions.setArrayDimensionsOfType(value(), texts("ZeroOrMore/Sequence")))).label("type");
53  	}
54  	
55  	/**
56  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#4.2">JLS section 4.2</a>
57  	 */
58  	public Rule primitiveType() {
59  		return Sequence(
60  				rawPrimitiveType(),
61  				set(actions.createPrimitiveType(lastText())),
62  				group.basics.optWS());
63  	}
64  	
65  	@SuppressSubnodes
66  	Rule rawPrimitiveType() {
67  		return Sequence(
68  				FirstOf("boolean", "int", "long", "double", "float", "short", "char", "byte", "void"),
69  				group.basics.testLexBreak());
70  	}
71  	
72  	/**
73  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#4.3">JLS section 4.3</a>
74  	 */
75  	public Rule referenceType() {
76  		return Sequence(
77  				referenceTypePart().label("head"),
78  				ZeroOrMore(dotReferenceTypePart().label("tail")),
79  				set(actions.createReferenceType(value("head"), values("ZeroOrMore/tail"))));
80  	}
81  	
82  	Rule dotReferenceTypePart() {
83  		return Sequence(
84  				Ch('.'), group.basics.optWS(),
85  				group.basics.identifier().label("partName"),
86  				Optional(typeArguments()),
87  				set(actions.createTypeReferencePart(node("partName"), value("Optional/typeArguments"))),
88  				group.basics.optWS());
89  	}
90  	
91  	Rule referenceTypePart() {
92  		return Sequence(
93  				group.basics.identifier().label("partName"),
94  				Optional(typeArguments()),
95  				set(actions.createTypeReferencePart(node("partName"), value("Optional/typeArguments"))),
96  				group.basics.optWS());
97  	}
98  	
99  	public Rule plainReferenceType() {
100 		return Sequence(
101 				plainReferenceTypePart().label("head"),
102 				ZeroOrMore(dotPlainReferenceTypePart().label("tail")),
103 				set(actions.createReferenceType(value("head"), values("ZeroOrMore/tail"))));
104 	}
105 	
106 	Rule plainReferenceTypePart() {
107 		return Sequence(
108 				group.basics.identifier().label("partName"),
109 				set(actions.createTypeReferencePart(node("partName"), null)),
110 				group.basics.optWS());
111 	}
112 	
113 	Rule dotPlainReferenceTypePart() {
114 		return Sequence(
115 				Ch('.'), group.basics.optWS(),
116 				group.basics.identifier().label("partName"),
117 				set(actions.createTypeReferencePart(node("partName"), null)),
118 				group.basics.optWS());
119 	}
120 	
121 	public Rule typeVariables() {
122 		return Optional(Sequence(
123 				Ch('<'),
124 				group.basics.optWS(),
125 				Optional(Sequence(
126 						typeVariable().label("head"),
127 						ZeroOrMore(Sequence(
128 								Ch(','),
129 								group.basics.optWS(),
130 								typeVariable().label("tail"))))),
131 				Ch('>'),
132 				set(actions.createTypeVariables(value("Optional/Sequence/head"), values("Optional/Sequence/ZeroOrMore/Sequence/tail"))),
133 				group.basics.optWS()));
134 	}
135 	
136 	Rule typeVariable() {
137 		return Sequence(
138 				group.basics.identifier(),
139 				Optional(Sequence(
140 						Sequence(
141 							String("extends"),
142 							group.basics.testLexBreak(),
143 							group.basics.optWS()),
144 						type(),
145 						ZeroOrMore(Sequence(
146 								Ch('&'), group.basics.optWS(),
147 								type())))),
148 				set(actions.createTypeVariable(value("identifier"), value("Optional/Sequence/type"), values("Optional/Sequence/ZeroOrMore/Sequence/type"))));
149 	}
150 	
151 	/**
152 	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#4.5">JLS section 4.5</a>
153 	 */
154 	public Rule typeArguments() {
155 		return Optional(Sequence(
156 				Ch('<'),
157 				group.basics.optWS(),
158 				Optional(Sequence(
159 						typeArgument().label("head"),
160 						ZeroOrMore(Sequence(
161 								Ch(','),
162 								group.basics.optWS(),
163 								typeArgument().label("tail"))))),
164 				Ch('>'),
165 				set(actions.createTypeArguments(value("Optional/Sequence/head"), values("Optional/Sequence/ZeroOrMore/Sequence/tail"))),
166 				group.basics.optWS())).label("typeArguments");
167 	}
168 	
169 	public Rule typeArgument() {
170 		return FirstOf(
171 				type(),
172 				Sequence(
173 						Ch('?').label("qmark"),
174 						group.basics.optWS(),
175 						FirstOf(String("extends"), String("super")).label("boundType"),
176 						group.basics.testLexBreak(),
177 						group.basics.optWS(),
178 						type(),
179 						set(actions.createWildcardedType(node("qmark"), node("boundType"), text("boundType"), value("type")))),
180 				Sequence(
181 						Ch('?').label("qmark"),
182 						set(actions.createUnboundedWildcardType(node("qmark"))),
183 						group.basics.optWS()));
184 	}
185 }
