22  package lombok.ast.printer;
23  
24  import lombok.ast.Node;
25  
26  public interface SourceFormatter {
27  	String FAIL = "?!?";
28  	
29  	/**
30  	 * The AST is not valid; it applies to the current level.
31  	 * 
32  	 * @param fail Human readable explanation of what's wrong with the AST at this level.
33  	 */
34  	void fail(String fail);
35  	
36  	void property(String name, Object value);
37  	void keyword(String text);
38  	void operator(String text);
39  	
40  	/**
41  	 * Generate an extra vertical spacer if the output format allows for it. Only occurs between block elements.
42  	 * Example: Between a package statement and the import statements, {@code verticalSpace} will be called.
43  	 */
44  	void verticalSpace();
45  	
46  	/**
47  	 * Generate horizontal space. Where relevant, the space should be 1 character wide.
48  	 */
49  	void space();
50  	
51  	/**
52  	 * Add raw text to the output stream.
53  	 * 
54  	 * NB: Parentheses, braces and brackets that are structurally relevant as opening and/or closing a group are always appended separately; therefore, if you
55  	 * want to do special highlighting on parens that go together, you only have to check if the text appended is a single character long and contains a brace, bracket, or paren.
56  	 * 
57  	 * @param text The raw text to print. If your output format requires it, it is the implementor's responsibility to escape this text.
58  	 */
59  	void append(String text);
60  	
61  	/**
62  	 * Opens a new AST level which is normally rendered inline (example: expressions, literals, modifiers, identifiers, etc).
63  	 * 
64  	 * @param node The AST node that this level represents. Can be {@code null} which signals that the current node has an inline substructure.
65  	 */
66  	void buildInline(Node node);
67  	
68  	/**
69  	 * Closes the previous {@link #buildInline(Node)} call.
70  	 */
71  	void closeInline();
72  	
73  	/**
74  	 * The next {@code buildBlock(Node)} call should <em>NOT</em> be rendered in its own vertical area, but should instead be treated as an {@code inline} element.
75  	 * Example: The execution statement is part of an if statement but has to be a statement, which is normally a block element. This call is generated before
76  	 * the execution statement so that the execution statement is printed inline with the if statement.
77  	 */
78  	void startSuppressBlock();
79  	
80  	/**
81  	 * Closes the previous {@link #startSuppressBlock()} call.
82  	 */
83  	void endSuppressBlock();
84  	
85  	/**
86  	 * The next {@code buildBlock(Node)} call should <em>NOT</em> indent.
87  	 * Example: The {@code case} or {@code default} statement inside a {@code switch} block.
88  	 */
89  	void startSuppressIndent();
90  	
91  	/**
92  	 * Closes the previous {@link #startSuppressIndent()} call.
93  	 */
94  	void endSuppressIndent();
95  	
96  	/**
97  	 * Opens a new AST level which is normally rendered as a block (example: statements, methods, type bodies).
98  	 * 
99  	 * @param node The AST node that this level represents. Can be {@code null} which signals that the current node has a block substructure.
100 	 */
101 	void buildBlock(Node node);
102 	
103 	/**
104 	 * Closes the previous {@link #buildBlock(Node)} call.
105 	 */
106 	void closeBlock();
107 	
108 	/**
109 	 * Registers a parse error with the formatter.
110 	 */
111 	void addError(int errorStart, int errorEnd, String errorMessage);
112 	
113 	/**
114 	 * Generate the source representation and return it as a string.
115 	 */
116 	String finish();
117 	
118 	/**
119 	 * Reports the total time taken in milliseconds by the parser.
120 	 */
121 	void setTimeTaken(long taken);
122 	
123 	/**
124 	 * The next {@link #buildInline(Node)} or {@link #buildBlock(Node)}'s relation to the current block is named by this call.
125 	 * 
126 	 * Not all elements will get a name, and the name only applies to the next {@code buildBlock/Inline} call, not to any further calls.
127 	 */
128 	void nameNextElement(String name);
129 }
