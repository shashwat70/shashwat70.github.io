16  public class Euler71Test {
17  
18      /**
19       * <strong>Problem 71: Ordered fractions</strong>
20       * <p>
21       * Consider the fraction, n/d, where n and d are positive integers. If n<d and HCF(n,d)=1, it is called a reduced proper fraction.
22       * <p>
23       * If we list the set of reduced proper fractions for d ≤ 8 in ascending order of size, we get:
24       * <p>
25       * 1/8, 1/7, 1/6, 1/5, 1/4, 2/7, 1/3, 3/8, 2/5, 3/7, 1/2, 4/7, 3/5, 5/8, 2/3, 5/7, 3/4, 4/5, 5/6, 6/7, 7/8
26       * <p>
27       * It can be seen that 2/5 is the fraction immediately to the left of 3/7.
28       * <p>
29       * By listing the set of reduced proper fractions for d ≤ 1,000,000 in ascending order of size, find the numerator of the fraction immediately to the left of 3/7.
30       * <p>
31       * See also <a href="https://projecteuler.net/problem=71">projecteuler.net problem 71</a>.
32       */
33      @Test
34      public void shouldSolveProblem71() {
35          assertThat(left37(8)).isEqualTo(2);
36          assertThat(left37(1_000_000)).isEqualTo(428_570);
37      }
38  
39      private static int left37(int maxDenominator) {
40          return Stream.iterate(Tuple.of(0, 1, 1, 1), (Tuple4<Integer, Integer, Integer, Integer> t) -> {
41              final int m1 = t._1 + t._3;
42              final int m2 = t._2 + t._4;
43              if (m1 * 7 >= m2 * 3) {
44                  return Tuple.of(t._1, t._2, m1, m2);
45              } else {
46                  return Tuple.of(m1, m2, t._3, t._4);
47              }
48          }).takeWhile(t -> t._2 <= maxDenominator && t._4 <= maxDenominator).last()._1;
49      }
