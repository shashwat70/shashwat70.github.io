7   package io.vavr.collection.euler;

9   import io.vavr.Tuple;
10  import io.vavr.collection.List;
11  import org.junit.Test;
12  
13  import static org.assertj.core.api.Assertions.assertThat;
14  
15  public class Euler09Test {
16  
17      /**
18       * <strong>Problem 9: Special Pythagorean triplet</strong>
19       * <p>
20       * A Pythagorean triplet is a set of three natural numbers, a < b < c, for which,
21       * a<sup>2</sup> + b<sup>2</sup> = c<sup>2</sup>
22       * <p>
23       * For example, 3<sup>2</sup> + 4<sup>2</sup> = 9 + 16 = 25 = 5<sup>2</sup>.
24       * <p>
25       * There exists exactly one Pythagorean triplet for which a + b + c = 1000.
26       * Find the product abc.
27       * <p>
28       * See also <a href="https://projecteuler.net/problem=9">projecteuler.net problem 9</a>.
29       */
30      @Test
31      public void shouldSolveProblem9() {
32          assertThat(abc(1_000)).isEqualTo(31_875_000);
33      }
34  
35      public int abc(int sum) {
36          return List.rangeClosed(1, sum)
37                  .crossProduct()
38                  .filter(t -> t._1 + t._2 < sum)
39                  .map(t -> Tuple.of(t._1, t._2, sum - t._1 - t._2))
40                  .filter(t -> t._1 * t._1 + t._2 * t._2 == t._3 * t._3)
41                  .map(t -> t._1 * t._2 * t._3)
42                  .head();
43      }
44  }
