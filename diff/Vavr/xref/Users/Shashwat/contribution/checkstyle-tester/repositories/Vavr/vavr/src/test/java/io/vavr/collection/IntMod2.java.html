9   /**
10   * An Int wrapper that implements equality by comparing int values modulo 2.
11   * <br>
12   * Examples:
13   * <ul>
14   * <li>IntMod2(0) equals IntMod2(2) equals IntMod2(4) ...</li>
15   * <li>IntMod2(1) equals IntMod2(3) equals IntMod2(5) ...</li>
16   * <li>IntMod2(0) &lt; IntMod2(1)</li>
17   * <li>IntMod2(_even_int_) &lt; IntMod2(_odd_int_)</li>
18   * </ul>
19   */
20  final class IntMod2 implements Comparable<IntMod2> {
21  
22      private final int val;
23  
24      IntMod2(int val) {
25          this.val = val;
26      }
27  
28      @Override
29      public int compareTo(IntMod2 that) {
30          return this.hashCode() - that.hashCode();
31      }
32  
33      @Override
34      public boolean equals(Object that) {
35          return that == this || (that instanceof IntMod2 && this.hashCode() == that.hashCode());
36      }
37  
38      @Override
39      public int hashCode() {
40          return val % 2;
41      }
42  
43      @Override
44      public String toString() {
45          return String.valueOf(val);
46      }
47  
48  }
