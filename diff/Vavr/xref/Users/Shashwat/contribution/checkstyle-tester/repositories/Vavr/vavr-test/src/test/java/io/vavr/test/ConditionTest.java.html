7   package io.vavr.test;
8   
9   import io.vavr.test.Property.Condition;
10  import org.junit.Test;
11  
12  import static org.assertj.core.api.Assertions.assertThat;
13  
14  public class ConditionTest {
15  
16      /**
17       * Def: A 'Condition' is the result of {@code p => q} where {@code p} is a pre-condition and {@code q} is a post-condition.
18       * <p>
19       * The following holds: {@code p => q ≡ ¬p ∨ q}
20       */
21      @Test
22      public void should() {
23          assertThat(cond(false, false)).isTrue();
24          assertThat(cond(false, true)).isTrue();
25          assertThat(cond(true, false)).isFalse();
26          assertThat(cond(true, true)).isTrue();
27      }
28  
29      private boolean cond(boolean p, boolean q) {
30          return !new Condition(p, q).isFalse();
31      }
32  }
