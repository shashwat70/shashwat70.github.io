7   package io.vavr.collection.euler;
8   
9   import io.vavr.collection.Stream;
10  import org.junit.Test;
11  
12  import static io.vavr.collection.euler.Utils.factors;
13  import static org.assertj.core.api.Assertions.assertThat;
14  
15  public class Euler12Test {
16  
17      /**
18       * <strong>Problem 12: Highly divisible triangular number</strong>
19       * <p>
20       * The sequence of triangle numbers is generated by adding the natural
21       * numbers. So the 7th triangle number would be 1 + 2 + 3 + 4 + 5 + 6 + 7 =
22       * 28. The first ten terms would be:
23       * <p>
24       * 1, 3, 6, 10, 15, 21, 28, 36, 45, 55, ...
25       * <p>
26       * Let us list the factors of the first seven triangle numbers:
27       * <pre>
28       * 1: 1
29       * 3: 1,3
30       * 6: 1,2,3,6
31       * 10: 1,2,5,10
32       * 15: 1,3,5,15
33       * 21: 1,3,7,21
34       * 28: 1,2,4,7,14,28
35       * </pre>
36       * <p>
37       * We can see that 28 is the first triangle number to have over five
38       * divisors.
39       * <p>
40       * What is the value of the first triangle number to have over five hundred
41       * divisors?
42       */
43      @Test
44      public void shouldSolveProblem12() {
45          assertThat(valueOfFirstTriangleNumberWithMoreDivisorsThan(5)).isEqualTo(28);
46          assertThat(valueOfFirstTriangleNumberWithMoreDivisorsThan(500)).isEqualTo(76_576_500);
47      }
48  
49      private static long valueOfFirstTriangleNumberWithMoreDivisorsThan(long divisorCount) {
50          return triangleNumbers()
51                  .find(t -> divisorCount(t) > divisorCount)
52                  .get();
53      }
54  
55      private static Stream<Long> triangleNumbers() {
56          return Stream.from(1L).scanLeft(0L, (a, l) -> a + l);
57      }
58  
59      private static long divisorCount(long number) {
60          return factors(number).length();
61      }
62  }
