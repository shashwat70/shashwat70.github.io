7   package io.vavr.collection.euler;

9   import io.vavr.collection.CharSeq;
10  import io.vavr.collection.List;
11  import org.junit.Test;
12  
13  import static io.vavr.API.*;
14  import static org.assertj.core.api.Assertions.assertThat;
15  
16  public class Euler38Test {
17  
18      /**
19       * <strong>Problem 38 Pandigital multiples</strong>
20       * <p>
21       * Take the number 192 and multiply it by each of 1, 2, and 3:
22       * <pre>
23       * 192 × 1 = 192
24       * 192 × 2 = 384
25       * 192 × 3 = 576
26       * </pre>
27       * <p>
28       * By concatenating each product we get the 1 to 9 pandigital, 192384576. We
29       * will call 192384576 the concatenated product of 192 and (1,2,3)
30       * <p>
31       * The same can be achieved by starting with 9 and multiplying by 1, 2, 3,
32       * 4, and 5, giving the pandigital, 918273645, which is the concatenated
33       * product of 9 and (1,2,3,4,5).
34       * <p>
35       * What is the largest 1 to 9 pandigital 9-digit number that can be formed
36       * as the concatenated product of an integer with (1,2, ... , n) where n >
37       * 1?
38       * <p>
39       * See also <a href="https://projecteuler.net/problem=38">projecteuler.net
40       * problem 38</a>.
41       */
42      @Test
43      public void shouldSolveProblem38() {
44          assertThat(isPandigitalMultiple(CharSeq.of("192384576"))).isTrue();
45          assertThat(isPandigitalMultiple(CharSeq.of("918273645"))).isTrue();
46  
47          assertThat(largets1To9PandigitalMultiple().mkString()).isEqualTo("932718654");
48      }
49  
50      private static CharSeq largets1To9PandigitalMultiple() {
51          return CharSeq.of("87654321")
52                  .permutations()
53                  .map(CharSeq::mkString)
54                  .map(Integer::valueOf)
55                  .sorted()
56                  .reverse()
57                  .map(i -> "9" + i) // Since 918273645 is known we don't have to investigate numbers not starting with a 9.
58                  .map(CharSeq::of)
59                  .find(Euler38Test::isPandigitalMultiple)
60                  .get();
61      }
62  
63      private static boolean isPandigitalMultiple(CharSeq pandigital) {
64          return List.rangeClosed(1, pandigital.length() - 1)
65                  .exists(i -> isPandigitalMultipleRest(pandigital.drop(i), Integer.valueOf(pandigital.take(i).mkString()), 2));
66      }
67  
68      private static boolean isPandigitalMultipleRest(CharSeq pandigitalRest, int multiplicand, int multiplicator) {
69          return Match(pandigitalRest.length()).of(
70                  Case($(0), true),
71                  Case($(), length -> List.rangeClosed(1, length)
72                          .find(i -> Integer.valueOf(pandigitalRest.take(i).mkString()) == multiplicand * multiplicator)
73                          .map(i -> isPandigitalMultipleRest(pandigitalRest.drop(i), multiplicand, multiplicator + 1))
74                          .getOrElse(false)
75                  )
76          );
77      }
