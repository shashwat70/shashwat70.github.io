7   package io.vavr;
8   
9   import java.util.NoSuchElementException;
10  
11  /**
12   * A {@link API.Match} throws a MatchError if no case matches the applied object.
13   *
14   * @author Daniel Dietrich
15   */
16  public class MatchError extends NoSuchElementException {
17  
18      private static final long serialVersionUID = 1L;
19  
20      private final Object obj;
21  
22      /**
23       * Internally called by {@link API.Match}.
24       *
25       * @param obj The object which could not be matched.
26       */
27      MatchError(Object obj) {
28          super((obj == null) ? "null" : "type: " + obj.getClass().getName() + ", value: " + obj);
29          this.obj = obj;
30      }
31  
32      /**
33       * Returns the object which could not be matched.
34       *
35       * @return An Object.
36       */
37      public Object getObject() {
38          return obj;
39      }
40  
41  }
