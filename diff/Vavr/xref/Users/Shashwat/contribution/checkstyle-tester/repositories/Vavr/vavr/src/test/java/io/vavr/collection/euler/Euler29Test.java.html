7   package io.vavr.collection.euler;
8   
9   import io.vavr.collection.Stream;
10  import org.junit.Test;
11  
12  import java.math.BigInteger;
13  
14  import static org.assertj.core.api.Assertions.assertThat;
15  
16  public class Euler29Test {
17  
18      /**
19       * <strong>Problem 29: Distinct powers</strong>
20       * <p>
21       * Consider all integer combinations of ab for 2 ≤ a ≤ 5 and 2 ≤ b ≤ 5:
22       * <ul>
23       * <li>2<sup>2</sup>=4, 2<sup>3</sup>=8, 2<sup>4</sup>=16, 2<sup>5</sup>=32</li>
24       * <li>3<sup>2</sup>=9, 3<sup>3</sup>=27, 3<sup>4</sup>=81, 3<sup>5</sup>=243</li>
25       * <li>4<sup>2</sup>=16, 4<sup>3</sup>=64, 4<sup>4</sup>=256, 4<sup>5</sup>=1024</li>
26       * <li>5<sup>2</sup>=25, 5<sup>3</sup>=125, 5<sup>4</sup>=625, 5<sup>5</sup>=3125</li>
27       * </ul>
28       * If they are then placed in numerical order, with any repeats removed, we get the following sequence of 15 distinct terms:
29       * <p>
30       * 4, 8, 9, 16, 25, 27, 32, 64, 81, 125, 243, 256, 625, 1024, 3125
31       * <p>
32       * How many distinct terms are in the sequence generated by ab for 2 ≤ a ≤ 100 and 2 ≤ b ≤ 100?
33       * <p>
34       * See also <a href="https://projecteuler.net/problem=29">projecteuler.net problem 29</a>.
35       */
36      @Test
37      public void shouldSolveProblem29() {
38          assertThat(cnt(5)).isEqualTo(15);
39          assertThat(cnt(100)).isEqualTo(9183);
40      }
41  
42      private static int cnt(int max) {
43          return Stream.rangeClosed(2, max)
44                  .map(BigInteger::valueOf)
45                  .flatMap(a -> Stream.rangeClosed(2, max).map(a::pow))
46                  .distinct()
47                  .length();
48      }
