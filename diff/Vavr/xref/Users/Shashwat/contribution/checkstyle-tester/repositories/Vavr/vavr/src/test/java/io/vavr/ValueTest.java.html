7   package io.vavr;
8   
9   import io.vavr.collection.List;
10  import io.vavr.control.Option;
11  import org.junit.Test;
12  
13  import java.util.ArrayList;
14  import java.util.stream.Collectors;
15  
16  import static org.assertj.core.api.Assertions.assertThat;
17  
18  public class ValueTest {
19  
20      @Test
21      public void shouldNarrowValue() {
22          final Value<Double> doubles = List.of(1.0d);
23          final Value<Number> numbers = Value.narrow(doubles);
24          assertThat(numbers.get()).isEqualTo(1.0d);
25      }
26  
27      @Test
28      public void collectWorkAsExpectedMultiValue() {
29          final Value<Double> doubles = List.of(1.0d, 2.0d);
30          final java.util.List<Double> result = doubles.collect(Collectors.toList());
31          assertThat(result).contains(1.0d, 2.0d);
32      }
33  
34      @Test
35      public void verboseCollectWorkAsExpectedMultiValue() {
36          final Value<Double> doubles = List.of(1.0d, 2.0d);
37          final java.util.List<Double> result = doubles.collect(ArrayList<Double>::new, ArrayList::add, ArrayList::addAll);
38          assertThat(result).contains(1.0d, 2.0d);
39      }
40  
41      @Test
42      public void collectWorkAsExpectedSingleValue() {
43          final Value<Double> doubles = Option.of(1.0d);
44          assertThat(doubles.collect(Collectors.toList()).get(0)).isEqualTo(1.0d);
45      }
46  
47      @Test
48      public void verboseCollectWorkAsExpectedSingleValue() {
49          final Value<Double> doubles = Option.of(1.0d);
50          assertThat(doubles.collect(ArrayList<Double>::new,
51                  ArrayList::add, ArrayList::addAll).get(0)).isEqualTo(1.0d);
52      }
53  }
