23  /**
24   * <strong>Problem 26: Reciprocal cycles</strong>
25   * <p>
26   * A unit fraction contains 1 in the numerator. The decimal representation of
27   * the unit fractions with denominators 2 to 10 are given:
28   * <pre>
29   * 1/2	= 0.5
30   * 1/3	= 0.(3)
31   * 1/4	= 0.25
32   * 1/5	= 0.2
33   * 1/6	= 0.1(6)
34   * 1/7	= 0.(142857)
35   * 1/8	= 0.125
36   * 1/9	= 0.(1)
37   * 1/10	= 0.1
38   * </pre> Where 0.1(6) means 0.166666..., and has a 1-digit recurring cycle. It
39   * can be seen that 1/7 has a 6-digit recurring cycle.
40   * <p>
41   * Find the value of d < 1000 for which 1/d contains the longest recurring cycle
42   * in its decimal fraction part. <p>
43   * See also
44   * <a href="https://projecteuler.net/problem=26">projecteuler.net problem 26
45   * </a>.
46   */
47  public class Euler26Test {
48  
49      @Test
50      public void shouldSolveProblem26() {
51          assertThat(recurringCycleLengthForDivisionOf1(2)._2).isEqualTo(0);
52          assertThat(recurringCycleLengthForDivisionOf1(3)._2).isEqualTo(1);
53          assertThat(recurringCycleLengthForDivisionOf1(4)._2).isEqualTo(0);
54          assertThat(recurringCycleLengthForDivisionOf1(5)._2).isEqualTo(0);
55          assertThat(recurringCycleLengthForDivisionOf1(6)._2).isEqualTo(1);
56          assertThat(recurringCycleLengthForDivisionOf1(7)._2).isEqualTo(6);
57          assertThat(recurringCycleLengthForDivisionOf1(8)._2).isEqualTo(0);
58          assertThat(recurringCycleLengthForDivisionOf1(9)._2).isEqualTo(1);
59          assertThat(recurringCycleLengthForDivisionOf1(10)._2).isEqualTo(0);
60          assertThat(deonominatorBelow1000WithTheLongetsRecurringCycleOfDecimalFractions()).isEqualTo(983);
61      }
62  
63      private static int deonominatorBelow1000WithTheLongetsRecurringCycleOfDecimalFractions() {
64          return List.range(2, 1000)
65                  .map(Euler26Test::recurringCycleLengthForDivisionOf1)
66                  .maxBy(Tuple2::_2)
67                  .get()._1;
68      }
69  
70      private static Tuple2<Integer, Integer> recurringCycleLengthForDivisionOf1(int divisor) {
71          return Tuple.of(
72                  divisor,
73                  recurringCycleLengthInDecimalFractionPart(
74                          CharSeq.of(BigDecimal.ONE.divide(BigDecimal.valueOf(divisor), 2000, RoundingMode.UP).toString())
75                                  .transform(removeLeadingZeroAndDecimalPoint())
76                                  .transform(removeRoundingDigit())
77                                  .transform(removeTrailingZeroes())
78                                  .mkString()
79                  ));
80      }
81  
82      private static int recurringCycleLengthInDecimalFractionPart(String decimalFractionPart) {
83          return CharSeq.of(decimalFractionPart)
84                  .reverse()
85                  .toStream() // Stream is lazy evaluated which ensures the rest is only evaluated until the recurring cycle is found.
86                  .transform(createCandidateCycles())
87                  .transform(removeCandidatesLongerThanHalfTheFullString(decimalFractionPart))
88                  .transform(findFirstRecurringCycle(decimalFractionPart))
89                  .map(String::length)
90                  .getOrElse(0);
91      }
92  
93      private static Function1<CharSeq, CharSeq> removeLeadingZeroAndDecimalPoint() {
94          return seq -> seq.drop(2);
95      }
96  
97      private static Function1<CharSeq, CharSeq> removeRoundingDigit() {
98          return seq -> seq.dropRight(1);
99      }
100 
101     private static Function1<CharSeq, CharSeq> removeTrailingZeroes() {
102         return seq -> seq
103                 .reverse()
104                 .dropWhile(c -> c == '0') //Remove any trailing zeroes
105                 .reverse();
106     }
107 
108     private static Function1<Stream<Character>, Stream<String>> createCandidateCycles() {
109         return reversedDecimalFractionPart -> reversedDecimalFractionPart
110                 .map(String::valueOf)
111                 .scan("", String::concat)
112                 .drop(1); // Drop the first emtpy string created by scan
113     }
114 
115     private static Function1<Stream<String>, Stream<String>> removeCandidatesLongerThanHalfTheFullString(String decimalFractionPart) {
116         return candidateCycles -> candidateCycles.filter(candidate -> decimalFractionPart.length() >= candidate.length() * 2);
117     }
118 
119     private static Function1<Stream<String>, Option<String>> findFirstRecurringCycle(String decimalFractionPart) {
120         return reversedCandidateCycles -> reversedCandidateCycles
121                 .map(s -> CharSeq.of(s).reverse().mkString())
122                 .find(candidate -> candidate.equals(decimalFractionPart.substring(decimalFractionPart.length() - (candidate.length() * 2), decimalFractionPart.length() - candidate.length())));
123     }
