7   package io.vavr.collection.euler;

9   import org.junit.Test;
10  
11  import static java.util.Arrays.asList;
12  import static io.vavr.collection.euler.Utils.file;
13  import static io.vavr.collection.euler.Utils.readLines;
14  import static org.assertj.core.api.Assertions.assertThat;
15  
16  public class Euler99Test {
17  
18      /**
19       * <strong>Problem 99: Largest exponential</strong>
20       * <p>
21       * Comparing two numbers written in index form like 2<sup>11</sup> and 3<sup>7</sup> is not difficult,
22       * as any calculator would confirm that 2<sup>11</sup> = 2048 &lt; 3<sup>7</sup> = 2187.
23       * <p>
24       * However, confirming that 632382<sup>518061</sup> &gt; 519432<sup>525806</sup> would be much more difficult,
25       * as both numbers contain over three million digits.
26       * <p>
27       * Using p099_base_exp.txt, a 22K text file containing one thousand lines with a base/exponent pair on each line,
28       * determine which line number has the greatest numerical value.
29       * <p>
30       * See also <a href="https://projecteuler.net/problem=99">projecteuler.net problem 99</a>.
31       */
32      @Test
33      public void shouldSolveProblem99() {
34          assertThat(solve()).isEqualTo(709);
35      }
36  
37      private static long solve() {
38          return readLines(file("p099_base_exp.txt"))
39                  .flatMap(s -> asList(s.split(",")))
40                  .map(Integer::parseInt)
41                  .grouped(2)
42                  .map(t -> t.get(1) * Math.log(t.get(0)))
43                  .zipWithIndex()
44                  .reduce((t1, t2) -> t1._1 > t2._1 ? t1 : t2)
45                  ._2 + 1;
46      }
47  }
