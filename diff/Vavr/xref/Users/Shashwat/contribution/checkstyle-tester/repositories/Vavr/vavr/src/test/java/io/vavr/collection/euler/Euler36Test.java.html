7   package io.vavr.collection.euler;

9   import io.vavr.collection.CharSeq;
10  import io.vavr.collection.Stream;
11  import org.junit.Test;
12  
13  import static org.assertj.core.api.Assertions.assertThat;
14  
15  /**
16   * <strong>Problem 36: Double-base palindromes</strong>
17   * <p>The decimal number, 585 = 1001001001<sub>2</sub> (binary), is palindromic in both bases.</p>
18   * <p>Find the sum of all numbers, less than one million, which are palindromic in base 10 and base 2.</p>
19   * <p class="info">(Please note that the palindromic number, in either base, may not include leading zeros.)</p>
20   * See also <a href="https://projecteuler.net/problem=36">projecteuler.net problem 36</a>.
21   */
22  public class Euler36Test {
23  
24      @Test
25      public void shouldSolveProblem36() {
26          assertThat(solve(1000000)).isEqualTo(872187);
27      }
28  
29      private static int solve(int n) {
30          return Stream.range(1, n)
31                  .filter(Euler36Test::isDoubleBasePalindrome)
32                  .sum().intValue();
33      }
34  
35      private static boolean isPalindrome(CharSeq seq) {
36          return seq.dropWhile(c -> c == '0').equals(seq.reverse().dropWhile(c -> c == '0'));
37      }
38  
39      private static boolean isDoubleBasePalindrome(int x) {
40          final CharSeq seq = CharSeq.of(Integer.toString(x));
41          final CharSeq rev = CharSeq.of(Integer.toBinaryString(x));
42          return isPalindrome(seq) && isPalindrome(rev);
43      }

45  }
