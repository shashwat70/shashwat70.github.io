7   package io.vavr.collection.euler;
8   
9   import io.vavr.collection.Stream;
10  import org.junit.Test;
11  
12  import static org.assertj.core.api.Assertions.assertThat;
13  
14  public class Euler48Test {
15  
16      private static final long MOD = 10_000_000_000L;
17  
18      /**
19       * <strong>Problem 48: Self powers</strong>
20       * <p>
21       * The series, 1<sup>1</sup> + 2<sup>2</sup> + 3<sup>3</sup> + ... + 10<sup>10</sup> = 10405071317.
22       * <p>
23       * Find the last ten digits of the series, 1<sup>1</sup> + 2<sup>2</sup> + 3<sup>3</sup> + ... + 1000<sup>1000</sup>.
24       * <p>
25       * See also <a href="https://projecteuler.net/problem=48">projecteuler.net problem 48</a>.
26       */
27      @Test
28      public void shouldSolveProblem48() {
29          assertThat(sumPowers(10)).isEqualTo(405_071_317L);
30          assertThat(sumPowers(1000)).isEqualTo(9_110_846_700L);
31      }
32  
33      private static long sumPowers(int max) {
34          return Stream.range(1, max)
35                  .map(Euler48Test::selfPower)
36                  .reduce(Euler48Test::sumMod);
37      }
38  
39      private static long selfPower(long v) {
40          final Stream<Long> powers = Stream.iterate(v, el -> multMod(el, el));
41          return bits(v)
42                  .map(powers::get)
43                  .prepend(1L)
44                  .reduce(Euler48Test::multMod);
45      }
46  
47      private static long multMod(long v1, long v2) {
48          final Stream<Long> shifts = Stream.iterate(v1, el -> sumMod(el, el));
49          return bits(v2)
50                  .map(shifts::get)
51                  .prepend(0L)
52                  .reduce(Euler48Test::sumMod);
53      }
54  
55      private static long sumMod(long v1, long v2) {
56          return (v1 + v2) % MOD;
57      }
58  
59      private static Stream<Integer> bits(long v) {
60          return Stream.from(0)
61                  .takeWhile(b -> (v >> b) > 0)
62                  .filter(b -> ((v >> b) & 1) != 0);
63      }
64  
65  }
