14  /**
15   * <strong>Problem 31: Coin sums</strong>
16   * <p>In England the currency is made up of pound, £, and pence, p, and there are eight coins in general circulation:</p>
17   * <blockquote>1p, 2p, 5p, 10p, 20p, 50p, £1 (100p) and £2 (200p).</blockquote>
18   * <p>It is possible to make £2 in the following way:</p>
19   * <blockquote>1×£1 + 1×50p + 2×20p + 1×5p + 1×2p + 3×1p</blockquote>
20   * <p>How many different ways can £2 be made using any number of coins?</p>
21   * See also <a href="https://projecteuler.net/problem=31">projecteuler.net problem 31</a>.
22   */
23  public class Euler31Test {
24  
25      @Test
26      public void shouldSolveProblem31() {
27          final List<Integer> coins = List.of(1, 2, 5, 10, 20, 50, 100, 200);
28          assertThat(coinSums(200, coins)).isEqualTo(73682);
29      }
30  
31      private static int coinSums(int n, List<Integer> coins) {
32          return (n == 0) ? 1 :
33                 (n < 0 || coins.isEmpty()) ? 0 :
34                 coinSums(n, coins.tail()) + coinSums(n - coins.head(), coins);
35      }
