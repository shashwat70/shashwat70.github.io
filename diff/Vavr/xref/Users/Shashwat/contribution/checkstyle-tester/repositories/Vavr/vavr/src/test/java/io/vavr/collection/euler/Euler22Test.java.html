17  /**
18   * <strong>Problem 22: Names scores</strong>
19   * <p>
20   * Using names.txt (right click and 'Save Link/Target As...'), a 46K text file
21   * containing over five-thousand first names, begin by sorting it into
22   * alphabetical order. Then working out the alphabetical value for each name,
23   * multiply this value by its alphabetical position in the list to obtain a name
24   * score.
25   * <p>
26   * For example, when the list is sorted into alphabetical order, COLIN, which is
27   * worth 3 + 15 + 12 + 9 + 14 = 53, is the 938th name in the list. So, COLIN
28   * would obtain a score of 938 × 53 = 49714.
29   * <p>
30   * What is the total of all the name scores in the file?
31   * <p>
32   * See also <a href="https://projecteuler.net/problem=22">projecteuler.net
33   * problem 22</a>.
34   */
35  public class Euler22Test {
36  
37      @Test
38      public void shouldSolveProblem22() {
39          assertThat(nameScore("COLIN", 938)).isEqualTo(49714);
40          assertThat(totalOfAllNameScores()).isEqualTo(871_198_282);
41      }
42  
43      private static long nameScore(String name, long position) {
44          return CharSeq.of(name)
45                  .map(c -> c - 'A' + 1)
46                  .sum().longValue() * position;
47      }
48  
49      private static long totalOfAllNameScores() {
50          return readLines(file("p022_names.txt"))
51                  .map(l -> l.replaceAll("\"", ""))
52                  .flatMap(l -> Stream.of(l.split(",")))
53                  .sorted()
54                  .zipWithIndex()
55                  .map(t -> nameScore(t._1, t._2 + 1))
56                  .sum().longValue();
57      }
