7   package io.vavr.collection.euler;
8   
9   import io.vavr.collection.Stream;
10  import io.vavr.collection.List;
11  import org.junit.Test;
12  
13  import static java.util.Comparator.reverseOrder;
14  import static io.vavr.collection.euler.Utils.isPrime;
15  import static org.assertj.core.api.Assertions.assertThat;
16  
17  public class Euler41Test {
18  
19      /**
20       * <strong>Problem 41 Pandigital prime</strong>
21       * <p>
22       * We shall say that an <i>n</i>-digit number is pandigital if it makes use
23       * of all the digits 1 to <i>n</i> exactly once. For example, 2143 is a
24       * 4-digit pandigital and is also prime.
25       *
26       * What is the largest <i>n</i>-digit pandigital prime that exists?
27       * <p>
28       * See also <a href="https://projecteuler.net/problem=41">projecteuler.net
29       * problem 41</a>.
30       */
31      @Test
32      public void shouldSolveProblem41() {
33          assertThat(nDigitPandigitalNumbers(4)).contains(2143);
34          assertThat(isPrime(2143)).isTrue();
35  
36          assertThat(largestNPandigitalPrime()).isEqualTo(7652413);
37      }
38  
39      private static int largestNPandigitalPrime() {
40          return Stream.rangeClosedBy(9, 1, -1)
41                  .flatMap(n -> nDigitPandigitalNumbers(n)
42                          .filter(Utils::isPrime)
43                          .sorted(reverseOrder()))
44                  .head();
45      }
46  
47      private static List<Integer> nDigitPandigitalNumbers(int n) {
48          return List.rangeClosed(1, n)
49                  .permutations()
50                  .map(List::mkString)
51                  .map(Integer::valueOf);
52      }
53  }
