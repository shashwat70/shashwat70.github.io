7   package io.vavr;
8   
9   import java.util.*;
10  
11  final class JavaCollections {
12  
13      private JavaCollections() {
14      }
15  
16      @SuppressWarnings("unchecked")
17      static <K, V> Map<K, V> javaMap(Object... pairs) {
18          Objects.requireNonNull(pairs, "pairs is null");
19          if ((pairs.length & 1) != 0) {
20              throw new IllegalArgumentException("Odd length of key-value pairs list");
21          }
22          final Map<K, V> map = new HashMap<>();
23          for (int i = 0; i < pairs.length; i += 2) {
24              map.put((K) pairs[i], (V) pairs[i + 1]);
25          }
26          return map;
27      }
28  
29      @SuppressWarnings("unchecked")
30      static <T> Set<T> javaSet(T... elements) {
31          Objects.requireNonNull(elements, "elements is null");
32          final Set<T> set = new HashSet<>();
33          Collections.addAll(set, elements);
34          return set;
35      }
36  }
