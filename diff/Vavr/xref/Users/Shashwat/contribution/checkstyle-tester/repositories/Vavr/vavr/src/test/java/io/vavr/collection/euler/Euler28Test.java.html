7   package io.vavr.collection.euler;
8   
9   import io.vavr.Tuple;
10  import io.vavr.collection.Stream;
11  import org.junit.Test;
12  
13  import static org.assertj.core.api.Assertions.assertThat;
14  
15  /**
16   * <strong>Problem 28: Number spiral diagonals</strong>
17   * <p>
18   * Starting with the number 1 and moving to the right in a clockwise direction a
19   * 5 by 5 spiral is formed as follows:
20   * <pre>
21   *             21 22 23 24 25
22   *             20  7  8  9 10
23   *             19  6  1  2 11
24   *             18  5  4  3 12
25   *             17 16 15 14 13
26   * </pre>
27   *
28   * It can be verified that the sum of the numbers on the diagonals is 101.
29   * <p>
30   * What is the sum of the numbers on the diagonals in a 1001 by 1001 spiral
31   * formed in the same way?
32   * <p>
33   * See also
34   * <a href="https://projecteuler.net/problem=28">projecteuler.net problem 28
35   * </a>.
36   */
37  public class Euler28Test {
38  
39      @Test
40      public void shouldSolveProblem28() {
41          assertThat(sumOfDiagonalInSpiralWithSide(5)).isEqualTo(101);
42          assertThat(sumOfDiagonalInSpiralWithSide(1001)).isEqualTo(669_171_001);
43      }
44  
45      private static long sumOfDiagonalInSpiralWithSide(long maxSideLength) {
46          return diagonalNumbersInSpiralWithSide(maxSideLength).sum().longValue();
47      }
48  
49      private static Stream<Long> diagonalNumbersInSpiralWithSide(long maxSideLength) {
50          return Stream.iterate(Tuple.of(1, center()), t -> Tuple.of(nextSideLength(t._1), nextRoundOfCorners(t._2.last(), nextSideLength(t._1))))
51                  .takeWhile(t -> t._1 <= maxSideLength)
52                  .flatMap(t -> t._2);
53      }
54  
55      private static Stream<Long> center() {
56          return Stream.of(1L);
57      }
58  
59      private static int nextSideLength(int currentSideLength) {
60          return currentSideLength + 2;
61      }
62  
63      private static Stream<Long> nextRoundOfCorners(long previousCorner, int currentSideLength) {
64          return Stream.iterate(previousCorner, n -> n + currentSideLength - 1)
65                  .drop(1)
66                  .take(4);
67      }
68  }
