9   import org.junit.Test;

11  import static io.vavr.collection.euler.Utils.fibonacci;
12  import static org.assertj.core.api.Assertions.assertThat;

14  /**
15   * <strong>Problem 25: 1000-digit Fibonacci number</strong>
16   * <p>
17   * The Fibonacci sequence is defined by the recurrence relation:
18   * <p>
19   * Fn = Fn−1 + Fn−2, where F1 = 1 and F2 = 1. Hence the first 12 terms will be:
20   * <pre>
21   * F1 = 1
22   * F2 = 1
23   * F3 = 2
24   * F4 = 3
25   * F5 = 5
26   * F6 = 8
27   * F7 = 13
28   * F8 = 21
29   * F9 = 34
30   * F10 = 55
31   * F11 = 89
32   * F12 = 144
33   * </pre>
34   * <p>
35   * The 12th term, F12, is the first term to contain three digits.
36   * <p>
37   * What is the first term in the Fibonacci sequence to contain 1000 digits?
38   * <p>
39   * See also <a href="https://projecteuler.net/problem=25">projecteuler.net
40   * problem 25</a>.
41   */
42  public class Euler25Test {
43  
44      @Test
45      public void shouldSolveProblem25() {
46          assertThat(firstFibonacciTermContaining(3)).isEqualTo(12);
47          assertThat(firstFibonacciTermContaining(1000)).isEqualTo(4782);
48      }
49  
50      private static int firstFibonacciTermContaining(int digits) {
51          return fibonacci()
52                  .zipWithIndex()
53                  .find(t -> t._1.toString().length() == digits)
54                  .get()._2;
55      }
56  }
