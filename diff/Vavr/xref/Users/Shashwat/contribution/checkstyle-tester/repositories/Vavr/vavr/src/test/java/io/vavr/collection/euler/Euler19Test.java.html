20  /**
21   * <strong>Problem 19:Counting Sundays</strong>
22   *
23   * <p>You are given the following information, but you may prefer to do some research for yourself.</p>
24   * <ul>
25   * <li>1 Jan 1900 was a Monday.</li>
26   * <li>Thirty days has September,<br />
27   * April, June and November.<br />
28   * All the rest have thirty-one,<br />
29   * Saving February alone,<br />
30   * Which has twenty-eight, rain or shine.<br />
31   * And on leap years, twenty-nine.</li>
32   * <li>A leap year occurs on any year evenly divisible by 4, but not on a century unless it is divisible by 400.</li>
33   * </ul>
34   * <p>How many Sundays fell on the first of the month during the twentieth century (1 Jan 1901 to 31 Dec 2000)?</p>
35   *
36   * See also <a href="https://projecteuler.net/problem=19">projecteuler.net problem 19</a>.
37   */
38  public class Euler19Test {
39  
40      @Test
41      public void shouldSolveProblem19() {
42          assertThat(findNumberOfFirstMonthDaysOnSunday(1901, 2000)).isEqualTo(171);
43      }
44  
45      private static boolean isFirstDayOfMonthSunday(int year, Month month) {
46          return LocalDate.of(year, month, 1).getDayOfWeek() == DayOfWeek.SUNDAY;
47      }
48  
49      private static int findNumberOfFirstMonthDaysOnSunday(int startYear, int endYear) {
50          return For(List.rangeClosed(startYear, endYear), List.of(Month.values()))
51                  .yield(Tuple::of)
52                  .filter(t -> isFirstDayOfMonthSunday(t._1, t._2))
53                  .length();
54      }
