7   package io.vavr.collection.euler;
8   
9   import io.vavr.Function1;
10  import io.vavr.collection.Stream;
11  import org.assertj.core.api.Assertions;
12  import org.junit.Test;
13  
14  import static org.assertj.core.api.Assertions.assertThat;
15  
16  public class Euler14Test {
17  
18      /**
19       * <strong>Problem 14: Longest Collatz sequence</strong>
20       * <p>
21       * The following iterative sequence is defined for the set of positive integers:
22       * <pre>
23       * <code>
24       * n → n/2 (n is even)
25       * n → 3n + 1 (n is odd)
26       * </code>
27       * </pre>
28       * Using the rule above and starting with 13, we generate the following sequence:
29       * <pre>
30       * <code>
31       * 13 → 40 → 20 → 10 → 5 → 16 → 8 → 4 → 2 → 1
32       * </code>
33       * </pre>
34       * It can be seen that this sequence (starting at 13 and finishing at 1) contains 10 terms.
35       * Although it has not been proved yet (Collatz Problem), it is thought that all starting numbers finish at 1.
36       * <p>
37       * Which starting number, under one million, produces the longest chain?
38       * <p>
39       * NOTE: Once the chain starts the terms are allowed to go above one million.
40       * <p>
41       * See also <a href="https://projecteuler.net/problem=14">projecteuler.net problem 14</a>.
42       */
43      @Test
44      public void shouldSolveProblem14() {
45          // equivalent to from(1L).take(1_000_000)
46          Assertions.assertThat(Stream.from(500_000L)
47                  .take(500_000)
48                  .maxBy(collatzSequenceLength)
49                  .get()).isEqualTo(837799);
50      }
51  
52      private final static Function1<Long, Long> collatzRecursive = n -> {
53          if (n == 1) {
54              return 1L;
55          } else {
56              if (n % 2 == 0) {
57                  return Euler14Test.collatzRecursive.apply(n / 2) + 1;
58              } else {
59                  return Euler14Test.collatzRecursive.apply(3 * n + 1) + 1;
60              }
61          }
62      };
63  
64      private final static Function1<Long, Long> collatzSequenceLength = collatzRecursive.memoized();
65  }
