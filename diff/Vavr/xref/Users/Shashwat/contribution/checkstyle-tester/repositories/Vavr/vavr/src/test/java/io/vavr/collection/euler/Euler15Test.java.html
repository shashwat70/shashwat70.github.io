7   package io.vavr.collection.euler;
8   
9   import org.junit.Test;
10  
11  import java.math.BigInteger;
12  
13  import static io.vavr.collection.euler.Utils.factorial;
14  import static org.assertj.core.api.Assertions.assertThat;
15  
16  public class Euler15Test {
17  
18      /**
19       * <strong>Problem 15: Lattice paths</strong>
20       * <p>
21       * Starting in the top left corner of a 2x2 grid, and only being able to move to the right and down,
22       * there are exactly 6 routes to the bottom right corner.
23       * <p>
24       * How many such routes are there through a 20×20 grid?
25       * <p>
26       * See also <a href="https://projecteuler.net/problem=15">projecteuler.net problem 15</a>.
27       */
28      @Test
29      public void shouldSolveProblem15() {
30          assertThat(solve(2)).isEqualTo(6);
31          assertThat(solve(20)).isEqualTo(137_846_528_820L);
32      }
33  
34      private static long solve(int n) {
35          final BigInteger f = factorial(n);
36          return factorial(2 * n).divide(f).divide(f).longValue();
37      }
38  }
