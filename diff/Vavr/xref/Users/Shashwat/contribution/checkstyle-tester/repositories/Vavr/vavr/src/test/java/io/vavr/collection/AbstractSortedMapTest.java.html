7   package io.vavr.collection;
8   
9   import org.junit.Test;
10  
11  import java.math.BigDecimal;
12  import java.util.Spliterator;
13  
14  public abstract class AbstractSortedMapTest extends AbstractMapTest {
15  
16      @Override
17      abstract protected <K extends Comparable<? super K>, V> SortedMap<K, V> mapOf(K k1, V v1);
18  
19      // -- isOrdered
20  
21      @Test
22      public void shouldReturnOrdered() {
23          final Map<Integer, String> actual = mapOf(1, "1", 1, "2");
24          assertThat(actual.isOrdered()).isTrue();
25      }
26  
27      // -- narrow
28  
29      @Test
30      public void shouldNarrowMap() {
31          final SortedMap<Integer, Double> int2doubleMap = mapOf(1, 1.0d);
32          final SortedMap<Integer, Number> number2numberMap = SortedMap.narrow(int2doubleMap);
33          final int actual = number2numberMap.put(2, new BigDecimal("2.0")).values().sum().intValue();
34          assertThat(actual).isEqualTo(3);
35      }
36  
37      // -- spliterator
38  
39      @Test
40      public void shouldHaveSortedSpliterator() {
41          assertThat(of(1, 2, 3).spliterator().hasCharacteristics(Spliterator.SORTED)).isTrue();
42      }
43  
44      @Test
45      public void shouldHaveOrderedSpliterator() {
46          assertThat(of(1, 2, 3).spliterator().hasCharacteristics(Spliterator.ORDERED)).isTrue();
47      }
48  
49      // -- isSequential()
50  
51      @Test
52      public void shouldReturnFalseWhenIsSequentialCalled() {
53          assertThat(of(1, 2, 3).isSequential()).isFalse();
54      }
55  
56  }
