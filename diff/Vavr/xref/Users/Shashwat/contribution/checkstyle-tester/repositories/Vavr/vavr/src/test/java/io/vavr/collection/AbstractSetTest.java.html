108     @Test
109     public void shouldObeyEqualityConstraints() {
110 
111         // sequential collections
112         assertThat(empty().equals(HashSet.empty())).isTrue();
113         assertThat(of(1).equals(HashSet.of(1))).isTrue();
114         assertThat(of(1, 2, 3).equals(HashSet.of(1, 2, 3))).isTrue();
115         assertThat(of(1, 2, 3).equals(HashSet.of(3, 2, 1))).isTrue();
116 
117         // other classes
118         assertThat(empty().equals(List.empty())).isFalse();
119         assertThat(empty().equals(HashMap.empty())).isFalse();
120         assertThat(empty().equals(HashMultimap.withSeq().empty())).isFalse();
121 
122         assertThat(empty().equals(LinkedHashMap.empty())).isFalse();
123         assertThat(empty().equals(LinkedHashMultimap.withSeq().empty())).isFalse();
124 
125         assertThat(empty().equals(TreeMap.empty())).isFalse();
126         assertThat(empty().equals(TreeMultimap.withSeq().empty())).isFalse();
127     }
