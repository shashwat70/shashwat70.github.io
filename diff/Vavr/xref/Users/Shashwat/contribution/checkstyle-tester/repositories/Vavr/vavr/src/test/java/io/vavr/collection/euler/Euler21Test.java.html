18  /**
19   * <strong>Problem 21: Amicable numbers</strong>
20   *
21   * <p>Let d(<i>n</i>) be defined as the sum of proper divisors of <i>n</i> (numbers less than <i>n</i> which divide evenly into <i>n</i>).<br />
22   * If d(<i>a</i>) = <i>b</i> and d(<i>b</i>) = <i>a</i>, where <i>a</i> ≠ <i>b</i>, then <i>a</i> and <i>b</i> are an amicable pair and each of <i>a</i> and <i>b</i> are called amicable numbers.</p>
23   * <p>For example, the proper divisors of 220 are 1, 2, 4, 5, 10, 11, 20, 22, 44, 55 and 110; therefore d(220) = 284. The proper divisors of 284 are 1, 2, 4, 71 and 142; so d(284) = 220.</p>
24   * <p>Evaluate the sum of all the amicable numbers under 10000</p>
25   * See also <a href="https://projecteuler.net/problem=21">projecteuler.net problem 21</a>.
26   */
27  
28  public class Euler21Test {
29  
30      @Test
31      public void shouldSolveProblem21() {
32          assertThat(sumOfDivisors(220)).isEqualTo(1 + 2 + 4 + 5 + 10 + 11 + 20 + 22 + 44 + 55 + 110);
33          assertThat(sumOfDivisors(284)).isEqualTo(1 + 2 + 4 + 71 + 142);
34          assertThat(sumOfAmicablePairs(10000)).isEqualTo(31626);
35      }
36  
37      private static int sumOfDivisors(int n) {
38          return 1 + Stream.rangeClosed(2, (int) Math.sqrt(n))
39                  .map(d -> Tuple.of(d, n / d))
40                  .filter(t -> t._1 * t._2 == n && !Objects.equals(t._1, t._2))
41                  .map(t -> t._1 + t._2)
42                  .foldLeft(0, (sum, x) -> sum + x);
43      }
44  
45      private static int sumOfAmicablePairs(int n) {
46          final Function1<Integer, Integer> mSumOfDivisors = Function1.of(Euler21Test::sumOfDivisors).memoized();
47          return Stream.range(1, n)
48                  .filter(x -> mSumOfDivisors.apply(mSumOfDivisors.apply(x)).intValue() == x && mSumOfDivisors.apply(x) > x)
49                  .foldLeft(0, (sum, x) -> sum + x + mSumOfDivisors.apply(x));
50      }
