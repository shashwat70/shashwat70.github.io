137     @Test
138     public void shouldNarrowHashMap() {
139         final HashMap<Integer, Double> int2doubleMap = mapOf(1, 1.0d);
140         final HashMap<Number, Number> number2numberMap = HashMap.narrow(int2doubleMap);
141         final int actual = number2numberMap.put(new BigDecimal("2"), new BigDecimal("2.0")).values().sum().intValue();
142         assertThat(actual).isEqualTo(3);
143     }

145     @Test
146     public void shouldWrapMap() {
147         final java.util.Map<Integer, Integer> source = new java.util.HashMap<>();
148         source.put(1, 2);
149         source.put(3, 4);
150         assertThat(HashMap.ofAll(source)).isEqualTo(emptyIntInt().put(1, 2).put(3, 4));
151     }
