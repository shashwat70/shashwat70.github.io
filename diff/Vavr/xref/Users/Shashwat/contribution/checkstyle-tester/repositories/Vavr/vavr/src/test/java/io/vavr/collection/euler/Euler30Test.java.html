17  /**
18   * <strong>Problem 30: Digit fifth powers</strong>
19   * <p>
20   * Surprisingly there are only three numbers that can be written as the sum of
21   * fourth powers of their digits:
22   * <pre>
23   * 1634 = 1^4 + 6^4 + 3^4 + 4^4
24   * 8208 = 8^4 + 2^4 + 0^4 + 8^4
25   * 9474 = 9^4 + 4^4 + 7^4 + 4^4
26   * </pre>
27   *
28   * As 1 = 1^4 is not a sum it is not included.
29   * <p>
30   * The sum of these numbers is 1634 + 8208 + 9474 = 19316.
31   * <p>
32   * Find the sum of all the numbers that can be written as the sum of fifth
33   * powers of their digits.
34   * <p>
35   * See also
36   * <a href="https://projecteuler.net/problem=30">projecteuler.net problem 30
37   * </a>.
38   */
39  public class Euler30Test {
40  
41      @Test
42      public void shouldSolveProblem26() {
43          assertThat(sumOfAllTheNumbersThatCanBeWrittenAsTheSumOfPowersOfTheirDigits(4)).isEqualTo(19316);
44          assertThat(sumOfAllTheNumbersThatCanBeWrittenAsTheSumOfPowersOfTheirDigits(5)).isEqualTo(443_839);
45      }
46  
47      private static long sumOfAllTheNumbersThatCanBeWrittenAsTheSumOfPowersOfTheirDigits(int powers) {
48          return List.rangeClosed(10, maximalSumForPowers(powers))
49                  .filter(i -> sumOfPowersOfDigits(powers, i) == i)
50                  .sum().longValue();
51      }
52  
53      private static long maximalSumForPowers(int powers) {
54          return Stream.from(1)
55                  .map(i -> Tuple.of((long) Math.pow(10, i) - 1, List.fill(i, () -> Math.pow(9, powers)).sum().longValue()))
56                  .find(t -> t._1 > t._2)
57                  .map(t -> t._1).get();
58      }
59  
60      private static long sumOfPowersOfDigits(int powers, long num) {
61          return CharSeq.of(Long.toString(num))
62                  .map(c -> Character.digit(c, 10))
63                  .map(d -> (long) Math.pow(d, powers))
64                  .sum().longValue();
65      }
