7   package io.vavr.collection.euler;
8   
9   import io.vavr.collection.Stream;
10  import org.junit.Test;
11  
12  import static org.assertj.core.api.Assertions.assertThat;
13  
14  public class Euler05Test {
15  
16      /**
17       * <strong>Problem 5: Smallest multiple</strong>
18       * <p>
19       * 2520 is the smallest number that can be divided by each of the numbers from 1
20       * to 10 without any remainder.
21       * <p>
22       * What is the smallest positive number that is evenly divisible by all of the
23       * numbers from 1 to 20?
24       * <p>
25       * See also <a href="https://projecteuler.net/problem=5">projecteuler.net problem 5</a>.
26       */
27      @Test
28      public void shouldSolveProblem5() {
29          assertThat(smallestPositiveNumberEvenlyDivisibleByAllNumbersFrom1To(10)).isEqualTo(2_520L);
30          assertThat(smallestPositiveNumberEvenlyDivisibleByAllNumbersFrom1To(20)).isEqualTo(232_792_560L);
31      }
32  
33      private static long smallestPositiveNumberEvenlyDivisibleByAllNumbersFrom1To(int max) {
34          return Stream.rangeClosed(2, max)
35                  .map(PrimeNumbers::factorization)
36                  .reduce((m1, m2) -> m1.merge(m2, Math::max))
37                  .foldLeft(1L, (xs, x) -> xs * pow(x._1, x._2));
38      }
39  
40      private static long pow(long a, long p) {
41          return Stream.rangeClosed(1, p).fold(1L, (xs, x) -> xs * a);
42      }
