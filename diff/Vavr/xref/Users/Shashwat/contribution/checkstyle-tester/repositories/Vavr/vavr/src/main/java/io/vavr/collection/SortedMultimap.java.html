7   package io.vavr.collection;

9   import io.vavr.control.Option;

16  /**
17   * An immutable {@code SortedMultimap} interface.
18   *
19   * @param <K> Key type
20   * @param <V> Value type
21   * @author Ruslan Sennov
22   */
23  public interface SortedMultimap<K, V> extends Multimap<K, V>, Ordered<K> {
24  
25      long serialVersionUID = 1L;
26  
27      /**
28       * Narrows a widened {@code SortedMultimap<? extends K, ? extends V>} to {@code SortedMultimap<K, V>}
29       * by performing a type-safe cast. This is eligible because immutable/read-only
30       * collections are covariant.
31       *
32       * @param map A {@code SortedMultimap}.
33       * @param <K> Key type
34       * @param <V> Value type
35       * @return the given {@code multimap} instance as narrowed type {@code SortedMultimap<K, V>}.
36       */
37      @SuppressWarnings("unchecked")
38      static <K, V> SortedMultimap<K, V> narrow(SortedMultimap<? extends K, ? extends V> map) {
39          return (SortedMultimap<K, V>) map;
40      }
41  
42      @Override
43      SortedMultimap<K, V> filter(BiPredicate<? super K, ? super V> predicate);
44  
45      @Override
46      SortedMultimap<K, V> filterKeys(Predicate<? super K> predicate);
47  
48      @Override
49      SortedMultimap<K, V> filterValues(Predicate<? super V> predicate);
50  
51      @Override
52      SortedSet<K> keySet();
53  
54      @Override
55      SortedMultimap<K, V> merge(Multimap<? extends K, ? extends V> that);
56  
57      @Override
58      <K2 extends K, V2 extends V> SortedMultimap<K, V> merge(Multimap<K2, V2> that, BiFunction<Traversable<V>, Traversable<V2>, Traversable<V>> collisionResolution);
59  
60      @Override
61      SortedMultimap<K, V> put(K key, V value);
62  
63      @Override
64      SortedMultimap<K, V> put(Tuple2<? extends K, ? extends V> entry);
65  
66      @Override
67      SortedMultimap<K, V> remove(K key);
68  
69      @Override
70      SortedMultimap<K, V> remove(K key, V value);
71  
72      @Override
73      SortedMultimap<K, V> removeAll(BiPredicate<? super K, ? super V> predicate);
74  
75      @Override
76      SortedMultimap<K, V> removeAll(Iterable<? extends K> keys);
77  
78      @Override
79      SortedMultimap<K, V> removeKeys(Predicate<? super K> predicate);
80  
81      @Override
82      SortedMultimap<K, V> removeValues(Predicate<? super V> predicate);
83  
84      @Override
85      java.util.SortedMap<K, Collection<V>> toJavaMap();
86  
87      @Override
88      SortedMultimap<K, V> distinct();
89  
90      @Override
91      SortedMultimap<K, V> distinctBy(Comparator<? super Tuple2<K, V>> comparator);
92  
93      @Override
94      <U> SortedMultimap<K, V> distinctBy(Function<? super Tuple2<K, V>, ? extends U> keyExtractor);
95  
96      @Override
97      SortedMultimap<K, V> drop(int n);
98  
99      @Override
100     SortedMultimap<K, V> dropRight(int n);
101 
102     @Override
103     SortedMultimap<K, V> dropUntil(Predicate<? super Tuple2<K, V>> predicate);
104 
105     @Override
106     SortedMultimap<K, V> dropWhile(Predicate<? super Tuple2<K, V>> predicate);
107 
108     @Override
109     SortedMultimap<K, V> filter(Predicate<? super Tuple2<K, V>> predicate);
110 
111     @Override
112     <C> Map<C, ? extends SortedMultimap<K, V>> groupBy(Function<? super Tuple2<K, V>, ? extends C> classifier);
113 
114     @Override
115     Iterator<? extends SortedMultimap<K, V>> grouped(int size);
116 
117     @Override
118     SortedMultimap<K, V> init();
119 
120     @Override
121     Option<? extends SortedMultimap<K, V>> initOption();
122 
123     @Override
124     SortedMultimap<K, V> orElse(Iterable<? extends Tuple2<K, V>> other);
125 
126     @Override
127     SortedMultimap<K, V> orElse(Supplier<? extends Iterable<? extends Tuple2<K, V>>> supplier);
128 
129     @Override
130     Tuple2<? extends SortedMultimap<K, V>, ? extends SortedMultimap<K, V>> partition(Predicate<? super Tuple2<K, V>> predicate);
131 
132     @Override
133     SortedMultimap<K, V> peek(Consumer<? super Tuple2<K, V>> action);
134 
135     @Override
136     SortedMultimap<K, V> replace(Tuple2<K, V> currentElement, Tuple2<K, V> newElement);
137 
138     @Override
139     SortedMultimap<K, V> replaceAll(Tuple2<K, V> currentElement, Tuple2<K, V> newElement);
140 
141     @Override
142     SortedMultimap<K, V> replaceValue(K key, V value);
143 
144     @Override
145     SortedMultimap<K, V> replace(K key, V oldValue, V newValue);
146 
147     @Override
148     SortedMultimap<K, V> replaceAll(BiFunction<? super K, ? super V, ? extends V> function);
149 
150     @Override
151     SortedMultimap<K, V> retainAll(Iterable<? extends Tuple2<K, V>> elements);
152 
153     @Override
154     SortedMultimap<K, V> scan(Tuple2<K, V> zero,
155                               BiFunction<? super Tuple2<K, V>, ? super Tuple2<K, V>, ? extends Tuple2<K, V>> operation);
156 
157     @Override
158     Iterator<? extends SortedMultimap<K, V>> slideBy(Function<? super Tuple2<K, V>, ?> classifier);
159 
160     @Override
161     Iterator<? extends SortedMultimap<K, V>> sliding(int size);
162 
163     @Override
164     Iterator<? extends SortedMultimap<K, V>> sliding(int size, int step);
165 
166     @Override
167     Tuple2<? extends SortedMultimap<K, V>, ? extends SortedMultimap<K, V>> span(Predicate<? super Tuple2<K, V>> predicate);
168 
169     @Override
170     SortedMultimap<K, V> tail();
171 
172     @Override
173     Option<? extends SortedMultimap<K, V>> tailOption();
174 
175     @Override
176     SortedMultimap<K, V> take(int n);
177 
178     @Override
179     SortedMultimap<K, V> takeRight(int n);
180 
181     @Override
182     SortedMultimap<K, V> takeUntil(Predicate<? super Tuple2<K, V>> predicate);
183 
184     @Override
185     SortedMultimap<K, V> takeWhile(Predicate<? super Tuple2<K, V>> predicate);
186 
187 }
