9   import io.vavr.collection.Stream;
10  import io.vavr.collection.List;

13  import java.math.BigInteger;

15  import static org.assertj.core.api.Assertions.assertThat;
16  
17  public class Euler55Test {
18  
19      /**
20       * <strong>Problem 55: Lychrel numbers</strong>
21       * <p>
22       * If we take 47, reverse and add, 47 + 74 = 121, which is palindromic.
23       * <p>
24       * Not all numbers produce palindromes so quickly. For example,
25       * <ul>
26       * <li>349 + 943 = 1292
27       * <li>1292 + 2921 = 4213
28       * <li>4213 + 3124 = 7337
29       * </ul>
30       * That is, 349 took three iterations to arrive at a palindrome.
31       * <p>
32       * Although no one has proved it yet, it is thought that some numbers, like 196, never produce a palindrome.
33       * A number that never forms a palindrome through the reverse and add process is called a Lychrel number.
34       * Due to the theoretical nature of these numbers, and for the purpose of this problem, we shall assume
35       * that a number is Lychrel until proven otherwise. In addition you are given that for every number
36       * below ten-thousand, it will either (i) become a palindrome in less than fifty iterations, or,
37       * (ii) no one, with all the computing power that exists, has managed so far to map it to a palindrome.
38       * In fact, 10677 is the first number to be shown to require over fifty iterations
39       * before producing a palindrome: 4668731596684224866951378664 (53 iterations, 28-digits).
40       * <p>
41       * Surprisingly, there are palindromic numbers that are themselves Lychrel numbers;
42       * the first example is 4994.
43       * <p>
44       * How many Lychrel numbers are there below ten-thousand?
45       * <p>
46       * See also <a href="https://projecteuler.net/problem=55">projecteuler.net problem 55</a>.
47       */
48      @Test
49      public void shouldSolveProblem55() {
50          assertThat(solve()).isEqualTo(249);
51      }
52  
53      private static int solve() {
54          return Stream.range(1, 10_000)
55                  .filter(Euler55Test::isLychrel)
56                  .length();
57      }
58  
59      private static boolean isLychrel(int n) {
60          return Stream.iterate(String.valueOf(n), Euler55Test::next)
61                  .tail()  // Surprisingly, there are palindromic numbers that are themselves Lychrel numbers
62                  .take(50)
63                  .find(Utils::isPalindrome)
64                  .isEmpty();
65      }
66  
67      private static String next(String s) {
68          return List.of(s, Utils.reverse(s))
69                  .map(BigInteger::new)
70                  .reduce(BigInteger::add)
71                  .toString();
72      }
73  }
