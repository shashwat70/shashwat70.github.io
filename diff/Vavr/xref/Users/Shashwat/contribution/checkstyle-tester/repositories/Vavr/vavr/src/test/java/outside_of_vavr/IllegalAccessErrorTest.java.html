7   package outside_of_vavr;
8   
9   import io.vavr.collection.Array;
10  import io.vavr.collection.BitSet;
11  import io.vavr.collection.HashMap;
12  import io.vavr.collection.List;
13  import org.junit.Test;
14  
15  import java.util.function.BiFunction;
16  
17  import static org.assertj.core.api.Assertions.assertThat;
18  
19  public class IllegalAccessErrorTest {
20  
21      @Test
22      public void shouldNotThrowIllegalAccessErrorWhenUsingHashMapMergeMethodReference() {
23          final BiFunction<HashMap<String, String>, HashMap<String, String>, HashMap<String, String>> merge = HashMap::merge;
24          final HashMap<String, String> reduced = Array.of("a", "b", "c")
25                  .map(t -> HashMap.of(t, t))
26                  .reduce(merge);
27          assertThat(reduced).isEqualTo(HashMap.of("a", "a", "b", "b", "c", "c"));
28      }
29  
30      @Test
31      public void shouldNotThrowIllegalAccessErrorWhenUsingBitSetAddAllMethodReference() {
32          final BiFunction<BitSet<Integer>, BitSet<Integer>, BitSet<Integer>> union = BitSet::union;
33          final BitSet<Integer> reduced = List.of(BitSet.of(1, 2, 3), BitSet.of(2, 3, 4)).reduce(union);
34          assertThat(reduced).isEqualTo(BitSet.of(1, 2, 3, 4));
35      }
36  }
