29      @SuppressWarnings("unchecked")
30      static <K, V, M extends Map<K, V>> Tuple2<V, M> computeIfAbsent(M map, K key, Function<? super K, ? extends V> mappingFunction) {
31          Objects.requireNonNull(mappingFunction, "mappingFunction is null");
32          final Option<V> value = map.get(key);
33          if (value.isDefined()) {
34              return Tuple.of(value.get(), map);
35          } else {
36              final V newValue = mappingFunction.apply(key);
37              final M newMap = (M) map.put(key, newValue);
38              return Tuple.of(newValue, newMap);
39          }
40      }

42      @SuppressWarnings("unchecked")
43      static <K, V, M extends Map<K, V>> Tuple2<Option<V>, M> computeIfPresent(M map, K key, BiFunction<? super K, ? super V, ? extends V> remappingFunction) {
44          final Option<V> value = map.get(key);
45          if (value.isDefined()) {
46              final V newValue = remappingFunction.apply(key, value.get());
47              final M newMap = (M) map.put(key, newValue);
48              return Tuple.of(Option.of(newValue), newMap);
49          } else {
50              return Tuple.of(Option.none(), map);
51          }
52      }

70      static <K, V, M extends Map<K, V>> M drop(M map, OfEntries<K, V, M> ofEntries, Supplier<M> emptySupplier, int n) {
71          if (n <= 0) {
72              return map;
73          } else if (n >= map.size()) {
74              return emptySupplier.get();
75          } else {
76              return ofEntries.apply(map.iterator().drop(n));
77          }
78      }

141     static <K, V, M extends Map<K, V>> M merge(M map, OfEntries<K, V, M> ofEntries,
142             Map<? extends K, ? extends V> that) {
143         Objects.requireNonNull(that, "that is null");
144         if (map.isEmpty()) {
145             return ofEntries.apply(Map.narrow(that));
146         } else if (that.isEmpty()) {
147             return map;
148         } else {
149             return that.foldLeft(map, (result, entry) -> !result.containsKey(entry._1) ? put(result, entry) : result);
150         }
151     }

153     @SuppressWarnings("unchecked")
154     static <K, V, U extends V, M extends Map<K, V>> M merge(
155             M map, OfEntries<K, V, M> ofEntries,
156             Map<? extends K, U> that, BiFunction<? super V, ? super U, ? extends V> collisionResolution) {
157         Objects.requireNonNull(that, "that is null");
158         Objects.requireNonNull(collisionResolution, "collisionResolution is null");
159         if (map.isEmpty()) {
160             return ofEntries.apply(Map.narrow(that));
161         } else if (that.isEmpty()) {
162             return map;
163         } else {
164             return that.foldLeft(map, (result, entry) -> {
165                 final K key = entry._1;
166                 final U value = entry._2;
167                 final V newValue = result.get(key).map(v -> (V) collisionResolution.apply(v, value)).getOrElse(value);
168                 return (M) result.put(key, newValue);
169             });
170         }
171     }

174     static <T, K, V, M extends Map<K, V>> M ofStream(M map, java.util.stream.Stream<? extends T> stream,

198     static <K, V, M extends Map<K, V>> M peek(M map, Consumer<? super Tuple2<K, V>> action) {
199         Objects.requireNonNull(action, "action is null");
200         if (!map.isEmpty()) {

202         }

206     @SuppressWarnings("unchecked")
207     static <K, V, U extends V, M extends Map<K, V>> M put(M map, K key, U value,
208             BiFunction<? super V, ? super U, ? extends V> merge) {
209         Objects.requireNonNull(merge, "the merge function is null");
210         final Option<V> currentValue = map.get(key);
211         if (currentValue.isEmpty()) {
212             return (M) map.put(key, value);
213         } else {
214             return (M) map.put(key, merge.apply(currentValue.get(), value));
215         }
216     }

219     static <K, V, M extends Map<K, V>> M put(M map, Tuple2<? extends K, ? extends V> entry) {
220         Objects.requireNonNull(entry, "entry is null");
221         return (M) map.put(entry._1, entry._2);
222     }

224     static <K, V, U extends V, M extends Map<K, V>> M put(M map, Tuple2<? extends K, U> entry,
225             BiFunction<? super V, ? super U, ? extends V> merge) {
226         Objects.requireNonNull(merge, "the merge function is null");
227         final Option<V> currentValue = map.get(entry._1);
228         if (currentValue.isEmpty()) {
229             return put(map, entry);
230         } else {
231             return put(map, entry.map2(value -> merge.apply(currentValue.get(), value)));
232         }
233     }

335         Objects.requireNonNull(predicate, "predicate is null");
