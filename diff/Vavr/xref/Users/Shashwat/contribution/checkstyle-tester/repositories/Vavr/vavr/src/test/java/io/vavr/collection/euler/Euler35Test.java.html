15  import java.util.function.Predicate;

17  import static org.assertj.core.api.Assertions.assertThat;

19  /**
20   * <strong>Problem 35: Circular primes</strong>
21   * <p>The number, 197, is called a circular prime because all rotations of the digits: 197, 971, and 719, are themselves prime.</p>
22   * <p>There are thirteen such primes below 100: 2, 3, 5, 7, 11, 13, 17, 31, 37, 71, 73, 79, and 97.</p>
23   * <p>How many circular primes are there below one million?</p>
24   * See also <a href="https://projecteuler.net/problem=35">projecteuler.net problem 35</a>.
25   */
26  public class Euler35Test {
27  
28      @Test
29      public void shouldSolveProblem35() {
30          assertThat(rotations(197)).isEqualTo(List.of(197, 971, 719));
31          assertThat(circularPrimes(100)).isEqualTo(13);
32          assertThat(circularPrimes(1000000)).isEqualTo(55);
33      }
34  
35      private static int circularPrimes(int n) {
36          final Predicate<Integer> memoizedIsPrime = Function1.of(Euler35Test::isPrime).memoized()::apply;
37          return Stream.rangeClosed(2, n)
38                  .filter(memoizedIsPrime)
39                  .map(Euler35Test::rotations)
40                  .filter(list -> list.forAll(memoizedIsPrime))
41                  .length();
42      }
43  
44      private static boolean isPrime(int n) {
45          return n == 2 || n % 2 != 0 &&
46                  Stream.rangeClosedBy(3, (int) Math.sqrt(n), 2)
47                          .find(x -> n % x == 0)
48                          .isEmpty();
49      }
50  
51      private static List<Integer> rotations(int n) {
52          final CharSeq seq = CharSeq.of(String.valueOf(n));
53          return Stream.range(0, seq.length())
54                  .map(i -> seq.drop(i).appendAll(seq.take(i)))
55                  .map(s -> Integer.valueOf(s.toString()))
56                  .toList();
57      }
