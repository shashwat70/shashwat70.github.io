18  /**
19   * <strong>Problem 27: Quadratic primes</strong>
20   * <p>
21   * Euler discovered the remarkable quadratic formula:
22   * <p>
23   * n² + n + 41
24   * <p>
25   * It turns out that the formula will produce 40 primes for the consecutive
26   * values n = 0 to 39. However, when n = 40, 40^2 + 40 + 41 = 40(40 + 1) + 41 is
27   * divisible by 41, and certainly when n = 41, 41² + 41 + 41 is clearly
28   * divisible by 41.
29   * <p>
30   * The incredible formula n² − 79n + 1601 was discovered, which produces 80
31   * primes for the consecutive values n = 0 to 79. The product of the
32   * coefficients, −79 and 1601, is −126479.
33   * <p>
34   * Considering quadratics of the form:
35   * <p>
36   * n² + an + b, where |a| < 1000 and |b| < 1000 <p>
37   * where |n| is the modulus/absolute value of n e.g. |11| = 11 and |−4| = 4
38   * <p>
39   * Find the product of the coefficients, a and b, for the quadratic expression
40   * that produces the maximum number of primes for consecutive values of n,
41   * starting with n = 0.
42   * <p>
43   * See also
44   * <a href="https://projecteuler.net/problem=27">projecteuler.net problem 27
45   * </a>.
46   */
47  public class Euler27Test {
48  
49      @Test
50      public void shouldSolveProblem27() {
51          assertThat(numberOfConsecutivePrimesProducedByFormulaWithCoefficients(1, 41)).isEqualTo(40);
52          assertThat(numberOfConsecutivePrimesProducedByFormulaWithCoefficients(-79, 1601)).isEqualTo(80);
53  
54          assertThat(productOfCoefficientsWithMostConsecutivePrimes(-999, 999)).isEqualTo(-59231);
55      }
56  
57      private static int productOfCoefficientsWithMostConsecutivePrimes(int coefficientsLowerBound, int coefficientsUpperBound) {
58          final List<Integer> coefficients = List.rangeClosed(coefficientsLowerBound, coefficientsUpperBound);
59          return API.For(coefficients, coefficients).yield(Tuple::of)
60                  .map(c -> Tuple.of(c._1, c._2, numberOfConsecutivePrimesProducedByFormulaWithCoefficients(c._1, c._2)))
61                  .fold(Tuple.of(0, 0, -1), (n, m) -> n._3 >= m._3 ? n : m)
62                  .apply((a, b, p) -> a * b);
63      }
64  
65      private static int numberOfConsecutivePrimesProducedByFormulaWithCoefficients(int a, int b) {
66          return Stream.from(0L)
67                  .map(n -> (long) Math.pow(n, 2) + a * n + b)
68                  .takeWhile(Utils.MEMOIZED_IS_PRIME::apply)
69                  .length();
70      }
