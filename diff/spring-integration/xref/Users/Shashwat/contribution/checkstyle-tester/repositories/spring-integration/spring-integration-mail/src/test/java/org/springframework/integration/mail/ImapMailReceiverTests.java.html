182 	public void testIdleWithServerGuts(ImapMailReceiver receiver, boolean mapped, boolean simple) throws Exception {
183 		Properties mailProps = new Properties();
184 		mailProps.put("mail.debug", "true");
185 		mailProps.put("mail.imap.connectionpool.debug", "true");
186 		receiver.setJavaMailProperties(mailProps);
187 		receiver.setMaxFetchSize(1);
188 		receiver.setShouldDeleteMessages(false);
189 		receiver.setShouldMarkMessagesAsRead(true);
190 		receiver.setCancelIdleInterval(1);
191 		ThreadPoolTaskScheduler taskScheduler = new ThreadPoolTaskScheduler();
192 		setUpScheduler(receiver, taskScheduler);
193 		receiver.setUserFlag("testSIUserFlag");
194 		receiver.afterPropertiesSet();
195 		Log logger = spy(TestUtils.getPropertyValue(receiver, "logger", Log.class));
196 		new DirectFieldAccessor(receiver).setPropertyValue("logger", logger);
197 		ImapIdleChannelAdapter adapter = new ImapIdleChannelAdapter(receiver);
198 		QueueChannel channel = new QueueChannel();
199 		adapter.setOutputChannel(channel);
200 		adapter.setTaskScheduler(taskScheduler);
201 		adapter.setReconnectDelay(1);
202 		adapter.start();
203 		if (!mapped) {
204 			@SuppressWarnings("unchecked")
205 			org.springframework.messaging.Message<MimeMessage> received =
206 					(org.springframework.messaging.Message<MimeMessage>) channel.receive(10000);
207 			assertThat(received).isNotNull();
208 			assertThat(received.getPayload().getReceivedDate()).isNotNull();
209 			assertThat(received.getPayload().getLineCount() > -1).isTrue();
210 			if (simple) {
211 				assertThat(received.getPayload().getContent())
212 						.isEqualTo(TestMailServer.MailServer.MailHandler.BODY + "\r\n");
213 			}
214 			else {
215 				assertThat(received.getPayload().getContent())
216 						.isEqualTo(TestMailServer.MailServer.MailHandler.MESSAGE + "\r\n");
217 			}
218 		}
219 		else {
220 			org.springframework.messaging.Message<?> received = channel.receive(10000);
221 			assertThat(received).isNotNull();
222 			MessageHeaders headers = received.getHeaders();
223 			assertThat(headers.get(MailHeaders.RAW_HEADERS)).isNotNull();
224 			assertThat(headers.get(MailHeaders.CONTENT_TYPE)).isEqualTo("TEXT/PLAIN; charset=ISO-8859-1");
225 			assertThat(headers.get(MessageHeaders.CONTENT_TYPE)).isEqualTo("TEXT/PLAIN; charset=ISO-8859-1");
226 			assertThat(headers.get(MailHeaders.FROM)).isEqualTo("Bar <bar@baz>");
227 			String[] toHeader = headers.get(MailHeaders.TO, String[].class);
228 			assertThat(toHeader).isNotEmpty();
229 			assertThat(toHeader[0]).isEqualTo("Foo <foo@bar>");
230 			assertThat(Arrays.toString(headers.get(MailHeaders.CC, String[].class))).isEqualTo("[a@b, c@d]");
231 			assertThat(Arrays.toString(headers.get(MailHeaders.BCC, String[].class))).isEqualTo("[e@f, g@h]");
232 			assertThat(headers.get(MailHeaders.SUBJECT)).isEqualTo("Test Email");
233 			if (simple) {
234 				assertThat(received.getPayload()).isEqualTo(TestMailServer.MailServer.MailHandler.BODY + "\r\n");
235 			}
236 			else {
237 				assertThat(received.getPayload()).isEqualTo(TestMailServer.MailServer.MailHandler.MESSAGE + "\r\n");
238 			}
239 		}
240 		assertThat(channel.receive(20000)).isNotNull(); // new message after idle
241 		assertThat(channel.receive(100)).isNull(); // no new message after second and third idle
242 		verify(logger).debug("Canceling IDLE");
243 
244 		adapter.stop();
245 		taskScheduler.shutdown();
246 		assertThat(this.imapIdleServer.assertReceived("storeUserFlag")).isTrue();
247 	}
