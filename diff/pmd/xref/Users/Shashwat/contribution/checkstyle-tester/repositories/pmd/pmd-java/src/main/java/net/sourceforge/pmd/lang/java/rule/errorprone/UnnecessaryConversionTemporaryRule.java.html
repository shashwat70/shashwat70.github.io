5   package net.sourceforge.pmd.lang.java.rule.errorprone;
6   
7   import java.util.Set;
8   
9   import net.sourceforge.pmd.lang.java.ast.ASTAllocationExpression;
10  import net.sourceforge.pmd.lang.java.ast.ASTClassOrInterfaceType;
11  import net.sourceforge.pmd.lang.java.ast.ASTPrimaryExpression;
12  import net.sourceforge.pmd.lang.java.ast.ASTPrimarySuffix;
13  import net.sourceforge.pmd.lang.java.rule.AbstractJavaRule;
14  import net.sourceforge.pmd.util.CollectionUtil;
15  
16  public class UnnecessaryConversionTemporaryRule extends AbstractJavaRule {
17  
18      private boolean inPrimaryExpressionContext;
19      private ASTPrimaryExpression primary;
20      private boolean usingPrimitiveWrapperAllocation;
21  
22      private static final Set<String> PRIMITIVE_WRAPPERS = CollectionUtil
23              .asSet(new String[] { "Integer", "Boolean", "Double", "Long", "Short", "Byte", "Float" });
24  
25      @Override
26      public Object visit(ASTPrimaryExpression node, Object data) {
27          if (node.getNumChildren() == 0 || (node.getChild(0)).getNumChildren() == 0
28                  || !(node.getChild(0).getChild(0) instanceof ASTAllocationExpression)) {
29              return super.visit(node, data);
30          }
31          // TODO... hmmm... is this inPrimaryExpressionContext gibberish
32          // necessary?
33          inPrimaryExpressionContext = true;
34          primary = node;
35          super.visit(node, data);
36          inPrimaryExpressionContext = false;
37          usingPrimitiveWrapperAllocation = false;
38          return data;
39      }
40  
41      @Override
42      public Object visit(ASTAllocationExpression node, Object data) {
43          if (!inPrimaryExpressionContext || !(node.getChild(0) instanceof ASTClassOrInterfaceType)) {
44              return super.visit(node, data);
45          }
46          if (!PRIMITIVE_WRAPPERS.contains(node.getChild(0).getImage())) {
47              return super.visit(node, data);
48          }
49          usingPrimitiveWrapperAllocation = true;
50          return super.visit(node, data);
51      }
52  
53      @Override
54      public Object visit(ASTPrimarySuffix node, Object data) {
55          if (inPrimaryExpressionContext && usingPrimitiveWrapperAllocation) {
56              if (node.hasImageEqualTo("toString")) {
57                  if (node.getParent() == primary) {
58                      addViolation(data, node);
59                  }
60              }
61          }
62          return super.visit(node, data);
63      }
64  
65  }
