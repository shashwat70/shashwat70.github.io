5   package net.sourceforge.pmd.lang.java.symboltable;
6   
7   import java.util.List;
8   
9   import net.sourceforge.pmd.lang.ast.Node;
10  import net.sourceforge.pmd.lang.java.JavaParsingHelper;
11  import net.sourceforge.pmd.lang.java.ast.ASTCompilationUnit;
12  
13  /**
14   * Base class for tests that usually need processing stages to run when
15   * parsing code.
16   */
17  public abstract class BaseNonParserTest {
18  
19      protected final JavaParsingHelper java = JavaParsingHelper.WITH_PROCESSING.withResourceContext(getClass());
20      protected final JavaParsingHelper java5 = java.withDefaultVersion("1.5");
21  
22  
23      protected ASTCompilationUnit parseCode(final String code) {
24          return java.parse(code);
25      }
26  
27      protected <T extends Node> List<T> getOrderedNodes(Class<T> target, String code) {
28          return JavaParsingHelper.WITH_PROCESSING.getNodes(target, code);
29      }
30  }
