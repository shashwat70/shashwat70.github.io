34  public class UnnecessaryFullyQualifiedNameRule extends AbstractJavaRule {
35  
36      private List<ASTImportDeclaration> imports = new ArrayList<>();
37      private String currentPackage;
38  
39      public UnnecessaryFullyQualifiedNameRule() {
40          super.addRuleChainVisit(ASTPackageDeclaration.class);
41          super.addRuleChainVisit(ASTImportDeclaration.class);
42          super.addRuleChainVisit(ASTClassOrInterfaceType.class);
43          super.addRuleChainVisit(ASTName.class);
44      }
45  
46      @Override
47      public void start(final RuleContext ctx) {
48          imports.clear();
49          currentPackage = null;
50      }
51  
52      @Override
53      public Object visit(ASTPackageDeclaration node, Object data) {
54          currentPackage = node.getPackageNameImage();
55          return data;
56      }
57  
58      @Override
59      public Object visit(ASTImportDeclaration node, Object data) {
60          imports.add(node);
61          return data;
62      }
63  
64      @Override
65      public Object visit(ASTClassOrInterfaceType node, Object data) {
66          // This name has no qualification, it can't be unnecessarily qualified
67          if (node.getImage().indexOf('.') < 0) {
68              return data;
69          }
70          checkImports(node, data);
71          return data;
72      }
73  
74      @Override
75      public Object visit(ASTName node, Object data) {
76          if (!(node.getParent() instanceof ASTImportDeclaration)
77                  && !(node.getParent() instanceof ASTPackageDeclaration)) {
78              // This name has no qualification, it can't be unnecessarily qualified
79              if (node.getImage().indexOf('.') < 0) {
80                  return data;
81              }
82              checkImports(node, data);
83          }
84          return data;
85      }

88      /**
89       * Returns true if the name could be imported by this declaration.
90       * The name must be fully qualified, the import is either on-demand
91       * or static, that is its {@link ASTImportDeclaration#getImportedName()}
92       * is the enclosing package or type name of the imported type or static member.
93       */
94      private boolean declarationMatches(ASTImportDeclaration decl, String name) {
95          return name.startsWith(decl.getImportedName())
96                  && name.lastIndexOf('.') == decl.getImportedName().length();
97      }

99      private boolean couldBeMethodCall(JavaNode node) {
100         if (node.getNthParent(2) instanceof ASTPrimaryExpression && node.getNthParent(1) instanceof ASTPrimaryPrefix) {
101             int nextSibling = node.getParent().getIndexInParent() + 1;
102             if (node.getNthParent(2).getNumChildren() > nextSibling) {
103                 return node.getNthParent(2).getChild(nextSibling) instanceof ASTPrimarySuffix;
104             }
105         }
106         return false;
107     }

109     private void checkImports(TypeNode node, Object data) {
110         final String name = node.getImage();
111 
112         // variable names shadow everything else
113         // If the first segment is a variable, then all
114         // the following are field accesses and it's not an FQCN
115         if (isVariable(node.getScope(), name)) {
116             return;
117         }
118 
119         List<ASTImportDeclaration> matches = new ArrayList<>();
120 
121         // Find all "matching" import declarations
122         for (ASTImportDeclaration importDeclaration : imports) {
123             if (!importDeclaration.isImportOnDemand()) {
124                 // Exact match of imported class
125                 if (name.equals(importDeclaration.getImportedName())) {
126                     matches.add(importDeclaration);
127                     continue;
128                 }
129             }
130             // On demand import exactly matches the package of the type
131             // Or match of static method call on imported class
132             if (declarationMatches(importDeclaration, name)) {
133                 matches.add(importDeclaration);
134             }
135         }
136 
137         // If there is no direct match, consider if we match the tail end of a
138         // direct static import, but also a static method on a class import.
139         // For example:
140         //
141         // import java.util.Arrays;
142         // import static java.util.Arrays.asList;
143         // static {
144         // List list1 = Arrays.asList("foo"); // Array class name not needed!
145         // List list2 = asList("foo"); // Preferred, used static import
146         // }
147         //
148         // Or: The usage of a FQN is correct, if there is another import with the same class.
149         // Example
150         // import foo.String;
151         // static {
152         // java.lang.String s = "a";
153         // }
154         if (matches.isEmpty()) {
155             for (ASTImportDeclaration importDeclaration : imports) {
156                 String[] importParts = importDeclaration.getImportedName().split("\\.");
157                 String[] nameParts = name.split("\\.");
158                 if (importDeclaration.isStatic()) {
159                     if (importDeclaration.isImportOnDemand()) {
160                         // Name class part matches class part of static import?
161                         if (nameParts[nameParts.length - 2].equals(importParts[importParts.length - 1])) {
162                             matches.add(importDeclaration);
163                         }
164                     } else {
165                         // Last 2 parts match? Class + Method name
166                         if (nameParts[nameParts.length - 1].equals(importParts[importParts.length - 1])
167                                 && nameParts[nameParts.length - 2].equals(importParts[importParts.length - 2])) {
168                             matches.add(importDeclaration);
169                         }
170                     }
171                 } else if (!importDeclaration.isImportOnDemand()) {
172                     // last part matches?
173                     if (nameParts[nameParts.length - 1].equals(importParts[importParts.length - 1])) {
174                         matches.add(importDeclaration);
175                     } else if (couldBeMethodCall(node)
176                             && nameParts.length > 1 && nameParts[nameParts.length - 2].equals(importParts[importParts.length - 1])) {
177                         // maybe the Name is part of a method call, then the second two last part needs to match
178                         matches.add(importDeclaration);
179                     }
180                 }
181             }
182         }
183 
184         if (matches.isEmpty()) {
185             if (isJavaLangImplicit(node)) {
186                 addViolation(data, node, new Object[] { node.getImage(), "java.lang.*", "implicit "});
187             } else if (isSamePackage(node, name)) {
188                 addViolation(data, node, new Object[] { node.getImage(), currentPackage + ".*", "same package "});
189             }
190         } else {
191             ASTImportDeclaration firstMatch = findFirstMatch(matches);
192 
193             if (!isReferencingInnerNonStaticClass(name, firstMatch)
194                     && !isAvoidingConflict(node, name, firstMatch)) {
195 
196                 String importStr = firstMatch.getImportedName() + (firstMatch.isImportOnDemand() ? ".*" : "");
197                 String type = firstMatch.isStatic() ? "static " : "";
198 
199                 addViolation(data, node, new Object[] { node.getImage(), importStr, type });
200             }
201         }
202     }

204     private ASTImportDeclaration findFirstMatch(List<ASTImportDeclaration> imports) {
205         // first search only static imports
206         ASTImportDeclaration result = null;
207         for (ASTImportDeclaration importDeclaration : imports) {
208             if (importDeclaration.isStatic()) {
209                 result = importDeclaration;
210                 break;
211             }
212         }
213 
214         // then search all non-static, if needed
215         if (result == null) {
216             for (ASTImportDeclaration importDeclaration : imports) {
217                 if (!importDeclaration.isStatic()) {
218                     result = importDeclaration;
219                     break;
220                 }
221             }
222         }
223 
224         return result;
225     }

227     private boolean isVariable(Scope scope, String name) {
228         String firstSegment = name.substring(0, name.indexOf('.'));
229 
230         while (scope != null) {
231 
232             for (Entry<VariableNameDeclaration, List<NameOccurrence>> entry : scope.getDeclarations(VariableNameDeclaration.class).entrySet()) {
233                 if (entry.getKey().getName().equals(firstSegment)) {
234                     return true;
235                 }
236             }
237 
238             scope = scope.getParent();
239         }
240 
241         return false;
242     }

244     private boolean isSamePackage(TypeNode node, String name) {
245         if (node.getType() != null) {
246             // with type resolution we can do an exact package match
247             Package packageOfType = node.getType().getPackage();
248             if (packageOfType != null) {
249 
250                 // get "package" candidate from name
251                 int i = name.lastIndexOf('.');
252                 if (i > 0) {
253                     name = name.substring(0, i);
254                 }
255 
256                 return node.getType().getPackage().getName().equals(currentPackage)
257                         && name.equals(currentPackage);
258             }
259         }
260 
261         int i = name.lastIndexOf('.');
262         if (i > 0) {
263             name = name.substring(0, i);
264             if (name.equals(currentPackage)) {
265                 return true;
266             }
267         }
268         // if it is a name used inside a primary prefix, then it is ambiguous, whether it references
269         // a type or a field or a type inside a subpackage
270         // we assume here, it won't be a subpackage the name references a field, e.g.
271         // package a;
272         // name: a.b.c.d(); -> we assume, b is a class, c is a field, d is a method.
273         // but it could very well be, that: a.b is a package and c is a class, d is a (static) method.
274         if (node.getParent() instanceof ASTPrimaryPrefix
275                 || node.getParent() instanceof ASTNameList
276                 || node instanceof ASTClassOrInterfaceType) {
277             return currentPackage != null && name.startsWith(currentPackage);
278         }
279 
280         return false;
281     }

283     private boolean isJavaLangImplicit(TypeNode node) {
284         String name = node.getImage();
285         boolean isJavaLang = name != null && name.startsWith("java.lang.");
286 
287         if (isJavaLang && node.getType() != null && node.getType().getPackage() != null) {
288             // valid would be ProcessBuilder.Redirect.PIPE but not java.lang.ProcessBuilder.Redirect.PIPE
289             String packageName = node.getType().getPackage() // package might be null, if type is an array type...
290                     .getName();
291             return "java.lang".equals(packageName);
292         } else if (isJavaLang) {
293             // only java.lang.* is implicitly imported, but not e.g. java.lang.reflection.*
294             return StringUtils.countMatches(name, '.') == 2;
295         }
296         return false;
297     }

299     private boolean isReferencingInnerNonStaticClass(final String name, final ASTImportDeclaration firstMatch) {
300         if (firstMatch.isImportOnDemand() && firstMatch.isStatic() && firstMatch.getType() != null) {
301             String[] nameParts = name.split("\\.");
302             String[] importParts = firstMatch.getImportedName().split("\\.");
303 
304             if (nameParts.length == 2 && importParts[importParts.length - 1].equals(nameParts[0])) {
305                 Class<?>[] declaredClasses = firstMatch.getType().getDeclaredClasses();
306                 for (Class<?> innerClass : declaredClasses) {
307                     if (nameParts[1].equals(innerClass.getSimpleName()) && (innerClass.getModifiers() & Modifier.STATIC) != Modifier.STATIC) {
308                         // the referenced inner class is not static, therefore the static import on demand doesn't match
309                         return true;
310                     }
311                 }
312             }
313         }
314         return false;
315     }

317     private boolean isAvoidingConflict(final TypeNode node, final String name,
318                                        final ASTImportDeclaration firstMatch) {
319         // is it a conflict between different imports?
320         if (firstMatch.isImportOnDemand() && firstMatch.isStatic()) {
321             final String methodCalled = name.substring(name.indexOf('.') + 1);
322 
323             // Is there any other static import conflictive?
324             for (final ASTImportDeclaration importDeclaration : imports) {
325                 if (!Objects.equals(importDeclaration, firstMatch) && importDeclaration.isStatic()) {
326                     if (declarationMatches(firstMatch, importDeclaration.getImportedName())) {
327                         // A conflict against the same class is not an excuse,
328                         // ie:
329                         // import java.util.Arrays;
330                         // import static java.util.Arrays.asList;
331                         continue;
332                     }
333 
334                     if (importDeclaration.isImportOnDemand()) {
335                         // We need type resolution to make sure there is a
336                         // conflicting method
337                         if (importDeclaration.getType() != null) {
338                             for (final Method m : importDeclaration.getType().getMethods()) {
339                                 if (m.getName().equals(methodCalled)) {
340                                     return true;
341                                 }
342                             }
343                         }
344                     } else if (importDeclaration.getImportedName().endsWith(methodCalled)) {
345                         return true;
346                     }
347                 }
348             }
349         }
350 
351         final String unqualifiedName = name.substring(name.lastIndexOf('.') + 1);
352         final int unqualifiedNameLength = unqualifiedName.length();
353 
354         // There could be a conflict between an import on demand and another import, e.g.
355         // import One.*;
356         // import Two.Problem;
357         // Where One.Problem is a legitimate qualification
358         if (firstMatch.isImportOnDemand() && !firstMatch.isStatic()) {
359             for (ASTImportDeclaration importDeclaration : imports) {
360                 if (importDeclaration != firstMatch     // NOPMD
361                         && !importDeclaration.isStatic()
362                         && !importDeclaration.isImportOnDemand()) {
363 
364                     // Duplicate imports are legal
365                     if (!importDeclaration.getPackageName().equals(firstMatch.getPackageName())
366                             && importDeclaration.getImportedSimpleName().equals(unqualifiedName)) {
367                         return true;
368                     }
369                 }
370             }
371         }
372 
373         // There could be a conflict between an import of a class with the same name as the FQN
374         String importName = firstMatch.getImportedName();
375         String importUnqualified = importName.substring(importName.lastIndexOf('.') + 1);
376         if (!firstMatch.isImportOnDemand() && !firstMatch.isStatic()) {
377             // the package is different, but the unqualified name is same
378             if (!firstMatch.getImportedName().equals(name) && importUnqualified.equals(unqualifiedName)) {
379                 return true;
380             }
381         }
382 
383         // There could be a conflict between an import of a class with the same name as the FQN, which
384         // could be a method call:
385         // import x.y.Thread;
386         // valid qualification (node): java.util.Thread.currentThread()
387         if (couldBeMethodCall(node)) {
388             String[] nameParts = name.split("\\.");
389             String fqnName = name.substring(0, name.lastIndexOf('.'));
390             // seems to be a static method call on a different FQN
391             if (!fqnName.equals(importName) && !firstMatch.isStatic() && !firstMatch.isImportOnDemand()
392                     && nameParts.length > 1 && nameParts[nameParts.length - 2].equals(importUnqualified)) {
393                 return true;
394             }
395         }
396 
397         // Is it a conflict with a class in the same file?
398         final Set<String> qualifiedTypes = node.getScope().getEnclosingScope(SourceFileScope.class)
399                                                .getQualifiedTypeNames().keySet();
400         for (final String qualified : qualifiedTypes) {
401             int fullLength = qualified.length();
402             if (qualified.endsWith(unqualifiedName)
403                     && (fullLength == unqualifiedNameLength || qualified.charAt(fullLength - unqualifiedNameLength - 1) == '.')) {
404                 return true;
405             }
406         }
407 
408         return false;
409     }
