5   package net.sourceforge.pmd.lang.apex.rule.design;
6   
7   
8   import static net.sourceforge.pmd.properties.constraints.NumericConstraints.positive;
9   
10  import java.util.Stack;
11  
12  import net.sourceforge.pmd.lang.apex.ast.ASTMethod;
13  import net.sourceforge.pmd.lang.apex.ast.ASTUserClass;
14  import net.sourceforge.pmd.lang.apex.ast.ASTUserTrigger;
15  import net.sourceforge.pmd.lang.apex.metrics.ApexMetrics;
16  import net.sourceforge.pmd.lang.apex.metrics.api.ApexClassMetricKey;
17  import net.sourceforge.pmd.lang.apex.metrics.api.ApexOperationMetricKey;
18  import net.sourceforge.pmd.lang.apex.rule.AbstractApexRule;
19  import net.sourceforge.pmd.lang.metrics.ResultOption;
20  import net.sourceforge.pmd.properties.PropertyDescriptor;
21  import net.sourceforge.pmd.properties.PropertyFactory;
22  
23  
24  /**
25   * Cyclomatic complexity rule using metrics. Uses Wmc to report classes.
26   * 
27   * @author Cl√©ment Fournier
28   */
29  public class CyclomaticComplexityRule extends AbstractApexRule {
30  
31      private static final PropertyDescriptor<Integer> CLASS_LEVEL_DESCRIPTOR
32          = PropertyFactory.intProperty("classReportLevel")
33                           .desc("Total class complexity reporting threshold")
34                           .require(positive())
35                           .defaultValue(40)
36                           .build();
37  
38      private static final PropertyDescriptor<Integer> METHOD_LEVEL_DESCRIPTOR
39          = PropertyFactory.intProperty("methodReportLevel")
40                           .desc("Cyclomatic complexity reporting threshold")
41                           .require(positive())
42                           .defaultValue(10)
43                           .build();
44  
45      private Stack<String> classNames = new Stack<>();
46      private boolean inTrigger;
47  
48  
49      public CyclomaticComplexityRule() {
50          definePropertyDescriptor(CLASS_LEVEL_DESCRIPTOR);
51          definePropertyDescriptor(METHOD_LEVEL_DESCRIPTOR);
52      }
53  
54  
55      @Override
56      public Object visit(ASTUserTrigger node, Object data) {
57          inTrigger = true;
58          super.visit(node, data);
59          inTrigger = false;
60          return data;
61      }
62  
63  
64      @Override
65      public Object visit(ASTUserClass node, Object data) {
66  
67          classNames.push(node.getImage());
68          super.visit(node, data);
69          classNames.pop();
70  
71          if (ApexClassMetricKey.WMC.supports(node)) {
72              int classWmc = (int) ApexMetrics.get(ApexClassMetricKey.WMC, node);
73  
74              if (classWmc >= getProperty(CLASS_LEVEL_DESCRIPTOR)) {
75                  int classHighest = (int) ApexMetrics.get(ApexOperationMetricKey.CYCLO, node, ResultOption.HIGHEST);
76  
77                  String[] messageParams = {"class",
78                                            node.getImage(),
79                                            " total",
80                                            classWmc + " (highest " + classHighest + ")", };
81  
82                  addViolation(data, node, messageParams);
83              }
84          }
85          return data;
86      }
87  
88  
89      @Override
90      public final Object visit(ASTMethod node, Object data) {
91  
92          int cyclo = (int) ApexMetrics.get(ApexOperationMetricKey.CYCLO, node);
93          if (cyclo >= getProperty(METHOD_LEVEL_DESCRIPTOR)) {
94              String opType = inTrigger ? "trigger"
95                                        : node.getImage().equals(classNames.peek()) ? "constructor"
96                                                                                    : "method";
97  
98              addViolation(data, node, new String[]{opType,
99                                                    node.getQualifiedName().getOperation(),
100                                                   "",
101                                                   "" + cyclo, });
102         }
103 
104         return data;
105     }
106 
107 }
