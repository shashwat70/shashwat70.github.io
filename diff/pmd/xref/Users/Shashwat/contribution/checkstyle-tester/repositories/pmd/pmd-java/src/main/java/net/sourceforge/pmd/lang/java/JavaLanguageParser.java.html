7   import java.io.Reader;

9   import net.sourceforge.pmd.annotation.InternalApi;
10  import net.sourceforge.pmd.lang.LanguageVersionHandler;
11  import net.sourceforge.pmd.lang.ParserOptions;
12  import net.sourceforge.pmd.lang.java.ast.JavaParser;
13  import net.sourceforge.pmd.lang.java.ast.ParseException;
14  
15  /**
16   * Adapter for the JavaParser, using the specified grammar version.
17   *
18   * @author Pieter_Van_Raemdonck - Application Engineers NV/SA - www.ae.be
19   * @author Andreas Dangel
20   *
21   * @deprecated This is internal API, use {@link LanguageVersionHandler#getParser(ParserOptions)}.
22   */
23  @InternalApi
24  @Deprecated
25  public class JavaLanguageParser extends AbstractJavaParser {
26      private final int jdkVersion;
27      private final boolean preview;
28  
29      public JavaLanguageParser(int jdkVersion, ParserOptions parserOptions) {
30          this(jdkVersion, false, parserOptions);
31      }
32  
33      public JavaLanguageParser(int jdkVersion, boolean preview, ParserOptions parserOptions) {
34          super(parserOptions);
35          this.jdkVersion = jdkVersion;
36          this.preview = preview;
37      }
38  
39      @Override
40      protected JavaParser createJavaParser(Reader source) throws ParseException {
41          JavaParser javaParser = super.createJavaParser(source);
42          javaParser.setJdkVersion(jdkVersion);
43          javaParser.setPreview(preview);
44          return javaParser;
45      }
46  }
