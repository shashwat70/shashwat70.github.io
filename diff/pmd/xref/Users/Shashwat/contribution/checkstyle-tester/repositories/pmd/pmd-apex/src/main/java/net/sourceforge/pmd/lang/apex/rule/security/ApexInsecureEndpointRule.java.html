5   package net.sourceforge.pmd.lang.apex.rule.security;
6   
7   import java.util.HashSet;
8   import java.util.Set;
9   import java.util.regex.Pattern;
10  
11  import net.sourceforge.pmd.lang.apex.ast.ASTAssignmentExpression;
12  import net.sourceforge.pmd.lang.apex.ast.ASTBinaryExpression;
13  import net.sourceforge.pmd.lang.apex.ast.ASTFieldDeclaration;
14  import net.sourceforge.pmd.lang.apex.ast.ASTLiteralExpression;
15  import net.sourceforge.pmd.lang.apex.ast.ASTMethodCallExpression;
16  import net.sourceforge.pmd.lang.apex.ast.ASTVariableDeclaration;
17  import net.sourceforge.pmd.lang.apex.ast.ASTVariableExpression;
18  import net.sourceforge.pmd.lang.apex.ast.AbstractApexNode;
19  import net.sourceforge.pmd.lang.apex.rule.AbstractApexRule;
20  import net.sourceforge.pmd.lang.apex.rule.internal.Helper;
21  
22  /**
23   * Insecure HTTP endpoints passed to (req.setEndpoint)
24   * req.setHeader('Authorization') should use named credentials
25   * 
26   * @author sergey.gorbaty
27   *
28   */
29  public class ApexInsecureEndpointRule extends AbstractApexRule {
30      private static final String SET_ENDPOINT = "setEndpoint";
31      private static final Pattern PATTERN = Pattern.compile("^http://.+?$", Pattern.CASE_INSENSITIVE);
32  
33      private final Set<String> httpEndpointStrings = new HashSet<>();
34  
35      public ApexInsecureEndpointRule() {
36          setProperty(CODECLIMATE_CATEGORIES, "Security");
37          setProperty(CODECLIMATE_REMEDIATION_MULTIPLIER, 100);
38          setProperty(CODECLIMATE_BLOCK_HIGHLIGHTING, false);
39      }
40  
41      @Override
42      public Object visit(ASTAssignmentExpression node, Object data) {
43          findInsecureEndpoints(node);
44          return data;
45      }
46  
47      @Override
48      public Object visit(ASTVariableDeclaration node, Object data) {
49          findInsecureEndpoints(node);
50          return data;
51      }
52  
53      @Override
54      public Object visit(ASTFieldDeclaration node, Object data) {
55          findInsecureEndpoints(node);
56          return data;
57      }
58  
59      private void findInsecureEndpoints(AbstractApexNode<?> node) {
60          ASTVariableExpression variableNode = node.getFirstChildOfType(ASTVariableExpression.class);
61          findInnerInsecureEndpoints(node, variableNode);
62  
63          ASTBinaryExpression binaryNode = node.getFirstChildOfType(ASTBinaryExpression.class);
64          if (binaryNode != null) {
65              findInnerInsecureEndpoints(binaryNode, variableNode);
66          }
67  
68      }
69  
70      private void findInnerInsecureEndpoints(AbstractApexNode<?> node, ASTVariableExpression variableNode) {
71          ASTLiteralExpression literalNode = node.getFirstChildOfType(ASTLiteralExpression.class);
72  
73          if (literalNode != null && variableNode != null) {
74              if (literalNode.isString()) {
75                  String literal = literalNode.getImage();
76                  if (PATTERN.matcher(literal).matches()) {
77                      httpEndpointStrings.add(Helper.getFQVariableName(variableNode));
78                  }
79              }
80          }
81      }
82  
83      @Override
84      public Object visit(ASTMethodCallExpression node, Object data) {
85          processInsecureEndpoint(node, data);
86          return data;
87      }
88  
89      private void processInsecureEndpoint(ASTMethodCallExpression node, Object data) {
90          if (!Helper.isMethodName(node, SET_ENDPOINT)) {
91              return;
92          }
93  
94          ASTBinaryExpression binaryNode = node.getFirstChildOfType(ASTBinaryExpression.class);
95          if (binaryNode != null) {
96              runChecks(binaryNode, data);
97          }
98  
99          runChecks(node, data);
100 
101     }
102 
103     private void runChecks(AbstractApexNode<?> node, Object data) {
104         ASTLiteralExpression literalNode = node.getFirstChildOfType(ASTLiteralExpression.class);
105         if (literalNode != null && literalNode.isString()) {
106             String literal = literalNode.getImage();
107             if (PATTERN.matcher(literal).matches()) {
108                 addViolation(data, literalNode);
109             }
110         }
111 
112         ASTVariableExpression variableNode = node.getFirstChildOfType(ASTVariableExpression.class);
113         if (variableNode != null) {
114             if (httpEndpointStrings.contains(Helper.getFQVariableName(variableNode))) {
115                 addViolation(data, variableNode);
116             }
117 
118         }
119     }
120 }
