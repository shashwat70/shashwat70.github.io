5   package net.sourceforge.pmd;

7   import net.sourceforge.pmd.properties.StringMultiProperty;
8   import net.sourceforge.pmd.properties.StringProperty;
9   
10  /**
11   * Sample rule that detect any node with an image of "Foo", similar to
12   * {@link FooRule}. It additionally has some properties in order to test the
13   * renderers. Used for testing.
14   */
15  public class RuleWithProperties extends FooRule {
16      public static final StringProperty STRING_PROPERTY_DESCRIPTOR = new StringProperty("stringProperty",
17              "simple string property", null, 1.0f);
18      public static final StringMultiProperty MULTI_STRING_PROPERTY_DESCRIPTOR = new StringMultiProperty("multiString",
19              "multi string property", new String[] { "default1", "default2" }, 2.0f, ',');
20  
21      public RuleWithProperties() {
22          definePropertyDescriptor(STRING_PROPERTY_DESCRIPTOR);
23          definePropertyDescriptor(MULTI_STRING_PROPERTY_DESCRIPTOR);
24      }
25  }
