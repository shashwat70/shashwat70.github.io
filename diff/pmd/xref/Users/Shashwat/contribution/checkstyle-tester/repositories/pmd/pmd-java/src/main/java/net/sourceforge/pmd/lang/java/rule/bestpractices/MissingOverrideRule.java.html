102     /**
103      * Returns a map of method name to methods with the same name (overloads).
104      * The map contains a MethodWrapper for all methods declared in this class.
105      *
106      * @param exploredType Type to explore
107      */
108     private MethodLookup getMethodLookup(Class<?> exploredType) {
109         if (exploredType == null) {
110             return null;
111         }
112 
113         try {
114             Set<Method> overridden = overriddenMethods(exploredType);
115             Map<String, Map<Integer, List<Method>>> result = new HashMap<>();
116 
117             for (Method m : exploredType.getDeclaredMethods()) {
118                 if (!result.containsKey(m.getName())) {
119                     result.put(m.getName(), new HashMap<Integer, List<Method>>());
120                 }
121 
122                 Map<Integer, List<Method>> pCountToOverloads = result.get(m.getName());
123 
124                 int paramCount = m.getParameterTypes().length;
125                 if (!pCountToOverloads.containsKey(paramCount)) {
126                     pCountToOverloads.put(paramCount, new ArrayList<Method>());
127                 }
128 
129                 pCountToOverloads.get(paramCount).add(m);
130             }
131 
132             return new MethodLookup(result, overridden);
133         } catch (final LinkageError e) {
134             // we may have an incomplete auxclasspath
135             return null;
136         }
137     }

140     /**
141      * Returns the set of methods declared in this type that are overridden.
142      *
143      * @param exploredType The type to explore
144      */
145     private Set<Method> overriddenMethods(Class<?> exploredType) {
146         return overriddenMethodsRec(exploredType, true, new HashSet<>(Arrays.asList(exploredType.getDeclaredMethods())), new HashSet<Method>(), new HashSet<Class<?>>(), false);
147     }

150     private Set<Method> overriddenMethodsRec(Class<?> exploredType, boolean skip, Set<Method> candidates, Set<Method> result, Set<Class<?>> alreadyExplored, boolean onlyPublic) {
151 
152         if (candidates.isEmpty() || alreadyExplored.contains(exploredType)) {
153             return result;
154         }
155 
156         alreadyExplored.add(exploredType);
157 
158         if (!skip) {
159             Set<Method> toRemove = new HashSet<>();
160             for (Method dm : exploredType.getDeclaredMethods()) {
161                 if (onlyPublic && !Modifier.isPublic(dm.getModifiers())
162                         || Modifier.isPrivate(dm.getModifiers())
163                         || Modifier.isStatic(dm.getModifiers())) {
164                     continue;
165                 }
166 
167                 for (Method cand : candidates) {
168                     if (Modifier.isPrivate(dm.getModifiers()) || Modifier.isStatic(dm.getModifiers())) {
169                         continue;
170                     }
171 
172                     if (cand.getName().equals(dm.getName()) && Arrays.equals(cand.getParameterTypes(), dm.getParameterTypes())) {
173                         // cand is overriden
174                         result.add(cand);
175                         toRemove.add(cand);
176                         // Several methods are eligible, because of return type covariance
177                         // We could do away with adding only the first one to the result,
178                         // but then the other would stay in the candidates and we'd explore
179                         // the rest of the tree unnecessarily
180                     }
181                 }
182                 candidates.removeAll(toRemove); // no need to look for it elsewhere
183             }
184         }
185 
186         if (candidates.isEmpty()) {
187             return result;
188         }
189 
190         Class<?> superClass = exploredType.getSuperclass();
191         if (superClass != null) {
192             overriddenMethodsRec(superClass, false, candidates, result, alreadyExplored, false);
193         }
194 
195         for (Class<?> iface : exploredType.getInterfaces()) {
196             overriddenMethodsRec(iface, false, candidates, result, alreadyExplored, false);
197         }
198 
199         if (exploredType.isInterface() && exploredType.getInterfaces().length == 0) {
200             // implicit public object member declarations
201             if (!alreadyExplored.contains(Object.class)) {
202                 overriddenMethodsRec(Object.class, false, candidates, result, alreadyExplored, true);
203             }
204         }
205 
206         return result;
207     }

210     @Override
211     public Object visit(ASTMethodDeclaration node, Object data) {
212         if (currentLookup.peek() == null) {
213             return super.visit(node, data);
214         }
215 
216         for (ASTAnnotation annot : node.getDeclaredAnnotations()) {
217             if (Override.class.equals(annot.getType())) {
218                 // we assume the compiler has already checked it, so it's correct
219                 return super.visit(node, data);
220             }
221         }
222 
223         try {
224             boolean overridden = currentLookup.peek().isOverridden(node.getName(), node.getFormalParameters());
225             if (overridden) {
226                 addViolation(data, node, new Object[]{PrettyPrintingUtil.displaySignature(node)});
227             }
228         } catch (NoSuchMethodException e) {
229             // may happen in the body of an enum constant,
230             // because the method lookup used is the one of
231             // the parent class.
232             LOG.fine("MissingOverride encountered unexpected method " + node.getName());
233             // throw new RuntimeException(e); // uncomment when enum constants are handled by typeres
234         }
235         return super.visit(node, data);
236     }

240     private static class MethodLookup {
241 
242         // method name, parameter count, methods
243         private final Map<String, Map<Integer, List<Method>>> map;
244         private final Set<Method> overridden;
245 
246 
247         private MethodLookup(Map<String, Map<Integer, List<Method>>> map, Set<Method> overridden) {
248             this.map = map;
249             this.overridden = overridden;
250 
251             for (Map<Integer, List<Method>> overloadSet : map.values()) {
252                 for (Entry<Integer, List<Method>> sameParamCountMethods : overloadSet.entrySet()) {
253                     // bridges have the same name and param count as the bridged method
254                     resolveBridges(sameParamCountMethods.getValue());
255                 }
256             }
257         }
258 
259 
260         private void resolveBridges(List<Method> overloads) {
261             if (overloads.size() <= 1) {
262                 return;
263             }
264 
265             // partition the overloads
266             List<Method> bridges = new ArrayList<>();
267             List<Method> notBridges = new ArrayList<>();
268 
269             for (Method m : overloads) {
270                 if (m.isBridge()) {
271                     bridges.add(m);
272                 } else {
273                     notBridges.add(m);
274                 }
275             }
276 
277             if (bridges.isEmpty()) {
278                 return;
279             }
280 
281             // each bridge necessarily calls another non-bridge method, which is overridden
282 
283             if (notBridges.size() == bridges.size()) {
284                 // This is a good heuristic, though not perfect.
285 
286                 // Most of the time, bridges is one-to-one to notBridges, and we can safely assume that
287                 // all non-bridge methods are overridden, since they need a bridge method
288 
289                 // This chokes on overloads which don't override a previous definition, in which case there's no
290                 // generated bridge for that overload. Short of statically analysing type parameters, or reading
291                 // the bytecode to find the delegation call, we have no way to find out which of the overloads is
292                 // overridden. An example of that is in RuleViolationComparator: there's one bridge
293                 // compare(Object, Object) for the overload compare(RV, RV), but we can't know because there's
294                 // another overload, compare(String, String) which is equally eligible
295 
296                 // It's also possible that several bridges are generated for the same method, when the method
297                 // was already redefined several times with a different bound. An example of that is in
298                 // PropertyDescriptorConversionWrapper.SingleValue.Packaged and similar subclasses: each of the
299                 // levels of the hierarchy redefine the original method with a tighter bound on the type parameter.
300                 // Three bridges are generated for populate() on concrete builder classes.
301 
302                 // The two situations could happen together, and if they balance out, that gives a false positive
303                 // with the current test (size are equal). If they don't balance out, then we don't report anything,
304                 // which is why this test seems the safest.
305 
306                 // Depending on the real-world frequency of those two situations happening together, we may rather
307                 // use notBridges.size() <= bridges.size(), to remove FNs caused by additional bridges.
308 
309                 overridden.addAll(notBridges);
310             }
311 
312             overloads.removeAll(bridges); // better prune the candidate overloads anyway
313         }

319         private List<Method> getMethods(String name, int paramCount) throws NoSuchMethodException {
320             Map<Integer, List<Method>> overloads = map.get(name);
321             if (overloads == null) {
322                 throw new NoSuchMethodException(name);
323             }
324 
325             List<Method> methods = overloads.get(paramCount);
326             if (methods == null || methods.isEmpty()) {
327                 throw new NoSuchMethodException(name);
328             }
329             return methods;
330         }
331 
332         /**
333          * Tries to determine if the method with the given name and parameter count is overridden
334          *
335          * @return True or false
336          *
337          * @throws NoSuchMethodException if no method is registered with this name and paramcount, which is a bug
338          */
339         boolean isOverridden(String name, ASTFormalParameters params) throws NoSuchMethodException {
340             List<Method> methods = getMethods(name, params.getParameterCount());
341 
342             if (methods.size() == 1) { // only one method with this name and parameter count, we can conclude
343                 return overridden.contains(methods.get(0));
344             } else { // several overloads with same name and count, cannot be determined without comparing parameters
345                 Class<?>[] paramTypes = getParameterTypes(params);
346                 if (paramTypes == null) {
347                     return false;
348                 }
349                 for (Method m : methods) {
350                     if (Arrays.equals(m.getParameterTypes(), paramTypes)) {
351                         // we found our overload
352                         return overridden.contains(m);
353                     }
354                 }
355                 return false;
356             }
357         }
358 
359 
360         private static Class<?>[] getParameterTypes(ASTFormalParameters params) {
361             Class<?>[] paramTypes = new Class[params.getParameterCount()];
362             int i = 0;
363             for (ASTFormalParameter p : params) {
364                 Class<?> pType = p.getType();
365                 if (pType == null) {
366                     // fail, couldn't resolve one parameter
367                     return null;
368                 }
369 
370                 paramTypes[i++] = pType;
371             }
372             return paramTypes;
373         }
