5   package net.sourceforge.pmd.lang.java.multifile.signature;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   
10  import net.sourceforge.pmd.lang.java.ast.ASTFieldDeclaration;
11  
12  /**
13   * Signature for a field.
14   *
15   * @author Cl√©ment Fournier
16   */
17  public final class JavaFieldSignature extends JavaSignature<ASTFieldDeclaration> {
18  
19      private static final Map<Integer, JavaFieldSignature> POOL = new HashMap<>();
20  
21      public final boolean isStatic;
22      public final boolean isFinal;
23  
24  
25      private JavaFieldSignature(Visibility visibility, boolean isStatic, boolean isFinal) {
26          super(visibility);
27          this.isStatic = isStatic;
28          this.isFinal = isFinal;
29      }
30  
31  
32      @Override
33      public String toString() {
34          return "JavaFieldSignature{"
35              + "isStatic=" + isStatic
36              + ", isFinal=" + isFinal
37              + ", visibility=" + visibility
38              + '}';
39      }
40  
41  
42      @Override
43      public boolean equals(Object o) {
44          return this == o;
45      }
46  
47  
48      @Override
49      public int hashCode() {
50          return code(visibility, isStatic, isFinal);
51      }
52  
53  
54      /** Used internally by the pooler. */
55      private static int code(Visibility visibility, boolean isStatic, boolean isFinal) {
56          return visibility.hashCode() * 31 + (isStatic ? 1 : 0) * 2 + (isFinal ? 1 : 0);
57      }
58  
59  
60      /**
61       * Builds a field signature from its AST node.
62       *
63       * @param node The AST node of the field
64       *
65       * @return The signature of the field
66       */
67      public static JavaFieldSignature buildFor(ASTFieldDeclaration node) {
68          int code = code(Visibility.get(node), node.isStatic(), node.isFinal());
69          if (!POOL.containsKey(code)) {
70              POOL.put(code, new JavaFieldSignature(Visibility.get(node), node.isStatic(), node.isFinal()));
71          }
72          return POOL.get(code);
73      }
74  }
