5   package net.sourceforge.pmd.lang.apex.rule.bestpractices;
6   
7   import net.sourceforge.pmd.lang.apex.ast.ASTMethodCallExpression;
8   import net.sourceforge.pmd.lang.apex.rule.AbstractApexUnitTestRule;
9   
10  public class ApexAssertionsShouldIncludeMessageRule extends AbstractApexUnitTestRule {
11  
12      private static final String ASSERT = "System.assert";
13      private static final String ASSERT_EQUALS = "System.assertEquals";
14      private static final String ASSERT_NOT_EQUALS = "System.assertNotEquals";
15  
16      @Override
17      public Object visit(ASTMethodCallExpression node, Object data) {
18          String methodName = node.getFullMethodName();
19  
20          if (ASSERT.equalsIgnoreCase(methodName) && node.getNumChildren() == 2) {
21              addViolationWithMessage(data, node,
22                      "''{0}'' should have 2 parameters.",
23                      new Object[] { ASSERT });
24          } else if ((ASSERT_EQUALS.equalsIgnoreCase(methodName)
25                  || ASSERT_NOT_EQUALS.equalsIgnoreCase(methodName))
26                  && node.getNumChildren() == 3) {
27              addViolationWithMessage(data, node,
28                      "''{0}'' should have 3 parameters.",
29                      new Object[] { methodName });
30          }
31          return data;
32      }
33  }
