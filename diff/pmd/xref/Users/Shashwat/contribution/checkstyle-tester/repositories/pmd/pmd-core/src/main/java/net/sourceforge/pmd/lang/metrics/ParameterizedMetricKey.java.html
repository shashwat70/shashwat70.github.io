5   package net.sourceforge.pmd.lang.metrics;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   
10  import net.sourceforge.pmd.lang.ast.Node;
11  
12  /**
13   * Represents a key parameterized with its options. Used to index memoization maps.
14   *
15   * @param <N> Type of node on which the memoized metric can be computed
16   *
17   * @author Cl√©ment Fournier
18   * @since 5.8.0
19   */
20  public final class ParameterizedMetricKey<N extends Node> {
21  
22      private static final Map<ParameterizedMetricKey<?>, ParameterizedMetricKey<?>> POOL = new HashMap<>();
23  
24      /** The metric key. */
25      public final MetricKey<N> key;
26      /** The options of the metric. */
27      public final MetricOptions options;
28  
29  
30      /** Used internally by the pooler. */
31      private ParameterizedMetricKey(MetricKey<N> key, MetricOptions options) {
32          this.key = key;
33          this.options = options;
34      }
35  
36  
37      @Override
38      public String toString() {
39          return "ParameterizedMetricKey{key=" + key.name() + ", options=" + options + '}';
40      }
41  
42  
43      @Override
44      public boolean equals(Object o) {
45          return o instanceof ParameterizedMetricKey
46              && ((ParameterizedMetricKey) o).key.equals(key)
47              && ((ParameterizedMetricKey) o).options.equals(options);
48      }
49  
50  
51      @Override
52      public int hashCode() {
53          return 31 * key.hashCode() + options.hashCode();
54      }
55  
56  
57      /**
58       * Builds a parameterized metric key.
59       *
60       * @param key     The key
61       * @param options The options
62       * @param <N>     The type of node of the metric key
63       *
64       * @return An instance of parameterized metric key corresponding to the parameters
65       */
66      @SuppressWarnings("PMD.SingletonClassReturningNewInstance")
67      public static <N extends Node> ParameterizedMetricKey<N> getInstance(MetricKey<N> key, MetricOptions options) {
68          ParameterizedMetricKey<N> tmp = new ParameterizedMetricKey<>(key, options);
69          if (!POOL.containsKey(tmp)) {
70              POOL.put(tmp, tmp);
71          }
72  
73          @SuppressWarnings("unchecked")
74          ParameterizedMetricKey<N> result = (ParameterizedMetricKey<N>) POOL.get(tmp);
75          return result;
76      }
77  }
