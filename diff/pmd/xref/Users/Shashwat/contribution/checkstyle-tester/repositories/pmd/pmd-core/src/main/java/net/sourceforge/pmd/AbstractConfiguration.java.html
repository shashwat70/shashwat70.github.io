5   package net.sourceforge.pmd;
6   
7   import java.nio.charset.Charset;
8   
9   /**
10   * Base configuration class for both PMD and CPD.
11   *
12   * @author Brian Remedios
13   */
14  public abstract class AbstractConfiguration {
15  
16      private Charset sourceEncoding = Charset.forName(System.getProperty("file.encoding"));
17      private boolean debug;
18  
19      /**
20       * Create a new abstract configuration.
21       */
22      protected AbstractConfiguration() {
23          super();
24      }
25  
26      /**
27       * Get the character encoding of source files.
28       *
29       * @return The character encoding.
30       */
31      public Charset getSourceEncoding() {
32          return sourceEncoding;
33      }
34  
35      /**
36       * Set the character encoding of source files.
37       *
38       * @param sourceEncoding
39       *            The character encoding.
40       */
41      public void setSourceEncoding(String sourceEncoding) {
42          this.sourceEncoding = Charset.forName(sourceEncoding);
43      }
44  
45      /**
46       * Return the debug indicator. If this value is <code>true</code> then PMD
47       * will log debug information.
48       *
49       * @return <code>true</code> if debug logging is enabled, <code>false</code>
50       *         otherwise.
51       */
52      public boolean isDebug() {
53          return debug;
54      }
55  
56      /**
57       * Set the debug indicator.
58       *
59       * @param debug
60       *            The debug indicator to set.
61       * @see #isDebug()
62       */
63      public void setDebug(boolean debug) {
64          this.debug = debug;
65      }
66  }
