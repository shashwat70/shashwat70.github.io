54  /**
55   * Makes sure you close your database connections. It does this by looking for
56   * code patterned like this:
57   *
58   * <pre>
59   *  Connection c = X;
60   *  try {
61   *   // do stuff, and maybe catch something
62   *  } finally {
63   *   c.close();
64   *  }
65   * </pre>
66   *
67   *  @author original author unknown
68   *  @author Contribution from Pierre Mathien
69   */
70  public class CloseResourceRule extends AbstractJavaRule {
71  
72      private Set<String> types = new HashSet<>();
73      private Set<String> simpleTypes = new HashSet<>();
74  
75      private Set<String> closeTargets = new HashSet<>();
76      private static final PropertyDescriptor<List<String>> CLOSE_TARGETS_DESCRIPTOR =
77              stringListProperty("closeTargets")
78                             .desc("Methods which may close this resource")
79                             .emptyDefaultValue()
80                             .delim(',').build();
81  
82      private static final PropertyDescriptor<List<String>> TYPES_DESCRIPTOR =
83              stringListProperty("types")
84                      .desc("Affected types")
85                      .defaultValues("java.lang.AutoCloseable", "java.sql.Connection", "java.sql.Statement", "java.sql.ResultSet")
86                      .delim(',').build();
87  
88      private static final PropertyDescriptor<Boolean> USE_CLOSE_AS_DEFAULT_TARGET =
89              booleanProperty("closeAsDefaultTarget")
90                      .desc("Consider 'close' as a target by default").defaultValue(true).build();
91  
92      private static final PropertyDescriptor<List<String>> ALLOWED_RESOURCE_TYPES =
93              stringListProperty("allowedResourceTypes")
94              .desc("Exact class names that do not need to be closed")
95              .defaultValues("java.io.ByteArrayOutputStream", "java.io.ByteArrayInputStream", "java.io.StringWriter",
96                      "java.io.CharArrayWriter", "java.util.stream.Stream")
97              .build();

100     public CloseResourceRule() {
101         definePropertyDescriptor(CLOSE_TARGETS_DESCRIPTOR);
102         definePropertyDescriptor(TYPES_DESCRIPTOR);
103         definePropertyDescriptor(USE_CLOSE_AS_DEFAULT_TARGET);
104         definePropertyDescriptor(ALLOWED_RESOURCE_TYPES);
105     }

107     @Override
108     public void start(RuleContext ctx) {
109         closeTargets.clear();
110         simpleTypes.clear();
111         types.clear();
112 
113         if (getProperty(CLOSE_TARGETS_DESCRIPTOR) != null) {
114             closeTargets.addAll(getProperty(CLOSE_TARGETS_DESCRIPTOR));
115         }
116         if (getProperty(USE_CLOSE_AS_DEFAULT_TARGET) && !closeTargets.contains("close")) {
117             closeTargets.add("close");
118         }
119         if (getProperty(TYPES_DESCRIPTOR) != null) {
120             types.addAll(getProperty(TYPES_DESCRIPTOR));
121             for (String type : getProperty(TYPES_DESCRIPTOR)) {
122                 simpleTypes.add(toSimpleType(type));
123             }
124         }
125     }

127     private static String toSimpleType(String fullyQualifiedClassName) {
128         int lastIndexOf = fullyQualifiedClassName.lastIndexOf('.');
129         if (lastIndexOf > -1) {
130             return fullyQualifiedClassName.substring(lastIndexOf + 1);
131         } else {
132             return fullyQualifiedClassName;
133         }
134     }

148     private void checkForResources(ASTMethodOrConstructorDeclaration node, Object data) {
149         List<ASTLocalVariableDeclaration> localVars = node.findDescendantsOfType(ASTLocalVariableDeclaration.class);
150         List<ASTVariableDeclarator> vars = new ArrayList<>();
151         Map<ASTVariableDeclaratorId, TypeNode> ids = new HashMap<>();
152 
153         // find all variable declarators
154         for (ASTLocalVariableDeclaration localVar : localVars) {
155             vars.addAll(localVar.findChildrenOfType(ASTVariableDeclarator.class));
156         }
157 
158         // find all variable references to Connection objects
159         for (ASTVariableDeclarator var : vars) {
160             // get the type of the local var declaration
161             TypeNode type = ((ASTLocalVariableDeclaration) var.getParent()).getTypeNode();
162 
163             if (type != null && isResourceTypeOrSubtype(type)) {
164                 if (var.hasInitializer()) {
165                     // figure out the runtime type. If the variable is initialized, take the type from there
166                     ASTExpression expression = var.getInitializer().getFirstChildOfType(ASTExpression.class);
167                     TypeNode runtimeType = expression;
168                     if (!isMethodCall(expression) && runtimeType != null && runtimeType.getType() != null) {
169                         type = runtimeType;
170                     }
171 
172                     // consider cases, when the streams are chained
173                     // assumes, that the underlaying stream is always the first argument in the
174                     // constructor call.
175                     ASTExpression firstArgument = getAllocationFirstArgument(expression);
176                     if (firstArgument != null) {
177                         type = firstArgument;
178                     }
179                 }
180 
181                 if (!isAllowedResourceType(type) && !isMethodParameter(var, node)) {
182                     ids.put(var.getVariableId(), type);
183                 }
184             }
185         }
186 
187         // if there are closables, ensure each is closed.
188         for (Map.Entry<ASTVariableDeclaratorId, TypeNode> entry : ids.entrySet()) {
189             ASTVariableDeclaratorId variableId = entry.getKey();
190             ensureClosed((ASTLocalVariableDeclaration) variableId.getParent().getParent(), variableId,
191                     entry.getValue(), data);
192         }
193     }

195     /**
196      * Checks whether the variable is initialized from a method parameter.
197      * @param var the variable that is being initialized
198      * @param methodOrCstor the method or constructor in which the variable is declared
199      * @return <code>true</code> if the variable is initialized from a method parameter. <code>false</code>
200      *         otherwise.
201      */
202     private boolean isMethodParameter(ASTVariableDeclarator var, ASTMethodOrConstructorDeclaration methodOrCstor) {
203         if (!var.hasInitializer()) {
204             return false;
205         }
206 
207         boolean result = false;
208         ASTVariableInitializer initializer = var.getInitializer();
209         ASTName name = initializer.getFirstDescendantOfType(ASTName.class);
210         if (name != null) {
211             ASTFormalParameters formalParameters = null;
212             if (methodOrCstor instanceof ASTMethodDeclaration) {
213                 formalParameters = ((ASTMethodDeclaration) methodOrCstor).getFormalParameters();
214             } else if (methodOrCstor instanceof ASTConstructorDeclaration) {
215                 formalParameters = ((ASTConstructorDeclaration) methodOrCstor).getFormalParameters();
216             }
217             if (formalParameters != null) {
218                 List<ASTVariableDeclaratorId> ids = formalParameters.findDescendantsOfType(ASTVariableDeclaratorId.class);
219                 for (ASTVariableDeclaratorId id : ids) {
220                     if (id.hasImageEqualTo(name.getImage()) && isResourceTypeOrSubtype(id)) {
221                         result = true;
222                         break;
223                     }
224                 }
225             }
226         }
227         return result;
228     }

230     private ASTExpression getAllocationFirstArgument(ASTExpression expression) {
231         List<ASTAllocationExpression> allocations = expression.findDescendantsOfType(ASTAllocationExpression.class);
232         ASTExpression firstArgument = null;
233 
234         if (!allocations.isEmpty()) {
235             ASTArgumentList argumentList = allocations.get(allocations.size() - 1).getFirstDescendantOfType(ASTArgumentList.class);
236             if (argumentList != null) {
237                 firstArgument = argumentList.getFirstChildOfType(ASTExpression.class);
238             }
239         }
240 
241         // the argument must not be a literal, it needs to be a Name referring to a variable
242         if (firstArgument != null && firstArgument.getFirstDescendantOfType(ASTName.class) != null) {
243             ASTName name = firstArgument.getFirstDescendantOfType(ASTName.class);
244 
245             Map<VariableNameDeclaration, List<NameOccurrence>> vars = firstArgument.getScope()
246                     .getDeclarations(VariableNameDeclaration.class);
247             for (VariableNameDeclaration nameDecl : vars.keySet()) {
248                 if (nameDecl.getName().equals(name.getImage()) && isResourceTypeOrSubtype(firstArgument)) {
249                     return firstArgument;
250                 }
251             }
252         }
253         return null;
254     }

256     private boolean isMethodCall(ASTExpression expression) {
257         return expression != null
258              && expression.getNumChildren() > 0
259              && expression.getChild(0) instanceof ASTPrimaryExpression
260              && expression.getChild(0).getFirstChildOfType(ASTPrimarySuffix.class) != null;
261     }

263     private boolean isResourceTypeOrSubtype(TypeNode refType) {
264         if (refType.getType() != null) {
265             for (String type : types) {
266                 if (TypeHelper.isA(refType, type)) {
267                     return true;
268                 }
269             }
270         } else if (refType.getNumChildren() > 0 && refType.getChild(0) instanceof ASTReferenceType) {
271             // no type information (probably missing auxclasspath) - use simple types
272             ASTReferenceType ref = (ASTReferenceType) refType.getChild(0);
273             if (ref.getChild(0) instanceof ASTClassOrInterfaceType) {
274                 ASTClassOrInterfaceType clazz = (ASTClassOrInterfaceType) ref.getChild(0);
275                 if (simpleTypes.contains(toSimpleType(clazz.getImage())) && !clazz.isReferenceToClassSameCompilationUnit()
276                         || types.contains(clazz.getImage()) && !clazz.isReferenceToClassSameCompilationUnit()) {
277                     return true;
278                 }
279             }
280         }
281         return false;
282     }

284     private boolean isAllowedResourceType(TypeNode refType) {
285         List<String> allowedResourceTypes = getProperty(ALLOWED_RESOURCE_TYPES);
286         if (refType.getType() != null && allowedResourceTypes != null) {
287             for (String type : allowedResourceTypes) {
288                 // the check here must be a exact type match, since subclasses may override close()
289                 // and actually require closing
290                 if (TypeHelper.isExactlyA(refType, type)) {
291                     return true;
292                 }
293             }
294         }
295         return false;
296     }

298     private boolean hasNullInitializer(ASTLocalVariableDeclaration var) {
299         ASTVariableInitializer init = var.getFirstDescendantOfType(ASTVariableInitializer.class);
300         if (init != null) {
301             try {
302                 List<?> nulls = init
303                         .findChildNodesWithXPath("Expression/PrimaryExpression/PrimaryPrefix/Literal/NullLiteral");
304                 return !nulls.isEmpty();
305             } catch (JaxenException e) {
306                 return false;
307             }
308         }
309         return false;
310     }

312     private void ensureClosed(ASTLocalVariableDeclaration var, ASTVariableDeclaratorId id, TypeNode type, Object data) {
313         // What are the chances of a Connection being instantiated in a
314         // for-loop init block? Anyway, I'm lazy!
315         String variableToClose = id.getImage();
316         Node n = var;
317 
318         while (!(n instanceof ASTBlock) && !(n instanceof ASTConstructorDeclaration)) {
319             n = n.getParent();
320         }
321 
322         Node top = n;
323 
324         List<ASTTryStatement> tryblocks = top.findDescendantsOfType(ASTTryStatement.class);
325 
326         boolean closed = false;
327 
328         ASTBlockStatement parentBlock = id.getFirstParentOfType(ASTBlockStatement.class);
329 
330         // look for try blocks below the line the variable was
331         // introduced and make sure there is a .close call in a finally
332         // block.
333         for (ASTTryStatement t : tryblocks) {
334 
335             // verifies that there are no critical statements between the
336             // variable declaration and
337             // the beginning of the try block.
338             ASTBlockStatement tryBlock = t.getFirstParentOfType(ASTBlockStatement.class);
339             // no need to check for critical statements, if
340             // the variable has been initialized with null
341             if (!hasNullInitializer(var) && parentBlock.getParent() == tryBlock.getParent()) {
342 
343                 List<ASTBlockStatement> blocks = parentBlock.getParent().findChildrenOfType(ASTBlockStatement.class);
344                 int parentBlockIndex = blocks.indexOf(parentBlock);
345                 int tryBlockIndex = blocks.indexOf(tryBlock);
346                 boolean criticalStatements = false;
347 
348                 for (int i = parentBlockIndex + 1; i < tryBlockIndex; i++) {
349                     // assume variable declarations are not critical and assignments are not critical
350                     ASTBlockStatement block = blocks.get(i);
351                     ASTLocalVariableDeclaration varDecl = block
352                             .getFirstDescendantOfType(ASTLocalVariableDeclaration.class);
353                     ASTStatementExpression statementExpression = block.getFirstDescendantOfType(ASTStatementExpression.class);
354 
355                     if (varDecl == null && (statementExpression == null
356                             || statementExpression.getFirstChildOfType(ASTAssignmentOperator.class) == null)) {
357                         criticalStatements = true;
358                         break;
359                     }
360                 }
361                 if (criticalStatements) {
362                     break;
363                 }
364             }
365 
366             ASTFinallyStatement finallyClause = t.getFinallyClause();
367             if (t.getBeginLine() > id.getBeginLine() && finallyClause != null) {
368                 ASTBlock finallyBody = finallyClause.getBody();
369                 List<ASTName> names = finallyBody.findDescendantsOfType(ASTName.class);
370                 for (ASTName oName : names) {
371                     String name = oName.getImage();
372                     if (name != null && name.contains(".")) {
373                         String[] parts = name.split("\\.");
374                         if (parts.length == 2) {
375                             String methodName = parts[1];
376                             String varName = parts[0];
377                             if (varName.equals(variableToClose) && closeTargets.contains(methodName)
378                                     && nullCheckIfCondition(finallyBody, oName, varName)) {
379                                 closed = true;
380                                 break;
381                             }
382 
383                         }
384                     }
385                 }
386                 if (closed) {
387                     break;
388                 }
389 
390                 List<ASTStatementExpression> exprs = finallyBody.findDescendantsOfType(ASTStatementExpression.class, true);
391                 for (ASTStatementExpression stmt : exprs) {
392                     ASTPrimaryExpression expr = stmt.getFirstChildOfType(ASTPrimaryExpression.class);
393                     if (expr != null) {
394                         ASTPrimaryPrefix prefix = expr.getFirstChildOfType(ASTPrimaryPrefix.class);
395                         ASTPrimarySuffix suffix = expr.getFirstChildOfType(ASTPrimarySuffix.class);
396                         if (prefix != null && suffix != null) {
397                             if (prefix.getImage() == null) {
398                                 ASTName prefixName = prefix.getFirstChildOfType(ASTName.class);
399                                 if (prefixName != null && closeTargets.contains(prefixName.getImage())) {
400                                     // Found a call to a "close target" that is
401                                     // a direct
402                                     // method call without a "ClassName."
403                                     // prefix.
404                                     closed = variableIsPassedToMethod(expr, variableToClose);
405                                     if (closed) {
406                                         break;
407                                     }
408                                 }
409                             } else if (suffix.getImage() != null) {
410                                 String prefixPlusSuffix = prefix.getImage() + "." + suffix.getImage();
411                                 if (closeTargets.contains(prefixPlusSuffix)) {
412                                     // Found a call to a "close target" that is
413                                     // a method call
414                                     // in the form "ClassName.methodName".
415                                     closed = variableIsPassedToMethod(expr, variableToClose);
416                                     if (closed) {
417                                         break;
418                                     }
419                                 }
420                             }
421                             // look for primary suffix containing the close
422                             // Targets elements.
423                             // If the .close is executed in another class
424                             // accessed by a method
425                             // this form :
426                             // getProviderInstance().closeConnexion(connexion)
427                             // For this use case, we assume the variable is
428                             // correctly closed
429                             // in the other class since there is no way to
430                             // really check it.
431                             if (!closed) {
432                                 List<ASTPrimarySuffix> suffixes = expr.findDescendantsOfType(ASTPrimarySuffix.class, true);
433                                 for (ASTPrimarySuffix oSuffix : suffixes) {
434                                     String suff = oSuffix.getImage();
435                                     if (closeTargets.contains(suff)) {
436                                         closed = variableIsPassedToMethod(expr, variableToClose);
437                                         if (closed) {
438                                             break;
439                                         }
440                                     }
441 
442                                 }
443                             }
444                         }
445                     }
446                 }
447                 if (closed) {
448                     break;
449                 }
450             } else if (t.isTryWithResources()) {
451                 // maybe the variable is used as a resource
452                 List<ASTName> names = t.getFirstChildOfType(ASTResourceSpecification.class).findDescendantsOfType(ASTName.class);
453                 for (ASTName potentialUsage : names) {
454                     if (potentialUsage.hasImageEqualTo(variableToClose)) {
455                         closed = true;
456                         break;
457                     }
458                 }
459             }
460         }
461 
462         if (!closed) {
463             // See if the variable is returned by the method, which means the
464             // method is a utility for creating the db resource, which means of
465             // course it can't be closed by the method, so it isn't an error.
466             List<ASTReturnStatement> returns = top.findDescendantsOfType(ASTReturnStatement.class, true);
467             for (ASTReturnStatement returnStatement : returns) {
468                 ASTName name = returnStatement.getFirstDescendantOfType(ASTName.class);
469                 if (name != null && name.getImage().equals(variableToClose)) {
470                     closed = true;
471                     break;
472                 }
473             }
474         }
475 
476         // if all is not well, complain
477         if (!closed) {
478             ASTLocalVariableDeclaration localVarDecl = id.getFirstParentOfType(ASTLocalVariableDeclaration.class);
479             Class<?> typeClass = type.getType();
480             if (typeClass != null) {
481                 addViolation(data, id, typeClass.getSimpleName());
482             } else if (localVarDecl != null && localVarDecl.getTypeNode() != null) {
483                 addViolation(data, id, localVarDecl.getTypeNode().getTypeImage());
484             } else {
485                 addViolation(data, id, id.getVariableName());
486             }
487         }
488     }

490     private boolean variableIsPassedToMethod(ASTPrimaryExpression expr, String variable) {
491         List<ASTName> methodParams = expr.findDescendantsOfType(ASTName.class, true);
492         for (ASTName pName : methodParams) {
493             String paramName = pName.getImage();
494             // also check if we've got the a parameter (i.e if it's an argument
495             // !)
496             ASTArgumentList parentParam = pName.getFirstParentOfType(ASTArgumentList.class);
497             if (paramName.equals(variable) && parentParam != null) {
498                 return true;
499             }
500         }
501         return false;
502     }

504     private ASTIfStatement findIfStatement(ASTBlock enclosingBlock, Node node) {
505         ASTIfStatement ifStatement = node.getFirstParentOfType(ASTIfStatement.class);
506         List<ASTIfStatement> allIfStatements = enclosingBlock.findDescendantsOfType(ASTIfStatement.class);
507         if (ifStatement != null && allIfStatements.contains(ifStatement)) {
508             return ifStatement;
509         }
510         return null;
511     }

513     /**
514      * Checks, whether the given node is inside a if condition, and if so,
515      * whether this is a null check for the given varName.
516      *
517      * @param enclosingBlock
518      *            where to search for if statements
519      * @param node
520      *            the node, where the call for the close is done
521      * @param varName
522      *            the variable, that is maybe null-checked
523      * @return <code>true</code> if no if condition is involved or if the if
524      *         condition is a null-check.
525      */
526     private boolean nullCheckIfCondition(ASTBlock enclosingBlock, Node node, String varName) {
527         ASTIfStatement ifStatement = findIfStatement(enclosingBlock, node);
528         if (ifStatement != null) {
529             try {
530                 // find expressions like: varName != null or null != varName
531                 List<?> nodes = ifStatement.findChildNodesWithXPath("Expression/EqualityExpression[@Image='!=']"
532                         + "  [PrimaryExpression/PrimaryPrefix/Name[@Image='" + varName + "']]"
533                         + "  [PrimaryExpression/PrimaryPrefix/Literal/NullLiteral]");
534                 return !nodes.isEmpty();
535             } catch (JaxenException e) {
536                 throw new RuntimeException(e);
537             }
538         }
539         return true;
540     }
