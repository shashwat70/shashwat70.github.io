92      public static boolean checkSubtypeability(MethodType method, MethodType subtypeableMethod) {
93          List<JavaTypeDefinition> subtypeableParams = subtypeableMethod.getParameterTypes();
94          List<JavaTypeDefinition> methodParams = method.getParameterTypes();
95  
96          // If we come from third-phase, both are varargs, otherwhise, treat all as fixed-arity
97          if (!method.getMethod().isVarArgs() || !subtypeableMethod.getMethod().isVarArgs()) {
98              for (int index = 0; index < subtypeableParams.size(); ++index) {
99                  if (!isSubtypeable(methodParams.get(index), subtypeableParams.get(index))) {
100                     return false;
101                 }
102             }
103         } else {
104             final int maxSize = Math.max(subtypeableParams.size(), methodParams.size());
105             for (int index = 0; index < maxSize; ++index) {
106                 if (!isSubtypeable(method.getArgTypeIncludingVararg(index),
107                                    subtypeableMethod.getArgTypeIncludingVararg(index))) {
108                     return false;
109                 }
110             }
111         }
112 
113         return true;
114     }

117      * Look for methods be subtypeability.
118      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-15.html#jls-15.12.2.2
119      */
120     public static List<MethodType> selectMethodsFirstPhase(JavaTypeDefinition context,
121                                                            List<MethodType> methodsToSearch, ASTArgumentList argList) {
122         // TODO: check if explicit type arguments are applicable to the type parameter bounds
123         List<MethodType> selectedMethods = new ArrayList<>();
124         final int argCount = argList == null ? 0 : argList.getNumChildren();
125 
126         outter:
127         for (int methodIndex = 0; methodIndex < methodsToSearch.size(); ++methodIndex) {
128             MethodType methodType = methodsToSearch.get(methodIndex);
129 
130             // vararg methods are considered fixed arity here, see 3rd phase
131             if (getArity(methodType.getMethod()) == argCount) {
132                 if (!methodType.isParameterized()) {
133                     // https://docs.oracle.com/javase/specs/jls/se8/html/jls-18.html#jls-18.5.1
134                     // ...
135                     //  To test for applicability by strict invocation:
136                     //  ... or if there exists an i (1 ≤ i ≤ n) such that ei is pertinent to applicability
137                     // (§15.12.2.2) and either i) ei is a standalone expression of a primitive type but Fi is a
138                     // reference type, or ii) Fi is a primitive type but ei is not a standalone expression of a
139                     // primitive type; then the method is not applicable and there is no need to proceed with inference.
140                     Class<?>[] methodParameterTypes = methodType.getMethod().getParameterTypes();
141                     for (int argIndex = 0; argIndex < argCount; ++argIndex) {
142                         if (((ASTExpression) argList.getChild(argIndex)).isStandAlonePrimitive()) {
143                             if (!methodParameterTypes[argIndex].isPrimitive()) {
144                                 continue outter; // this method is not applicable
145                             }
146                         } else if (methodParameterTypes[argIndex].isPrimitive()) {
147                             continue outter; // this method is not applicable
148                         }
149                     }
150 
151                     methodType = parameterizeInvocation(context, methodType.getMethod(), argList);
152 
153                     // May be null if the method call is not applicable
154                     if (methodType == null) {
155                         continue;
156                     }
157                 }
158 
159                 // TODO : Is this needed? parameterizeInvocation already performs inference to check applicability...
160                 // check subtypeability of each argument to the corresponding parameter
161                 boolean methodIsApplicable = true;
162 
163                 // try each arguments if it's subtypeable
164                 for (int argIndex = 0; argIndex < argCount; ++argIndex) {
165                     if (!isSubtypeable(methodType.getParameterTypes().get(argIndex),
166                                        (ASTExpression) argList.getChild(argIndex))) {
167                         methodIsApplicable = false;
168                         break;
169                     }
170                     // TODO: add unchecked conversion in an else if branch
171                 }
172 
173                 if (methodIsApplicable) {
174                     selectedMethods.add(methodType);
175                 }
176 
177             }
178 
179         }
180 
181         return selectedMethods;
182     }

184     public static MethodType parameterizeInvocation(JavaTypeDefinition context, Method method,
185                                                     ASTArgumentList argList) {
186 
187         // variables are set up by the call to produceInitialBounds
188         List<Variable> variables = new ArrayList<>();
189         List<Bound> initialBounds = new ArrayList<>();
190         produceInitialBounds(method, context, variables, initialBounds);
191 
192         List<JavaTypeDefinition> resolvedTypeParameters = TypeInferenceResolver
193                 .inferTypes(produceInitialConstraints(method, argList, variables), initialBounds, variables);
194 
195         // Is the method applicable?
196         if (resolvedTypeParameters == null) {
197             return null;
198         }
199 
200         return getTypeDefOfMethod(context, method, resolvedTypeParameters);
201     }

203     public static List<Constraint> produceInitialConstraints(Method method, ASTArgumentList argList,
204                                                              List<Variable> variables) {
205         List<Constraint> result = new ArrayList<>();
206 
207         Type[] methodParameters = method.getGenericParameterTypes();
208         TypeVariable<Method>[] methodTypeParameters = method.getTypeParameters();
209 
210         // TODO: add support for variable arity methods
211         for (int i = 0; i < methodParameters.length; i++) {
212             int typeParamIndex = -1;
213             if (methodParameters[i] instanceof TypeVariable) {
214                 typeParamIndex = JavaTypeDefinition
215                         .getGenericTypeIndex(methodTypeParameters, ((TypeVariable<?>) methodParameters[i]).getName());
216             }
217 
218             if (typeParamIndex != -1) {
219                 // TODO: we are cheating here, it should be a contraint of the form 'var -> expression' not 'var->type'
220                 result.add(new Constraint(((TypeNode) argList.getChild(i)).getTypeDefinition(),
221                                           variables.get(typeParamIndex), LOOSE_INVOCATION));
222             }
223         }
224 
225         return result;
226     }

229     public static void produceInitialBounds(Method method, JavaTypeDefinition context,
230                                             List<Variable> variables, List<Bound> initialBounds) {
231         // https://docs.oracle.com/javase/specs/jls/se8/html/jls-18.html#jls-18.1.3
232         // When inference begins, a bound set is typically generated from a list of type parameter declarations P1,
233         // ..., Pp and associated inference variables α1, ..., αp. Such a bound set is constructed as follows. For
234         // each l (1 ≤ l ≤ p):
235 
236         TypeVariable<Method>[] typeVariables = method.getTypeParameters();
237 
238         variables.clear();
239         for (int i = 0; i < typeVariables.length; ++i) {
240             variables.add(new Variable());
241         }
242 
243         for (int currVarIndex = 0; currVarIndex < typeVariables.length; ++currVarIndex) {
244             Type[] bounds = typeVariables[currVarIndex].getBounds();
245             boolean currVarHasNoProperUpperBound = true;
246 
247             for (Type bound : bounds) {
248                 // Otherwise, for each type T delimited by & in the TypeBound, the bound αl <: T[P1:=α1, ..., Pp:=αp]
249                 // appears in the set; if this results in no proper upper bounds for αl (only dependencies), then the
250                 // bound α <: Object also appears in the set.
251 
252                 int boundVarIndex = -1;
253                 if (bound instanceof TypeVariable) {
254                     boundVarIndex =
255                             JavaTypeDefinition.getGenericTypeIndex(typeVariables, ((TypeVariable<?>) bound).getName());
256                 }
257 
258                 if (boundVarIndex != -1) {
259                     initialBounds.add(new Bound(variables.get(currVarIndex), variables.get(boundVarIndex), SUBTYPE));
260                 } else {
261                     currVarHasNoProperUpperBound = false;
262                     initialBounds.add(new Bound(variables.get(currVarIndex), context.resolveTypeDefinition(bound),
263                                                 SUBTYPE));
264                 }
265             }
266 
267             // If Pl has no TypeBound, the bound αl <: Object appears in the set.
268             if (currVarHasNoProperUpperBound) {
269                 initialBounds.add(new Bound(variables.get(currVarIndex), JavaTypeDefinition.forClass(Object.class),
270                                             SUBTYPE));
271             }
272         }
273     }

278      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-15.html#jls-15.12.2.3
279      */
280     public static List<MethodType> selectMethodsSecondPhase(List<MethodType> methodsToSearch, ASTArgumentList argList) {
281         // TODO: check if explicit type arguments are applicable to the type parameter bounds
282         List<MethodType> selectedMethods = new ArrayList<>();
283         final int argCount = argList == null ? 0 : argList.getNumChildren();
284 
285         for (int methodIndex = 0; methodIndex < methodsToSearch.size(); ++methodIndex) {
286             MethodType methodType = methodsToSearch.get(methodIndex);
287             if (!methodType.isParameterized()) {
288                 throw new ResolutionFailedException();
289             }
290 
291             // vararg methods are considered fixed arity here, see 3rd phase
292             if (getArity(methodType.getMethod()) == argCount) {
293                 // check method convertability of each argument to the corresponding parameter
294                 boolean methodIsApplicable = true;
295 
296                 // try each arguments if it's method convertible
297                 for (int argIndex = 0; argIndex < argCount; ++argIndex) {
298                     if (!isMethodConvertible(methodType.getParameterTypes().get(argIndex),
299                                              (ASTExpression) argList.getChild(argIndex))) {
300                         methodIsApplicable = false;
301                         break;
302                     }
303                     // TODO: add unchecked conversion in an else if branch
304                 }
305 
306                 if (methodIsApplicable) {
307                     selectedMethods.add(methodType);
308                 }
309             }
310 
311         }
312 
313         return selectedMethods;
314     }

317     /**
318      * Look for methods considering varargs as well.
319      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-15.html#jls-15.12.2.4
320      */
321     public static List<MethodType> selectMethodsThirdPhase(List<MethodType> methodsToSearch, ASTArgumentList argList) {
322         // TODO: check if explicit type arguments are applicable to the type parameter bounds
323         List<MethodType> selectedMethods = new ArrayList<>();
324 
325         for (int methodIndex = 0; methodIndex < methodsToSearch.size(); ++methodIndex) {
326             MethodType methodType = methodsToSearch.get(methodIndex);
327             if (!methodType.isParameterized()) {
328                 throw new ResolutionFailedException();
329             }
330 
331             // now we consider varargs as not fixed arity
332             // if we reach here and the method is not a vararg, then we didn't find a resolution in earlier phases
333             if (methodType.isVararg()) { // check subtypeability of each argument to the corresponding parameter
334                 boolean methodIsApplicable = true;
335 
336                 List<JavaTypeDefinition> methodParameters = methodType.getParameterTypes();
337                 JavaTypeDefinition varargComponentType = methodType.getVarargComponentType();
338 
339                 if (argList == null) {
340                     // There are no arguments, make sure the method has only a vararg
341                     methodIsApplicable = getArity(methodType.getMethod()) == 1;
342                 } else {
343                     // try each arguments if it's method convertible
344                     for (int argIndex = 0; argIndex < argList.getNumChildren(); ++argIndex) {
345                         JavaTypeDefinition parameterType = argIndex < methodParameters.size() - 1
346                                 ? methodParameters.get(argIndex) : varargComponentType;
347 
348                         if (!isMethodConvertible(parameterType, (ASTExpression) argList.getChild(argIndex))) {
349                             methodIsApplicable = false;
350                             break;
351                         }
352 
353                         // TODO: If k != n, or if k = n and An cannot be converted by method invocation conversion to
354                         // Sn[], then the type which is the erasure (§4.6) of Sn is accessible at the point of invocation.
355 
356                         // TODO: add unchecked conversion in an else if branch
357                     }
358                 }
359 
360                 if (methodIsApplicable) {
361                     selectedMethods.add(methodType);
362                 }
363             } else {
364                 // TODO: Remove check for vararg here, once we can detect and use return types of method calls
365                 LOG.log(Level.FINE, "Method {0} couldn't be resolved", String.valueOf(methodType));
366             }
367         }
368 
369         return selectedMethods;
370     }

373     /**
374      * Searches a list of methods by trying the three phases of method overload resolution.
375      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-15.html#jls-15.12.2
376      */
377     public static JavaTypeDefinition getBestMethodReturnType(JavaTypeDefinition context, List<MethodType> methods,
378                                                              ASTArgumentList arguments) {
379 
380         try {
381             List<MethodType> selectedMethods = selectMethodsFirstPhase(context, methods, arguments);
382             if (!selectedMethods.isEmpty()) {
383                 return selectMostSpecificMethod(selectedMethods).getReturnType();
384             }
385 
386             selectedMethods = selectMethodsSecondPhase(methods, arguments);
387             if (!selectedMethods.isEmpty()) {
388                 return selectMostSpecificMethod(selectedMethods).getReturnType();
389             }
390 
391             selectedMethods = selectMethodsThirdPhase(methods, arguments);
392             if (!selectedMethods.isEmpty()) {
393                 return selectMostSpecificMethod(selectedMethods).getReturnType();
394             }
395 
396             return null;
397         } catch (ResolutionFailedException e) {
398             return null;
399         }
400     }

402     /**
403      * Most specific method selection.
404      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-15.html#jls-15.12.2.5
405      */
406     public static MethodType selectMostSpecificMethod(List<MethodType> selectedMethods) {
407 
408         MethodType mostSpecific = selectedMethods.get(0);
409 
410         for (int methodIndex = 1; methodIndex < selectedMethods.size(); ++methodIndex) {
411             MethodType nextMethod = selectedMethods.get(methodIndex);
412 
413             if (checkSubtypeability(mostSpecific, nextMethod)) {
414                 if (checkSubtypeability(nextMethod, mostSpecific)) { // both are maximally specific
415                     mostSpecific = selectAmongMaximallySpecific(mostSpecific, nextMethod);
416                 } else {
417                     mostSpecific = nextMethod;
418                 }
419             }
420         }
421 
422         return mostSpecific;
423     }

452     /**
453      * Looks for potentially applicable methods in a given type definition.
454      */
455     public static List<MethodType> getApplicableMethods(JavaTypeDefinition context,
456                                                         String methodName,
457                                                         List<JavaTypeDefinition> typeArguments,
458                                                         int argArity,
459                                                         Class<?> accessingClass) {
460         List<MethodType> result = new ArrayList<>();
461 
462         if (context == null) {
463             return result;
464         }
465 
466         // TODO: shadowing, overriding
467         // TODO: add multiple upper bounds
468 
469         Class<?> contextClass = context.getType();
470 
471         // search the class
472         try {
473             for (Method method : contextClass.getDeclaredMethods()) {
474                 if (isMethodApplicable(method, methodName, argArity, accessingClass, typeArguments)) {
475                     result.add(getTypeDefOfMethod(context, method, typeArguments));
476                 }
477             }
478         } catch (final LinkageError ignored) {
479             // TODO : This is an incomplete classpath, report the missing class
480         }
481 
482         // search it's supertype
483         if (!contextClass.equals(Object.class)) {
484             List<MethodType> inheritedMethods = getApplicableMethods(context.resolveTypeDefinition(contextClass.getGenericSuperclass()),
485                                                methodName, typeArguments, argArity, accessingClass);
486 
487             // but only add the found methods of the supertype, if they have not been overridden
488             // TODO: verify whether this simplified overriding detection is good enough and at the correct place
489             for (MethodType inherited : inheritedMethods) {
490                 if (!result.contains(inherited)) {
491                     result.add(inherited);
492                 }
493             }
494         }
495 
496         // search it's interfaces
497         for (Type interfaceType : contextClass.getGenericInterfaces()) {
498             result.addAll(getApplicableMethods(context.resolveTypeDefinition(interfaceType),
499                                                methodName, typeArguments, argArity, accessingClass));
500         }
501 
502         return result;
503     }

506     public static MethodType getTypeDefOfMethod(JavaTypeDefinition context, Method method,
507                                                 List<JavaTypeDefinition> typeArguments) {
508         if (typeArguments.isEmpty() && isGeneric(method)) {
509             return MethodType.build(method);
510         }
511 
512         JavaTypeDefinition returnType = context.resolveTypeDefinition(method.getGenericReturnType(),
513                                                                       method, typeArguments);
514         List<JavaTypeDefinition> argTypes = new ArrayList<>();
515 
516         for (Type argType : method.getGenericParameterTypes()) {
517             argTypes.add(context.resolveTypeDefinition(argType, method, typeArguments));
518         }
519 
520         return MethodType.build(returnType, argTypes, method);
521     }

524     /**
525      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-15.html#jls-15.12.2.1
526      * Potential applicability.
527      */
528     public static boolean isMethodApplicable(Method method, String methodName, int argArity,
529                                              Class<?> accessingClass, List<JavaTypeDefinition> typeArguments) {
530 
531         return method.getName().equals(methodName) // name matches
532                 // is visible
533                 && isMemberVisibleFromClass(method.getDeclaringClass(), method.getModifiers(), accessingClass)
534                 // if method is vararg with arity n, then the invocation's arity >= n - 1
535                 && (!method.isVarArgs() || argArity >= getArity(method) - 1)
536                 // if the method isn't vararg, then arity matches
537                 && (method.isVarArgs() || argArity == getArity(method))
538                 // isn't generic or arity of type arguments matches that of parameters
539                 && (!isGeneric(method) || typeArguments.isEmpty()
540                 || method.getTypeParameters().length == typeArguments.size());
541     }

544     /**
545      * Given a class, the modifiers of on of it's member and the class that is trying to access that member,
546      * returns true is the member is accessible from the accessingClass Class.
547      *
548      * @param classWithMember The Class with the member.
549      * @param modifiers       The modifiers of that member.
550      * @param accessingClass  The Class trying to access the member.
551      * @return True if the member is visible from the accessingClass Class.
552      */
553     public static boolean isMemberVisibleFromClass(Class<?> classWithMember, int modifiers, Class<?> accessingClass) {
554         if (accessingClass == null) {
555             return false;
556         }
557 
558         // public members
559         if (Modifier.isPublic(modifiers)) {
560             return true;
561         }
562 
563         boolean areInTheSamePackage = false;
564 
565         if (accessingClass.getPackage() != null) { // if null, then it's in the default package
566             // if calssWithMember.getPackage() is null, result will be false
567             areInTheSamePackage = accessingClass.getPackage().getName().equals(
568                     classWithMember.getPackage().getName());
569         }
570 
571         // protected members
572         if (Modifier.isProtected(modifiers)) {
573             if (areInTheSamePackage || classWithMember.isAssignableFrom(accessingClass)) {
574                 return true;
575             }
576             // private members
577         } else if (Modifier.isPrivate(modifiers)) {
578             if (classWithMember.equals(accessingClass)) {
579                 return true;
580             }
581             // package private members
582         } else if (areInTheSamePackage) {
583             return true;
584         }
585 
586         return false;
587     }

609     /**
610      * Method invocation conversion rules.
611      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-5.html#jls-5.3
612      */
613     public static boolean isMethodConvertible(JavaTypeDefinition parameter, JavaTypeDefinition argument) {
614         // covers identity conversion, widening primitive conversion, widening reference conversion, null
615         // covers if both are primitive or bot are boxed primitive
616         if (isSubtypeable(parameter, argument)) {
617             return true;
618         }
619 
620         // covers boxing
621         int indexInPrimitive = PRIMITIVE_SUBTYPE_ORDER.indexOf(argument.getType());
622 
623         if (indexInPrimitive != -1 // arg is primitive
624                 && isSubtypeable(parameter,
625                                  JavaTypeDefinition.forClass(BOXED_PRIMITIVE_SUBTYPE_ORDER.get(indexInPrimitive)))) {
626             return true;
627         }
628 
629         // covers unboxing
630         int indexInBoxed = BOXED_PRIMITIVE_SUBTYPE_ORDER.indexOf(argument.getType());
631 
632         return indexInBoxed != -1 // arg is boxed primitive
633                 && isSubtypeable(parameter,
634                                  JavaTypeDefinition.forClass(PRIMITIVE_SUBTYPE_ORDER.get(indexInBoxed)));
635         // TODO: add raw unchecked conversion part
636     }

650     /**
651      * Subtypeability rules.
652      * https://docs.oracle.com/javase/specs/jls/se7/html/jls-4.html#jls-4.10
653      */
654     public static boolean isSubtypeable(JavaTypeDefinition parameter, JavaTypeDefinition argument) {
655         // null types are always applicable
656         if (argument.getType() == null) {
657             return true;
658         }
659 
660         // this covers arrays, simple class/interface cases
661         if (parameter.getType().isAssignableFrom(argument.getType())) {
662             if (!parameter.isGeneric() || parameter.isRawType() || argument.isRawType()) {
663                 return true;
664             }
665 
666             // parameter is a non-raw generic type
667             // argument is a non-generic or a non-raw generic type
668 
669             // example result: List<String>.getAsSuper(Collection) becomes Collection<String>
670             JavaTypeDefinition argSuper = argument.getAsSuper(parameter.getType());
671             // argSuper can't be null because isAssignableFrom check above returned true
672 
673             // right now we only check if generic arguments are the same
674             // TODO: add support for wildcard types
675             // (future note: can't call subtype as it is recursively, infinite types)
676             //return parameter.equals(argSuper);
677 
678             // TODO: this ignores the check for generic types!!
679             return parameter.getType().equals(argSuper.getType());
680         }
681 
682         int indexOfParameter = PRIMITIVE_SUBTYPE_ORDER.indexOf(parameter.getType());
683 
684         if (indexOfParameter != -1) {
685             if (argument.getType() == char.class) {
686                 if (indexOfParameter <= 3) { // <= 3 because short and byte are not compatible with char
687                     return true;
688                 }
689             } else {
690                 int indexOfArg = PRIMITIVE_SUBTYPE_ORDER.indexOf(argument.getType());
691                 if (indexOfArg != -1 && indexOfParameter <= indexOfArg) {
692                     return true;
693                 }
694             }
695         }
696 
697         return false;
698     }

704     public static List<JavaTypeDefinition> getMethodExplicitTypeArugments(Node node) {
705         ASTMemberSelector memberSelector = node.getFirstChildOfType(ASTMemberSelector.class);
706         if (memberSelector == null) {
707             return Collections.emptyList();
708         }
709 
710         ASTTypeArguments typeArguments = memberSelector.getFirstChildOfType(ASTTypeArguments.class);
711         if (typeArguments == null) {
712             return Collections.emptyList();
713         }
714 
715         List<JavaTypeDefinition> result = new ArrayList<>();
716 
717         for (int childIndex = 0; childIndex < typeArguments.getNumChildren(); ++childIndex) {
718             result.add(((TypeNode) typeArguments.getChild(childIndex)).getTypeDefinition());
719         }
720 
721         return result;
722     }
