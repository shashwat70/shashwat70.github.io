356     public void runSampleFile(final FieldModel type, final String inputFile, final String outputFile)
357         throws Exception {
358 
359         final BufferedReader inReader = new BufferedReader(new InputStreamReader(getResource(inputFile)));
360         final BufferedReader outReader = new BufferedReader(new InputStreamReader(getResource(outputFile)));
361 
362         // read header line
363         outReader.readLine();
364 
365         String line = null;
366         while ((line = inReader.readLine()) != null) {
367             if (line.trim().length() == 0) {
368                 break;
369             }
370 
371             final StringTokenizer st = new StringTokenizer(line);
372 
373             final double year = getYear(st.nextToken());
374             final String heightType = st.nextToken();
375             final String heightStr = st.nextToken();
376             final double lat = getLatLon(st.nextToken());
377             final double lon = getLatLon(st.nextToken());
378 
379             final GeoMagneticField field = GeoMagneticFieldFactory.getField(type, year);
380 
381             double height = Double.valueOf(heightStr.substring(1));
382             if (heightStr.startsWith("M")) {
383                 // convert from m to km
384                 height /= 1000d;
385             } else if (heightStr.startsWith("F")) {
386                 // convert from feet to km
387                 height *= 3.048e-4;
388             }
389 
390             final GeoMagneticElements ge = field.calculateField(lat, lon, height);
391             final String validateLine = outReader.readLine();
392             // geocentric altitude is not yet supported, ignore by now
393             if (!heightType.startsWith("C")) {
394                 validate(ge, validateLine);
395             }
396 
397             String geString = ge.toString();
398             Assert.assertNotNull(geString);
399             Assert.assertFalse(geString.isEmpty());
400         }
401     }

403     private double getYear(final String yearStr) {
404 
405         if (yearStr.contains(",")) {
406             final StringTokenizer st = new StringTokenizer(yearStr, ",");
407             final String y = st.nextToken();
408             final String m = st.nextToken();
409             final String d = st.nextToken();
410             return GeoMagneticField.getDecimalYear(Integer.valueOf(d), Integer.valueOf(m), Integer.valueOf(y));
411         } else {
412             return Double.valueOf(yearStr);
413         }
414     }
415 
416     private double getLatLon(final String str) {
417 
418         if (str.contains(",")) {
419             final StringTokenizer st = new StringTokenizer(str, ",");
420             final int d = Integer.valueOf(st.nextToken());
421             final int m = Integer.valueOf(st.nextToken());
422             int s = 0;
423             if (st.hasMoreTokens()) {
424                 s = Integer.valueOf(st.nextToken());
425             }
426             double deg = FastMath.abs(d) + m / 60d + s / 3600d;
427             if (d < 0) {
428                 deg = -deg;
429             }
430             return deg;
431         } else {
432             return Double.valueOf(str);
433         }
434     }
435 
436     private double getDegree(final String degree, final String minute) {
437         double result = Double.valueOf(degree.substring(0, degree.length() - 1));
438         final double min = Double.valueOf(minute.substring(0, minute.length() - 1)) / 60d;
439         result += (result < 0) ? -min : min;
440         return result;
441     }

443     @SuppressWarnings("unused")
444     private void validate(final GeoMagneticElements ge, final String outputLine)
445         throws Exception {
446 
447         final StringTokenizer st = new StringTokenizer(outputLine);
448 
449         final double year = getYear(st.nextToken());
450 
451         final String coord = st.nextToken();
452         final String heightStr = st.nextToken();
453         final String latStr = st.nextToken();
454         final String lonStr = st.nextToken();
455 
456         final double dec = getDegree(st.nextToken(), st.nextToken());
457         final double inc = getDegree(st.nextToken(), st.nextToken());
458 
459         final double h = Double.valueOf(st.nextToken());
460         final double x = Double.valueOf(st.nextToken());
461         final double y = Double.valueOf(st.nextToken());
462         final double z = Double.valueOf(st.nextToken());
463         final double f = Double.valueOf(st.nextToken());
464 
465         final double eps = 1e-1;
466         Assert.assertEquals(h, ge.getHorizontalIntensity(), eps);
467         Assert.assertEquals(f, ge.getTotalIntensity(), eps);
468         Assert.assertEquals(x, ge.getFieldVector().getX(), eps);
469         Assert.assertEquals(y, ge.getFieldVector().getY(), eps);
470         Assert.assertEquals(z, ge.getFieldVector().getZ(), eps);
471         Assert.assertEquals(dec, ge.getDeclination(), eps);
472         Assert.assertEquals(inc, ge.getInclination(), eps);
473     }

475     private InputStream getResource(final String name) throws FileNotFoundException {
476         // the data path has multiple components, the resources are in the first one
477         final String separator = System.getProperty("path.separator");
478         final String dataPath = System.getProperty(DataProvidersManager.OREKIT_DATA_PATH).split(separator)[0];
479         return new FileInputStream(new File(dataPath, name));
480     }
