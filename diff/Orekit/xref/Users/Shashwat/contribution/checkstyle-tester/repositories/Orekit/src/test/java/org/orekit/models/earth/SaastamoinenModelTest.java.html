39      private static double epsilon = 1e-6;

41      @Test
42      public void testFixedElevation() throws OrekitException {
43          Utils.setDataRoot("atmosphere");
44          SaastamoinenModel model = SaastamoinenModel.getStandardModel();
45          double lastDelay = Double.MAX_VALUE;
46          // delay shall decline with increasing height of the station
47          for (double height = 0; height < 5000; height += 100) {
48              final double delay = model.pathDelay(FastMath.toRadians(5), height);
49              Assert.assertTrue(Precision.compareTo(delay, lastDelay, epsilon) < 0);
50              lastDelay = delay;
51          }
52      }

54      @Test
55      public void testFixedHeight() throws OrekitException {
56          Utils.setDataRoot("atmosphere");
57          SaastamoinenModel model = SaastamoinenModel.getStandardModel();
58          double lastDelay = Double.MAX_VALUE;
59          // delay shall decline with increasing elevation angle
60          for (double elev = 10d; elev < 90d; elev += 8d) {
61              final double delay = model.pathDelay(FastMath.toRadians(elev), 350);
62              Assert.assertTrue(Precision.compareTo(delay, lastDelay, epsilon) < 0);
63              lastDelay = delay;
64          }
65      }

79      @Test
80      public void testSerialization()
81        throws OrekitException, IOException, ClassNotFoundException, NoSuchFieldException, IllegalAccessException {
82          Utils.setDataRoot("atmosphere");
83          SaastamoinenModel model = SaastamoinenModel.getStandardModel();
84  
85          ByteArrayOutputStream bos = new ByteArrayOutputStream();
86          ObjectOutputStream    oos = new ObjectOutputStream(bos);
87          oos.writeObject(model);
88  
89          Assert.assertTrue(bos.size() > 1400);
90          Assert.assertTrue(bos.size() < 1500);
91  
92          ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
93          ObjectInputStream     ois = new ObjectInputStream(bis);
94          SaastamoinenModel deserialized  = (SaastamoinenModel) ois.readObject();
95  
96          double[] heights = new double[] {
97              0.0, 250.0, 500.0, 750.0, 1000.0, 1250.0, 1500.0, 1750.0, 2000.0, 2250.0, 2500.0, 2750.0, 3000.0, 3250.0,
98              3500.0, 3750.0, 4000.0, 4250.0, 4500.0, 4750.0, 5000.0
99          };
100         double[] elevations = new double[] {
101             FastMath.toRadians(10.0), FastMath.toRadians(15.0), FastMath.toRadians(20.0),
102             FastMath.toRadians(25.0), FastMath.toRadians(30.0), FastMath.toRadians(35.0),
103             FastMath.toRadians(40.0), FastMath.toRadians(45.0), FastMath.toRadians(50.0),
104             FastMath.toRadians(55.0), FastMath.toRadians(60.0), FastMath.toRadians(65.0),
105             FastMath.toRadians(70.0), FastMath.toRadians(75.0), FastMath.toRadians(80.0),
106             FastMath.toRadians(85.0), FastMath.toRadians(90.0)
107         };
108         for (int h = 0; h < heights.length; h++) {
109             for (int e = 0; e < elevations.length; e++) {
110                 double height = heights[h];
111                 double elevation = elevations[e];
112                 double expectedValue = model.pathDelay(elevation, height);
113                 double actualValue = deserialized.pathDelay(elevation, height);
114                 assertEquals("For height=" + height + " elevation = " + elevation + " precision not met",
115                              expectedValue, actualValue, epsilon);
116             }
117         }
118 
119     }
