142     /**
143      * Test several pre-computed points from the Online Geoid Height Evaluation
144      * tool, which takes into account terrain.
145      *
146      * @throws OrekitException on error
147      * @see <a href="http://geographiclib.sourceforge.net/cgi-bin/GeoidEval">Online
148      * Geoid Height Evaluation</a>
149      * @see <a href="http://earth-info.nga.mil/GandG/wgs84/gravitymod/egm96/egm96.html">Geoid
150      * height for WGS84 and EGM96</a>
151      */
152     @Test
153     public void testGetUndulation() throws OrekitException {
154         /*
155          * allow 3 meter of error, which is what the approximations would
156          * suggest, see the comment for Geoid.
157          */
158         final double maxError = 3;
159 
160         // run the test
161         Geoid geoid = getComponent();
162         for (double[] row : reference) {
163             double lat = row[0];
164             double lon = row[1];
165             double undulation = geoid.getUndulation(FastMath.toRadians(lat),
166                     FastMath.toRadians(lon), date);
167             double expected = row[2];
168             // System.out.format("%10g %10g %10g %10g%n", lat, lon, expected,
169             // undulation - expected);
170             Assert.assertEquals(String.format("lat: %5g, lon: %5g", lat, lon),
171                     undulation, expected, maxError);
172         }
173     }

175     /**
176      * check {@link Geoid#getIntersectionPoint(Line, Vector3D, Frame,
177      * AbsoluteDate)} with several points.
178      *
179      * @throws OrekitException on error
180      */
181     @Test
182     public void testGetIntersectionPoint() throws OrekitException {
183         // setup
184         Geoid geoid = getComponent();
185         Frame frame = geoid.getBodyFrame();
186 
187         for (double[] point : reference) {
188             GeodeticPoint gp = new GeodeticPoint(FastMath.toRadians(point[0]),
189                     FastMath.toRadians(point[1]), 0);
190             Vector3D expected = geoid.transform(gp);
191             // glancing line: 10% vertical and 90% north (~6 deg elevation)
192             Vector3D slope = gp.getZenith().scalarMultiply(0.1)
193                     .add(gp.getNorth().scalarMultiply(0.9));
194             Vector3D close = expected.add(slope.scalarMultiply(100e3));
195             Vector3D pointOnLine = expected.add(slope);
196             Line line = new Line(close, pointOnLine, 0);
197             // line directed the other way
198             Line otherDirection = new Line(pointOnLine, close, 0);
199 
200             // action
201             GeodeticPoint actual = geoid.getIntersectionPoint(line, close,
202                     frame, date);
203             // other direction
204             GeodeticPoint actualReversed = geoid.getIntersectionPoint(
205                     otherDirection, close, frame, date);
206 
207             // verify
208             String message = String.format("point: %s%n",
209                     Arrays.toString(point));
210             // position accuracy on Earth's surface to 1 um.
211             assertThat(message, actualReversed, geodeticPointCloseTo(gp, 1e-6));
212             assertThat(message, actual, geodeticPointCloseTo(gp, 1e-6));
213         }
214     }

216     /**
217      * check {@link Geoid#getIntersectionPoint(Line, Vector3D, Frame,
218      * AbsoluteDate)} handles frame transformations correctly
219      *
220      * @throws OrekitException on error
221      */
222     @Test
223     public void testGetIntersectionPointFrame() throws OrekitException {
224         // setup
225         Geoid geoid = getComponent();
226         Frame frame = new Frame(
227                 geoid.getBodyFrame(),
228                 new Transform(
229                         date,
230                         new Transform(
231                                 date,
232                                 new Vector3D(-1, 2, -3),
233                                 new Vector3D(4, -5, 6)),
234                         new Transform(
235                                 date,
236                                 new Rotation(-7, 8, -9, 10, true),
237                                 new Vector3D(-11, 12, -13))),
238                 "test frame");
239         GeodeticPoint gp = new GeodeticPoint(FastMath.toRadians(46.8743190),
240                 FastMath.toRadians(102.4487290), 0);
241         Vector3D expected = geoid.transform(gp);
242         // glancing line: 10% vertical and 90% north (~6 deg elevation)
243         Vector3D slope = gp.getZenith().scalarMultiply(0.1)
244                 .add(gp.getNorth().scalarMultiply(0.9));
245         Vector3D close = expected.add(slope.scalarMultiply(100));
246         Line line = new Line(expected.add(slope), close, 0);
247         Transform xform = geoid.getBodyFrame().getTransformTo(frame, date);
248         // transform to test frame
249         close = xform.transformPosition(close);
250         line = xform.transformLine(line);
251 
252         // action
253         GeodeticPoint actual = geoid.getIntersectionPoint(line, close, frame,
254                 date);
255 
256         // verify, 1 um position accuracy at Earth's surface
257         assertThat(actual, geodeticPointCloseTo(gp, 1e-6));
258     }

281     /**
282      * check altitude is referenced to the geoid. h<sub>ellipse</sub> =
283      * h<sub>geoid</sub> + N. Where N is the undulation of the geoid.
284      *
285      * @throws OrekitException on error
286      */
287     @Test
288     public void testTransformVector3DFrameAbsoluteDate()
289             throws OrekitException {
290         // frame and date are the same
291         Frame frame = FramesFactory.getGCRF();
292         AbsoluteDate date = AbsoluteDate.CCSDS_EPOCH;
293 
294         Geoid geoid = getComponent();
295         // test point at 0,0,0
296         Vector3D point = new Vector3D(WGS84.getEquatorialRadius(), 0, 0);
297         double undulation = geoid.getUndulation(0, 0, date);
298 
299         // check ellipsoidal points and geoidal points differ by undulation
300         GeodeticPoint ellipsoidal = geoid.getEllipsoid().transform(
301                 point, frame, date);
302         GeodeticPoint geoidal = geoid.transform(point, frame, date);
303         assertThat(ellipsoidal.getAltitude() - geoidal.getAltitude(),
304                 is(undulation));
305 
306         // check it is the reverse of transform(GeodeticPoint)
307         point = new Vector3D(0.5, 0.4, 0.31).scalarMultiply(WGS84
308                 .getEquatorialRadius());
309         Vector3D expected = geoid
310                 .transform(geoid.transform(point, frame, date));
311         // allow 2 upls of error
312         assertThat(point, vectorCloseTo(expected, 2));
313 
314     }

316     /**
317      * check that the altitude is referenced to the geoid (includes
318      * undulation).
319      *
320      * @throws OrekitException on error
321      */
322     @Test
323     public void testTransformGeodeticPoint() throws OrekitException {
324         // geoid
325         Geoid geoid = getComponent();
326         // ellipsoid
327         ReferenceEllipsoid ellipsoid = geoid.getEllipsoid();
328         // point to test with orthometric height
329         GeodeticPoint orthometric = new GeodeticPoint(0, 75, 5);
330         // undulation at point
331         double undulation = geoid.getUndulation(orthometric.getLatitude(),
332                 orthometric.getLongitude(), date);
333         // same point with height referenced to ellipsoid
334         GeodeticPoint point = new GeodeticPoint(orthometric.getLatitude(),
335                 orthometric.getLongitude(), orthometric.getAltitude()
336                 + undulation);
337 
338         // test they are the same
339         Vector3D expected = ellipsoid.transform(point);
340         Vector3D actual = geoid.transform(orthometric);
341         assertThat(actual, is(expected));
342 
343         // test the point 0,0,0
344         expected = new Vector3D(WGS84.getEquatorialRadius()
345                 + geoid.getUndulation(0, 0, date), 0, 0);
346         actual = geoid.transform(new GeodeticPoint(0, 0, 0));
347         assertThat(actual, vectorCloseTo(expected, 0));
348     }
