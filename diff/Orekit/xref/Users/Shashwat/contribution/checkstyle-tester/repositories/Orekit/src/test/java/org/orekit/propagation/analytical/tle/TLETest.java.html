46      public void testTLEFormat() throws OrekitException {
47  
48          String line1 = "1 27421U 02021A   02124.48976499 -.00021470  00000-0 -89879-2 0    20";
49          String line2 = "2 27421  98.7490 199.5121 0001333 133.9522 226.1918 14.26113993    62";
50  
51          Assert.assertTrue(TLE.isFormatOK(line1, line2));
52  
53          TLE tle = new TLE(line1, line2);
54          Assert.assertEquals(27421, tle.getSatelliteNumber(), 0);
55          Assert.assertEquals(2002, tle.getLaunchYear());
56          Assert.assertEquals(21, tle.getLaunchNumber());
57          Assert.assertEquals("A", tle.getLaunchPiece());
58          Assert.assertEquals(-0.0089879, tle.getBStar(), 0);
59          Assert.assertEquals(0, tle.getEphemerisType());
60          Assert.assertEquals(98.749, FastMath.toDegrees(tle.getI()), 1e-10);
61          Assert.assertEquals(199.5121, FastMath.toDegrees(tle.getRaan()), 1e-10);
62          Assert.assertEquals(0.0001333, tle.getE(), 1e-10);
63          Assert.assertEquals(133.9522, FastMath.toDegrees(tle.getPerigeeArgument()), 1e-10);
64          Assert.assertEquals(226.1918, FastMath.toDegrees(tle.getMeanAnomaly()), 1e-10);
65          Assert.assertEquals(14.26113993, tle.getMeanMotion() * Constants.JULIAN_DAY / (2 * FastMath.PI), 0);
66          Assert.assertEquals(tle.getRevolutionNumberAtEpoch(), 6, 0);
67          Assert.assertEquals(tle.getElementNumber(), 2 ,0);
68  
69          line1 = "1 27421U 02021A   02124.48976499 -.00021470  00000-0 -89879-2 0    20";
70          line2 = "2 27421  98.7490 199.5121 0001333 133.9522 226.1918 14*26113993    62";
71          Assert.assertFalse(TLE.isFormatOK(line1, line2));
72  
73          line1 = "1 27421 02021A   02124.48976499 -.00021470  00000-0 -89879-2 0    20";
74          line2 = "2 27421  98.7490 199.5121 0001333 133.9522 226.1918 14.26113993    62";
75          Assert.assertFalse(TLE.isFormatOK(line1, line2));
76  
77          line1 = "1 27421U 02021A   02124.48976499 -.00021470  00000-0 -89879-2 0    20";
78          line2 = "2 27421  98.7490 199.5121 0001333 133.9522 226.1918 10006113993    62";
79          Assert.assertFalse(TLE.isFormatOK(line1, line2));
80  
81          line1 = "1 27421U 02021A   02124.48976499 -.00021470  00000-0 -89879 2 0    20";
82          line2 = "2 27421  98.7490 199.5121 0001333 133.9522 226.1918 14.26113993    62";
83          Assert.assertFalse(TLE.isFormatOK(line1, line2));
84      }

122     private void checkSymmetry(String line1, String line2) throws OrekitException {
123         TLE tleRef = new TLE(line1, line2);
124         TLE tle = new TLE(tleRef.getSatelliteNumber(), tleRef.getClassification(),
125                           tleRef.getLaunchYear(), tleRef.getLaunchNumber(), tleRef.getLaunchPiece(),
126                           tleRef.getEphemerisType(), tleRef.getElementNumber(), tleRef.getDate(),
127                           tleRef.getMeanMotion(), tleRef.getMeanMotionFirstDerivative(),
128                           tleRef.getMeanMotionSecondDerivative(), tleRef.getE(), tleRef.getI(),
129                           tleRef.getPerigeeArgument(), tleRef.getRaan(), tleRef.getMeanAnomaly(),
130                           tleRef.getRevolutionNumberAtEpoch(), tleRef.getBStar());
131         Assert.assertEquals(line1, tle.getLine1());
132         Assert.assertEquals(line2, tle.getLine2());
133     }

269     final double taylorConvert(final double m, final int n) {
270         // convert one term of TLE mean motion Taylor series
271         return  m * 2 * FastMath.PI * CombinatoricsUtils.factorial(n) / FastMath.pow(Constants.JULIAN_DAY, n);
272     }

298     @Test
299     public void testSatCodeCompliance() throws IOException, OrekitException, ParseException {
300 
301         BufferedReader rEntry = null;
302         BufferedReader rResults = null;
303 
304         InputStream inEntry =
305             TLETest.class.getResourceAsStream("/tle/extrapolationTest-data/SatCode-entry");
306         rEntry = new BufferedReader(new InputStreamReader(inEntry));
307 
308         try {
309             InputStream inResults =
310                 TLETest.class.getResourceAsStream("/tle/extrapolationTest-data/SatCode-results");
311             rResults = new BufferedReader(new InputStreamReader(inResults));
312 
313             try {
314                 double cumulated = 0; // sum of all differences between test cases and OREKIT results
315                 boolean stop = false;
316 
317                 String rline = rResults.readLine();
318 
319                 while (!stop) {
320                     if (rline == null) break;
321 
322                     String[] title = rline.split(" ");
323 
324                     if (title[0].matches("r")) {
325 
326                         String eline;
327                         int count = 0;
328                         String[] header = new String[4];
329                         for (eline = rEntry.readLine(); (eline != null) && (eline.charAt(0)=='#'); eline = rEntry.readLine()) {
330                             header[count++] = eline;
331                         }
332                         String line1 = eline;
333                         String line2 = rEntry.readLine();
334                         Assert.assertTrue(TLE.isFormatOK(line1, line2));
335 
336                         TLE tle = new TLE(line1, line2);
337 
338                         int satNum = Integer.parseInt(title[1]);
339                         Assert.assertTrue(satNum==tle.getSatelliteNumber());
340                         TLEPropagator ex = TLEPropagator.selectExtrapolator(tle);
341 
342                         for (rline = rResults.readLine(); (rline!=null)&&(rline.charAt(0)!='r'); rline = rResults.readLine()) {
343 
344                             String[] data = rline.split(" ");
345                             double minFromStart = Double.parseDouble(data[0]);
346                             double pX = 1000*Double.parseDouble(data[1]);
347                             double pY = 1000*Double.parseDouble(data[2]);
348                             double pZ = 1000*Double.parseDouble(data[3]);
349                             double vX = 1000*Double.parseDouble(data[4]);
350                             double vY = 1000*Double.parseDouble(data[5]);
351                             double vZ = 1000*Double.parseDouble(data[6]);
352                             Vector3D testPos = new Vector3D(pX, pY, pZ);
353                             Vector3D testVel = new Vector3D(vX, vY, vZ);
354 
355                             AbsoluteDate date = tle.getDate().shiftedBy(minFromStart * 60);
356                             PVCoordinates results = ex.getPVCoordinates(date);
357                             double normDifPos = testPos.subtract(results.getPosition()).getNorm();
358                             double normDifVel = testVel.subtract(results.getVelocity()).getNorm();
359 
360                             cumulated += normDifPos;
361                             Assert.assertEquals(0, normDifPos, 2e-3);;
362                             Assert.assertEquals(0, normDifVel, 1e-5);
363 
364 
365                         }
366                     }
367                 }
368                 Assert.assertEquals(0, cumulated, 0.026);
369             } finally {
370                 if (rResults != null) {
371                     rResults.close();
372                 }
373             }
374         } finally {
375             if (rEntry != null) {
376                 rEntry.close();
377             }
378         }
379     }
