100     public void testIntegrators() throws OrekitException {

138     protected void checkFit(final Orbit orbit, final double duration,
139                             final double stepSize, final double threshold,
140                             final double expectedRMS,
141                             final String... freeParameters)
142         throws OrekitException, IOException, ParseException {
143 
144         NumericalPropagatorBuilder builder =
145                         new NumericalPropagatorBuilder(OrbitType.CARTESIAN.convertType(orbit),
146                                                        new DormandPrince853IntegratorBuilder(minStep, maxStep, dP),
147                                                        PositionAngle.TRUE, dP);
148 
149         ForceModel guessedDrag = drag;
150         ForceModel guessedGravity = gravity;
151         for (String param: freeParameters) {
152             if (DragSensitive.DRAG_COEFFICIENT.equals(param)) {
153                 // we want to adjust drag coefficient, we need to start from a wrong value
154                 ParameterDriver driver = drag.getParameterDriver(param);
155                 double coeff = driver.getReferenceValue() - driver.getScale();
156                 guessedDrag = new DragForce(atmosphere, new IsotropicDrag(crossSection, coeff));
157             } else if (NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT.equals(param)) {
158                 // we want to adjust mu, we need to start from  a wrong value
159                 guessedGravity = new HolmesFeatherstoneAttractionModel(FramesFactory.getITRF(IERSConventions.IERS_2010, true),
160                                                                        GravityFieldFactory.getNormalizedProvider(2, 0));
161                 ParameterDriver driver = guessedGravity.getParameterDriver(param);
162                 driver.setValue(driver.getReferenceValue() + driver.getScale());
163             }
164         }
165         builder.addForceModel(guessedDrag);
166         builder.addForceModel(guessedGravity);
167 
168         JacobianPropagatorConverter fitter = new JacobianPropagatorConverter(builder,
169                                                                              threshold,
170                                                                              5000);
171 
172         fitter.convert(propagator, duration, 1 + (int) (duration / stepSize), freeParameters);
173 
174         NumericalPropagator prop = (NumericalPropagator)fitter.getAdaptedPropagator();
175         Orbit fitted = prop.getInitialState().getOrbit();
176 
177         for (String param: freeParameters) {
178             for (ForceModel force: propagator.getAllForceModels()) {
179                 if (force.isSupported(param)) {
180                     for (ForceModel model: prop.getAllForceModels()) {
181                         if (model.isSupported(param)) {
182                             Assert.assertEquals(force.getParameterDriver(param).getValue(),
183                                                 model.getParameterDriver(param).getValue(),
184                                                 3.0e-4 * FastMath.abs(force.getParameterDriver(param).getValue()));
185                         }
186                     }
187                 }
188             }
189         }
190 
191         Assert.assertEquals(expectedRMS, fitter.getRMS(), 0.01 * expectedRMS);
192 
193         Assert.assertEquals(orbit.getPVCoordinates().getPosition().getX(),
194                             fitted.getPVCoordinates().getPosition().getX(),
195                             1.1);
196         Assert.assertEquals(orbit.getPVCoordinates().getPosition().getY(),
197                             fitted.getPVCoordinates().getPosition().getY(),
198                             1.1);
199         Assert.assertEquals(orbit.getPVCoordinates().getPosition().getZ(),
200                             fitted.getPVCoordinates().getPosition().getZ(),
201                             1.1);
202 
203         Assert.assertEquals(orbit.getPVCoordinates().getVelocity().getX(),
204                             fitted.getPVCoordinates().getVelocity().getX(),
205                             0.0005);
206         Assert.assertEquals(orbit.getPVCoordinates().getVelocity().getY(),
207                             fitted.getPVCoordinates().getVelocity().getY(),
208                             0.0005);
209         Assert.assertEquals(orbit.getPVCoordinates().getVelocity().getZ(),
210                             fitted.getPVCoordinates().getVelocity().getZ(),
211                             0.0005);
212     }

214     protected void checkFit(final ODEIntegratorBuilder foiBuilder) throws OrekitException {
215 
216         NumericalPropagatorBuilder builder = new NumericalPropagatorBuilder(OrbitType.CARTESIAN.convertType(orbit),
217                                                                             foiBuilder,
218                                                                             PositionAngle.TRUE,
219                                                                             1.0);
220 
221         builder.addForceModel(drag);
222         builder.addForceModel(gravity);
223         builder.setAttitudeProvider(InertialProvider.EME2000_ALIGNED);
224         builder.setMass(1000.0);
225 
226         JacobianPropagatorConverter fitter = new JacobianPropagatorConverter(builder, 1.0, 500);
227 
228         fitter.convert(propagator, 1000., 11);
229 
230         NumericalPropagator prop = (NumericalPropagator)fitter.getAdaptedPropagator();
231         Orbit fitted = prop.getInitialState().getOrbit();
232 
233         final double peps = 1.e-1;
234         Assert.assertEquals(orbit.getPVCoordinates().getPosition().getX(),
235                             fitted.getPVCoordinates().getPosition().getX(),
236                             peps * FastMath.abs(orbit.getPVCoordinates().getPosition().getX()));
237         Assert.assertEquals(orbit.getPVCoordinates().getPosition().getY(),
238                             fitted.getPVCoordinates().getPosition().getY(),
239                             peps * FastMath.abs(orbit.getPVCoordinates().getPosition().getY()));
240         Assert.assertEquals(orbit.getPVCoordinates().getPosition().getZ(),
241                             fitted.getPVCoordinates().getPosition().getZ(),
242                             peps * FastMath.abs(orbit.getPVCoordinates().getPosition().getZ()));
243 
244         final double veps = 5.e-1;
245         Assert.assertEquals(orbit.getPVCoordinates().getVelocity().getX(),
246                             fitted.getPVCoordinates().getVelocity().getX(),
247                             veps * FastMath.abs(orbit.getPVCoordinates().getVelocity().getX()));
248         Assert.assertEquals(orbit.getPVCoordinates().getVelocity().getY(),
249                             fitted.getPVCoordinates().getVelocity().getY(),
250                             veps * FastMath.abs(orbit.getPVCoordinates().getVelocity().getY()));
251         Assert.assertEquals(orbit.getPVCoordinates().getVelocity().getZ(),
252                             fitted.getPVCoordinates().getVelocity().getZ(),
253                             veps * FastMath.abs(orbit.getPVCoordinates().getVelocity().getZ()));
254     }

256     @Before
257     public void setUp() throws OrekitException, IOException, ParseException {
258 
259         Utils.setDataRoot("regular-data:potential/shm-format");
260         gravity = new HolmesFeatherstoneAttractionModel(FramesFactory.getITRF(IERSConventions.IERS_2010, true),
261                                                         GravityFieldFactory.getNormalizedProvider(2, 0));
262         mu = gravity.getParameterDriver(NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT).getValue();
263         minStep = 1.0;
264         maxStep = 600.0;
265         dP = 10.0;
266 
267         // use a orbit that comes close to Earth so the drag coefficient has an effect
268         final Vector3D position     = new Vector3D(7.0e6, 1.0e6, 4.0e6).normalize()
269                 .scalarMultiply(Constants.WGS84_EARTH_EQUATORIAL_RADIUS + 300e3);
270         final Vector3D velocity     = new Vector3D(-500.0, 8000.0, 1000.0);
271         final AbsoluteDate initDate = new AbsoluteDate(2010, 10, 10, 10, 10, 10.0, TimeScalesFactory.getUTC());
272         orbit = new EquinoctialOrbit(new PVCoordinates(position,  velocity),
273                                      FramesFactory.getEME2000(), initDate, mu);
274 
275         final double[][] tol = NumericalPropagator.tolerances(dP, orbit, OrbitType.CARTESIAN);
276         propagator = new NumericalPropagator(new DormandPrince853Integrator(minStep, maxStep, tol[0], tol[1]));
277         propagator.setInitialState(new SpacecraftState(orbit));
278         propagator.setOrbitType(OrbitType.CARTESIAN);
279 
280 
281         final OneAxisEllipsoid earth = new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
282                                                             Constants.WGS84_EARTH_FLATTENING,
283                                                             FramesFactory.getITRF(IERSConventions.IERS_2010, true));
284         earth.setAngularThreshold(1.e-7);
285         atmosphere = new SimpleExponentialAtmosphere(earth, 0.0004, 42000.0, 7500.0);
286         final double dragCoef = 2.0;
287         crossSection = 10.0;
288         drag = new DragForce(atmosphere, new IsotropicDrag(crossSection, dragCoef));
289 
290         propagator.addForceModel(gravity);
291         propagator.addForceModel(drag);
292     }
