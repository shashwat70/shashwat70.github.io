57      @Test
58      public void testModifiedData() throws OrekitException, IOException {
59  
60          String directory = "/assets/org/orekit/IERS-conventions/";
61          InputStream is = getClass().getResourceAsStream(directory + "2010/nutation-arguments.txt");
62          BufferedReader reader = new BufferedReader(new InputStreamReader(is, "UTF-8"));
63          StringBuilder builder = new StringBuilder();
64          for (String line = reader.readLine(); line != null; line = reader.readLine()) {
65              builder.append(line);
66              builder.append('\n');
67          }
68          reader.close();
69          try {
70              // remove the line:
71              // F5 ≡ Ω = 125.04455501◦ − 6962890.5431″t + 7.4722″t² + 0.007702″t³ − 0.00005939″t⁴
72              String modified = builder.toString().replaceAll("F5[^\\n]+", "");
73              new FundamentalNutationArguments(IERSConventions.IERS_2010, null,
74                                               new ByteArrayInputStream(modified.getBytes()),
75                                               "modified-data");
76              Assert.fail("an exception should have been thrown");
77          } catch (OrekitException oe) {
78              Assert.assertEquals(OrekitMessages.NOT_A_SUPPORTED_IERS_DATA_FILE, oe.getSpecifier());
79          }
80      }

82      @Test
83      public void testEnum() throws NoSuchMethodException, SecurityException,
84                                    IllegalAccessException, IllegalArgumentException,
85                                    InvocationTargetException {
86          Class<?> e = null;
87          for (final Class<?> c : FundamentalNutationArguments.class.getDeclaredClasses()) {
88              if (c.getName().endsWith("FundamentalName")) {
89                  e = c;
90              }
91          }
92          Method m = e.getDeclaredMethod("valueOf", String.class);
93          m.setAccessible(true);
94          for (String n : Arrays.asList("L", "L_PRIME", "F", "D", "OMEGA",
95                                        "L_ME", "L_VE", "L_E", "L_MA", "L_J", "L_SA", "L_U", "L_NE", "PA")) {
96              Assert.assertEquals(n, m.invoke(null, n).toString());
97          }
98          try {
99              m.invoke(null, "inexistent");
100             Assert.fail("an exception should have been thrown");
101         } catch (InvocationTargetException ite) {
102             Assert.assertTrue(ite.getCause() instanceof IllegalArgumentException);
103         }
104     }

126     private void checkSerialization(int low, int high, FundamentalNutationArguments nutation)
127         throws OrekitException, IOException, ClassNotFoundException {
128         
129         ByteArrayOutputStream bos = new ByteArrayOutputStream();
130         ObjectOutputStream    oos = new ObjectOutputStream(bos);
131         oos.writeObject(nutation);
132 
133         Assert.assertTrue(bos.size() > low);
134         Assert.assertTrue(bos.size() < high);
135 
136         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
137         ObjectInputStream     ois = new ObjectInputStream(bis);
138         FundamentalNutationArguments deserialized  = (FundamentalNutationArguments) ois.readObject();
139         for (double dt = 0; dt < Constants.JULIAN_DAY; dt += 3600) {
140             AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt);
141             BodiesElements be1 = nutation.evaluateAll(date);
142             BodiesElements be2 = deserialized.evaluateAll(date);
143             Assert.assertEquals(be1.getGamma(),  be2.getGamma(),  1.0e-15);
144             Assert.assertEquals(be1.getL(),      be2.getL(),      1.0e-15);
145             Assert.assertEquals(be1.getLPrime(), be2.getLPrime(), 1.0e-15);
146             Assert.assertEquals(be1.getF(),      be2.getF(),      1.0e-15);
147             Assert.assertEquals(be1.getD(),      be2.getD(),      1.0e-15);
148             Assert.assertEquals(be1.getOmega(),  be2.getOmega(),  1.0e-15);
149             Assert.assertEquals(be1.getLMe(),    be2.getLMe(),    1.0e-15);
150             Assert.assertEquals(be1.getLVe(),    be2.getLVe(),    1.0e-15);
151             Assert.assertEquals(be1.getLE(),     be2.getLE(),     1.0e-15);
152             Assert.assertEquals(be1.getLMa(),    be2.getLMa(),    1.0e-15);
153             Assert.assertEquals(be1.getLJu(),    be2.getLJu(),    1.0e-15);
154             Assert.assertEquals(be1.getLSa(),    be2.getLSa(),    1.0e-15);
155             Assert.assertEquals(be1.getLUr(),    be2.getLUr(),    1.0e-15);
156             Assert.assertEquals(be1.getLNe(),    be2.getLNe(),    1.0e-15);
157             Assert.assertEquals(be1.getPa(),     be2.getPa(),     1.0e-15);
158         }
159 
160     }
