17  package org.orekit.files.ccsds;
18  
19  import org.junit.Assert;
20  import org.junit.Test;
21  import org.orekit.errors.OrekitException;
22  import org.orekit.errors.OrekitMessages;
23  
24  
25  public class KeyValueTest {
26  
27      @Test
28      public void testUppercase() {
29          for (final Keyword keyword : Keyword.values()) {
30              KeyValue kv = new KeyValue("  " + keyword + " = " + "aaa", 12, "dummy");
31              Assert.assertEquals(keyword, kv.getKeyword());
32              Assert.assertEquals(keyword.toString(), kv.getKey());
33              if (keyword == Keyword.COMMENT) {
34                  Assert.assertEquals("aaa", kv.getValue());
35              } else {
36                  Assert.assertEquals("AAA", kv.getValue());
37              }
38              try {
39                  kv.getDoubleValue();
40                  Assert.fail("an exception should have been thrown");
41              } catch (OrekitException oe) {
42                  Assert.assertEquals(OrekitMessages.UNABLE_TO_PARSE_LINE_IN_FILE, oe.getSpecifier());
43                  Assert.assertEquals(12, ((Integer) oe.getParts()[0]).intValue());
44                  Assert.assertEquals("dummy", ((String) oe.getParts()[1]));
45              }
46          }
47      }
48  
49      @Test
50      public void testUserDefined() throws OrekitException {
51          KeyValue kv = new KeyValue("USER_DEFINED_TEST_KEY = Orekit", 3, "dummy");
52          Assert.assertEquals(Keyword.USER_DEFINED_X, kv.getKeyword());
53          Assert.assertEquals("USER_DEFINED_TEST_KEY", kv.getKey());
54          Assert.assertEquals("OREKIT", kv.getValue());
55      }
56  
57      @Test
58      public void testUnknownKey() throws OrekitException {
59          KeyValue kv = new KeyValue("INEXISTENT_KEY = 1.0e-5", 5, "dummy");
60          Assert.assertNull(kv.getKeyword());
61          Assert.assertEquals("INEXISTENT_KEY", kv.getKey());
62          Assert.assertEquals("1.0E-5", kv.getValue());
63          Assert.assertEquals(1.0e-5, kv.getDoubleValue(), 1.0e-15);
64      }
65  
66      @Test
67      public void testBadSyntax() throws OrekitException {
68          KeyValue kv = new KeyValue("there are no equal sign in this line", 17, "dummy");
69          Assert.assertNull(kv.getKeyword());
70          Assert.assertEquals("", kv.getKey());
71          Assert.assertEquals("", kv.getValue());
72      }
73  
74      @Test
75      public void testUnderscoreEqualsBlank() throws OrekitException {
76          KeyValue kv = new KeyValue("CENTER_NAME = EARTH_BARYCENTER", 1, "dummy");
77          Assert.assertEquals(Keyword.CENTER_NAME, kv.getKeyword());
78          Assert.assertEquals("EARTH BARYCENTER", kv.getValue());
79      }
80  
81      @Test
82      public void testCoalescedBlanks() throws OrekitException {
83          KeyValue kv = new KeyValue("CENTER_NAME = EARTH    ___     BARYCENTER    ", 1, "dummy");
84          Assert.assertEquals(Keyword.CENTER_NAME, kv.getKeyword());
85          Assert.assertEquals("EARTH BARYCENTER", kv.getValue());
86      }
87  
88  }
