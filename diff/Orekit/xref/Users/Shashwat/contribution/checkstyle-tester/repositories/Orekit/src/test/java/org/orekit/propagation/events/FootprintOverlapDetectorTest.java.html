68      @Test
69      public void testRightForwardView() throws OrekitException, IOException {
70  
71          propagator.setAttitudeProvider(new LofOffset(initialOrbit.getFrame(), LOFType.VVLH,
72                                                        RotationOrder.XYZ,
73                                                        FastMath.toRadians(-20.0),
74                                                        FastMath.toRadians(+20.0),
75                                                        0.0));
76  
77          // observe continental France plus Corsica
78          final SphericalPolygonsSet france = buildFrance();
79  
80          // square field of view along Z axis (which is pointing sideways), aperture 5°, 0° margin
81          final FieldOfView fov = new FieldOfView(Vector3D.PLUS_K, Vector3D.PLUS_I,
82                                                  FastMath.toRadians(2.5), 4, 0.0);
83          final FootprintOverlapDetector detector =
84                  new FootprintOverlapDetector(fov, earth, france, 50000.0).
85                  withMaxCheck(1.0).
86                  withThreshold(1.0e-6).
87                  withHandler(new ContinueOnEvent<FootprintOverlapDetector>());
88          final EventsLogger logger = new EventsLogger();
89          propagator.addEventDetector(logger.monitorDetector(detector));
90  
91          // Extrapolate from the initial to the final date
92          propagator.propagate(initialOrbit.getDate().shiftedBy(635000),
93                               initialOrbit.getDate().shiftedBy(735000));
94  
95          List<LoggedEvent> events = logger.getLoggedEvents();
96          Assert.assertEquals(8, events.size());
97  
98          // the first two consecutive close events occur during the same ascending orbit
99          // we first see Corsica, then lose visibility over the see, then see continental France
100 
101         // above Mediterranean see, between Illes Balears and Sardigna,
102         // pointing to Corsica towards North-East
103         checkEventPair(events.get(0),  events.get(1),
104                        639010.0775,  33.9434, 39.2168,  6.5980, 42.0671,  9.0543);
105 
106         // above Saint-Chamond (Loire), pointing near Saint-Dié-des-Vosges (Vosges) towards North-East
107         checkEventPair(events.get(2),  events.get(3),
108                        639111.1399,  40.8032, 45.4637,  4.5075, 48.3487,  7.1733);
109 
110         // event is on a descending orbit, so the pointing direction,
111         // taking roll and pitch offsets, is towards South-West with respect to spacecraft
112         // above English Channel, pointing near Hanvec (Finistère) towards South-West
113         checkEventPair(events.get(4),  events.get(5),
114                        687772.4531,  27.0852, 50.2693,  0.0493, 48.3243, -4.1510);
115 
116         // event on an ascending orbit
117         // above Atlantic ocean, pointing near to île d'Oléron (Charente-Maritime) towards North-East
118         checkEventPair(events.get(6),  events.get(7),
119                        727696.1033, 113.8829, 42.9785, -4.0426, 45.8492, -1.4656);
120 
121     }

123     private void checkEventPair(final LoggedEvent start, final LoggedEvent end,
124                                 final double expectedStart, final double expectedDuration,
125                                 final double spacecraftLatitude, final double spacecraftLongitude,
126                                 final double fovCenterLatitude, final double fovCenterLongitude)
127         throws OrekitException {
128 
129         Assert.assertFalse(start.isIncreasing());
130         Assert.assertTrue(end.isIncreasing());
131         Assert.assertEquals(expectedStart,
132                             start.getState().getDate().durationFrom(initialOrbit.getDate()),
133                             0.001);
134         Assert.assertEquals(expectedDuration,
135                             end.getState().getDate().durationFrom(start.getState().getDate()),
136                             0.001);
137 
138         SpacecraftState middle = start.getState().shiftedBy(0.5 * expectedDuration);
139 
140         // sub-satellite point
141         Vector3D p = middle.getPVCoordinates().getPosition();
142         GeodeticPoint gpSat = earth.transform(p, middle.getFrame(), middle.getDate());
143         Assert.assertEquals(spacecraftLatitude,  FastMath.toDegrees(gpSat.getLatitude()),  0.001);
144         Assert.assertEquals(spacecraftLongitude, FastMath.toDegrees(gpSat.getLongitude()), 0.001);
145 
146         // point at center of Field Of View
147         final Transform scToInert = middle.toTransform().getInverse();
148         GeodeticPoint gpFOV =
149                 earth.getIntersectionPoint(new Line(p, scToInert.transformPosition(Vector3D.PLUS_K), 1.0e-6),
150                                            middle.getPVCoordinates().getPosition(),
151                                            middle.getFrame(), middle.getDate());
152         Assert.assertEquals(fovCenterLatitude,  FastMath.toDegrees(gpFOV.getLatitude()),  0.001);
153         Assert.assertEquals(fovCenterLongitude, FastMath.toDegrees(gpFOV.getLongitude()), 0.001);
154 
155     }

157     @Test
158     public void testSerialization()
159       throws IOException, ClassNotFoundException, NoSuchFieldException, IllegalAccessException, OrekitException {
160 
161         // observe continental France plus Corsica
162         final SphericalPolygonsSet france = buildFrance();
163 
164         // square field of view along Z axis (which is pointing sideways),
165         // aperture 5° (hence half-aperture 2.5°), 0.001 radians margin
166         final double alpha = FastMath.toRadians(2.5);
167         final FieldOfView fov = new FieldOfView(Vector3D.PLUS_K, Vector3D.PLUS_I,
168                                                 alpha, 4, 0.001);
169         double eta = FastMath.acos(FastMath.sin(alpha) * FastMath.sin(alpha));
170         double theoreticalArea = MathUtils.TWO_PI - 4 * eta;
171         final FootprintOverlapDetector detector =
172                 new FootprintOverlapDetector(fov, earth, france, 50000.0).
173                 withMaxCheck(1.0).
174                 withThreshold(1.0e-6).
175                 withHandler(new ContinueOnEvent<FootprintOverlapDetector>());
176 
177         Assert.assertEquals(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
178                             ((OneAxisEllipsoid) detector.getBody()).getEquatorialRadius(),
179                             1.0e-12);
180         Assert.assertEquals(0.001, detector.getFieldOfView().getMargin(), 1.0e-12);
181         Assert.assertEquals(theoreticalArea, detector.getFieldOfView().getZone().getSize(), 1.0e-12);
182 
183         ByteArrayOutputStream bos = new ByteArrayOutputStream();
184         ObjectOutputStream    oos = new ObjectOutputStream(bos);
185         oos.writeObject(detector);
186 
187         Assert.assertTrue(bos.size() > 2400);
188         Assert.assertTrue(bos.size() < 2500);
189 
190         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
191         ObjectInputStream     ois = new ObjectInputStream(bis);
192         FootprintOverlapDetector deserialized  = (FootprintOverlapDetector) ois.readObject();
193 
194         Assert.assertEquals(detector.getZone().getSize(),         deserialized.getZone().getSize(),         1.0e-14);
195         Assert.assertEquals(detector.getZone().getBoundarySize(), deserialized.getZone().getBoundarySize(), 1.0e-14);
196         Assert.assertEquals(detector.getZone().getTolerance(),    deserialized.getZone().getTolerance(),    1.0e-15);
197         Assert.assertEquals(detector.getMaxCheckInterval(),       deserialized.getMaxCheckInterval(),       1.0e-15);
198         Assert.assertEquals(detector.getThreshold(),              deserialized.getThreshold(),              1.0e-15);
199         Assert.assertEquals(detector.getMaxIterationCount(),      deserialized.getMaxIterationCount());
200 
201         Assert.assertTrue(new RegionFactory<Sphere2D>().difference(detector.getZone(), deserialized.getZone()).isEmpty());
202 
203     }

261     private SphericalPolygonsSet buildSimpleZone(double[][] points) {
262         final S2Point[] vertices = new S2Point[points.length];
263         for (int i = 0; i < points.length; ++i) {
264             vertices[i] = new S2Point(FastMath.toRadians(points[i][1]),         // points[i][1] is longitude
265                                       FastMath.toRadians(90.0 - points[i][0])); // points[i][0] is latitude
266         }
267         return new SphericalPolygonsSet(1.0e-10, vertices);
268     }
