27  /** Finder for satellite eclipse related events.
28   * <p>This class finds eclipse events, i.e. satellite within umbra (total
29   * eclipse) or penumbra (partial eclipse).</p>
30   * <p>The default implementation behavior is to {@link
31   * org.orekit.propagation.events.handlers.EventHandler.Action#CONTINUE continue}
32   * propagation when entering the eclipse and to {@link
33   * org.orekit.propagation.events.handlers.EventHandler.Action#STOP stop} propagation
34   * when exiting the eclipse. This can be changed by calling {@link
35   * #withHandler(EventHandler)} after construction.</p>
36   * @see org.orekit.propagation.Propagator#addEventDetector(EventDetector)
37   * @author Pascal Parraud
38   */
39  public class EclipseDetector extends AbstractDetector<EclipseDetector> {
40  
41      /** Serializable UID. */
42      private static final long serialVersionUID = 20131118L;
43  
44      /** Occulting body. */
45      private final PVCoordinatesProvider occulting;
46  
47      /** Occulting body radius (m). */
48      private final double occultingRadius;
49  
50      /** Occulted body. */
51      private final PVCoordinatesProvider occulted;
52  
53      /** Occulted body radius (m). */
54      private final double occultedRadius;
55  
56      /** Umbra, if true, or penumbra, if false, detection flag. */
57      private boolean totalEclipse;
58  
59      /** Build a new eclipse detector.
60       * <p>The new instance is a total eclipse (umbra) detector with default
61       * values for maximal checking interval ({@link #DEFAULT_MAXCHECK})
62       * and convergence threshold ({@link #DEFAULT_THRESHOLD}).</p>
63       * @param occulted the body to be occulted
64       * @param occultedRadius the radius of the body to be occulted (m)
65       * @param occulting the occulting body
66       * @param occultingRadius the occulting body radius (m)
67       */
68      public EclipseDetector(final PVCoordinatesProvider occulted,  final double occultedRadius,
69                             final PVCoordinatesProvider occulting, final double occultingRadius) {
70          this(DEFAULT_MAXCHECK, DEFAULT_THRESHOLD,
71               occulted, occultedRadius, occulting, occultingRadius);
72      }
73  
74      /** Build a new eclipse detector.
75       * <p>The new instance is a total eclipse (umbra) detector with default
76       * value for convergence threshold ({@link #DEFAULT_THRESHOLD}).</p>
77       * <p>The maximal interval between eclipse checks should be smaller than
78       * the half duration of the minimal pass to handle, otherwise some short
79       * passes could be missed.</p>
80       * @param maxCheck maximal checking interval (s)
81       * @param occulted the body to be occulted
82       * @param occultedRadius the radius of the body to be occulted in meters
83       * @param occulting the occulting body
84       * @param occultingRadius the occulting body radius in meters
85       */
86      public EclipseDetector(final double maxCheck,
87                             final PVCoordinatesProvider occulted,  final double occultedRadius,
88                             final PVCoordinatesProvider occulting, final double occultingRadius) {
89          this(maxCheck, DEFAULT_THRESHOLD,
90               occulted, occultedRadius, occulting, occultingRadius);
91      }
92  
93      /** Build a new eclipse detector.
94       * <p>The new instance is a total eclipse (umbra) detector.</p>
95       * <p>The maximal interval between eclipse checks should be smaller than
96       * the half duration of the minimal pass to handle, otherwise some short
97       * passes could be missed.</p>
98       * @param maxCheck maximal checking interval (s)
99       * @param threshold convergence threshold (s)
100      * @param occulted the body to be occulted
101      * @param occultedRadius the radius of the body to be occulted in meters
102      * @param occulting the occulting body
103      * @param occultingRadius the occulting body radius in meters
104      */
105     public EclipseDetector(final double maxCheck, final double threshold,
106                            final PVCoordinatesProvider occulted,  final double occultedRadius,
107                            final PVCoordinatesProvider occulting, final double occultingRadius) {
108         this(maxCheck, threshold, DEFAULT_MAX_ITER, new StopOnIncreasing<EclipseDetector>(),
109              occulted, occultedRadius, occulting, occultingRadius, true);
110     }
111 
112     /** Private constructor with full parameters.
113      * <p>
114      * This constructor is private as users are expected to use the builder
115      * API with the various {@code withXxx()} methods to set up the instance
116      * in a readable manner without using a huge amount of parameters.
117      * </p>
118      * @param maxCheck maximum checking interval (s)
119      * @param threshold convergence threshold (s)
120      * @param maxIter maximum number of iterations in the event time search
121      * @param handler event handler to call at event occurrences
122      * @param occulted the body to be occulted
123      * @param occultedRadius the radius of the body to be occulted in meters
124      * @param occulting the occulting body
125      * @param occultingRadius the occulting body radius in meters
126      * @param totalEclipse umbra (true) or penumbra (false) detection flag
127      * @since 6.1
128      */
129     private EclipseDetector(final double maxCheck, final double threshold,
130                             final int maxIter, final EventHandler<? super EclipseDetector> handler,
131                             final PVCoordinatesProvider occulted,  final double occultedRadius,
132                             final PVCoordinatesProvider occulting, final double occultingRadius,
133                             final boolean totalEclipse) {
134         super(maxCheck, threshold, maxIter, handler);
135         this.occulted        = occulted;
136         this.occultedRadius  = FastMath.abs(occultedRadius);
137         this.occulting       = occulting;
138         this.occultingRadius = FastMath.abs(occultingRadius);
139         this.totalEclipse    = totalEclipse;
140     }
141 
142     /** {@inheritDoc} */
143     @Override
144     protected EclipseDetector create(final double newMaxCheck, final double newThreshold,
145                                      final int nawMaxIter, final EventHandler<? super EclipseDetector> newHandler) {
146         return new EclipseDetector(newMaxCheck, newThreshold, nawMaxIter, newHandler,
147                                    occulted, occultedRadius, occulting, occultingRadius, totalEclipse);
148     }

152      * <p>
153      * This will override a penumbra/umbra flag if it has been configured previously.
154      * </p>
155      * @return a new detector with updated configuration (the instance is not changed)
156      * @see #withPenumbra()
157      * @since 6.1
158      */
159     public EclipseDetector withUmbra() {
160         return new EclipseDetector(getMaxCheckInterval(), getThreshold(), getMaxIterationCount(), getHandler(),
161                                    occulted, occultedRadius, occulting, occultingRadius,
162                                    true);
163     }

165     /**
166      * Setup the detector to penumbra detection.
167      * <p>
168      * This will override a penumbra/umbra flag if it has been configured previously.
169      * </p>
170      * @return a new detector with updated configuration (the instance is not changed)
171      * @see #withUmbra()
172      * @since 6.1
173      */
174     public EclipseDetector withPenumbra() {
175         return new EclipseDetector(getMaxCheckInterval(), getThreshold(), getMaxIterationCount(), getHandler(),
176                                    occulted, occultedRadius, occulting, occultingRadius,
177                                    false);

216     /** Compute the value of the switching function.
217      * This function becomes negative when entering the region of shadow
218      * and positive when exiting.
219      * @param s the current state information: date, kinematics, attitude
220      * @return value of the switching function
221      * @exception OrekitException if some specific error occurs
222      */
223     public double g(final SpacecraftState s) throws OrekitException {
224         final Vector3D pted = occulted.getPVCoordinates(s.getDate(), s.getFrame()).getPosition();
225         final Vector3D ping = occulting.getPVCoordinates(s.getDate(), s.getFrame()).getPosition();
226         final Vector3D psat = s.getPVCoordinates().getPosition();
227         final Vector3D ps   = pted.subtract(psat);
228         final Vector3D po   = ping.subtract(psat);
229         final double angle  = Vector3D.angle(ps, po);
230         final double rs     = FastMath.asin(occultedRadius / ps.getNorm());
231         if (Double.isNaN(rs)) {
232             return FastMath.PI;
233         }
234         final double ro     = FastMath.asin(occultingRadius / po.getNorm());
235         if (Double.isNaN(ro)) {
236             return -FastMath.PI;
237         }
238         return totalEclipse ? (angle - ro + rs) : (angle - ro - rs);
239     }
