71      /**
72       * check the acceleration from relitivity
73       *
74       * @throws OrekitException on error
75       */
76      @Test
77      public void testAcceleration() throws OrekitException {
78          double gm = Constants.EIGEN5C_EARTH_MU;
79          Relativity relativity = new Relativity(gm);
80          AccelerationRetriever adder = new AccelerationRetriever();
81          final Vector3D p = new Vector3D(3777828.75000531, -5543949.549783845, 2563117.448578311);
82          final Vector3D v = new Vector3D(489.0060271721, -2849.9328929417, -6866.4671013153);
83          SpacecraftState s = new SpacecraftState(new CartesianOrbit(
84                  new PVCoordinates(p, v),
85                  frame,
86                  date,
87                  gm
88          ));
89  
90          //action
91          relativity.addContribution(s, adder);
92  
93          //verify
94          //force is ~1e-8 so this give ~3 sig figs.
95          double tol = 2e-11;
96          Vector3D circularApproximation = p.normalize().scalarMultiply(
97                  gm / p.getNormSq() * 3 * v.getNormSq() / (c * c));
98          Assert.assertEquals(
99                  0,
100                 adder.getAcceleration().subtract(circularApproximation).getNorm(),
101                 tol);
102         //check derivatives
103         final DerivativeStructure mass = new DerivativeStructure(7, 1, 0);
104         final Vector3D actualDerivatives = relativity
105                 .accelerationDerivatives(date, frame, ds(p, 0), ds(v, 3), identity, mass)
106                 .toVector3D();
107         Assert.assertEquals(
108                 0,
109                 actualDerivatives.subtract(circularApproximation).getNorm(),
110                 tol);
111     }

113     /**
114      * Check a nearly circular orbit.
115      *
116      * @throws OrekitException on error
117      */
118     @Test
119     public void testAccelerationCircular() throws OrekitException {
120         double gm = Constants.EIGEN5C_EARTH_MU;
121         double re = Constants.WGS84_EARTH_EQUATORIAL_RADIUS;
122         Relativity relativity = new Relativity(gm);
123         AccelerationRetriever adder = new AccelerationRetriever();
124         final CircularOrbit orbit = new CircularOrbit(
125                 re + 500e3, 0, 0, FastMath.toRadians(41.2), -1, 3, PositionAngle.TRUE,
126                 frame,
127                 date,
128                 gm
129         );
130         SpacecraftState state = new SpacecraftState(orbit);
131 
132         //action
133         relativity.addContribution(state, adder);
134 
135         //verify
136         //force is ~1e-8 so this give ~7 sig figs.
137         double tol = 2e-10;
138         PVCoordinates pv = state.getPVCoordinates();
139         Vector3D p = pv.getPosition();
140         Vector3D v = pv.getVelocity();
141         Vector3D circularApproximation = p.normalize().scalarMultiply(
142                 gm / p.getNormSq() * 3 * v.getNormSq() / (c * c));
143         Assert.assertEquals(
144                 0,
145                 adder.getAcceleration().subtract(circularApproximation).getNorm(),
146                 tol);
147         //check derivatives
148         DerivativeStructure mass = new DerivativeStructure(7, 1, 6, 1);
149         final FieldVector3D<DerivativeStructure> pDS = ds(p, 0);
150         final FieldVector3D<DerivativeStructure> vDS = ds(v, 3);
151         FieldVector3D<DerivativeStructure> gradient =
152                 relativity.accelerationDerivatives(date, frame, pDS, vDS, identity, mass);
153         Assert.assertEquals(
154                 0,
155                 gradient.toVector3D().subtract(circularApproximation).getNorm(),
156                 tol);
157         double r = p.getNorm();
158         double s = v.getNorm();
159         final double[] actualdx = gradient.getX().getAllDerivatives();
160         final double x = p.getX();
161         final double vx = v.getX();
162         double expectedDxDx = gm / (c * c * r * r * r * r * r) *
163                 (-13 * x * x * s * s + 3 * r * r * s * s + 4 * r * r * vx * vx);
164         Assert.assertEquals(expectedDxDx, actualdx[1], 2);
165     }

167     /**
168      * create a DS version of a vector.
169      *
170      * @param v the vector
171      * @param i the start index
172      * @return v as a DS vector
173      */
174     private static FieldVector3D<DerivativeStructure> ds(Vector3D v, int i) {
175         return new FieldVector3D<DerivativeStructure>(
176                 new DerivativeStructure(7, 1, i, v.getX()),
177                 new DerivativeStructure(7, 1, i + 1, v.getY()),
178                 new DerivativeStructure(7, 1, i + 2, v.getZ())
179         );
180     }

182     /**
183      * check against example in Tapley, Schutz, and Born, p 65-66. They predict a
184      * progression of perigee of 11 arcsec/year. To get the same results we must set the
185      * propagation tolerances to 1e-5.
186      *
187      * @throws OrekitException on error
188      */
189     @Test
190     public void testSmallEffectOnOrbit() throws OrekitException {
191         //setup
192         final double gm = Constants.EIGEN5C_EARTH_MU;
193         Orbit orbit =
194                 new KeplerianOrbit(
195                         7500e3, 0.025, FastMath.toRadians(41.2), 0, 0, 0, PositionAngle.TRUE,
196                         frame,
197                         date,
198                         gm
199                 );
200         double[][] tol = NumericalPropagator.tolerances(0.00001, orbit, OrbitType.CARTESIAN);
201         AbstractIntegrator integrator = new DormandPrince853Integrator(1, 3600, tol[0], tol[1]);
202         NumericalPropagator propagator = new NumericalPropagator(integrator);
203         propagator.setOrbitType(OrbitType.CARTESIAN);
204         propagator.addForceModel(new Relativity(gm));
205         propagator.setInitialState(new SpacecraftState(orbit));
206 
207         //action: propagate a period
208         AbsoluteDate end = orbit.getDate().shiftedBy(30 * Constants.JULIAN_DAY);
209         PVCoordinates actual = propagator.getPVCoordinates(end, frame);
210 
211         //verify
212         KeplerianOrbit endOrbit = new KeplerianOrbit(actual, frame, end, gm);
213         KeplerianOrbit startOrbit = new KeplerianOrbit(orbit);
214         double dp = endOrbit.getPerigeeArgument() - startOrbit.getPerigeeArgument();
215         double dtYears = end.durationFrom(orbit.getDate()) / Constants.JULIAN_YEAR;
216         double dpDeg = FastMath.toDegrees(dp);
217         //change in argument of perigee in arcseconds per year
218         double arcsecPerYear = dpDeg * 3600 / dtYears;
219         Assert.assertEquals(11, arcsecPerYear, 0.5);
220     }

240                 0);
