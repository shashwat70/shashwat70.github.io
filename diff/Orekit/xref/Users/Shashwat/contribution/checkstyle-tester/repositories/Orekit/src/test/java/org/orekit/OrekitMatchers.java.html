43      /**
44       * Match a geodetic point
45       *
46       * @param lat latitude matcher, in radians
47       * @param lon longitude matcher, in radians
48       * @param alt altitude matcher, in meters
49       * @return a {@link GeodeticPoint} matcher
50       */
51      public static Matcher<GeodeticPoint> geodeticPoint(
52              final Matcher<Double> lat,
53              final Matcher<Double> lon,
54              final Matcher<Double> alt) {
55          return new TypeSafeDiagnosingMatcher<GeodeticPoint>() {
56              @Override
57              public void describeTo(Description description) {
58                  description.appendList("GeodeticPoint[", ",", "]",
59                          Arrays.<SelfDescribing>asList(lat, lon, alt));
60              }
61  
62              @Override
63              protected boolean matchesSafely(GeodeticPoint item,
64                                              Description mismatchDescription) {
65                  if (!lat.matches(item.getLatitude())) {
66                      mismatchDescription.appendText("the latitude ");
67                      lat.describeMismatch(item.getLatitude(),
68                              mismatchDescription);
69                      return false;
70                  }
71                  if (!lon.matches(item.getLongitude())) {
72                      mismatchDescription.appendText("the longitude ");
73                      lon.describeMismatch(item.getLongitude(),
74                              mismatchDescription);
75                      return false;
76                  }
77                  if (!alt.matches(item.getAltitude())) {
78                      mismatchDescription.appendText("the altitude ");
79                      alt.describeMismatch(item.getAltitude(),
80                              mismatchDescription);
81                      return false;
82                  }
83                  return true;
84              }
85          };
86      }

88      /**
89       * Match a geodetic point
90       *
91       * @param lat latitude, in radians
92       * @param lon longitude, in radians
93       * @param alt altitude, in meters
94       * @return matcher of a {@link GeodeticPoint}
95       */
96      public static Matcher<GeodeticPoint> geodeticPoint(double lat,
97                                                         double lon,
98                                                         double alt) {
99          return geodeticPoint(is(lat), is(lon), is(alt));
100     }

102     /**
103      * Match a geodetic point by comparing it with another one.
104      *
105      * @param expected the expected value
106      * @param absTol   the absolute tolerance on the comparison, in meters.
107      *                 Differences less than this value will be ignored.
108      * @return a {@link GeodeticPoint} matcher
109      */
110     public static Matcher<GeodeticPoint> geodeticPointCloseTo(
111             GeodeticPoint expected, double absTol) {
112         double angularAbsTol = absTol / Constants.WGS84_EARTH_EQUATORIAL_RADIUS;
113         return geodeticPoint(closeTo(expected.getLatitude(), angularAbsTol),
114                 closeTo(expected.getLongitude(), angularAbsTol),
115                 closeTo(expected.getAltitude(), absTol));
116     }

118     /**
119      * Match a geodetic point by comparing it with another one.
120      *
121      * @param expected the expected value
122      * @param ulps     the ulps difference allowed
123      * @return a {@link GeodeticPoint} matcher
124      * @see #relativelyCloseTo
125      */
126     public static Matcher<GeodeticPoint> geodeticPointCloseTo(
127             GeodeticPoint expected, int ulps) {
128         return geodeticPoint(relativelyCloseTo(expected.getLatitude(), ulps),
129                 relativelyCloseTo(expected.getLongitude(), ulps),
130                 relativelyCloseTo(expected.getAltitude(), ulps));
131     }

133     /**
134      * Matches a {@link Vector3D} based on its three coordinates.
135      *
136      * @param x matcher for the x coordinate
137      * @param y matcher for the y coordinate
138      * @param z matcher for the z coordinate
139      * @return a vector matcher
140      */
141     public static Matcher<Vector3D> vector(final Matcher<Double> x,
142                                            final Matcher<Double> y,
143                                            final Matcher<Double> z) {
144         return new TypeSafeDiagnosingMatcher<Vector3D>() {
145             @Override
146             public void describeTo(Description description) {
147                 description.appendList("Vector3D[", ",", "]",
148                         Arrays.<SelfDescribing>asList(x, y, z));
149             }
150 
151             @Override
152             protected boolean matchesSafely(Vector3D item,
153                                             Description mismatchDescription) {
154                 if (!x.matches(item.getX())) {
155                     mismatchDescription.appendText("the x coordinate ");
156                     x.describeMismatch(item.getX(), mismatchDescription);
157                     return false;
158                 }
159                 if (!y.matches(item.getY())) {
160                     mismatchDescription.appendText("the y coordinate ");
161                     y.describeMismatch(item.getY(), mismatchDescription);
162                     return false;
163                 }
164                 if (!z.matches(item.getZ())) {
165                     mismatchDescription.appendText("the z coordinate ");
166                     z.describeMismatch(item.getZ(), mismatchDescription);
167                     return false;
168                 }
169                 return true;
170             }
171         };
172     }

174     /**
175      * Matches a {@link Vector3D} close to another one.
176      *
177      * @param vector the reference vector
178      * @param absTol absolute tolerance of comparison, in each dimension
179      * @return a vector matcher.
180      */
181     public static Matcher<Vector3D> vectorCloseTo(Vector3D vector, double absTol) {
182         return vector(closeTo(vector.getX(), absTol),
183                 closeTo(vector.getY(), absTol), closeTo(vector.getZ(), absTol));
184     }

186     /**
187      * Matches a {@link Vector3D} close to another one.
188      *
189      * @param vector the reference vector
190      * @param ulps   the relative tolerance, in units in last place, of the
191      *               Comparison of each dimension.
192      * @return a vector matcher.
193      */
194     public static Matcher<Vector3D> vectorCloseTo(Vector3D vector, int ulps) {
195         return vector(relativelyCloseTo(vector.getX(), ulps),
196                 relativelyCloseTo(vector.getY(), ulps),
197                 relativelyCloseTo(vector.getZ(), ulps));
198     }

213     /**
214      * Matches a {@link Vector3D} to another one.
215      *
216      * @param vector the reference vector
217      * @param absTol the absolute tolerance of comparison, in each dimension.
218      * @param ulps   the relative tolerance of comparison in each dimension, in
219      *               units in last place.
220      * @return a matcher that matches if either the absolute or relative
221      * comparison matches in each dimension.
222      */
223     public static Matcher<Vector3D> vectorCloseTo(Vector3D vector,
224                                                   double absTol, int ulps) {
225         return vector(numberCloseTo(vector.getX(), absTol, ulps),
226                 numberCloseTo(vector.getY(), absTol, ulps),
227                 numberCloseTo(vector.getZ(), absTol, ulps));
228     }

230     /**
231      * Match a {@link PVCoordinates}
232      *
233      * @param position matcher for the position
234      * @param velocity matcher for the velocity
235      * @return a matcher of {@link PVCoordinates}
236      */
237     public static Matcher<PVCoordinates> pvIs(
238             final Matcher<? super Vector3D> position,
239             final Matcher<? super Vector3D> velocity) {
240         return new TypeSafeDiagnosingMatcher<PVCoordinates>() {
241 
242             @Override
243             public void describeTo(Description description) {
244                 description.appendText("position ");
245                 description.appendDescriptionOf(position);
246                 description.appendText(" and velocity ");
247                 description.appendDescriptionOf(velocity);
248             }
249 
250             @Override
251             protected boolean matchesSafely(PVCoordinates item,
252                     Description mismatchDescription) {
253                 if (!position.matches(item.getPosition())) {
254                     // position doesn't match
255                     mismatchDescription.appendText("position ");
256                     position.describeMismatch(item.getPosition(), mismatchDescription);
257                     return false;
258                 } else if (!velocity.matches(item.getVelocity())) {
259                     // velocity doesn't match
260                     mismatchDescription.appendText("velocity ");
261                     velocity.describeMismatch(item.getVelocity(), mismatchDescription);
262                     return false;
263                 } else {
264                     // both p and v matched
265                     return true;
266                 }
267             }
268         };
269     }

271     /**
272      * Check that a {@link PVCoordinates} is the same as another one.
273      *
274      * @param pv the reference {@link PVCoordinates}
275      * @return a {@link PVCoordinates} {@link Matcher}
276      */
277     public static Matcher<PVCoordinates> pvIs(PVCoordinates pv) {
278         return pvCloseTo(pv, 0);
279     }

281     /**
282      * Match a {@link PVCoordinates} close to another one.
283      *
284      * @param pv     the reference {@link PVCoordinates}
285      * @param absTol distance a matched {@link PVCoordinates} can be from the reference in
286      *               any one coordinate.
287      * @return a matcher of {@link PVCoordinates}.
288      */
289     public static Matcher<PVCoordinates> pvCloseTo(PVCoordinates pv,
290                                                    double absTol) {
291         return pvIs(vectorCloseTo(pv.getPosition(), absTol),
292                 vectorCloseTo(pv.getVelocity(), absTol));
293     }

296      * Match a {@link PVCoordinates} close to another one.
297      *
298      * @param pv   the reference {@link PVCoordinates}
299      * @param ulps the units in last place any coordinate can be off by.
300      * @return a matcher of {@link PVCoordinates}.

302     public static Matcher<PVCoordinates> pvCloseTo(PVCoordinates pv, int ulps) {
303         return pvIs(vectorCloseTo(pv.getPosition(), ulps),
304                 vectorCloseTo(pv.getVelocity(), ulps));

307     /**
308      * Checks if two numbers are relatively close to each other. For absolute
309      * comparisons, see {@link #closeTo(double, double)}.
310      *
311      * @param expected the expected value in the relative comparison
312      * @param ulps     the units in last place of {@code expected} the two
313      *                 numbers can be off by.
314      * @return a matcher of numbers
315      */
316     public static Matcher<Double> relativelyCloseTo(final double expected,
317                                                     final int ulps) {
318         return new TypeSafeDiagnosingMatcher<Double>() {
319 
320             @Override
321             public void describeTo(Description description) {
322                 description.appendText("a numeric value within ")
323                         .appendValue(ulps).appendText(" ulps of ")
324                         .appendValue(expected);
325             }
326 
327             @Override
328             protected boolean matchesSafely(Double item,
329                                             Description mismatchDescription) {
330                 if (!Precision.equals(item, expected, ulps)) {
331                     mismatchDescription
332                             .appendValue(item)
333                             .appendText(" was off by ")
334                             .appendValue(
335                                     Double.doubleToLongBits(item)
336                                             - Double.doubleToLongBits(expected))
337                             .appendText(" ulps");
338                     return false;
339                 }
340                 return true;
341             }
342         };
343     }
344 
345     /**
346      * Check a number is close to another number using a relative
347      * <strong>or</strong> absolute comparison.
348      *
349      * @param number the expected value
350      * @param absTol absolute tolerance of comparison
351      * @param ulps   units in last place tolerance for relative comparison
352      * @return a matcher that matches if the differences is less than or equal
353      * to absTol <strong>or</strong> the two numbers differ by less or equal
354      * ulps.
355      */
356     public static Matcher<Double> numberCloseTo(double number, double absTol,
357                                                 int ulps) {
358         return either(closeTo(number, absTol)).or(
359                 relativelyCloseTo(number, ulps));
360     }

366     /**
367      * Creates a matcher of {@link Double}s that matches when an examined double
368      * is equal to the specified <code>operand</code>, within a range of +/-
369      * <code>error</code>. <p/> For example:
370      * <pre>assertThat(1.03, is(closeTo(1.0, 0.03)))</pre>
371      *
372      * @param value the expected value of matching doubles
373      * @param delta the delta (+/-) within which matches will be allowed
374      * @return a double matcher.
375      */
376     public static Matcher<Double> closeTo(final double value,
377                                           final double delta) {
378 
379         return new TypeSafeMatcher<Double>() {
380             @Override
381             public boolean matchesSafely(Double item) {
382                 return actualDelta(item) <= 0.0;
383             }
384 
385             @Override
386             public void describeMismatchSafely(Double item, Description mismatchDescription) {
387                 mismatchDescription.appendValue(item)
388                         .appendText(" differed by ")
389                         .appendValue(actualDelta(item));
390             }
391 
392             @Override
393             public void describeTo(Description description) {
394                 description.appendText("a numeric value within ")
395                         .appendValue(delta)
396                         .appendText(" of ")
397                         .appendValue(value);
398             }
399 
400             private double actualDelta(Double item) {
401                 return (FastMath.abs((item - value)) - delta);
402             }
403         };
