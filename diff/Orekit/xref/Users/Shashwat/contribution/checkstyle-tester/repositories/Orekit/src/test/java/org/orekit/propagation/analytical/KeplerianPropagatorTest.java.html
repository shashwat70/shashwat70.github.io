93                  PositionAngle.TRUE, FramesFactory.getGCRF(), initDate, mu);
94          Propagator propagator = new KeplerianPropagator(ic);
95          AbsoluteDate end = initDate.shiftedBy(90 * 60);
96  
97          // action
98          final List<SpacecraftState> states = new ArrayList<>();
99          propagator.setEphemerisMode((interpolator, isLast) -> {
100             states.add(interpolator.getCurrentState());
101             states.add(interpolator.getPreviousState());

103         propagator.propagate(end);
104         final BoundedPropagator ephemeris = propagator.getGeneratedEphemeris();
105 
106         //verify

108         for (SpacecraftState state : states) {
109             PVCoordinates actual =
110                     ephemeris.propagate(state.getDate()).getPVCoordinates();
111             Assert.assertThat(actual, OrekitMatchers.pvIs(state.getPVCoordinates()));

188     @Test
189     public void propagatedCartesian() throws OrekitException {
190 
191         // Definition of initial conditions with position and velocity
192         //------------------------------------------------------------
193         Vector3D position = new Vector3D(7.0e6, 1.0e6, 4.0e6);
194         Vector3D velocity = new Vector3D(-500.0, 8000.0, 1000.0);
195         double mu = 3.9860047e14;
196 
197         AbsoluteDate initDate = AbsoluteDate.J2000_EPOCH.shiftedBy(584.);
198         Orbit initialOrbit = new EquinoctialOrbit(new PVCoordinates(position, velocity),
199                                                   FramesFactory.getEME2000(), initDate, mu);
200 
201         // Extrapolator definition
202         // -----------------------
203         KeplerianPropagator extrapolator = new KeplerianPropagator(initialOrbit);
204 
205         // Extrapolation at a final date different from initial date
206         // ---------------------------------------------------------
207         double delta_t = 100000.0; // extrapolation duration in seconds
208         AbsoluteDate extrapDate = initDate.shiftedBy(delta_t);
209 
210         SpacecraftState finalOrbit = extrapolator.propagate(extrapDate);
211 
212 
213         // computation of (M final - M initial) with another method
214         double a = finalOrbit.getA();
215         // another way to compute n
216         double n = FastMath.sqrt(finalOrbit.getMu()/FastMath.pow(a, 3));
217 
218         Assert.assertEquals(n * delta_t,
219                             finalOrbit.getLM() - initialOrbit.getLM(),
220                             Utils.epsilonAngle);
221 
222         // computation of M final orbit
223         double LM = finalOrbit.getLE()
224         - finalOrbit.getEquinoctialEx()*FastMath.sin(finalOrbit.getLE())
225         + finalOrbit.getEquinoctialEy()*FastMath.cos(finalOrbit.getLE());
226 
227         Assert.assertEquals(LM , finalOrbit.getLM() , Utils.epsilonAngle);
228 
229         // test of tan ((LE - Lv)/2) :
230         Assert.assertEquals(FastMath.tan((finalOrbit.getLE() - finalOrbit.getLv())/2.),
231                      tangLEmLv(finalOrbit.getLv(),finalOrbit.getEquinoctialEx(),finalOrbit.getEquinoctialEy()),
232                      Utils.epsilonAngle);
233 
234         // test of evolution of M vs E: LM = LE - ex*sin(LE) + ey*cos(LE)
235         // with ex and ey the same for initial and final orbit
236         double deltaM = finalOrbit.getLM() - initialOrbit.getLM();
237         double deltaE = finalOrbit.getLE() - initialOrbit.getLE();
238         double delta  = finalOrbit.getEquinoctialEx() * (FastMath.sin(finalOrbit.getLE()) - FastMath.sin(initialOrbit.getLE()))
239         - finalOrbit.getEquinoctialEy() * (FastMath.cos(finalOrbit.getLE()) - FastMath.cos(initialOrbit.getLE()));
240 
241         Assert.assertEquals(deltaM, deltaE - delta, Utils.epsilonAngle);
242 
243         // the orbital elements except for Mean/True/Eccentric latitude arguments are the same
244         Assert.assertEquals(finalOrbit.getA(), initialOrbit.getA(), Utils.epsilonTest * initialOrbit.getA());
245         Assert.assertEquals(finalOrbit.getEquinoctialEx(), initialOrbit.getEquinoctialEx(), Utils.epsilonE);
246         Assert.assertEquals(finalOrbit.getEquinoctialEy(), initialOrbit.getEquinoctialEy(), Utils.epsilonE);
247         Assert.assertEquals(finalOrbit.getHx(), initialOrbit.getHx(), Utils.epsilonAngle);
248         Assert.assertEquals(finalOrbit.getHy(), initialOrbit.getHy(), Utils.epsilonAngle);
249 
250         // for final orbit
251         double ex = finalOrbit.getEquinoctialEx();
252         double ey = finalOrbit.getEquinoctialEy();
253         double hx = finalOrbit.getHx();
254         double hy = finalOrbit.getHy();
255         double LE = finalOrbit.getLE();
256 
257         double ex2 = ex*ex;
258         double ey2 = ey*ey;
259         double hx2 = hx*hx;
260         double hy2 = hy*hy;
261         double h2p1 = 1. + hx2 + hy2;
262         double beta = 1. / (1. + FastMath.sqrt(1. - ex2 - ey2));
263 
264         double x3 = -ex + (1.- beta*ey2)*FastMath.cos(LE) + beta*ex*ey*FastMath.sin(LE);
265         double y3 = -ey + (1. -beta*ex2)*FastMath.sin(LE) + beta*ex*ey*FastMath.cos(LE);
266 
267         Vector3D U = new Vector3D((1. + hx2 - hy2)/ h2p1,
268                                   (2.*hx*hy)/h2p1,
269                                   (-2.*hy)/h2p1);
270 
271         Vector3D V = new Vector3D((2.*hx*hy)/ h2p1,
272                                   (1.- hx2+ hy2)/h2p1,
273                                   (2.*hx)/h2p1);
274 
275         Vector3D r = new Vector3D(finalOrbit.getA(),(new Vector3D(x3,U,y3,V)));
276 
277         Assert.assertEquals(finalOrbit.getPVCoordinates().getPosition().getNorm(), r.getNorm(), Utils.epsilonTest * r.getNorm());
278 
279     }
