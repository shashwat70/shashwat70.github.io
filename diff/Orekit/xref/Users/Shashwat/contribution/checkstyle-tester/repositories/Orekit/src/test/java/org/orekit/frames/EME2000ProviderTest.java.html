98      @Test
99      public void testSofaBp00() throws OrekitException {
100 
101         // the reference value has been computed using the March 2012 version of the SOFA library
102         // http://www.iausofa.org/2012_0301_C.html, with the following code
103         //
104         //        double utc1, utc2, tai1, tai2, tt1, tt2, rb[3][3], rp[3][3], rbp[3][3];
105         //        
106         //        // 2004-02-14:00:00:00Z, MJD = 53049, UT1-UTC = -0.4093509
107         //        utc1  = DJM0 + 53049.0;
108         //        utc2  = 0.0;
109         //        iauUtctai(utc1, utc2, &tai1, &tai2);
110         //        iauTaitt(tai1, tai2, &tt1, &tt2);
111         //
112         //        iauBp00(tt1, tt2, rb, rp, rbp);
113         //
114         //        printf("iauBp00(%.20g, %.20g, rb, rp, rbp)\n"
115         //               "  rb  --> %.20g %.20g %.20g\n          %.20g %.20g %.20g\n          %.20g %.20g %.20g\n"
116         //               "  rp  --> %.20g %.20g %.20g\n          %.20g %.20g %.20g\n          %.20g %.20g %.20g\n"
117         //               "  rbp --> %.20g %.20g %.20g\n          %.20g %.20g %.20g\n          %.20g %.20g %.20g\n",
118         //               tt1, tt2,
119         //               rb[0][0],  rb[0][1],  rb[0][2],
120         //               rb[1][0],  rb[1][1],  rb[1][2],
121         //               rb[2][0],  rb[2][1],  rb[2][2],
122         //               rp[0][0],  rp[0][1],  rp[0][2],
123         //               rp[1][0],  rp[1][1],  rp[1][2],
124         //               rp[2][0],  rp[2][1],  rp[2][2],
125         //               rbp[0][0], rbp[0][1], rbp[0][2],
126         //               rbp[1][0], rbp[1][1], rbp[1][2],
127         //               rbp[2][0], rbp[2][1], rbp[2][2]);
128         //
129         // the output of this test reads:
130         //        iauBp00(2453049.5, 0.00074287037037037029902, rb, rp, rbp)
131         //        rb  --> 0.99999999999999422684 -7.0782797441991980175e-08 8.0562171469761337802e-08
132         //                7.0782794778573375197e-08 0.99999999999999689138 3.3060414542221364117e-08
133         //                -8.0562173809869716745e-08 -3.3060408839805516801e-08 0.99999999999999622524
134         //        rp  --> 0.99999949573309343531 -0.00092105778423522924759 -0.00040023257863225548568
135         //                0.00092105778625203805956 0.99999957582617116092 -1.7927962069881782439e-07
136         //                0.00040023257399096032498 -1.8935780260465051583e-07 0.99999991990692216337
137         //        rbp --> 0.99999949570013624278 -0.00092112855376512230675 -0.00040015204695196122638
138         //                0.00092112856903123034591 0.99999957576097886491 -1.4614501776464880046e-07
139         //                0.00040015201181019732432 -2.2244653837776004327e-07 0.99999991993915571253
140 
141         AbsoluteDate date = new AbsoluteDate(2004, 2, 14, TimeScalesFactory.getUTC());
142         Frame eme2000 = FramesFactory.getFrame(Predefined.EME2000);
143         Frame gcrf    = FramesFactory.getFrame(Predefined.GCRF);
144         checkRotation(new double[][] {
145             {  0.99999999999999422684,    -7.0782797441991980175e-08, 8.0562171469761337802e-08 },
146             {  7.0782794778573375197e-08,  0.99999999999999689138,    3.3060414542221364117e-08 },
147             { -8.0562173809869716745e-08, -3.3060408839805516801e-08, 0.99999999999999622524    }
148 
149         }, gcrf.getTransformTo(eme2000, date), 2.5e-16);
150 
151     }

158     private void checkPV(PVCoordinates reference,
159                          PVCoordinates result, double positionThreshold,
160                          double velocityThreshold) {
161 
162         Vector3D dP = result.getPosition().subtract(reference.getPosition());
163         Vector3D dV = result.getVelocity().subtract(reference.getVelocity());
164         Assert.assertEquals(0, dP.getNorm(), positionThreshold);
165         Assert.assertEquals(0, dV.getNorm(), velocityThreshold);
166     }
167 
168     private void checkRotation(double[][] reference, Transform t, double epsilon) {
169         double[][] mat = t.getRotation().getMatrix();
170         for (int i = 0; i < 3; ++i) {
171             for (int j = 0; j < 3; ++j) {
172                 Assert.assertEquals(reference[i][j], mat[i][j], epsilon * FastMath.abs(reference[i][j]));
173                 
174             }
175         }
176     }
