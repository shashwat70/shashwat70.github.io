246     @Test
247     public void testDoppler()
248         throws OrekitException {
249 
250         // Surface point at latitude 45, longitude 5
251         final GeodeticPoint point = new GeodeticPoint(FastMath.toRadians(45.), FastMath.toRadians(5.), 0.);
252         final TopocentricFrame topoFrame = new TopocentricFrame(earthSpheric, point, "lon 5 lat 45");
253 
254         // Point at 30 deg longitude
255         // ***************************
256         final CircularOrbit orbit =
257             new CircularOrbit(7178000.0, 0.5e-8, -0.5e-8, FastMath.toRadians(50.), FastMath.toRadians(120.),
258                                    FastMath.toRadians(90.), PositionAngle.MEAN,
259                                    FramesFactory.getEME2000(), date, mu);
260 
261         // Transform satellite position to position/velocity parameters in body frame
262         final Transform eme2000ToItrf = FramesFactory.getEME2000().getTransformTo(earthSpheric.getBodyFrame(), date);
263         final PVCoordinates pvSatItrf = eme2000ToItrf.transformPVCoordinates(orbit.getPVCoordinates());
264 
265         // Compute range rate directly
266         //********************************************
267         final double dop = topoFrame.getRangeRate(pvSatItrf, earthSpheric.getBodyFrame(), date);
268 
269         // Compare to finite difference computation (2 points)
270         //*****************************************************
271         final double dt = 0.1;
272         KeplerianPropagator extrapolator = new KeplerianPropagator(orbit);
273 
274         // Extrapolate satellite position a short while after reference date
275         AbsoluteDate dateP = date.shiftedBy(dt);
276         Transform j2000ToItrfP = FramesFactory.getEME2000().getTransformTo(earthSpheric.getBodyFrame(), dateP);
277         SpacecraftState orbitP = extrapolator.propagate(dateP);
278         Vector3D satPointGeoP = j2000ToItrfP.transformPVCoordinates(orbitP.getPVCoordinates()).getPosition();
279 
280         // Retropolate satellite position a short while before reference date
281         AbsoluteDate dateM = date.shiftedBy(-dt);
282         Transform j2000ToItrfM = FramesFactory.getEME2000().getTransformTo(earthSpheric.getBodyFrame(), dateM);
283         SpacecraftState orbitM = extrapolator.propagate(dateM);
284         Vector3D satPointGeoM = j2000ToItrfM.transformPVCoordinates(orbitM.getPVCoordinates()).getPosition();
285 
286         // Compute ranges at both instants
287         double rangeP = topoFrame.getRange(satPointGeoP, earthSpheric.getBodyFrame(), dateP);
288         double rangeM = topoFrame.getRange(satPointGeoM, earthSpheric.getBodyFrame(), dateM);
289         final double dopRef2 = (rangeP - rangeM) / (2. * dt);
290         Assert.assertEquals(dopRef2, dop, 1.e-3);
291 
292     }

398     @Test
399     public void testPointAtDistance() throws OrekitException {
400 
401         RandomGenerator random = new Well1024a(0xa1e6bd5cd0578779l);
402         final OneAxisEllipsoid earth =
403             new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS, Constants.WGS84_EARTH_FLATTENING,
404                                  itrf);
405         final AbsoluteDate date = AbsoluteDate.J2000_EPOCH;
406 
407         for (int i = 0; i < 20; ++i) {
408             // we don't need uniform point on the sphere, just a few different test configurations
409             double latitude  = FastMath.PI * (0.5 - random.nextDouble());
410             double longitude = 2 * FastMath.PI * random.nextDouble();
411             TopocentricFrame topo = new TopocentricFrame(earth,
412                                                          new GeodeticPoint(latitude, longitude, 0.0),
413                                                          "topo");
414             Transform transform = earth.getBodyFrame().getTransformTo(topo, date);
415             for (int j = 0; j < 20; ++j) {
416                 double elevation      = FastMath.PI * (0.5 - random.nextDouble());
417                 double azimuth        = 2 * FastMath.PI * random.nextDouble();
418                 double range          = 500000.0 * (1.0 + random.nextDouble());
419                 Vector3D absolutePoint = earth.transform(topo.pointAtDistance(azimuth, elevation, range));
420                 Vector3D relativePoint = transform.transformPosition(absolutePoint);
421                 double rebuiltElevation = topo.getElevation(relativePoint, topo, AbsoluteDate.J2000_EPOCH);
422                 double rebuiltAzimuth   = topo.getAzimuth(relativePoint, topo, AbsoluteDate.J2000_EPOCH);
423                 double rebuiltRange     = topo.getRange(relativePoint, topo, AbsoluteDate.J2000_EPOCH);
424                 Assert.assertEquals(elevation, rebuiltElevation, 1.0e-12);
425                 Assert.assertEquals(azimuth, MathUtils.normalizeAngle(rebuiltAzimuth, azimuth), 1.0e-12);
426                 Assert.assertEquals(range, rebuiltRange, 1.0e-12 * range);
427             }
428         }
429     }
