148 	@Test
149 	@TestForIssue( jiraKey = "HHH-12875")
150 	public void testInitializeFromUniqueAssociationTable() {
151 		doInHibernate(
152 				this::sessionFactory, session -> {
153 
154 					Material material = session.get( Material.class, 1 );
155 					assertEquals( "plastic", material.getName() );
156 
157 					// Material#ratings is mapped with lazy="true"
158 					assertFalse( Hibernate.isInitialized( material.getRatings() ) );
159 					assertEquals( 1, material.getRatings().size() );
160 					assertTrue( Hibernate.isInitialized( material.getRatings() ) );
161 
162 					final Rating rating = material.getRatings().iterator().next();
163 					assertEquals( "high", rating.getName() );
164 
165 					Building building = session.get( Building.class, 1 );
166 					assertEquals( "house", building.getName() );
167 
168 					// Building#ratings is mapped with lazy="true"
169 					assertFalse( Hibernate.isInitialized( building.getMediumOrHighRatings() ) );
170 					checkMediumOrHighRatings( building.getMediumOrHighRatings() );
171 				}
172 		);
173 	}

175 	@Test
176 	@TestForIssue( jiraKey = "HHH-12875")
177 	public void testInitializeFromNonUniqueAssociationTable() {
178 		doInHibernate(
179 				this::sessionFactory, session -> {
180 
181 					Material material = session.get( Material.class, 1 );
182 					assertEquals( "plastic", material.getName() );
183 
184 					// Material#mediumOrHighRatingsFromCombined is mapped with lazy="true"
185 					assertFalse( Hibernate.isInitialized( material.getMediumOrHighRatingsFromCombined() ) );
186 					checkMediumOrHighRatings( material.getMediumOrHighRatingsFromCombined() );
187 					Rating highRating = null;
188 					for ( Rating rating : material.getMediumOrHighRatingsFromCombined() ) {
189 						if ( "high".equals( rating.getName() ) ) {
190 							highRating = rating;
191 						}
192 					}
193 					assertNotNull( highRating );
194 
195 					// Material#sizesFromCombined is mapped with lazy="true"
196 					assertFalse( Hibernate.isInitialized( material.getSizesFromCombined() ) );
197 					assertEquals( 1, material.getSizesFromCombined().size() );
198 					assertTrue( Hibernate.isInitialized( material.getSizesFromCombined() ) );
199 
200 					final Size size = material.getSizesFromCombined().iterator().next();
201 					assertEquals( "medium", size.getName() );
202 
203 					Building building = session.get( Building.class, 1 );
204 
205 					// building.ratingsFromCombined is mapped with lazy="true"
206 					assertFalse( Hibernate.isInitialized( building.getRatingsFromCombined() ) );
207 					assertEquals( 1, building.getRatingsFromCombined().size() );
208 					assertTrue( Hibernate.isInitialized( building.getRatingsFromCombined() ) );
209 					assertSame( highRating, building.getRatingsFromCombined().iterator().next() );
210 
211 					// Building#sizesFromCombined is mapped with lazy="true"
212 					assertFalse( Hibernate.isInitialized( building.getSizesFromCombined() ) );
213 					assertEquals( 1, building.getSizesFromCombined().size() );
214 					assertTrue( Hibernate.isInitialized( building.getSizesFromCombined() ) );
215 					assertEquals( "small", building.getSizesFromCombined().iterator().next().getName() );
216 				}
217 		);
218 	}

220 	private void checkMediumOrHighRatings(List<Rating> mediumOrHighRatings) {
221 		assertEquals( 2, mediumOrHighRatings.size() );
222 
223 		final Iterator<Rating> iterator = mediumOrHighRatings.iterator();
224 		final Rating firstRating = iterator.next();
225 		final Rating secondRating = iterator.next();
226 		if ( "high".equals( firstRating.getName() ) ) {
227 			assertEquals( "medium", secondRating.getName() );
228 		}
229 		else if ( "medium".equals( firstRating.getName() ) ) {
230 			assertEquals( "high", secondRating.getName() );
231 		}
232 		else {
233 			fail( "unexpected rating" );
234 		}
235 	}
