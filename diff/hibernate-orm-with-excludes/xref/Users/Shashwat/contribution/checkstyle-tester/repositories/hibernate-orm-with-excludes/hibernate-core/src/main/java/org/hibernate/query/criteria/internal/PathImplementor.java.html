7   package org.hibernate.query.criteria.internal;
8   
9   import javax.persistence.criteria.Path;
10  import javax.persistence.metamodel.Attribute;
11  
12  /**
13   * Implementation contract for the JPA {@link Path} interface.
14   *
15   * @author Steve Ebersole
16   */
17  public interface PathImplementor<X> extends ExpressionImplementor<X>, Path<X>, PathSource<X>, Renderable {
18  	/**
19  	 * Retrieve reference to the attribute this path represents.
20  	 *
21  	 * @return The metamodel attribute.
22  	 */
23  	public Attribute<?, ?> getAttribute();
24  
25  	/**
26  	 * Defines handling for the JPA 2.1 TREAT down-casting feature.
27  	 *
28  	 * @param treatAsType The type to treat the path as.
29  	 * @param <T> The parameterized type representation of treatAsType.
30  	 *
31  	 * @return The properly typed view of this path.
32  	 */
33  	public <T extends X> PathImplementor<T> treatAs(Class<T> treatAsType);
34  }
