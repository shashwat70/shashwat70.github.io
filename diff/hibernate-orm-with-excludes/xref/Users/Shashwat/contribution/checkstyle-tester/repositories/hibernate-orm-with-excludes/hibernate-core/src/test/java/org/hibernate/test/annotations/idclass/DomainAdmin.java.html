23  package org.hibernate.test.annotations.idclass;
24  import java.io.Serializable;
25  import javax.persistence.Column;
26  import javax.persistence.Entity;
27  import javax.persistence.Id;
28  import javax.persistence.IdClass;
29  import javax.persistence.NamedNativeQuery;
30  import javax.persistence.Table;
31  
32  
33  /**
34   * A DomainAdmin.
35   *
36   * @author <a href="mailto:stale.pedersen@jboss.org">Stale W. Pedersen</a>
37   */
38  @Entity
39  @Table(name = "domainadmin")
40  @IdClass(DomainAdminId.class)
41  @NamedNativeQuery(name = "DomainAdmin.testQuery",
42  		query = "select * from domainadmin da where da.domain_name = 'org'",
43  		resultClass = org.hibernate.test.annotations.idclass.DomainAdmin.class)
44  public class DomainAdmin implements Serializable {
45  
46  	@Id
47  	@Column(name = "domain_name")
48  	private String domainName;
49  
50  	@Id
51  	@Column(name = "admin_user")
52  	private String adminUser;
53  
54  	@Column(name = "nick_name")
55  	private String nickName;
56  
57  	public DomainAdmin() {
58  	}
59  
60  	public String getDomainName() {
61  		return domainName;
62  	}
63  
64  	public void setDomainName(String domainName) {
65  		this.domainName = domainName;
66  	}
67  
68  	public String getAdminUser() {
69  		return adminUser;
70  	}
71  
72  	public void setAdminUser(String adminUser) {
73  		this.adminUser = adminUser;
74  	}
75  
76  	public String getNickName() {
77  		return nickName;
78  	}
79  
80  	public void setNickName(String nickName) {
81  		this.nickName = nickName;
82  	}
83  }
