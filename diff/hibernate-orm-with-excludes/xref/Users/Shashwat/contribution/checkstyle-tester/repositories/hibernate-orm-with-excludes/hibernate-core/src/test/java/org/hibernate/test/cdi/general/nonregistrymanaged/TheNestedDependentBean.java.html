7   package org.hibernate.test.cdi.general.nonregistrymanaged;
8   
9   import javax.annotation.PostConstruct;
10  import javax.annotation.PreDestroy;
11  import javax.enterprise.context.Dependent;
12  
13  /**
14   * A dependent bean required by other beans, but never requested directly
15   * to the {@link org.hibernate.resource.beans.spi.ManagedBeanRegistry}.
16   *
17   * @author Yoann Rodiere
18   */
19  @Dependent
20  public class TheNestedDependentBean {
21  
22  	public TheNestedDependentBean() {
23  		Monitor.theNestedDependentBean().instantiated();
24  	}
25  
26  	public void ensureInitialized() {
27  		// Nothing to do: if this method is called, all surrounding proxies have been initialized
28  	}
29  
30  	@PostConstruct
31  	public void postConstruct() {
32  		Monitor.theNestedDependentBean().postConstructCalled();
33  	}
34  
35  	@PreDestroy
36  	public void preDestroy() {
37  		Monitor.theNestedDependentBean().preDestroyCalled();
38  	}
39  }
