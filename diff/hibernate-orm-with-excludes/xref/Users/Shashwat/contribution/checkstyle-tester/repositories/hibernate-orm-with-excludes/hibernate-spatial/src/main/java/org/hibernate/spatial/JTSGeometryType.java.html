7   package org.hibernate.spatial;
8   
9   import org.hibernate.type.AbstractSingleColumnStandardBasicType;
10  import org.hibernate.type.descriptor.sql.SqlTypeDescriptor;
11  
12  import org.locationtech.jts.geom.Geometry;
13  
14  /**
15   * A {@code Type} that maps between the database geometry type and JTS {@code Geometry}.
16   *
17   * @author Karel Maesen
18   */
19  public class JTSGeometryType extends AbstractSingleColumnStandardBasicType<Geometry> implements Spatial {
20  
21  	/**
22  	 * Constructs an instance with the specified {@code SqlTypeDescriptor}
23  	 *
24  	 * @param sqlTypeDescriptor The descriptor for the type used by the database for geometries.
25  	 */
26  	public JTSGeometryType(SqlTypeDescriptor sqlTypeDescriptor) {
27  		super( sqlTypeDescriptor, JTSGeometryJavaTypeDescriptor.INSTANCE );
28  	}
29  
30  	@Override
31  	public String[] getRegistrationKeys() {
32  		return new String[] {
33  				org.locationtech.jts.geom.Geometry.class.getCanonicalName(),
34  				org.locationtech.jts.geom.Point.class.getCanonicalName(),
35  				org.locationtech.jts.geom.Polygon.class.getCanonicalName(),
36  				org.locationtech.jts.geom.MultiPolygon.class.getCanonicalName(),
37  				org.locationtech.jts.geom.LineString.class.getCanonicalName(),
38  				org.locationtech.jts.geom.MultiLineString.class.getCanonicalName(),
39  				org.locationtech.jts.geom.MultiPoint.class.getCanonicalName(),
40  				org.locationtech.jts.geom.GeometryCollection.class.getCanonicalName(),
41  				"jts_geometry"
42  		};
43  	}
44  
45  
46  	@Override
47  	public String getName() {
48  		return "jts_geometry";
49  	}
50  
51  }
