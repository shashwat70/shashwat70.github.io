18  import org.hibernate.boot.registry.StandardServiceRegistryBuilder;

24  import org.hibernate.testing.DialectChecks;

30  /**
31   * @author Steve Ebersole
32   */
33  @RequiresDialectFeature( value = {DialectChecks.SupportCatalogCreation.class})
34  public class ForeignKeyMigrationTest extends BaseUnitTestCase {
35  	@Test
36  	@TestForIssue( jiraKey = "HHH-9716" )
37  //	@FailureExpected( jiraKey = "HHH-9716" )
38  	public void testMigrationOfForeignKeys() {
39  		StandardServiceRegistry ssr = new StandardServiceRegistryBuilder().build();
40  		try {
41  			final MetadataImplementor metadata = (MetadataImplementor) new MetadataSources( ssr )
42  					.addAnnotatedClass( Box.class )
43  					.addAnnotatedClass( Thing.class )
44  					.buildMetadata();
45  			metadata.validate();
46  
47  			// first create the schema...
48  			new SchemaExport().create( EnumSet.of( TargetType.DATABASE ), metadata );
49  
50  			try {
51  				// try to update the just created schema
52  				new SchemaUpdate().execute( EnumSet.of( TargetType.DATABASE ), metadata );
53  			}
54  			finally {
55  				// clean up
56  				new SchemaExport().drop( EnumSet.of( TargetType.DATABASE ), metadata );
57  			}
58  		}
59  		finally {
60  			StandardServiceRegistryBuilder.destroy( ssr );
61  		}
62  	}
63  
64  	@Entity(name = "Box")
65  	@Table(name = "Box", schema = "PUBLIC", catalog = "DB1")
66  	public static class Box {
67  		@Id
68  		public Integer id;
69  		@ManyToOne
70  		@JoinColumn
71  		public Thing thing1;
72  	}
73  
74  	@Entity(name = "Thing")
75  	@Table(name = "Thing", schema = "PUBLIC", catalog = "DB1")
76  	public static class Thing {
77  		@Id
78  		public Integer id;
79  	}
80  }
