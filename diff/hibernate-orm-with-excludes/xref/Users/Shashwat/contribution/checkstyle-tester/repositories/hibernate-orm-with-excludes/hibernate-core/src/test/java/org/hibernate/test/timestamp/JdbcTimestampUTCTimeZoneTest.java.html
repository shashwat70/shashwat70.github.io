7   package org.hibernate.test.timestamp;
8   
9   import java.util.Map;
10  import java.util.TimeZone;
11  
12  import org.hibernate.cfg.AvailableSettings;
13  import org.hibernate.dialect.PostgreSQL82Dialect;
14  
15  import org.hibernate.testing.RequiresDialect;
16  import org.hibernate.test.util.jdbc.TimeZoneConnectionProvider;
17  
18  /**
19   * @author Vlad Mihalcea
20   */
21  @RequiresDialect(value = PostgreSQL82Dialect.class)
22  public class JdbcTimestampUTCTimeZoneTest
23  		extends JdbcTimestampWithoutUTCTimeZoneTest {
24  
25  	private TimeZoneConnectionProvider connectionProvider = new TimeZoneConnectionProvider(
26  			"America/Los_Angeles" );
27  
28  	private static final TimeZone TIME_ZONE = TimeZone.getTimeZone( "UTC" );
29  
30  	@Override
31  	protected void addSettings(Map settings) {
32  		settings.put(
33  				AvailableSettings.CONNECTION_PROVIDER,
34  				connectionProvider
35  		);
36  		settings.put(
37  				AvailableSettings.JDBC_TIME_ZONE,
38  				TIME_ZONE
39  		);
40  	}
41  
42  	@Override
43  	protected void releaseResources() {
44  		super.releaseResources();
45  		connectionProvider.stop();
46  	}
47  
48  	protected String expectedTimestampValue() {
49  		return "2000-01-01 00:00:00.000000";
50  	}
51  }
