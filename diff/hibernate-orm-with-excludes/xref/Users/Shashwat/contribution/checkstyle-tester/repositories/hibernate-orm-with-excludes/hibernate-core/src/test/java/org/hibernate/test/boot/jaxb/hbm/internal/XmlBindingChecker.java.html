7   package org.hibernate.test.boot.jaxb.hbm.internal;
8   
9   import java.io.ByteArrayInputStream;
10  import java.io.ByteArrayOutputStream;
11  
12  import javax.xml.bind.JAXBContext;
13  import javax.xml.bind.Marshaller;
14  
15  import org.hibernate.boot.jaxb.hbm.spi.JaxbHbmHibernateMapping;
16  import org.hibernate.boot.registry.StandardServiceRegistryBuilder;
17  import org.hibernate.boot.spi.XmlMappingBinderAccess;
18  import org.hibernate.service.ServiceRegistry;
19  
20  /**
21   * @author Jean-Fran√ßois Boeuf
22   */
23  public class XmlBindingChecker {
24  
25  	public static void checkValidGeneration(JaxbHbmHibernateMapping hbmMapping)
26  			throws Exception {
27  		JAXBContext jaxbContext = JAXBContext
28  				.newInstance( JaxbHbmHibernateMapping.class );
29  
30  		Marshaller jaxbMarshaller = jaxbContext.createMarshaller();
31  		jaxbMarshaller.setProperty( Marshaller.JAXB_FORMATTED_OUTPUT, true );
32  		ByteArrayOutputStream bos = new ByteArrayOutputStream();
33  		jaxbMarshaller.marshal( hbmMapping, bos );
34  		ByteArrayInputStream is = new ByteArrayInputStream( bos.toByteArray() );
35  		ServiceRegistry sr = new StandardServiceRegistryBuilder().build();
36  		new XmlMappingBinderAccess( sr ).bind( is );
37  	}
38  }
