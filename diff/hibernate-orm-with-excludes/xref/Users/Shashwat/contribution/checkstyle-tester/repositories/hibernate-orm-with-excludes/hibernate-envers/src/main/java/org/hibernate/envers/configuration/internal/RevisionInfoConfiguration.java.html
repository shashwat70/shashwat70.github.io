50  public class RevisionInfoConfiguration {
51  	private String revisionInfoEntityName;
52  	private PropertyData revisionInfoIdData;

61  	public RevisionInfoConfiguration(GlobalConfiguration globalCfg) {
62  		this.globalCfg = globalCfg;
63  		if ( globalCfg.isUseRevisionEntityWithNativeId() ) {
64  			revisionInfoEntityName = "org.hibernate.envers.DefaultRevisionEntity";
65  		}
66  		else {
67  			revisionInfoEntityName = "org.hibernate.envers.enhanced.SequenceIdRevisionEntity";
68  		}
69  		revisionInfoIdData = new PropertyData( "id", "id", "field", null );
70  		revisionInfoTimestampData = new PropertyData( "timestamp", "timestamp", "field", null );
71  		modifiedEntityNamesData = new PropertyData( "modifiedEntityNames", "modifiedEntityNames", "field", null );
72  		revisionInfoTimestampType = new LongType();
73  
74  		revisionPropType = "integer";
75  	}

77  	private Document generateDefaultRevisionInfoXmlMapping() {
78  		final Document document = globalCfg.getEnversService().getXmlHelper().getDocumentFactory().createDocument();
79  
80  		final Element classMapping = MetadataTools.createEntity(
81  				document,
82  				new AuditTableData( null, null, globalCfg.getDefaultSchemaName(), globalCfg.getDefaultCatalogName() ),
83  				null,
84  				null
85  		);
86  
87  		classMapping.addAttribute( "name", revisionInfoEntityName );
88  		classMapping.addAttribute( "table", "REVINFO" );
89  
90  		final Element idProperty = MetadataTools.addNativelyGeneratedId(
91  				classMapping,
92  				revisionInfoIdData.getName(),
93  				revisionPropType,
94  				globalCfg.isUseRevisionEntityWithNativeId()
95  		);
96  		MetadataTools.addColumn( idProperty, "REV", null, null, null, null, null, null, false );
97  
98  		final Element timestampProperty = MetadataTools.addProperty(
99  				classMapping,
100 				revisionInfoTimestampData.getName(),
101 				revisionInfoTimestampType.getName(),
102 				true,
103 				false
104 		);
105 		MetadataTools.addColumn( timestampProperty, "REVTSTMP", null, null, null, null, null, null, false );
106 
107 		if ( globalCfg.isTrackEntitiesChangedInRevision() ) {
108 			generateEntityNamesTrackingTableMapping(
109 					classMapping,
110 					"modifiedEntityNames",
111 					globalCfg.getDefaultSchemaName(),
112 					globalCfg.getDefaultCatalogName(),
113 					"REVCHANGES",
114 					"REV",
115 					"ENTITYNAME",
116 					"string"
117 			);
118 		}
119 
120 		return document;
121 	}

123 	/**
124 	 * Generates mapping that represents a set of primitive types.<br />
125 	 * <code>
126 	 * &lt;set name="propertyName" table="joinTableName" schema="joinTableSchema" catalog="joinTableCatalog"
127 	 * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;cascade="persist, delete" lazy="false" fetch="join"&gt;<br />
128 	 * &nbsp;&nbsp;&nbsp;&lt;key column="joinTablePrimaryKeyColumnName" /&gt;<br />
129 	 * &nbsp;&nbsp;&nbsp;&lt;element type="joinTableValueColumnType"&gt;<br />
130 	 * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;column name="joinTableValueColumnName" /&gt;<br />
131 	 * &nbsp;&nbsp;&nbsp;&lt;/element&gt;<br />
132 	 * &lt;/set&gt;
133 	 * </code>
134 	 */
135 	private void generateEntityNamesTrackingTableMapping(
136 			Element classMapping,
137 			String propertyName,
138 			String joinTableSchema,
139 			String joinTableCatalog,
140 			String joinTableName,
141 			String joinTablePrimaryKeyColumnName,
142 			String joinTableValueColumnName,
143 			String joinTableValueColumnType) {
144 		final Element set = classMapping.addElement( "set" );
145 		set.addAttribute( "name", propertyName );
146 		set.addAttribute( "table", joinTableName );
147 		set.addAttribute( "schema", joinTableSchema );
148 		set.addAttribute( "catalog", joinTableCatalog );
149 		set.addAttribute( "cascade", "persist, delete" );
150 		set.addAttribute( "fetch", "join" );
151 		set.addAttribute( "lazy", "false" );
152 		final Element key = set.addElement( "key" );
153 		key.addAttribute( "column", joinTablePrimaryKeyColumnName );
154 		final Element element = set.addElement( "element" );
155 		element.addAttribute( "type", joinTableValueColumnType );
156 		final Element column = element.addElement( "column" );
157 		column.addAttribute( "name", joinTableValueColumnName );
158 	}

160 	private Element generateRevisionInfoRelationMapping() {
161 		final Document document = globalCfg.getEnversService().getXmlHelper().getDocumentFactory().createDocument();
162 		final Element revRelMapping = document.addElement( "key-many-to-one" );
163 		revRelMapping.addAttribute( "type", revisionPropType );
164 		revRelMapping.addAttribute( "class", revisionInfoEntityName );
165 
166 		if ( revisionPropSqlType != null ) {
167 			// Putting a fake name to make Hibernate happy. It will be replaced later anyway.
168 			MetadataTools.addColumn( revRelMapping, "*", null, null, null, revisionPropSqlType, null, null, false );
169 		}
170 
171 		return revRelMapping;
172 	}

174 	private void searchForRevisionInfoCfgInProperties(
175 			XClass clazz,
176 			ReflectionManager reflectionManager,
177 			MutableBoolean revisionNumberFound,
178 			MutableBoolean revisionTimestampFound,
179 			MutableBoolean modifiedEntityNamesFound,
180 			String accessType) {
181 		for ( XProperty property : clazz.getDeclaredProperties( accessType ) ) {
182 			final RevisionNumber revisionNumber = property.getAnnotation( RevisionNumber.class );
183 			final RevisionTimestamp revisionTimestamp = property.getAnnotation( RevisionTimestamp.class );
184 			final ModifiedEntityNames modifiedEntityNames = property.getAnnotation( ModifiedEntityNames.class );
185 
186 			if ( revisionNumber != null ) {
187 				if ( revisionNumberFound.isSet() ) {
188 					throw new MappingException( "Only one property may be annotated with @RevisionNumber!" );
189 				}
190 
191 				final XClass revisionNumberClass = property.getType();
192 				if ( reflectionManager.equals( revisionNumberClass, Integer.class ) ||
193 						reflectionManager.equals( revisionNumberClass, Integer.TYPE ) ) {
194 					revisionInfoIdData = new PropertyData( property.getName(), property.getName(), accessType, null );
195 					revisionNumberFound.set();
196 				}
197 				else if ( reflectionManager.equals( revisionNumberClass, Long.class ) ||
198 						reflectionManager.equals( revisionNumberClass, Long.TYPE ) ) {
199 					revisionInfoIdData = new PropertyData( property.getName(), property.getName(), accessType, null );
200 					revisionNumberFound.set();
201 
202 					// The default is integer
203 					revisionPropType = "long";
204 				}
205 				else {
206 					throw new MappingException(
207 							"The field annotated with @RevisionNumber must be of type " +
208 									"int, Integer, long or Long"
209 					);
210 				}
211 
212 				// Getting the @Column definition of the revision number property, to later use that info to
213 				// generate the same mapping for the relation from an audit table's revision number to the
214 				// revision entity revision number.
215 				final Column revisionPropColumn = property.getAnnotation( Column.class );
216 				if ( revisionPropColumn != null ) {
217 					revisionPropSqlType = revisionPropColumn.columnDefinition();
218 				}
219 			}
220 
221 			if ( revisionTimestamp != null ) {
222 				if ( revisionTimestampFound.isSet() ) {
223 					throw new MappingException( "Only one property may be annotated with @RevisionTimestamp!" );
224 				}
225 
226 				final XClass revisionTimestampClass = property.getType();
227 				if ( reflectionManager.equals( revisionTimestampClass, Long.class ) ||
228 						reflectionManager.equals( revisionTimestampClass, Long.TYPE ) ||
229 						reflectionManager.equals( revisionTimestampClass, Date.class ) ||
230 						reflectionManager.equals( revisionTimestampClass, java.sql.Date.class ) ) {
231 					revisionInfoTimestampData = new PropertyData(
232 							property.getName(),
233 							property.getName(),
234 							accessType,
235 							null
236 					);
237 					revisionTimestampFound.set();
238 				}
239 				else {
240 					throw new MappingException(
241 							"The field annotated with @RevisionTimestamp must be of type " +
242 									"long, Long, java.util.Date or java.sql.Date"
243 					);
244 				}
245 			}
246 
247 			if ( modifiedEntityNames != null ) {
248 				if ( modifiedEntityNamesFound.isSet() ) {
249 					throw new MappingException( "Only one property may be annotated with @ModifiedEntityNames!" );
250 				}
251 				final XClass modifiedEntityNamesClass = property.getType();
252 				if ( reflectionManager.equals( modifiedEntityNamesClass, Set.class ) &&
253 						reflectionManager.equals( property.getElementClass(), String.class ) ) {
254 					modifiedEntityNamesData = new PropertyData(
255 							property.getName(),
256 							property.getName(),
257 							accessType,
258 							null
259 					);
260 					modifiedEntityNamesFound.set();
261 				}
262 				else {
263 					throw new MappingException(
264 							"The field annotated with @ModifiedEntityNames must be of Set<String> type."
265 					);
266 				}
267 			}
268 		}
269 	}

271 	private void searchForRevisionInfoCfg(
272 			XClass clazz, ReflectionManager reflectionManager,
273 			MutableBoolean revisionNumberFound, MutableBoolean revisionTimestampFound,
274 			MutableBoolean modifiedEntityNamesFound) {
275 		final XClass superclazz = clazz.getSuperclass();
276 		if ( !"java.lang.Object".equals( superclazz.getName() ) ) {
277 			searchForRevisionInfoCfg(
278 					superclazz,
279 					reflectionManager,
280 					revisionNumberFound,
281 					revisionTimestampFound,
282 					modifiedEntityNamesFound
283 			);
284 		}
285 
286 		searchForRevisionInfoCfgInProperties(
287 				clazz, reflectionManager, revisionNumberFound, revisionTimestampFound,
288 				modifiedEntityNamesFound, "field"
289 		);
290 		searchForRevisionInfoCfgInProperties(
291 				clazz, reflectionManager, revisionNumberFound, revisionTimestampFound,
292 				modifiedEntityNamesFound, "property"
293 		);
294 	}

296 	public RevisionInfoConfigurationResult configure(MetadataImplementor metadata, ReflectionManager reflectionManager) {
297 		boolean revisionEntityFound = false;
298 		RevisionInfoGenerator revisionInfoGenerator = null;
299 		Class<?> revisionInfoClass = null;
300 
301 		for ( PersistentClass persistentClass : metadata.getEntityBindings() ) {
302 			// Ensure we're in POJO, not dynamic model, mapping.
303 			if (persistentClass.getClassName() != null) {
304 				XClass clazz;
305 				try {
306 					clazz = reflectionManager.classForName( persistentClass.getClassName() );
307 				}
308 				catch (ClassLoadingException e) {
309 					throw new MappingException( e );
310 				}
311 
312 				final RevisionEntity revisionEntity = clazz.getAnnotation( RevisionEntity.class );
313 				if ( revisionEntity != null ) {
314 					if (revisionEntityFound) {
315 						throw new MappingException("Only one entity may be annotated with @RevisionEntity!");
316 					}
317 
318 					// Checking if custom revision entity isn't audited
319 					if ( clazz.getAnnotation( Audited.class ) != null ) {
320 						throw new MappingException("An entity annotated with @RevisionEntity cannot be audited!");
321 					}
322 
323 					revisionEntityFound = true;
324 
325 					final MutableBoolean revisionNumberFound = new MutableBoolean();
326 					final MutableBoolean revisionTimestampFound = new MutableBoolean();
327 					final MutableBoolean modifiedEntityNamesFound = new MutableBoolean();
328 
329 					searchForRevisionInfoCfg(
330 							clazz,
331 							reflectionManager,
332 							revisionNumberFound,
333 							revisionTimestampFound,
334 							modifiedEntityNamesFound
335 					);
336 
337 					if (!revisionNumberFound.isSet()) {
338 						throw new MappingException(
339 								"An entity annotated with @RevisionEntity must have a field annotated " +
340 										"with @RevisionNumber!"
341 						);
342 					}
343 
344 					if (!revisionTimestampFound.isSet()) {
345 						throw new MappingException(
346 								"An entity annotated with @RevisionEntity must have a field annotated " +
347 										"with @RevisionTimestamp!"
348 						);
349 					}
350 
351 					revisionInfoEntityName = persistentClass.getEntityName();
352 					revisionInfoClass = persistentClass.getMappedClass();
353 					final Class<? extends RevisionListener> revisionListenerClass = getRevisionListenerClass( revisionEntity.value() );
354 					revisionInfoTimestampType = persistentClass.getProperty( revisionInfoTimestampData.getName() ).getType();
355 					if ( globalCfg.isTrackEntitiesChangedInRevision()
356 							|| ( globalCfg.isUseRevisionEntityWithNativeId() && DefaultTrackingModifiedEntitiesRevisionEntity.class
357 							.isAssignableFrom( revisionInfoClass ) )
358 							|| ( !globalCfg.isUseRevisionEntityWithNativeId() && SequenceIdTrackingModifiedEntitiesRevisionEntity.class
359 							.isAssignableFrom( revisionInfoClass ) )
360 							|| modifiedEntityNamesFound.isSet() ) {
361 						// If tracking modified entities parameter is enabled, custom revision info entity is a subtype
362 						// of DefaultTrackingModifiedEntitiesRevisionEntity class, or @ModifiedEntityNames annotation is used.
363 						revisionInfoGenerator = new DefaultTrackingModifiedEntitiesRevisionInfoGenerator(
364 								revisionInfoEntityName,
365 								revisionInfoClass,
366 								revisionListenerClass,
367 								revisionInfoTimestampData,
368 								isTimestampAsDate(),
369 								modifiedEntityNamesData,
370 								metadata.getMetadataBuildingOptions().getServiceRegistry()
371 						);
372 						globalCfg.setTrackEntitiesChangedInRevision( true );
373 					}
374 					else {
375 						revisionInfoGenerator = new DefaultRevisionInfoGenerator(
376 								revisionInfoEntityName,
377 								revisionInfoClass,
378 								revisionListenerClass,
379 								revisionInfoTimestampData,
380 								isTimestampAsDate(),
381 								metadata.getMetadataBuildingOptions().getServiceRegistry()
382 						);
383 					}
384 				}
385 			}
386 		}
387 
388 		// In case of a custom revision info generator, the mapping will be null.
389 		Document revisionInfoXmlMapping = null;
390 
391 		final Class<? extends RevisionListener> revisionListenerClass = getRevisionListenerClass( RevisionListener.class );
392 
393 		if ( revisionInfoGenerator == null ) {
394 			if ( globalCfg.isTrackEntitiesChangedInRevision() ) {
395 				revisionInfoClass = globalCfg.isUseRevisionEntityWithNativeId()
396 						? DefaultTrackingModifiedEntitiesRevisionEntity.class
397 						: SequenceIdTrackingModifiedEntitiesRevisionEntity.class;
398 				revisionInfoEntityName = revisionInfoClass.getName();
399 				revisionInfoGenerator = new DefaultTrackingModifiedEntitiesRevisionInfoGenerator(
400 						revisionInfoEntityName,
401 						revisionInfoClass,
402 						revisionListenerClass,
403 						revisionInfoTimestampData,
404 						isTimestampAsDate(),
405 						modifiedEntityNamesData,
406 						metadata.getMetadataBuildingOptions().getServiceRegistry()
407 				);
408 			}
409 			else {
410 				revisionInfoClass = globalCfg.isUseRevisionEntityWithNativeId()
411 						? DefaultRevisionEntity.class
412 						: SequenceIdRevisionEntity.class;
413 				revisionInfoGenerator = new DefaultRevisionInfoGenerator(
414 						revisionInfoEntityName,
415 						revisionInfoClass,
416 						revisionListenerClass,
417 						revisionInfoTimestampData,
418 						isTimestampAsDate(),
419 						metadata.getMetadataBuildingOptions().getServiceRegistry()
420 				);
421 			}
422 			revisionInfoXmlMapping = generateDefaultRevisionInfoXmlMapping();
423 		}
424 
425 		final RevisionInfoNumberReader revisionInfoNumberReader = new RevisionInfoNumberReader(
426 				revisionInfoClass,
427 				revisionInfoIdData,
428 				metadata.getMetadataBuildingOptions().getServiceRegistry()
429 		);
430 
431 		revisionInfoGenerator.setRevisionInfoNumberReader( revisionInfoNumberReader );
432 
433 		return new RevisionInfoConfigurationResult(
434 				revisionInfoGenerator, revisionInfoXmlMapping,
435 				new RevisionInfoQueryCreator(
436 						revisionInfoEntityName, revisionInfoIdData.getName(),
437 						revisionInfoTimestampData.getName(), isTimestampAsDate()
438 				),
439 				generateRevisionInfoRelationMapping(),
440 				revisionInfoNumberReader,
441 				globalCfg.isTrackEntitiesChangedInRevision()
442 						? new ModifiedEntityNamesReader( revisionInfoClass, modifiedEntityNamesData, metadata.getMetadataBuildingOptions().getServiceRegistry() )
443 						: null,
444 				revisionInfoEntityName, revisionInfoClass, revisionInfoTimestampData
445 		);
446 	}
447 
448 	private boolean isTimestampAsDate() {
449 		final String typename = revisionInfoTimestampType.getName();
450 		return "date".equals( typename ) || "time".equals( typename ) || "timestamp".equals( typename );
451 	}

453 	/**
454 	 * @param defaultListener Revision listener that shall be applied if {@code org.hibernate.envers.revision_listener}
455 	 * parameter has not been set.
456 	 *
457 	 * @return Revision listener.
458 	 */
459 	private Class<? extends RevisionListener> getRevisionListenerClass(Class<? extends RevisionListener> defaultListener) {
460 		if ( globalCfg.getRevisionListenerClass() != null ) {
461 			return globalCfg.getRevisionListenerClass();
462 		}
463 		return defaultListener;
464 	}
