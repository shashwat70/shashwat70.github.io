2   package org.hibernate.test.annotations;
3   import java.io.Serializable;
4   import java.util.Collection;
5   import java.util.SortedSet;
6   import javax.persistence.CascadeType;
7   import javax.persistence.Entity;
8   import javax.persistence.FetchType;
9   import javax.persistence.GeneratedValue;
10  import javax.persistence.Id;
11  import javax.persistence.JoinColumn;
12  import javax.persistence.OneToMany;
13  import javax.persistence.OneToOne;
14  
15  import org.hibernate.annotations.Cascade;
16  import org.hibernate.annotations.Sort;
17  import org.hibernate.annotations.SortType;
18  
19  import static org.hibernate.annotations.CascadeType.ALL;
20  
21  
22  /**
23   * Company customer
24   *
25   * @author Emmanuel Bernard
26   */
27  @Entity
28  public class Customer implements Serializable {
29  	Long id;
30  	String name;
31  	SortedSet<Ticket> tickets;
32  	Collection discountTickets;
33  	Passport passport;
34  
35  	public Customer() {
36  	}
37  
38  	@Id
39  	@GeneratedValue
40  	public Long getId() {
41  		return id;
42  	}
43  
44  	public String getName() {
45  		return name;
46  	}
47  
48  	public void setId(Long long1) {
49  		id = long1;
50  	}
51  
52  	public void setName(String string) {
53  		name = string;
54  	}
55  
56  	@OneToMany(cascade = CascadeType.ALL, fetch = FetchType.EAGER)
57  	@JoinColumn(name = "CUST_ID")
58  	@Sort(type = SortType.COMPARATOR, comparator = TicketComparator.class)
59  	public SortedSet<Ticket> getTickets() {
60  		return tickets;
61  	}
62  
63  	public void setTickets(SortedSet<Ticket> tickets) {
64  		this.tickets = tickets;
65  	}
66  
67  	@OneToMany(targetEntity = org.hibernate.test.annotations.Discount.class,
68  			cascade = CascadeType.ALL, mappedBy = "owner")
69  	@Cascade({ALL})
70  	public Collection getDiscountTickets() {
71  		return discountTickets;
72  	}
73  
74  	public void setDiscountTickets(Collection collection) {
75  		discountTickets = collection;
76  	}
77  
78  	@OneToOne(cascade = CascadeType.ALL)
79  	public Passport getPassport() {
80  		return passport;
81  	}
82  
83  	public void setPassport(Passport passport) {
84  		this.passport = passport;
85  	}
86  
87  }
