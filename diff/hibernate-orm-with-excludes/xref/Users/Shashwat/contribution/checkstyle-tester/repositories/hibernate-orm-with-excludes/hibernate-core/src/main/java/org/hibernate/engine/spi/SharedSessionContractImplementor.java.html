46  /**
47   * Defines the internal contract shared between {@link org.hibernate.Session} and
48   * {@link org.hibernate.StatelessSession} as used by other parts of Hibernate (such as
49   * {@link org.hibernate.type.Type}, {@link EntityPersister} and
50   * {@link org.hibernate.persister.collection.CollectionPersister} implementors
51   *
52   * A Session, through this interface and SharedSessionContractImplementor, implements:<ul>
53   *     <li>
54   *         {@link org.hibernate.resource.jdbc.spi.JdbcSessionOwner} to drive the behavior of a "JDBC session".
55   *         Can therefor be used to construct a JdbcCoordinator, which (for now) models a "JDBC session"
56   *     </li>
57   *     <li>
58   *         {@link Options}
59   *         to drive the creation of the {@link TransactionCoordinator} delegate.
60   *         This allows it to be passed along to
61   *         {@link org.hibernate.resource.transaction.spi.TransactionCoordinatorBuilder#buildTransactionCoordinator}
62   *     </li>
63   *     <li>
64   *         {@link org.hibernate.engine.jdbc.LobCreationContext} to act as the context for JDBC LOB instance creation
65   *     </li>
66   *     <li>
67   *         {@link org.hibernate.type.descriptor.WrapperOptions} to fulfill the behavior needed while
68   *         binding/extracting values to/from JDBC as part of the Type contracts
69   *     </li>
70   * </ul>
71   *
72   * @author Gavin King
73   * @author Steve Ebersole
74   */
75  public interface SharedSessionContractImplementor
76  		extends SharedSessionContract, JdbcSessionOwner, Options, LobCreationContext, WrapperOptions, QueryProducerImplementor {

392 	/**
393 	 * Set the flush mode for this session.
394 	 * <p/>
395 	 * The flush mode determines the points at which the session is flushed.
396 	 * <i>Flushing</i> is the process of synchronizing the underlying persistent
397 	 * store with persistable state held in memory.
398 	 * <p/>
399 	 * For a logically "read only" session, it is reasonable to set the session's
400 	 * flush mode to {@link FlushMode#MANUAL} at the start of the session (in
401 	 * order to achieve some extra performance).
402 	 *
403 	 * @param flushMode the new flush mode
404 	 *
405 	 * @deprecated (since 5.2) use {@link #setHibernateFlushMode(FlushMode)} instead
406 	 */

477 	 * Get the currently configured JDBC batch size either at the Session-level or SessionFactory-level.

479 	 * If the Session-level JDBC batch size was not configured, return the SessionFactory-level one.
480 	 *
481 	 * @return Session-level or or SessionFactory-level JDBC batch size.
482 	 *

489 		final Integer sessionJdbcBatchSize = getJdbcBatchSize();
490 
491 		return sessionJdbcBatchSize == null ?
492 			ConfigurationHelper.getInt(
493 					Environment.STATEMENT_BATCH_SIZE,

495 					1
496 			) :
497 			sessionJdbcBatchSize;
498 	}
