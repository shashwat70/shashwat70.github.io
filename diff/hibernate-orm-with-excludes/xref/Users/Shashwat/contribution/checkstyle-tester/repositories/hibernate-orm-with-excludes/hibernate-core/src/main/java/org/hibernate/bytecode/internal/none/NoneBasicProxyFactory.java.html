7   package org.hibernate.bytecode.internal.none;
8   
9   import java.util.Arrays;
10  
11  import org.hibernate.HibernateException;
12  import org.hibernate.bytecode.spi.BasicProxyFactory;
13  
14  /**
15   * This is used to instantiate "components" which are either Abstract or defined by an Interface.
16   */
17  final class NoneBasicProxyFactory implements BasicProxyFactory {
18  
19  	private final Class superClass;
20  	private final Class[] interfaces;
21  
22  	public NoneBasicProxyFactory(Class superClass, Class[] interfaces) {
23  		this.superClass = superClass;
24  		this.interfaces = interfaces;
25  	}
26  
27  	@Override
28  	public Object getProxy() {
29  		throw new HibernateException( "NoneBasicProxyFactory is unable to generate a BasicProxy for type " + superClass + " and interfaces " + Arrays.toString( interfaces ) + ". Enable a different BytecodeProvider." );
30  	}
31  
32  }
