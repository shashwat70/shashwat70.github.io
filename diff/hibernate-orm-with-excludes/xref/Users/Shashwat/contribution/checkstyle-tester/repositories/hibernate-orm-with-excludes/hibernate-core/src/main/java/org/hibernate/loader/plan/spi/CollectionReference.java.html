7   package org.hibernate.loader.plan.spi;
8   
9   import org.hibernate.loader.PropertyPath;
10  import org.hibernate.persister.collection.CollectionPersister;
11  
12  /**
13   * Represents a reference to a persistent collection either as a Return or as a {@link CollectionAttributeFetch}.
14   *
15   * @author Steve Ebersole
16   */
17  public interface CollectionReference {
18  	/**
19  	 * Obtain the UID of the QuerySpace (specifically a {@link CollectionQuerySpace}) that this CollectionReference
20  	 * refers to.
21  	 *
22  	 * @return The UID
23  	 */
24  	public String getQuerySpaceUid();
25  
26  	/**
27  	 * Retrieves the CollectionPersister describing the collection associated with this CollectionReference.
28  	 *
29  	 * @return The CollectionPersister.
30  	 */
31  	public CollectionPersister getCollectionPersister();
32  
33  	/**
34  	 * Retrieve the metadata about the index of this collection *as a FetchSource*.  Will return
35  	 * {@code null} when:<ul>
36  	 *     <li>the collection is not indexed</li>
37  	 *     <li>the index is not a composite, entity, or "any" (cannot act as a FetchSource)</li>
38  	 * </ul>
39  	 * <p/>
40  	 * Works only for map keys, since a List index (int type) cannot act as a FetchSource.
41  	 * <p/>
42  	 *
43  	 * @return The collection index metadata as a FetchSource, or {@code null}.
44  	 */
45  	public CollectionFetchableIndex getIndexGraph();
46  
47  	/**
48  	 * Retrieve the metadata about the elements of this collection *as a FetchSource*.  Will return
49  	 * {@code null} when the element is not a composite, entity, or "any" (cannot act as a FetchSource).
50  	 * Works only for map keys, since a List index cannot be anything other than an int which cannot be a FetchSource.
51  	 * <p/>
52  	 *
53  	 * @return The collection element metadata as a FetchSource, or {@code null}.
54  	 */
55  	public CollectionFetchableElement getElementGraph();
56  
57  	/**
58  	 * Retrieve the PropertyPath to this reference.
59  	 *
60  	 * @return The PropertyPath
61  	 */
62  	public PropertyPath getPropertyPath();
63  
64  	/**
65  	 * Should a collection element join be allowed? Returning <code>true</code>
66  	 * indicates that an element join can safely be added.
67  	 *
68  	 * @return true, if a collection index join is allowed.
69  	 */
70  	public boolean allowElementJoin();
71  
72  	/**
73  	 * Should a collection index join be allowed? Returning <code>true</code>
74  	 * indicates that an index join can safely be added.
75  	 *
76  	 * @return true, if a collection index join is allowed.
77  	 */
78  	public boolean allowIndexJoin();
79  }
