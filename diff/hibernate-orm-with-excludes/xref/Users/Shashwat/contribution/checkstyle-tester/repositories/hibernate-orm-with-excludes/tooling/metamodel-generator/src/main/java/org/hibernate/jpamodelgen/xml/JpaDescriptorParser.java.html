41  /**
42   * Parser for JPA XML descriptors (persistence.xml and referenced mapping files).
43   *
44   * @author Hardy Ferentschik
45   */
46  public class JpaDescriptorParser {
47  	private static final String DEFAULT_ORM_XML_LOCATION = "/META-INF/orm.xml";
48  	private static final String SERIALIZATION_FILE_NAME = "Hibernate-Static-Metamodel-Generator.tmp";
49  
50  	private static final String PERSISTENCE_SCHEMA = "persistence_2_1.xsd";
51  	private static final String ORM_SCHEMA = "orm_2_1.xsd";
52  
53  	private final Context context;
54  	private final List<EntityMappings> entityMappings;
55  	private final XmlParserHelper xmlParserHelper;
56  
57  	public JpaDescriptorParser(Context context) {
58  		this.context = context;
59  		this.entityMappings = new ArrayList<EntityMappings>();
60  		this.xmlParserHelper = new XmlParserHelper( context );
61  	}
62  
63  	public void parseXml() {
64  		Collection<String> mappingFileNames = determineMappingFileNames();
65  
66  		if ( context.doLazyXmlParsing() && mappingFilesUnchanged( mappingFileNames ) ) {
67  			return;
68  		}
69  
70  		loadEntityMappings( mappingFileNames );
71  		determineDefaultAccessTypeAndMetaCompleteness();
72  		determineXmlAccessTypes();
73  		if ( !context.isFullyXmlConfigured() ) {
74  			// need to take annotations into consideration, since they can override xml settings
75  			// we have to at least determine whether any of the xml configured entities is influenced by annotations
76  			determineAnnotationAccessTypes();
77  		}
78  
79  		for ( EntityMappings mappings : entityMappings ) {
80  			String defaultPackageName = mappings.getPackage();
81  			parseEntities( mappings.getEntity(), defaultPackageName );
82  			parseEmbeddable( mappings.getEmbeddable(), defaultPackageName );
83  			parseMappedSuperClass( mappings.getMappedSuperclass(), defaultPackageName );
84  		}
85  	}
86  
87  	private Collection<String> determineMappingFileNames() {
88  		Collection<String> mappingFileNames = new ArrayList<String>();
89  
90  		Persistence persistence = getPersistence();
91  		if ( persistence != null ) {
92  			// get mapping file names from persistence.xml
93  			List<Persistence.PersistenceUnit> persistenceUnits = persistence.getPersistenceUnit();
94  			for ( Persistence.PersistenceUnit unit : persistenceUnits ) {
95  				mappingFileNames.addAll( unit.getMappingFile() );
96  			}
97  		}
98  
99  		// /META-INF/orm.xml is implicit
100 		mappingFileNames.add( DEFAULT_ORM_XML_LOCATION );
101 
102 		// not really part of the official spec, but the processor allows to specify mapping files directly as
103 		// command line options
104 		mappingFileNames.addAll( context.getOrmXmlFiles() );
105 		return mappingFileNames;
106 	}
107 
108 	private Persistence getPersistence() {
109 		Persistence persistence = null;
110 		String persistenceXmlLocation = context.getPersistenceXmlLocation();
111 		InputStream stream = xmlParserHelper.getInputStreamForResource( persistenceXmlLocation );
112 		if ( stream == null ) {
113 			return null;
114 		}
115 
116 		try {
117 			Schema schema = xmlParserHelper.getSchema( PERSISTENCE_SCHEMA );
118 			persistence = xmlParserHelper.getJaxbRoot( stream, Persistence.class, schema );
119 		}
120 		catch (XmlParsingException e) {
121 			context.logMessage(
122 					Diagnostic.Kind.WARNING, "Unable to parse persistence.xml: " + e.getMessage()
123 			);
124 		}
125 
126 		try {
127 			stream.close();
128 		}
129 		catch (IOException e) {
130 			// eat it
131 		}
132 
133 		return persistence;
134 	}
135 
136 	private void loadEntityMappings(Collection<String> mappingFileNames) {
137 		for ( String mappingFile : mappingFileNames ) {
138 			InputStream stream = xmlParserHelper.getInputStreamForResource( mappingFile );
139 			if ( stream == null ) {
140 				continue;
141 			}
142 			EntityMappings mapping = null;
143 			try {
144 				Schema schema = xmlParserHelper.getSchema( ORM_SCHEMA );
145 				mapping = xmlParserHelper.getJaxbRoot( stream, EntityMappings.class, schema );
146 			}
147 			catch (XmlParsingException e) {
148 				context.logMessage(
149 						Diagnostic.Kind.WARNING, "Unable to parse " + mappingFile + ": " + e.getMessage()
150 				);
151 			}
152 			if ( mapping != null ) {
153 				entityMappings.add( mapping );
154 			}
155 
156 			try {
157 				stream.close();
158 			}
159 			catch (IOException e) {
160 				// eat it
161 			}
162 		}
163 	}

165 	private boolean mappingFilesUnchanged(Collection<String> mappingFileNames) {
166 		boolean mappingFilesUnchanged = false;
167 		FileTimeStampChecker fileStampCheck = new FileTimeStampChecker();
168 		for ( String mappingFile : mappingFileNames ) {
169 			try {
170 				URL url = this.getClass().getResource( mappingFile );
171 				if ( url == null ) {
172 					continue;
173 				}
174 				File file = new File( url.toURI() );
175 				context.logMessage( Diagnostic.Kind.OTHER, "Check file  " + mappingFile );
176 				if ( file.exists() ) {
177 					fileStampCheck.add( mappingFile, file.lastModified() );
178 				}
179 			}
180 			catch (URISyntaxException e) {
181 				// in doubt return false
182 				return false;
183 			}
184 		}
185 
186 		FileTimeStampChecker serializedTimeStampCheck = loadTimeStampCache();
187 		if ( serializedTimeStampCheck.equals( fileStampCheck ) ) {
188 			context.logMessage( Diagnostic.Kind.OTHER, "XML parsing will be skipped due to unchanged xml files" );
189 			mappingFilesUnchanged = true;
190 		}
191 		else {
192 			saveTimeStampCache( fileStampCheck );
193 		}
194 
195 		return mappingFilesUnchanged;
196 	}
197 
198 	private void saveTimeStampCache(FileTimeStampChecker fileStampCheck) {
199 		try {
200 			File file = getSerializationTmpFile();
201 			ObjectOutput out = new ObjectOutputStream( new FileOutputStream( file ) );
202 			out.writeObject( fileStampCheck );
203 			out.close();
204 			context.logMessage(
205 					Diagnostic.Kind.OTHER, "Serialized " + fileStampCheck + " into " + file.getAbsolutePath()
206 			);
207 		}
208 		catch (IOException e) {
209 			// ignore - if the serialization failed we just have to keep parsing the xml
210 			context.logMessage( Diagnostic.Kind.OTHER, "Error serializing  " + fileStampCheck );
211 		}
212 	}
213 
214 	private File getSerializationTmpFile() {
215 		File tmpDir = new File( System.getProperty( "java.io.tmpdir" ) );
216 		return new File( tmpDir, SERIALIZATION_FILE_NAME );
217 	}
218 
219 	private FileTimeStampChecker loadTimeStampCache() {
220 		FileTimeStampChecker serializedTimeStampCheck = new FileTimeStampChecker();
221 		File file = null;
222 		try {
223 			file = getSerializationTmpFile();
224 			if ( file.exists() ) {
225 				ObjectInputStream in = new ObjectInputStream( new FileInputStream( file ) );
226 				serializedTimeStampCheck = (FileTimeStampChecker) in.readObject();
227 				in.close();
228 			}
229 		}
230 		catch (Exception e) {
231 			// ignore - if the de-serialization failed we just have to keep parsing the xml
232 			context.logMessage( Diagnostic.Kind.OTHER, "Error de-serializing  " + file );
233 		}
234 		return serializedTimeStampCheck;
235 	}

237 	private void parseEntities(Collection<Entity> entities, String defaultPackageName) {
238 		for ( Entity entity : entities ) {
239 			String fqcn = StringUtil.determineFullyQualifiedClassName( defaultPackageName, entity.getClazz() );
240 
241 			if ( !xmlMappedTypeExists( fqcn ) ) {
242 				context.logMessage(
243 						Diagnostic.Kind.WARNING,
244 						fqcn + " is mapped in xml, but class does not exist. Skipping meta model generation."
245 				);
246 				continue;
247 			}
248 
249 			XmlMetaEntity metaEntity = new XmlMetaEntity(
250 					entity, defaultPackageName, getXmlMappedType( fqcn ), context
251 			);
252 			if ( context.containsMetaEntity( fqcn ) ) {
253 				context.logMessage(
254 						Diagnostic.Kind.WARNING,
255 						fqcn + " was already processed once. Skipping second occurrence."
256 				);
257 			}
258 			context.addMetaEntity( fqcn, metaEntity );
259 		}
260 	}
261 
262 	private void parseEmbeddable(
263 			Collection<org.hibernate.jpamodelgen.xml.jaxb.Embeddable> embeddables,
264 			String defaultPackageName) {
265 		for ( org.hibernate.jpamodelgen.xml.jaxb.Embeddable embeddable : embeddables ) {
266 			String fqcn = StringUtil.determineFullyQualifiedClassName( defaultPackageName, embeddable.getClazz() );
267 			// we have to extract the package name from the fqcn. Maybe the entity was setting a fqcn directly
268 			String pkg = StringUtil.packageNameFromFqcn( fqcn );
269 
270 			if ( !xmlMappedTypeExists( fqcn ) ) {
271 				context.logMessage(
272 						Diagnostic.Kind.WARNING,
273 						fqcn + " is mapped in xml, but class does not exist. Skipping meta model generation."
274 				);
275 				continue;
276 			}
277 
278 			XmlMetaEntity metaEntity = new XmlMetaEntity( embeddable, pkg, getXmlMappedType( fqcn ), context );
279 			if ( context.containsMetaEmbeddable( fqcn ) ) {
280 				context.logMessage(
281 						Diagnostic.Kind.WARNING,
282 						fqcn + " was already processed once. Skipping second occurrence."
283 				);
284 			}
285 			context.addMetaEmbeddable( fqcn, metaEntity );
286 		}
287 	}

289 	private void parseMappedSuperClass(
290 			Collection<org.hibernate.jpamodelgen.xml.jaxb.MappedSuperclass> mappedSuperClasses,
291 			String defaultPackageName) {
292 		for ( org.hibernate.jpamodelgen.xml.jaxb.MappedSuperclass mappedSuperClass : mappedSuperClasses ) {
293 			String fqcn = StringUtil.determineFullyQualifiedClassName(
294 					defaultPackageName, mappedSuperClass.getClazz()
295 			);
296 			// we have to extract the package name from the fqcn. Maybe the entity was setting a fqcn directly
297 			String pkg = StringUtil.packageNameFromFqcn( fqcn );
298 
299 			if ( !xmlMappedTypeExists( fqcn ) ) {
300 				context.logMessage(
301 						Diagnostic.Kind.WARNING,
302 						fqcn + " is mapped in xml, but class does not exist. Skipping meta model generation."
303 				);
304 				continue;
305 			}
306 
307 			XmlMetaEntity metaEntity = new XmlMetaEntity(
308 					mappedSuperClass, pkg, getXmlMappedType( fqcn ), context
309 			);
310 
311 			if ( context.containsMetaEntity( fqcn ) ) {
312 				context.logMessage(
313 						Diagnostic.Kind.WARNING,
314 						fqcn + " was already processed once. Skipping second occurrence."
315 				);
316 			}
317 			context.addMetaEntity( fqcn, metaEntity );
318 		}
319 	}

321 	private boolean xmlMappedTypeExists(String fullyQualifiedClassName) {
322 		Elements utils = context.getElementUtils();
323 		return utils.getTypeElement( fullyQualifiedClassName ) != null;
324 	}
325 
326 	private TypeElement getXmlMappedType(String fullyQualifiedClassName) {
327 		Elements utils = context.getElementUtils();
328 		return utils.getTypeElement( fullyQualifiedClassName );
329 	}

331 	private AccessType determineEntityAccessType(EntityMappings mappings) {
332 		AccessType accessType = context.getPersistenceUnitDefaultAccessType();
333 		if ( mappings.getAccess() != null ) {
334 			accessType = mapXmlAccessTypeToJpaAccessType( mappings.getAccess() );
335 		}
336 		return accessType;
337 	}
338 
339 	private void determineXmlAccessTypes() {
340 		for ( EntityMappings mappings : entityMappings ) {
341 			String fqcn;
342 			String packageName = mappings.getPackage();
343 			AccessType defaultAccessType = determineEntityAccessType( mappings );
344 
345 			for ( Entity entity : mappings.getEntity() ) {
346 				String name = entity.getClazz();
347 				fqcn = StringUtil.determineFullyQualifiedClassName( packageName, name );
348 				AccessType explicitAccessType = null;
349 				org.hibernate.jpamodelgen.xml.jaxb.AccessType type = entity.getAccess();
350 				if ( type != null ) {
351 					explicitAccessType = mapXmlAccessTypeToJpaAccessType( type );
352 				}
353 				AccessTypeInformation accessInfo = new AccessTypeInformation(
354 						fqcn, explicitAccessType, defaultAccessType
355 				);
356 				context.addAccessTypeInformation( fqcn, accessInfo );
357 			}
358 
359 			for ( org.hibernate.jpamodelgen.xml.jaxb.MappedSuperclass mappedSuperClass : mappings.getMappedSuperclass() ) {
360 				String name = mappedSuperClass.getClazz();
361 				fqcn = StringUtil.determineFullyQualifiedClassName( packageName, name );
362 				AccessType explicitAccessType = null;
363 				org.hibernate.jpamodelgen.xml.jaxb.AccessType type = mappedSuperClass.getAccess();
364 				if ( type != null ) {
365 					explicitAccessType = mapXmlAccessTypeToJpaAccessType( type );
366 				}
367 				AccessTypeInformation accessInfo = new AccessTypeInformation(
368 						fqcn, explicitAccessType, defaultAccessType
369 				);
370 				context.addAccessTypeInformation( fqcn, accessInfo );
371 			}
372 
373 			for ( org.hibernate.jpamodelgen.xml.jaxb.Embeddable embeddable : mappings.getEmbeddable() ) {
374 				String name = embeddable.getClazz();
375 				fqcn = StringUtil.determineFullyQualifiedClassName( packageName, name );
376 				AccessType explicitAccessType = null;
377 				org.hibernate.jpamodelgen.xml.jaxb.AccessType type = embeddable.getAccess();
378 				if ( type != null ) {
379 					explicitAccessType = mapXmlAccessTypeToJpaAccessType( type );
380 				}
381 				AccessTypeInformation accessInfo = new AccessTypeInformation(
382 						fqcn, explicitAccessType, defaultAccessType
383 				);
384 				context.addAccessTypeInformation( fqcn, accessInfo );
385 			}
386 		}
387 	}

389 	private void determineAnnotationAccessTypes() {
390 		for ( EntityMappings mappings : entityMappings ) {
391 			String fqcn;
392 			String packageName = mappings.getPackage();
393 
394 			for ( Entity entity : mappings.getEntity() ) {
395 				String name = entity.getClazz();
396 				fqcn = StringUtil.determineFullyQualifiedClassName( packageName, name );
397 				TypeElement element = context.getTypeElementForFullyQualifiedName( fqcn );
398 				if ( element != null ) {
399 					TypeUtils.determineAccessTypeForHierarchy( element, context );
400 				}
401 			}
402 
403 			for ( org.hibernate.jpamodelgen.xml.jaxb.MappedSuperclass mappedSuperClass : mappings.getMappedSuperclass() ) {
404 				String name = mappedSuperClass.getClazz();
405 				fqcn = StringUtil.determineFullyQualifiedClassName( packageName, name );
406 				TypeElement element = context.getTypeElementForFullyQualifiedName( fqcn );
407 				if ( element != null ) {
408 					TypeUtils.determineAccessTypeForHierarchy( element, context );
409 				}
410 			}
411 		}
412 	}

414 	/**
415 	 * Determines the default access type as specified in the <i>persistence-unit-defaults</i> as well as whether the
416 	 * xml configuration is complete and annotations should be ignored.
417 	 * <p/>
418 	 * Note, the spec says:
419 	 * <ul>
420 	 * <li>The persistence-unit-metadata element contains metadata for the entire persistence unit. It is
421 	 * undefined if this element occurs in multiple mapping files within the same persistence unit.</li>
422 	 * <li>If the xml-mapping-metadata-complete subelement is specified, the complete set of mapping
423 	 * metadata for the persistence unit is contained in the XML mapping files for the persistence unit, and any
424 	 * persistence annotations on the classes are ignored.</li>
425 	 * <li>When the xml-mapping-metadata-complete element is specified, any metadata-complete attributes specified
426 	 * within the entity, mapped-superclass, and embeddable elements are ignored.<li>
427 	 * </ul>
428 	 */
429 	private void determineDefaultAccessTypeAndMetaCompleteness() {
430 		for ( EntityMappings mappings : entityMappings ) {
431 			PersistenceUnitMetadata meta = mappings.getPersistenceUnitMetadata();
432 			if ( meta != null ) {
433 				if ( meta.getXmlMappingMetadataComplete() != null ) {
434 					context.mappingDocumentFullyXmlConfigured( true );
435 				}
436 				else {
437 					context.mappingDocumentFullyXmlConfigured( false );
438 				}
439 
440 				PersistenceUnitDefaults persistenceUnitDefaults = meta.getPersistenceUnitDefaults();
441 				if ( persistenceUnitDefaults != null ) {
442 					org.hibernate.jpamodelgen.xml.jaxb.AccessType xmlAccessType = persistenceUnitDefaults.getAccess();
443 					if ( xmlAccessType != null ) {
444 						context.setPersistenceUnitDefaultAccessType( mapXmlAccessTypeToJpaAccessType( xmlAccessType ) );
445 					}
446 				}
447 			}
448 			else {
449 				context.mappingDocumentFullyXmlConfigured( false );
450 			}
451 		}
452 	}
453 
454 	private AccessType mapXmlAccessTypeToJpaAccessType(org.hibernate.jpamodelgen.xml.jaxb.AccessType xmlAccessType) {
455 		switch ( xmlAccessType ) {
456 			case FIELD: {
457 				return AccessType.FIELD;
458 			}
459 			case PROPERTY: {
460 				return AccessType.PROPERTY;
461 			}
462 			default: {
463 			}
464 		}
465 		return null;
466 	}
