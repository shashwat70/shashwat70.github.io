7   package org.hibernate.boot.model.source.spi;
8   
9   /**
10   * Describes source information about the key of a persistent map.  At high
11   * level this broken down further into 2 categories:<ul>
12   *     <li>{@link PluralAttributeMapKeySourceEntityAttribute}</li>
13   *     <li>
14   *         <ul>
15   *             <li>{@link PluralAttributeMapKeySourceBasic}</li>
16   *             <li>{@link PluralAttributeMapKeySourceEmbedded}</li>
17   *             <li>{@link PluralAttributeMapKeyManyToManySource}</li>
18   *         </ul>
19   *     </li>
20   * </ul>
21   * <p/>
22   * {@link PluralAttributeMapKeySourceEntityAttribute} is only relevant from
23   * annotations when using {@link javax.persistence.MapKey}.
24   *
25   * @author Steve Ebersole
26   */
27  public interface PluralAttributeMapKeySource extends PluralAttributeIndexSource {
28  	public static enum Nature {
29  		BASIC,
30  		EMBEDDED,
31  		MANY_TO_MANY,
32  		ANY
33  	}
34  
35  	public Nature getMapKeyNature();
36  
37  	/**
38  	 * Is this plural attribute index source for an attribute of the referenced entity
39  	 * (relevant only for one-to-many and many-to-many associations)?
40  	 *
41  	 * If this method returns {@code true}, then this object can safely
42  	 * be cast to {@link PluralAttributeMapKeySourceEntityAttribute}.
43  	 *
44  	 * @return true, if this plural attribute index source for an attribute of the referenced
45  	 * entity; false, otherwise.
46  	 */
47  	public boolean isReferencedEntityAttribute();
48  }
