7   package org.hibernate.dialect.pagination;
8   
9   import org.hibernate.engine.spi.RowSelection;
10  
11  /**
12   * LIMIT clause handler compatible with ISO and ANSI SQL:2008 standard.
13   * 
14   * @author zhouyanming (zhouyanming@gmail.com)
15   */
16  public class SQL2008StandardLimitHandler extends AbstractLimitHandler {
17  
18  	public static final SQL2008StandardLimitHandler INSTANCE = new SQL2008StandardLimitHandler();
19  
20  	/**
21  	 * Constructs a SQL2008StandardLimitHandler
22  	 */
23  	private SQL2008StandardLimitHandler() {
24  		// NOP
25  	}
26  
27  	@Override
28  	public boolean supportsLimit() {
29  		return true;
30  	}
31  
32  	@Override
33  	public String processSql(String sql, RowSelection selection) {
34  		if (LimitHelper.useLimit( this, selection )) {
35  			return sql + (LimitHelper.hasFirstRow( selection ) ?
36  					" offset ? rows fetch next ? rows only" : " fetch first ? rows only");
37  		}
38  		else {
39  			// or return unaltered SQL
40  			return sql;
41  		}
42  	}
43  
44  }
