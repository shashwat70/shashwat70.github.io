7   package org.hibernate.loader.plan.build.internal.spaces;
8   
9   import org.hibernate.loader.plan.build.spi.ExpandingQuerySpace;
10  import org.hibernate.loader.plan.build.spi.ExpandingQuerySpaces;
11  import org.hibernate.loader.plan.spi.Join;
12  
13  /**
14   * @author Gail Badner
15   */
16  public abstract class AbstractExpandingSourceQuerySpace extends AbstractQuerySpace implements ExpandingQuerySpace {
17  
18  	public AbstractExpandingSourceQuerySpace(
19  			String uid,
20  			Disposition disposition,
21  			ExpandingQuerySpaces querySpaces,
22  			boolean canJoinsBeRequired) {
23  		super( uid, disposition, querySpaces, canJoinsBeRequired );
24  	}
25  
26  	@Override
27  	public void addJoin(Join join) {
28  		internalGetJoins().add( join );
29  	}
30  
31  	@Override
32  	public ExpandingQuerySpaces getExpandingQuerySpaces() {
33  		return super.getExpandingQuerySpaces();
34  	}
35  }
