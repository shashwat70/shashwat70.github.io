7   package org.hibernate.boot.registry;
8   
9   import java.util.Map;
10  
11  import org.hibernate.service.Service;
12  import org.hibernate.service.spi.ServiceInitiator;
13  import org.hibernate.service.spi.ServiceRegistryImplementor;
14  
15  /**
16   * Contract for an initiator of services that target the standard {@link org.hibernate.service.ServiceRegistry}.
17   *
18   * @param <R> The type of the service initiated.
19   *
20   * @author Steve Ebersole
21   */
22  public interface StandardServiceInitiator<R extends Service> extends ServiceInitiator<R> {
23  	/**
24  	 * Initiates the managed service.
25  	 *
26  	 * @param configurationValues The configuration values in effect
27  	 * @param registry The service registry.  Can be used to locate services needed to fulfill initiation.
28  	 *
29  	 * @return The initiated service.
30  	 */
31  	public R initiateService(Map configurationValues, ServiceRegistryImplementor registry);
32  }
