7   package org.hibernate.loader.plan.build.internal.spaces;
8   
9   import org.hibernate.engine.spi.SessionFactoryImplementor;
10  import org.hibernate.loader.plan.spi.CollectionQuerySpace;
11  import org.hibernate.loader.plan.spi.CompositeQuerySpace;
12  import org.hibernate.loader.plan.spi.EntityQuerySpace;
13  import org.hibernate.loader.plan.spi.JoinDefinedByMetadata;
14  import org.hibernate.loader.plan.spi.QuerySpace;
15  import org.hibernate.persister.entity.Joinable;
16  import org.hibernate.persister.entity.PropertyMapping;
17  import org.hibernate.type.CollectionType;
18  import org.hibernate.type.CompositeType;
19  import org.hibernate.type.EntityType;
20  
21  /**
22   * @author Steve Ebersole
23   * @author Gail Badner
24   */
25  public class JoinHelper {
26  	/**
27  	 * Singleton access
28  	 */
29  	public static final JoinHelper INSTANCE = new JoinHelper();
30  
31  	private JoinHelper() {
32  	}
33  
34  	public JoinDefinedByMetadata createEntityJoin(
35  			QuerySpace leftHandSide,
36  			String lhsPropertyName,
37  			EntityQuerySpace rightHandSide,
38  			boolean rightHandSideRequired,
39  			EntityType joinedPropertyType,
40  			SessionFactoryImplementor sessionFactory) {
41  		return new JoinImpl(
42  				leftHandSide,
43  				lhsPropertyName,
44  				rightHandSide,
45  				determineRhsColumnNames( joinedPropertyType, sessionFactory ),
46  				joinedPropertyType,
47  				rightHandSideRequired
48  		);
49  	}
50  
51  	public JoinDefinedByMetadata createCollectionJoin(
52  			QuerySpace leftHandSide,
53  			String lhsPropertyName,
54  			CollectionQuerySpace rightHandSide,
55  			boolean rightHandSideRequired,
56  			CollectionType joinedPropertyType,
57  			SessionFactoryImplementor sessionFactory) {
58  		return new JoinImpl(
59  				leftHandSide,
60  				lhsPropertyName,
61  				rightHandSide,
62  				joinedPropertyType.getAssociatedJoinable( sessionFactory ).getKeyColumnNames(),
63  				joinedPropertyType,
64  				rightHandSideRequired
65  		);
66  	}
67  
68  	public JoinDefinedByMetadata createCompositeJoin(
69  			QuerySpace leftHandSide,
70  			String lhsPropertyName,
71  			CompositeQuerySpace rightHandSide,
72  			boolean rightHandSideRequired,
73  			CompositeType joinedPropertyType) {
74  		return new JoinImpl(
75  				leftHandSide,
76  				lhsPropertyName,
77  				rightHandSide,
78  				null,
79  				joinedPropertyType,
80  				rightHandSideRequired
81  		);
82  	}
83  
84  	private static String[] determineRhsColumnNames(EntityType entityType, SessionFactoryImplementor sessionFactory) {
85  		final Joinable persister = entityType.getAssociatedJoinable( sessionFactory );
86  		return entityType.getRHSUniqueKeyPropertyName() == null ?
87  				persister.getKeyColumnNames() :
88  				( (PropertyMapping) persister ).toColumns( entityType.getRHSUniqueKeyPropertyName() );
89  	}
90  }
