7   package org.hibernate.resource.transaction.backend.jta.internal.synchronization;
8   
9   import javax.transaction.Synchronization;
10  
11  import org.jboss.logging.Logger;
12  
13  import static org.hibernate.internal.CoreLogging.logger;
14  
15  /**
16   * The JTA {@link javax.transaction.Synchronization} Hibernate registers when needed for JTA callbacks.
17   * <p/>
18   * Note that we split the notion of the registered Synchronization and the processing of the Synchronization callbacks
19   * mainly to account for "separation of concerns", but also so that the transaction engine does not have to hold
20   * reference to the actual Synchronization that gets registered with the JTA system.
21   *
22   * @author Steve Ebersole
23   */
24  public class RegisteredSynchronization implements Synchronization {
25  	private static final Logger log = logger( RegisteredSynchronization.class );
26  
27  	private final SynchronizationCallbackCoordinator synchronizationCallbackCoordinator;
28  
29  	public RegisteredSynchronization(SynchronizationCallbackCoordinator synchronizationCallbackCoordinator) {
30  		this.synchronizationCallbackCoordinator = synchronizationCallbackCoordinator;
31  	}
32  
33  	@Override
34  	public void beforeCompletion() {
35  		log.trace( "Registered JTA Synchronization : beforeCompletion()" );
36  
37  		synchronizationCallbackCoordinator.beforeCompletion();
38  	}
39  
40  	@Override
41  	public void afterCompletion(int status) {
42  		log.tracef( "Registered JTA Synchronization : afterCompletion(%s)", status );
43  
44  		synchronizationCallbackCoordinator.afterCompletion( status );
45  	}
46  }
