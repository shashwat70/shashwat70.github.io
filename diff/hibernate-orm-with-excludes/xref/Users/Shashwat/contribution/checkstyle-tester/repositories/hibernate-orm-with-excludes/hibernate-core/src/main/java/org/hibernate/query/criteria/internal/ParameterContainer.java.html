7   package org.hibernate.query.criteria.internal;
8   import javax.persistence.criteria.Selection;
9   
10  /**
11   * Contract for query components capable of either being a parameter or containing parameters.
12   *
13   * @author Steve Ebersole
14   */
15  public interface ParameterContainer {
16  	/**
17  	 * Register any parameters contained within this query component with the given registry.
18  	 *
19  	 * @param registry The parameter registry with which to register.
20  	 */
21  	public void registerParameters(ParameterRegistry registry);
22  
23  	/**
24  	 * Helper to deal with potential parameter container nodes.
25  	 */
26  	public static class Helper {
27  		public static void possibleParameter(Selection selection, ParameterRegistry registry) {
28  			if ( ParameterContainer.class.isInstance( selection ) ) {
29  				( (ParameterContainer) selection ).registerParameters( registry );
30  			}
31  		}
32  	}
33  }
