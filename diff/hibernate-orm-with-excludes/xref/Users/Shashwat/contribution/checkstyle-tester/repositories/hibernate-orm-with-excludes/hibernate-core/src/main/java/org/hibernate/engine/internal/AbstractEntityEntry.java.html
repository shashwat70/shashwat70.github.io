55  	/**
56  	 * Holds several boolean and enum typed attributes in a very compact manner. Enum values are stored in 4 bits
57  	 * (where 0 represents {@code null}, and each enum value is represented by its ordinal value + 1), thus allowing
58  	 * for up to 15 values per enum. Boolean values are stored in one bit.
59  	 * <p>
60  	 * The value is structured as follows:
61  	 *
62  	 * <pre>
63  	 * 1 - Lock mode
64  	 * 2 - Status
65  	 * 3 - Previous Status
66  	 * 4 - existsInDatabase
67  	 * 5 - isBeingReplicated
68  	 * 6 - loadedWithLazyPropertiesUnfetched; NOTE: this is not updated when properties are fetched lazily!
69  	 *
70  	 * 0000 0000 | 0000 0000 | 0654 3333 | 2222 1111
71  	 * </pre>
72  	 * Use {@link #setCompressedValue(org.hibernate.engine.internal.AbstractEntityEntry.EnumState, Enum)},
73  	 * {@link #getCompressedValue(org.hibernate.engine.internal.AbstractEntityEntry.EnumState)} etc
74  	 * to access the enums and booleans stored in this value.
75  	 * <p>
76  	 * Representing enum values by their ordinal value is acceptable for our case as this value itself is never
77  	 * serialized or deserialized and thus is not affected should ordinal values change.
78  	 */
79  	private transient int compressedState;

104 	public AbstractEntityEntry(
105 			final Status status,
106 			final Object[] loadedState,
107 			final Object rowId,
108 			final Serializable id,
109 			final Object version,
110 			final LockMode lockMode,
111 			final boolean existsInDatabase,
112 			final EntityPersister persister,
113 			final boolean disableVersionIncrement,
114 			final PersistenceContext persistenceContext) {
115 		setCompressedValue( EnumState.STATUS, status );
116 		// not useful strictly speaking but more explicit
117 		setCompressedValue( EnumState.PREVIOUS_STATUS, null );
118 		// only retain loaded state if the status is not Status.READ_ONLY
119 		if ( status != Status.READ_ONLY ) {
120 			this.loadedState = loadedState;
121 		}
122 		this.id=id;
123 		this.rowId=rowId;
124 		setCompressedValue( BooleanState.EXISTS_IN_DATABASE, existsInDatabase );
125 		this.version=version;
126 		setCompressedValue( EnumState.LOCK_MODE, lockMode );
127 		setCompressedValue( BooleanState.IS_BEING_REPLICATED, disableVersionIncrement );
128 		this.persister=persister;
129 		this.persistenceContext = persistenceContext;
130 	}

135 	@SuppressWarnings( {"JavaDoc"})
136 	protected AbstractEntityEntry(
137 			final SessionFactoryImplementor factory,
138 			final String entityName,
139 			final Serializable id,
140 			final Status status,
141 			final Status previousStatus,
142 			final Object[] loadedState,
143 			final Object[] deletedState,
144 			final Object version,
145 			final LockMode lockMode,
146 			final boolean existsInDatabase,
147 			final boolean isBeingReplicated,
148 			final PersistenceContext persistenceContext) {
149 		this.persister = ( factory == null ? null : factory.getEntityPersister( entityName ) );
150 		this.id = id;
151 		setCompressedValue( EnumState.STATUS, status );
152 		setCompressedValue( EnumState.PREVIOUS_STATUS, previousStatus );
153 		this.loadedState = loadedState;
154 		setDeletedState( deletedState );
155 		this.version = version;
156 		setCompressedValue( EnumState.LOCK_MODE, lockMode );
157 		setCompressedValue( BooleanState.EXISTS_IN_DATABASE, existsInDatabase );
158 		setCompressedValue( BooleanState.IS_BEING_REPLICATED, isBeingReplicated );
159 		this.rowId = null; // this is equivalent to the old behavior...
160 		this.persistenceContext = persistenceContext;
161 	}

183 	@Override
184 	public void setStatus(Status status) {
185 		if ( status == Status.READ_ONLY ) {
186 			//memory optimization
187 			loadedState = null;
188 		}
189 
190 		final Status currentStatus = this.getStatus();
191 
192 		if ( currentStatus != status ) {
193 			setCompressedValue( EnumState.PREVIOUS_STATUS, currentStatus );
194 			setCompressedValue( EnumState.STATUS, status );
195 		}
196 	}

218 		EntityEntryExtraStateHolder extra = getExtraState( EntityEntryExtraStateHolder.class );
219 		if ( extra == null && deletedState == DEFAULT_DELETED_STATE ) {
220 			//this is the default value and we do not store the extra state
221 			return;
222 		}
223 		if ( extra == null ) {
224 			extra = new EntityEntryExtraStateHolder();
225 			addExtraState( extra );
226 		}
227 		extra.setDeletedState( deletedState );
228 	}

272 	@Override
273 	public void postUpdate(Object entity, Object[] updatedState, Object nextVersion) {
274 		this.loadedState = updatedState;
275 		setLockMode( LockMode.WRITE );
276 
277 		if ( getPersister().isVersioned() ) {
278 			this.version = nextVersion;
279 			getPersister().setPropertyValue( entity, getPersister().getVersionProperty(), nextVersion );
280 		}
281 
282 		if( entity instanceof SelfDirtinessTracker ) {
283 			( (SelfDirtinessTracker) entity ).$$_hibernate_clearDirtyAttributes();
284 		}
285 
286 		getPersistenceContext().getSession()
287 				.getFactory()
288 				.getCustomEntityDirtinessStrategy()
289 				.resetDirty( entity, getPersister(), (Session) getPersistenceContext().getSession() );
290 	}

305 	public boolean isNullifiable(boolean earlyInsert, SharedSessionContractImplementor session) {
306 		if ( getStatus() == Status.SAVING ) {
307 			return true;
308 		}
309 		else if ( earlyInsert ) {
310 			return !isExistsInDatabase();

346 	@SuppressWarnings( {"SimplifiableIfStatement"})
347 	private boolean isUnequivocallyNonDirty(Object entity) {
348 		if ( entity instanceof SelfDirtinessTracker ) {
349 			return ! persister.hasCollections() && ! ( (SelfDirtinessTracker) entity ).$$_hibernate_hasDirtyAttributes();
350 		}
351 
352 		if ( entity instanceof PersistentAttributeInterceptable ) {
353 			final PersistentAttributeInterceptable interceptable = (PersistentAttributeInterceptable) entity;
354 			final PersistentAttributeInterceptor interceptor = interceptable.$$_hibernate_getInterceptor();
355 			if ( interceptor instanceof EnhancementAsProxyLazinessInterceptor ) {
356 				// we never have to check an uninitialized proxy
357 				return true;
358 			}
359 		}
360 
361 		final CustomEntityDirtinessStrategy customEntityDirtinessStrategy =
362 				getPersistenceContext().getSession().getFactory().getCustomEntityDirtinessStrategy();
363 		if ( customEntityDirtinessStrategy.canDirtyCheck( entity, getPersister(), (Session) getPersistenceContext().getSession() ) ) {
364 			return ! customEntityDirtinessStrategy.isDirty( entity, getPersister(), (Session) getPersistenceContext().getSession() );
365 		}
366 
367 		if ( getPersister().hasMutableProperties() ) {
368 			return false;
369 		}
370 
371 		return false;
372 	}

374 	@Override
375 	public boolean isModifiableEntity() {
376 		final Status status = getStatus();
377 		final Status previousStatus = getPreviousStatus();
378 		return getPersister().isMutable()
379 				&& status != Status.READ_ONLY
380 				&& ! ( status == Status.DELETED && previousStatus == Status.READ_ONLY );
381 	}

385 		version = nextVersion;
386 		loadedState[ persister.getVersionProperty() ] = version;
387 		// TODO:  use LockMode.PESSIMISTIC_FORCE_INCREMENT
388 		//noinspection deprecation
389 		setLockMode( LockMode.FORCE );

393 	@Override
394 	public boolean isReadOnly() {
395 		final Status status = getStatus();
396 		if (status != Status.MANAGED && status != Status.READ_ONLY) {
397 			throw new HibernateException("instance was not in a valid state");
398 		}
399 		return status == Status.READ_ONLY;
400 	}
401 
402 	@Override
403 	public void setReadOnly(boolean readOnly, Object entity) {
404 		if ( readOnly == isReadOnly() ) {
405 			// simply return since the status is not being changed
406 			return;
407 		}
408 		if ( readOnly ) {
409 			setStatus( Status.READ_ONLY );
410 			loadedState = null;
411 		}
412 		else {
413 			if ( ! persister.isMutable() ) {
414 				throw new IllegalStateException( "Cannot make an immutable entity modifiable." );
415 			}
416 			setStatus( Status.MANAGED );
417 			loadedState = getPersister().getPropertyValues( entity );
418 			getPersistenceContext().getNaturalIdHelper().manageLocalNaturalIdCrossReference(
419 					persister,
420 					id,
421 					loadedState,
422 					null,
423 					CachedNaturalIdValueSource.LOAD
424 			);
425 		}
426 	}

435 	@Override
436 	public void serialize(ObjectOutputStream oos) throws IOException {
437 		final Status previousStatus = getPreviousStatus();
438 		oos.writeObject( getEntityName() );
439 		oos.writeObject( id );
440 		oos.writeObject( getStatus().name() );
441 		oos.writeObject( (previousStatus == null ? "" : previousStatus.name()) );
442 		// todo : potentially look at optimizing these two arrays
443 		oos.writeObject( loadedState );
444 		oos.writeObject( getDeletedState() );
445 		oos.writeObject( version );
446 		oos.writeObject( getLockMode().toString() );
447 		oos.writeBoolean( isExistsInDatabase() );
448 		oos.writeBoolean( isBeingReplicated() );
449 	}

479 	/**
480 	 * Saves the value for the given enum property.
481 	 *
482 	 * @param state
483 	 *            identifies the value to store
484 	 * @param value
485 	 *            the value to store; The caller must make sure that it matches
486 	 *            the given identifier
487 	 */
488 	protected <E extends Enum<E>> void setCompressedValue(EnumState<E> state, E value) {
489 		// reset the bits for the given property to 0
490 		compressedState &= state.getUnsetMask();
491 		// store the numeric representation of the enum value at the right offset
492 		compressedState |= ( state.getValue( value ) << state.getOffset() );
493 	}
494 
495 	/**
496 	 * Gets the current value of the given enum property.
497 	 *
498 	 * @param state
499 	 *            identifies the value to store
500 	 * @return the current value of the specified property
501 	 */
502 	protected <E extends Enum<E>> E getCompressedValue(EnumState<E> state) {
503 		// restore the numeric value from the bits at the right offset and return the corresponding enum constant
504 		final int index = ( ( compressedState & state.getMask() ) >> state.getOffset() ) - 1;
505 		return index == - 1 ? null : state.getEnumConstants()[index];
506 	}
507 
508 	/**
509 	 * Saves the value for the given boolean flag.
510 	 *
511 	 * @param state
512 	 *            identifies the value to store
513 	 * @param value
514 	 *            the value to store
515 	 */
516 	protected void setCompressedValue(BooleanState state, boolean value) {
517 		compressedState &= state.getUnsetMask();
518 		compressedState |= ( state.getValue( value ) << state.getOffset() );
519 	}
520 
521 	/**
522 	 * Gets the current value of the given boolean flag.
523 	 *
524 	 * @param state
525 	 *            identifies the value to store
526 	 * @return the current value of the specified flag
527 	 */
528 	protected boolean getCompressedValue(BooleanState state) {
529 		return ( ( compressedState & state.getMask() ) >> state.getOffset() ) == 1;
530 	}

532 	/**
533 	 * Represents an enum value stored within a number value, using four bits starting at a specified offset.
534 	 *
535 	 * @author Gunnar Morling
536 	 */
537 	protected static class EnumState<E extends Enum<E>> {
538 
539 		protected static final EnumState<LockMode> LOCK_MODE = new EnumState<LockMode>( 0, LockMode.class );
540 		protected static final EnumState<Status> STATUS = new EnumState<Status>( 4, Status.class );
541 		protected static final EnumState<Status> PREVIOUS_STATUS = new EnumState<Status>( 8, Status.class );
542 
543 		protected final int offset;
544 		protected final E[] enumConstants;
545 		protected final int mask;
546 		protected final int unsetMask;
547 
548 		private EnumState(int offset, Class<E> enumType) {
549 			final E[] enumConstants = enumType.getEnumConstants();
550 
551 			// In case any of the enums cannot be stored in 4 bits anymore, we'd have to re-structure the compressed
552 			// state int
553 			if ( enumConstants.length > 15 ) {
554 				throw new AssertionFailure( "Cannot store enum type " + enumType.getName() + " in compressed state as"
555 						+ " it has too many values." );
556 			}
557 
558 			this.offset = offset;
559 			this.enumConstants = enumConstants;
560 
561 			// a mask for reading the four bits, starting at the right offset
562 			this.mask = 0xF << offset;
563 
564 			// a mask for setting the four bits at the right offset to 0
565 			this.unsetMask = 0xFFFF & ~mask;
566 		}
567 
568 		/**
569 		 * Returns the numeric value to be stored for the given enum value.
570 		 */
571 		private int getValue(E value) {
572 			return value != null ? value.ordinal() + 1 : 0;
573 		}
574 
575 		/**
576 		 * Returns the offset within the number value at which this enum value is stored.
577 		 */
578 		private int getOffset() {
579 			return offset;
580 		}
581 
582 		/**
583 		 * Returns the bit mask for reading this enum value from the number value storing it.
584 		 */
585 		private int getMask() {
586 			return mask;
587 		}
588 
589 		/**
590 		 * Returns the bit mask for resetting this enum value from the number value storing it.
591 		 */
592 		private int getUnsetMask() {
593 			return unsetMask;
594 		}
595 
596 		/**
597 		 * Returns the constants of the represented enum which is cached for performance reasons.
598 		 */
599 		private E[] getEnumConstants() {
600 			return enumConstants;
601 		}
602 	}

604 	/**
605 	 * Represents a boolean flag stored within a number value, using one bit at a specified offset.
606 	 *
607 	 * @author Gunnar Morling
608 	 */
609 	protected enum BooleanState {
610 
611 		EXISTS_IN_DATABASE(13),
612 		IS_BEING_REPLICATED(14);
613 
614 		private final int offset;
615 		private final int mask;
616 		private final int unsetMask;
617 
618 		private BooleanState(int offset) {
619 			this.offset = offset;
620 			this.mask = 0x1 << offset;
621 			this.unsetMask = 0xFFFF & ~mask;
622 		}
623 
624 		private int getValue(boolean value) {
625 			return value ? 1 : 0;
626 		}
627 
628 		/**
629 		 * Returns the offset within the number value at which this boolean flag is stored.
630 		 */
631 		private int getOffset() {
632 			return offset;
633 		}
634 
635 		/**
636 		 * Returns the bit mask for reading this flag from the number value storing it.
637 		 */
638 		private int getMask() {
639 			return mask;
640 		}
641 
642 		/**
643 		 * Returns the bit mask for resetting this flag from the number value storing it.
644 		 */
645 		private int getUnsetMask() {
646 			return unsetMask;
647 		}
648 	}
