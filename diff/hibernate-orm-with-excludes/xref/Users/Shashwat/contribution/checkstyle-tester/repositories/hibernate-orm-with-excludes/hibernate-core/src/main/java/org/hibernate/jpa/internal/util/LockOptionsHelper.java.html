7   package org.hibernate.jpa.internal.util;
8   
9   import java.util.Map;
10  import java.util.function.Supplier;
11  import javax.persistence.PersistenceException;
12  import javax.persistence.PessimisticLockScope;
13  
14  import org.hibernate.LockOptions;
15  
16  import static org.hibernate.cfg.AvailableSettings.JPA_LOCK_SCOPE;
17  import static org.hibernate.cfg.AvailableSettings.JPA_LOCK_TIMEOUT;
18  
19  public final class LockOptionsHelper {
20  
21  	private LockOptionsHelper() {
22  		//utility class, not to be constructed
23  	}
24  
25  	/**
26  	 * Applies configuration properties on a {@link LockOptions} instance, passed as a supplier
27  	 * so to make it possible to skip allocating the {@link LockOptions} instance if there's
28  	 * nothing to set.
29  	 *
30  	 * @param props The configuration properties
31  	 * @param lockOptionsSupplier The reference to the lock to modify
32  	 */
33  	public static void applyPropertiesToLockOptions(final Map<String, Object> props, final Supplier<LockOptions> lockOptionsSupplier) {
34  		Object lockScope = props.get( JPA_LOCK_SCOPE );
35  		if ( lockScope instanceof String && PessimisticLockScope.valueOf( (String) lockScope ) == PessimisticLockScope.EXTENDED ) {
36  			lockOptionsSupplier.get().setScope( true );
37  		}
38  		else if ( lockScope instanceof PessimisticLockScope ) {
39  			boolean extended = PessimisticLockScope.EXTENDED.equals( lockScope );
40  			lockOptionsSupplier.get().setScope( extended );
41  		}
42  		else if ( lockScope != null ) {
43  			throw new PersistenceException( "Unable to parse " + JPA_LOCK_SCOPE + ": " + lockScope );
44  		}
45  
46  		Object lockTimeout = props.get( JPA_LOCK_TIMEOUT );
47  		int timeout = 0;
48  		boolean timeoutSet = false;
49  		if ( lockTimeout instanceof String ) {
50  			timeout = Integer.parseInt( (String) lockTimeout );
51  			timeoutSet = true;
52  		}
53  		else if ( lockTimeout instanceof Number ) {
54  			timeout = ( (Number) lockTimeout ).intValue();
55  			timeoutSet = true;
56  		}
57  		else if ( lockTimeout != null ) {
58  			throw new PersistenceException( "Unable to parse " + JPA_LOCK_TIMEOUT + ": " + lockTimeout );
59  		}
60  
61  		if ( timeoutSet ) {
62  			if ( timeout == LockOptions.SKIP_LOCKED ) {
63  				lockOptionsSupplier.get().setTimeOut( LockOptions.SKIP_LOCKED );
64  			}
65  			else if ( timeout < 0 ) {
66  				lockOptionsSupplier.get().setTimeOut( LockOptions.WAIT_FOREVER );
67  			}
68  			else if ( timeout == 0 ) {
69  				lockOptionsSupplier.get().setTimeOut( LockOptions.NO_WAIT );
70  			}
71  			else {
72  				lockOptionsSupplier.get().setTimeOut( timeout );
73  			}
74  		}
75  	}
76  
77  }
