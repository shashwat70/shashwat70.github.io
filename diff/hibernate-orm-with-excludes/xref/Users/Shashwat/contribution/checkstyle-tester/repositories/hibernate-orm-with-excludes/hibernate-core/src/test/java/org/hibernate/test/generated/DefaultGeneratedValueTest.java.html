135 	@Test
136 	@TestForIssue(jiraKey = "HHH-2907")
137 	public void testUpdateTimestampGeneration() {
138 		Session s = openSession();
139 		s.beginTransaction();
140 		TheEntity theEntity = new TheEntity( 1 );
141 		assertNull( theEntity.updated );
142 		s.save( theEntity );
143 
144 		//TODO: Actually the value should be non-null after save
145 		assertNull( theEntity.updated );
146 		s.getTransaction().commit();
147 		s.close();
148 
149 		Timestamp created = theEntity.vmCreatedSqlTimestamp;
150 		Timestamp updated = theEntity.updated;
151 		assertNotNull( updated );
152 		assertNotNull( created );
153 
154 		s = openSession();
155 		s.beginTransaction();
156 
157 		theEntity = (TheEntity) s.get( TheEntity.class, 1 );
158 		theEntity.lastName = "Smith";
159 		try {
160 			Thread.sleep( 1 );
161 		}
162 		catch (InterruptedException ignore) {}
163 		s.getTransaction().commit();
164 		s.close();
165 
166 		s = openSession();
167 		s.beginTransaction();
168 
169 		theEntity = (TheEntity) s.get( TheEntity.class, 1 );
170 
171 		assertEquals( "Creation timestamp should not change on update", created, theEntity.vmCreatedSqlTimestamp );
172 		assertTrue( "Update timestamp should have changed due to update", theEntity.updated.after( updated ) );
173 
174 		s.delete( theEntity );
175 		s.getTransaction().commit();
176 		s.close();
177 	}
