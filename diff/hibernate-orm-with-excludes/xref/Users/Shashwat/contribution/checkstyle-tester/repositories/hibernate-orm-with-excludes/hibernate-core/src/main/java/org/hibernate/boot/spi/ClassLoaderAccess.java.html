7   package org.hibernate.boot.spi;
8   
9   import java.net.URL;
10  
11  /**
12   * During the process of building this metamodel, accessing the ClassLoader
13   * is very discouraged.  However, sometimes it is needed.  This contract helps
14   * mitigate accessing the ClassLoader in these cases.
15   *
16   * @author Steve Ebersole
17   *
18   * @since 5.0
19   */
20  public interface ClassLoaderAccess {
21  	/**
22  	 * Obtain a Class reference by name
23  	 *
24  	 * @param name The name of the Class to get a reference to.
25  	 *
26  	 * @return The Class.
27  	 */
28  	public <T> Class<T> classForName(String name);
29  
30  	/**
31  	 * Locate a resource by name
32  	 *
33  	 * @param resourceName The name of the resource to resolve.
34  	 *
35  	 * @return The located resource; may return {@code null} to indicate the resource was not found
36  	 */
37  	public URL locateResource(String resourceName);
38  }
