7   package org.hibernate.boot.model.source.spi;
8   
9   import org.hibernate.boot.model.naming.ImplicitDiscriminatorColumnNameSource;
10  
11  /**
12   * Contract for sources of information about a mapped discriminator.
13   *
14   * @author Hardy Ferentschik
15   * @author Steve Ebersole
16   */
17  public interface DiscriminatorSource extends ImplicitDiscriminatorColumnNameSource {
18  	/**
19  	 * Obtain the column/formula information about the discriminator.
20  	 *
21  	 * @return The column/formula information
22  	 */
23  	public RelationalValueSource getDiscriminatorRelationalValueSource();
24  
25  	/**
26  	 * Obtain the discriminator type.  Unlike the type of attributes, implementors
27  	 * here should generally specify the type in case of no user selection rather than
28  	 * return null because we cannot infer it from any physical java member.
29  	 *
30  	 * @return The discriminator type
31  	 */
32  	public String getExplicitHibernateTypeName();
33  
34  	/**
35  	 * "Forces" Hibernate to specify the allowed discriminator values, even when retrieving
36  	 * all instances of the root class.
37  	 *
38  	 * @return {@code true} in case the discriminator value should be forces, {@code false}
39  	 * otherwise. Default is {@code false}.
40  	 */
41  	boolean isForced();
42  
43  	/**
44  	 * Set this to {@code false}, if your discriminator column is also part of a mapped
45  	 * composite identifier.  It tells Hibernate not to include the column in SQL INSERTs.
46  	 *
47  	 * @return {@code true} in case the discriminator value should be included in inserts,
48  	 * {@code false} otherwise.  Default is {@code true}.
49  	 */
50  	boolean isInserted();
51  }
