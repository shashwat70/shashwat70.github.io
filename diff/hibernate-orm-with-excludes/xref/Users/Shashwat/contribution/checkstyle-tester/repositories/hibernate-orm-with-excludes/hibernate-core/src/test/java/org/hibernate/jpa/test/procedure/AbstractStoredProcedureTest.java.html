7   package org.hibernate.jpa.test.procedure;
8   
9   import java.util.List;
10  import javax.persistence.EntityManager;
11  
12  import org.hibernate.engine.spi.SessionFactoryImplementor;
13  import org.hibernate.jpa.test.BaseEntityManagerFunctionalTestCase;
14  import org.hibernate.procedure.internal.ProcedureCallMementoImpl;
15  import org.hibernate.procedure.spi.ParameterStrategy;
16  import org.hibernate.type.IntegerType;
17  import org.hibernate.type.LongType;
18  import org.hibernate.type.StringType;
19  
20  import org.junit.Test;
21  
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.assertNotNull;
24  
25  /**
26   * @author Strong Liu <stliu@hibernate.org>
27   */
28  public abstract class AbstractStoredProcedureTest extends BaseEntityManagerFunctionalTestCase {
29  	@Test
30  	public void testNamedStoredProcedureBinding() {
31  		EntityManager em = getOrCreateEntityManager();
32  		SessionFactoryImplementor sf = em.getEntityManagerFactory().unwrap( SessionFactoryImplementor.class );
33  		final ProcedureCallMementoImpl m1 = (ProcedureCallMementoImpl) sf.getNamedQueryRepository()
34  				.getNamedProcedureCallMemento( "s1" );
35  		assertNotNull( m1 );
36  		assertEquals( "p1", m1.getProcedureName() );
37  		assertEquals( ParameterStrategy.NAMED, m1.getParameterStrategy() );
38  		List<ProcedureCallMementoImpl.ParameterMemento> list = m1.getParameterDeclarations();
39  		assertEquals( 2, list.size() );
40  		ProcedureCallMementoImpl.ParameterMemento memento = list.get( 0 );
41  		assertEquals( "p11", memento.getName() );
42  		assertEquals( javax.persistence.ParameterMode.IN, memento.getMode() );
43  		assertEquals( IntegerType.INSTANCE, memento.getHibernateType() );
44  		assertEquals( Integer.class, memento.getType() );
45  
46  		memento = list.get( 1 );
47  		assertEquals( "p12", memento.getName() );
48  		assertEquals( javax.persistence.ParameterMode.IN, memento.getMode() );
49  		assertEquals( IntegerType.INSTANCE, memento.getHibernateType() );
50  		assertEquals( Integer.class, memento.getType() );
51  
52  
53  
54  		final ProcedureCallMementoImpl m2 = (ProcedureCallMementoImpl) sf.getNamedQueryRepository()
55  				.getNamedProcedureCallMemento( "s2" );
56  		assertNotNull( m2 );
57  		assertEquals( "p2", m2.getProcedureName() );
58  		assertEquals( ParameterStrategy.POSITIONAL, m2.getParameterStrategy() );
59  		list = m2.getParameterDeclarations();
60  
61  		memento = list.get( 0 );
62  		assertEquals( Integer.valueOf( 1 ), memento.getPosition() );
63  		assertEquals( javax.persistence.ParameterMode.INOUT, memento.getMode() );
64  		assertEquals( StringType.INSTANCE, memento.getHibernateType() );
65  		assertEquals( String.class, memento.getType() );
66  
67  		memento = list.get( 1 );
68  		assertEquals( Integer.valueOf( 2 ), memento.getPosition() );
69  		assertEquals( javax.persistence.ParameterMode.INOUT, memento.getMode() );
70  		assertEquals( LongType.INSTANCE, memento.getHibernateType() );
71  		assertEquals( Long.class, memento.getType() );
72  
73  	}
74  }
