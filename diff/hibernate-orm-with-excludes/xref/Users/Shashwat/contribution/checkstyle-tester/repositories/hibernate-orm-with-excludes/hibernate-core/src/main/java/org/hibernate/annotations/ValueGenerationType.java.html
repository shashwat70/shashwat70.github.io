7   package org.hibernate.annotations;
8   
9   import java.lang.annotation.ElementType;
10  import java.lang.annotation.Retention;
11  import java.lang.annotation.RetentionPolicy;
12  import java.lang.annotation.Target;
13  
14  import org.hibernate.tuple.AnnotationValueGeneration;
15  
16  /**
17   * Marks an annotation type as a generator annotation type.
18   * <p>
19   * Adding a generator annotation to an entity property causes the value of the property to be generated upon insert
20   * or update of the owning entity. Not more than one generator annotation may be placed on a given property.
21   * <p>
22   * Each generator annotation type is associated with a {@link AnnotationValueGeneration} which implements the strategy
23   * for generating the value. Generator annotation types may define arbitrary custom attributes, e.g. allowing the
24   * client to configure the generation timing (if applicable) or other settings taking an effect on the value generation.
25   * The corresponding implementation can retrieve these settings from the annotation instance passed to
26   * {@link AnnotationValueGeneration#initialize(java.lang.annotation.Annotation, Class)}.
27   * <p>
28   * Custom generator annotation types must have retention policy {@link RetentionPolicy#RUNTIME}.
29  
30   * @author Gunnar Morling
31   */
32  @Target( value = ElementType.ANNOTATION_TYPE )
33  @Retention( RetentionPolicy.RUNTIME )
34  public @interface ValueGenerationType {
35  
36  	/**
37  	 * The type of value generation associated with the annotated value generator annotation type. The referenced
38  	 * generation type must be parameterized with the type of the given generator annotation.
39  	 *
40  	 * @return the value generation type
41  	 */
42  	Class<? extends AnnotationValueGeneration<?>> generatedBy();
43  }
