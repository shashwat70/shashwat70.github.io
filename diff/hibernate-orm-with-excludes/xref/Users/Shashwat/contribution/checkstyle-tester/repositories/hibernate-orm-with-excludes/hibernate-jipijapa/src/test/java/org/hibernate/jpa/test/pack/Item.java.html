7   package org.hibernate.jpa.test.pack;
8   
9   import java.io.Serializable;
10  import java.util.HashSet;
11  import java.util.Set;
12  import javax.persistence.Column;
13  import javax.persistence.Entity;
14  import javax.persistence.EntityResult;
15  import javax.persistence.FieldResult;
16  import javax.persistence.Id;
17  import javax.persistence.LockModeType;
18  import javax.persistence.NamedNativeQueries;
19  import javax.persistence.NamedNativeQuery;
20  import javax.persistence.NamedQueries;
21  import javax.persistence.NamedQuery;
22  import javax.persistence.OneToMany;
23  import javax.persistence.QueryHint;
24  import javax.persistence.SqlResultSetMapping;
25  
26  import org.hibernate.annotations.QueryHints;
27  
28  /**
29   * @author Gavin King
30   */
31  @Entity(name = "Item")
32  @SqlResultSetMapping(name = "getItem", entities =
33  @EntityResult(entityClass = Item.class, fields = {
34  		@FieldResult(name = "name", column = "itemname"),
35  		@FieldResult(name = "descr", column = "itemdescription")
36  })
37  )
38  @NamedNativeQueries({
39  		@NamedNativeQuery(
40  				name = "nativeItem1",
41  				query = "select name as itemname, descr as itemdescription from Item",
42  				resultSetMapping = "getItem"
43  		),
44  		@NamedNativeQuery(
45  				name = "nativeItem2",
46  				query = "select * from Item",
47  				resultClass = Item.class
48  		)
49  })
50  @NamedQueries({
51  		@NamedQuery(
52  				name = "itemJpaQueryWithLockModeAndHints",
53  				query = "select i from Item i",
54  				lockMode = LockModeType.PESSIMISTIC_WRITE,
55  				hints = {
56  						@QueryHint(name = QueryHints.TIMEOUT_JPA, value = "3000"),
57  						@QueryHint(name = QueryHints.CACHE_MODE, value = "ignore"),
58  						@QueryHint(name = QueryHints.CACHEABLE, value = "true"),
59  						@QueryHint(name = QueryHints.READ_ONLY, value = "true"),
60  						@QueryHint(name = QueryHints.COMMENT, value = "custom static comment"),
61  						@QueryHint(name = QueryHints.FETCH_SIZE, value = "512"),
62  						@QueryHint(name = QueryHints.FLUSH_MODE, value = "manual")
63  				}
64  		),
65  		@NamedQuery(name = "query-construct", query = "select new Item(i.name,i.descr) from Item i")
66  })
67  public class Item implements Serializable {
68  
69  	private String name;
70  	private String descr;
71  	private Set<Distributor> distributors = new HashSet<Distributor>();
72  
73  	public Item() {
74  	}
75  
76  	public Item(String name, String desc) {
77  		this.name = name;
78  		this.descr = desc;
79  	}
80  
81  	@Column(length = 200)
82  	public String getDescr() {
83  		return descr;
84  	}
85  
86  	public void setDescr(String desc) {
87  		this.descr = desc;
88  	}
89  
90  	@Id
91  	@Column(length = 30)
92  	public String getName() {
93  		return name;
94  	}
95  
96  	public void setName(String name) {
97  		this.name = name;
98  	}
99  
100 	@OneToMany
101 	public Set<Distributor> getDistributors() {
102 		return distributors;
103 	}
104 
105 	public void setDistributors(Set<Distributor> distributors) {
106 		this.distributors = distributors;
107 	}
108 
109 	public void addDistributor(Distributor d) {
110 		if ( distributors == null ) {
111 			distributors = new HashSet();
112 		}
113 		distributors.add( d );
114 	}
115 }
