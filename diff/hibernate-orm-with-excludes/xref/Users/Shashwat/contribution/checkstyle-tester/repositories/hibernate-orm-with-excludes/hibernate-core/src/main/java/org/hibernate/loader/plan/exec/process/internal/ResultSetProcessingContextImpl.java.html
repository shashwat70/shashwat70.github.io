62  	/**
63  	 * Builds a ResultSetProcessingContextImpl
64  	 *
65  	 * @param shouldUseOptionalEntityInformation There are times when the "optional entity information" on
66  	 * QueryParameters should be used and times when they should not.  Collection initializers, batch loaders, etc
67  	 * are times when it should NOT be used.
68  	 */
69  	public ResultSetProcessingContextImpl(
70  			final ResultSet resultSet,
71  			final SharedSessionContractImplementor session,
72  			final LoadPlan loadPlan,
73  			final AliasResolutionContext aliasResolutionContext,
74  			final boolean readOnly,
75  			final boolean shouldUseOptionalEntityInformation,
76  			final boolean shouldReturnProxies,
77  			final QueryParameters queryParameters,
78  			final NamedParameterContext namedParameterContext,
79  			final boolean hadSubselectFetches) {
80  		this.resultSet = resultSet;
81  		this.session = session;
82  		this.loadPlan = loadPlan;
83  		this.aliasResolutionContext = aliasResolutionContext;
84  		this.readOnly = readOnly;
85  		this.shouldUseOptionalEntityInformation = shouldUseOptionalEntityInformation;
86  		this.shouldReturnProxies = shouldReturnProxies;
87  		this.queryParameters = queryParameters;
88  		this.namedParameterContext = namedParameterContext;
89  		this.hadSubselectFetches = hadSubselectFetches;
90  
91  		if ( shouldUseOptionalEntityInformation ) {
92  			if ( queryParameters.getOptionalId() != null ) {
93  				// make sure we have only one return
94  				if ( loadPlan.getReturns().size() > 1 ) {
95  					throw new IllegalStateException( "Cannot specify 'optional entity' values with multi-return load plans" );
96  				}
97  			}
98  		}
99  	}

132 		if ( queryParameters.getLockOptions() != null && queryParameters.getLockOptions()

139 	@Override
140 	public EntityReferenceProcessingState getProcessingState(final EntityReference entityReference) {
141 		EntityReferenceProcessingState context;
142 		if ( identifierResolutionContextMap == null ) {
143 			//The default expected size of IdentityHashMap is 21, which is likely to allocate larger arrays than what is typically necessary.
144 			//Reducing to 5, as a reasonable estimate for typical use: any larger query can better justify the need to resize,
145 			//while single loads shouldn't pay such an high cost.
146 			//This can save a lot of memory as it reduces the internal table of IdentityHashMap from a 64 slot array, to 16 slots:
147 			//that's a 75% memory cost reduction for usage patterns which do many individual loads.
148 			identifierResolutionContextMap = new IdentityHashMap<>(5);
149 			context = null;
150 		}
151 		else {
152 			context = identifierResolutionContextMap.get( entityReference );
153 		}
154 
155 		if ( context == null ) {
156 			context = new EntityReferenceProcessingState() {
157 				private boolean wasMissingIdentifier;
158 				private Object identifierHydratedForm;
159 				private EntityKey entityKey;
160 				private Object[] hydratedState;
161 				private Object entityInstance;
162 
163 				@Override
164 				public EntityReference getEntityReference() {
165 					return entityReference;
166 				}
167 
168 				@Override
169 				public void registerMissingIdentifier() {
170 					if ( !EntityFetch.class.isInstance( entityReference ) ) {
171 						throw new IllegalStateException( "Missing return row identifier" );
172 					}
173 					ResultSetProcessingContextImpl.this.registerNonExists( (EntityFetch) entityReference );
174 					wasMissingIdentifier = true;
175 				}
176 
177 				@Override
178 				public boolean isMissingIdentifier() {
179 					return wasMissingIdentifier;
180 				}
181 
182 				@Override
183 				public void registerIdentifierHydratedForm(Object identifierHydratedForm) {
184 					this.identifierHydratedForm = identifierHydratedForm;
185 				}
186 
187 				@Override
188 				public Object getIdentifierHydratedForm() {
189 					return identifierHydratedForm;
190 				}
191 
192 				@Override
193 				public void registerEntityKey(EntityKey entityKey) {
194 					this.entityKey = entityKey;
195 				}
196 
197 				@Override
198 				public EntityKey getEntityKey() {
199 					return entityKey;
200 				}
201 
202 				@Override
203 				public void registerHydratedState(Object[] hydratedState) {
204 					this.hydratedState = hydratedState;
205 				}
206 
207 				@Override
208 				public Object[] getHydratedState() {
209 					return hydratedState;
210 				}
211 
212 				@Override
213 				public void registerEntityInstance(Object entityInstance) {
214 					this.entityInstance = entityInstance;
215 				}
216 
217 				@Override
218 				public Object getEntityInstance() {
219 					return entityInstance;
220 				}
221 			};
222 			identifierResolutionContextMap.put( entityReference, context );
223 		}
224 
225 		return context;
226 	}

228 	private void registerNonExists(EntityFetch fetch) {
229 		final EntityType fetchedType = fetch.getFetchedType();
230 		if ( ! fetchedType.isOneToOne() ) {
231 			return;
232 		}
233 
234 		final EntityReferenceProcessingState fetchOwnerState = getOwnerProcessingState( fetch );
235 		if ( fetchOwnerState == null ) {
236 			throw new IllegalStateException( "Could not locate fetch owner state" );
237 		}
238 
239 		final EntityKey ownerEntityKey = fetchOwnerState.getEntityKey();
240 		if ( ownerEntityKey == null ) {
241 			throw new IllegalStateException( "Could not locate fetch owner EntityKey" );
242 		}
243 
244 		session.getPersistenceContextInternal().addNullProperty(
245 				ownerEntityKey,
246 				fetchedType.getPropertyName()
247 		);
248 	}
249 
250 	@Override
251 	public EntityReferenceProcessingState getOwnerProcessingState(Fetch fetch) {
252 		return getProcessingState( fetch.getSource().resolveEntityReference() );
253 	}

255 	@Override
256 	public void registerHydratedEntity(EntityReference entityReference, EntityKey entityKey, Object entityInstance) {
257 		if ( currentRowHydratedEntityRegistrationList == null ) {
258 			currentRowHydratedEntityRegistrationList = new ArrayList<>();
259 		}
260 		currentRowHydratedEntityRegistrationList.add(
261 				new HydratedEntityRegistration(
262 						entityReference,
263 						entityKey,
264 						entityInstance
265 				)
266 		);
267 	}
268 
269 	/**
270 	 * Package-protected
271 	 */
272 	void finishUpRow() {
273 		nRowsRead++;
274 
275 		if ( currentRowHydratedEntityRegistrationList == null ) {
276 			if ( identifierResolutionContextMap != null ) {
277 				identifierResolutionContextMap.clear();
278 			}
279 			return;
280 		}
281 
282 
283 		// managing the running list of registrations ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
284 		final int sizeHint = currentRowHydratedEntityRegistrationList.size();
285 		if ( hydratedEntityRegistrationList == null ) {
286 			hydratedEntityRegistrationList = new ArrayList<>( sizeHint );
287 		}
288 		hydratedEntityRegistrationList.addAll( currentRowHydratedEntityRegistrationList );
289 
290 
291 		// managing the map forms needed for subselect fetch generation ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
292 		if ( hadSubselectFetches ) {
293 			if ( subselectLoadableEntityKeyMap == null ) {
294 				subselectLoadableEntityKeyMap = new HashMap<>();
295 			}
296 			for ( HydratedEntityRegistration registration : currentRowHydratedEntityRegistrationList ) {
297 				Set<EntityKey> entityKeys = subselectLoadableEntityKeyMap.get(
298 						registration.getEntityReference()
299 				);
300 				if ( entityKeys == null ) {
301 					entityKeys = new HashSet<>();
302 					subselectLoadableEntityKeyMap.put( registration.getEntityReference(), entityKeys );
303 				}
304 				entityKeys.add( registration.getKey() );
305 			}
306 		}
307 
308 		// release the currentRowHydratedEntityRegistrationList entries
309 		currentRowHydratedEntityRegistrationList.clear();
310 
311 		identifierResolutionContextMap.clear();
312 	}
313 
314 	public List<HydratedEntityRegistration> getHydratedEntityRegistrationList() {
315 		return hydratedEntityRegistrationList;
316 	}
317 
318 	/**
319 	 * Package-protected
320 	 */
321 	void wrapUp() {
322 		createSubselects();
323 
324 		if ( hydratedEntityRegistrationList != null ) {
325 			hydratedEntityRegistrationList = null;
326 		}
327 
328 		if ( subselectLoadableEntityKeyMap != null ) {
329 			subselectLoadableEntityKeyMap = null;
330 		}
331 	}
332 
333 	private void createSubselects() {
334 		if ( subselectLoadableEntityKeyMap == null || nRowsRead <= 1 ) {
335 			LOG.tracef(
336 					"Skipping create subselects because there are fewer than 2 results, so query by key is more efficient.",
337 					getClass().getName()
338 			);
339 			return; // early return
340 		}
341 
342 		final Map<String, int[]> namedParameterLocMap =
343 				ResultSetProcessorHelper.buildNamedParameterLocMap( queryParameters, namedParameterContext );
344 
345 		final String subselectQueryString = SubselectFetch.createSubselectFetchQueryFragment( queryParameters );
346 		for ( Map.Entry<EntityReference, Set<EntityKey>> entry : subselectLoadableEntityKeyMap.entrySet() ) {
347 			if ( ! entry.getKey().getEntityPersister().hasSubselectLoadableCollections() ) {
348 				continue;
349 			}
350 
351 			SubselectFetch subselectFetch = new SubselectFetch(
352 					subselectQueryString,
353 					aliasResolutionContext.resolveSqlTableAliasFromQuerySpaceUid( entry.getKey().getQuerySpaceUid() ),
354 					(Loadable) entry.getKey().getEntityPersister(),
355 					queryParameters,
356 					entry.getValue(),
357 					namedParameterLocMap
358 			);
359 
360 			final PersistenceContext persistenceContext = session.getPersistenceContextInternal();
361 			for ( EntityKey key : entry.getValue() ) {
362 				persistenceContext.getBatchFetchQueue().addSubselect( key, subselectFetch );
363 			}
364 		}
365 	}
