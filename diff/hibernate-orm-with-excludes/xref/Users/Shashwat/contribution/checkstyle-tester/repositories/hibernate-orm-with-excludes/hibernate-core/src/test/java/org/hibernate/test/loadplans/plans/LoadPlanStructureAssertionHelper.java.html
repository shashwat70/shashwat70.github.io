9   import org.hibernate.LockMode;
10  import org.hibernate.LockOptions;

13  import org.hibernate.loader.JoinWalker;
14  import org.hibernate.loader.entity.EntityJoinWalker;

18  import org.hibernate.loader.plan.exec.query.internal.QueryBuildingParametersImpl;

24  /**
25   * Perform assertions based on a LoadPlan, specifically against the outputs/expectations of the legacy Loader approach.
26   * <p/>
27   * Mainly this is intended to be a transitory set of help since it is expected that Loader will go away replaced by
28   * LoadPlans, QueryBuilders and ResultSetProcessors.  For now I want to make sure that the outputs (e.g., the SQL,
29   * the extraction aliases) are the same given the same input.  That makes sure we have the best possibility of success
30   * in designing and implementing the "replacement parts".
31   *
32   * @author Steve Ebersole
33   */
34  public class LoadPlanStructureAssertionHelper {
35  	/**
36  	 * Singleton access to the helper
37  	 */
38  	public static final LoadPlanStructureAssertionHelper INSTANCE = new LoadPlanStructureAssertionHelper();
39  
40  	/**
41  	 * Performs a basic comparison.  Builds a LoadPlan for the given persister and compares it against the
42  	 * expectations according to the Loader/Walker corollary.
43  	 *
44  	 * @param sf The SessionFactory
45  	 * @param persister The entity persister for which to build a LoadPlan and compare against the Loader/Walker
46  	 * expectations.
47  	 */
48  	public void performBasicComparison(SessionFactoryImplementor sf, OuterJoinLoadable persister) {
49  		// todo : allow these to be passed in by tests?
50  		final LoadQueryInfluencers influencers = LoadQueryInfluencers.NONE;
51  		final LockMode lockMode = LockMode.NONE;
52  		final int batchSize = 1;
53  
54  		// legacy Loader-based contracts...
55  		final EntityJoinWalker walker = new EntityJoinWalker(
56  				persister,
57  				persister.getKeyColumnNames(),
58  				batchSize,
59  				lockMode,
60  				sf,
61  				influencers
62  		);
63  //		final EntityLoader loader = new EntityLoader( persister, lockMode, sf, influencers );
64  
65  		LoadPlan plan = buildLoadPlan( sf, persister, influencers, lockMode );
66  		LoadQueryDetails details = BatchingLoadQueryDetailsFactory.INSTANCE.makeEntityLoadQueryDetails(
67  				plan,
68  				persister.getKeyColumnNames(),
69  				new QueryBuildingParametersImpl(
70  						influencers,
71  						batchSize,
72  						lockMode,
73  						null
74  
75  				),
76  				sf
77  		);
78  
79  		compare( walker, details );
80  	}
81  
82  	public LoadPlan buildLoadPlan(
83  			SessionFactoryImplementor sf,
84  			OuterJoinLoadable persister,
85  			LoadQueryInfluencers influencers,
86  			LockMode lockMode) {
87  		FetchStyleLoadPlanBuildingAssociationVisitationStrategy strategy = new FetchStyleLoadPlanBuildingAssociationVisitationStrategy(
88  				sf,
89  				influencers,
90  				lockMode
91  				);
92  		return MetamodelDrivenLoadPlanBuilder.buildRootEntityLoadPlan( strategy, persister );
93  	}
94  
95  	public LoadPlan buildLoadPlan(SessionFactoryImplementor sf, OuterJoinLoadable persister) {
96  		return buildLoadPlan( sf, persister, LoadQueryInfluencers.NONE, LockMode.NONE );
97  	}
98  
99  	private void compare(JoinWalker walker, LoadQueryDetails details) {
100 		System.out.println( "------ SQL -----------------------------------------------------------------" );
101 		System.out.println( "WALKER    : " + walker.getSQLString() );
102 		System.out.println( "LOAD-PLAN : " + details.getSqlStatement() );
103 		System.out.println( "----------------------------------------------------------------------------" );
104 		System.out.println( );
105 		System.out.println( "------ SUFFIXES ------------------------------------------------------------" );
106 		System.out.println( "WALKER    : " + String.join( ", ",  walker.getSuffixes() ) + " : "
107 									+ String.join( ", ", walker.getCollectionSuffixes() ) );
108 		System.out.println( "----------------------------------------------------------------------------" );
109 		System.out.println( );
110 	}
111 }
