338 	class DuplicateSecondaryTableException extends HibernateException {
339 		private final Identifier tableName;
340 
341 		public DuplicateSecondaryTableException(Identifier tableName) {
342 			super(
343 					String.format(
344 							Locale.ENGLISH,
345 							"Table with that name [%s] already associated with entity",
346 							tableName.render()
347 					)
348 			);
349 			this.tableName = tableName;
350 		}
351 	}
