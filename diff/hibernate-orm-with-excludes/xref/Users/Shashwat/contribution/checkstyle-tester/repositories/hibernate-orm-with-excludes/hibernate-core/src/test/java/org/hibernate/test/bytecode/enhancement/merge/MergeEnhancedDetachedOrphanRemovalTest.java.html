7   package org.hibernate.test.bytecode.enhancement.merge;
8   
9   import org.junit.Test;
10  import org.junit.runner.RunWith;
11  
12  import org.hibernate.testing.TestForIssue;
13  import org.hibernate.testing.bytecode.enhancement.BytecodeEnhancerRunner;
14  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
15  
16  import static org.hibernate.testing.transaction.TransactionUtil.doInHibernate;
17  import static org.junit.Assert.assertNotSame;
18  
19  /**
20   * @author Chris Cranford
21   */
22  @TestForIssue(jiraKey = "HHH-12592")
23  @RunWith(BytecodeEnhancerRunner.class)
24  public class MergeEnhancedDetachedOrphanRemovalTest extends BaseCoreFunctionalTestCase {
25  	@Override
26  	protected Class<?>[] getAnnotatedClasses() {
27  		return new Class<?>[] { Leaf.class, Root.class };
28  	}
29  
30  	@Test
31  	public void testMergeDetachedOrphanRemoval() {
32  		final Root entity = doInHibernate( this::sessionFactory, session -> {
33  			Root root = new Root();
34  			root.setName( "new" );
35  			session.save( root );
36  			return root;
37  		} );
38  
39  		doInHibernate( this::sessionFactory, session -> {
40  			entity.setName( "updated" );
41  			Root entityMerged = (Root) session.merge( entity );
42  			assertNotSame( entity, entityMerged );
43  			assertNotSame( entity.getLeaves(), entityMerged.getLeaves() );
44  		} );
45  	}
46  
47  	@Test
48  	public void testMergeDetachedNonEmptyCollection() {
49  		final Root entity = doInHibernate( this::sessionFactory, session -> {
50  			Root root = new Root();
51  			root.setName( "new" );
52  			Leaf leaf = new Leaf();
53  			leaf.setRoot( root );
54  			root.getLeaves().add( leaf );
55  			session.save( root );
56  			return root;
57  		} );
58  
59  		doInHibernate( this::sessionFactory, session -> {
60  			entity.setName( "updated" );
61  			Root entityMerged = (Root) session.merge( entity );
62  			assertNotSame( entity, entityMerged );
63  			assertNotSame( entity.getLeaves(), entityMerged.getLeaves() );
64  		} );
65  	}
66  }
