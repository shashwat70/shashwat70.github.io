65  	@Test
66  	@TestForIssue( jiraKey = "HHH-8637" )
67  	public void testFilteringDiscriminatorSubclasses() {
68  		Session s = openSession();
69  		s.beginTransaction();
70  		DiscriminatorEntity root = new DiscriminatorEntity( 1, "root" );
71  		s.save( root );
72  		DiscriminatorEntitySubclass child = new DiscriminatorEntitySubclass( 2, "child", root );
73  		s.save( child );
74  		s.getTransaction().commit();
75  		s.close();
76  
77  		s = openSession();
78  		s.beginTransaction();
79  
80  		// in select clause
81  		List result = s.createQuery( "select e from DiscriminatorEntity e" ).list();
82  		assertEquals( 2, result.size() );
83  		result = s.createQuery( "select treat (e as DiscriminatorEntitySubclass) from DiscriminatorEntity e" ).list();
84  		assertEquals( 1, result.size() );
85  		result = s.createQuery( "select treat (e as DiscriminatorEntitySubSubclass) from DiscriminatorEntity e" ).list();
86  		assertEquals( 0, result.size() );
87  
88  		// in join
89  		result = s.createQuery( "from DiscriminatorEntity e inner join e.other" ).list();
90  		assertEquals( 1, result.size() );
91  		result = s.createQuery( "from DiscriminatorEntity e inner join treat (e.other as DiscriminatorEntitySubclass)" ).list();
92  		assertEquals( 0, result.size() );
93  		result = s.createQuery( "from DiscriminatorEntity e inner join treat (e.other as DiscriminatorEntitySubSubclass)" ).list();
94  		assertEquals( 0, result.size() );
95  
96  		s.close();
97  
98  		s = openSession();
99  		s.beginTransaction();
100 		s.delete( root );
101 		s.delete( child );
102 		s.getTransaction().commit();
103 		s.close();
104 	}

106 	@Test
107 	@TestForIssue( jiraKey = "HHH-8637" )
108 	public void testFilteringJoinedSubclasses() {
109 		Session s = openSession();
110 		s.beginTransaction();
111 		JoinedEntity root = new JoinedEntity( 1, "root" );
112 		s.save( root );
113 		JoinedEntitySubclass child = new JoinedEntitySubclass( 2, "child", root );
114 		s.save( child );
115 		s.getTransaction().commit();
116 		s.close();
117 
118 		s = openSession();
119 		s.beginTransaction();
120 
121 		// in the select clause which causes an implicit inclusion of subclass joins, the test here makes sure that
122 		// the TREAT-AS effects the join-type used.
123 		List result = s.createQuery( "select e from JoinedEntity e" ).list();
124 		assertEquals( 2, result.size() );
125 		result = s.createQuery( "select treat (e as JoinedEntitySubclass) from JoinedEntity e" ).list();
126 		assertEquals( 1, result.size() );
127 		result = s.createQuery( "select treat (e as JoinedEntitySubSubclass) from JoinedEntity e" ).list();
128 		assertEquals( 0, result.size() );
129 
130 		// in join
131 		result = s.createQuery( "from JoinedEntity e inner join e.other" ).list();
132 		assertEquals( 1, result.size() );
133 		result = s.createQuery( "from JoinedEntity e inner join treat (e.other as JoinedEntitySubclass)" ).list();
134 		assertEquals( 0, result.size() );
135 		result = s.createQuery( "from JoinedEntity e inner join treat (e.other as JoinedEntitySubSubclass)" ).list();
136 		assertEquals( 0, result.size() );
137 
138 		s.close();
139 
140 		s = openSession();
141 		s.beginTransaction();
142 		s.delete( child );
143 		s.delete( root );
144 		s.getTransaction().commit();
145 		s.close();
146 	}

148 	@Test
149 	@TestForIssue( jiraKey = "HHH-9862" )
150 	@FailureExpected( jiraKey = "HHH-9862" )
151 	public void testRestrictionsOnJoinedSubclasses() {
152 		Session s = openSession();
153 		s.beginTransaction();
154 		JoinedEntity root = new JoinedEntity( 1, "root" );
155 		s.save( root );
156 		JoinedEntitySubclass child1 = new JoinedEntitySubclass( 2, "child1", root );
157 		s.save( child1 );
158 		JoinedEntitySubclass2 child2 = new JoinedEntitySubclass2( 3, "child2", root );
159 		s.save( child2 );
160 		s.getTransaction().commit();
161 		s.close();
162 
163 		s = openSession();
164 		s.beginTransaction();
165 
166 		List result = s.createQuery( "select e from JoinedEntity e where treat (e as JoinedEntitySubclass ).name = 'child1'" ).list();
167 		assertEquals( 1, result.size() );
168 		assertTrue( JoinedEntitySubclass.class.isInstance( result.get( 0 ) ) );
169 
170 		result = s.createQuery( "select e from JoinedEntity e where treat (e as JoinedEntitySubclass2 ).name = 'child1'" ).list();
171 		assertEquals( 0, result.size() );
172 
173 		result = s.createQuery( "select e from JoinedEntity e where treat (e as JoinedEntitySubclass2 ).name = 'child2'" ).list();
174 		assertEquals( 1, result.size() );
175 		assertTrue( JoinedEntitySubclass2.class.isInstance( result.get( 0 ) ) );
176 
177 		result = s.createQuery( "select e from JoinedEntity e where treat (e as JoinedEntitySubclass ).name = 'child1' or treat (e as JoinedEntitySubclass2 ).name = 'child2'" ).list();
178 		assertEquals( 2, result.size() );
179 
180 		s.close();
181 
182 		s = openSession();
183 		s.beginTransaction();
184 		s.delete( child1 );
185 		s.delete( child2 );
186 		s.delete( root );
187 		s.getTransaction().commit();
188 		s.close();
189 	}
