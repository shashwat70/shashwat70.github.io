25  package org.hibernate.proxy;
26  import java.io.Serializable;
27  
28  import org.hibernate.HibernateException;
29  import org.hibernate.engine.spi.SessionImplementor;
30  
31  /**
32   * Handles fetching of the underlying entity for a proxy
33   *
34   * @author Gavin King
35   * @author Steve Ebersole
36   */
37  public interface LazyInitializer {
38  	/**
39  	 * Initialize the proxy, fetching the target entity if necessary.
40  	 *
41  	 * @throws HibernateException Indicates a problem initializing the proxy.
42  	 */
43  	public void initialize() throws HibernateException;
44  
45  	/**
46  	 * Retrieve the identifier value for the entity our owning proxy represents.
47  	 *
48  	 * @return The identifier value.
49  	 */
50  	public Serializable getIdentifier();
51  
52  	/**
53  	 * Set the identifier value for the entity our owning proxy represents.
54  	 *
55  	 * @param id The identifier value.
56  	 */
57  	public void setIdentifier(Serializable id);
58  
59  	/**
60  	 * The entity-name of the entity our owning proxy represents.
61  	 *
62  	 * @return The entity-name.
63  	 */
64  	public String getEntityName();
65  
66  	/**
67  	 * Get the actual class of the entity.  Generally, {@link #getEntityName()} should be used instead.
68  	 *
69  	 * @return The actual entity class.
70  	 */
71  	public Class getPersistentClass();
72  
73  	/**
74  	 * Is the proxy uninitialzed?
75  	 *
76  	 * @return True if uninitialized; false otherwise.
77  	 */
78  	public boolean isUninitialized();
79  
80  	/**
81  	 * Return the underlying persistent object, initializing if necessary
82  	 *
83  	 * @return The underlying target entity.
84  	 */
85  	public Object getImplementation();
86  
87  	/**
88  	 * Return the underlying persistent object in the given session, or null if not contained in this session's
89  	 * persistence context.
90  	 *
91  	 * @param session The session to check
92  	 *
93  	 * @return The target, or null.
94  	 *
95  	 * @throws HibernateException Indicates problem locating the target.
96  	 */
97  	public Object getImplementation(SessionImplementor session) throws HibernateException;
98  
99  	/**
100 	 * Initialize the proxy manually by injecting its target.
101 	 *
102 	 * @param target The proxy target (the actual entity being proxied).
103 	 */
104 	public void setImplementation(Object target);
105 
106 	/**
107 	 * Is the proxy's read-only/modifiable setting available?
108 	 * @return true, if the setting is available
109 	 *         false, if the proxy is detached or its associated session is closed
110 	 */
111 	public boolean isReadOnlySettingAvailable();
112 
113 	/**
114 	 * Is the proxy read-only?.
115 	 *
116 	 * The read-only/modifiable setting is not available when the proxy is
117 	 * detached or its associated session is closed.
118 	 *
119 	 * To check if the read-only/modifiable setting is available:
120 	 *
121 	 * @return true, if this proxy is read-only; false, otherwise
122 	 * @throws org.hibernate.TransientObjectException if the proxy is detached (getSession() == null)
123 	 * @throws org.hibernate.SessionException if the proxy is associated with a sesssion that is closed
124 	 *
125 	 * @see org.hibernate.proxy.LazyInitializer#isReadOnlySettingAvailable()
126 	 * @see org.hibernate.Session#isReadOnly(Object entityOrProxy)
127 	 */
128 	public boolean isReadOnly();
129 
130 	/**
131 	 * Set an associated modifiable proxy to read-only mode, or a read-only
132 	 * proxy to modifiable mode. If the proxy is currently initialized, its
133 	 * implementation will be set to the same mode; otherwise, when the
134 	 * proxy is initialized, its implementation will have the same read-only/
135 	 * modifiable setting as the proxy. In read-only mode, no snapshot is
136 	 * maintained and the instance is never dirty checked.
137 	 *
138 	 * If the associated proxy already has the specified read-only/modifiable
139 	 * setting, then this method does nothing.
140 	 *
141 	 * @param readOnly if true, the associated proxy is made read-only;
142 	 *                  if false, the associated proxy is made modifiable.
143 	 * @throws org.hibernate.TransientObjectException if the proxy is not association with a session
144 	 * @throws org.hibernate.SessionException if the proxy is associated with a session that is closed
145 	 * 
146 	 * @see org.hibernate.Session#setReadOnly(Object entityOrProxy, boolean readOnly)
147 	 */
148 	public void setReadOnly(boolean readOnly);
149 
150 	/**
151 	 * Get the session to which this proxy is associated, or null if it is not attached.
152 	 *
153 	 * @return The associated session.
154 	 */
155 	public SessionImplementor getSession();
156 
157 	/**
158 	 * Associate the proxy with the given session.
159 	 * <p/>
160 	 * Care should be given to make certain that the proxy is added to the session's persistence context as well
161 	 * to maintain the symetry of the association.  That must be done seperately as this method simply sets an
162 	 * internal reference.  We do also check that if there is already an associated session that the proxy
163 	 * reference was removed from that previous session's persistence contet.
164 	 *
165 	 * @param session The session
166 	 * @throws HibernateException Indicates that the proxy was still contained in the persistence context of the
167 	 * "previous session".
168 	 */
169 	public void setSession(SessionImplementor session) throws HibernateException;
170 
171 	/**
172 	 * Unset this initializer's reference to session.  It is assumed that the caller is also taking care or
173 	 * cleaning up the owning proxy's reference in the persistence context.
174 	 * <p/>
175 	 * Generally speaking this is intended to be called only during {@link org.hibernate.Session#evict} and
176 	 * {@link org.hibernate.Session#clear} processing; most other use-cases should call {@link #setSession} instead.
177 	 */
178 	public void unsetSession();
179 	
180 	public void setUnwrap(boolean unwrap);
181 	public boolean isUnwrap();
182 }
