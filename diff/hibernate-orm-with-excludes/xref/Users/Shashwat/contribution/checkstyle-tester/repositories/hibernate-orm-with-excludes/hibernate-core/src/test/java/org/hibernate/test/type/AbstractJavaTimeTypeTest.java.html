120 	@TestForIssue(jiraKey = "HHH-13266")
121 	public void writeThenRead() {
122 		withDefaultTimeZone( () -> {
123 			inTransaction( session -> {
124 				session.persist( createEntityForHibernateWrite( 1 ) );
125 			} );
126 			inTransaction( session -> {
127 				T read = getActualPropertyValue( session.find( getEntityType(), 1 ) );
128 				assertEquals(
129 						"Writing then reading a value should return the original value",
130 						getExpectedPropertyValueAfterHibernateRead(), read
131 				);
132 			} );
133 		} );

136 	@Test
137 	@TestForIssue(jiraKey = "HHH-13266")
138 	public void writeThenNativeRead() {
139 		assumeNoJdbcTimeZone();
140 
141 		withDefaultTimeZone( () -> {
142 			inTransaction( session -> {
143 				session.persist( createEntityForHibernateWrite( 1 ) );
144 			} );
145 			inTransaction( session -> {
146 				session.doWork( connection -> {
147 					final PreparedStatement statement = connection.prepareStatement(
148 							"SELECT " + PROPERTY_COLUMN_NAME + " FROM " + ENTITY_NAME + " WHERE " + ID_COLUMN_NAME + " = ?"
149 					);
150 					statement.setInt( 1, 1 );
151 					statement.execute();
152 					final ResultSet resultSet = statement.getResultSet();
153 					resultSet.next();
154 					Object nativeRead = getActualJdbcValue( resultSet, 1 );
155 					assertEquals(
156 							"Values written by Hibernate ORM should match the original value (same day, hour, ...)",
157 							getExpectedJdbcValueAfterHibernateWrite(),
158 							nativeRead
159 					);
160 				} );
161 			} );
162 		} );
163 	}

165 	@Test
166 	@TestForIssue(jiraKey = "HHH-13266")
167 	public void nativeWriteThenRead() {
168 		assumeNoJdbcTimeZone();
169 
170 		withDefaultTimeZone( () -> {
171 			inTransaction( session -> {
172 				session.doWork( connection -> {
173 					final PreparedStatement statement = connection.prepareStatement(
174 							"INSERT INTO " + ENTITY_NAME + " (" + ID_COLUMN_NAME + ", " + PROPERTY_COLUMN_NAME + ") "
175 							+ " VALUES ( ? , ? )"
176 					);
177 					statement.setInt( 1, 1 );
178 					setJdbcValueForNonHibernateWrite( statement, 2 );
179 					statement.execute();
180 				} );
181 			} );
182 			inTransaction( session -> {
183 				T read = getActualPropertyValue( session.find( getEntityType(), 1 ) );
184 				assertEquals(
185 						"Values written without Hibernate ORM should be read correctly by Hibernate ORM",
186 						getExpectedPropertyValueAfterHibernateRead(), read
187 				);
188 			} );
189 		} );
190 	}

192 	protected final void withDefaultTimeZone(Runnable runnable) {
193 		TimeZone timeZoneBefore = TimeZone.getDefault();
194 		TimeZone.setDefault( toTimeZone( env.defaultJvmTimeZone ) );
195 		/*
196 		 * Run the code in a new thread, because some libraries (looking at you, h2 JDBC driver)
197 		 * cache data dependent on the default timezone in thread local variables,
198 		 * and we want this data to be reinitialized with the new default time zone.
199 		 */
200 		try {
201 			ExecutorService executor = Executors.newSingleThreadExecutor();
202 			Future<?> future = executor.submit( runnable );
203 			executor.shutdown();
204 			future.get();
205 		}
206 		catch (InterruptedException e) {
207 			throw new IllegalStateException( "Interrupted while testing", e );
208 		}
209 		catch (ExecutionException e) {
210 			Throwable cause = e.getCause();
211 			if ( cause instanceof RuntimeException ) {
212 				throw (RuntimeException) cause;
213 			}
214 			else if ( cause instanceof Error ) {
215 				throw (Error) cause;
216 			}
217 			else {
218 				throw new IllegalStateException( "Unexpected exception while testing", cause );
219 			}
220 		}
221 		finally {
222 			TimeZone.setDefault( timeZoneBefore );
223 		}
224 	}

226 	private static TimeZone toTimeZone(ZoneId zoneId) {
227 		String idString = zoneId.getId();
228 		if ( idString.startsWith( "UTC+" ) || idString.startsWith( "UTC-" ) ) {
229 			// Apparently TimeZone doesn't understand UTC+XXX nor UTC-XXX
230 			// Using GMT+XXX or GMT-XXX as a fallback
231 			idString = "GMT" + idString.substring( "UTC".length() );
232 		}
233 
234 		TimeZone result = TimeZone.getTimeZone( idString );
235 		if ( !idString.equals( result.getID() ) ) {
236 			// If the timezone is not understood, getTimeZone returns GMT and the condition above is true
237 			throw new IllegalStateException( "Attempting to test an unsupported timezone: " + zoneId );
238 		}
239 
240 		return result;
241 	}

247 	protected void assumeNoJdbcTimeZone() {
248 		Assume.assumeTrue(
249 				"Tests with native read/writes are only relevant when not using " + AvailableSettings.JDBC_TIME_ZONE
250 						+ ", because the expectations do not take that time zone into account."
251 						+ " When this property is set, we only test that a write by Hibernate followed by "

253 				env.hibernateJdbcTimeZone == null
254 		);

257 	protected static abstract class AbstractParametersBuilder<S extends AbstractParametersBuilder<S>> {
258 
259 		private final Dialect dialect;
260 
261 		private final List<Object[]> result = new ArrayList<>();
262 
263 		private final List<Class<? extends AbstractRemappingH2Dialect>> remappingDialectClasses = new ArrayList<>();
264 
265 		private ZoneId forcedJdbcTimeZone = null;
266 
267 		protected AbstractParametersBuilder() {
268 			dialect = determineDialect();
269 			remappingDialectClasses.add( null ); // Always test without remapping
270 		}
271 
272 		public S skippedForDialects(List<Class<?>> dialectClasses, Consumer<S> skippedIfDialectMatchesClasses) {
273 			boolean skip = false;
274 			for ( Class<?> dialectClass : dialectClasses ) {
275 				if ( dialectClass.isInstance( dialect ) ) {
276 					skip = true;
277 				}
278 			}
279 			if ( !skip ) {
280 				skippedIfDialectMatchesClasses.accept( thisAsS() );
281 			}
282 			return thisAsS();
283 		}
284 
285 		public S withForcedJdbcTimezone(String zoneIdString, Consumer<S> contributor) {
286 			ZoneId zoneId = ZoneId.of( zoneIdString );
287 			this.forcedJdbcTimeZone = zoneId;
288 			try {
289 				contributor.accept( thisAsS() );
290 			}
291 			finally {
292 				this.forcedJdbcTimeZone = null;
293 			}
294 			return thisAsS();
295 		}
296 
297 		@SafeVarargs
298 		public final S alsoTestRemappingsWithH2(Class<? extends AbstractRemappingH2Dialect> ... dialectClasses) {
299 			if ( dialect instanceof H2Dialect ) {
300 				// Only test remappings with H2
301 				Collections.addAll( remappingDialectClasses, dialectClasses );
302 			}
303 			return thisAsS();
304 		}
305 
306 		protected final boolean isNanosecondPrecisionSupported() {
307 			// Most databases apparently don't support nanosecond precision correctly
308 			return dialect instanceof H2Dialect;
309 		}
310 
311 		protected final S add(ZoneId defaultJvmTimeZone, Object ... subClassParameters) {
312 			for ( Class<? extends AbstractRemappingH2Dialect> remappingDialectClass : remappingDialectClasses ) {
313 				List<Object> parameters = new ArrayList<>();
314 				parameters.add(
315 						new EnvironmentParameters(
316 								defaultJvmTimeZone,
317 								forcedJdbcTimeZone,
318 								remappingDialectClass
319 						)
320 				);
321 				Collections.addAll( parameters, subClassParameters );
322 				result.add( parameters.toArray() );
323 			}
324 			if ( forcedJdbcTimeZone == null ) {
325 				for ( ZoneId hibernateJdbcTimeZone : getHibernateJdbcTimeZonesToTest() ) {
326 					List<Object> parameters = new ArrayList<>();
327 					parameters.add(
328 							new EnvironmentParameters(
329 									defaultJvmTimeZone,
330 									hibernateJdbcTimeZone,
331 									null
332 							)
333 					);
334 					Collections.addAll( parameters, subClassParameters );
335 					result.add( parameters.toArray() );
336 				}
337 			}
338 			return thisAsS();
339 		}
340 
341 		protected Iterable<? extends ZoneId> getHibernateJdbcTimeZonesToTest() {
342 			return Arrays.asList( ZONE_GMT, ZONE_OSLO );
343 		}
344 
345 		private S thisAsS() {
346 			return (S) this;
347 		}
348 
349 		public List<Object[]> build() {
350 			return result;
351 		}

353 	}

355 	protected final static class EnvironmentParameters {
356 
357 		/*
358 		 * The default timezone affects conversions done using java.util,
359 		 * which is why we take it into account even with timezone-independent types such as Instant.
360 		 */
361 		private final ZoneId defaultJvmTimeZone;
362 
363 		/**
364 		 * The Hibernate setting, {@link AvailableSettings#JDBC_TIME_ZONE},
365 		 * may affect a lot of time-related types,
366 		 * which is why we take it into account even with timezone-independent types such as Instant.
367 		 */
368 		private final ZoneId hibernateJdbcTimeZone;
369 
370 		private final Class<? extends AbstractRemappingH2Dialect> remappingDialectClass;
371 
372 		private EnvironmentParameters(ZoneId defaultJvmTimeZone, ZoneId hibernateJdbcTimeZone,
373 				Class<? extends AbstractRemappingH2Dialect> remappingDialectClass) {
374 			this.defaultJvmTimeZone = defaultJvmTimeZone;
375 			this.hibernateJdbcTimeZone = hibernateJdbcTimeZone;
376 			this.remappingDialectClass = remappingDialectClass;
377 		}
378 
379 		@Override
380 		public String toString() {
381 			return String.format(
382 					"[JVM TZ: %s, JDBC TZ: %s, remapping dialect: %s]",
383 					defaultJvmTimeZone,
384 					hibernateJdbcTimeZone,
385 					remappingDialectClass == null ? null : remappingDialectClass.getSimpleName()
386 			);
387 		}
388 	}

390 	protected static class AbstractRemappingH2Dialect extends H2Dialect {
391 		private final int overriddenSqlTypeCode;
392 		private final SqlTypeDescriptor overriddenSqlTypeDescriptor;
393 
394 		public AbstractRemappingH2Dialect(int overriddenSqlTypeCode, SqlTypeDescriptor overriddenSqlTypeDescriptor) {
395 			this.overriddenSqlTypeCode = overriddenSqlTypeCode;
396 			this.overriddenSqlTypeDescriptor = overriddenSqlTypeDescriptor;
397 		}
398 
399 		@Override
400 		protected SqlTypeDescriptor getSqlTypeDescriptorOverride(int sqlCode) {
401 			if ( overriddenSqlTypeCode == sqlCode ) {
402 				return overriddenSqlTypeDescriptor;
403 			}
404 			else {
405 				return super.getSqlTypeDescriptorOverride( sqlCode );
406 			}
407 		}
