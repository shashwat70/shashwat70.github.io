7   package org.hibernate.dialect.identity;
8   
9   import org.hibernate.MappingException;
10  
11  /**
12   * @author Andrea Boriero
13   */
14  public class Chache71IdentityColumnSupport extends IdentityColumnSupportImpl {
15  	@Override
16  	public boolean supportsIdentityColumns() {
17  		return true;
18  	}
19  
20  	@Override
21  	public boolean hasDataTypeInIdentityColumn() {
22  		// Whether this dialect has an Identity clause added to the data type or a completely seperate identity
23  		// data type
24  		return true;
25  	}
26  
27  	@Override
28  	public String getIdentityColumnString(int type) throws MappingException {
29  		// The keyword used to specify an identity column, if identity column key generation is supported.
30  		return "identity";
31  	}
32  
33  	@Override
34  	public String getIdentitySelectString(String table, String column, int type) {
35  		return "SELECT LAST_IDENTITY() FROM %TSQL_sys.snf";
36  	}
37  }
