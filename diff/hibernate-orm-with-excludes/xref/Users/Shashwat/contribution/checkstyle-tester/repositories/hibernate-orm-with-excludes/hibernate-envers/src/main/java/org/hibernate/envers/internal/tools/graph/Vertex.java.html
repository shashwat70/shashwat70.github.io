7   package org.hibernate.envers.internal.tools.graph;
8   
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  /**
13   * A graph vertex - stores its representation, neighbours, start and end time in (D|B)FS.
14   *
15   * @author Adam Warski (adam at warski dot org)
16   */
17  public class Vertex<R> {
18  	private final R representation;
19  	private final List<Vertex<R>> neighbours;
20  
21  	private int startTime;
22  	private int endTime;
23  
24  	public Vertex(R representation) {
25  		this.representation = representation;
26  		this.neighbours = new ArrayList<>();
27  		this.startTime = 0;
28  		this.endTime = 0;
29  	}
30  
31  	public R getRepresentation() {
32  		return representation;
33  	}
34  
35  	public List<Vertex<R>> getNeighbours() {
36  		return neighbours;
37  	}
38  
39  	public void addNeighbour(Vertex<R> n) {
40  		neighbours.add( n );
41  	}
42  
43  	public int getStartTime() {
44  		return startTime;
45  	}
46  
47  	public void setStartTime(int startTime) {
48  		this.startTime = startTime;
49  	}
50  
51  	public int getEndTime() {
52  		return endTime;
53  	}
54  
55  	public void setEndTime(int endTime) {
56  		this.endTime = endTime;
57  	}
58  }
