7   package org.hibernate.envers.internal.revisioninfo;
8   
9   import java.util.Set;
10  
11  import org.hibernate.envers.internal.entities.PropertyData;
12  import org.hibernate.envers.internal.tools.ReflectionTools;
13  import org.hibernate.property.access.spi.Getter;
14  import org.hibernate.service.ServiceRegistry;
15  
16  /**
17   * Returns modified entity names from a persisted revision info entity.
18   *
19   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
20   */
21  public class ModifiedEntityNamesReader {
22  	private final Getter modifiedEntityNamesGetter;
23  
24  	public ModifiedEntityNamesReader(
25  			Class<?> revisionInfoClass,
26  			PropertyData modifiedEntityNamesData,
27  			ServiceRegistry serviceRegistry) {
28  		modifiedEntityNamesGetter = ReflectionTools.getGetter( revisionInfoClass, modifiedEntityNamesData, serviceRegistry );
29  	}
30  
31  	@SuppressWarnings({"unchecked"})
32  	public Set<String> getModifiedEntityNames(Object revisionEntity) {
33  		return (Set<String>) modifiedEntityNamesGetter.get( revisionEntity );
34  	}
35  }
