48  	/**
49  	 * Build a ResultSetMappingDefinition given a containing element for the "return-XXX" elements.
50  	 * <p/>
51  	 * This form is used for ResultSet mappings defined outside the context of any specific entity.
52  	 * For {@code hbm.xml} this means at the root of the document.  For annotations, this means at
53  	 * the package level.
54  	 *
55  	 * @param resultSetMappingSource The XML data as a JAXB binding
56  	 * @param context The mapping state
57  	 *
58  	 * @return The ResultSet mapping descriptor
59  	 */
60  	public static ResultSetMappingDefinition bind(
61  			ResultSetMappingBindingDefinition resultSetMappingSource,
62  			HbmLocalMetadataBuildingContext context) {
63  		if ( resultSetMappingSource.getName() == null ) {
64  			throw new MappingException(
65  					"ResultSet mapping did not specify name",
66  					context.getOrigin()
67  			);
68  		}
69  
70  		final ResultSetMappingDefinition binding = new ResultSetMappingDefinition( resultSetMappingSource.getName() );
71  		bind( resultSetMappingSource, binding, context );
72  		return binding;
73  	}
74  
75  	/**
76  	 * Build a ResultSetMappingDefinition given a containing element for the "return-XXX" elements
77  	 *
78  	 * @param resultSetMappingSource The XML data as a JAXB binding
79  	 * @param context The mapping state
80  	 * @param prefix A prefix to apply to named ResultSet mapping; this is either {@code null} for
81  	 * ResultSet mappings defined outside of any entity, or the name of the containing entity
82  	 * if defined within the context of an entity
83  	 *
84  	 * @return The ResultSet mapping descriptor
85  	 */
86  	public static ResultSetMappingDefinition bind(
87  			ResultSetMappingBindingDefinition resultSetMappingSource,
88  			HbmLocalMetadataBuildingContext context,
89  			String prefix) {
90  		if ( StringHelper.isEmpty( prefix ) ) {
91  			throw new AssertionFailure( "Passed prefix was null; perhaps you meant to call the alternate #bind form?" );
92  		}
93  
94  		final String resultSetName = prefix + '.' + resultSetMappingSource.getName();
95  		final ResultSetMappingDefinition binding = new ResultSetMappingDefinition( resultSetName );
96  		bind( resultSetMappingSource, binding, context );
97  		return binding;
98  	}

100 	private static void bind(
101 			ResultSetMappingBindingDefinition resultSetMappingSource,
102 			ResultSetMappingDefinition binding,
103 			HbmLocalMetadataBuildingContext context) {
104 
105 		int cnt = 0;
106 
107 		for ( Object valueMappingSource : resultSetMappingSource.getValueMappingSources() ) {
108 			if ( JaxbHbmNativeQueryReturnType.class.isInstance( valueMappingSource ) ) {
109 				binding.addQueryReturn(
110 						extractReturnDescription( (JaxbHbmNativeQueryReturnType) valueMappingSource, context, cnt++ )
111 				);
112 			}
113 			else if ( JaxbHbmNativeQueryCollectionLoadReturnType.class.isInstance( valueMappingSource ) ) {
114 				binding.addQueryReturn(
115 						extractReturnDescription( (JaxbHbmNativeQueryCollectionLoadReturnType) valueMappingSource, context, cnt++ )
116 				);
117 			}
118 			else if ( JaxbHbmNativeQueryJoinReturnType.class.isInstance( valueMappingSource ) ) {
119 				binding.addQueryReturn(
120 						extractReturnDescription( (JaxbHbmNativeQueryJoinReturnType) valueMappingSource, context, cnt++ )
121 				);
122 			}
123 			else if ( JaxbHbmNativeQueryScalarReturnType.class.isInstance( valueMappingSource ) ) {
124 				binding.addQueryReturn(
125 						extractReturnDescription( (JaxbHbmNativeQueryScalarReturnType) valueMappingSource, context )
126 				);
127 			}
128 		}
129 	}

139 	public static NativeSQLQueryScalarReturn extractReturnDescription(
140 			JaxbHbmNativeQueryScalarReturnType rtnSource,
141 			HbmLocalMetadataBuildingContext context) {
142 		final String column = rtnSource.getColumn();
143 		final String typeName = rtnSource.getType();
144 		Type type = null;
145 		if ( typeName != null ) {
146 			type = context.getMetadataCollector().getTypeResolver().heuristicType( typeName );
147 			if ( type == null ) {
148 				throw new MappingException(
149 						String.format(
150 								"Unable to resolve type [%s] specified for native query scalar return",
151 								typeName
152 						),
153 						context.getOrigin()
154 				);
155 			}
156 		}
157 		return new NativeSQLQueryScalarReturn( column, type );
158 	}

161 	public static NativeSQLQueryRootReturn extractReturnDescription(
162 			JaxbHbmNativeQueryReturnType rtnSource,
163 			HbmLocalMetadataBuildingContext context,
164 			int queryReturnPosition) {
165 		String alias = rtnSource.getAlias();
166 		if ( StringHelper.isEmpty( alias ) ) {
167 			// hack-around as sqlquery impl depend on having a key.
168 			alias = "alias_" + queryReturnPosition;
169 		}
170 		final String entityName = context.determineEntityName(
171 				rtnSource.getEntityName(),
172 				rtnSource.getClazz()
173 		);
174 		final PersistentClass pc = context.getMetadataCollector().getEntityBinding( entityName );
175 
176 		return new NativeSQLQueryRootReturn(
177 				alias,
178 				entityName,
179 				extractPropertyResults( alias, rtnSource, pc, context ),
180 				rtnSource.getLockMode()
181 		);
182 	}

184 	public static NativeSQLQueryJoinReturn extractReturnDescription(
185 			JaxbHbmNativeQueryJoinReturnType rtnSource,
186 			HbmLocalMetadataBuildingContext context,
187 			int queryReturnPosition) {
188 		final int dot = rtnSource.getProperty().lastIndexOf( '.' );
189 		if ( dot == -1 ) {
190 			throw new MappingException(
191 					String.format(
192 							Locale.ENGLISH,
193 							"Role attribute for sql query return [%s] not formatted correctly {owningAlias.propertyName}",
194 							rtnSource.getAlias()
195 					),
196 					context.getOrigin()
197 			);
198 		}
199 
200 		String roleOwnerAlias = rtnSource.getProperty().substring( 0, dot );
201 		String roleProperty = rtnSource.getProperty().substring( dot + 1 );
202 
203 		return new NativeSQLQueryJoinReturn(
204 				rtnSource.getAlias(),
205 				roleOwnerAlias,
206 				roleProperty,
207 				//FIXME: get the PersistentClass
208 				extractPropertyResults( rtnSource.getAlias(), rtnSource, null, context ),
209 				rtnSource.getLockMode()
210 		);
211 	}

213 	public static NativeSQLQueryReturn extractReturnDescription(
214 			JaxbHbmNativeQueryCollectionLoadReturnType rtnSource,
215 			HbmLocalMetadataBuildingContext context,
216 			int queryReturnPosition) {
217 		final int dot = rtnSource.getRole().lastIndexOf( '.' );
218 		if ( dot == -1 ) {
219 			throw new MappingException(
220 					String.format(
221 							Locale.ENGLISH,
222 							"Collection attribute for sql query return [%s] not formatted correctly {OwnerClassName.propertyName}",
223 							rtnSource.getAlias()
224 					),
225 					context.getOrigin()
226 			);
227 		}
228 
229 		String ownerClassName = context.findEntityBinding( null, rtnSource.getRole().substring( 0, dot ) )
230 				.getClassName();
231 		String ownerPropertyName = rtnSource.getRole().substring( dot + 1 );
232 
233 		return new NativeSQLQueryCollectionReturn(
234 				rtnSource.getAlias(),
235 				ownerClassName,
236 				ownerPropertyName,
237 				// FIXME: get the PersistentClass
238 				extractPropertyResults( rtnSource.getAlias(), rtnSource, null, context ),
239 				rtnSource.getLockMode()
240 		);
241 	}

243 	/**
244 	 * return-property extraction/binding specific to an entity return to account for
245 	 * discriminator.
246 	 *
247 	 * @param alias The return alias
248 	 * @param rtnSource The entity return jaxb binding
249 	 * @param pc The located PersistentClass
250 	 * @param context The hbm context
251 	 *
252 	 * @return The extracted property mappings
253 	 */
254 	private static Map<String, String[]> extractPropertyResults(
255 			String alias,
256 			JaxbHbmNativeQueryReturnType rtnSource,
257 			PersistentClass pc,
258 			HbmLocalMetadataBuildingContext context) {
259 		Map<String, String[]> results = extractPropertyResults(
260 				alias,
261 				(NativeQueryNonScalarRootReturn) rtnSource,
262 				pc,
263 				context
264 		);
265 
266 		if ( rtnSource.getReturnDiscriminator() != null ) {
267 			if ( results == null ) {
268 				results = new HashMap<String, String[]>();
269 			}
270 
271 			final String column = rtnSource.getReturnDiscriminator().getColumn();
272 			if ( column == null ) {
273 				throw new MappingException(
274 						String.format(
275 								Locale.ENGLISH,
276 								"return-discriminator [%s (%s)] did not specify column",
277 								pc.getEntityName(),
278 								alias
279 						),
280 						context.getOrigin()
281 				);
282 			}
283 
284 			results.put( "class", new String[] {column} );
285 		}
286 
287 		return results;
288 	}

290 	@SuppressWarnings("unchecked")
291 	private static Map<String, String[]> extractPropertyResults(
292 			String alias,
293 			NativeQueryNonScalarRootReturn rtnSource,
294 			PersistentClass pc,
295 			HbmLocalMetadataBuildingContext context) {
296 		if ( CollectionHelper.isEmpty( rtnSource.getReturnProperty() ) ) {
297 			return null;
298 		}
299 
300 		final HashMap results = new HashMap();
301 
302 		List<JaxbHbmNativeQueryPropertyReturnType> propertyReturnSources = new ArrayList<JaxbHbmNativeQueryPropertyReturnType>();
303 		List<String> propertyNames = new ArrayList<String>();
304 
305 
306 		for ( JaxbHbmNativeQueryPropertyReturnType propertyReturnSource : rtnSource.getReturnProperty() ) {
307 			final int dotPosition = propertyReturnSource.getName().lastIndexOf( '.' );
308 			if ( pc == null || dotPosition == -1 ) {
309 				propertyReturnSources.add( propertyReturnSource );
310 				propertyNames.add( propertyReturnSource.getName() );
311 			}
312 			else {
313 				final String reducedName = propertyReturnSource.getName().substring( 0, dotPosition );
314 				final Value value = pc.getRecursiveProperty( reducedName ).getValue();
315 
316 				Iterator parentPropItr;
317 				if ( value instanceof Component ) {
318 					final Component comp = (Component) value;
319 					parentPropItr = comp.getPropertyIterator();
320 				}
321 				else if ( value instanceof ToOne ) {
322 					final ToOne toOne = (ToOne) value;
323 					final PersistentClass referencedPc = context.getMetadataCollector()
324 							.getEntityBinding( toOne.getReferencedEntityName() );
325 					if ( toOne.getReferencedPropertyName() != null ) {
326 						try {
327 							parentPropItr = ( (Component) referencedPc.getRecursiveProperty( toOne.getReferencedPropertyName() )
328 									.getValue() ).getPropertyIterator();
329 						}
330 						catch (ClassCastException e) {
331 							throw new MappingException(
332 									"dotted notation reference neither a component nor a many/one to one",
333 									e,
334 									context.getOrigin()
335 							);
336 						}
337 					}
338 					else {
339 						try {
340 							if ( referencedPc.getIdentifierMapper() == null ) {
341 								parentPropItr = ( (Component) referencedPc.getIdentifierProperty()
342 										.getValue() ).getPropertyIterator();
343 							}
344 							else {
345 								parentPropItr = referencedPc.getIdentifierMapper().getPropertyIterator();
346 							}
347 						}
348 						catch (ClassCastException e) {
349 							throw new MappingException(
350 									"dotted notation reference neither a component nor a many/one to one",
351 									e,
352 									context.getOrigin()
353 							);
354 						}
355 					}
356 				}
357 				else {
358 					throw new MappingException(
359 							"dotted notation reference neither a component nor a many/one to one",
360 							context.getOrigin()
361 					);
362 				}
363 
364 				boolean hasFollowers = false;
365 				List<String> followers = new ArrayList<String>();
366 				while ( parentPropItr.hasNext() ) {
367 					final Property parentProperty = (Property) parentPropItr.next();
368 					final String currentPropertyName = parentProperty.getName();
369 					final String currentName = reducedName + '.' + currentPropertyName;
370 					if ( hasFollowers ) {
371 						followers.add( currentName );
372 					}
373 					if ( propertyReturnSource.getName().equals( currentName ) ) {
374 						hasFollowers = true;
375 					}
376 				}
377 
378 				int index = propertyNames.size();
379 				for ( String follower : followers ) {
380 					int currentIndex = getIndexOfFirstMatchingProperty( propertyNames, follower );
381 					index = currentIndex != -1 && currentIndex < index ? currentIndex : index;
382 				}
383 				propertyNames.add( index, propertyReturnSource.getName() );
384 				propertyReturnSources.add( index, propertyReturnSource );
385 			}
386 		}
387 
388 		Set<String> uniqueReturnProperty = new HashSet<String>();
389 		for ( JaxbHbmNativeQueryPropertyReturnType propertyReturnBinding : propertyReturnSources ) {
390 			final String name = propertyReturnBinding.getName();
391 			if ( "class".equals( name ) ) {
392 				throw new MappingException(
393 						"class is not a valid property name to use in a <return-property>, use <return-discriminator> instead",
394 						context.getOrigin()
395 				);
396 			}
397 			//TODO: validate existing of property with the chosen name. (secondpass )
398 			ArrayList allResultColumns = extractResultColumns( propertyReturnBinding );
399 
400 			if ( allResultColumns.isEmpty() ) {
401 				throw new MappingException(
402 						String.format(
403 								Locale.ENGLISH,
404 								"return-property [alias=%s, property=%s] must specify at least one column or return-column name",
405 								alias,
406 								propertyReturnBinding.getName()
407 						),
408 						context.getOrigin()
409 				);
410 			}
411 			if ( uniqueReturnProperty.contains( name ) ) {
412 				throw new MappingException(
413 						String.format(
414 								Locale.ENGLISH,
415 								"Duplicate return-property [alias=%s] : %s",
416 								alias,
417 								propertyReturnBinding.getName()
418 						),
419 						context.getOrigin()
420 				);
421 			}
422 			uniqueReturnProperty.add( name );
423 
424 			// the issue here is that for <return-join/> representing an entity collection,
425 			// the collection element values (the property values of the associated entity)
426 			// are represented as 'element.{propertyname}'.  Thus the StringHelper.root()
427 			// here puts everything under 'element' (which additionally has significant
428 			// meaning).  Probably what we need to do is to something like this instead:
429 			//      String root = StringHelper.root( name );
430 			//      String key = root; // by default
431 			//      if ( !root.equals( name ) ) {
432 			//	        // we had a dot
433 			//          if ( !root.equals( alias ) {
434 			//              // the root does not apply to the specific alias
435 			//              if ( "elements".equals( root ) {
436 			//                  // we specifically have a <return-join/> representing an entity collection
437 			//                  // and this <return-property/> is one of that entity's properties
438 			//                  key = name;
439 			//              }
440 			//          }
441 			//      }
442 			// but I am not clear enough on the intended purpose of this code block, especially
443 			// in relation to the "Reorder properties" code block above...
444 			//			String key = StringHelper.root( name );
445 			ArrayList intermediateResults = (ArrayList) results.get( name );
446 			if ( intermediateResults == null ) {
447 				results.put( name, allResultColumns );
448 			}
449 			else {
450 				intermediateResults.addAll( allResultColumns );
451 			}
452 		}
453 
454 		for ( Object o : results.entrySet() ) {
455 			Map.Entry entry = (Map.Entry) o;
456 			if ( entry.getValue() instanceof ArrayList ) {
457 				ArrayList list = (ArrayList) entry.getValue();
458 				entry.setValue( list.toArray( new String[list.size()] ) );
459 			}
460 		}
461 		return results.isEmpty() ? Collections.EMPTY_MAP : results;
462 	}

464 	private static int getIndexOfFirstMatchingProperty(List propertyNames, String follower) {
465 		int propertySize = propertyNames.size();
466 		for ( int propIndex = 0; propIndex < propertySize; propIndex++ ) {
467 			if ( ( (String) propertyNames.get( propIndex ) ).startsWith( follower ) ) {
468 				return propIndex;
469 			}
470 		}
471 		return -1;
472 	}

474 	private static ArrayList<String> extractResultColumns(JaxbHbmNativeQueryPropertyReturnType propertyReturnSource) {
475 		final String column = unquote( propertyReturnSource.getColumn() );
476 		ArrayList<String> allResultColumns = new ArrayList<String>();
477 		if ( column != null ) {
478 			allResultColumns.add( column );
479 		}
480 		for ( JaxbHbmNativeQueryPropertyReturnType.JaxbHbmReturnColumn returnColumnSource : propertyReturnSource.getReturnColumn() ) {
481 			allResultColumns.add( unquote( returnColumnSource.getName() ) );
482 		}
483 		return allResultColumns;
484 	}
485 
486 	private static String unquote(String name) {
487 		if ( name != null && name.charAt( 0 ) == '`' ) {
488 			name = name.substring( 1, name.length() - 1 );
489 		}
490 		return name;
491 	}
