14  package org.hibernate.test.bytecode.enhancement.lazy.proxy;
15  
16  import java.io.Serializable;
17  import java.util.LinkedHashSet;
18  import java.util.Set;
19  import javax.persistence.Column;
20  import javax.persistence.Entity;
21  import javax.persistence.FetchType;
22  import javax.persistence.Inheritance;
23  import javax.persistence.InheritanceType;
24  import javax.persistence.JoinColumn;
25  import javax.persistence.ManyToOne;
26  import javax.persistence.OneToMany;
27  import javax.persistence.Table;
28  
29  import org.hibernate.annotations.LazyGroup;
30  import org.hibernate.annotations.LazyToOne;
31  import org.hibernate.annotations.LazyToOneOption;
32  
33  @Entity
34  @Inheritance(strategy = InheritanceType.TABLE_PER_CLASS)
35  @Table(name = "PP_DCKey")
36  public abstract class AbstractKey extends ModelEntity
37  		implements Serializable {
38  
39  	@Column(name = "Name")
40  	String name;
41  
42  	@OneToMany(targetEntity = RoleEntity.class, mappedBy = "key", fetch = FetchType.LAZY)
43  	@LazyToOne(LazyToOneOption.NO_PROXY)
44  	@LazyGroup("R")
45  	protected Set<RoleEntity> roles = new LinkedHashSet<>();
46  
47  	@ManyToOne(fetch = FetchType.LAZY)
48  	@LazyToOne(LazyToOneOption.NO_PROXY)
49  	@LazyGroup("PR")
50  	@JoinColumn
51  	protected AbstractKey register;
52  
53  	@OneToMany(targetEntity = AbstractKey.class, mappedBy = "register", fetch = FetchType.LAZY)
54  	@LazyToOne(LazyToOneOption.NO_PROXY)
55  	@LazyGroup("RK")
56  	protected Set<AbstractKey> keys = new LinkedHashSet();
57  
58  	@ManyToOne(fetch = FetchType.LAZY)
59  	@LazyToOne(LazyToOneOption.NO_PROXY)
60  	@LazyGroup("PP")
61  	@JoinColumn
62  	protected AbstractKey parent;
63  
64  	@OneToMany(targetEntity = AbstractKey.class, mappedBy = "parent", fetch = FetchType.LAZY)
65  	@LazyToOne(LazyToOneOption.NO_PROXY)
66  	@LazyGroup("PK")
67  	protected Set<AbstractKey> otherKeys = new LinkedHashSet();
68  
69  
70  	public String getName() {
71  		return name;
72  	}
73  
74  	public void setName(String name) {
75  		this.name = name;
76  	}
77  
78  	public Set<RoleEntity> getRoles() {
79  		return roles;
80  	}
81  
82  	public void setRoles(Set<RoleEntity> role) {
83  		this.roles = role;
84  	}
85  
86  	public void addRole(RoleEntity role) {
87  		this.roles.add( role );
88  	}
89  
90  	public AbstractKey getRegister() {
91  		return register;
92  	}
93  
94  	public void setRegister(AbstractKey register) {
95  		this.register = register;
96  	}
97  
98  	public Set<AbstractKey> getKeys() {
99  		return keys;
100 	}
101 
102 	public void setKeys(Set<AbstractKey> keys) {
103 		this.keys = keys;
104 	}
105 
106 	public void addRegisterKey(AbstractKey registerKey) {
107 		keys.add( registerKey );
108 	}
109 
110 	public AbstractKey getParent() {
111 		return parent;
112 	}
113 
114 	public void setParent(AbstractKey parent) {
115 		this.parent = parent;
116 	}
117 
118 	public Set<AbstractKey> getOtherKeys() {
119 		return otherKeys;
120 	}
121 
122 	public void setOtherKeys(Set<AbstractKey> otherKeys) {
123 		this.otherKeys = otherKeys;
124 	}
125 
126 	public void addPanelKey(AbstractKey panelKey) {
127 		this.otherKeys.add( panelKey );
128 	}
129 
130 }
