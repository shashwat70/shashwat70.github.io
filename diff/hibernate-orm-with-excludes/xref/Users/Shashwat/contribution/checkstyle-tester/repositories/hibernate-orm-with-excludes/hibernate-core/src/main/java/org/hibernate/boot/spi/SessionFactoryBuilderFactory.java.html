7   package org.hibernate.boot.spi;
8   
9   import org.hibernate.boot.SessionFactoryBuilder;
10  
11  /**
12   * An extension point for integrators that wish to hook into the process of how a SessionFactory
13   * is built.  Intended as a "discoverable service" ({@link java.util.ServiceLoader}).  There can
14   * be at most one implementation discovered that returns a non-null SessionFactoryBuilder.
15   *
16   * @author Steve Ebersole
17   */
18  public interface SessionFactoryBuilderFactory {
19  	/**
20  	 * The contract method.  Return the SessionFactoryBuilder.  May return {@code null}
21  	 *
22  	 * @param metadata The metadata from which we will be building a SessionFactory.
23  	 * @param defaultBuilder The default SessionFactoryBuilder instance.  If the SessionFactoryBuilder being built
24  	 * here needs to use this passed SessionFactoryBuilder instance it is the responsibility of the built
25  	 * SessionFactoryBuilder impl to delegate configuration calls to the passed default impl.
26  	 *
27  	 * @return The SessionFactoryBuilder, or {@code null}
28  	 */
29  	public SessionFactoryBuilder getSessionFactoryBuilder(MetadataImplementor metadata, SessionFactoryBuilderImplementor defaultBuilder);
30  }
