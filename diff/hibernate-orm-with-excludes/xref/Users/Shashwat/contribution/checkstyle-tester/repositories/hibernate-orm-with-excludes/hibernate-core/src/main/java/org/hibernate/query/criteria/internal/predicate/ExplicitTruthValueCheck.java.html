7   package org.hibernate.query.criteria.internal.predicate;
8   
9   import java.io.Serializable;
10  import javax.persistence.criteria.Expression;
11  
12  import org.hibernate.query.criteria.internal.CriteriaBuilderImpl;
13  import org.hibernate.query.criteria.internal.ParameterRegistry;
14  import org.hibernate.query.criteria.internal.Renderable;
15  import org.hibernate.query.criteria.internal.compile.RenderingContext;
16  
17  /**
18   * ANSI-SQL defines <tt>TRUE</tt>, <tt>FALSE</tt> and <tt>UNKNOWN</tt> as <i>truth values</i>.  These
19   * <i>truth values</i> are used to explicitly check the result of a boolean expression (the syntax is like
20   * <tt>a > b IS TRUE</tt>.  <tt>IS TRUE</tt> is the assumed default.
21   * <p/>
22   * JPA defines support for only <tt>IS TRUE</tt> and <tt>IS FALSE</tt>, not <tt>IS UNKNOWN</tt> (<tt>a > NULL</tt>
23   * is an example where the result would be UNKNOWN.
24   *
25   * @author Steve Ebersole
26   */
27  public class ExplicitTruthValueCheck
28  		extends AbstractSimplePredicate
29  		implements Serializable {
30  	// TODO : given that JPA supports only TRUE and FALSE, can this be handled just with negation?
31  	private final Expression<Boolean> booleanExpression;
32  	private final TruthValue truthValue;
33  
34  	public ExplicitTruthValueCheck(CriteriaBuilderImpl criteriaBuilder, Expression<Boolean> booleanExpression, TruthValue truthValue) {
35  		super( criteriaBuilder );
36  		this.booleanExpression = booleanExpression;
37  		this.truthValue = truthValue;
38  	}
39  
40  	public Expression<Boolean> getBooleanExpression() {
41  		return booleanExpression;
42  	}
43  
44  	public TruthValue getTruthValue() {
45  		return truthValue;
46  	}
47  
48  	@Override
49  	public void registerParameters(ParameterRegistry registry) {
50  		Helper.possibleParameter( getBooleanExpression(), registry );
51  	}
52  
53  	@Override
54  	public String render(boolean isNegated, RenderingContext renderingContext) {
55  		return ( (Renderable) getBooleanExpression() ).render( renderingContext )
56  				+ ( isNegated ? " <> " : " = " )
57  				+ ( getTruthValue() == TruthValue.TRUE ? "true" : "false" );
58  	}
59  }
