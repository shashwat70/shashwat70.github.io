7   package org.hibernate.test.annotations.reflection;
8   
9   import org.hibernate.testing.TestForIssue;
10  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
11  import org.junit.Test;
12  
13  import static org.hibernate.testing.transaction.TransactionUtil.doInHibernate;
14  import static org.junit.Assert.assertEquals;
15  
16  @TestForIssue( jiraKey = "HHH-11924")
17  public class ElementCollectionConverterTest extends BaseCoreFunctionalTestCase {
18  
19  	@Override
20  	protected Class[] getAnnotatedClasses() {
21  		return new Class[] {
22  			Company.class,
23  		};
24  	}
25  
26  	@Override
27  	protected String[] getXmlFiles() {
28  		return new String[] { "org/hibernate/test/annotations/reflection/element-collection-converter-orm.xml" };
29  	}
30  
31  
32  	@Test
33  	public void testConverterIsAppliedToElementCollection() {
34  		doInHibernate( this::sessionFactory, session -> {
35  			Company company = new Company();
36  			company.setId( 1L );
37  
38  			Organization org1 = new Organization();
39  			org1.setOrganizationId( "ACME" );
40  
41  			company.getOrganizations().add( org1 );
42  
43  			session.persist( company );
44  		} );
45  
46  		doInHibernate( this::sessionFactory, session -> {
47  			String organizationId = (String) session
48  					.createNativeQuery( "select organizations from Company_organizations" )
49  					.getSingleResult();
50  			assertEquals( "ORG-ACME", organizationId );
51  
52  			Company company = session.find( Company.class, 1L );
53  
54  			assertEquals( 1, company.getOrganizations().size() );
55  			assertEquals( "ACME" , company.getOrganizations().get( 0 ).getOrganizationId());
56  		} );
57  	}
58  }
