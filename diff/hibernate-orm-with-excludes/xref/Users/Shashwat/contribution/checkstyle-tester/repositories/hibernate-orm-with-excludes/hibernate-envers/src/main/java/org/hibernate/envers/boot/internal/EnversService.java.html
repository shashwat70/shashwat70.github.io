8   
9   import org.hibernate.boot.registry.classloading.spi.ClassLoaderService;
10  import org.hibernate.boot.spi.MetadataImplementor;
11  import org.hibernate.envers.configuration.internal.AuditEntitiesConfiguration;
12  import org.hibernate.envers.configuration.internal.GlobalConfiguration;
13  import org.hibernate.envers.configuration.internal.MappingCollector;
14  import org.hibernate.envers.internal.entities.EntitiesConfigurations;
15  import org.hibernate.envers.internal.revisioninfo.ModifiedEntityNamesReader;
16  import org.hibernate.envers.internal.revisioninfo.RevisionInfoNumberReader;
17  import org.hibernate.envers.internal.revisioninfo.RevisionInfoQueryCreator;

20  import org.hibernate.internal.util.xml.XMLHelper;
21  import org.hibernate.service.Service;
22  import org.hibernate.service.ServiceRegistry;
23  
24  /**
25   * Provides central access to Envers' configuration.
26   *
27   * In many ways, this replaces the legacy static map Envers used originally as
28   * a means to share the old AuditConfiguration.
29   *
30   * @author Steve Ebersole
31   */
32  public interface EnversService extends Service {
33  	/**
34  	 * The name of the configuration setting used to control whether the Envers integration
35  	 * is enabled.  Default is true
36  	 */
37  	String INTEGRATION_ENABLED = "hibernate.integration.envers.enabled";
38  
39  	/**
40  	 * Is the Envers integration enabled?  This is generally used as a
41  	 * protection for other Envers services (in the ServiceLoader sense)
42  	 * determine whether they should do their work.
43  	 *
44  	 * @return {@code true} If the integration is enabled; {@code false} otherwise.
45  	 */
46  	boolean isEnabled();
47  
48  	/**
49  	 * Assuming {@link #isEnabled()} is {@code true}, has {@link #initialize}
50  	 * been called yet?
51  	 *
52  	 * @return {@code true} indicates {@link #initialize} has been called; {@code false}
53  	 * indicates that {@link #initialize} has not (yet) been called.
54  	 */
55  	boolean isInitialized();
56  
57  	void initialize(MetadataImplementor metadata, MappingCollector mappingCollector);
58  
59  	XMLHelper getXmlHelper();
60  
61  	GlobalConfiguration getGlobalConfiguration();
62  
63  	AuditEntitiesConfiguration getAuditEntitiesConfiguration();
64  
65  	AuditProcessManager getAuditProcessManager();
66  
67  	AuditStrategy getAuditStrategy();
68  
69  	EntitiesConfigurations getEntitiesConfigurations();
70  
71  	RevisionInfoQueryCreator getRevisionInfoQueryCreator();
72  
73  	RevisionInfoNumberReader getRevisionInfoNumberReader();
74  
75  	ModifiedEntityNamesReader getModifiedEntityNamesReader();
76  
77  	ClassLoaderService getClassLoaderService();
78  
79  	ServiceRegistry getServiceRegistry();
80  }
