7   package org.hibernate;
8   
9   /**
10   * @author Steve Ebersole
11   */
12  public class PropertySetterAccessException extends PropertyAccessException {
13  	/**
14  	 * Constructs a PropertyAccessException using the specified information.
15  	 *
16  	 * @param cause The underlying cause
17  	 * @param persistentClass The class which is supposed to contain the property in question
18  	 * @param propertyName The name of the property.
19  	 * @param expectedType The expected property type
20  	 * @param target The target, which should be of type 'persistentClass'
21  	 * @param value The property value we are trying to set
22  	 */
23  	public PropertySetterAccessException(
24  			Throwable cause,
25  			Class persistentClass,
26  			String propertyName,
27  			Class expectedType,
28  			Object target,
29  			Object value) {
30  		super(
31  				cause,
32  				String.format(
33  						"IllegalArgumentException occurred while calling setter for property [%s.%s (expected type = %s)]; " +
34  								"target = [%s], property value = [%s]",
35  						persistentClass.getName(),
36  						propertyName,
37  						expectedType.getName(),
38  						target,
39  						value
40  				),
41  				true,
42  				persistentClass,
43  				propertyName
44  		);
45  	}
46  
47  	@Override
48  	public String toString() {
49  		return super.originalMessage();
50  	}
51  }
