7   package org.hibernate.type;
8   
9   import java.util.Calendar;
10  
11  import org.hibernate.type.descriptor.java.CalendarTimeTypeDescriptor;
12  import org.hibernate.type.descriptor.sql.TimeTypeDescriptor;
13  
14  /**
15   * A type mapping {@link java.sql.Types#TIME TIME} and {@link Calendar}.
16   * <p/>
17   * For example, a Calendar attribute annotated with {@link javax.persistence.Temporal} and specifying
18   * {@link javax.persistence.TemporalType#TIME}
19   *
20   * @author Steve Ebersole
21   */
22  public class CalendarTimeType extends AbstractSingleColumnStandardBasicType<Calendar> {
23  	public static final CalendarTimeType INSTANCE = new CalendarTimeType();
24  
25  	public CalendarTimeType() {
26  		super( TimeTypeDescriptor.INSTANCE, CalendarTimeTypeDescriptor.INSTANCE );
27  	}
28  
29  	public String getName() {
30  		return "calendar_time";
31  	}
32  }
