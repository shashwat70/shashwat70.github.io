7   package org.hibernate.boot.model.convert.spi;
8   
9   import javax.persistence.AttributeConverter;
10  
11  import org.hibernate.metamodel.model.convert.spi.JpaAttributeConverter;
12  
13  import com.fasterxml.classmate.ResolvedType;
14  
15  /**
16   * Boot-time descriptor of a JPA AttributeConverter
17   *
18   * @author Steve Ebersole
19   */
20  public interface ConverterDescriptor {
21  	/**
22  	 * The AttributeConverter class
23  	 */
24  	Class<? extends AttributeConverter> getAttributeConverterClass();
25  
26  	/**
27  	 * The resolved Classmate type descriptor for the conversion's domain type
28  	 */
29  	ResolvedType getDomainValueResolvedType();
30  
31  	/**
32  	 * The resolved Classmate type descriptor for the conversion's relational type
33  	 */
34  	ResolvedType getRelationalValueResolvedType();
35  
36  	/**
37  	 * Get the auto-apply checker for this converter.  Should never return `null` - prefer
38  	 * {@link org.hibernate.boot.model.convert.internal.AutoApplicableConverterDescriptorBypassedImpl#INSTANCE}
39  	 * instead.
40  	 */
41  	AutoApplicableConverterDescriptor getAutoApplyDescriptor();
42  
43  	/**
44  	 * Factory for the runtime representation of the converter
45  	 */
46  	JpaAttributeConverter createJpaAttributeConverter(JpaAttributeConverterCreationContext context);
47  }
