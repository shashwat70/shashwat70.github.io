24  package org.hibernate.bytecode.spi;
25  
26  import org.hibernate.bytecode.buildtime.spi.ClassFilter;
27  import org.hibernate.bytecode.buildtime.spi.FieldFilter;
28  
29  /**
30   * Contract for providers of bytecode services to Hibernate.
31   * <p/>
32   * Bytecode requirements break down into basically 3 areas<ol>
33   *     <li>proxy generation (both for runtime-lazy-loading and basic proxy generation) {@link #getProxyFactoryFactory()}</li>
34   *     <li>bean reflection optimization {@link #getReflectionOptimizer}</li>
35   *     <li>field-access instrumentation {@link #getTransformer}</li>
36   * </ol>
37   *
38   * @author Steve Ebersole
39   */
40  public interface BytecodeProvider {
41  	/**
42  	 * Retrieve the specific factory for this provider capable of
43  	 * generating run-time proxies for lazy-loading purposes.
44  	 *
45  	 * @return The provider specific factory.
46  	 */
47  	public ProxyFactoryFactory getProxyFactoryFactory();
48  
49  	/**
50  	 * Retrieve the ReflectionOptimizer delegate for this provider
51  	 * capable of generating reflection optimization components.
52  	 *
53  	 * @param clazz The class to be reflected upon.
54  	 * @param getterNames Names of all property getters to be accessed via reflection.
55  	 * @param setterNames Names of all property setters to be accessed via reflection.
56  	 * @param types The types of all properties to be accessed.
57  	 * @return The reflection optimization delegate.
58  	 */
59  	public ReflectionOptimizer getReflectionOptimizer(Class clazz, String[] getterNames, String[] setterNames, Class[] types);
60  
61  	/**
62  	 * Generate a ClassTransformer capable of performing bytecode manipulation.
63  	 *
64  	 * @param classFilter filter used to limit which classes are to be instrumented
65  	 * via this ClassTransformer.
66  	 * @param fieldFilter filter used to limit which fields are to be instrumented
67  	 * via this ClassTransformer.
68  	 * @return The appropriate ClassTransformer.
69  	 */
70  	public ClassTransformer getTransformer(ClassFilter classFilter, FieldFilter fieldFilter);
71  
72  	/**
73  	 * Retrieve the interception metadata for the particular entity type.
74  	 *
75  	 * @param entityClass The entity class.  Note: we pass class here instead of the usual "entity name" because
76  	 * only real classes can be instrumented.
77  	 *
78  	 * @return The metadata
79  	 */
80  	public EntityInstrumentationMetadata getEntityInstrumentationMetadata(Class entityClass);
81  }
