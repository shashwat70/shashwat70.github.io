9   package org.hibernate.jpa.test.emops;
10  import java.util.ArrayList;
11  import java.util.List;
12  import javax.persistence.CascadeType;
13  import javax.persistence.Entity;
14  import javax.persistence.FetchType;
15  import javax.persistence.GeneratedValue;
16  import javax.persistence.Id;
17  import javax.persistence.OneToMany;
18  import javax.persistence.OrderColumn;
19  
20  /**
21   * @author Emmanuel Bernard
22   */
23  @Entity
24  public class Race {
25  	@Id @GeneratedValue public Integer id;
26  	@OrderColumn( name="index_" ) @OneToMany(cascade = CascadeType.ALL, fetch = FetchType.EAGER)
27  	@org.hibernate.annotations.Cascade( { org.hibernate.annotations.CascadeType.ALL, org.hibernate.annotations.CascadeType.DELETE_ORPHAN }) 
28  	public List<Competitor> competitors = new ArrayList<Competitor>();
29  	public String name;
30  }
