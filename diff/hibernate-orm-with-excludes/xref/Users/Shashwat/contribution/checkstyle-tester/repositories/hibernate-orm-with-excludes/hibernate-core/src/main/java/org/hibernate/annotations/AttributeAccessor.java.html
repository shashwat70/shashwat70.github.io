7   package org.hibernate.annotations;
8   
9   import java.lang.annotation.Retention;
10  
11  import static java.lang.annotation.ElementType.FIELD;
12  import static java.lang.annotation.ElementType.METHOD;
13  import static java.lang.annotation.ElementType.TYPE;
14  import static java.lang.annotation.RetentionPolicy.RUNTIME;
15  
16  /**
17   * Names a persistent property access strategy ({@link org.hibernate.property.access.spi.PropertyAccessStrategy}) to use.
18   *
19   * Can be specified at either:<ul>
20   *     <li>
21   *         <strong>TYPE</strong> level, which will act as naming the default accessor strategy for
22   *         all attributes on the class which do not explicitly name an accessor strategy
23   *     </li>
24   *     <li>
25   *         <strong>METHOD/FIELD</strong> level, which will be in effect for just that attribute.
26   *     </li>
27   * </ul>
28   *
29   * Should only be used to name custom {@link org.hibernate.property.access.spi.PropertyAccessStrategy}.  For
30   * {@code property/field} access, the JPA {@link javax.persistence.Access} annotation should be preferred
31   * using the appropriate {@link javax.persistence.AccessType}.  However, if this annotation is used with
32   * either {@code value="property"} or {@code value="field"}, it will act just as the corresponding usage
33   * of {@link javax.persistence.Access}.
34   *
35   * @see org.hibernate.property.access.spi.PropertyAccessStrategy
36   * @see org.hibernate.property.access.spi.PropertyAccessStrategyResolver
37   *
38   * @author Steve Ebersole
39   * @author Emmanuel Bernard
40   */
41  @java.lang.annotation.Target({ TYPE, METHOD, FIELD })
42  @Retention(RUNTIME)
43  public @interface AttributeAccessor {
44  	/**
45  	 * Names the {@link org.hibernate.property.access.spi.PropertyAccessStrategy} strategy.
46  	 */
47  	String value();
48  }
