196 @Entity
197 public class Person {
198 
199     @Id
200     @GeneratedValue
201     private Long id;
202 
203     private String name;
204 
205     private String nickName;
206 
207     private String address;
208 
209     @Temporal(TemporalType.TIMESTAMP )
210     private Date createdOn;
211 
212     @OneToMany(mappedBy = "person", cascade = CascadeType.ALL)
213     @OrderColumn(name = "order_id")
214     private List<Phone> phones = new ArrayList<>();
215 
216     @ElementCollection
217     @MapKeyEnumerated(EnumType.STRING)
218     private Map<AddressType, String> addresses = new HashMap<>();
219 
220     @Version
221     private int version;
222 
223     //Getters and setters are omitted for brevity
224 
225 //end::hql-examples-domain-model-example[]
226 
227     public Person() {}
228 
229     public Person(String name) {
230         this.name = name;
231     }
232 
233     public Long getId() {
234         return id;
235     }
236 
237     public String getName() {
238         return name;
239     }
240 
241     public String getNickName() {
242         return nickName;
243     }
244 
245     public void setNickName(String nickName) {
246         this.nickName = nickName;
247     }
248 
249     public String getAddress() {
250         return address;
251     }
252 
253     public void setAddress(String address) {
254         this.address = address;
255     }
256 
257     public Date getCreatedOn() {
258         return createdOn;
259     }
260 
261     public void setCreatedOn(Date createdOn) {
262         this.createdOn = createdOn;
263     }
264 
265     public List<Phone> getPhones() {
266         return phones;
267     }
268 
269     public Map<AddressType, String> getAddresses() {
270         return addresses;
271     }
272 
273     public void addPhone(Phone phone) {
274         phones.add( phone );
275         phone.setPerson( this );
276     }

278 }
