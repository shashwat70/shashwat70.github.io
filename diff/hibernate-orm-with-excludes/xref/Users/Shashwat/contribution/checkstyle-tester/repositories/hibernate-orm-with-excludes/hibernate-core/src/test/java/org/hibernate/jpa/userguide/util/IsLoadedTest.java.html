7   package org.hibernate.jpa.userguide.util;
8   
9   import javax.persistence.EntityManager;
10  
11  import org.junit.Test;
12  
13  import org.hibernate.jpa.test.BaseEntityManagerFunctionalTestCase;
14  
15  import static org.junit.Assert.assertFalse;
16  import static org.junit.Assert.assertTrue;
17  
18  /**
19   * @author Emmanuel Bernard
20   */
21  public class IsLoadedTest extends BaseEntityManagerFunctionalTestCase {
22  	@Test
23  	public void testIsLoadedOnPrivateSuperclassProperty() {
24  		EntityManager em = entityManagerFactory().createEntityManager();
25  		em.getTransaction().begin();
26  		try {
27  			Author a = new Author();
28  			Book book = new Book( a );
29  			em.persist( a );
30  			em.persist( book );
31  			em.flush();
32  			em.clear();
33  			book = em.find( Book.class, book.getId() );
34  			assertTrue( em.getEntityManagerFactory().getPersistenceUnitUtil().isLoaded( book ) );
35  			assertFalse( em.getEntityManagerFactory().getPersistenceUnitUtil().isLoaded( book, "author" ) );
36  		}
37  		finally {
38  			em.getTransaction().rollback();
39  			em.close();
40  		}
41  	}
42  
43  	@Override
44  	protected Class<?>[] getAnnotatedClasses() {
45  		return new Class<?>[] {
46  				Author.class,
47  				Book.class,
48  				CopyrightableContent.class
49  		};
50  	}
51  }
