7   package org.hibernate.graph;
8   
9   import java.util.Map;
10  import javax.persistence.Subgraph;
11  import javax.persistence.metamodel.Attribute;
12  
13  /**
14   * Hibernate extension to the JPA entity-graph AttributeNode contract.
15   *
16   * @author Strong Liu <stliu@hibernate.org>
17   * @author Steve Ebersole
18   * @author Andrea Boriero
19   */
20  public interface AttributeNode<J> extends GraphNode<J>, javax.persistence.AttributeNode<J> {
21  	Attribute<?, J> getAttributeDescriptor();
22  
23  	Map<Class<? extends J>, SubGraph<? extends J>> getSubGraphs();
24  	Map<Class<? extends J>, SubGraph<? extends J>> getKeySubGraphs();
25  
26  	@Override
27  	@SuppressWarnings("unchecked")
28  	default Map<Class, Subgraph> getSubgraphs() {
29  		return (Map) getSubGraphs();
30  	}
31  
32  	@Override
33  	@SuppressWarnings("unchecked")
34  	default Map<Class, Subgraph> getKeySubgraphs() {
35  		return (Map) getKeySubGraphs();
36  	}
37  
38  	<S extends J> void addSubGraph(Class<S> subType, SubGraph<S> subGraph);
39  	<S extends J> void addKeySubGraph(Class<S> subType, SubGraph<S> subGraph);
40  
41  	SubGraph<J> makeSubGraph();
42  	SubGraph<J> makeKeySubGraph();
43  
44  	<S extends J> SubGraph<S> makeSubGraph(Class<S> subtype);
45  	<S extends J> SubGraph<S> makeKeySubGraph(Class<S> subtype);
46  }
