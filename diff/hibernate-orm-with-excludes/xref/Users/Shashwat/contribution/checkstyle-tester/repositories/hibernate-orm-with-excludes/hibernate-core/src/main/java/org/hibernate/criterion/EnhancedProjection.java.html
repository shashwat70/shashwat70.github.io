25  package org.hibernate.criterion;
26  import org.hibernate.Criteria;
27  
28  /**
29   * An "enhanced" Projection for a {@link Criteria} query.
30   *
31   * @author Gail Badner
32   * @see Projection
33   * @see Criteria
34   */
35  public interface EnhancedProjection extends Projection {
36  
37  	/**
38  	 * Get the SQL column aliases used by this projection for the columns it writes for inclusion into the
39  	 * <tt>SELECT</tt> clause ({@link #toSqlString}.  Hibernate always uses column aliases to extract data from the
40  	 * JDBC {@link java.sql.ResultSet}, so it is important that these be implemented correctly in order for
41  	 * Hibernate to be able to extract these val;ues correctly.
42  	 *
43  	 * @param position Just as in {@link #toSqlString}, represents the number of <b>columns</b> rendered
44  	 * prior to this projection.
45  	 * @param criteria The local criteria to which this project is attached (for resolution).
46  	 * @param criteriaQuery The overall criteria query instance.
47  	 * @return The columns aliases.
48  	 */
49  	public String[] getColumnAliases(int position, Criteria criteria, CriteriaQuery criteriaQuery);
50  
51  	/**
52  	 * Get the SQL column aliases used by this projection for the columns it writes for inclusion into the
53  	 * <tt>SELECT</tt> clause ({@link #toSqlString} <i>for a particular criteria-level alias</i>.
54  	 *
55  	 * @param alias The criteria-level alias
56  	 * @param position Just as in {@link #toSqlString}, represents the number of <b>columns</b> rendered
57  	 * prior to this projection.
58  	 * @param criteria The local criteria to which this project is attached (for resolution).
59  	 * @param criteriaQuery The overall criteria query instance.
60  	 * @return The columns aliases pertaining to a particular criteria-level alias; expected to return null if
61  	 * this projection does not understand this alias.
62  	 */
63  	public String[] getColumnAliases(String alias, int position, Criteria criteria, CriteriaQuery criteriaQuery);
64  }
