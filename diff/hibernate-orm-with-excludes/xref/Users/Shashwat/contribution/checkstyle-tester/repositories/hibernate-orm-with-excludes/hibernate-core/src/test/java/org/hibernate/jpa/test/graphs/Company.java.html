7   package org.hibernate.jpa.test.graphs;
8   
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  import javax.persistence.ElementCollection;
13  import javax.persistence.Entity;
14  import javax.persistence.FetchType;
15  import javax.persistence.GeneratedValue;
16  import javax.persistence.Id;
17  import javax.persistence.OneToMany;
18  import javax.persistence.OneToOne;
19  
20  
21  /**
22   * @author Brett Meyer
23   */
24  @Entity
25  public class Company {
26  	@Id @GeneratedValue
27  	public long id;
28  	
29  	@OneToMany
30  	public Set<Employee> employees = new HashSet<Employee>();
31  	
32  	@OneToOne(fetch = FetchType.LAZY)
33  	public Location location;
34  	
35  	@ElementCollection
36  	public Set<Market> markets = new HashSet<Market>();
37  	
38  	@ElementCollection(fetch = FetchType.EAGER)
39  	public Set<String> phoneNumbers = new HashSet<String>();
40  
41  	public Location getLocation() {
42  		return location;
43  	}
44  }
