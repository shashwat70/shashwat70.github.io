7   package org.hibernate.tuple;
8   
9   import java.io.Serializable;
10  
11  /**
12   * Describes the generation of property values.
13   *
14   * @author Steve Ebersole
15   */
16  public interface ValueGeneration extends Serializable {
17  	/**
18  	 * When is this value generated : NEVER, INSERT, ALWAYS (INSERT+UPDATE)
19  	 *
20  	 * @return When the value is generated.
21  	 */
22  	public GenerationTiming getGenerationTiming();
23  
24  	/**
25  	 * Obtain the in-VM value generator.
26  	 * <p/>
27  	 * May return {@code null}.  In fact for values that are generated "in the database" via execution of the
28  	 * INSERT/UPDATE statement, the expectation is that {@code null} be returned here
29  	 *
30  	 * @return The strategy for performing in-VM value generation
31  	 */
32  	public ValueGenerator<?> getValueGenerator();
33  
34  	/**
35  	 * For values which are generated in the database ({@link #getValueGenerator()} == {@code null}), should the
36  	 * column be referenced in the INSERT / UPDATE SQL?
37  	 * <p/>
38  	 * This will be false most often to have a DDL-defined DEFAULT value be applied on INSERT
39  	 *
40  	 * @return {@code true} indicates the column should be included in the SQL.
41  	 */
42  	public boolean referenceColumnInSql();
43  
44  	/**
45  	 * For values which are generated in the database ({@link #getValueGenerator} == {@code null}), if the
46  	 * column will be referenced in the SQL ({@link #referenceColumnInSql()} == {@code true}), what value should be
47  	 * used in the SQL as the column value.
48  	 * <p/>
49  	 * Generally this will be a function call or a marker (DEFAULTS).
50  	 * <p/>
51  	 * NOTE : for in-VM generation, this will not be called and the column value will implicitly be a JDBC parameter ('?')
52  	 *
53  	 * @return The column value to be used in the SQL.
54  	 */
55  	public String getDatabaseGeneratedReferencedColumnValue();
56  }
