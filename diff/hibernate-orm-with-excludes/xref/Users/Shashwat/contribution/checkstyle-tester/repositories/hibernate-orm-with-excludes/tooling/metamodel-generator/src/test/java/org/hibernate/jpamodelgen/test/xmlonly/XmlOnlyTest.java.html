7   package org.hibernate.jpamodelgen.test.xmlonly;
8   
9   import org.hibernate.jpamodelgen.JPAMetaModelEntityProcessor;
10  import org.hibernate.jpamodelgen.test.util.CompilationTest;
11  import org.hibernate.jpamodelgen.test.util.WithClasses;
12  import org.hibernate.jpamodelgen.test.util.WithProcessorOption;
13  import org.junit.Test;
14  
15  import static org.hibernate.jpamodelgen.test.util.TestUtil.assertMetamodelClassGeneratedFor;
16  import static org.hibernate.jpamodelgen.test.util.TestUtil.assertPresenceOfFieldInMetamodelFor;
17  
18  /**
19   * @author Hardy Ferentschik
20   */
21  @WithClasses({ Car.class, Course.class, Option.class, Period.class, Teacher.class, Tire.class, XmlOnly.class })
22  @WithProcessorOption(key = JPAMetaModelEntityProcessor.PERSISTENCE_XML_OPTION,
23  		value = "org/hibernate/jpamodelgen/test/xmlonly/persistence.xml")
24  public class XmlOnlyTest extends CompilationTest {
25  	@Test
26  	public void testMetaModelGeneratedForXmlConfiguredEntity() {
27  		assertMetamodelClassGeneratedFor( XmlOnly.class );
28  	}
29  
30  	@Test
31  	public void testMetaModelGeneratedForManyToManyFieldAccessWithoutTargetEntity() {
32  		assertPresenceOfFieldInMetamodelFor( Course.class, "qualifiedTeachers", "Type should be inferred from field" );
33  		assertPresenceOfFieldInMetamodelFor( Teacher.class, "qualifiedFor", "Type should be inferred from field" );
34  	}
35  
36  	@Test
37  	public void testMetaModelGeneratedForOneToManyPropertyAccessWithoutTargetEntity() {
38  		assertPresenceOfFieldInMetamodelFor( Car.class, "tires", "Type should be inferred from field" );
39  		assertPresenceOfFieldInMetamodelFor( Tire.class, "car", "Type should be inferred from field" );
40  	}
41  
42  	@Test
43  	public void testMetaModelGeneratedForEmbeddable() {
44  		assertPresenceOfFieldInMetamodelFor( Option.class, "period", "Embedded expected" );
45  		assertPresenceOfFieldInMetamodelFor( Period.class, "start", "Embedded expected" );
46  		assertPresenceOfFieldInMetamodelFor( Period.class, "end", "Embedded expected" );
47  	}
48  }
