172 
173 		private byte[] bytes = null;
174 
175 		public MaterializedBlob(byte[] bytes) {
176 			this.setBytes( bytes );
177 		}
178 
179 		@Override
180 		public long length() throws SQLException {
181 			return this.getBytes().length;
182 		}
183 
184 		@Override
185 		public byte[] getBytes(long pos, int length) throws SQLException {
186 			return Arrays.copyOfRange( this.bytes, (int) ( pos - 1 ), (int) ( pos - 1 + length ) );
187 		}
188 
189 		@Override
190 		public InputStream getBinaryStream() throws SQLException {
191 			return new ByteArrayInputStream( this.getBytes() );
192 		}
193 
194 		@Override
195 		public long position(byte[] pattern, long start) throws SQLException {
196 			throw new SQLFeatureNotSupportedException();
197 		}
198 
199 		@Override
200 		public long position(Blob pattern, long start) throws SQLException {
201 			throw new SQLFeatureNotSupportedException();
202 		}
203 
204 		@Override
205 		public int setBytes(long pos, byte[] bytes) throws SQLException {
206 			int bytesSet = 0;
207 			if ( this.bytes.length < pos - 1 + bytes.length ) {
208 				this.bytes = Arrays.copyOf( this.bytes, (int) ( pos - 1 + bytes.length ) );
209 			}
210 			for ( int i = 0; i < bytes.length && i < this.bytes.length; i++, bytesSet++ ) {
211 				this.bytes[(int) ( i + pos - 1 )] = bytes[i];
212 			}
213 			return bytesSet;
214 		}
215 
216 		@Override
217 		public int setBytes(long pos, byte[] bytes, int offset, int len) throws SQLException {
218 			int bytesSet = 0;
219 			if ( this.bytes.length < pos - 1 + len ) {
220 				this.bytes = Arrays.copyOf( this.bytes, (int) ( pos - 1 + len ) );
221 			}
222 			for ( int i = offset; i < len && i < this.bytes.length; i++, bytesSet++ ) {
223 				this.bytes[(int) ( i + pos - 1 )] = bytes[i];
224 			}
225 			return bytesSet;
226 		}
227 
228 		@Override
229 		public OutputStream setBinaryStream(long pos) throws SQLException {
230 			return new ByteArrayOutputStream() {
231 
232 				{
233 					this.buf = getBytes();
234 				}
235 			};
236 		}
237 
238 		@Override
239 		public void truncate(long len) throws SQLException {
240 			this.setBytes( Arrays.copyOf( this.getBytes(), (int) len ) );
241 		}
242 
243 		@Override
244 		public void free() throws SQLException {
245 			this.setBytes( null );
246 		}
247 
248 		@Override
249 		public InputStream getBinaryStream(long pos, long length) throws SQLException {
250 			return new ByteArrayInputStream( this.getBytes(), (int) ( pos - 1 ), (int) length );
251 		}
252 
253 		byte[] getBytes() {
254 			return this.bytes;
255 		}
256 
257 		void setBytes(byte[] bytes) {
258 			this.bytes = bytes;
259 		}
