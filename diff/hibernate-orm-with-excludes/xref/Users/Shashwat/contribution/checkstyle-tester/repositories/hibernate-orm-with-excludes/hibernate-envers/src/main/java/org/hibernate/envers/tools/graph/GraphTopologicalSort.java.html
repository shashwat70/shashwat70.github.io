24  package org.hibernate.envers.tools.graph;
25  import java.util.ArrayList;
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  /**
31   * @author Adam Warski (adam at warski dot org)
32   */
33  public class GraphTopologicalSort {
34      /**
35       * Sorts a graph topologically.
36       * @param definer Defines a graph (values and representations) to sort.
37       * @return Values of the graph, sorted topologically.
38       */
39      public static <V, R> List<V> sort(GraphDefiner<V, R> definer) {
40          List<V> values = definer.getValues();
41          Map<R, Vertex<R>> vertices = new HashMap<R, Vertex<R>>();
42  
43          // Creating a vertex for each representation
44          for (V v : values) {
45              R rep = definer.getRepresentation(v);
46              vertices.put(rep, new Vertex<R>(rep));
47          }
48  
49          // Connecting neighbourhooding vertices
50          for (V v : values) {
51              for (V vn : definer.getNeighbours(v)) {
52                  vertices.get(definer.getRepresentation(v)).addNeighbour(vertices.get(definer.getRepresentation(vn)));
53              }
54          }
55  
56          // Sorting the representations
57          List<R> sortedReps = new TopologicalSort<R>().sort(vertices.values());
58  
59          // Transforming the sorted representations to sorted values 
60          List<V> sortedValues = new ArrayList<V>(sortedReps.size());
61          for (R rep : sortedReps) {
62              sortedValues.add(definer.getValue(rep));
63          }
64  
65          return sortedValues;
66      }
67  }
