18  /**

21  public class RevisionInfoConfigurationResult {
22  	private final RevisionInfoGenerator revisionInfoGenerator;
23  	private final Document revisionInfoXmlMapping;
24  	private final RevisionInfoQueryCreator revisionInfoQueryCreator;
25  	private final Element revisionInfoRelationMapping;
26  	private final RevisionInfoNumberReader revisionInfoNumberReader;
27  	private final ModifiedEntityNamesReader modifiedEntityNamesReader;
28  	private final String revisionInfoEntityName;
29  	private final Class<?> revisionInfoClass;
30  	private final PropertyData revisionInfoTimestampData;
31  
32  	RevisionInfoConfigurationResult(
33  			RevisionInfoGenerator revisionInfoGenerator,
34  			Document revisionInfoXmlMapping, RevisionInfoQueryCreator revisionInfoQueryCreator,
35  			Element revisionInfoRelationMapping, RevisionInfoNumberReader revisionInfoNumberReader,
36  			ModifiedEntityNamesReader modifiedEntityNamesReader, String revisionInfoEntityName,
37  			Class<?> revisionInfoClass, PropertyData revisionInfoTimestampData) {
38  		this.revisionInfoGenerator = revisionInfoGenerator;
39  		this.revisionInfoXmlMapping = revisionInfoXmlMapping;
40  		this.revisionInfoQueryCreator = revisionInfoQueryCreator;
41  		this.revisionInfoRelationMapping = revisionInfoRelationMapping;
42  		this.revisionInfoNumberReader = revisionInfoNumberReader;
43  		this.modifiedEntityNamesReader = modifiedEntityNamesReader;
44  		this.revisionInfoEntityName = revisionInfoEntityName;
45  		this.revisionInfoClass = revisionInfoClass;
46  		this.revisionInfoTimestampData = revisionInfoTimestampData;
47  	}
48  
49  	public RevisionInfoGenerator getRevisionInfoGenerator() {
50  		return revisionInfoGenerator;
51  	}
52  
53  	public Document getRevisionInfoXmlMapping() {
54  		return revisionInfoXmlMapping;
55  	}
56  
57  	public RevisionInfoQueryCreator getRevisionInfoQueryCreator() {
58  		return revisionInfoQueryCreator;
59  	}
60  
61  	public Element getRevisionInfoRelationMapping() {
62  		return revisionInfoRelationMapping;
63  	}
64  
65  	public RevisionInfoNumberReader getRevisionInfoNumberReader() {
66  		return revisionInfoNumberReader;
67  	}
68  
69  	public String getRevisionInfoEntityName() {
70  		return revisionInfoEntityName;
71  	}
72  
73  	public Class<?> getRevisionInfoClass() {

76  
77  	public PropertyData getRevisionInfoTimestampData() {

79  	}
80  
81  	public ModifiedEntityNamesReader getModifiedEntityNamesReader() {
