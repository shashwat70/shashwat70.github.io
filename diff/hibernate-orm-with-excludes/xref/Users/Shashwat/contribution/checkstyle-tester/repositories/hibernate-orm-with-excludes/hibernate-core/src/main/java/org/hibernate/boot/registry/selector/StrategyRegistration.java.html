7   package org.hibernate.boot.registry.selector;
8   
9   /**
10   * Describes the registration of a named strategy implementation.  A strategy + selector name should resolve
11   * to a single implementation.
12   *
13   * @param <T> The type of the strategy described by this implementation registration.
14   *
15   * @author Steve Ebersole
16   */
17  public interface StrategyRegistration<T> {
18  	/**
19  	 * The strategy role.  Best practice says this should be an interface.
20  	 *
21  	 * @return The strategy contract/role.
22  	 */
23  	public Class<T> getStrategyRole();
24  
25  	/**
26  	 * Any registered names for this strategy registration.
27  	 *
28  	 * @return The registered selection names.
29  	 */
30  	public Iterable<String> getSelectorNames();
31  
32  	/**
33  	 * The strategy implementation class.
34  	 *
35  	 * @return The strategy implementation.
36  	 */
37  	public Class<? extends T> getStrategyImplementation();
38  }
