7   package org.hibernate.persister.walking.spi;
8   
9   import org.hibernate.type.AnyType;
10  import org.hibernate.type.Type;
11  
12  /**
13   * Describes an ANY mapping
14   *
15   * @author Steve Ebersole
16   */
17  public interface AnyMappingDefinition {
18  	/**
19  	 * Access to the mapping's AnyType
20  	 *
21  	 * @return The AnyType
22  	 */
23  	public AnyType getType();
24  
25  	/**
26  	 * Was the mapping defined as lazy?
27  	 *
28  	 * @return true/false
29  	 */
30  	public boolean isLazy();
31  
32  	/**
33  	 * Access to the type of the value that makes up the identifier portion of the AnyType.
34  	 *
35  	 * @return The identifier type
36  	 *
37  	 * @see org.hibernate.annotations.AnyMetaDef#idType()
38  	 */
39  	public Type getIdentifierType();
40  
41  	/**
42  	 * Access to the type of the value that makes up the discriminator portion of the AnyType.  The discriminator is
43  	 * historically called the "meta".
44  	 * <p/>
45  	 * NOTE : If explicit discriminator mappings are given, the type here will be a {@link org.hibernate.type.MetaType}.
46  	 *
47  	 * @return The discriminator type
48  	 *
49  	 * @see org.hibernate.annotations.Any#metaColumn()
50  	 * @see org.hibernate.annotations.AnyMetaDef#metaType()
51  	 */
52  	public Type getDiscriminatorType();
53  
54  	/**
55  	 * Access to discriminator mappings explicitly defined in the mapping metadata.
56  	 *
57  	 * There are 2 flavors of discrimination:<ol>
58  	 *     <li>
59  	 *         The database holds the concrete entity names.  This is an implicit form, meaning that the discriminator
60  	 *         mappings do not have to be defined in the mapping metadata.  In this case, an empty iterable is returned
61  	 *         here
62  	 *     </li>
63  	 *     <li>
64  	 *         The database holds discriminator values that are interpreted to corresponding entity names based on
65  	 *         discriminator mappings explicitly supplied in the mapping metadata (see
66  	 *         {@link org.hibernate.annotations.AnyMetaDef#metaValues()}).  In this case, this method gives access
67  	 *         to those explicitly defined mappings.
68  	 *     </li>
69  	 * </ol>
70  	 *
71  	 * @return The explicitly defined discriminator value mappings.
72  	 */
73  	public Iterable<DiscriminatorMapping> getMappingDefinedDiscriminatorMappings();
74  
75  	/**
76  	 * Models a single discriminator mapping definition
77  	 */
78  	public static interface DiscriminatorMapping {
79  		/**
80  		 * Access to the defined discriminator value (the database value) being mapped.
81  		 *
82  		 * @return The defined discriminator value
83  		 */
84  		public Object getDiscriminatorValue();
85  
86  		/**
87  		 * Access to the defined entity name corresponding to the defined {@link #getDiscriminatorValue()}
88  		 *
89  		 * @return The defined entity name
90  		 */
91  		public String getEntityName();
92  	}
93  }
