1   package org.hibernate.test.schemaupdate;
2   
3   import java.util.EnumSet;
4   
5   import org.hibernate.boot.MetadataSources;
6   import org.hibernate.boot.registry.StandardServiceRegistry;
7   import org.hibernate.boot.registry.StandardServiceRegistryBuilder;
8   import org.hibernate.boot.spi.MetadataImplementor;
9   import org.hibernate.cfg.AvailableSettings;
10  import org.hibernate.dialect.H2Dialect;
11  import org.hibernate.tool.hbm2ddl.SchemaUpdate;
12  import org.hibernate.tool.schema.TargetType;
13  
14  import org.hibernate.testing.RequiresDialect;
15  import org.hibernate.testing.TestForIssue;
16  import org.hibernate.testing.junit4.BaseUnitTestCase;
17  import org.junit.After;
18  import org.junit.Test;
19  
20  @RequiresDialect(H2Dialect.class)
21  @TestForIssue(jiraKey = "HHH-13597")
22  public class H2DialectDataBaseToUpperTest extends BaseUnitTestCase {
23  
24  	private StandardServiceRegistry ssr;
25  	private MetadataImplementor metadata;
26  
27  	@Test
28  	public void hibernateShouldStartUpWithH2AutoUpdateAndDatabaseToUpperFalse() {
29  		setUp( "false" );
30  		new SchemaUpdate().setHaltOnError( true )
31  				.execute( EnumSet.of( TargetType.DATABASE ), metadata );
32  	}
33  
34  	@Test
35  	public void hibernateShouldStartUpWithH2AutoUpdateAndDatabaseToUpperTrue() {
36  		setUp( "true" );
37  		new SchemaUpdate().setHaltOnError( true )
38  				.execute( EnumSet.of( TargetType.DATABASE ), metadata );
39  	}
40  
41  	private void setUp(String databaseToUpper) {
42  		ssr = new StandardServiceRegistryBuilder()
43  				.applySetting(
44  						AvailableSettings.URL,
45  						"jdbc:h2:mem:databaseToUpper;DATABASE_TO_UPPER=" + databaseToUpper
46  				)
47  				.build();
48  		final MetadataSources metadataSources = new MetadataSources( ssr );
49  		metadata = (MetadataImplementor) metadataSources.buildMetadata();
50  	}
51  
52  	@After
53  	public void tearDown() {
54  		if ( ssr != null ) {
55  			StandardServiceRegistryBuilder.destroy( ssr );
56  		}
57  	}
58  }
