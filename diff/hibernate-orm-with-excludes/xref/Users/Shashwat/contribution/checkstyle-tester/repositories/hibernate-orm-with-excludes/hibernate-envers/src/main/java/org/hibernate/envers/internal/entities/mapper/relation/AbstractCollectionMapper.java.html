41  public abstract class AbstractCollectionMapper<T> extends AbstractPropertyMapper {
42  	protected final CommonCollectionMapperData commonCollectionMapperData;
43  	protected final Class<? extends T> collectionClass;
44  	protected final boolean ordinalInId;
45  	protected final boolean revisionTypeInId;
46  
47  	private final Constructor<? extends T> proxyConstructor;
48  
49  	protected AbstractCollectionMapper(
50  			CommonCollectionMapperData commonCollectionMapperData,
51  			Class<? extends T> collectionClass,
52  			Class<? extends T> proxyClass,
53  			boolean ordinalInId,
54  			boolean revisionTypeInId) {
55  		this.commonCollectionMapperData = commonCollectionMapperData;
56  		this.collectionClass = collectionClass;
57  		this.ordinalInId = ordinalInId;
58  		this.revisionTypeInId = revisionTypeInId;
59  
60  		try {
61  			proxyConstructor = proxyClass.getConstructor( Initializor.class );
62  		}
63  		catch (NoSuchMethodException e) {
64  			throw new AuditException( e );
65  		}
66  	}

74  	/**
75  	 * Maps the changed collection element to the given map.
76  	 *
77  	 * @param idData Map to which composite-id data should be added.
78  	 * @param data Where to map the data.
79  	 * @param changed The changed collection element to map.
80  	 */
81  	protected abstract void mapToMapFromObject(
82  			SessionImplementor session,
83  			Map<String, Object> idData,
84  			Map<String, Object> data,
85  			Object changed);

87  	/**
88  	 * Creates map for storing identifier data. Ordinal parameter guarantees uniqueness of primary key.
89  	 * Composite primary key cannot contain embeddable properties since they might be nullable.
90  	 *
91  	 * @param ordinal Iteration ordinal.
92  	 *
93  	 * @return Map for holding identifier data.
94  	 */
95  	protected Map<String, Object> createIdMap(int ordinal) {
96  		final Map<String, Object> idMap = new HashMap<>();
97  		if ( ordinalInId ) {
98  			idMap.put( commonCollectionMapperData.getVerEntCfg().getEmbeddableSetOrdinalPropertyName(), ordinal );
99  		}
100 		return idMap;
101 	}
102 
103 	protected void addCollectionChanges(
104 			SessionImplementor session,
105 			List<PersistentCollectionChangeData> collectionChanges,
106 			Set<Object> changed,
107 			RevisionType revisionType,
108 			Serializable id) {
109 		int ordinal = 0;
110 
111 		for ( Object changedObj : changed ) {
112 			final Map<String, Object> entityData = new HashMap<>();
113 			final Map<String, Object> originalId = createIdMap( ordinal++ );
114 			entityData.put( commonCollectionMapperData.getVerEntCfg().getOriginalIdPropName(), originalId );
115 
116 			collectionChanges.add(
117 					new PersistentCollectionChangeData(
118 							commonCollectionMapperData.getVersionsMiddleEntityName(), entityData, changedObj
119 					)
120 			);
121 			// Mapping the collection owner's id.
122 			commonCollectionMapperData.getReferencingIdData().getPrefixedMapper().mapToMapFromId( originalId, id );
123 
124 			// Mapping collection element and index (if present).
125 			mapToMapFromObject( session, originalId, entityData, changedObj );
126 
127 			( revisionTypeInId ? originalId : entityData ).put(
128 					commonCollectionMapperData.getVerEntCfg()
129 							.getRevisionTypePropName(), revisionType
130 			);
131 		}
132 	}

134 	@Override
135 	@SuppressWarnings({"unchecked"})
136 	public List<PersistentCollectionChangeData> mapCollectionChanges(
137 			SessionImplementor session,
138 			String referencingPropertyName,
139 			PersistentCollection newColl,
140 			Serializable oldColl,
141 			Serializable id) {
142 		final PropertyData propertyData = commonCollectionMapperData.getCollectionReferencingPropertyData();
143 		if ( !propertyData.getName().equals( referencingPropertyName ) ) {
144 			return null;
145 		}
146 
147 		return mapCollectionChanges( session, newColl, oldColl, id );
148 	}

160 	@Override
161 	public void mapModifiedFlagsToMapFromEntity(
162 			SessionImplementor session,
163 			Map<String, Object> data,
164 			Object newObj,
165 			Object oldObj) {
166 		final PropertyData propertyData = commonCollectionMapperData.getCollectionReferencingPropertyData();
167 		if ( propertyData.isUsingModifiedFlag() ) {
168 			if ( isNotPersistentCollection( newObj ) || isNotPersistentCollection( oldObj ) ) {
169 				// Compare POJOs.
170 				data.put( propertyData.getModifiedFlagPropertyName(), !Objects.deepEquals( newObj, oldObj ) );
171 			}
172 			else if ( isFromNullToEmptyOrFromEmptyToNull( (PersistentCollection) newObj, (Serializable) oldObj ) ) {
173 				data.put( propertyData.getModifiedFlagPropertyName(), true );
174 			}
175 			else {
176 				// HHH-7949 - Performance optimization to avoid lazy-fetching collections that have
177 				// not been changed for deriving the modified flags value.
178 				final PersistentCollection pc = (PersistentCollection) newObj;
179 				if ( ( pc != null && !pc.isDirty() ) || ( newObj == null && oldObj == null ) ) {
180 					data.put( propertyData.getModifiedFlagPropertyName(), false );
181 					return;
182 				}
183 
184 				final List<PersistentCollectionChangeData> changes = mapCollectionChanges(
185 						session,
186 						commonCollectionMapperData.getCollectionReferencingPropertyData().getName(),
187 						pc,
188 						(Serializable) oldObj,
189 						null
190 				);
191 				data.put( propertyData.getModifiedFlagPropertyName(), !changes.isEmpty() );
192 			}
193 		}
194 	}
195 
196 	private boolean isNotPersistentCollection(Object obj) {
197 		return obj != null && !(obj instanceof PersistentCollection);
198 	}
199 
200 	private boolean isFromNullToEmptyOrFromEmptyToNull(PersistentCollection newColl, Serializable oldColl) {
201 		// Comparing new and old collection content.
202 		final Collection newCollection = getNewCollectionContent( newColl );
203 		final Collection oldCollection = getOldCollectionContent( oldColl );
204 
205 		return oldCollection == null && newCollection != null && newCollection.isEmpty()
206 				|| newCollection == null && oldCollection != null && oldCollection.isEmpty();
207 	}

209 	@Override
210 	public void mapModifiedFlagsToMapForCollectionChange(String collectionPropertyName, Map<String, Object> data) {
211 		final PropertyData propertyData = commonCollectionMapperData.getCollectionReferencingPropertyData();
212 		if ( propertyData.isUsingModifiedFlag() ) {
213 			data.put(
214 					propertyData.getModifiedFlagPropertyName(),
215 					propertyData.getName().equals( collectionPropertyName )
216 			);
217 		}
218 	}

227 	protected CollectionPersister resolveCollectionPersister(
228 			SessionImplementor session,
229 			PersistentCollection collection) {
230 		// First attempt to resolve the persister from the collection entry
231 		if ( collection != null ) {
232 			CollectionEntry collectionEntry = session.getPersistenceContextInternal().getCollectionEntry( collection );
233 			if ( collectionEntry != null ) {
234 				CollectionPersister collectionPersister = collectionEntry.getCurrentPersister();
235 				if ( collectionPersister != null ) {
236 					return collectionPersister;
237 				}
238 			}
239 		}
240 
241 		// Fallback to resolving the persister from the collection role
242 		final CollectionPersister collectionPersister = session.getFactory()
243 				.getMetamodel()
244 				.collectionPersister( commonCollectionMapperData.getRole() );
245 
246 		if ( collectionPersister == null ) {
247 			throw new AuditException(
248 					String.format(
249 							Locale.ROOT,
250 							"Failed to locate CollectionPersister for collection [%s]",
251 							commonCollectionMapperData.getRole()
252 					)
253 			);
254 		}
255 
256 		return collectionPersister;
257 	}

259 	/**
260 	 * Checks whether the old collection element and new collection element are the same.
261 	 * By default, this delegates to the collection persister's {@link CollectionPersister#getElementType()}.
262 	 *
263 	 * @param collectionPersister The collection persister.
264 	 * @param oldObject The collection element from the old persistent collection.
265 	 * @param newObject The collection element from the new persistent collection.
266 	 *
267 	 * @return true if the two objects are the same, false otherwise.
268 	 */
269 	protected boolean isSame(CollectionPersister collectionPersister, Object oldObject, Object newObject) {
270 		return collectionPersister.getElementType().isSame( oldObject, newObject );
271 	}

273 	@Override
274 	public void mapToEntityFromMap(
275 			final EnversService enversService,
276 			final Object obj,
277 			final Map data,
278 			final Object primaryKey,
279 			final AuditReaderImplementor versionsReader,
280 			final Number revision) {
281 		final String revisionTypePropertyName = enversService.getAuditEntitiesConfiguration().getRevisionTypePropName();
282 
283 		// construct the collection proxy
284 		final Object collectionProxy;
285 		try {
286 			collectionProxy = proxyConstructor.newInstance(
287 					getInitializor(
288 							enversService,
289 							versionsReader,
290 							primaryKey,
291 							revision,
292 							RevisionType.DEL.equals( data.get( revisionTypePropertyName ) )
293 					)
294 			);
295 		}
296 		catch ( Exception e ) {
297 			throw new AuditException( "Failed to construct collection proxy", e );
298 		}
299 
300 		final PropertyData collectionPropertyData = commonCollectionMapperData.getCollectionReferencingPropertyData();
301 
302 		if ( isDynamicComponentMap() ) {
303 			final Map<String, Object> map = (Map<String, Object>) obj;
304 			map.put( collectionPropertyData.getBeanName(), collectionProxy );
305 		}
306 		else {
307 			AccessController.doPrivileged(
308 					new PrivilegedAction<Object>() {
309 						@Override
310 						public Object run() {
311 							final Setter setter = ReflectionTools.getSetter(
312 									obj.getClass(),
313 									collectionPropertyData,
314 									enversService.getServiceRegistry()
315 							);
316 
317 							setter.set( obj, collectionProxy, null );
318 
319 							return null;
320 						}
321 					}
322 			);
323 		}
324 	}

342 	public boolean hasPropertiesWithModifiedFlag() {
343 		if ( commonCollectionMapperData != null ) {
