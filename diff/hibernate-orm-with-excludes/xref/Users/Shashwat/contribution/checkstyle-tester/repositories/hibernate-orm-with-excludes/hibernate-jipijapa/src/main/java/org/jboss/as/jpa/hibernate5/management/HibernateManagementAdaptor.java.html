8   package org.jboss.as.jpa.hibernate5.management;
9   
10  import org.jipijapa.management.spi.Statistics;
11  import org.jipijapa.plugin.spi.ManagementAdaptor;
12  
13  /**
14   * Contains management support for Hibernate
15   *
16   * @author Scott Marlow
17   */
18  public class HibernateManagementAdaptor implements ManagementAdaptor {
19  
20  	// shared (per classloader) instance for all Hibernate 4.3 JPA deployments
21  	private static final HibernateManagementAdaptor INSTANCE = new HibernateManagementAdaptor();
22  
23  	private final Statistics statistics = new HibernateStatistics();
24  
25  	private static final String PROVIDER_LABEL = "hibernate-persistence-unit";
26  	private static final String VERSION = "Hibernate ORM 4.3.x";
27  
28  	/**
29  	 * The management statistics are shared across all Hibernate 4 JPA deployments
30  	 *
31  	 * @return shared instance for all Hibernate 4 JPA deployments
32  	 */
33  	public static HibernateManagementAdaptor getInstance() {
34  		return INSTANCE;
35  	}
36  
37  	@Override
38  	public String getIdentificationLabel() {
39  		return PROVIDER_LABEL;
40  	}
41  
42  	@Override
43  	public String getVersion() {
44  		return VERSION;
45  	}
46  
47  	@Override
48  	public Statistics getStatistics() {
49  		return statistics;
50  	}
51  
52  
53  }
