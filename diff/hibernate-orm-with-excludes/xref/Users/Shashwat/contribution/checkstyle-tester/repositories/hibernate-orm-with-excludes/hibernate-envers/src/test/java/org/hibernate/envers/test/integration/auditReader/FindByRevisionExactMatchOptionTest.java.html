7   package org.hibernate.envers.test.integration.auditReader;
8   
9   import java.util.Arrays;
10  import java.util.Map;
11  
12  import org.hibernate.envers.AuditReader;
13  import org.hibernate.envers.configuration.EnversSettings;
14  import org.hibernate.envers.test.BaseEnversJPAFunctionalTestCase;
15  import org.hibernate.envers.test.Priority;
16  import org.hibernate.envers.test.entities.IntNoAutoIdTestEntity;
17  import org.junit.Test;
18  
19  import org.hibernate.testing.TestForIssue;
20  
21  import static junit.framework.TestCase.assertNull;
22  import static org.hibernate.testing.transaction.TransactionUtil.doInJPA;
23  import static org.junit.Assert.assertEquals;
24  
25  /**
26   * A test which verifies the behavior of the various {@link AuditReader} find implementations when the
27   * configuration option {@link EnversSettings#FIND_BY_REVISION_EXACT_MATCH} is enabled.
28   *
29   * @author Chris Cranford
30   */
31  @TestForIssue(jiraKey = "HHH-13500")
32  public class FindByRevisionExactMatchOptionTest extends BaseEnversJPAFunctionalTestCase {
33  	@Override
34  	protected void addConfigOptions(Map options) {
35  		super.addConfigOptions( options );
36  
37  		options.put( EnversSettings.FIND_BY_REVISION_EXACT_MATCH, "true" );
38  	}
39  
40  	@Override
41  	protected Class<?>[] getAnnotatedClasses() {
42  		return new Class<?>[] { IntNoAutoIdTestEntity.class };
43  	}
44  
45  	@Priority(10)
46  	@Test
47  	public void initData() {
48  		// Insert entity with id=1, numVal=1, revision 1
49  		doInJPA( this::entityManagerFactory, entityManager -> {
50  			final IntNoAutoIdTestEntity entity = new IntNoAutoIdTestEntity( 1, 1 );
51  			entityManager.persist( entity );
52  		} );
53  
54  		// Update entity with id=1, setting numVal=11, revision 2
55  		doInJPA( this::entityManagerFactory, entityManager -> {
56  			final IntNoAutoIdTestEntity entity = entityManager.find( IntNoAutoIdTestEntity.class, 1 );
57  			entity.setNumVal( 11 );
58  			entityManager.merge( entity );
59  		} );
60  
61  		// Insert entity with id=2, numVal=2, revision 3
62  		doInJPA( this::entityManagerFactory, entityManager -> {
63  			final IntNoAutoIdTestEntity entity = new IntNoAutoIdTestEntity( 2, 2 );
64  			entityManager.persist( entity );
65  		} );
66  
67  		// Update entity with id=2, setting numVal=22, revision 4
68  		doInJPA( this::entityManagerFactory, entityManager -> {
69  			final IntNoAutoIdTestEntity entity = entityManager.find( IntNoAutoIdTestEntity.class, 2 );
70  			entity.setNumVal( 22 );
71  			entityManager.merge( entity );
72  		} );
73  	}
74  
75  	@Test
76  	public void testRevisionCounts() {
77  		assertEquals( Arrays.asList( 1, 2 ), getAuditReader().getRevisions( IntNoAutoIdTestEntity.class, 1 ) );
78  		assertEquals( Arrays.asList( 3, 4 ), getAuditReader().getRevisions( IntNoAutoIdTestEntity.class, 2 ) );
79  	}
80  
81  	@Test
82  	public void testFindEntityId1() {
83  		final AuditReader auditReader = getAuditReader();
84  		assertEquals( new IntNoAutoIdTestEntity( 1, 1 ), auditReader.find( IntNoAutoIdTestEntity.class, 1, 1 ) );
85  		assertEquals( new IntNoAutoIdTestEntity( 11, 1 ), auditReader.find( IntNoAutoIdTestEntity.class, 1, 2 ) );
86  		assertNull( auditReader.find( IntNoAutoIdTestEntity.class, 1, 3 ) );
87  		assertNull( auditReader.find( IntNoAutoIdTestEntity.class, 1, 4 ) );
88  	}
89  
90  	@Test
91  	public void testFindEntityId2() {
92  		final AuditReader auditReader = getAuditReader();
93  		assertNull( auditReader.find( IntNoAutoIdTestEntity.class, 2, 1 ) );
94  		assertNull( auditReader.find( IntNoAutoIdTestEntity.class, 2, 2 ) );
95  		assertEquals( new IntNoAutoIdTestEntity( 2, 2 ), auditReader.find( IntNoAutoIdTestEntity.class, 2, 3 ) );
96  		assertEquals( new IntNoAutoIdTestEntity( 22, 2 ), auditReader.find( IntNoAutoIdTestEntity.class, 2, 4 ) );
97  	}
98  }
