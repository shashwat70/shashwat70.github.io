39  @TestForIssue( jiraKey = "HHH-13365" )
40  @RequiresDialect( Oracle9iDialect.class )
41  public class JoinedSequenceIdentityBatchTest extends BaseNonConfigCoreFunctionalTestCase {
42  
43  	@Test
44  	public void testInsertAndUpdate() {
45  		doInHibernate(
46  				this::sessionFactory,
47  				session -> {
48  					FolderResource folder = new FolderResource();
49  					folder.name = "PARENT";
50  					session.persist( folder );
51  				}
52  		);
53  
54  		doInHibernate(
55  				this::sessionFactory,
56  				session -> {
57  					List<FolderResource> folderResources = session.createQuery( "from FolderResource" ).getResultList();
58  					assertEquals( 1, folderResources.size() );
59  					final FolderResource folderResource = folderResources.get( 0 );
60  					assertNull( folderResource.description );
61  					folderResource.description = "A folder resource";
62  			}
63  		);
64  
65  		doInHibernate(
66  				this::sessionFactory,
67  				session -> {
68  					List<FolderResource> folderResources = session.createQuery( "from FolderResource" ).getResultList();
69  					assertEquals( 1, folderResources.size() );
70  					final FolderResource folderResource = folderResources.get( 0 );
71  					assertEquals( "A folder resource", folderResource.description );
72  				}
73  		);
74  	}
75  
76  	@Override
77  	@SuppressWarnings( "unchecked" )
78  	protected void addSettings(Map settings) {
79  		super.addSettings( settings );
80  		settings.put( AvailableSettings.STATEMENT_BATCH_SIZE, "5" );
81  		settings.put( AvailableSettings.USE_GET_GENERATED_KEYS, "true" );
82  	}
83  
84  	@Override
85  	protected Class[] getAnnotatedClasses() {
86  		return new Class[] { Resource.class, FolderResource.class };
87  	}
88  
89  	@Entity(name = "Resource")
90  	@Inheritance(strategy = InheritanceType.JOINED)
91  	@Table(name = "WORKSPACE_RESOURCE")
92  	@DiscriminatorColumn(name = "type", discriminatorType = DiscriminatorType.STRING)
93  	public static class Resource {
94  
95  		@Id
96      		@GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "SEQ_WORKSPACE_RESOURCE")
97      		@GenericGenerator(name = "SEQ_WORKSPACE_RESOURCE", strategy = "sequence-identity",
98  					parameters = {@Parameter(name = "sequence", value = "SEQ_WORKSPACE_RESOURCE")})
99  		@Column(name = "ID_WORKSPACE_RESOURCE", nullable = false, precision = 18)
100 		private Long id;
101 
102 		@Column(name = "NAME", nullable = false, length = 256)
103 		protected String name;
104 
105 		@Column(name = "RESOURCE_TYPE", nullable = false, length = 20)
106 		protected String type;
107 	}
108 
109 	@Entity(name = "FolderResource")
110 	@Table(name = "WORKSPACE_RESOURCE_FOLDER")
111 	@PrimaryKeyJoinColumn(name = "ID_WORKSPACE_RESOURCE_FOLDER")
112 	@DiscriminatorValue("FOLDER")
113 	public static class FolderResource extends Resource implements Serializable {
114 		private String description;
115 
116 		public FolderResource()
117 		{
118 			super();
119 			type = "FOLDER";
120 		}
121 	}
122 }
