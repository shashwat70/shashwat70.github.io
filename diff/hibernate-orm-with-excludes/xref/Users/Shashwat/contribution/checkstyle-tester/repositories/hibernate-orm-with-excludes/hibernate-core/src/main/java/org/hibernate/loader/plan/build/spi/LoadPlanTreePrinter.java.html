7   package org.hibernate.loader.plan.build.spi;
8   
9   import java.io.ByteArrayOutputStream;
10  import java.io.PrintStream;
11  import java.io.PrintWriter;
12  
13  import org.hibernate.internal.CoreLogging;
14  import org.hibernate.loader.plan.exec.spi.AliasResolutionContext;
15  import org.hibernate.loader.plan.spi.CollectionReturn;
16  import org.hibernate.loader.plan.spi.EntityReturn;
17  import org.hibernate.loader.plan.spi.LoadPlan;
18  import org.hibernate.loader.plan.spi.Return;
19  
20  import org.jboss.logging.Logger;
21  
22  /**
23   * Prints a {@link org.hibernate.loader.plan.spi.LoadPlan} graph and its
24   * {@link org.hibernate.loader.plan.spi.QuerySpaces} graph as tree structures.
25   * <p/>
26   * Intended for use in debugging, logging, etc.
27   * <p/>
28   * Aggregates calls to the {@link QuerySpaceTreePrinter} and {@link ReturnGraphTreePrinter}
29   *
30   * @author Steve Ebersole
31   */
32  public class LoadPlanTreePrinter {
33  	private static final Logger log = CoreLogging.logger( LoadPlanTreePrinter.class );
34  
35  	/**
36  	 * Singleton access
37  	 */
38  	public static final LoadPlanTreePrinter INSTANCE = new LoadPlanTreePrinter();
39  
40  	private LoadPlanTreePrinter() {
41  	}
42  
43  	/**
44  	 * Logs the specified {@link org.hibernate.loader.plan.spi.LoadPlan} graph and its
45  	 * {@link org.hibernate.loader.plan.spi.QuerySpaces} graph as tree structures.
46  	 *
47  	 * @param loadPlan The load plan.
48  	 * @param aliasResolutionContext The context for resolving table and column aliases/
49  	 */
50  	public void logTree(LoadPlan loadPlan, AliasResolutionContext aliasResolutionContext) {
51  		if ( ! log.isDebugEnabled() ) {
52  			return;
53  		}
54  
55  		log.debug( toString( loadPlan, aliasResolutionContext ) );
56  	}
57  
58  	private String toString(LoadPlan loadPlan, AliasResolutionContext aliasResolutionContext) {
59  		final ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
60  		final PrintStream printStream = new PrintStream( byteArrayOutputStream );
61  		final PrintWriter printWriter = new PrintWriter( printStream );
62  
63  		logTree( loadPlan, aliasResolutionContext, printWriter );
64  
65  		printWriter.flush();
66  		printStream.flush();
67  
68  		return new String( byteArrayOutputStream.toByteArray() );
69  	}
70  
71  	private void logTree(
72  			LoadPlan loadPlan,
73  			AliasResolutionContext aliasResolutionContext,
74  			PrintWriter printWriter) {
75  		printWriter.println( "LoadPlan(" + extractDetails( loadPlan ) + ")" );
76  		printWriter.println( TreePrinterHelper.INSTANCE.generateNodePrefix( 1 ) + "Returns" );
77  		for ( Return rtn : loadPlan.getReturns() ) {
78  			ReturnGraphTreePrinter.INSTANCE.write( rtn, 2, printWriter );
79  			printWriter.flush();
80  		}
81  
82  		QuerySpaceTreePrinter.INSTANCE.write( loadPlan.getQuerySpaces(), 1, aliasResolutionContext, printWriter );
83  
84  		printWriter.flush();
85  	}
86  
87  	private String extractDetails(LoadPlan loadPlan) {
88  		switch ( loadPlan.getDisposition() ) {
89  			case MIXED: {
90  				return "mixed";
91  			}
92  			case ENTITY_LOADER: {
93  				return "entity=" + ( (EntityReturn) loadPlan.getReturns().get( 0 ) ).getEntityPersister().getEntityName();
94  			}
95  			case COLLECTION_INITIALIZER: {
96  				return "collection=" + ( (CollectionReturn) loadPlan.getReturns().get( 0 ) ).getCollectionPersister().getRole();
97  			}
98  			default: {
99  				return "???";
100 			}
101 		}
102 	}
103 }
