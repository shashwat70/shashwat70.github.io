7   package org.hibernate.resource.jdbc;
8   
9   import java.sql.Connection;
10  
11  /**
12   * Models the logical notion of a JDBC Connection.  We may release/re-acquire physical JDBC connections under the
13   * covers, but this logically represents the overall access to the JDBC Connection.
14   *
15   * @author Steve Ebersole
16   */
17  public interface LogicalConnection {
18  	/**
19  	 * Is this (logical) JDBC Connection still open/active.  In other words, has {@link #close} not been called yet?
20  	 *
21  	 * @return {@code true} if still open ({@link #close} has not been called yet); {@code false} if not open
22  	 * ({@link #close} has been called).
23  	 */
24  	boolean isOpen();
25  
26  	/**
27  	 * Closes the JdbcSession, making it inactive and forcing release of any held resources
28  	 *
29  	 * @return Legacy :(  Returns the JDBC Connection *if* the user passed in a Connection originally.
30  	 */
31  	Connection close();
32  
33  	/**
34  	 * Is this JdbcSession currently physically connected (meaning does it currently hold a JDBC Connection)?
35  	 *
36  	 * @return {@code true} if the JdbcSession currently hold a JDBC Connection; {@code false} if it does not.
37  	 */
38  	boolean isPhysicallyConnected();
39  
40  	/**
41  	 * Provides access to the registry of JDBC resources associated with this LogicalConnection.
42  	 *
43  	 * @return The JDBC resource registry.
44  	 *
45  	 * @throws org.hibernate.ResourceClosedException if the LogicalConnection is closed
46  	 */
47  	ResourceRegistry getResourceRegistry();
48  
49  }
