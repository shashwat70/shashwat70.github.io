22  
23  import static org.hibernate.testing.transaction.TransactionUtil.doInHibernate;
24  import static org.junit.Assert.assertEquals;
25  import static org.junit.Assert.assertFalse;
26  import static org.junit.Assert.assertNotNull;
27  
28  /**
29   * @author Vlad Mihalcea
30   */
31  public class AtomikosJtaLazyLoadingTest
32  		extends BaseCoreFunctionalTestCase {
33  
34  	private static final int CHILDREN_SIZE = 3;
35  	private Long parentID;
36  	private Long lastChildID;
37  
38  	protected void configure(Configuration cfg) {
39  		super.configure( cfg );
40  		cfg.setProperty( Environment.ENABLE_LAZY_LOAD_NO_TRANS, "true" );
41  
42  		TestingJtaBootstrap.prepare( cfg.getProperties() );
43  		cfg.setProperty( AvailableSettings.JTA_PLATFORM, "Atomikos" );
44  	}
45  
46  	protected Class<?>[] getAnnotatedClasses() {
47  		return new Class<?>[] {
48  				Parent.class,
49  				Child.class
50  		};
51  	}
52  
53  	protected void prepareTest()
54  			throws Exception {
55  		doInHibernate( this::sessionFactory, session -> {
56  			Parent p = new Parent();
57  			for ( int i = 0; i < CHILDREN_SIZE; i++ ) {
58  				final Child child = p.makeChild();
59  				session.persist( child );
60  				lastChildID = child.getId();
61  			}
62  			session.persist( p );
63  			parentID = p.getId();
64  		} );
65  	}
66  
67  	@Test
68  	@TestForIssue(jiraKey = "HHH-7971")
69  	public void testLazyCollectionLoadingAfterEndTransaction() {
70  		Parent loadedParent = doInHibernate( this::sessionFactory, session -> {
71  			return session.load( Parent.class, parentID );
72  		} );
73  
74  		assertFalse( Hibernate.isInitialized( loadedParent.getChildren() ) );
75  
76  		int i = 0;
77  		for ( Child child : loadedParent.getChildren() ) {
78  			i++;
79  			assertNotNull( child );
80  		}
81  
82  		assertEquals( CHILDREN_SIZE, i );
83  
84  		Child loadedChild = doInHibernate( this::sessionFactory, session -> {
85  			return session.load( Child.class, lastChildID );
86  		} );
87  
88  		Parent p = loadedChild.getParent();
89  		int j = 0;
90  		for ( Child child : p.getChildren() ) {
91  			j++;
92  			assertNotNull( child );
93  		}
94  
95  		assertEquals( CHILDREN_SIZE, j );
96  	}
97  
98  }
