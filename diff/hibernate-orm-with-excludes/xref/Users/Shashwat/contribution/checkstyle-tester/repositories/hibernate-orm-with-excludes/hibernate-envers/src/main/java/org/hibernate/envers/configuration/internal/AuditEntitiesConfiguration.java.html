7   package org.hibernate.envers.configuration.internal;
8   
9   import java.util.HashMap;
10  import java.util.Map;
11  import java.util.Properties;
12  
13  import org.hibernate.envers.boot.internal.EnversService;
14  import org.hibernate.envers.configuration.EnversSettings;
15  import org.hibernate.envers.strategy.DefaultAuditStrategy;
16  import org.hibernate.internal.util.config.ConfigurationHelper;
17  
18  /**
19   * Configuration of versions entities - names of fields, entities and tables created to store versioning information.
20   *
21   * @author Adam Warski (adam at warski dot org)
22   * @author Stephanie Pau at Markit Group Plc
23   * @author Chris Cranford
24   */
25  public class  AuditEntitiesConfiguration {
26  	private final String auditTablePrefix;
27  	private final String auditTableSuffix;
28  
29  	private final String auditStrategyName;
30  	private final String originalIdPropName;
31  
32  	private final String revisionFieldName;
33  	private final String revisionNumberPath;
34  	private final String revisionPropBasePath;
35  
36  	private final String revisionTypePropName;
37  	private final String revisionTypePropType;
38  
39  	private final String revisionInfoEntityName;
40  
41  	private final Map<String, String> customAuditTablesNames;
42  
43  	private final String revisionEndFieldName;
44  
45  	private final boolean revisionEndTimestampEnabled;
46  	private final String revisionEndTimestampFieldName;
47  
48  	private final String embeddableSetOrdinalPropertyName;
49  	private final EnversService enversService;
50  
51  	public AuditEntitiesConfiguration(
52  			Properties properties,
53  			String revisionInfoEntityName,
54  			EnversService enversService) {
55  		this.revisionInfoEntityName = revisionInfoEntityName;
56  		this.enversService = enversService;
57  
58  		auditTablePrefix = ConfigurationHelper.getString( EnversSettings.AUDIT_TABLE_PREFIX, properties, "" );
59  		auditTableSuffix = ConfigurationHelper.getString( EnversSettings.AUDIT_TABLE_SUFFIX, properties, "_AUD" );
60  
61  		auditStrategyName = ConfigurationHelper.getString(
62  				EnversSettings.AUDIT_STRATEGY, properties, DefaultAuditStrategy.class.getName()
63  		);
64  
65  		originalIdPropName = ConfigurationHelper.getString(
66  				EnversSettings.ORIGINAL_ID_PROP_NAME, properties, "originalId"
67  		);
68  
69  		revisionFieldName = ConfigurationHelper.getString( EnversSettings.REVISION_FIELD_NAME, properties, "REV" );
70  
71  		revisionTypePropName = ConfigurationHelper.getString(
72  				EnversSettings.REVISION_TYPE_FIELD_NAME, properties, "REVTYPE"
73  		);
74  		revisionTypePropType = "byte";
75  
76  		revisionEndFieldName = ConfigurationHelper.getString(
77  				EnversSettings.AUDIT_STRATEGY_VALIDITY_END_REV_FIELD_NAME, properties, "REVEND"
78  		);
79  
80  		revisionEndTimestampEnabled = ConfigurationHelper.getBoolean(
81  				EnversSettings.AUDIT_STRATEGY_VALIDITY_STORE_REVEND_TIMESTAMP, properties, false
82  		);
83  
84  		if ( revisionEndTimestampEnabled ) {
85  			revisionEndTimestampFieldName = ConfigurationHelper.getString(
86  					EnversSettings.AUDIT_STRATEGY_VALIDITY_REVEND_TIMESTAMP_FIELD_NAME, properties, "REVEND_TSTMP"
87  			);
88  		}
89  		else {
90  			revisionEndTimestampFieldName = null;
91  		}
92  
93  		customAuditTablesNames = new HashMap<>();
94  
95  		revisionNumberPath = originalIdPropName + "." + revisionFieldName + ".id";
96  		revisionPropBasePath = originalIdPropName + "." + revisionFieldName + ".";
97  
98  		embeddableSetOrdinalPropertyName = ConfigurationHelper.getString(
99  				EnversSettings.EMBEDDABLE_SET_ORDINAL_FIELD_NAME, properties, "SETORDINAL"
100 		);
101 	}
102 
103 	public String getOriginalIdPropName() {
104 		return originalIdPropName;
105 	}
106 
107 	public String getRevisionFieldName() {
108 		return revisionFieldName;
109 	}
110 
111 	public boolean isRevisionEndTimestampEnabled() {
112 		return revisionEndTimestampEnabled;
113 	}
114 
115 	public String getRevisionEndTimestampFieldName() {
116 		return revisionEndTimestampFieldName;
117 	}
118 
119 	public String getRevisionNumberPath() {
120 		return revisionNumberPath;
121 	}
122 
123 	/**
124 	 * @param propertyName Property of the revision entity.
125 	 *
126 	 * @return A path to the given property of the revision entity associated with an audit entity.
127 	 */
128 	public String getRevisionPropPath(String propertyName) {
129 		return revisionPropBasePath + propertyName;
130 	}
131 
132 	public String getRevisionTypePropName() {
133 		return revisionTypePropName;
134 	}
135 
136 	public String getRevisionTypePropType() {
137 		return revisionTypePropType;
138 	}
139 
140 	public String getRevisionInfoEntityName() {
141 		return revisionInfoEntityName;
142 	}
143 
144 	public void addCustomAuditTableName(String entityName, String tableName) {
145 		customAuditTablesNames.put( entityName, tableName );
146 	}
147 
148 	public String getAuditEntityName(String entityName) {
149 		return auditTablePrefix + entityName + auditTableSuffix;
150 	}
151 
152 	public String getAuditTableName(String entityName, String tableName) {
153 		final String customHistoryTableName = customAuditTablesNames.get( entityName );
154 		if ( customHistoryTableName == null ) {
155 			return auditTablePrefix + tableName + auditTableSuffix;
156 		}
157 
158 		return customHistoryTableName;
159 	}
160 
161 	public String getAuditStrategyName() {
162 		return auditStrategyName;
163 	}
164 
165 	public String getRevisionEndFieldName() {
166 		return revisionEndFieldName;
167 	}
168 
169 	public String getEmbeddableSetOrdinalPropertyName() {
170 		return embeddableSetOrdinalPropertyName;
171 	}
172 
173 	/**
174 	 * @deprecated (since 5.2.1), while actually added in 5.2.1, this was added to cleanup the
175 	 * audit strategy interface temporarily.
176 	 */
177 	@Deprecated
178 	public EnversService getEnversService() {
179 		return enversService;
180 	}
181 }
