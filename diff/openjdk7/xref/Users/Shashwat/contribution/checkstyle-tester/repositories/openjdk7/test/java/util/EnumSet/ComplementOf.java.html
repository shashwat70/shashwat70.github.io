33  public class ComplementOf {
34      public static void main(String[] args) throws Exception {
35          test(Test0.class);
36          test(MonthTest10.class);
37          test(Test127.class);
38          test(Test128.class);
39      }

41      static <T extends Enum<T>> void test(Class<T> enumClass) {
42          EnumSet<T> set = EnumSet.allOf(enumClass);
43          EnumSet<T> setComplement = EnumSet.complementOf(set);
44          if (setComplement.size() != 0)
45              throw new RuntimeException(setComplement.size() + " != 0");
46          setComplement.toString();
47      }
