93      /**
94       * Write and read class descriptors to/from a stream.
95       * Arguments: <# cycles>
96       */
97      public long run(String[] args) throws Exception {
98          int ncycles = Integer.parseInt(args[0]);
99          StreamBuffer sbuf = new StreamBuffer();
100         ObjectOutputStream oout =
101             new ObjectOutputStream(sbuf.getOutputStream());
102         ObjectInputStream oin =
103             new ObjectInputStream(sbuf.getInputStream());
104         ObjectStreamClass desc = ObjectStreamClass.lookup(Dummy50.class);
105 
106         doReps(oout, oin, sbuf, desc, 1);       // warmup
107 
108         long start = System.currentTimeMillis();
109         doReps(oout, oin, sbuf, desc, ncycles);
110         return System.currentTimeMillis() - start;
111     }
112 
113     /**
114      * Run benchmark for given number of cycles.
115      */
116     void doReps(ObjectOutputStream oout, ObjectInputStream oin,
117                 StreamBuffer sbuf, ObjectStreamClass desc, int ncycles)
118         throws Exception
119     {
120         for (int i = 0; i < ncycles; i++) {
121             sbuf.reset();
122             oout.reset();
123             oout.writeObject(desc);
124             oout.flush();
125             oin.readObject();
126         }
127     }
