26  package sun.awt.windows;
27  
28  import java.awt.KeyboardFocusManager;
29  import java.awt.Window;
30  import java.awt.Component;
31  import java.awt.peer.ComponentPeer;
32  import sun.awt.KeyboardFocusManagerPeerImpl;
33  import sun.awt.CausedFocusEvent;
34  
35  class WKeyboardFocusManagerPeer extends KeyboardFocusManagerPeerImpl {
36      static native void setNativeFocusOwner(ComponentPeer peer);
37      static native Component getNativeFocusOwner();
38      static native Window getNativeFocusedWindow();
39  
40      WKeyboardFocusManagerPeer(KeyboardFocusManager manager) {
41          super(manager);
42      }
43  
44      @Override
45      public void setCurrentFocusOwner(Component comp) {
46          setNativeFocusOwner(comp != null ? comp.getPeer() : null);
47      }
48  
49      @Override
50      public Component getCurrentFocusOwner() {
51          return getNativeFocusOwner();
52      }
53  
54      @Override
55      public Window getCurrentFocusedWindow() {
56          return getNativeFocusedWindow();
57      }
58  
59      public static boolean deliverFocus(Component lightweightChild,
60                                         Component target,
61                                         boolean temporary,
62                                         boolean focusedWindowChangeAllowed,
63                                         long time,
64                                         CausedFocusEvent.Cause cause)
65      {
66          // TODO: do something to eliminate this forwarding
67          return KeyboardFocusManagerPeerImpl.deliverFocus(lightweightChild,
68                                                           target,
69                                                           temporary,
70                                                           focusedWindowChangeAllowed,
71                                                           time,
72                                                           cause,
73                                                           getNativeFocusOwner());
74      }
75  }
