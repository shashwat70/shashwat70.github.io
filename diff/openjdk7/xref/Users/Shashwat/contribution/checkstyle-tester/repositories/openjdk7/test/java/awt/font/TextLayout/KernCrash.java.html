24  import java.io.*;
25  import java.awt.*;
26  import java.awt.font.*;
27  import java.util.*;
28  
29  /**
30   * Shows (top) with kerning, (middle) without, (bottom) also without.
31   *
32   * @bug 7017324
33   */
34  public class KernCrash extends Frame {
35      private static Font font0;
36      private static Font font1;
37      private static Font font2;
38  
39      public static void main(String[] args) throws Exception {
40          HashMap attrs = new HashMap();
41          font0 = Font.createFont(Font.TRUETYPE_FONT, new File("Vera.ttf"));
42          System.out.println("using " + font0);
43          attrs.put(TextAttribute.SIZE, new Float(58f));
44          font1 = font0.deriveFont(attrs);
45          attrs.put(TextAttribute.KERNING, TextAttribute.KERNING_ON);
46          font2 = font0.deriveFont(attrs);
47  
48          KernCrash f = new KernCrash();
49          f.setTitle("Kerning Crash");
50          f.setSize(600, 300);
51          f.setForeground(Color.black);
52          f.show();
53      }
54  
55      public void paint(Graphics g) {
56          Graphics2D g2 = (Graphics2D)g;
57          FontRenderContext frc = g2.getFontRenderContext();
58          TextLayout layout = new TextLayout("text", font2, frc);
59          layout.draw(g2, 10, 150);
60  
61          String s = "WAVATastic";
62          TextLayout layout2 = new TextLayout(s, font1, frc);
63          layout2.draw(g2, 10, 200);
64          TextLayout layout3 = new TextLayout(s, font2, frc);
65          layout3.draw(g2, 10, 100);
66      }
67  }
