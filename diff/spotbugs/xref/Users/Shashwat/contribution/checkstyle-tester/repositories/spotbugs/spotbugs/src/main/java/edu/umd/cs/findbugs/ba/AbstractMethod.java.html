19  package edu.umd.cs.findbugs.ba;
20  
21  import org.apache.bcel.Const;
22  
23  import edu.umd.cs.findbugs.classfile.DescriptorFactory;
24  import edu.umd.cs.findbugs.classfile.MethodDescriptor;
25  import edu.umd.cs.findbugs.internalAnnotations.DottedClassName;
26  import edu.umd.cs.findbugs.util.ClassName;
27  
28  public abstract class AbstractMethod extends AbstractClassMember implements XMethod {
29  
30      protected AbstractMethod(@DottedClassName String className, String methodName, String methodSig, int accessFlags) {
31          super(className, methodName, methodSig, accessFlags);
32      }
33  
34      @Override
35      public int getNumParams() {
36          // FIXME: cache this?
37          return new SignatureParser(getSignature()).getNumParameters();
38      }
39  
40      @Override
41      public boolean isNative() {
42          return (getAccessFlags() & Const.ACC_NATIVE) != 0;
43      }
44  
45      @Override
46      public boolean isSynchronized() {
47          return (getAccessFlags() & Const.ACC_SYNCHRONIZED) != 0;
48      }
49  
50      @Override
51      public boolean isBridge() {
52          return (getAccessFlags() & Const.ACC_BRIDGE) != 0;
53      }
54  
55      @Override
56      public String toString() {
57          return SignatureConverter.convertMethodSignature(this);
58      }
59  
60      /*
61       * (non-Javadoc)
62       *
63       * @see edu.umd.cs.findbugs.ba.XMethod#getMethodDescriptor()
64       */
65      @Override
66      public MethodDescriptor getMethodDescriptor() {
67          return DescriptorFactory.instance().getMethodDescriptor(ClassName.toSlashedClassName(getClassName()), getName(),
68                  getSignature(), isStatic());
69      }
70  
71      @Override
72      public XMethod resolveAccessMethodForMethod() {
73          MethodDescriptor access = getAccessMethodForMethod();
74          if (access != null) {
75              return XFactory.createXMethod(access);
76          }
77          return this;
78  
79      }
80  
81  }
