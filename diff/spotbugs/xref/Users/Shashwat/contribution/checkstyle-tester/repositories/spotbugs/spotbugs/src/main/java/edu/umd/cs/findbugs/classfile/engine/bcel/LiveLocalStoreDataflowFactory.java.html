19  package edu.umd.cs.findbugs.classfile.engine.bcel;
20  
21  import org.apache.bcel.generic.MethodGen;
22  
23  import edu.umd.cs.findbugs.ba.CFG;
24  import edu.umd.cs.findbugs.ba.ClassContext;
25  import edu.umd.cs.findbugs.ba.LiveLocalStoreAnalysis;
26  import edu.umd.cs.findbugs.ba.LiveLocalStoreDataflow;
27  import edu.umd.cs.findbugs.ba.ReverseDepthFirstSearch;
28  import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;
29  import edu.umd.cs.findbugs.classfile.IAnalysisCache;
30  import edu.umd.cs.findbugs.classfile.MethodDescriptor;
31  
32  /**
33   * Analysis engine to produce LiveLocalStoreDataflow objects for analyzed
34   * methods.
35   *
36   * @author David Hovemeyer
37   */
38  public class LiveLocalStoreDataflowFactory extends AnalysisFactory<LiveLocalStoreDataflow> {
39      /**
40       * Constructor.
41       */
42      public LiveLocalStoreDataflowFactory() {
43          super("live local stores analysis", LiveLocalStoreDataflow.class);
44      }
45  
46      /*
47       * (non-Javadoc)
48       *
49       * @see
50       * edu.umd.cs.findbugs.classfile.IAnalysisEngine#analyze(edu.umd.cs.findbugs
51       * .classfile.IAnalysisCache, java.lang.Object)
52       */
53      @Override
54      public LiveLocalStoreDataflow analyze(IAnalysisCache analysisCache, MethodDescriptor descriptor)
55              throws CheckedAnalysisException {
56          MethodGen methodGen = getMethodGen(analysisCache, descriptor);
57          if (methodGen == null) {
58              return null;
59          }
60          CFG cfg = getCFG(analysisCache, descriptor);
61  
62          ReverseDepthFirstSearch rdfs = getReverseDepthFirstSearch(analysisCache, descriptor);
63  
64          LiveLocalStoreAnalysis analysis = new LiveLocalStoreAnalysis(methodGen, rdfs, getDepthFirstSearch(analysisCache,
65                  descriptor));
66          LiveLocalStoreDataflow dataflow = new LiveLocalStoreDataflow(cfg, analysis);
67  
68          dataflow.execute();
69          if (ClassContext.DUMP_DATAFLOW_ANALYSIS) {
70              ClassContext.dumpLiveLocalStoreDataflow(descriptor, cfg, dataflow);
71  
72          }
73          return dataflow;
74      }
75  }
