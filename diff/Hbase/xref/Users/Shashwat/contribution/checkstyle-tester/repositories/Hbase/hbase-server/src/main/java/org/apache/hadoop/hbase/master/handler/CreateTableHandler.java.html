60  /**
61   * Handler to create a table.
62   */
63  @InterfaceAudience.Private
64  public class CreateTableHandler extends EventHandler {
65    private static final Log LOG = LogFactory.getLog(CreateTableHandler.class);
66    protected final MasterFileSystem fileSystemManager;
67    protected final HTableDescriptor hTableDescriptor;
68    protected final Configuration conf;
69    private final AssignmentManager assignmentManager;
70    private final TableLockManager tableLockManager;
71    private final HRegionInfo [] newRegions;
72    private final TableLock tableLock;
73    private User activeUser;
74  
75    public CreateTableHandler(Server server, MasterFileSystem fileSystemManager,
76        HTableDescriptor hTableDescriptor, Configuration conf, HRegionInfo [] newRegions,
77        MasterServices masterServices) {
78      super(server, EventType.C_M_CREATE_TABLE);
79  
80      this.fileSystemManager = fileSystemManager;
81      this.hTableDescriptor = hTableDescriptor;
82      this.conf = conf;
83      this.newRegions = newRegions;
84      this.assignmentManager = masterServices.getAssignmentManager();
85      this.tableLockManager = masterServices.getTableLockManager();
86  
87      this.tableLock = this.tableLockManager.writeLock(this.hTableDescriptor.getTableName()
88          , EventType.C_M_CREATE_TABLE.toString());
89    }
90  
91    @Override
92    public CreateTableHandler prepare()
93        throws NotAllMetaRegionsOnlineException, TableExistsException, IOException {
94      int timeout = conf.getInt("hbase.client.catalog.timeout", 10000);
95      // Need hbase:meta availability to create a table
96      try {
97        if (server.getMetaTableLocator().waitMetaRegionLocation(
98            server.getZooKeeper(), timeout) == null) {
99          throw new NotAllMetaRegionsOnlineException();
100       }
101       // If we are creating the table in service to an RPC request, record the
102       // active user for later, so proper permissions will be applied to the
103       // new table by the AccessController if it is active
104       this.activeUser = RpcServer.getRequestUser();
105       if (this.activeUser == null) {
106         this.activeUser = UserProvider.instantiate(conf).getCurrent();
107       }
108     } catch (InterruptedException e) {
109       LOG.warn("Interrupted waiting for meta availability", e);
110       InterruptedIOException ie = new InterruptedIOException(e.getMessage());
111       ie.initCause(e);
112       throw ie;
113     }
114 
115     //acquire the table write lock, blocking. Make sure that it is released.
116     this.tableLock.acquire();
117     boolean success = false;
118     try {
119       TableName tableName = this.hTableDescriptor.getTableName();
120       if (MetaTableAccessor.tableExists(this.server.getConnection(), tableName)) {
121         throw new TableExistsException(tableName);
122       }
123 
124       checkAndSetEnablingTable(assignmentManager, tableName);
125       success = true;
126     } finally {
127       if (!success) {
128         releaseTableLock();
129       }
130     }
131     return this;
132   }
133 
134   static void checkAndSetEnablingTable(final AssignmentManager assignmentManager,
135       final TableName tableName) throws IOException {
136     // If we have multiple client threads trying to create the table at the
137     // same time, given the async nature of the operation, the table
138     // could be in a state where hbase:meta table hasn't been updated yet in
139     // the process() function.
140     // Use enabling state to tell if there is already a request for the same
141     // table in progress. This will introduce a new zookeeper call. Given
142     // createTable isn't a frequent operation, that should be ok.
143     // TODO: now that we have table locks, re-evaluate above -- table locks are not enough.
144     // We could have cleared the hbase.rootdir and not zk.  How can we detect this case?
145     // Having to clean zk AND hdfs is awkward.
146     try {
147       if (!assignmentManager.getTableStateManager().setTableStateIfNotInStates(tableName,
148         ZooKeeperProtos.Table.State.ENABLING,
149         ZooKeeperProtos.Table.State.ENABLING,
150         ZooKeeperProtos.Table.State.ENABLED)) {
151         throw new TableExistsException(tableName);
152       }
153     } catch (CoordinatedStateException e) {
154       throw new IOException("Unable to ensure that the table will be" +
155         " enabling because of a ZooKeeper issue", e);
156     }
157   }
158 
159   static void removeEnablingTable(final AssignmentManager assignmentManager,
160       final TableName tableName) {
161     // Try deleting the enabling node in case of error
162     // If this does not happen then if the client tries to create the table
163     // again with the same Active master
164     // It will block the creation saying TableAlreadyExists.
165     try {
166       assignmentManager.getTableStateManager().checkAndRemoveTableState(tableName,
167         ZooKeeperProtos.Table.State.ENABLING, false);
168     } catch (CoordinatedStateException e) {
169       // Keeper exception should not happen here
170       LOG.error("Got a keeper exception while removing the ENABLING table znode "
171           + tableName, e);
172     }
173   }

175   @Override
176   public String toString() {
177     String name = "UnknownServerName";
178     if(server != null && server.getServerName() != null) {
179       name = server.getServerName().toString();
180     }
181     return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" +
182       this.hTableDescriptor.getTableName();
183   }
184 
185   @Override
186   public void process() {
187     TableName tableName = this.hTableDescriptor.getTableName();
188     LOG.info("Create table " + tableName);
189     HMaster master = ((HMaster) this.server);
190     try {
191       final MasterCoprocessorHost cpHost = master.getMasterCoprocessorHost();
192       if (cpHost != null) {
193         cpHost.preCreateTableHandler(this.hTableDescriptor, this.newRegions);
194       }
195       handleCreateTable(tableName);
196       completed(null);
197       if (cpHost != null) {
198         this.activeUser.runAs(new PrivilegedExceptionAction<Void>() {
199           @Override
200           public Void run() throws Exception {
201             cpHost.postCreateTableHandler(hTableDescriptor, newRegions);
202             return null;
203           }
204         });
205       }
206     } catch (Throwable e) {
207       LOG.error("Error trying to create the table " + tableName, e);
208       if (master.isInitialized()) {
209         try {
210           ((HMaster) this.server).getMasterQuotaManager().removeTableFromNamespaceQuota(
211             hTableDescriptor.getTableName());
212         } catch (IOException e1) {
213           LOG.error("Error trying to update namespace quota " + e1);
214         }
215       }
216       completed(e);
217     }
218   }
219 
220   /**
221    * Called after that process() is completed.
222    * @param exception null if process() is successful or not null if something has failed.
223    */
224   protected void completed(final Throwable exception) {
225     releaseTableLock();
226     LOG.info("Table, " + this.hTableDescriptor.getTableName() + ", creation " +
227         (exception == null ? "successful" : "failed. " + exception));
228     if (exception != null) {
229       removeEnablingTable(this.assignmentManager, this.hTableDescriptor.getTableName());
230     }
231   }

233   /**
234    * Responsible of table creation (on-disk and META) and assignment.
235    * - Create the table directory and descriptor (temp folder)
236    * - Create the on-disk regions (temp folder)
237    *   [If something fails here: we've just some trash in temp]
238    * - Move the table from temp to the root directory
239    *   [If something fails here: we've the table in place but some of the rows required
240    *    present in META. (hbck needed)]
241    * - Add regions to META
242    *   [If something fails here: we don't have regions assigned: table disabled]
243    * - Assign regions to Region Servers
244    *   [If something fails here: we still have the table in disabled state]
245    * - Update ZooKeeper with the enabled state
246    */
247   private void handleCreateTable(TableName tableName)
248       throws IOException, CoordinatedStateException {
249     Path tempdir = fileSystemManager.getTempDir();
250     FileSystem fs = fileSystemManager.getFileSystem();
251 
252     // 1. Create Table Descriptor
253     Path tempTableDir = FSUtils.getTableDir(tempdir, tableName);
254     new FSTableDescriptors(this.conf).createTableDescriptorForTableDirectory(
255       tempTableDir, this.hTableDescriptor, false);
256     Path tableDir = FSUtils.getTableDir(fileSystemManager.getRootDir(), tableName);
257 
258     // 2. Create Regions
259     List<HRegionInfo> regionInfos = handleCreateHdfsRegions(tempdir, tableName);
260     // 3. Move Table temp directory to the hbase root location
261     if (!fs.rename(tempTableDir, tableDir)) {
262       throw new IOException("Unable to move table from temp=" + tempTableDir +
263         " to hbase root=" + tableDir);
264     }
265 
266     if (regionInfos != null && regionInfos.size() > 0) {
267       // 4. Add regions to META
268       addRegionsToMeta(regionInfos, hTableDescriptor.getRegionReplication());
269       // 5. Add replicas if needed
270       regionInfos = addReplicas(hTableDescriptor, regionInfos);
271 
272       // 6. Setup replication for region replicas if needed
273       if (hTableDescriptor.getRegionReplication() > 1) {
274         ServerRegionReplicaUtil.setupRegionReplicaReplication(conf);
275       }
276 
277       // 7. Trigger immediate assignment of the regions in round-robin fashion
278       ModifyRegionUtils.assignRegions(assignmentManager, regionInfos);
279     }
280 
281     // 8. Set table enabled flag up in zk.
282     try {
283       assignmentManager.getTableStateManager().setTableState(tableName,
284         ZooKeeperProtos.Table.State.ENABLED);
285     } catch (CoordinatedStateException e) {
286       throw new IOException("Unable to ensure that " + tableName + " will be" +
287         " enabled because of a ZooKeeper issue", e);
288     }
289 
290     // 8. Update the tabledescriptor cache.
291     ((HMaster) this.server).getTableDescriptors().get(tableName);
292   }

294   /**
295    * Create any replicas for the regions (the default replicas that was
296    * already created is passed to the method)
297    * @param hTableDescriptor
298    * @param regions default replicas
299    * @return the combined list of default and non-default replicas
300    */
301   protected List<HRegionInfo> addReplicas(HTableDescriptor hTableDescriptor,
302       List<HRegionInfo> regions) {
303     int numRegionReplicas = hTableDescriptor.getRegionReplication() - 1;
304     if (numRegionReplicas <= 0) {
305       return regions;
306     }
307     List<HRegionInfo> hRegionInfos =
308         new ArrayList<HRegionInfo>((numRegionReplicas+1)*regions.size());
309     for (int i = 0; i < regions.size(); i++) {
310       for (int j = 1; j <= numRegionReplicas; j++) {
311         hRegionInfos.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(i), j));
312       }
313     }
314     hRegionInfos.addAll(regions);
315     return hRegionInfos;
316   }
317 
318   private void releaseTableLock() {
319     if (this.tableLock != null) {
320       try {
321         this.tableLock.release();
322       } catch (IOException ex) {
323         LOG.warn("Could not release the table lock", ex);
324       }
325     }
326   }

328   /**
329    * Create the on-disk structure for the table, and returns the regions info.
330    * @param tableRootDir directory where the table is being created
331    * @param tableName name of the table under construction
332    * @return the list of regions created
333    */
334   protected List<HRegionInfo> handleCreateHdfsRegions(final Path tableRootDir,
335     final TableName tableName)
336       throws IOException {
337     return ModifyRegionUtils.createRegions(conf, tableRootDir,
338         hTableDescriptor, newRegions, null);
339   }
340 
341   /**
342    * Add the specified set of regions to the hbase:meta table.
343    */
344   protected void addRegionsToMeta(final List<HRegionInfo> regionInfos, int regionReplication)
345       throws IOException {
346     MetaTableAccessor.addRegionsToMeta(this.server.getConnection(), regionInfos, regionReplication);
347   }
