11  package org.apache.hadoop.hbase.replication;
12  
13  import org.apache.hadoop.hbase.classification.InterfaceAudience;
14  
15  /**
16   * A HBase ReplicationLoad to present MetricsSink information
17   */
18  @InterfaceAudience.Private
19  public class ReplicationLoadSink {
20    private long ageOfLastAppliedOp;
21    private long timeStampsOfLastAppliedOp;
22  
23    public ReplicationLoadSink(long age, long timeStamp) {
24      this.ageOfLastAppliedOp = age;
25      this.timeStampsOfLastAppliedOp = timeStamp;
26    }
27  
28    public long getAgeOfLastAppliedOp() {
29      return this.ageOfLastAppliedOp;
30    }
31  
32    public long getTimeStampsOfLastAppliedOp() {
33      return this.timeStampsOfLastAppliedOp;
34    }
35  
36  }
