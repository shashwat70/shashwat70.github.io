87    @Test(timeout=60000)
88    public void testWalRecoverLease() throws Exception {
89      final ProcedureStore masterStore = getMasterProcedureExecutor().getStore();
90      assertTrue("expected WALStore for this test", masterStore instanceof WALProcedureStore);
91  
92      HMaster firstMaster = UTIL.getHBaseCluster().getMaster();
93      // Abort Latch for the master store
94      final CountDownLatch masterStoreAbort = new CountDownLatch(1);
95      masterStore.registerListener(new ProcedureStore.ProcedureStoreListener() {
96        @Override
97        public void abortProcess() {
98          LOG.debug("Abort store of Master");
99          masterStoreAbort.countDown();
100       }
101     });
102 
103     // startup a fake master the new WAL store will take the lease
104     // and the active master should abort.
105     HMaster backupMaster3 = Mockito.mock(HMaster.class);
106     Mockito.doReturn(firstMaster.getConfiguration()).when(backupMaster3).getConfiguration();
107     Mockito.doReturn(true).when(backupMaster3).isActiveMaster();
108     final WALProcedureStore backupStore3 = new WALProcedureStore(firstMaster.getConfiguration(),
109         firstMaster.getMasterFileSystem().getFileSystem(),
110         ((WALProcedureStore)masterStore).getLogDir(),
111         new MasterProcedureEnv.WALStoreLeaseRecovery(backupMaster3));
112     // Abort Latch for the test store
113     final CountDownLatch backupStore3Abort = new CountDownLatch(1);
114     backupStore3.registerListener(new ProcedureStore.ProcedureStoreListener() {
115       @Override
116       public void abortProcess() {
117         LOG.debug("Abort store of backupMaster3");
118         backupStore3Abort.countDown();
119         backupStore3.stop(true);
120       }
121     });
122     backupStore3.start(1);
123     backupStore3.recoverLease();
124 
125     // Try to trigger a command on the master (WAL lease expired on the active one)
126     HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(TableName.valueOf("mtb"), "f");
127     HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, null);
128     LOG.debug("submit proc");
129     getMasterProcedureExecutor().submitProcedure(
130       new CreateTableProcedure(getMasterProcedureExecutor().getEnvironment(), htd, regions));
131     LOG.debug("wait master store abort");
132     masterStoreAbort.await();
133 
134     // Now the real backup master should start up
135     LOG.debug("wait backup master to startup");
136     waitBackupMaster(UTIL, firstMaster);
137     assertEquals(true, firstMaster.isStopped());
138 
139     // wait the store in here to abort (the test will fail due to timeout if it doesn't)
140     LOG.debug("wait the store to abort");
141     backupStore3.getStoreTracker().setDeleted(1, false);
142     backupStore3.delete(1);
143     backupStore3Abort.await();
144   }

230   private void testTruncateWithFailoverAtStep(final boolean preserveSplits, final int step)
231       throws Exception {
232     final TableName tableName = TableName.valueOf("testTruncateWithFailoverAtStep" + step);
233 
234     // create the table
235     final String[] families = new String[] { "f1", "f2" };
236     final byte[][] splitKeys = new byte[][] {
237       Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
238     };
239     HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
240       getMasterProcedureExecutor(), tableName, splitKeys, families);
241     // load and verify that there are rows in the table
242     MasterProcedureTestingUtility.loadData(
243       UTIL.getConnection(), tableName, 100, splitKeys, families);
244     assertEquals(100, UTIL.countRows(tableName));
245     // disable the table
246     UTIL.getHBaseAdmin().disableTable(tableName);
247 
248     ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
249     ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
250 
251     // Start the Truncate procedure && kill the executor
252     long procId = procExec.submitProcedure(
253       new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
254     testRecoveryAndDoubleExecution(UTIL, procId, step, TruncateTableState.values());
255 
256     ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
257     UTIL.waitUntilAllRegionsAssigned(tableName);
258 
259     // validate the table regions and layout
260     if (preserveSplits) {
261       assertEquals(1 + splitKeys.length, UTIL.getHBaseAdmin().getTableRegions(tableName).size());
262     } else {
263       regions = UTIL.getHBaseAdmin().getTableRegions(tableName).toArray(new HRegionInfo[1]);
264       assertEquals(1, regions.length);
265     }
266     MasterProcedureTestingUtility.validateTableCreation(
267       UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
268 
269     // verify that there are no rows in the table
270     assertEquals(0, UTIL.countRows(tableName));
271 
272     // verify that the table is read/writable
273     MasterProcedureTestingUtility.loadData(
274       UTIL.getConnection(), tableName, 50, splitKeys, families);
275     assertEquals(50, UTIL.countRows(tableName));
276   }

351   // ==========================================================================
352   //  Test Helpers
353   // ==========================================================================
354   public static <TState> void testRecoveryAndDoubleExecution(final HBaseTestingUtility testUtil,
355       final long procId, final int lastStepBeforeFailover, TState[] states) throws Exception {
356     ProcedureExecutor<MasterProcedureEnv> procExec =
357       testUtil.getHBaseCluster().getMaster().getMasterProcedureExecutor();
358     ProcedureTestingUtility.waitProcedure(procExec, procId);
359 
360     for (int i = 0; i < lastStepBeforeFailover; ++i) {
361       LOG.info("Restart "+ i +" exec state: " + states[i]);
362       ProcedureTestingUtility.assertProcNotYetCompleted(procExec, procId);
363       ProcedureTestingUtility.restart(procExec);
364       ProcedureTestingUtility.waitProcedure(procExec, procId);
365     }
366     ProcedureTestingUtility.assertProcNotYetCompleted(procExec, procId);
367 
368     LOG.info("Trigger master failover");
369     masterFailover(testUtil);
370 
371     procExec = testUtil.getHBaseCluster().getMaster().getMasterProcedureExecutor();
372     ProcedureTestingUtility.waitProcedure(procExec, procId);
373     ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
374   }

377   //  Master failover utils
378   // ==========================================================================
379   public static void masterFailover(final HBaseTestingUtility testUtil)
380       throws Exception {
381     MiniHBaseCluster cluster = testUtil.getMiniHBaseCluster();
382 
383     // Kill the master
384     HMaster oldMaster = cluster.getMaster();
385     cluster.killMaster(cluster.getMaster().getServerName());
386 
387     // Wait the secondary
388     waitBackupMaster(testUtil, oldMaster);
389   }
390 
391   public static void waitBackupMaster(final HBaseTestingUtility testUtil,
392       final HMaster oldMaster) throws Exception {
393     MiniHBaseCluster cluster = testUtil.getMiniHBaseCluster();
394 
395     HMaster newMaster = cluster.getMaster();
396     while (newMaster == null || newMaster == oldMaster) {
397       Thread.sleep(250);
398       newMaster = cluster.getMaster();
399     }
400 
401     while (!(newMaster.isActiveMaster() && newMaster.isInitialized())) {
402       Thread.sleep(250);
403     }
404   }
