19  package org.apache.struts2.jasper.compiler.tagplugin;
20  
21  /**
22   * This interface allows the plugin author to make inqueries about the
23   * properties of the current tag, and to use Jasper resources to generate
24   * direct Java codes in place of tag handler invocations.
25   */
26  
27  public interface TagPluginContext {
28      /**
29       * @return true if the body of the tag is scriptless.
30       */
31      boolean isScriptless();
32  
33      /**
34       * @param attribute Name of the attribute
35       * @return true if the attribute is specified in the tag
36       */
37      boolean isAttributeSpecified(String attribute);
38  
39      /**
40       * @return An unique temporary variable name that the plugin can use.
41       */
42      String getTemporaryVariableName();
43  
44      /**
45       * Generate an import statement
46       * @param s Name of the import class, '*' allowed.
47       */
48      void generateImport(String s);
49  
50      /**
51       * Generate a declaration in the of the generated class.  This can be
52       * used to declare an innter class, a method, or a class variable.
53       * @param id An unique ID identifying the declaration.  It is not
54       *           part of the declaration, and is used to ensure that the
55       *           declaration will only appear once.  If this method is
56       *           invoked with the same id more than once in the translation
57       *           unit, only the first declaration will be taken.
58       * @param text The text of the declaration.
59       **/
60      void generateDeclaration(String id, String text);
61  
62      /**
63       * Generate Java source codes
64       *
65       * @param s string
66       */
67      void generateJavaSource(String s);
68  
69      /**
70       * @param attribute the attribute to check
71       * @return true if the attribute is specified and its value is a
72       *         translation-time constant.
73       */
74      boolean isConstantAttribute(String attribute);
75  
76      /**
77       * @param attribute The specified attribute
78       *
79       * @return A string that is the value of a constant attribute.  Undefined
80       *         if the attribute is not a (translation-time) constant.
81       *         null if the attribute is not specified.
82       */
83      String getConstantAttribute(String attribute);
84  
85      /**
86       * Generate codesto evaluate value of a attribute in the custom tag
87       * The codes is a Java expression.
88       * NOTE: Currently cannot handle attributes that are fragments.
89       * @param attribute The specified attribute
90       */
91      void generateAttribute(String attribute);
92  
93      /**
94       * Generate codes for the body of the custom tag
95       */
96      void generateBody();
97  
98      /**
99       * Abandon optimization for this tag handler, and instruct
100      * Jasper to generate the tag handler calls, as usual.
101      * Should be invoked if errors are detected, or when the tag body
102      * is deemed too compilicated for optimization.
103      */
104     void dontUseTagPlugin();
105 
106     /**
107      * Get the PluginContext for the parent of this custom tag.  NOTE:
108      * The operations available for PluginContext so obtained is limited
109      * to getPluginAttribute and setPluginAttribute, and queries (e.g.
110      * isScriptless().  There should be no calls to generate*().
111      * @return The pluginContext for the parent node.
112      *         null if the parent is not a custom tag, or if the pluginConxt
113      *         if not available (because useTagPlugin is false, e.g).
114      */
115     TagPluginContext getParentContext();
116 
117     /**
118      * Associate the attribute with a value in the current tagplugin context.
119      * The plugin attributes can be used for communication among tags that
120      * must work together as a group.  See &lt;c:when&gt; for an example.
121      *
122      * @param attr The specified attribute
123      * @param value the value
124      */
125     void setPluginAttribute(String attr, Object value);
126 
127     /**
128      * @param attr The specified attribute
129      * @return  the value of an attribute in the current tagplugin context.
130      */
131     Object getPluginAttribute(String attr);
132 }
