19  package org.apache.struts2.spring;
20  
21  import com.opensymphony.xwork2.inject.Container;
22  import junit.framework.TestCase;
23  import org.easymock.EasyMock;
24  import org.springframework.beans.factory.config.AutowireCapableBeanFactory;
25  import org.springframework.mock.web.MockServletContext;
26  import org.springframework.web.context.ConfigurableWebApplicationContext;
27  import org.springframework.web.context.WebApplicationContext;
28  import org.springframework.web.context.support.XmlWebApplicationContext;
29  
30  import javax.servlet.ServletContext;
31  
32  /**
33   * Unit test for {@link StrutsSpringObjectFactory}.
34   *
35   */
36  public class StrutsSpringObjectFactoryTest extends TestCase {
37  
38      public void testNoSpringContext() throws Exception {
39          // to cover situations where there will be logged an error
40          Container container = EasyMock.createNiceMock(Container.class);
41          EasyMock.replay(container);
42          
43          StrutsSpringObjectFactory fac = new StrutsSpringObjectFactory(null, null, null, null, new MockServletContext(), null, container);
44  
45          assertEquals(AutowireCapableBeanFactory.AUTOWIRE_BY_NAME, fac.getAutowireStrategy());
46      }
47  
48      public void testWithSpringContext() throws Exception {
49          Container container = EasyMock.createNiceMock(Container.class);
50          EasyMock.replay(container);
51  
52          ConfigurableWebApplicationContext ac = new XmlWebApplicationContext();
53          ServletContext msc = new MockServletContext();
54          msc.setAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE, ac);
55          ac.setServletContext(msc);
56          ac.setConfigLocations(new String[] {"org/apache/struts2/spring/StrutsSpringObjectFactoryTest-applicationContext.xml"});
57          ac.refresh();
58          StrutsSpringObjectFactory fac = new StrutsSpringObjectFactory("constructor", null, null, null, msc, null, container);
59  
60          assertEquals(AutowireCapableBeanFactory.AUTOWIRE_CONSTRUCTOR, fac.getAutowireStrategy());
61      }
62  
63  
64  }
