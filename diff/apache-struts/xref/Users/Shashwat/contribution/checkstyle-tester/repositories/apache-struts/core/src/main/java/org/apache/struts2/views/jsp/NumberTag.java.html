19  package org.apache.struts2.views.jsp;
20  
21  import com.opensymphony.xwork2.util.ValueStack;
22  import org.apache.struts2.components.Component;
23  import org.apache.struts2.components.Number;
24  
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  
28  /**
29   * @see Number
30   */
31  public class NumberTag extends ContextBeanTag {
32  
33      private static final long serialVersionUID = -6216963123295613440L;
34  
35      private String name;
36      private String currency;
37      private String type;
38      private Boolean groupingUsed;
39      private Integer maximumFractionDigits;
40      private Integer maximumIntegerDigits;
41      private Integer minimumFractionDigits;
42      private Integer minimumIntegerDigits;
43      private Boolean parseIntegerOnly;
44      private String roundingMode;
45  
46      public Component getBean(ValueStack stack, HttpServletRequest req, HttpServletResponse res) {
47          return new Number(stack);
48      }
49  
50      protected void populateParams() {
51          super.populateParams();
52          Number n = (Number) component;
53          n.setName(name);
54          n.setCurrency(currency);
55          n.setType(type);
56          n.setGroupingUsed(groupingUsed);
57          n.setMaximumFractionDigits(maximumFractionDigits);
58          n.setMaximumIntegerDigits(maximumIntegerDigits);
59          n.setMinimumFractionDigits(minimumFractionDigits);
60          n.setMinimumIntegerDigits(minimumIntegerDigits);
61          n.setParseIntegerOnly(parseIntegerOnly);
62          n.setRoundingMode(roundingMode);
63  
64      }
65  
66      /**
67       * @param name the name to set
68       */
69      public void setName(String name) {
70          this.name = name;
71      }
72  
73      /**
74       * @param currency the currency to set
75       */
76      public void setCurrency(String currency) {
77          this.currency = currency;
78      }
79  
80      /**
81       * @param type the type to set
82       */
83      public void setType(String type) {
84          this.type = type;
85      }
86  
87      /**
88       * @param groupingUsed the groupingUsed to set
89       */
90      public void setGroupingUsed(Boolean groupingUsed) {
91          this.groupingUsed = groupingUsed;
92      }
93  
94      /**
95       * @param maximumFractionDigits the maximumFractionDigits to set
96       */
97      public void setMaximumFractionDigits(Integer maximumFractionDigits) {
98          this.maximumFractionDigits = maximumFractionDigits;
99      }
100 
101     /**
102      * @param maximumIntegerDigits the maximumIntegerDigits to set
103      */
104     public void setMaximumIntegerDigits(Integer maximumIntegerDigits) {
105         this.maximumIntegerDigits = maximumIntegerDigits;
106     }
107 
108     /**
109      * @param minimumFractionDigits the minimumFractionDigits to set
110      */
111     public void setMinimumFractionDigits(Integer minimumFractionDigits) {
112         this.minimumFractionDigits = minimumFractionDigits;
113     }
114 
115     /**
116      * @param minimumIntegerDigits the minimumIntegerDigits to set
117      */
118     public void setMinimumIntegerDigits(Integer minimumIntegerDigits) {
119         this.minimumIntegerDigits = minimumIntegerDigits;
120     }
121 
122     /**
123      * @param parseIntegerOnly the parseIntegerOnly to set
124      */
125     public void setParseIntegerOnly(Boolean parseIntegerOnly) {
126         this.parseIntegerOnly = parseIntegerOnly;
127     }
128 
129     /**
130      * @param roundingMode the roundingMode to set
131      */
132     public void setRoundingMode(String roundingMode) {
133         this.roundingMode = roundingMode;
134     }
135 
136 }
