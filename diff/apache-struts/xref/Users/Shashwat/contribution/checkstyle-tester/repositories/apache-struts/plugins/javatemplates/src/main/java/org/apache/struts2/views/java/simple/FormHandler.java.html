19  package org.apache.struts2.views.java.simple;
20  
21  import org.apache.struts2.views.java.Attributes;
22  import org.apache.struts2.views.java.TagGenerator;
23  
24  import java.io.IOException;
25  import java.util.Map;
26  
27  public class FormHandler extends AbstractTagHandler implements TagGenerator {
28      public void generate() throws IOException {
29          Map<String, Object> params = context.getParameters();
30          Attributes attrs = new Attributes();
31  
32          attrs.addIfExists("name", params.get("name"))
33                  .addIfExists("id", params.get("id"))
34                  .addIfExists("onsubmit", params.get("onsubmit"))
35                  .addIfExists("onreset", params.get("onreset"))
36                  .addIfExists("action", params.get("action"))
37                  .addIfExists("target", params.get("target"))
38                  .addIfExists("enctype", params.get("enctype"))
39                  .addIfExists("class", params.get("cssClass"))
40                  .addIfExists("style", params.get("cssStyle"))
41                  .addIfExists("title", params.get("title"))
42                  .addIfExists("accept-charset", params.get("acceptcharset"));
43          attrs.add("method", params.containsKey("method") ? (String) params.get("method") : "post");
44          start("form", attrs);
45      }
46  
47      public static class CloseHandler extends AbstractTagHandler implements TagGenerator {
48          public void generate() throws IOException {
49              end("form");
50          }
51      }
52  }
