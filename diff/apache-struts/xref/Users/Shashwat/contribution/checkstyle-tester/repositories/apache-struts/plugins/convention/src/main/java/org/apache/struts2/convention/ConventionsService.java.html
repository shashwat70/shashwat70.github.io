19  package org.apache.struts2.convention;
20  
21  import java.util.Map;
22  
23  import com.opensymphony.xwork2.config.entities.ActionConfig;
24  import com.opensymphony.xwork2.config.entities.PackageConfig;
25  import com.opensymphony.xwork2.config.entities.ResultTypeConfig;
26  
27  /**
28   * <p>
29   * This interface defines the conventions that are used by the convention plugin.
30   * In most cases the methods on this class will provide the best default for any
31   * values and also handle locating overrides of the default via the annotations
32   * that are part of the plugin.
33   * </p>
34   */
35  public interface ConventionsService {
36      /**
37       * Locates the result location from annotations on the action class or the package or returns the
38       * default if no annotations are present.
39       *
40       * @param   actionClass The action class.
41       * @return  The result location if it is set in the annotations. Otherwise, the default result
42       *          location is returned.
43       */
44      String determineResultPath(Class<?> actionClass);
45  
46      /**
47       * Delegates to the other method but first looks up the Action's class using the given class name.
48       *
49       * @param   actionConfig (Optional) The configuration for the action that the result is being
50       *          built for or null if the default result path is needed.
51       * @return  The result location if it is set in the annotations for the class of the ActionConfig.
52       *          Otherwise, the default result location is returned. If null is passed in, the default
53       *          is returned,
54       */
55      String determineResultPath(ActionConfig actionConfig);
56  
57      /**
58       * Returns a mapping between the result type strings and the {@link ResultTypeConfig} instances
59       * based on the {@link PackageConfig} given.
60       *
61       * @param   packageConfig The PackageConfig to get the result types for.
62       * @return  The result types or an empty Map of nothing is configured.
63       */
64      Map<String, ResultTypeConfig> getResultTypesByExtension(PackageConfig packageConfig);
65  }
