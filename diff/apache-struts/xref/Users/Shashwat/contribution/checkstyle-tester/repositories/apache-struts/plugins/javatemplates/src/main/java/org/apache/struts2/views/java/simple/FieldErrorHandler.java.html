19  package org.apache.struts2.views.java.simple;
20  
21  import org.apache.struts2.views.java.Attributes;
22  import org.apache.struts2.views.java.TagGenerator;
23  
24  import java.io.IOException;
25  import java.util.List;
26  import java.util.Map;
27  
28  public class FieldErrorHandler extends AbstractTagHandler implements TagGenerator {
29      public void generate() throws IOException {
30          Map<String, Object> params = context.getParameters();
31          Map<String, List<String>> errors = (Map<String, List<String>>) findValue("fieldErrors");
32          List<String> fieldErrorFieldNames = (List<String>) params.get("errorFieldNames");
33  
34          if (fieldErrorFieldNames != null && !fieldErrorFieldNames.isEmpty() && errors != null && !errors.isEmpty()) {
35              startUL(params);
36  
37              //iterate over field error names
38              for (String fieldErrorFieldName : fieldErrorFieldNames) {
39                  List<String> fieldErrors = errors.get(fieldErrorFieldName);
40                  if (fieldErrors != null) {
41                      for (String fieldError : fieldErrors) {
42                          writeError(params, fieldError);
43                      }
44                  }
45              }
46  
47              endUL();
48          } else if (errors != null && !errors.isEmpty()) {
49              startUL(params);
50  
51              for (Map.Entry<String, List<String>> errorEntry : errors.entrySet()) {
52                  for (String fieldError : errorEntry.getValue()) {
53                      writeError(params, fieldError);
54                  }
55              }
56  
57              endUL();
58          }
59      }
60  
61      private void endUL() throws IOException {
62          end("ul");
63      }
64  
65      private void writeError(Map<String, Object> params, String fieldError) throws IOException {
66          start("li", null);
67          start("span", null);
68          characters(fieldError, params.containsKey("escape") ? (Boolean) params.get("escape") : true);
69          end("span");
70          end("li");
71      }
72  
73      private void startUL(Map<String, Object> params) throws IOException {
74          Attributes attrs = new Attributes();
75          attrs.addIfExists("style", params.get("cssStyle"))
76                  .add("class", params.containsKey("cssClass") ? (String) params.get("cssClass") : "errorMessage");
77          start("ul", attrs);
78      }
79  }
