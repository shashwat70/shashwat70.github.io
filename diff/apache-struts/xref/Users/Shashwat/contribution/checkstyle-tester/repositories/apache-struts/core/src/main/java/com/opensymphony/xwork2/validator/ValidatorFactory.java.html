19  package com.opensymphony.xwork2.validator;
20  
21  /**
22   * ValidatorFactory
23   *
24   * <p>
25   * <!-- START SNIPPET: javadoc -->
26   * Validation rules are handled by validators, which must be registered with
27   * the ValidatorFactory (using the registerValidator method). The simplest way to do so is to add a file name
28   * validators.xml in the root of the classpath (/WEB-INF/classes) that declares
29   * all the validators you intend to use.
30   * <!-- END SNIPPET: javadoc -->
31   * </p>
32   *
33   *
34   * <p>
35   * <b>INFORMATION</b>
36   * <!-- START SNIPPET: information -->
37   * validators.xml if being defined should be available in the classpath. However
38   * this is not necessary, if no custom validator is needed. Predefined sets of validators
39   * will automatically be picked up when defined in
40   * com/opensymphony/xwork2/validator/validators/default.xml packaged in
41   * in the xwork jar file. See ValidatorFactory static block for details.
42   * <!-- END SNIPPET: information -->
43   * </p>
44   *
45   * <p>
46   * <b>WARNING</b>
47   * <!-- START SNIPPET: warning -->
48   * If custom validator is being defined and a validators.xml is created and
49   * place in the classpath, do remember to copy all the other pre-defined validators
50   * that is needed into the validators.xml as if not they will not be registered.
51   * Once a validators.xml is detected in the classpath, the default one
52   * (com/opensymphony/xwork2/validator/validators/default.xml) will not be loaded.
53   * It is only loaded when a custom validators.xml cannot be found in the classpath.
54   *  Be careful.
55   * <!-- END SNIPPET: warning -->
56   * </p>
57   *
58   * <p><b>Note:</b>
59   * <!-- START SNIPPET: turningOnValidators -->
60   * The default validationWorkflowStack already includes this.<br>
61   * All that is required to enable validation for an Action is to put the
62   * ValidationInterceptor in the interceptor refs of the action (see xwork.xml) like so:
63   * <!-- END SNIPPET: turningOnValidators -->
64   * </p>
65   *
66   * <pre>
67   * <!-- START SNIPPET: exTurnOnValidators -->
68   *     &lt;interceptor name="validator" class="com.opensymphony.xwork2.validator.ValidationInterceptor"/&gt;
69   * <!-- END SNIPPET: exTurnOnValidators -->
70   * </pre>
71   *
72   * <p><b>Field Validators</b>
73   * <!-- START SNIPPET: fieldValidators -->
74   * Field validators, as the name indicate, act on single fields accessible through an action.
75   * A validator, in contrast, is more generic and can do validations in the full action context,
76   * involving more than one field (or even no field at all) in validation rule.
77   * Most validations can be defined on per field basis. This should be preferred over
78   * non-field validation wherever possible, as field validator messages are bound to the
79   * related field and will be presented next to the corresponding input element in the
80   * respecting view.
81   * <!-- END SNIPPET: fieldValidators -->
82   * </p>
83   *
84   * <p><b>Non Field Validators</b>
85   * <!-- START SNIPPET: nonFieldValidators -->
86   * Non-field validators only add action level messages. Non-field validators
87   * are mostly domain specific and therefore offer custom implementations.
88   * The most important standard non-field validator provided by XWork
89   * is ExpressionValidator.
90   * <!-- END SNIPPET: nonFieldValidators -->
91   * </p>
92   *
93   * <p><b>NOTE:</b>
94   * <!-- START SNIPPET: validatorsNote -->
95   * Non-field validators takes precedence over field validators
96   * regardless of the order they are defined in *-validation.xml. If a non-field
97   * validator is short-circuited, it will causes its non-field validator to not
98   * being executed. See validation framework documentation for more info.
99   * <!-- END SNIPPET: validatorsNote -->
100  * </p>
101  *
102  * <p><b>VALIDATION RULES:</b></p>
103  * <!-- START SNIPPET: validationRules1 -->
104  * <p>Validation rules can be specified:</p>
105  * <ol>
106  *  <li> Per Action class: in a file named ActionName-validation.xml</li>
107  *  <li> Per Action alias: in a file named ActionName-alias-validation.xml</li>
108  *  <li> Inheritance hierarchy and interfaces implemented by Action class:
109  *  XWork searches up the inheritance tree of the action to find default
110  *  validations for parent classes of the Action and interfaces implemented</li>
111  * </ol>
112  * <p>Here is an example for SimpleAction-validation.xml:</p>
113  * <!-- END SNIPPET: validationRules1 -->
114  *
115  * <pre>
116  * <!-- START SNIPPET: exValidationRules1 -->
117  * &lt;!DOCTYPE validators PUBLIC "-//Apache Struts//XWork Validator 1.0.3//EN"
118    		"http://struts.apache.org/dtds/xwork-validator-1.0.3.dtd"&gt;
119  * &lt;validators&gt;
120  *   &lt;field name="bar"&gt;
121  *       &lt;field-validator type="required"&gt;
122  *           &lt;message&gt;You must enter a value for bar.&lt;/message&gt;
123  *       &lt;/field-validator&gt;
124  *       &lt;field-validator type="int"&gt;
125  *           &lt;param name="min"&gt;6&lt;/param&gt;
126  *           &lt;param name="max"&gt;10&lt;/param&gt;
127  *           &lt;message&gt;bar must be between ${min} and ${max}, current value is ${bar}.&lt;/message&gt;
128  *       &lt;/field-validator&gt;
129  *   &lt;/field&gt;
130  *   &lt;field name="bar2"&gt;
131  *       &lt;field-validator type="regex"&gt;
132  *           &lt;param name="expression"&gt;[0-9],[0-9]&lt;/param&gt;
133  *           &lt;message&gt;The value of bar2 must be in the format "x, y", where x and y are between 0 and 9&lt;/message&gt;
134  *      &lt;/field-validator&gt;
135  *   &lt;/field&gt;
136  *   &lt;field name="date"&gt;
137  *       &lt;field-validator type="date"&gt;
138  *           &lt;param name="min"&gt;12/22/2002&lt;/param&gt;
139  *           &lt;param name="max"&gt;12/25/2002&lt;/param&gt;
140  *           &lt;message&gt;The date must be between 12-22-2002 and 12-25-2002.&lt;/message&gt;
141  *       &lt;/field-validator&gt;
142  *   &lt;/field&gt;
143  *   &lt;field name="foo"&gt;
144  *       &lt;field-validator type="int"&gt;
145  *           &lt;param name="min"&gt;0&lt;/param&gt;
146  *           &lt;param name="max"&gt;100&lt;/param&gt;
147  *           &lt;message key="foo.range"&gt;Could not find foo.range!&lt;/message&gt;
148  *       &lt;/field-validator&gt;
149  *   &lt;/field&gt;
150  *   &lt;validator type="expression"&gt;
151  *       &lt;param name="expression"&gt;foo lt bar &lt;/param&gt;
152  *       &lt;message&gt;Foo must be greater than Bar. Foo = ${foo}, Bar = ${bar}.&lt;/message&gt;
153  *   &lt;/validator&gt;
154  * &lt;/validators&gt;
155  * <!-- END SNIPPET: exValidationRules1 -->
156  * </pre>
157  *
158  *
159  * <p>
160  * <!-- START SNIPPET: validationRules2 -->
161  * Here we can see the configuration of validators for the SimpleAction class.
162  * Validators (and field-validators) must have a type attribute, which refers
163  * to a name of an Validator registered with the ValidatorFactory as above.
164  * Validator elements may also have &lt;param&gt; elements with name and value attributes
165  * to set arbitrary parameters into the Validator instance. See below for discussion
166  * of the message element.
167  * <!-- END SNIPPET: validationRules2 -->
168  * </p>
169  *
170  *
171  *
172  * <!-- START SNIPPET: validationRules3 -->
173  * <p>Each Validator or Field-Validator element must define one message element inside
174  * the validator element body. The message element has 1 attributes, key which is not
175  * required. The body of the message tag is taken as the default message which should
176  * be added to the Action if the validator fails. Key gives a message key to look up
177  * in the Action's ResourceBundles using getText() from LocaleAware if the Action
178  * implements that interface (as ActionSupport does). This provides for Localized
179  * messages based on the Locale of the user making the request (or whatever Locale
180  * you've set into the LocaleAware Action). After either retrieving the message from
181  * the ResourceBundle using the Key value, or using the Default message, the current
182  * Validator is pushed onto the ValueStack, then the message is parsed for \$\{...\}
183  * sections which are replaced with the evaluated value of the string between the
184  * \$\{ and \}. This allows you to parameterize your messages with values from the
185  * Validator, the Action, or both.</p>
186  *
187  *
188  * <p>If the validator fails, the validator is pushed onto the ValueStack and the
189  * message - either the default or the locale-specific one if the key attribute is
190  * defined (and such a message exists) - is parsed for ${...} sections which are
191  * replaced with the evaluated value of the string between the ${ and }. This
192  * allows you to parameterize your messages with values from the validator, the
193  * Action, or both. </p>
194  *
195  * <p><b>NOTE:</b> Since validation rules are in an XML file, you must make sure
196  * you escape special characters. For example, notice that in the expression
197  * validator rule above we use "&amp;gt;" instead of "&gt;". Consult a resource on XML
198  * for the full list of characters that must be escaped. The most commonly used
199  * characters that must be escaped are: &amp; (use &amp;amp;), &gt; (user &amp;gt;), and &lt; (use &amp;lt;).</p>
200  *
201  * <p>Here is an example of a parameterized message:</p>
202  * <p>This will pull the min and max parameters from the IntRangeFieldValidator and
203  * the value of bar from the Action.</p>
204  * <!-- END SNIPPET: validationRules3 -->
205  *
206  * <pre>
207  * <!-- START SNIPPET: exValidationRules3 -->
208  *    bar must be between ${min} and ${max}, current value is ${bar}.
209  * <!-- END SNIPPET: exValidationRules3 -->
210  * </pre>
211  *
212  * <!-- START SNIPPET: validationRules4 -->
213  * <p>Another notable fact is that the provided message value is capable of containing OGNL expressions.
214  * Keeping this in mind, it is possible to construct quite sophisticated messages.</p>
215  * <p>See the following example to get an impression:</p>
216  *
217  * <!-- END SNIPPET: validationRules4 -->
218  *
219  * <pre>
220  * <!-- START SNIPPET: exValidationRules4 -->
221  *    &lt;message&gt;${getText("validation.failednotice")}! ${getText("reason")}: ${getText("validation.inputrequired")}&lt;/message&gt;
222  * <!-- END SNIPPET: exValidationRules4 -->
223  * </pre>
224  *
225  * @author Jason Carreira
226  * @author James House
227  */
228 public interface ValidatorFactory {
229 
230     /**
231      * Get a Validator that matches the given configuration.
232      *
233      * @param cfg  the configurator.
234      * @return  the validator.
235      */
236     Validator getValidator(ValidatorConfig cfg);
237 
238     /**
239      * Registers the given validator to the existing map of validators.
240      * This will <b>add</b> to the existing list.
241      *
242      * @param name    name of validator to add.
243      * @param className   the FQ classname of the validator.
244      */
245     void registerValidator(String name, String className);
246 
247     /**
248      * Lookup to get the FQ classname of the given validator name.
249      *
250      * @param name   name of validator to lookup.
251      * @return  the found FQ classname
252      * @throws IllegalArgumentException is thrown if the name is not found.
253      */
254     String lookupRegisteredValidatorType(String name);
255 }
