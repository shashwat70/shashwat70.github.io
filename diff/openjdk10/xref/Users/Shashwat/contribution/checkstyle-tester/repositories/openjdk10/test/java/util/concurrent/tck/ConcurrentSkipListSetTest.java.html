63      /**
64       * Returns a new set of given size containing consecutive
65       * Integers 0 ... n - 1.
66       */
67      private static ConcurrentSkipListSet<Integer> populatedSet(int n) {
68          ConcurrentSkipListSet<Integer> q =
69              new ConcurrentSkipListSet<Integer>();
70          assertTrue(q.isEmpty());
71          for (int i = n - 1; i >= 0; i -= 2)
72              assertTrue(q.add(new Integer(i)));
73          for (int i = (n & 1); i < n; i += 2)
74              assertTrue(q.add(new Integer(i)));
75          assertFalse(q.isEmpty());
76          assertEquals(n, q.size());
77          return q;
78      }

745     static NavigableSet<Integer> newSet(Class cl) throws Exception {
746         NavigableSet<Integer> result =
747             (NavigableSet<Integer>) cl.getConstructor().newInstance();
748         assertEquals(0, result.size());

753     void populate(NavigableSet<Integer> set, int limit, BitSet bs) {
754         for (int i = 0, n = 2 * limit / 3; i < n; i++) {
755             int element = rnd.nextInt(limit);
756             put(set, element, bs);
757         }
758     }

760     void mutateSet(NavigableSet<Integer> set, int min, int max, BitSet bs) {
761         int size = set.size();
762         int rangeSize = max - min + 1;
763 
764         // Remove a bunch of entries directly
765         for (int i = 0, n = rangeSize / 2; i < n; i++) {
766             remove(set, min - 5 + rnd.nextInt(rangeSize + 10), bs);
767         }
768 
769         // Remove a bunch of entries with iterator
770         for (Iterator<Integer> it = set.iterator(); it.hasNext(); ) {
771             if (rnd.nextBoolean()) {
772                 bs.clear(it.next());
773                 it.remove();
774             }
775         }
776 
777         // Add entries till we're back to original size
778         while (set.size() < size) {
779             int element = min + rnd.nextInt(rangeSize);
780             assertTrue(element >= min && element <= max);
781             put(set, element, bs);
782         }
783     }

785     void mutateSubSet(NavigableSet<Integer> set, int min, int max,
786                       BitSet bs) {
787         int size = set.size();
788         int rangeSize = max - min + 1;
789 
790         // Remove a bunch of entries directly
791         for (int i = 0, n = rangeSize / 2; i < n; i++) {
792             remove(set, min - 5 + rnd.nextInt(rangeSize + 10), bs);
793         }
794 
795         // Remove a bunch of entries with iterator
796         for (Iterator<Integer> it = set.iterator(); it.hasNext(); ) {
797             if (rnd.nextBoolean()) {
798                 bs.clear(it.next());
799                 it.remove();
800             }
801         }
802 
803         // Add entries till we're back to original size
804         while (set.size() < size) {
805             int element = min - 5 + rnd.nextInt(rangeSize + 10);
806             if (element >= min && element <= max) {
807                 put(set, element, bs);
808             } else {
809                 try {
810                     set.add(element);
811                     shouldThrow();
812                 } catch (IllegalArgumentException success) {}
813             }
814         }
815     }

827     void bashSubSet(NavigableSet<Integer> set,
828                     int min, int max, boolean ascending,
829                     BitSet bs) {
830         check(set, min, max, ascending, bs);
831         check(set.descendingSet(), min, max, !ascending, bs);
832 
833         mutateSubSet(set, min, max, bs);
834         check(set, min, max, ascending, bs);
835         check(set.descendingSet(), min, max, !ascending, bs);
836 
837         // Recurse
838         if (max - min < 2)
839             return;
840         int midPoint = (min + max) / 2;
841 
842         // headSet - pick direction and endpoint inclusion randomly
843         boolean incl = rnd.nextBoolean();
844         NavigableSet<Integer> hm = set.headSet(midPoint, incl);
845         if (ascending) {
846             if (rnd.nextBoolean())
847                 bashSubSet(hm, min, midPoint - (incl ? 0 : 1), true, bs);
848             else
849                 bashSubSet(hm.descendingSet(), min, midPoint - (incl ? 0 : 1),
850                            false, bs);
851         } else {
852             if (rnd.nextBoolean())
853                 bashSubSet(hm, midPoint + (incl ? 0 : 1), max, false, bs);
854             else
855                 bashSubSet(hm.descendingSet(), midPoint + (incl ? 0 : 1), max,
856                            true, bs);
857         }
858 
859         // tailSet - pick direction and endpoint inclusion randomly
860         incl = rnd.nextBoolean();
861         NavigableSet<Integer> tm = set.tailSet(midPoint,incl);
862         if (ascending) {
863             if (rnd.nextBoolean())
864                 bashSubSet(tm, midPoint + (incl ? 0 : 1), max, true, bs);
865             else
866                 bashSubSet(tm.descendingSet(), midPoint + (incl ? 0 : 1), max,
867                            false, bs);
868         } else {
869             if (rnd.nextBoolean()) {
870                 bashSubSet(tm, min, midPoint - (incl ? 0 : 1), false, bs);
871             } else {
872                 bashSubSet(tm.descendingSet(), min, midPoint - (incl ? 0 : 1),
873                            true, bs);
874             }
875         }
876 
877         // subSet - pick direction and endpoint inclusion randomly
878         int rangeSize = max - min + 1;
879         int[] endpoints = new int[2];
880         endpoints[0] = min + rnd.nextInt(rangeSize);
881         endpoints[1] = min + rnd.nextInt(rangeSize);
882         Arrays.sort(endpoints);
883         boolean lowIncl = rnd.nextBoolean();
884         boolean highIncl = rnd.nextBoolean();
885         if (ascending) {
886             NavigableSet<Integer> sm = set.subSet(
887                 endpoints[0], lowIncl, endpoints[1], highIncl);
888             if (rnd.nextBoolean())
889                 bashSubSet(sm, endpoints[0] + (lowIncl ? 0 : 1),
890                            endpoints[1] - (highIncl ? 0 : 1), true, bs);
891             else
892                 bashSubSet(sm.descendingSet(), endpoints[0] + (lowIncl ? 0 : 1),
893                            endpoints[1] - (highIncl ? 0 : 1), false, bs);
894         } else {
895             NavigableSet<Integer> sm = set.subSet(
896                 endpoints[1], highIncl, endpoints[0], lowIncl);
897             if (rnd.nextBoolean())
898                 bashSubSet(sm, endpoints[0] + (lowIncl ? 0 : 1),
899                            endpoints[1] - (highIncl ? 0 : 1), false, bs);
900             else
901                 bashSubSet(sm.descendingSet(), endpoints[0] + (lowIncl ? 0 : 1),
902                            endpoints[1] - (highIncl ? 0 : 1), true, bs);
903         }
904     }

906     /**
907      * min and max are both inclusive.  If max < min, interval is empty.
908      */
909     void check(NavigableSet<Integer> set,
910                final int min, final int max, final boolean ascending,
911                final BitSet bs) {
912         class ReferenceSet {
913             int lower(int element) {
914                 return ascending ?
915                     lowerAscending(element) : higherAscending(element);
916             }
917             int floor(int element) {
918                 return ascending ?
919                     floorAscending(element) : ceilingAscending(element);
920             }
921             int ceiling(int element) {
922                 return ascending ?
923                     ceilingAscending(element) : floorAscending(element);
924             }
925             int higher(int element) {
926                 return ascending ?
927                     higherAscending(element) : lowerAscending(element);
928             }
929             int first() {
930                 return ascending ? firstAscending() : lastAscending();
931             }
932             int last() {
933                 return ascending ? lastAscending() : firstAscending();
934             }
935             int lowerAscending(int element) {
936                 return floorAscending(element - 1);
937             }
938             int floorAscending(int element) {
939                 if (element < min)
940                     return -1;
941                 else if (element > max)
942                     element = max;
943 
944                 // BitSet should support this! Test would run much faster
945                 while (element >= min) {
946                     if (bs.get(element))
947                         return element;
948                     element--;
949                 }
950                 return -1;
951             }
952             int ceilingAscending(int element) {
953                 if (element < min)
954                     element = min;
955                 else if (element > max)
956                     return -1;
957                 int result = bs.nextSetBit(element);
958                 return result > max ? -1 : result;
959             }
960             int higherAscending(int element) {
961                 return ceilingAscending(element + 1);
962             }
963             private int firstAscending() {
964                 int result = ceilingAscending(min);
965                 return result > max ? -1 : result;
966             }
967             private int lastAscending() {
968                 int result = floorAscending(max);
969                 return result < min ? -1 : result;
970             }
971         }
972         ReferenceSet rs = new ReferenceSet();
973 
974         // Test contents using containsElement
975         int size = 0;
976         for (int i = min; i <= max; i++) {
977             boolean bsContainsI = bs.get(i);
978             assertEquals(bsContainsI, set.contains(i));
979             if (bsContainsI)
980                 size++;
981         }
982         assertEquals(size, set.size());
983 
984         // Test contents using contains elementSet iterator
985         int size2 = 0;
986         int previousElement = -1;
987         for (int element : set) {
988             assertTrue(bs.get(element));
989             size2++;
990             assertTrue(previousElement < 0 || (ascending ?
991                 element - previousElement > 0 : element - previousElement < 0));
992             previousElement = element;
993         }
994         assertEquals(size2, size);
995 
996         // Test navigation ops
997         for (int element = min - 1; element <= max + 1; element++) {
998             assertEq(set.lower(element), rs.lower(element));
999             assertEq(set.floor(element), rs.floor(element));
1000             assertEq(set.higher(element), rs.higher(element));
1001             assertEq(set.ceiling(element), rs.ceiling(element));
1002         }
1003 
1004         // Test extrema
1005         if (set.size() != 0) {
1006             assertEq(set.first(), rs.first());
1007             assertEq(set.last(), rs.last());
1008         } else {
1009             assertEq(rs.first(), -1);
1010             assertEq(rs.last(),  -1);
1011             try {
1012                 set.first();
1013                 shouldThrow();
1014             } catch (NoSuchElementException success) {}
1015             try {
1016                 set.last();
1017                 shouldThrow();
1018             } catch (NoSuchElementException success) {}
1019         }
1020     }
