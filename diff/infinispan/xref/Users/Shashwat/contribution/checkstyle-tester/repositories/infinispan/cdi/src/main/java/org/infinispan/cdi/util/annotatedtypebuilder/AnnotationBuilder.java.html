1   package org.infinispan.cdi.util.annotatedtypebuilder;
2   
3   import java.lang.annotation.Annotation;
4   import java.util.HashMap;
5   import java.util.HashSet;
6   import java.util.Iterator;
7   import java.util.Map;
8   import java.util.Set;
9   
10  import org.infinispan.cdi.util.Reflections;
11  import org.infinispan.cdi.util.logging.Log;
12  import org.infinispan.util.logging.LogFactory;
13  
14  /**
15   * Helper class used to build annotation stores
16   *
17   * @author Stuart Douglas
18   */
19  public class AnnotationBuilder {
20  
21     private static final Log log = LogFactory.getLog(AnnotationBuilder.class, Log.class);
22  
23     private final Map<Class<? extends Annotation>, Annotation> annotationMap;
24     private final Set<Annotation> annotationSet;
25  
26     AnnotationBuilder() {
27        this.annotationMap = new HashMap<Class<? extends Annotation>, Annotation>();
28        this.annotationSet = new HashSet<Annotation>();
29     }
30  
31     public AnnotationBuilder add(Annotation annotation) {
32        if (annotation == null) {
33           throw log.parameterMustNotBeNull("annotation");
34        }
35        annotationSet.add(annotation);
36        annotationMap.put(annotation.annotationType(), annotation);
37        return this;
38     }
39  
40     public AnnotationBuilder remove(Class<? extends Annotation> annotationType) {
41        if (annotationType == null) {
42           throw log.parameterMustNotBeNull("annotationType");
43        }
44  
45        Iterator<Annotation> it = annotationSet.iterator();
46        while (it.hasNext()) {
47           Annotation an = it.next();
48           if (annotationType.isAssignableFrom(an.annotationType())) {
49              it.remove();
50           }
51        }
52        annotationMap.remove(annotationType);
53        return this;
54     }
55  
56     AnnotationStore create() {
57        return new AnnotationStore(annotationMap, annotationSet);
58     }
59  
60     public AnnotationBuilder addAll(AnnotationStore annotations) {
61        for (Annotation annotation : annotations.getAnnotations()) {
62           add(annotation);
63        }
64        return this;
65     }
66  
67     public <T extends Annotation> T getAnnotation(Class<T> anType) {
68        return Reflections.cast(annotationMap.get(anType));
69     }
70  
71     public boolean isAnnotationPresent(Class<?> annotationType) {
72        return annotationMap.containsKey(annotationType);
73     }
74  
75     @Override
76     public String toString() {
77        return annotationSet.toString();
78     }
79  
80  }
