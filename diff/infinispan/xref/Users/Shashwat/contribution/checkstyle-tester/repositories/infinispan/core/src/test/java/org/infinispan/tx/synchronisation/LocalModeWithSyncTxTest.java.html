4   import org.infinispan.manager.EmbeddedCacheManager;
5   import org.infinispan.test.fwk.TestCacheManagerFactory;
6   import org.infinispan.transaction.lookup.DummyTransactionManagerLookup;
7   import org.infinispan.transaction.tm.DummyTransaction;
8   import org.infinispan.tx.LocalModeTxTest;
9   import org.testng.annotations.Test;
10  
11  import javax.transaction.NotSupportedException;
12  import javax.transaction.SystemException;
13  
14  import static org.testng.Assert.assertEquals;
15  
16  /**
17   * @author Mircea.Markus@jboss.com
18   * @since 5.0
19   */
20  @Test(groups = "functional", testName = "tx.synchronisation.LocalModeWithSyncTxTest")
21  public class LocalModeWithSyncTxTest extends LocalModeTxTest {
22  
23     @Override
24     protected EmbeddedCacheManager createCacheManager() {
25        ConfigurationBuilder config = getDefaultStandaloneCacheConfig(true);
26        config.transaction().transactionManagerLookup(new DummyTransactionManagerLookup()).useSynchronization(true);
27        return TestCacheManagerFactory.createCacheManager(config);
28     }
29  
30     public void testSyncRegisteredWithCommit() throws Exception {
31        DummyTransaction dt = startTx();
32        tm().commit();
33        assertEquals(0, dt.getEnlistedResources().size());
34        assertEquals(0, dt.getEnlistedSynchronization().size());
35        assertEquals("v", cache.get("k"));
36     }
37  
38     public void testSyncRegisteredWithRollback() throws Exception {
39        DummyTransaction dt = startTx();
40        tm().rollback();
41        assertEquals(null, cache.get("k"));
42        assertEquals(0, dt.getEnlistedResources().size());
43        assertEquals(0, dt.getEnlistedSynchronization().size());
44     }
45  
46     private DummyTransaction startTx() throws NotSupportedException, SystemException {
47        tm().begin();
48        cache.put("k","v");
49        DummyTransaction dt = (DummyTransaction) tm().getTransaction();
50        assertEquals(0, dt.getEnlistedResources().size());
51        assertEquals(1, dt.getEnlistedSynchronization().size());
52        cache.put("k2","v2");
53        assertEquals(0, dt.getEnlistedResources().size());
54        assertEquals(1, dt.getEnlistedSynchronization().size());
55        return dt;
56     }
57  }
