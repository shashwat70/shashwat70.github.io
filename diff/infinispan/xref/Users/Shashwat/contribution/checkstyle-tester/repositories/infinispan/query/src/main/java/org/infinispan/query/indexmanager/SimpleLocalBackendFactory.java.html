1   package org.infinispan.query.indexmanager;
2   
3   import java.util.Properties;
4   
5   import org.hibernate.search.backend.BackendFactory;
6   import org.hibernate.search.backend.spi.BackendQueueProcessor;
7   import org.hibernate.search.indexes.spi.DirectoryBasedIndexManager;
8   import org.hibernate.search.spi.WorkerBuildContext;
9   
10  /**
11   * Factory of local backends to simplify lazy initialization.
12   *
13   * @author Sanne Grinovero <sanne@hibernate.org> (C) 2014 Red Hat Inc.
14   * @since 7.0
15   */
16  public class SimpleLocalBackendFactory implements LocalBackendFactory {
17  
18     private final DirectoryBasedIndexManager indexManager;
19     private Properties cfg;
20     private WorkerBuildContext buildContext;
21  
22  
23     SimpleLocalBackendFactory(DirectoryBasedIndexManager indexManager, Properties cfg, WorkerBuildContext buildContext) {
24        this.indexManager = indexManager;
25        this.cfg = cfg;//TODO wrap to override some dangerous properties?
26        this.buildContext = buildContext; //TODO deferring the buildContext in Hibernate Search is unexpected: think of a better way
27     }
28  
29     @Override
30     public IndexingBackend createLocalIndexingBackend() {
31        //Force to use the "lucene" backend?
32        BackendQueueProcessor localBackend = BackendFactory.createBackend(indexManager, buildContext, cfg);
33        IndexingBackend backend = new LocalIndexingBackend(localBackend);
34        return backend;
35     }
36  
37  }
