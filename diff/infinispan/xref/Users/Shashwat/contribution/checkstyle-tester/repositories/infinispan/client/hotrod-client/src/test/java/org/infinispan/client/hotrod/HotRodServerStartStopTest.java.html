20  @Test(testName = "client.hotrod.HotRodServerStartStopTest", groups = "functional")
21  public class HotRodServerStartStopTest extends MultipleCacheManagersTest {
22     private HotRodServer hotRodServer1;
23     private HotRodServer hotRodServer2;
24  
25     @AfterMethod
26     @Override
27     protected void clearContent() throws Throwable {
28     }
29  
30     @Override
31     protected void createCacheManagers() throws Throwable {
32        ConfigurationBuilder builder = hotRodCacheConfiguration(
33              getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false));
34        addClusterEnabledCacheManager(builder);
35        addClusterEnabledCacheManager(builder);
36  
37        hotRodServer1 = HotRodClientTestingUtil.startHotRodServer(manager(0));
38        hotRodServer2 = HotRodClientTestingUtil.startHotRodServer(manager(1));
39  
40        assert manager(0).getCache() != null;
41        assert manager(1).getCache() != null;
42  
43        waitForClusterToForm();
44     }
45  
46     public void testTouchServer() {
47        RemoteCacheManager remoteCacheManager = new RemoteCacheManager("localhost", hotRodServer1.getPort(), true);
48        RemoteCache<Object, Object> remoteCache = remoteCacheManager.getCache();
49        remoteCache.put("k", "v");
50        assertEquals("v", remoteCache.get("k"));
51        killRemoteCacheManager(remoteCacheManager);
52     }
53  
54     @Test (dependsOnMethods = "testTouchServer")
55     public void testHrServerStop() {
56        killServers(hotRodServer1, hotRodServer2);
57     }
