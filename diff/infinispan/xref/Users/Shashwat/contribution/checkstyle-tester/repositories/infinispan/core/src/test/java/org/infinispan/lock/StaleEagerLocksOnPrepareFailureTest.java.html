12  import org.infinispan.test.TestingUtil;

15  import org.infinispan.transaction.LockingMode;
16  import org.testng.annotations.Test;
17  
18  import static org.testng.Assert.assertNull;
19  
20  @Test(testName = "lock.StaleEagerLocksOnPrepareFailureTest", groups = "functional")
21  @CleanupAfterMethod
22  public class StaleEagerLocksOnPrepareFailureTest extends MultipleCacheManagersTest {
23  
24     Cache<MagicKey, String> c1, c2;
25  
26     @Override
27     protected void createCacheManagers() throws Throwable {
28        ConfigurationBuilder cfg = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
29        cfg
30           .transaction()
31              .lockingMode(LockingMode.PESSIMISTIC)
32              .useSynchronization(false)
33              .recovery()
34                 .disable()
35           .locking()
36              .lockAcquisitionTimeout(100);
37        EmbeddedCacheManager cm1 = TestCacheManagerFactory.createClusteredCacheManager(cfg);
38        EmbeddedCacheManager cm2 = TestCacheManagerFactory.createClusteredCacheManager(cfg);
39        registerCacheManager(cm1, cm2);
40        c1 = cm1.getCache();
41        c2 = cm2.getCache();
42        waitForClusterToForm();
43     }
44  
45     public void testNoModsCommit() throws Exception {
46        doTest(false);
47     }
48  
49     public void testModsCommit() throws Exception {
50        doTest(true);
51     }
52  
53     private void doTest(boolean mods) throws Exception {
54        // force the prepare command to fail on c2
55        FailInterceptor interceptor = new FailInterceptor();
56        interceptor.failFor(PrepareCommand.class);
57        InterceptorChain ic = TestingUtil.extractComponent(c2, InterceptorChain.class);
58        ic.addInterceptorBefore(interceptor, TxDistributionInterceptor.class);
59  
60        MagicKey k1 = new MagicKey("k1", c1);
61        MagicKey k2 = new MagicKey("k2", c2);
62  
63        tm(c1).begin();
64        if (mods) {
65           c1.put(k1, "v1");
66           c1.put(k2, "v2");
67  
68           assertKeyLockedCorrectly(k1);
69           assertKeyLockedCorrectly(k2);
70        } else {
71           c1.getAdvancedCache().lock(k1);
72           c1.getAdvancedCache().lock(k2);
73  
74           assertNull(c1.get(k1));
75           assertNull(c1.get(k2));
76  
77           assertKeyLockedCorrectly(k1);
78           assertKeyLockedCorrectly(k2);
79        }
80  
81        try {
82           tm(c1).commit();
83           assert false : "Commit should have failed";
84        } catch (Exception e) {
85           // expected
86        }
87  
88        assertNotLocked(c1, k1);
89        assertNotLocked(c2, k1);
90        assertNotLocked(c1, k2);
91        assertNotLocked(c2, k2);
92     }
93  }
