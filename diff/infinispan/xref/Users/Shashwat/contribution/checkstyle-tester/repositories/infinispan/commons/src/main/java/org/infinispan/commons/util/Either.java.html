1   package org.infinispan.commons.util;
2   
3   import java.util.NoSuchElementException;
4   
5   public abstract class Either<A, B> {
6   
7      public abstract Type type();
8      public abstract A left();
9      public abstract B right();
10  
11     public static <A, B> Either<A, B> newLeft(A a) { return new Left<>(a); }
12     public static <A, B> Either<A, B> newRight(B b) { return new Right<>(b); }
13  
14     public static enum Type {
15        LEFT, RIGHT
16     }
17  
18     private static class Left<A, B> extends Either<A, B> {
19        private A leftValue;
20        Left(A a) { leftValue = a; }
21        @Override public Type type() { return Type.LEFT; }
22        @Override public A left() { return leftValue; }
23        @Override public B right() { throw new NoSuchElementException("Either.right() called on Left"); }
24     }
25  
26     private static class Right<A, B> extends Either<A, B> {
27        private B rightValue;
28        Right(B b) { rightValue = b; }
29        @Override public Type type() { return Type.RIGHT; }
30        @Override public A left() { throw new NoSuchElementException("Either.left() called on Right"); }
31        @Override public B right() {  return rightValue; }
32     }
33  
34  }
