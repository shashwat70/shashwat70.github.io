1   package org.infinispan.server.endpoint.subsystem;
2   
3   import java.security.Principal;
4   import java.util.Collection;
5   
6   import javax.security.auth.Subject;
7   
8   import org.infinispan.server.core.security.SubjectUserInfo;
9   
10  /**
11   * RealmSubjectUserInfo.
12   *
13   * @author Tristan Tarrant
14   * @since 7.0
15   */
16  public class RealmSubjectUserInfo implements SubjectUserInfo {
17     private final String userName;
18     private final Subject subject;
19  
20     public RealmSubjectUserInfo(org.jboss.as.core.security.SubjectUserInfo userInfo) {
21         this.userName = userInfo.getUserName();
22         this.subject = userInfo.getSubject();
23     }
24  
25     RealmSubjectUserInfo(String userName, Subject subject) {
26        this.userName = userName;
27        this.subject = subject;
28     }
29  
30     @Override
31     public String getUserName() {
32        return userName;
33     }
34  
35     @Override
36     public Collection<Principal> getPrincipals() {
37        return subject.getPrincipals();
38     }
39  
40     @Override
41     public Subject getSubject() {
42        return subject;
43     }
44  
45  }
