8   /**

18     private final String saslMechanism;

25        this.clientSubject = clientSubject;
26        this.saslMechanism = saslMechanism;
27        this.saslProperties = saslProperties;
28        this.serverName = serverName;

33     }
34

37     }
38  
39     public String saslMechanism() {

42

45     }
46  
47     public String serverName() {
48        return serverName;

52        return clientSubject;

55  }
