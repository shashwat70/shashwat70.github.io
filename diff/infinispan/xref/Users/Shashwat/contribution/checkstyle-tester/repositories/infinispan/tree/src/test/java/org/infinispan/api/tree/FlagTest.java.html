21  @Test(groups = "functional", testName = "api.tree.FlagTest")
22  public class FlagTest extends MultipleCacheManagersTest {
23     private Cache<String, String> cache1, cache2;
24     private TreeCache<String, String> treeCache1, treeCache2;
25     private static final String KEY = "key";
26     private static final Log log = LogFactory.getLog(FlagTest.class);
27  
28     @Override
29     protected void createCacheManagers() throws Throwable {
30        ConfigurationBuilder cb = getDefaultClusteredCacheConfig(CacheMode.INVALIDATION_SYNC, true);
31        cb.invocationBatching().enable();
32        createClusteredCaches(2, "invalidatedFlagCache", cb);
33        cache1 = cache(0, "invalidatedFlagCache");
34        cache2 = cache(1, "invalidatedFlagCache");
35        TreeCacheFactory tcf = new TreeCacheFactory();
36        treeCache1 = tcf.createTreeCache(cache1);
37        treeCache2 = tcf.createTreeCache(cache2);
38     }
39  
40     public void testTreeCacheLocalPut() throws Exception {
41        final Fqn fqn = Fqn.fromElements("TEST");
42        treeCache1.put(fqn, KEY, "1", Flag.CACHE_MODE_LOCAL);
43        treeCache2.put(fqn, KEY, "2", Flag.CACHE_MODE_LOCAL);
44        assert "2".equals(treeCache2.get(fqn, KEY)) : "treeCache2 was updated locally";
45        assert "1".equals(treeCache1.get(fqn, KEY)) : "treeCache1 should not be invalidated in case of LOCAL put in treeCache2";
46  
47        String fqnString = "fqnAsString";
48        treeCache1.put(fqnString, KEY, "3", Flag.CACHE_MODE_LOCAL);
49        treeCache2.put(fqnString, KEY, "4", Flag.CACHE_MODE_LOCAL);
50        assert "4".equals(treeCache2.get(fqnString, KEY)) : "treeCache2 was updated locally";
51        assert "3".equals(treeCache1.get(fqnString, KEY)) : "treeCache1 should not be invalidated in case of LOCAL put in treeCache2";
52     }
53  
54     public void testWithFlags() {
55        AdvancedCache<String, String> localCache1 = cache1.getAdvancedCache().withFlags(Flag.CACHE_MODE_LOCAL);
56        AdvancedCache<String, String> localCache2 = cache2.getAdvancedCache().withFlags(Flag.CACHE_MODE_LOCAL);
57        TreeCache<String, String> treeCache1 = new TreeCacheFactory().createTreeCache(localCache1);
58        TreeCache<String, String> treeCache2 = new TreeCacheFactory().createTreeCache(localCache2);
59        final Fqn fqn = Fqn.fromElements("TEST_WITH_FLAGS");
60        treeCache1.put(fqn, KEY, "1");
61        treeCache2.put(fqn, KEY, "2");
62        assert "2".equals(treeCache2.get(fqn, KEY)) : "treeCache2 was updated locally";
63        assert "1".equals(treeCache1.get(fqn, KEY)) : "treeCache1 should not be invalidated in case of LOCAL put in treeCache2";
64     }
