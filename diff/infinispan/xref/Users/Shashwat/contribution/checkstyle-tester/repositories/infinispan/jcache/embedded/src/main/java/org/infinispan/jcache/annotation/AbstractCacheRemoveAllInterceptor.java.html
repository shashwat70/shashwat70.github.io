1   package org.infinispan.jcache.annotation;
2   
3   import java.io.Serializable;
4   
5   import javax.cache.Cache;
6   import javax.cache.annotation.CacheKeyInvocationContext;
7   import javax.cache.annotation.CacheRemoveAll;
8   import javax.cache.annotation.CacheResolver;
9   import javax.cache.annotation.GeneratedCacheKey;
10  import javax.interceptor.InvocationContext;
11  
12  import org.infinispan.jcache.logging.Log;
13  
14  /**
15   * <p>{@link javax.cache.annotation.CacheRemoveAll} interceptor implementation. This interceptor uses the following algorithm describes in
16   * JSR-107.</p>
17   *
18   * <p>The interceptor that intercepts method annotated with {@code @CacheRemoveAll} must do the following, remove all
19   * entries associated with the cache. The removeAll occurs after the method body is executed. This can be overridden by
20   * specifying a afterInvocation attribute value of false. If afterInvocation is true and the annotated method throws an
21   * exception, the removeAll will not happen.</p>
22   *
23   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
24   */
25  public abstract class AbstractCacheRemoveAllInterceptor implements Serializable {
26  
27     private static final long serialVersionUID = -8763819640664021763L;
28  
29     private final CacheResolver cacheResolver;
30     private final CacheKeyInvocationContextFactory contextFactory;
31  
32     public AbstractCacheRemoveAllInterceptor(CacheResolver cacheResolver, CacheKeyInvocationContextFactory contextFactory) {
33        this.cacheResolver = cacheResolver;
34        this.contextFactory = contextFactory;
35     }
36  
37     public Object cacheRemoveAll(InvocationContext invocationContext) throws Exception {
38        if (getLog().isTraceEnabled()) {
39           getLog().tracef("Interception of method named '%s'", invocationContext.getMethod().getName());
40        }
41  
42        final CacheKeyInvocationContext<CacheRemoveAll> cacheKeyInvocationContext =
43              contextFactory.getCacheKeyInvocationContext(invocationContext);
44        final CacheRemoveAll cacheRemoveAll = cacheKeyInvocationContext.getCacheAnnotation();
45        final Cache<GeneratedCacheKey, Object> cache = cacheResolver.resolveCache(cacheKeyInvocationContext);
46  
47        if (!cacheRemoveAll.afterInvocation()) {
48           cache.clear();
49           if (getLog().isTraceEnabled()) {
50              getLog().tracef("Clear cache '%s' before method invocation", cache.getName());
51           }
52        }
53  
54        final Object result = invocationContext.proceed();
55  
56        if (cacheRemoveAll.afterInvocation()) {
57           cache.clear();
58           if (getLog().isTraceEnabled()) {
59              getLog().tracef("Clear cache '%s' after method invocation", cache.getName());
60           }
61        }
62  
63        return result;
64     }
65  
66     protected abstract Log getLog();
67  
68  }
