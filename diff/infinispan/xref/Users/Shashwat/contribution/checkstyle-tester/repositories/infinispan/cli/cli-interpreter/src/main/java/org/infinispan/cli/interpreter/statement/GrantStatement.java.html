1   package org.infinispan.cli.interpreter.statement;
2   
3   import org.infinispan.cli.interpreter.logging.Log;
4   import org.infinispan.cli.interpreter.result.EmptyResult;
5   import org.infinispan.cli.interpreter.result.Result;
6   import org.infinispan.cli.interpreter.result.StatementException;
7   import org.infinispan.cli.interpreter.session.Session;
8   import org.infinispan.configuration.global.GlobalAuthorizationConfiguration;
9   import org.infinispan.security.impl.ClusterRoleMapper;
10  import org.infinispan.util.logging.LogFactory;
11  
12  /**
13   *
14   * GrantStatement adds a role mapping to a user
15   *
16   * @author Tristan Tarrant
17   * @since 7.0
18   */
19  public class GrantStatement implements Statement {
20     private static final Log log = LogFactory.getLog(GrantStatement.class, Log.class);
21  
22     private final String principalName;
23     private final String roleName;
24  
25     public GrantStatement(String roleName, String principalName) {
26        this.roleName = roleName;
27        this.principalName = principalName;
28     }
29  
30     @Override
31     public Result execute(Session session) throws StatementException {
32        GlobalAuthorizationConfiguration gac = session.getCacheManager().getCacheManagerConfiguration().security().authorization();
33        if (!gac.enabled()) {
34           throw log.authorizationNotEnabledOnContainer();
35        }
36        if (!(gac.principalRoleMapper() instanceof ClusterRoleMapper)) {
37           throw log.noClusterPrincipalMapper("GRANT");
38        }
39        ClusterRoleMapper cpm = (ClusterRoleMapper) gac.principalRoleMapper();
40        cpm.grant(roleName, principalName);
41        return EmptyResult.RESULT;
42     }
43  
44  }
