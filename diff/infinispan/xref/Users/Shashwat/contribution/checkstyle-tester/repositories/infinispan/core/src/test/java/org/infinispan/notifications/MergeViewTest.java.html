1   package org.infinispan.notifications;
2   
3   import org.infinispan.configuration.cache.CacheMode;
4   import org.infinispan.notifications.cachemanagerlistener.annotation.Merged;
5   import org.infinispan.notifications.cachemanagerlistener.event.MergeEvent;
6   import org.infinispan.test.MultipleCacheManagersTest;
7   import org.infinispan.test.TestingUtil;
8   import org.infinispan.test.fwk.TransportFlags;
9   import org.infinispan.util.logging.Log;
10  import org.infinispan.util.logging.LogFactory;
11  import org.jgroups.protocols.DISCARD;
12  import org.testng.annotations.Test;
13  
14  import static org.testng.AssertJUnit.assertEquals;
15  
16  /**
17   * @author Mircea.Markus@jboss.com
18   */
19  @Test(groups = "functional", testName = "notifications.MergeViewTest")
20  public class MergeViewTest extends MultipleCacheManagersTest {
21  
22     private static final Log log = LogFactory.getLog(MergeViewTest.class);
23  
24     private DISCARD discard;
25     private MergeListener ml0;
26     private MergeListener ml1;
27  
28     @Override
29     protected void createCacheManagers() throws Throwable {
30        addClusterEnabledCacheManager(getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true),
31                                      new TransportFlags().withMerge(true));
32  
33        ml0 = new MergeListener();
34        manager(0).addListener(ml0);
35  
36        discard = TestingUtil.getDiscardForCache(cache(0));
37        discard.setDiscardAll(true);
38  
39        addClusterEnabledCacheManager(getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true),
40                                      new TransportFlags().withMerge(true));
41        ml1 = new MergeListener();
42        manager(1).addListener(ml1);
43  
44        cache(0).put("k", "v0");
45        cache(1).put("k", "v1");
46        Thread.sleep(2000);
47  
48  
49        assert advancedCache(0).getRpcManager().getTransport().getMembers().size() == 1;
50        assert advancedCache(1).getRpcManager().getTransport().getMembers().size() == 1;
51     }
52  
53     public void testMergeViewHappens() {
54        discard.setDiscardAll(false);
55        TestingUtil.blockUntilViewsReceived(60000, cache(0), cache(1));
56        TestingUtil.waitForRehashToComplete(cache(0), cache(1));
57  
58        assert ml0.isMerged && ml1.isMerged;
59  
60        cache(0).put("k", "v2");
61        assertEquals(cache(0).get("k"), "v2");
62        assertEquals(cache(1).get("k"), "v2");
63     }
64  
65     @Listener
66     public static class MergeListener {
67        volatile boolean isMerged;
68  
69        @Merged
70        public void viewMerged(MergeEvent vce) {
71           log.info("vce = " + vce);
72           isMerged = true;
73        }
74     }
75  }
