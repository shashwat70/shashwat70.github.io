22  package org.jboss.as.clustering.jgroups.subsystem;
23  
24  import java.util.HashMap;
25  import java.util.Map;
26  
27  /**
28   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
29   */
30  public enum Element {
31      // must be first
32      UNKNOWN(null),
33  
34      PROPERTY(ModelKeys.PROPERTY),
35      PROTOCOL(ModelKeys.PROTOCOL),
36      RELAY(ModelKeys.RELAY),
37      REMOTE_SITE(ModelKeys.REMOTE_SITE),
38      SASL(ModelKeys.SASL),
39      STACK(ModelKeys.STACK),
40      TRANSPORT(ModelKeys.TRANSPORT),
41      ;
42  
43      private final String name;
44  
45      Element(final String name) {
46          this.name = name;
47      }
48  
49      /**
50       * Get the local name of this element.
51       *
52       * @return the local name
53       */
54      public String getLocalName() {
55          return name;
56      }
57  
58      private static final Map<String, Element> elements;
59  
60      static {
61          final Map<String, Element> map = new HashMap<String, Element>();
62          for (Element element : values()) {
63              final String name = element.getLocalName();
64              if (name != null) map.put(name, element);
65          }
66          elements = map;
67      }
68  
69      public static Element forName(String localName) {
70          final Element element = elements.get(localName);
71          return element == null ? UNKNOWN : element;
72      }
73  }
