29  @Test(groups = "functional" , testName="statetransfer.StateTransferLargeObjectTest")
30  public class StateTransferLargeObjectTest extends MultipleCacheManagersTest {
31  
32     private static final Log log = LogFactory.getLog(StateTransferLargeObjectTest.class);
33  
34     private Cache<Integer, BigObject> c0;
35     private Cache<Integer, BigObject> c1;
36     private Cache<Integer, BigObject> c2;
37     private Cache<Integer, BigObject> c3;
38     private Map<Integer, BigObject> cache;
39  
40     private final Random rnd = new Random();
41  
42     @Override
43     protected void createCacheManagers() throws Throwable {
44        ConfigurationBuilder builder = TestCacheManagerFactory.getDefaultCacheConfiguration(true);
45        builder.clustering().cacheMode(CacheMode.DIST_SYNC)
46              .l1().disable()
47              .clustering().stateTransfer().fetchInMemoryState(true)
48              .locking().useLockStriping(false)
49              .clustering().hash().numOwners(3).numSegments(60);
50        createCluster(builder, 4);
51  
52        c0 = cache(0);
53        c1 = cache(1);
54        c2 = cache(2);
55        c3 = cache(3);
56        waitForClusterToForm();
57        log.debug("Rehash is complete!");
58        cache = new HashMap<Integer, BigObject>();
59     }
60  
61     public void testForFailure() {
62        final int num = 1000;
63        for (int i = 0; i < num; i++) {
64           BigObject bigObject = createBigObject(i, "prefix");
65           cache.put(i, bigObject);
66           c0.put(i, bigObject);
67        }
68  
69        for (int i = 0; i < num; i++) {
70           assertTrue(c0.get(i) instanceof BigObject);
71           assertTrue(c1.get(i) instanceof BigObject);
72           assertTrue(c2.get(i) instanceof BigObject);
73           assertTrue(c3.get(i) instanceof BigObject);
74        }
75  
76        log.info("Before stopping a cache!");
77        fork(new Runnable() {
78           @Override
79           public void run() {
80              log.debug("About to stop " + c3.getAdvancedCache().getRpcManager().getAddress());
81              c3.stop();
82              c3.getCacheManager().stop();
83              log.debug("Cache stopped async!");
84           }
85        });
86  
87        for (int i = 0; i < num; i++) {
88           log.debug("----Running a get on " + i);
89           assertValue(i, c0.get(i));
90           assertValue(i, c1.get(i));
91           assertValue(i, c2.get(i));
92        }
93        log.debug("Before stopping cache managers!");
94        TestingUtil.killCacheManagers(manager(2));
95        log.debug("2 killed");
96        TestingUtil.killCacheManagers(manager(1));
97        log.debug("1 killed");
98        TestingUtil.killCacheManagers(manager(0));
99        log.debug("0 killed");
100    }
101 
102    private void assertValue(int i, Object o) {
103       assertNotNull(o);
104       assertTrue(o instanceof BigObject);
105       assertEquals(o, cache.get(i));
106    }
107 
108    private BigObject createBigObject(int num, String prefix) {
109       BigObject obj = new BigObject();
110       obj.setName("[" + num + "|" + prefix + "|" +  (num*3) + "|" + (num*7) + "]");
111       obj.setValue(generateLargeString());
112       return obj;
113    }
114 
115    private String generateLargeString() {
116       byte[] bytes = new byte[20 * 100];
117       rnd.nextBytes(bytes);
118       return new String(bytes);
119    }
