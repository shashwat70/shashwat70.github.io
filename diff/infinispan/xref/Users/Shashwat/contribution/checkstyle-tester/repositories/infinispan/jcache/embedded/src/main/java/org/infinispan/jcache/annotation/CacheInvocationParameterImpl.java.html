1   package org.infinispan.jcache.annotation;
2   
3   import java.lang.annotation.Annotation;
4   import java.util.Set;
5   
6   import javax.cache.annotation.CacheInvocationParameter;
7   
8   /**
9    * The {@link javax.cache.annotation.CacheInvocationParameter} implementation.
10   *
11   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
12   * @author Galder Zamarre√±o
13   */
14  public class CacheInvocationParameterImpl implements CacheInvocationParameter {
15  
16     private final ParameterMetaData parameterMetaData;
17     private final Object parameterValue;
18  
19     public CacheInvocationParameterImpl(ParameterMetaData parameterMetaData, Object parameterValue) {
20        this.parameterMetaData = parameterMetaData;
21        this.parameterValue = parameterValue;
22     }
23  
24     @Override
25     public Class<?> getRawType() {
26        return parameterMetaData.getRawType();
27     }
28  
29     @Override
30     public Object getValue() {
31        return parameterValue;
32     }
33  
34     @Override
35     public Set<Annotation> getAnnotations() {
36        return parameterMetaData.getAnnotations();
37     }
38  
39     @Override
40     public int getParameterPosition() {
41        return parameterMetaData.getPosition();
42     }
43  
44     @Override
45     public String toString() {
46        return new StringBuilder()
47              .append("CacheInvocationParameterImpl{")
48              .append("parameterMetaData=").append(parameterMetaData)
49              .append(", parameterValue=").append(parameterValue)
50              .append('}')
51              .toString();
52     }
53  }
