1   package org.infinispan.nearcache.jms;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.CacheException;
5   import org.infinispan.commons.marshall.StreamingMarshaller;
6   import org.infinispan.util.logging.Log;
7   import org.infinispan.util.logging.LogFactory;
8   
9   import javax.jms.BytesMessage;
10  import javax.jms.Message;
11  import javax.jms.MessageListener;
12  
13  /**
14   * JMS listener that listens for remote cache events and invalidates cache
15   * contents.
16   *
17   * @author Galder Zamarre√±o
18   * @since 5.1
19   */
20  public class RemoteEventListener implements MessageListener {
21  
22     private static final Log log = LogFactory.getLog(RemoteEventListener.class);
23  
24     final Cache<Object, Object> cache;
25     final StreamingMarshaller marshaller;
26  
27     public RemoteEventListener(Cache<Object, Object> cache,
28                                StreamingMarshaller marshaller) {
29        this.cache = cache;
30        this.marshaller = marshaller;
31     }
32  
33     @Override
34     public void onMessage(Message msg) {
35        try {
36           // Remove the key from the cache
37           Object key = getKey(msg);
38           log.infof("Received invalidation message[%s] for key=%s, remove from cache",
39                     msg.getJMSMessageID(), key);
40           cache.remove(key);
41        } catch (Exception e) {
42           throw new CacheException("Unable to process remote cache event", e);
43        }
44     }
45  
46     private Object getKey(Message message) throws Exception {
47        BytesMessage msg = (BytesMessage) message;
48        // Transform a Hot Rod binary key into the local Java equivalent
49        byte[] keyBytes = new byte[(int) msg.getBodyLength()];
50        msg.readBytes(keyBytes);
51        // Since Hot Rod stores keys in byte[], it needs to be unmarshalled
52        return marshaller.objectFromByteBuffer(keyBytes);
53     }
54  
55  }
