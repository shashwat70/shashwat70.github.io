21  /**
22   * Tests functionality related to getting multiple entries from a HotRod server
23   * in bulk.
24   *
25   * @author <a href="mailto:rtsang@redhat.com">Ray Tsang</a>
26   * @since 5.2
27   */
28  @Test(groups = "functional")
29  public abstract class BaseBulkGetKeysTest extends MultipleCacheManagersTest {
30  
31     protected HotRodServer[] hotrodServers;
32     protected RemoteCacheManager remoteCacheManager;
33     protected RemoteCache<Object, Object> remoteCache;
34  
35     abstract protected int numberOfHotRodServers();
36  
37     abstract protected ConfigurationBuilder clusterConfig();
38  
39     @Override
40     protected void createCacheManagers() throws Throwable {
41        final int numServers = numberOfHotRodServers();
42        hotrodServers = new HotRodServer[numServers];
43  
44        createCluster(hotRodCacheConfiguration(clusterConfig()), numberOfHotRodServers());
45  
46        for (int i = 0; i < numServers; i++) {
47           EmbeddedCacheManager cm = cacheManagers.get(i);
48           hotrodServers[i] = HotRodClientTestingUtil.startHotRodServer(cm);
49        }
50  
51        String servers = HotRodClientTestingUtil.getServersString(hotrodServers);
52  
53        remoteCacheManager = new RemoteCacheManager(servers);
54        remoteCache = remoteCacheManager.getCache();
55     }
56  
57     @AfterClass(alwaysRun = true)
58     public void release() {
59        killRemoteCacheManager(remoteCacheManager);
60        killServers(hotrodServers);
61     }
62  
63     protected void populateCacheManager() {
64        for (int i = 0; i < 100; i++) {
65           remoteCache.put(i, i);
66        }
67     }
68  
69     public void testBulkGetKeys() {
70        populateCacheManager();
71        Set<Object> set = remoteCache.keySet();
72        assert set.size() == 100;
73        for (int i = 0; i < 100; i++) {
74           assert set.contains(i);
75        }
76     }
77  
78     public void testBulkGetAfterLifespanExpire() throws InterruptedException {
79        Map<String, String> dataIn = new HashMap<String, String>();
80        dataIn.put("aKey", "aValue");
81        dataIn.put("bKey", "bValue");
82        final long startTime = System.currentTimeMillis();
83        final long lifespan = 10000;
84        remoteCache.putAll(dataIn, lifespan, TimeUnit.MILLISECONDS);
85  
86        Set<Object> dataOut = new HashSet<Object>();
87        while (true) {
88           dataOut = remoteCache.keySet();
89           if (System.currentTimeMillis() >= startTime + lifespan)
90              break;
91           assert dataOut.size() == dataIn.size() : String
92                 .format("Data size not the same, put in %s elements, keySet has %s elements",
93                       dataIn.size(), dataOut.size());
94           for (Object outKey : dataOut) {
95              assert dataIn.containsKey(outKey);
96           }
97           Thread.sleep(100);
98        }
99  
100       // Make sure that in the next 30 secs data is removed
101       while (System.currentTimeMillis() < startTime + lifespan + 30000) {
102          dataOut = remoteCache.keySet();
103          if (dataOut.size() == 0)
104             return;
105       }
106 
107       assert dataOut.size() == 0 : String.format(
108             "Data not empty, it contains: %s elements", dataOut.size());
109    }
