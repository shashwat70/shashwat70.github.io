1   package org.infinispan.commons.hash;
2   
3   import net.jcip.annotations.Immutable;
4   import net.jcip.annotations.ThreadSafe;
5   
6   import org.infinispan.commons.marshall.exts.NoStateExternalizer;
7   import org.infinispan.commons.util.Util;
8   import org.infinispan.commons.marshall.Ids;
9   
10  import java.io.ObjectInput;
11  import java.util.Set;
12  
13  /**
14   * An implementation of Austin Appleby's MurmurHash2.0 algorithm, as documented on <a href="http://sites.google.com/site/murmurhash/">his website</a>.
15   * <p />
16   * This implementation is based on the slower, endian-neutral version of the algorithm as documented on the site,
17   * ported from Austin Appleby's original C++ version <a href="http://sites.google.com/site/murmurhash/MurmurHashNeutral2.cpp">MurmurHashNeutral2.cpp</a>.
18   * <p />
19   * Other implementations are documented on Wikipedia's <a href="http://en.wikipedia.org/wiki/MurmurHash">MurmurHash</a> page.
20   * <p />
21   * <b>Note</b>: This is the backward compatible version of this hash.  The correct version of this hash is implemented as
22   * {@link MurmurHash2}.  This version contains a slight bug in that it only takes into account 31 bits of the 32 bit range,
23   * which will result in poorer distribution.  It is still maintained in the source tree to provide backward compatibility
24   * with existing clusters prior to 4.2.1, when the bug was identified and fixed.  See <a href="https://issues.jboss.org/browse/ISPN-873">ISPN-873</a> for more details.
25   *
26   * @see <a href="http://sites.google.com/site/murmurhash/">MurmurHash website</a>
27   * @see <a href="http://en.wikipedia.org/wiki/MurmurHash">MurmurHash entry on Wikipedia</a>
28   * @see MurmurHash2
29   * @author Manik Surtani
30   * @version 4.1
31   */
32  @ThreadSafe
33  @Immutable
34  public class MurmurHash2Compat implements Hash {
35     private static final int M = 0x5bd1e995;
36     private static final int R = 24;
37     private static final int H = -1;
38  
39     @Override
40     public final int hash(byte[] payload) {
41        int h = H;
42        int len = payload.length;
43        int offset = 0;
44        while (len >= 4) {
45           int k = payload[offset];
46           k |= payload[offset + 1] << 8;
47           k |= payload[offset + 2] << 16;
48           k |= payload[offset + 3] << 24;
49  
50           k *= M;
51           k ^= k >> R;
52           k *= M;
53           h *= M;
54           h ^= k;
55  
56           len -= 4;
57           offset += 4;
58        }
59  
60        switch (len) {
61           case 3:
62              h ^= payload[offset + 2] << 16;
63           case 2:
64              h ^= payload[offset + 1] << 8;
65           case 1:
66              h ^= payload[offset];
67              h *= M;
68        }
69  
70        h ^= h >> 13;
71        h *= M;
72        h ^= h >> 15;
73  
74        return h;
75     }
76  
77     @Override
78     public final int hash(int hashcode) {
79        byte[] b = new byte[4];
80        b[0] = (byte) hashcode;
81        b[1] = (byte) (hashcode >> 8);
82        b[2] = (byte) (hashcode >> 16);
83        b[3] = (byte) (hashcode >> 24);
84        return hash(b);
85     }
86  
87     @Override
88     public final int hash(Object o) {
89        if (o instanceof byte[])
90           return hash((byte[]) o);
91        else if (o instanceof String)
92           return hash(((String) o).getBytes());
93        else
94           return hash(o.hashCode());
95     }
96  
97     @Override
98     public boolean equals(Object other) {
99        return other != null && other.getClass() == getClass();
100    }
101 
102    @Override
103    public int hashCode() {
104       return 0;
105    }
106 
107    @Override
108    public String toString() {
109       return "MurmurHash2Compat";
110    }
111 
112    public static class Externalizer extends NoStateExternalizer<MurmurHash2Compat> {
113       @Override
114       @SuppressWarnings("unchecked")
115       public Set<Class<? extends MurmurHash2Compat>> getTypeClasses() {
116          return Util.<Class<? extends MurmurHash2Compat>>asSet(MurmurHash2Compat.class);
117       }
118 
119       @Override
120       public MurmurHash2Compat readObject(ObjectInput input) {
121          return new MurmurHash2Compat();
122       }
123 
124       @Override
125       public Integer getId() {
126          return Ids.MURMURHASH_2_COMPAT;
127       }
128    }
129 }
