1   package org.infinispan.persistence.remote.configuration;
2   
3   import org.infinispan.commons.configuration.Builder;
4   import org.infinispan.configuration.global.GlobalConfiguration;
5   
6   /**
7    *
8    * ConnectionPoolConfigurationBuilder. Specified connection pooling properties for the HotRod client
9    *
10   * @author Tristan Tarrant
11   * @since 5.2
12   */
13  public class ConnectionPoolConfigurationBuilder extends AbstractRemoteStoreConfigurationChildBuilder<RemoteStoreConfigurationBuilder> implements
14        Builder<ConnectionPoolConfiguration> {
15     private ExhaustedAction exhaustedAction = ExhaustedAction.WAIT;
16     private int maxActive = -1;
17     private int maxTotal = -1;
18     private int maxIdle = -1;
19     private int minIdle = 1;
20     private long timeBetweenEvictionRuns = 120000;
21     private long minEvictableIdleTime = 1800000;
22     private boolean testWhileIdle = true;
23  
24     ConnectionPoolConfigurationBuilder(RemoteStoreConfigurationBuilder builder) {
25        super(builder);
26     }
27  
28     /**
29      * Specifies what happens when asking for a connection from a server's pool, and that pool is
30      * exhausted.
31      */
32     public ConnectionPoolConfigurationBuilder exhaustedAction(ExhaustedAction exhaustedAction) {
33        this.exhaustedAction = exhaustedAction;
34        return this;
35     }
36  
37     /**
38      * Controls the maximum number of connections per server that are allocated (checked out to
39      * client threads, or idle in the pool) at one time. When non-positive, there is no limit to the
40      * number of connections per server. When maxActive is reached, the connection pool for that
41      * server is said to be exhausted. The default setting for this parameter is -1, i.e. there is no
42      * limit.
43      */
44     public ConnectionPoolConfigurationBuilder maxActive(int maxActive) {
45        this.maxActive = maxActive;
46        return this;
47     }
48  
49     /**
50      * Sets a global limit on the number persistent connections that can be in circulation within the
51      * combined set of servers. When non-positive, there is no limit to the total number of
52      * persistent connections in circulation. When maxTotal is exceeded, all connections pools are
53      * exhausted. The default setting for this parameter is -1 (no limit).
54      */
55     public ConnectionPoolConfigurationBuilder maxTotal(int maxTotal) {
56        this.maxTotal = maxTotal;
57        return this;
58     }
59  
60     /**
61      * Controls the maximum number of idle persistent connections, per server, at any time. When
62      * negative, there is no limit to the number of connections that may be idle per server. The
63      * default setting for this parameter is -1.
64      */
65     public ConnectionPoolConfigurationBuilder maxIdle(int maxIdle) {
66        this.maxIdle = maxIdle;
67        return this;
68     }
69  
70     /**
71      * Sets a target value for the minimum number of idle connections (per server) that should always
72      * be available. If this parameter is set to a positive number and timeBetweenEvictionRunsMillis
73      * > 0, each time the idle connection eviction thread runs, it will try to create enough idle
74      * instances so that there will be minIdle idle instances available for each server. The default
75      * setting for this parameter is 1.
76      */
77     public ConnectionPoolConfigurationBuilder minIdle(int minIdle) {
78        this.minIdle = minIdle;
79        return this;
80     }
81  
82     /**
83      * Indicates how long the eviction thread should sleep before "runs" of examining idle
84      * connections. When non-positive, no eviction thread will be launched. The default setting for
85      * this parameter is 2 minutes.
86      */
87     public ConnectionPoolConfigurationBuilder timeBetweenEvictionRuns(long timeBetweenEvictionRuns) {
88        this.timeBetweenEvictionRuns = timeBetweenEvictionRuns;
89        return this;
90     }
91  
92     /**
93      * Specifies the minimum amount of time that an connection may sit idle in the pool before it is
94      * eligible for eviction due to idle time. When non-positive, no connection will be dropped from
95      * the pool due to idle time alone. This setting has no effect unless
96      * timeBetweenEvictionRunsMillis > 0. The default setting for this parameter is 1800000(30
97      * minutes).
98      */
99     public ConnectionPoolConfigurationBuilder minEvictableIdleTime(long minEvictableIdleTime) {
100       this.minEvictableIdleTime = minEvictableIdleTime;
101       return this;
102    }
103 
104    /**
105     * Indicates whether or not idle connections should be validated by sending an TCP packet to the
106     * server, during idle connection eviction runs. Connections that fail to validate will be
107     * dropped from the pool. This setting has no effect unless timeBetweenEvictionRunsMillis > 0.
108     * The default setting for this parameter is true.
109     */
110    public ConnectionPoolConfigurationBuilder testWhileIdle(boolean testWhileIdle) {
111       this.testWhileIdle = testWhileIdle;
112       return this;
113    }
114 
115    @Override
116    public void validate() {
117    }
118 
119    @Override
120    public void validate(GlobalConfiguration globalConfig) {
121    }
122 
123    @Override
124    public ConnectionPoolConfiguration create() {
125       return new ConnectionPoolConfiguration(exhaustedAction, maxActive, maxTotal, maxIdle, minIdle, timeBetweenEvictionRuns, minEvictableIdleTime, testWhileIdle);
126    }
127 
128    @Override
129    public ConnectionPoolConfigurationBuilder read(ConnectionPoolConfiguration template) {
130       exhaustedAction = template.exhaustedAction();
131       maxActive = template.maxActive();
132       maxTotal = template.maxTotal();
133       maxIdle = template.maxIdle();
134       minIdle = template.minIdle();
135       timeBetweenEvictionRuns = template.timeBetweenEvictionRuns();
136       minEvictableIdleTime = template.minEvictableIdleTime();
137       testWhileIdle = template.testWhileIdle();
138       return this;
139    }
140 }
