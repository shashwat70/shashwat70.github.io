22  package org.jboss.as.clustering.jgroups;
23  
24  import java.io.IOException;
25  import java.net.DatagramSocket;
26  import java.net.InetAddress;
27  import java.net.MulticastSocket;
28  import java.net.ServerSocket;
29  import java.net.Socket;
30  import java.net.SocketAddress;
31  import java.net.SocketException;
32  import java.util.Map;
33  
34  import org.jboss.as.network.SocketBindingManager;
35  import org.jgroups.util.SocketFactory;
36  
37  /**
38   * Manages registration of all JGroups sockets with a {@link SocketBindingManager}.
39   * @author Paul Ferraro
40   */
41  public class ManagedSocketFactory implements SocketFactory {
42  
43      private final SocketFactory factory;
44      private final SocketBindingManager manager;
45  
46      public ManagedSocketFactory(SocketFactory factory, SocketBindingManager manager) {
47          this.factory = factory;
48          this.manager = manager;
49      }
50  
51      @Override
52      public Socket createSocket(String name) throws IOException {
53          return this.register(this.factory.createSocket(name));
54      }
55  
56      @Override
57      public Socket createSocket(String name, String host, int port) throws IOException {
58          return this.register(this.factory.createSocket(name, host, port));
59      }
60  
61      @Override
62      public Socket createSocket(String name, InetAddress address, int port) throws IOException {
63          return this.register(this.factory.createSocket(name, address, port));
64      }
65  
66      @Override
67      public Socket createSocket(String name, String host, int port, InetAddress localAddress, int localPort) throws IOException {
68          return this.register(this.factory.createSocket(name, host, port, localAddress, localPort));
69      }
70  
71      @Override
72      public Socket createSocket(String name, InetAddress address, int port, InetAddress localAddress, int localPort) throws IOException {
73          return this.register(this.factory.createSocket(name, address, port, localAddress, localPort));
74      }
75  
76      private Socket register(final Socket socket) {
77          this.manager.getUnnamedRegistry().registerSocket(socket);
78          return socket;
79      }
80  
81      @Override
82      public ServerSocket createServerSocket(String name) throws IOException {
83          return this.register(this.factory.createServerSocket(name));
84      }
85  
86      @Override
87      public ServerSocket createServerSocket(String name, int port) throws IOException {
88          return this.register(this.factory.createServerSocket(name, port));
89      }
90  
91      @Override
92      public ServerSocket createServerSocket(String name, int port, int backlog) throws IOException {
93          return this.register(this.factory.createServerSocket(name, port, backlog));
94      }
95  
96      @Override
97      public ServerSocket createServerSocket(String name, int port, int backlog, InetAddress bindAddr) throws IOException {
98          return this.register(this.factory.createServerSocket(name, port, backlog, bindAddr));
99      }
100 
101     private ServerSocket register(final ServerSocket socket) {
102         this.manager.getUnnamedRegistry().registerSocket(socket);
103         return socket;
104     }
105 
106     @Override
107     public DatagramSocket createDatagramSocket(String name) throws SocketException {
108         return this.register(this.factory.createDatagramSocket(name));
109     }
110 
111     @Override
112     public DatagramSocket createDatagramSocket(String name, SocketAddress bindAddress) throws SocketException {
113         return this.register(this.factory.createDatagramSocket(name, bindAddress));
114     }
115 
116     @Override
117     public DatagramSocket createDatagramSocket(String name, int port) throws SocketException {
118         return this.register(this.factory.createDatagramSocket(name, port));
119     }
120 
121     @Override
122     public DatagramSocket createDatagramSocket(String name, int port, InetAddress localAddress) throws SocketException {
123         return this.register(this.factory.createDatagramSocket(name, port, localAddress));
124     }
125 
126     private DatagramSocket register(final DatagramSocket socket) {
127         this.manager.getUnnamedRegistry().registerSocket(socket);
128         return socket;
129     }
130 
131     @Override
132     public MulticastSocket createMulticastSocket(String name) throws IOException {
133         return this.register(this.factory.createMulticastSocket(name));
134     }
135 
136     @Override
137     public MulticastSocket createMulticastSocket(String name, int port) throws IOException {
138         return this.register(this.factory.createMulticastSocket(name, port));
139     }
140 
141     @Override
142     public MulticastSocket createMulticastSocket(String name, SocketAddress bindAddress) throws IOException {
143         return this.register(this.factory.createMulticastSocket(name, bindAddress));
144     }
145 
146     private MulticastSocket register(final MulticastSocket socket) {
147         this.manager.getUnnamedRegistry().registerSocket(socket);
148         return socket;
149     }
150 
151     @Override
152     public void close(Socket socket) throws IOException {
153         if (socket != null) {
154             this.manager.getUnnamedRegistry().unregisterSocket(socket);
155             this.factory.close(socket);
156         }
157     }
158 
159     @Override
160     public void close(ServerSocket socket) throws IOException {
161         if (socket != null) {
162             this.manager.getUnnamedRegistry().unregisterSocket(socket);
163             this.factory.close(socket);
164         }
165     }
166 
167     @Override
168     public void close(DatagramSocket socket) {
169         if (socket != null) {
170             this.manager.getUnnamedRegistry().unregisterSocket(socket);
171             this.factory.close(socket);
172         }
173     }
174 
175     @Override
176     public Map<Object, String> getSockets() {
177         return this.factory.getSockets();
178     }
179 }
