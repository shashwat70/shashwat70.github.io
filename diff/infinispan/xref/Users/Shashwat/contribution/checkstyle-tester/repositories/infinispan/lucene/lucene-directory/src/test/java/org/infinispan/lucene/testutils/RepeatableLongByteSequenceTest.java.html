1   package org.infinispan.lucene.testutils;
2   
3   import org.testng.annotations.Test;
4   
5   /**
6    * Test for {@link RepeatableLongByteSequence}
7    *
8    * @author Dan Berindei
9    * @since 5.2
10   */
11  @Test(groups = "unit", testName = "lucene.testutils.RepeatableLongByteSequenceTest")
12  public class RepeatableLongByteSequenceTest {
13     @Test(description="To verify the RepeatableLongByteSequence meets the requirement of producing "
14        + "always the same values when using the single nextByte()")
15     public void verifyRepeatability() {
16        RepeatableLongByteSequence src1 = new RepeatableLongByteSequence();
17        RepeatableLongByteSequence src2 = new RepeatableLongByteSequence();
18        for (int i = 0; i < 1000; i++) {
19           assert src1.nextByte() == src2.nextByte();
20        }
21     }
22  
23     @Test(description="To verify the RepeatableLongByteSequence meets the requirement of producing "
24           + "always the same values when using the multivalued nextBytes()")
25     public void verifyEquality() {
26        RepeatableLongByteSequence src1 = new RepeatableLongByteSequence();
27        RepeatableLongByteSequence src2 = new RepeatableLongByteSequence();
28        final int arrayLength = 10;
29        byte[] b = new byte[arrayLength];
30        for (int i = 0; i < 1000; i++) {
31           if((i % arrayLength) == 0) {
32              src1.nextBytes(b);
33           }
34           assert b[i % arrayLength] == src2.nextByte();
35        }
36     }
37  }
