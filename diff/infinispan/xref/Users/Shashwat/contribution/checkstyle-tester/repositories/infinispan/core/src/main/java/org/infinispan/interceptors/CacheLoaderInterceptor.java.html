110    @Override
111    public Object visitGetAllCommand(InvocationContext ctx, GetAllCommand command) throws Throwable {
112       if (enabled) {
113          for (Object key : command.getKeys()) {
114             loadIfNeeded(ctx, key, command);
115          }
116       }
117       return invokeNextInterceptor(ctx, command);
118    }

120    @Override
121    public Object visitInvalidateCommand(InvocationContext ctx, InvalidateCommand command) throws Throwable {
122       if (enabled) {
123          Object[] keys;
124          if ((keys = command.getKeys()) != null && keys.length > 0) {
125             for (Object key : command.getKeys()) {
126                loadIfNeeded(ctx, key, command);
127             }
128          }
129       }
130       return invokeNextInterceptor(ctx, command);
131    }

143    private Object visitDataCommand(InvocationContext ctx, AbstractDataCommand command) throws Throwable {
144       if (enabled) {
145          Object key;
146          if ((key = command.getKey()) != null) {
147             loadIfNeeded(ctx, key, command);
148          }
149       }
150       return invokeNextInterceptor(ctx, command);
151    }
152 
153    @Override
154    public Object visitGetKeysInGroupCommand(final InvocationContext ctx, GetKeysInGroupCommand command) throws Throwable {
155       final String groupName = command.getGroupName();
156       if (!command.isGroupOwner() || !enabled || hasSkipLoadFlag(command)) {
157          return invokeNextInterceptor(ctx, command);
158       }
159 
160       final KeyFilter<Object> keyFilter = new CompositeKeyFilter<>(new GroupFilter<>(groupName, groupManager),
161                                                                    new CollectionKeyFilter<>(ctx.getLookedUpEntries().keySet()));
162       persistenceManager.processOnAllStores(keyFilter, new AdvancedCacheLoader.CacheLoaderTask() {
163          @Override
164          public void processEntry(MarshalledEntry marshalledEntry, AdvancedCacheLoader.TaskContext taskContext) throws InterruptedException {
165             synchronized (ctx) {
166                //the process can be made in multiple threads, so we need to synchronize in the context.
167                entryFactory.wrapEntryForReading(ctx, marshalledEntry.getKey(), convert(marshalledEntry, iceFactory)).setSkipLookup(true);
168             }
169          }
170       }, true, true);
171       return invokeNextInterceptor(ctx, command);
172    }

199    /**
200     * Loads from the cache loader the entry for the given key.  A found value is loaded into the current context.  The
201     * method returns whether the value was found or not, or even if the cache loader was checked.
202     * @param ctx The current invocation's context
203     * @param key The key for the entry to look up
204     * @param cmd The command that was called that now wants to query the cache loader
205     * @return Whether or not the entry was found in the cache loader.  A value of null means the cache loader was never
206     *         queried for the value, so it was neither a hit or a miss.
207     * @throws Throwable
208     */
209    protected final Boolean loadIfNeeded(final InvocationContext ctx, Object key, final FlagAffectedCommand cmd) throws Throwable {
210       if (skipLoad(cmd, key, ctx)) {
211          return null;
212       }
213 
214       final boolean isDelta = cmd instanceof ApplyDeltaCommand;
215       final AtomicReference<Boolean> isLoaded = new AtomicReference<>();
216       InternalCacheEntry entry = PersistenceUtil.loadAndStoreInDataContainer(dataContainer, persistenceManager, key,
217                                                                              ctx, timeService, isLoaded);
218       Boolean isLoadedValue = isLoaded.get();
219       if (trace) {
220          log.tracef("Entry was loaded? %s", isLoadedValue);
221       }
222       if (getStatisticsEnabled()) {
223          if (isLoadedValue == null) {
224             // the entry was in data container, we haven't touched cache store
225          } else if (isLoadedValue) {
226             cacheLoads.incrementAndGet();
227          } else {
228             cacheMisses.incrementAndGet();
229          }
230       }
231 
232       if (entry != null) {
233          CacheEntry wrappedEntry = wrapInternalCacheEntry(ctx, key, cmd, entry, isDelta);
234          if (isLoadedValue != null && isLoadedValue.booleanValue() && wrappedEntry != null) {
235             recordLoadedEntry(ctx, key, wrappedEntry, entry, cmd);
236          }
237       }
238       return isLoadedValue;
239    }

241    private boolean skipLoad(FlagAffectedCommand cmd, Object key, InvocationContext ctx) {
242       if (!shouldAttemptLookup(ctx.lookupEntry(key))) {
243          if (trace) {
244             log.tracef("Skip load for command %s. Entry already exists in context.", cmd);
245          }
246          return true;
247       }
248 
249       if (!canLoad(key)) {
250          if (trace) {
251             log.tracef("Skip load for command %s. Cannot load the key.", cmd);
252          }
253          return true;
254       }
255 
256       boolean skip;
257       if (cmd instanceof WriteCommand) {
258          skip = skipLoadForWriteCommand((WriteCommand) cmd, key, ctx);
259          if (trace) {
260             log.tracef("Skip load for write command %s? %s", cmd, skip);
261          }
262       } else {
263          //read command
264          skip = hasSkipLoadFlag(cmd);
265          if (trace) {
266             log.tracef("Skip load for command %s?. %s", cmd, skip);
267          }
268       }
269       return skip;
270    }
271 
272    protected boolean skipLoadForWriteCommand(WriteCommand cmd, Object key, InvocationContext ctx) {
273       if (isDeltaWrite(cmd)) {
274          if (trace) {
275             log.tracef("Don't skip load for command %s. Value is needed!", cmd);
276          }
277          return false;
278       } else if (isConditional(cmd)) {
279          boolean skip = hasSkipLoadFlag(cmd);
280          if (trace) {
281             log.tracef("Skip load for conditional command %s? %s", cmd, skip);
282          }
283          return skip;
284       }
285       return hasSkipLoadFlag(cmd) || hasIgnoreReturnValueFlag(cmd);
286    }

288    private CacheEntry wrapInternalCacheEntry(InvocationContext ctx, Object key, FlagAffectedCommand cmd,
289                                              InternalCacheEntry ice, boolean isDelta) {
290       if (isDelta) {
291          ctx.putLookedUpEntry(key, ice);
292          return entryFactory.wrapEntryForDelta(ctx, key, ((ApplyDeltaCommand) cmd).getDelta());
293       } else {
294          return entryFactory.wrapEntryForPut(ctx, key, ice, false, cmd, false);
295       }
296    }

298    /**
299     * Only perform if context doesn't have a value found (Read Committed) or if we can do a remote
300     * get only if the value is null (Repeatable Read)
301     */
302    private boolean shouldAttemptLookup(CacheEntry e) {
303       return e == null || (e.isNull() || e.getValue() == null) && !e.skipLookup();
304    }

306    /**
307     * This method records a loaded entry, performing the following steps: <ol>
308     * <li>updates the 'entry' reference (an entry in the current thread's InvocationContext) with the contents
309     * of 'loadedEntry' (freshly loaded from the CacheStore) so that the loaded details will be flushed to the
310     * DataContainer when the call returns (in the LockingInterceptor, when locks are released)</li>
311     * <li>notifies listeners</li> </ol>
312     */
313    private void recordLoadedEntry(InvocationContext ctx, Object key,
314                                   CacheEntry entry, InternalCacheEntry loadedEntry, FlagAffectedCommand cmd) {
315       final Object value = loadedEntry.getValue();
316       // FIXME: There's no point to trigger the entryLoaded/Activated event twice.
317       sendNotification(key, value, true, ctx, cmd);
318       entry.setValue(value);
319 
320       Metadata metadata = cmd.getMetadata();
321       Metadata loadedMetadata = loadedEntry.getMetadata();
322       if (metadata != null && loadedMetadata != null)
323          metadata = Metadatas.applyVersion(loadedMetadata, metadata);
324       else if (metadata == null)
325          metadata = loadedMetadata;
326 
327       entry.setMetadata(metadata);
328       // TODO shouldn't we also be setting last used and created timestamps?
329       entry.setValid(true);
330 
331       sendNotification(key, value, false, ctx, cmd);
332    }
333 
334    protected void sendNotification(Object key, Object value, boolean pre,
335          InvocationContext ctx, FlagAffectedCommand cmd) {
336       notifier.notifyCacheEntryLoaded(key, value, pre, ctx, cmd);
337    }
