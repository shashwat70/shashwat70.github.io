1   package org.jboss.as.clustering.infinispan.cs.configuration;
2   
3   import org.infinispan.commons.configuration.BuiltBy;
4   import org.infinispan.configuration.cache.*;
5   
6   import java.util.Properties;
7   
8   /**
9    * Configuration which operates only on class names instead of class objects.
10   *
11   * @author Sebastian Laskawiec
12   * @since 7.2
13   */
14  @BuiltBy(DeployedStoreConfigurationBuilder.class)
15  public class DeployedStoreConfiguration extends AbstractStoreConfiguration {
16  
17     private PersistenceConfigurationBuilder persistenceConfigurationBuilder;
18     private String customStoreClassName;
19  
20     public DeployedStoreConfiguration(boolean purgeOnStartup, boolean fetchPersistentState, boolean ignoreModifications,
21                                       AsyncStoreConfiguration async, SingletonStoreConfiguration singletonStore, boolean preload,
22                                       boolean shared, Properties properties, PersistenceConfigurationBuilder persistenceConfigurationBuilder, String customStoreClassName) {
23        super(purgeOnStartup, fetchPersistentState, ignoreModifications, async, singletonStore, preload, shared, properties);
24        this.persistenceConfigurationBuilder = persistenceConfigurationBuilder;
25        this.customStoreClassName = customStoreClassName;
26     }
27  
28     public PersistenceConfigurationBuilder getPersistenceConfigurationBuilder() {
29        return persistenceConfigurationBuilder;
30     }
31  
32     public void setPersistenceConfigurationBuilder(PersistenceConfigurationBuilder persistenceConfigurationBuilder) {
33        this.persistenceConfigurationBuilder = persistenceConfigurationBuilder;
34     }
35  
36     public String getCustomStoreClassName() {
37        return customStoreClassName;
38     }
39  
40     public void setCustomStoreClassName(String customStoreClassName) {
41        this.customStoreClassName = customStoreClassName;
42     }
43  }
