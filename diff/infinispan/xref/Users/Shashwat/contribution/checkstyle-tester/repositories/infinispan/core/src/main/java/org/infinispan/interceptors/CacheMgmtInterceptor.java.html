97     private Object visitDataReadCommand(InvocationContext ctx, AbstractDataCommand command) throws Throwable {
98        long start = 0;
99        boolean statisticsEnabled = getStatisticsEnabled(command);
100       if (statisticsEnabled)
101          start = timeService.time();
102 
103       Object retval = invokeNextInterceptor(ctx, command);
104 
105       if (statisticsEnabled && ctx.isOriginLocal()) {
106          long intervalMilliseconds = timeService.timeDuration(start, TimeUnit.MILLISECONDS);
107          if (retval == null) {
108             missTimes.add(intervalMilliseconds);
109             misses.increment();
110          } else {
111             hitTimes.add(intervalMilliseconds);
112             hits.increment();
113          }
114       }
115 
116       return retval;
117    }

119    @SuppressWarnings("unchecked")
120    @Override
121    public Object visitGetAllCommand(InvocationContext ctx, GetAllCommand command) throws Throwable {
122       long start = 0;
123       boolean statisticsEnabled = getStatisticsEnabled(command);
124       if (statisticsEnabled)
125          start = timeService.time();
126 
127       Object retval = invokeNextInterceptor(ctx, command);
128 
129       if (statisticsEnabled && ctx.isOriginLocal()) {
130          long intervalMilliseconds = timeService.timeDuration(start, TimeUnit.MILLISECONDS);
131          int requests = command.getKeys().size();
132          int hitCount = ((Map<Object, Object>) retval).size();
133          for (Entry<Object, Object> entry : ((Map<Object, Object>) retval).entrySet()) {
134             if (entry.getValue() != null) {
135                hitCount++;
136             }
137          }
138          
139          int missCount = requests - hitCount;
140          if (hitCount > 0) {
141             hits.add(hitCount);
142             hitTimes.add(intervalMilliseconds * hitCount / requests);
143          }
144          if (missCount > 0) {
145             misses.add(missCount);
146             missTimes.add(intervalMilliseconds * missCount / requests);
147          }
148       }
149 
150       return retval;
151    }

153    @Override
154    public Object visitPutMapCommand(InvocationContext ctx, PutMapCommand command) throws Throwable {
155       long start = 0;
156       boolean statisticsEnabled = getStatisticsEnabled(command);
157       if (statisticsEnabled)
158          start = timeService.time();
159 
160       final Object retval = invokeNextInterceptor(ctx, command);
161 
162       if (statisticsEnabled && ctx.isOriginLocal()) {
163          final long intervalMilliseconds = timeService.timeDuration(start, TimeUnit.MILLISECONDS);
164          final Map<Object, Object> data = command.getMap();
165          if (data != null && !data.isEmpty()) {
166             storeTimes.add(intervalMilliseconds);
167             stores.add(data.size());
168          }
169       }
170 
171       return retval;
172    }

185    private Object updateStoreStatistics(InvocationContext ctx, WriteCommand command) throws Throwable {
186       long start = 0;
187       boolean statisticsEnabled = getStatisticsEnabled(command);
188       if (statisticsEnabled)
189          start = timeService.time();
190 
191       Object retval = invokeNextInterceptor(ctx, command);
192 
193       if (statisticsEnabled && ctx.isOriginLocal() && command.isSuccessful()) {
194          long intervalMilliseconds = timeService.timeDuration(start, TimeUnit.MILLISECONDS);
195          storeTimes.add(intervalMilliseconds);
196          stores.increment();
197       }
198 
199       return retval;
200    }
201 
202    @Override
203    public Object visitRemoveCommand(InvocationContext ctx, RemoveCommand command) throws Throwable {
204       long start = 0;
205       boolean statisticsEnabled = getStatisticsEnabled(command);
206       if (statisticsEnabled)
207          start = timeService.time();
208 
209       Object retval = invokeNextInterceptor(ctx, command);
210 
211       if (statisticsEnabled && ctx.isOriginLocal()) {
212          if (command.isConditional()) {
213             if (command.isSuccessful())
214                increaseRemoveHits(start);
215             else
216                increaseRemoveMisses();
217          } else {
218             if (retval == null)
219                increaseRemoveMisses();
220             else
221                increaseRemoveHits(start);
222          }
223       }
224 
225       return retval;
226    }
227 
228    private void increaseRemoveHits(long start) {
229       long intervalMilliseconds = timeService.timeDuration(start, TimeUnit.MILLISECONDS);
230       removeTimes.add(intervalMilliseconds);
231       removeHits.increment();
232    }
233 
234    private void increaseRemoveMisses() {
235       removeMisses.increment();
236    }

297    @ManagedAttribute(
298          description = "Percentage hit/(hit+miss) ratio for the cache",
299          displayName = "Hit ratio",
300          units = Units.PERCENTAGE,
301          displayType = DisplayType.SUMMARY
302    )
303    @SuppressWarnings("unused")
304    public double getHitRatio() {
305       long hitsL = hits.sum();
306       double total = hitsL + misses.sum();
307       // The reason for <= is that equality checks
308       // should be avoided for floating point numbers.
309       if (total <= 0)
310          return 0;
311       return (hitsL / total);
312    }
313 
314    @ManagedAttribute(
315          description = "read/writes ratio for the cache",
316          displayName = "Read/write ratio",
317          units = Units.PERCENTAGE,
318          displayType = DisplayType.SUMMARY
319    )
320    @SuppressWarnings("unused")
321    public double getReadWriteRatio() {
322       long sum = stores.sum();
323       if (sum == 0)
324          return 0;
325       return (((double) (hits.sum() + misses.sum()) / (double) sum));
326    }

328    @ManagedAttribute(
329          description = "Average number of milliseconds for a read operation on the cache",
330          displayName = "Average read time",
331          units = Units.MILLISECONDS,
332          displayType = DisplayType.SUMMARY
333    )
334    @SuppressWarnings("unused")
335    public long getAverageReadTime() {
336       long total = hits.sum() + misses.sum();
337       if (total == 0)
338          return 0;
339       return (hitTimes.sum() + missTimes.sum()) / total;
340    }

342    @ManagedAttribute(
343          description = "Average number of milliseconds for a write operation in the cache",
344          displayName = "Average write time",
345          units = Units.MILLISECONDS,
346          displayType = DisplayType.SUMMARY
347    )
348    @SuppressWarnings("unused")
349    public long getAverageWriteTime() {
350       long sum = stores.sum();
351       if (sum == 0)
352          return 0;
353       return (storeTimes.sum()) / sum;
354    }

357          description = "Average number of milliseconds for a remove operation in the cache",
358          displayName = "Average remove time",
359          units = Units.MILLISECONDS,
360          displayType = DisplayType.SUMMARY
361    )
362    @SuppressWarnings("unused")
363    public long getAverageRemoveTime() {
364       long removes = getRemoveHits();
365       if (removes == 0)
366          return 0;
367       return (removeTimes.sum()) / removes;
368    }

420    @Override
421    @ManagedOperation(
422          description = "Resets statistics gathered by this component",
423          displayName = "Reset Statistics (Statistics)"
424    )
425    public void resetStatistics() {
426       hits.reset();
427       misses.reset();
428       stores.reset();
429       evictions.reset();
430       hitTimes.reset();
431       missTimes.reset();
432       storeTimes.reset();
433       removeHits.reset();
434       removeTimes.reset();
435       removeMisses.reset();
436       resetNanoseconds.set(timeService.time());
437    }

439    private boolean getStatisticsEnabled(FlagAffectedCommand cmd) {
440       return super.getStatisticsEnabled() && !cmd.hasFlag(Flag.SKIP_STATISTICS);
