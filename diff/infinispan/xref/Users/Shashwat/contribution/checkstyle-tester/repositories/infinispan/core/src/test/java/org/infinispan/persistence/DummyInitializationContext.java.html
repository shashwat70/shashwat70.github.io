1   package org.infinispan.persistence;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.io.ByteBufferFactory;
5   import org.infinispan.commons.marshall.StreamingMarshaller;
6   import org.infinispan.configuration.cache.StoreConfiguration;
7   import org.infinispan.marshall.core.MarshalledEntryFactory;
8   import org.infinispan.persistence.spi.InitializationContext;
9   import org.infinispan.util.TimeService;
10  
11  /**
12   * @author Mircea Markus
13   * @since 6.0
14   */
15  public class DummyInitializationContext implements InitializationContext {
16     StoreConfiguration clc;
17     Cache cache;
18     StreamingMarshaller marshaller;
19  
20     ByteBufferFactory byteBufferFactory;
21     MarshalledEntryFactory marshalledEntryFactory;
22  
23     public DummyInitializationContext() {
24     }
25  
26     public DummyInitializationContext(StoreConfiguration clc, Cache cache, StreamingMarshaller marshaller,
27                                       ByteBufferFactory byteBufferFactory, MarshalledEntryFactory marshalledEntryFactory) {
28        this.clc = clc;
29        this.cache = cache;
30        this.marshaller = marshaller;
31        this.byteBufferFactory = byteBufferFactory;
32        this.marshalledEntryFactory = marshalledEntryFactory;
33     }
34  
35     @Override
36     public StoreConfiguration getConfiguration() {
37        return clc;
38     }
39  
40     @Override
41     public Cache getCache() {
42        return cache;
43     }
44  
45     @Override
46     public StreamingMarshaller getMarshaller() {
47        return marshaller;
48     }
49  
50     @Override
51     public TimeService getTimeService() {
52        return cache.getAdvancedCache().getComponentRegistry().getTimeService();
53     }
54  
55     @Override
56     public ByteBufferFactory getByteBufferFactory() {
57        return byteBufferFactory;
58     }
59  
60     @Override
61     public MarshalledEntryFactory getMarshalledEntryFactory() {
62        return marshalledEntryFactory;
63     }
64  }
