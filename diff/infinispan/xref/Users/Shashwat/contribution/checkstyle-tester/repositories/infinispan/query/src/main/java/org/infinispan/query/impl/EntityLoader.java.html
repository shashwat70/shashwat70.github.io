1   package org.infinispan.query.impl;
2   
3   import java.util.ArrayList;
4   import java.util.LinkedHashSet;
5   import java.util.List;
6   import java.util.Map;
7   import java.util.Set;
8   
9   import org.hibernate.search.query.engine.spi.EntityInfo;
10  import org.infinispan.AdvancedCache;
11  import org.infinispan.query.backend.KeyTransformationHandler;
12  
13  /**
14   * @author Sanne Grinovero <sanne@hibernate.org> (C) 2011 Red Hat Inc.
15   * @author Marko Luksa
16   * @since 5.0
17   */
18  public class EntityLoader implements QueryResultLoader {
19  
20     private final AdvancedCache<?, ?> cache;
21     private final KeyTransformationHandler keyTransformationHandler;
22  
23     public EntityLoader(AdvancedCache<?, ?> cache, KeyTransformationHandler keyTransformationHandler) {
24        this.keyTransformationHandler = keyTransformationHandler;
25        this.cache = cache;
26     }
27  
28     private Object decodeKey(EntityInfo entityInfo) {
29        return keyTransformationHandler.stringToKey(entityInfo.getId().toString(), cache.getClassLoader());
30     }
31  
32     public Object load(EntityInfo entityInfo) {
33        return cache.get(decodeKey(entityInfo));
34     }
35  
36     public List<Object> load(List<EntityInfo> entityInfos) {
37        int entitiesSize = entityInfos.size();
38        Set<Object> keys = new LinkedHashSet<>(entitiesSize);
39        for (EntityInfo e : entityInfos) {
40           keys.add(decodeKey(e));
41        }
42        Map<?, ?> entries = cache.getAll(keys);
43        return new ArrayList<>(entries.values());
44     }
45  
46  }
