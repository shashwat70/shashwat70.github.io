478    class NonIndexedClass implements Serializable {
479       public String description;
480 
481       NonIndexedClass(String description) {
482          this.description = description;
483       }
484 
485       @Override
486       public boolean equals(Object o) {
487          if (this == o) return true;
488          if (o == null || getClass() != o.getClass()) return false;
489 
490          NonIndexedClass that = (NonIndexedClass) o;
491 
492          if (!description.equals(that.description)) return false;
493 
494          return true;
495       }
496 
497       @Override
498       public int hashCode() {
499          return description.hashCode();
500       }
501    }

503    class SimpleInterceptor extends CommandInterceptor {
504       private boolean putOkay;
505 
506       @Override
507       public Object visitPutKeyValueCommand(InvocationContext ctx, PutKeyValueCommand command) throws Throwable {
508          if (isRightType(ctx)) putOkay = true;
509          return super.visitPutKeyValueCommand(ctx, command);
510       }
511 
512       private boolean isRightType(InvocationContext ctx) {
513          return ctx instanceof SingleKeyNonTxInvocationContext;
514       }
515    }
