1   package org.infinispan.jcache.annotation;
2   
3   import java.io.Serializable;
4   
5   import javax.cache.Cache;
6   import javax.cache.annotation.CacheKeyGenerator;
7   import javax.cache.annotation.CacheKeyInvocationContext;
8   import javax.cache.annotation.CacheResolver;
9   import javax.cache.annotation.CacheResult;
10  import javax.cache.annotation.GeneratedCacheKey;
11  import javax.interceptor.InvocationContext;
12  
13  import org.infinispan.jcache.logging.Log;
14  
15  /**
16   * <p>{@link javax.cache.annotation.CacheResult} interceptor implementation. This interceptor uses the following algorithm describes in
17   * JSR-107.</p>
18   *
19   * <p>When a method annotated with {@link javax.cache.annotation.CacheResult} is invoked the following must occur.
20   * <ol>
21   *    <li>Generate a key based on InvocationContext using the specified {@linkplain javax.cache.annotation.CacheKeyGenerator}.</li>
22   *    <li>Use this key to look up the entry in the cache.</li>
23   *    <li>If an entry is found return it as the result and do not call the annotated method.</li>
24   *    <li>If no entry is found invoke the method.</li>
25   *    <li>Use the result to populate the cache with this key/result pair.</li>
26   * </ol>
27   *
28   * There is a skipGet attribute which if set to true will cause the method body to always be invoked and the return
29   * value put into the cache. The cache is not checked for the key before method body invocation, skipping steps 2 and 3
30   * from the list above. This can be used for annotating methods that do a cache.put() with no other consequences.</p>
31   *
32   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
33   * @author Galder Zamarre√±o
34   */
35  public abstract class AbstractCacheResultInterceptor implements Serializable {
36  
37     private static final long serialVersionUID = 5275055951121834315L;
38  
39     private final CacheResolver cacheResolver;
40     private final CacheKeyInvocationContextFactory contextFactory;
41  
42     public AbstractCacheResultInterceptor(CacheResolver cacheResolver, CacheKeyInvocationContextFactory contextFactory) {
43        this.cacheResolver = cacheResolver;
44        this.contextFactory = contextFactory;
45     }
46  
47     public Object cacheResult(InvocationContext invocationContext) throws Exception {
48        if (getLog().isTraceEnabled()) {
49           getLog().tracef("Interception of method named '%s'", invocationContext.getMethod().getName());
50        }
51  
52        final CacheKeyInvocationContext<CacheResult> cacheKeyInvocationContext = contextFactory.getCacheKeyInvocationContext(invocationContext);
53        final CacheKeyGenerator cacheKeyGenerator = cacheKeyInvocationContext.unwrap(CacheKeyInvocationContextImpl.class).getCacheKeyGenerator();
54        final CacheResult cacheResult = cacheKeyInvocationContext.getCacheAnnotation();
55        final GeneratedCacheKey cacheKey = cacheKeyGenerator.generateCacheKey(cacheKeyInvocationContext);
56        final Cache<GeneratedCacheKey, Object> cache = cacheResolver.resolveCache(cacheKeyInvocationContext);
57  
58        Object result = null;
59  
60        if (!cacheResult.skipGet()) {
61           result = cache.get(cacheKey);
62           if (getLog().isTraceEnabled()) {
63              getLog().tracef("Entry with value '%s' has been found in cache '%s' with key '%s'", result, cache.getName(), cacheKey);
64           }
65        }
66  
67        if (result == null) {
68           result = invocationContext.proceed();
69           if (result != null) {
70              cache.put(cacheKey, result);
71              if (getLog().isTraceEnabled()) {
72                 getLog().tracef("Value '%s' cached in cache '%s' with key '%s'", result, cache.getName(), cacheKey);
73              }
74           }
75        }
76  
77        return result;
78     }
79  
80     protected abstract Log getLog();
81  
82  }
