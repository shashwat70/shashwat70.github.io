1   package org.infinispan.remoting.responses;
2   
3   import org.infinispan.remoting.rpc.ResponseFilter;
4   import org.infinispan.remoting.transport.Address;
5   
6   import java.util.Collection;
7   import java.util.HashSet;
8   import java.util.Set;
9   
10  /**
11   * A filter that only expects responses from an initial set of targets.
12   *
13   * Useful when sending a command to {@code null} to ensure we don't wait for responses from
14   * cluster members that weren't properly started when the command was sent.
15   *
16   * JGroups calls our handler while holding a lock, so we don't need any synchronization.
17   *
18   * @author Dan Berindei &lt;dan@infinispan.org&gt;
19   * @since 5.1
20   */
21  public final class IgnoreExtraResponsesValidityFilter implements ResponseFilter {
22  
23     private final Set<Address> targets;
24     private int missingResponses;
25  
26     public IgnoreExtraResponsesValidityFilter(Collection<Address> targets, Address self, boolean removeSelf) {
27        this.targets = new HashSet<Address>(targets);
28        this.missingResponses = targets.size();
29        if (removeSelf && this.targets.contains(self)) {
30           missingResponses--;
31        }
32     }
33  
34     @Override
35     public boolean isAcceptable(Response response, Address address) {
36        if (targets.contains(address)) {
37           missingResponses--;
38        }
39  
40        // always return true to make sure a response is logged by the JGroups RpcDispatcher.
41        return true;
42     }
43  
44     @Override
45     public boolean needMoreResponses() {
46        return missingResponses > 0;
47     }
48  
49  }
