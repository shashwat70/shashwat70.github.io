1   package org.infinispan.cdi.util.annotatedtypebuilder;
2   
3   import java.lang.annotation.Annotation;
4   import java.util.Map;
5   import java.util.Set;
6   
7   import static java.util.Collections.emptyMap;
8   import static java.util.Collections.emptySet;
9   import static java.util.Collections.unmodifiableSet;
10  
11  /**
12   * A helper class used to hold annotations on a type or member.
13   *
14   * @author Stuart Douglas
15   */
16  class AnnotationStore {
17  
18     private final Map<Class<? extends Annotation>, Annotation> annotationMap;
19     private final Set<Annotation> annotationSet;
20  
21     AnnotationStore(Map<Class<? extends Annotation>, Annotation> annotationMap, Set<Annotation> annotationSet) {
22        this.annotationMap = annotationMap;
23        this.annotationSet = unmodifiableSet(annotationSet);
24     }
25  
26     AnnotationStore() {
27        this.annotationMap = emptyMap();
28        this.annotationSet = emptySet();
29     }
30  
31     <T extends Annotation> T getAnnotation(Class<T> annotationType) {
32        return annotationType.cast(annotationMap.get(annotationType));
33     }
34  
35     Set<Annotation> getAnnotations() {
36        return annotationSet;
37     }
38  
39     boolean isAnnotationPresent(Class<? extends Annotation> annotationType) {
40        return annotationMap.containsKey(annotationType);
41     }
42  
43  }
