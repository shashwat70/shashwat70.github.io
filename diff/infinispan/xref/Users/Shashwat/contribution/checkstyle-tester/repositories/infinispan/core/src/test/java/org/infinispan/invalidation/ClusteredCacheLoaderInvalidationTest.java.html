1   package org.infinispan.invalidation;
2   
3   import org.infinispan.configuration.cache.CacheMode;
4   import org.infinispan.configuration.cache.ClusterLoaderConfigurationBuilder;
5   import org.infinispan.configuration.cache.ConfigurationBuilder;
6   import org.infinispan.test.MultipleCacheManagersTest;
7   import org.testng.annotations.Test;
8   
9   import static org.testng.AssertJUnit.assertEquals;
10  import static org.testng.AssertJUnit.assertFalse;
11  import static org.testng.AssertJUnit.assertNull;
12  
13  /**
14   * Tests to ensure that invalidation caches with a cluster cache loader properly retrieve the value from the remote
15   * cache
16   *
17   * @author William burns
18   * @since 6.0
19   */
20  public class ClusteredCacheLoaderInvalidationTest extends MultipleCacheManagersTest {
21     private static final String key = "key";
22     private static final String value = "value";
23     private static final String changedValue = "changed-value";
24  
25     private static final String cacheName = "inval-write-cache-store";
26  
27     @Override
28     protected void createCacheManagers() throws Throwable {
29        ConfigurationBuilder cb = getDefaultClusteredCacheConfig(CacheMode.INVALIDATION_SYNC, false);
30        cb.persistence().addStore(ClusterLoaderConfigurationBuilder.class);
31        createClusteredCaches(2, cacheName, cb);
32     }
33  
34     @Test
35     public void testCacheLoaderBeforeAfterInvalidation() {
36        assertNull(cache(0, cacheName).get(key));
37  
38        cache(0, cacheName).put(key, value);
39  
40        assertFalse("Invalidation should not contain the value in memory",
41                    cache(1, cacheName).getAdvancedCache().getDataContainer().containsKey(key));
42  
43        assertEquals(value, cache(1, cacheName).get(key));
44  
45        cache(1, cacheName).put(key, changedValue);
46  
47        assertFalse("Invalidation should not contain the value in memory after other node put",
48                    cache(0, cacheName).getAdvancedCache().getDataContainer().containsKey(key));
49  
50        assertEquals(changedValue, cache(0, cacheName).get(key));
51     }
52  }
