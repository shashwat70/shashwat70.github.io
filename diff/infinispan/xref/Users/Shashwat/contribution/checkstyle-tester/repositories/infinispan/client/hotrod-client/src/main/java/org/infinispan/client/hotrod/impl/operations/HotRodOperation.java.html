1   package org.infinispan.client.hotrod.impl.operations;
2   
3   import java.util.concurrent.atomic.AtomicInteger;
4   
5   import net.jcip.annotations.Immutable;
6   
7   import org.infinispan.client.hotrod.Flag;
8   import org.infinispan.client.hotrod.impl.protocol.Codec;
9   import org.infinispan.client.hotrod.impl.protocol.HeaderParams;
10  import org.infinispan.client.hotrod.impl.protocol.HotRodConstants;
11  import org.infinispan.client.hotrod.impl.transport.Transport;
12  
13  /**
14   * Generic Hot Rod operation. It is aware of {@link org.infinispan.client.hotrod.Flag}s and it is targeted against a
15   * cache name. This base class encapsulates the knowledge of writing and reading a header, as described in the
16   * <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod protocol specification</a>
17   *
18   * @author Mircea.Markus@jboss.com
19   * @since 4.1
20   */
21  @Immutable
22  public abstract class HotRodOperation implements HotRodConstants {
23  
24     protected final Flag[] flags;
25  
26     public final byte[] cacheName;
27  
28     protected final AtomicInteger topologyId;
29  
30     protected final Codec codec;
31  
32     private static final byte NO_TX = 0;
33     private static final byte XA_TX = 1;
34  
35     protected HotRodOperation(Codec codec, Flag[] flags, byte[] cacheName, AtomicInteger topologyId) {
36        this.flags = flags;
37        this.cacheName = cacheName;
38        this.topologyId = topologyId;
39        this.codec = codec;
40     }
41  
42     public abstract Object execute();
43  
44     protected final HeaderParams writeHeader(Transport transport, short operationCode) {
45        HeaderParams params = new HeaderParams()
46              .opCode(operationCode).cacheName(cacheName).flags(flags)
47              .clientIntel(CLIENT_INTELLIGENCE_HASH_DISTRIBUTION_AWARE)
48              .topologyId(topologyId).txMarker(NO_TX);
49        return codec.writeHeader(transport, params);
50     }
51  
52     /**
53      * Magic	| Message Id | Op code | Status | Topology Change Marker
54      */
55     protected short readHeaderAndValidate(Transport transport, HeaderParams params) {
56        return codec.readHeader(transport, params);
57     }
58  
59  }
