1   package org.infinispan.invalidation;
2   
3   import org.infinispan.configuration.cache.CacheMode;
4   import org.infinispan.configuration.cache.ConfigurationBuilder;
5   import org.infinispan.persistence.dummy.DummyInMemoryStoreConfigurationBuilder;
6   import org.infinispan.test.MultipleCacheManagersTest;
7   import org.testng.annotations.Test;
8   
9   import static org.testng.AssertJUnit.assertEquals;
10  import static org.testng.AssertJUnit.assertNull;
11  import static org.testng.AssertJUnit.assertFalse;
12  
13  /**
14   * Tests to ensure that invalidation caches operate properly with a shared cache store with various operations.
15   *
16   * @author William Burns
17   * @since 6.0
18   */
19  public class WriteStoreInvalidationTest extends MultipleCacheManagersTest {
20     private static final String key = "key";
21     private static final String value = "value";
22     private static final String changedValue = "changed-value";
23  
24     private static final String cacheName = "inval-write-cache-store";
25  
26     @Override
27     protected void createCacheManagers() throws Throwable {
28        ConfigurationBuilder cb = getDefaultClusteredCacheConfig(CacheMode.INVALIDATION_SYNC, false);
29        cb.persistence().addStore(DummyInMemoryStoreConfigurationBuilder.class).shared(true)
30              .storeName(getClass().getSimpleName());
31        createClusteredCaches(2, cacheName, cb);
32     }
33  
34     @Test
35     public void testSharedCacheStoreAfterInvalidation() {
36        assertNull(cache(0, cacheName).get(key));
37  
38        cache(0, cacheName).put(key, value);
39  
40        assertEquals(value, cache(1, cacheName).get(key));
41  
42        cache(1, cacheName).put(key, changedValue);
43  
44        assertFalse(cache(0, cacheName).getAdvancedCache().getDataContainer().containsKey(key));
45  
46        assertEquals(changedValue, cache(0, cacheName).get(key));
47     }
48  }
