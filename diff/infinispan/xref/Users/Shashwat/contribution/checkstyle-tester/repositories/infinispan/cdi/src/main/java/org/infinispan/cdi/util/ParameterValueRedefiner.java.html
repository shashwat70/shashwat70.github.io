1   package org.infinispan.cdi.util;
2   
3   import javax.enterprise.context.spi.CreationalContext;
4   import javax.enterprise.inject.spi.BeanManager;
5   import javax.enterprise.inject.spi.InjectionPoint;
6   
7   /**
8    * Provides the ability to redefine the value of a parameter on an
9    * {@link InjectableMethod} via the
10   * {@link #redefineParameterValue(ParameterValue)} callback.
11   *
12   * @author Pete Muir
13   * @see InjectableMethod
14   */
15  public interface ParameterValueRedefiner {
16  
17      /**
18       * Provides the default parameter's value, along with metadata about the
19       * parameter to a parameter redefinition.
20       *
21       * @author Pete Muir
22       * @see ParameterValueRedefiner
23       * @see InjectableMethod
24       */
25      public static class ParameterValue {
26  
27          private final int position;
28          private final InjectionPoint injectionPoint;
29          private final BeanManager beanManager;
30  
31          ParameterValue(int position, InjectionPoint injectionPoint, BeanManager beanManager) {
32              this.position = position;
33              this.injectionPoint = injectionPoint;
34              this.beanManager = beanManager;
35          }
36  
37          /**
38           * Get the position of the parameter in the member's parameter list.
39           *
40           * @return the position of the parameter
41           */
42          public int getPosition() {
43              return position;
44          }
45  
46          /**
47           * Get the {@link InjectionPoint} for the parameter.
48           *
49           * @return the injection point
50           */
51          public InjectionPoint getInjectionPoint() {
52              return injectionPoint;
53          }
54  
55          /**
56           * Get the default value of the parameter. The default value is that which
57           * would be injected according to the CDI type safe resolution rules.
58           *
59           * @param creationalContext the creationalContext to use to obtain the
60           *                          injectable reference.
61           * @return the default value
62           */
63          public Object getDefaultValue(CreationalContext<?> creationalContext) {
64              return beanManager.getInjectableReference(injectionPoint, creationalContext);
65          }
66  
67      }
68  
69      /**
70       * Callback allowing the default parameter value (that which would be
71       * injected according to the CDI type safe resolution rules) to be
72       * overridden.
73       *
74       * @param value the default value
75       * @return the overridden value
76       */
77      public Object redefineParameterValue(ParameterValue value);
78  
79  }
