1   package org.infinispan.server.hotrod.configuration;
2   
3   import org.infinispan.commons.configuration.Builder;
4   import org.infinispan.configuration.cache.LockingConfigurationBuilder;
5   import org.infinispan.configuration.cache.StateTransferConfigurationBuilder;
6   import org.infinispan.configuration.cache.SyncConfigurationBuilder;
7   import org.infinispan.server.core.configuration.ProtocolServerConfigurationBuilder;
8   
9   /**
10   * HotRodServerConfigurationBuilder.
11   *
12   * @author Tristan Tarrant
13   * @since 5.3
14   */
15  public class HotRodServerConfigurationBuilder extends ProtocolServerConfigurationBuilder<HotRodServerConfiguration, HotRodServerConfigurationBuilder> implements
16        Builder<HotRodServerConfiguration>, HotRodServerChildConfigurationBuilder {
17     private final AuthenticationConfigurationBuilder authentication = new AuthenticationConfigurationBuilder(this);
18     private String proxyHost;
19     private int proxyPort = -1;
20     private long topologyLockTimeout = 10000L;
21     private long topologyReplTimeout = 10000L;
22     private boolean topologyAwaitInitialTransfer = true;
23     private boolean topologyStateTransfer = true;
24  
25     public HotRodServerConfigurationBuilder() {
26        super(11222);
27     }
28  
29     @Override
30     public HotRodServerConfigurationBuilder self() {
31        return this;
32     }
33  
34     @Override
35     public AuthenticationConfigurationBuilder authentication() {
36        return authentication;
37     }
38  
39     /**
40      * Sets the external address of this node, i.e. the address which clients will connect to
41      */
42     @Override
43     public HotRodServerConfigurationBuilder proxyHost(String proxyHost) {
44        this.proxyHost = proxyHost;
45        return this;
46     }
47  
48     /**
49      * Sets the external port of this node, i.e. the port which clients will connect to
50      */
51     @Override
52     public HotRodServerConfigurationBuilder proxyPort(int proxyPort) {
53        this.proxyPort = proxyPort;
54        return this;
55     }
56  
57     /**
58      * Configures the lock acquisition timeout for the topology cache. See {@link LockingConfigurationBuilder#lockAcquisitionTimeout(long)}. Defaults to 10 seconds
59      */
60     @Override
61     public HotRodServerConfigurationBuilder topologyLockTimeout(long topologyLockTimeout) {
62        this.topologyLockTimeout = topologyLockTimeout;
63        return this;
64     }
65  
66     /**
67      * Configures the replication timeout for the topology cache. See {@link SyncConfigurationBuilder#replTimeout(long)}. Defaults to 10 seconds
68      */
69     @Override
70     public HotRodServerConfigurationBuilder topologyReplTimeout(long topologyReplTimeout) {
71        this.topologyReplTimeout = topologyReplTimeout;
72        return this;
73     }
74  
75     /**
76      * Configures whether to enable waiting for initial state transfer for the topology cache. See {@link StateTransferConfigurationBuilder#awaitInitialTransfer(boolean)}
77      */
78     @Override
79     public HotRodServerConfigurationBuilder topologyAwaitInitialTransfer(boolean topologyAwaitInitialTransfer) {
80        this.topologyAwaitInitialTransfer = topologyAwaitInitialTransfer;
81        return this;
82     }
83  
84     /**
85      * Configures whether to enable state transfer for the topology cache. If disabled, a {@link org.infinispan.persistence.cluster.ClusterLoader} will be used to lazily retrieve topology information from the other nodes.
86      * Defaults to true.
87      */
88     @Override
89     public HotRodServerConfigurationBuilder topologyStateTransfer(boolean topologyStateTransfer) {
90        this.topologyStateTransfer = topologyStateTransfer;
91        return this;
92     }
93  
94     @Override
95     public HotRodServerConfiguration create() {
96        return new HotRodServerConfiguration(defaultCacheName, proxyHost, proxyPort, topologyLockTimeout, topologyReplTimeout, topologyAwaitInitialTransfer, topologyStateTransfer, name, host, port, idleTimeout,
97              recvBufSize, sendBufSize, ssl.create(), tcpNoDelay, workerThreads, authentication.create());
98     }
99  
100    @Override
101    public HotRodServerConfigurationBuilder read(HotRodServerConfiguration template) {
102       super.read(template);
103       this.proxyHost = template.proxyHost();
104       this.proxyPort = template.proxyPort();
105       this.topologyLockTimeout = template.topologyLockTimeout();
106       this.topologyReplTimeout = template.topologyReplTimeout();
107       this.topologyAwaitInitialTransfer = template.topologyAwaitInitialTransfer();
108       this.topologyStateTransfer = template.topologyStateTransfer();
109       return this;
110    }
111 
112    @Override
113    public void validate() {
114       super.validate();
115       if (proxyHost == null) {
116          proxyHost = host;
117       }
118       if (proxyPort < 0) {
119          proxyPort = port;
120       }
121       authentication.validate();
122    }
123 
124    public HotRodServerConfiguration build(boolean validate) {
125       if (validate) {
126          validate();
127       }
128       return create();
129    }
130 
131    @Override
132    public HotRodServerConfiguration build() {
133       return build(true);
134    }
135 
136 }
