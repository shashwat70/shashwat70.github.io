19  package org.jsecurity.util;
20  
21  /**
22   * Internal helper class used to find the Java/JDK version
23   * that JSecurity is operating within, to allow for automatically
24   * adapting to the present platform's capabilities.
25   *
26   * <p>Note that JSecurity does not support 1.2 or earlier JVMs - only 1.3 and later.
27   *
28   * <p><em>This class was borrowed and heavily based upon a nearly identical version found in
29   * the <a href="http://www.springframework.org/">Spring Framework</a>, with minor modifications.
30   * The original author names and copyright (Apache 2.0) has been left in place.  A special
31   * thanks to Rod Johnson, Juergen Hoeller, and Rick Evans for making this available.</em>
32   *
33   * @author Rod Johnson
34   * @author Juergen Hoeller
35   * @author Rick Evans
36   * @author Les Hazlewood
37   * @since 0.2
38   */
39  public abstract class JavaEnvironment {
40  
41      /**
42       * Constant identifying the 1.3.x JVM (JDK 1.3).
43       */
44      public static final int JAVA_13 = 0;
45  
46      /**
47       * Constant identifying the 1.4.x JVM (J2SE 1.4).
48       */
49      public static final int JAVA_14 = 1;
50  
51      /**
52       * Constant identifying the 1.5 JVM (Java 5).
53       */
54      public static final int JAVA_15 = 2;
55  
56      /**
57       * Constant identifying the 1.6 JVM (Java 6).
58       */
59      public static final int JAVA_16 = 3;
60  
61      /**
62       * Constant identifying the 1.7 JVM.
63       */
64      public static final int JAVA_17 = 4;
65  
66  
67      private static final String version;
68  
69      private static final int majorVersion;
70  
71      static {
72          version = System.getProperty("java.version");
73          // version String should look like "1.4.2_10"
74          if (version.indexOf("1.7.") != -1) {
75              majorVersion = JAVA_17;
76          } else if (version.indexOf("1.6.") != -1) {
77              majorVersion = JAVA_16;
78          } else if (version.indexOf("1.5.") != -1) {
79              majorVersion = JAVA_15;
80          } else if (version.indexOf("1.4.") != -1) {
81              majorVersion = JAVA_14;
82          } else {
83              // else leave 1.3 as default (it's either 1.3 or unknown)
84              majorVersion = JAVA_13;
85          }
86      }
87  
88  
89      /**
90       * Return the full Java version string, as returned by
91       * <code>System.getProperty("java.version")</code>.
92       *
93       * @return the full Java version string
94       * @see System#getProperty(String)
95       */
96      public static String getVersion() {
97          return version;
98      }
99  
100     /**
101      * Get the major version code. This means we can do things like
102      * <code>if (getMajorVersion() < JAVA_14)</code>.
103      *
104      * @return a code comparable to the JAVA_XX codes in this class
105      * @see #JAVA_13
106      * @see #JAVA_14
107      * @see #JAVA_15
108      * @see #JAVA_16
109      * @see #JAVA_17
110      */
111     public static int getMajorVersion() {
112         return majorVersion;
113     }
114 
115     /**
116      * Convenience method to determine if the current JVM is at least Java 1.4.
117      *
118      * @return <code>true</code> if the current JVM is at least Java 1.4
119      * @see #getMajorVersion()
120      * @see #JAVA_14
121      * @see #JAVA_15
122      * @see #JAVA_16
123      * @see #JAVA_17
124      */
125     public static boolean isAtLeastVersion14() {
126         return getMajorVersion() >= JAVA_14;
127     }
128 
129     /**
130      * Convenience method to determine if the current JVM is at least
131      * Java 1.5 (Java 5).
132      *
133      * @return <code>true</code> if the current JVM is at least Java 1.5
134      * @see #getMajorVersion()
135      * @see #JAVA_15
136      * @see #JAVA_16
137      * @see #JAVA_17
138      */
139     public static boolean isAtLeastVersion15() {
140         return getMajorVersion() >= JAVA_15;
141     }
142 }
