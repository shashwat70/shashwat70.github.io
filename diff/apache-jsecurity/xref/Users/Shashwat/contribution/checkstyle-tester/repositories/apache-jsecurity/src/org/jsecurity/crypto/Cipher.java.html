19  package org.jsecurity.crypto;
20  
21  /**
22   * A <tt>Cipher</tt> is an algorithm used in cryptography that converts an original input source using a <tt>Key</tt> to
23   * an uninterpretable format.  The resulting encrypted output is only able to be converted back to original form with
24   * a <tt>Key</tt> as well.
25   *
26   * <p>In what is known as <em>Symmetric</em> <tt>Cipher</tt>s, the <tt>Key</tt> used to encrypt the source is the same
27   * as (or trivially similar to) the <tt>Key</tt> used to decrypt it.
28   *
29   * <p>In <em>Assymetric</em> <tt>Cipher</tt>s, the encryption <tt>Key</tt> is not the same as the decryption <tt>Key</tt>.
30   * The most common type of Assymetric Ciphers are based on what is called public/private key pairs:
31   *
32   * <p>A <em>private</em> key is known only to a single party, and as its name implies, is supposed be kept very private
33   * and secure.  A <em>public</em> key that is associated with the private key can be disseminated freely to anyone.
34   * Then data encrypted by the public key can only be decrypted by the private key and vice versa, but neither party
35   * need share their private key with anyone else.  By not sharing a private key, you can guarantee no 3rd party can
36   * intercept the key and therefore use it to decrypt a message.
37   *
38   * <p>This assymetric key technology was created as a
39   * more secure alternative to symmetric ciphers that sometimes suffer from man-in-the-middle attacks since, for
40   * data shared between two parties, the same Key must also be shared and may be compromised.
41   *
42   * <p>Note that a symmetric cipher is perfectly fine to use if you just want to encode data in a format no one else
43   * can understand and you never give away the key.  JSecurity uses a symmetric cipher when using certain
44   * HTTP Cookies for example - because it is often undesireable to have user's identity stored in a plain-text cookie,
45   * that identity can be converted via a symmetric cipher.  Since the the same exact JSecurity application will receive
46   * the cookie, it can decrypt it via the same <tt>Key</tt> and there is no potential for discovery since that Key
47   * is never shared with anyone.
48   *
49   * @author Les Hazlewood
50   * @see BlowfishCipher
51   * @since 0.9
52   */
53  public interface Cipher {
54  
55      /**
56       * Encrypts data via the specified Cipher key.  Note that the key must be in a format understood by
57       * the Cipher implementation.
58       *
59       * @param raw           the data to encrypt
60       * @param encryptionKey the Cipher key used during encryption.
61       * @return an encrypted representation of the specified raw data.
62       */
63      byte[] encrypt(byte[] raw, byte[] encryptionKey);
64  
65      /**
66       * Decrypts encrypted data via the specified Cipher key and returns the original (pre-encrypted) data.
67       * Note that the key must be in a format understood by the Cipher implementation.
68       *
69       * @param encrypted     the previously encrypted data to decrypt
70       * @param decryptionKey the Cipher key used during decryption.
71       * @return the original form of the specified encrypted data.
72       */
73      byte[] decrypt(byte[] encrypted, byte[] decryptionKey);
74  
75  }
