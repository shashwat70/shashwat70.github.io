18  package org.apache.tools.ant.util.regexp;
19  
20  import org.apache.tools.ant.Project;
21  import org.apache.tools.ant.MagicNames;
22  import org.apache.tools.ant.BuildException;
23  import org.apache.tools.ant.util.ClasspathUtils;
24  
25  /**
26   * Simple Factory Class that produces an implementation of RegexpMatcher based on the system
27   * property <code>ant.regexp.regexpimpl</code> and the classes available.
28   *
29   * <p>
30   * In a more general framework this class would be abstract and have a static newInstance method.
31   * </p>
32   *
33   */
34  public class RegexpMatcherFactory {
35  
36      /** Constructor for RegexpMatcherFactory. */
37      public RegexpMatcherFactory() {
38      }
39  
40      /***
41       * Create a new regular expression instance.
42       * @return the matcher
43       * @throws BuildException on error
44       */
45      public RegexpMatcher newRegexpMatcher() throws BuildException {
46          return newRegexpMatcher(null);
47      }
48  
49      /***
50       * Create a new regular expression instance.
51       *
52       * @param p Project whose ant.regexp.regexpimpl property will be used.
53       * @return the matcher
54       * @throws BuildException on error
55       */
56      public RegexpMatcher newRegexpMatcher(Project p) throws BuildException {
57          String systemDefault = null;
58          if (p == null) {
59              systemDefault = System.getProperty(MagicNames.REGEXP_IMPL);
60          } else {
61              systemDefault = p.getProperty(MagicNames.REGEXP_IMPL);
62          }
63  
64          if (systemDefault != null) {
65              return createInstance(systemDefault);
66              // TODO     should we silently catch possible exceptions and try to
67              //         load a different implementation?
68          }
69  
70          return new Jdk14RegexpMatcher();
71      }
72  
73      /**
74       * Create an instance of a matcher from a classname.
75       *
76       * @param className a <code>String</code> value
77       * @return a <code>RegexpMatcher</code> value
78       * @exception BuildException if an error occurs
79       */
80      protected RegexpMatcher createInstance(String className) throws BuildException {
81          return (RegexpMatcher) ClasspathUtils.newInstance(className, RegexpMatcherFactory.class
82                  .getClassLoader(), RegexpMatcher.class);
83      }
84  
85      /**
86       * Test if a particular class is available to be used.
87       *
88       * @param className a <code>String</code> value
89       * @exception BuildException if an error occurs
90       */
91      protected void testAvailability(String className) throws BuildException {
92          try {
93              Class.forName(className);
94          } catch (Throwable t) {
95              throw new BuildException(t);
96          }
97      }
98  
99      /**
100      * Checks if a RegExp-Matcher is available.
101      * @param project  The project to check for (may be <code>null</code>)
102      * @return <code>true</code> if available otherwise <code>false</code>
103      */
104     public static boolean regexpMatcherPresent(Project project) {
105         try {
106             // The factory throws a BuildException if no usable matcher
107             // cant be instantiated. We dont need the matcher itself here.
108             new RegexpMatcherFactory().newRegexpMatcher(project);
109             return true;
110         } catch (Throwable ex) {
111             return false;
112         }
113     }
114 }
