80  	protected List executeLoad(
81  			SharedSessionContractImplementor session,
82  			QueryParameters queryParameters,
83  			LoadQueryDetails loadQueryDetails,
84  			boolean returnProxies,
85  			ResultTransformer forcedResultTransformer) throws SQLException {
86  		final PersistenceContext persistenceContext = session.getPersistenceContextInternal();
87  		final boolean defaultReadOnlyOrig = persistenceContext.isDefaultReadOnly();
88  		if ( queryParameters.isReadOnlyInitialized() ) {
89  			// The read-only/modifiable mode for the query was explicitly set.
90  			// Temporarily set the default read-only/modifiable setting to the query's setting.
91  			persistenceContext.setDefaultReadOnly( queryParameters.isReadOnly() );
92  		}
93  		else {
94  			// The read-only/modifiable setting for the query was not initialized.
95  			// Use the default read-only/modifiable from the persistence context instead.
96  			queryParameters.setReadOnly( persistenceContext.isDefaultReadOnly() );
97  		}
98  		persistenceContext.beforeLoad();
99  		try {
100 			final List results;
101 			final String sql = loadQueryDetails.getSqlStatement();
102 			SqlStatementWrapper wrapper = null;
103 			try {
104 				wrapper = executeQueryStatement( sql, queryParameters, false, session );
105 				results = loadQueryDetails.getResultSetProcessor().extractResults(
106 						wrapper.getResultSet(),
107 						session,
108 						queryParameters,
109 						new NamedParameterContext() {
110 							@Override
111 							public int[] getNamedParameterLocations(String name) {
112 								return AbstractLoadPlanBasedLoader.this.getNamedParameterLocs( name );
113 							}
114 						},
115 						returnProxies,
116 						queryParameters.isReadOnly(),
117 						forcedResultTransformer,
118 						Collections.EMPTY_LIST
119 				);
120 			}
121 			finally {
122 				if ( wrapper != null ) {
123 					final JdbcCoordinator jdbcCoordinator = session.getJdbcCoordinator();
124 					final ResourceRegistry resourceRegistry = jdbcCoordinator.getResourceRegistry();
125 					resourceRegistry.release( wrapper.getStatement() );
126 					jdbcCoordinator.afterStatementExecution();
127 				}
128 				persistenceContext.afterLoad();
129 			}
130 			persistenceContext.initializeNonLazyCollections();
131 			return results;
132 		}
133 		finally {
134 			// Restore the original default
135 			persistenceContext.setDefaultReadOnly( defaultReadOnlyOrig );
136 		}
137 	}

139 	protected SqlStatementWrapper executeQueryStatement(
140 			String sqlStatement,
141 			QueryParameters queryParameters,
142 			boolean scroll,
143 			SharedSessionContractImplementor session) throws SQLException {
144 
145 		// Processing query filters.
146 		queryParameters.processFilters( sqlStatement, session );
147 
148 		// Applying LIMIT clause.
149 		final LimitHandler limitHandler = getLimitHandler(
150 				queryParameters.getRowSelection()
151 		);
152 		String sql = limitHandler.processSql( queryParameters.getFilteredSQL(), queryParameters.getRowSelection() );
153 
154 		// Adding locks and comments.
155 		sql = session.getJdbcServices().getJdbcEnvironment().getDialect()
156 				.addSqlHintOrComment(
157 					sql,
158 					queryParameters,
159 					session.getFactory().getSessionFactoryOptions().isCommentsEnabled()
160 				);
161 
162 		final PreparedStatement st = prepareQueryStatement( sql, queryParameters, limitHandler, scroll, session );
163 		return new SqlStatementWrapper( st, getResultSet( st, queryParameters.getRowSelection(), limitHandler, queryParameters.hasAutoDiscoverScalarTypes(), session ) );
164 	}

166 	/**
167 	 * Build LIMIT clause handler applicable for given selection criteria. Returns {@link org.hibernate.dialect.pagination.NoopLimitHandler} delegate
168 	 * if dialect does not support LIMIT expression or processed query does not use pagination.
169 	 *
170 	 * @param selection Selection criteria.
171 	 * @return LIMIT clause delegate.
172 	 */
173 	protected LimitHandler getLimitHandler(RowSelection selection) {
174 		final LimitHandler limitHandler = getFactory().getDialect().getLimitHandler();
175 		return LimitHelper.useLimit( limitHandler, selection ) ? limitHandler : NoopLimitHandler.INSTANCE;
176 	}

178 	/**
179 	 * Obtain a <tt>PreparedStatement</tt> with all parameters pre-bound.
180 	 * Bind JDBC-style <tt>?</tt> parameters, named parameters, and
181 	 * limit parameters.
182 	 */
183 	protected final PreparedStatement prepareQueryStatement(
184 			final String sql,
185 			final QueryParameters queryParameters,
186 			final LimitHandler limitHandler,
187 			final boolean scroll,
188 			final SharedSessionContractImplementor session) throws SQLException, HibernateException {
189 		final Dialect dialect = session.getJdbcServices().getJdbcEnvironment().getDialect();
190 		final RowSelection selection = queryParameters.getRowSelection();
191 		final boolean useLimit = LimitHelper.useLimit( limitHandler, selection );
192 		final boolean hasFirstRow = LimitHelper.hasFirstRow( selection );
193 		final boolean useLimitOffset = hasFirstRow && useLimit && limitHandler.supportsLimitOffset();
194 		final boolean callable = queryParameters.isCallable();
195 		final ScrollMode scrollMode = getScrollMode( scroll, hasFirstRow, useLimitOffset, queryParameters );
196 
197 		final PreparedStatement st = session.getJdbcCoordinator()
198 				.getStatementPreparer().prepareQueryStatement( sql, callable, scrollMode );
199 
200 		try {
201 
202 			int col = 1;
203 			//TODO: can we limit stored procedures ?!
204 			col += limitHandler.bindLimitParametersAtStartOfQuery( selection, st, col );
205 
206 			if (callable) {
207 				col = dialect.registerResultSetOutParameter( (CallableStatement)st, col );
208 			}
209 
210 			col += bindParameterValues( st, queryParameters, col, session );
211 
212 			col += limitHandler.bindLimitParametersAtEndOfQuery( selection, st, col );
213 
214 			limitHandler.setMaxRows( selection, st );
215 
216 			if ( selection != null ) {
217 				if ( selection.getTimeout() != null ) {
218 					st.setQueryTimeout( selection.getTimeout() );
219 				}
220 				if ( selection.getFetchSize() != null ) {
221 					st.setFetchSize( selection.getFetchSize() );
222 				}
223 			}
224 
225 			// handle lock timeout...
226 			final LockOptions lockOptions = queryParameters.getLockOptions();
227 			if ( lockOptions != null ) {
228 				if ( lockOptions.getTimeOut() != LockOptions.WAIT_FOREVER ) {
229 					if ( !dialect.supportsLockTimeouts() ) {
230 						if ( log.isDebugEnabled() ) {
231 							log.debugf(
232 									"Lock timeout [%s] requested but dialect reported to not support lock timeouts",
233 									lockOptions.getTimeOut()
234 							);
235 						}
236 					}
237 					else if ( dialect.isLockTimeoutParameterized() ) {
238 						st.setInt( col++, lockOptions.getTimeOut() );
239 					}
240 				}
241 			}
242 
243 			if ( log.isTraceEnabled() ) {
244 				log.tracev( "Bound [{0}] parameters total", col );
245 			}
246 		}
247 		catch ( SQLException sqle ) {
248 			session.getJdbcCoordinator().getResourceRegistry().release( st );
249 			session.getJdbcCoordinator().afterStatementExecution();
250 			throw sqle;
251 		}
252 		catch ( HibernateException he ) {
253 			session.getJdbcCoordinator().getResourceRegistry().release( st );
254 			session.getJdbcCoordinator().afterStatementExecution();
255 			throw he;
256 		}
257 
258 		return st;
259 	}
260 
261 	protected ScrollMode getScrollMode(boolean scroll, boolean hasFirstRow, boolean useLimitOffSet, QueryParameters queryParameters) {
262 		final boolean canScroll = getFactory().getSettings().isScrollableResultSetsEnabled();
263 		if ( canScroll ) {
264 			if ( scroll ) {
265 				return queryParameters.getScrollMode();
266 			}
267 			if ( hasFirstRow && !useLimitOffSet ) {
268 				return ScrollMode.SCROLL_INSENSITIVE;
269 			}
270 		}
271 		return null;
272 	}

274 	/**
275 	 * Bind all parameter values into the prepared statement in preparation
276 	 * for execution.
277 	 *
278 	 * @param statement The JDBC prepared statement
279 	 * @param queryParameters The encapsulation of the parameter values to be bound.
280 	 * @param startIndex The position from which to start binding parameter values.
281 	 * @param session The originating session.
282 	 * @return The number of JDBC bind positions actually bound during this method execution.
283 	 * @throws SQLException Indicates problems performing the binding.
284 	 */
285 	protected int bindParameterValues(
286 			PreparedStatement statement,
287 			QueryParameters queryParameters,
288 			int startIndex,
289 			SharedSessionContractImplementor session) throws SQLException {
290 		int span = 0;
291 		span += bindPositionalParameters( statement, queryParameters, startIndex, session );
292 		span += bindNamedParameters( statement, queryParameters.getNamedParameters(), startIndex + span, session );
293 		return span;
294 	}

296 	/**
297 	 * Bind positional parameter values to the JDBC prepared statement.
298 	 * <p/>
299 	 * Positional parameters are those specified by JDBC-style ? parameters
300 	 * in the source query.  It is (currently) expected that these come
301 	 * before any named parameters in the source query.
302 	 *
303 	 * @param statement The JDBC prepared statement
304 	 * @param queryParameters The encapsulation of the parameter values to be bound.
305 	 * @param startIndex The position from which to start binding parameter values.
306 	 * @param session The originating session.
307 	 * @return The number of JDBC bind positions actually bound during this method execution.
308 	 * @throws SQLException Indicates problems performing the binding.
309 	 * @throws org.hibernate.HibernateException Indicates problems delegating binding to the types.
310 	 */
311 	protected int bindPositionalParameters(
312 			final PreparedStatement statement,
313 			final QueryParameters queryParameters,
314 			final int startIndex,
315 			final SharedSessionContractImplementor session) throws SQLException, HibernateException {
316 		final Object[] values = queryParameters.getFilteredPositionalParameterValues();
317 		final Type[] types = queryParameters.getFilteredPositionalParameterTypes();
318 		int span = 0;
319 		for ( int i = 0; i < values.length; i++ ) {
320 			types[i].nullSafeSet( statement, values[i], startIndex + span, session );
321 			span += types[i].getColumnSpan( getFactory() );
322 		}
323 		return span;
324 	}

326 	/**
327 	 * Bind named parameters to the JDBC prepared statement.
328 	 * <p/>
329 	 * This is a generic implementation, the problem being that in the
330 	 * general case we do not know enough information about the named
331 	 * parameters to perform this in a complete manner here.  Thus this
332 	 * is generally overridden on subclasses allowing named parameters to
333 	 * apply the specific behavior.  The most usual limitation here is that
334 	 * we need to assume the type span is always one...
335 	 *
336 	 * @param statement The JDBC prepared statement
337 	 * @param namedParams A map of parameter names to values
338 	 * @param startIndex The position from which to start binding parameter values.
339 	 * @param session The originating session.
340 	 * @return The number of JDBC bind positions actually bound during this method execution.
341 	 * @throws SQLException Indicates problems performing the binding.
342 	 * @throws org.hibernate.HibernateException Indicates problems delegating binding to the types.
343 	 */
344 	protected int bindNamedParameters(
345 			final PreparedStatement statement,
346 			final Map namedParams,
347 			final int startIndex,
348 			final SharedSessionContractImplementor session) throws SQLException, HibernateException {
349 		if ( namedParams != null ) {
350 			// assumes that types are all of span 1
351 			final Iterator itr = namedParams.entrySet().iterator();
352 			int result = 0;
353 			while ( itr.hasNext() ) {
354 				final Map.Entry e = (Map.Entry) itr.next();
355 				final String name = (String) e.getKey();
356 				final TypedValue typedval = (TypedValue) e.getValue();
357 				final int[] locs = getNamedParameterLocs( name );
358 				for ( int loc : locs ) {
359 					if ( log.isDebugEnabled() ) {
360 						log.debugf(
361 								"bindNamedParameters() %s -> %s [%s]",
362 								typedval.getValue(),
363 								name,
364 								loc + startIndex
365 						);
366 					}
367 					typedval.getType().nullSafeSet( statement, typedval.getValue(), loc + startIndex, session );
368 				}
369 				result += locs.length;
370 			}
371 			return result;
372 		}
373 		else {
374 			return 0;
375 		}
376 	}

378 	/**
379 	 * Execute given <tt>PreparedStatement</tt>, advance to the first result and return SQL <tt>ResultSet</tt>.
380 	 */
381 	protected final ResultSet getResultSet(
382 			final PreparedStatement st,
383 			final RowSelection selection,
384 			final LimitHandler limitHandler,
385 			final boolean autodiscovertypes,
386 			final SharedSessionContractImplementor session)
387 			throws SQLException, HibernateException {
388 
389 		try {
390 			ResultSet rs = session.getJdbcCoordinator().getResultSetReturn().extract( st );
391 			rs = wrapResultSetIfEnabled( rs , session );
392 
393 			if ( !limitHandler.supportsLimitOffset() || !LimitHelper.useLimit( limitHandler, selection ) ) {
394 				advance( rs, selection );
395 			}
396 
397 			if ( autodiscovertypes ) {
398 				autoDiscoverTypes( rs );
399 			}
400 			return rs;
401 		}
402 		catch (SQLException | HibernateException ex) {
403 			session.getJdbcCoordinator().getResourceRegistry().release( st );
404 			session.getJdbcCoordinator().afterStatementExecution();
405 			throw ex;
406 		}
407 	}

409 	/**
410 	 * Advance the cursor to the first required row of the <tt>ResultSet</tt>
411 	 */
412 	protected void advance(final ResultSet rs, final RowSelection selection) throws SQLException {
413 		final int firstRow = LimitHelper.getFirstRow( selection );
414 		if ( firstRow != 0 ) {
415 			if ( getFactory().getSettings().isScrollableResultSetsEnabled() ) {
416 				// we can go straight to the first required row
417 				rs.absolute( firstRow );
418 			}
419 			else {
420 				// we need to step through the rows one row at a time (slow)
421 				for ( int m = 0; m < firstRow; m++ ) {
422 					rs.next();
423 				}
424 			}
425 		}
426 	}

428 	private ResultSet wrapResultSetIfEnabled(final ResultSet rs, final SharedSessionContractImplementor session) {
429 		if ( session.getFactory().getSessionFactoryOptions().isWrapResultSetsEnabled() ) {
430 			try {
431 				if ( log.isDebugEnabled() ) {
432 					log.debugf( "Wrapping result set [%s]", rs );
433 				}
434 				ResultSetWrapper wrapper = session.getFactory()
435 						.getServiceRegistry()
436 						.getService( JdbcServices.class )
437 						.getResultSetWrapper();
438 				// synchronized to avoid multi-thread access issues
439 				// Apparently the comment about this needing synchronization was introduced when AbstractLoadPlanBasedLoader first appeared
440 				// in version control. Would need to investigate if it's still needed?
441 				synchronized ( this ) {
442 					return wrapper.wrap( rs, retreiveColumnNameToIndexCache( rs ) );
443 				}
444 			}
445 			catch(SQLException e) {
446 				log.unableToWrapResultSet( e );
447 				return rs;
448 			}
449 		}
450 		else {
451 			return rs;
452 		}
453 	}
454 
455 	private ColumnNameCache retreiveColumnNameToIndexCache(ResultSet rs) throws SQLException {
456 		if ( columnNameCache == null ) {
457 			log.trace( "Building columnName->columnIndex cache" );
458 			columnNameCache = new ColumnNameCache( rs.getMetaData().getColumnCount() );
459 		}
460 
461 		return columnNameCache;
462 	}

464 	/**
465 	 * Wrapper class for {@link java.sql.Statement} and associated {@link java.sql.ResultSet}.
466 	 */
467 	protected static class SqlStatementWrapper {
468 		private final Statement statement;
469 		private final ResultSet resultSet;
470 
471 		private SqlStatementWrapper(Statement statement, ResultSet resultSet) {
472 			this.resultSet = resultSet;
473 			this.statement = statement;
474 		}
475 
476 		public ResultSet getResultSet() {
477 			return resultSet;
478 		}
479 
480 		public Statement getStatement() {
481 			return statement;
