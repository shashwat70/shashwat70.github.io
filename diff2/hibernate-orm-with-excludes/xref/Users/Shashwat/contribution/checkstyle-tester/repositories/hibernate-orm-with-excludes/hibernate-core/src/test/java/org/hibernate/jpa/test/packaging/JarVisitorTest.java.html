80  	private ScanResult standardScan(URL url) {
81  		ScanEnvironment env = new ScanEnvironmentImpl( url );
82  		return new StandardScanner().scan(
83  				env,
84  				new StandardScanOptions(),
85  				StandardScanParameters.INSTANCE
86  		);
87  	}

126 	private void validateResults(ScanResult scanResult, Class... expectedClasses) throws IOException {
127 		assertEquals( 3, scanResult.getLocatedClasses().size() );
128 		for ( Class expectedClass : expectedClasses ) {
129 			assertTrue(
130 					scanResult.getLocatedClasses().contains(
131 							new ClassDescriptorImpl( expectedClass.getName(), ClassDescriptor.Categorization.MODEL, null )
132 					)
133 			);
134 		}
135 
136 		assertEquals( 2, scanResult.getLocatedMappingFiles().size() );
137 		for ( MappingFileDescriptor mappingFileDescriptor : scanResult.getLocatedMappingFiles() ) {
138 			assertNotNull( mappingFileDescriptor.getStreamAccess() );
139 			final InputStream stream = mappingFileDescriptor.getStreamAccess().accessInputStream();
140 			assertNotNull( stream );
141 			stream.close();
142 		}

279 	@TestForIssue(jiraKey = "HHH-6806")
280 	public void testJarVisitorFactory() throws Exception {
281 		final File explodedPar = buildExplodedPar();
282 		final File defaultPar = buildDefaultPar();
283 		addPackageToClasspath( explodedPar, defaultPar );
284 
285 		//setting URL to accept vfs based protocol
286 		URL.setURLStreamHandlerFactory(new URLStreamHandlerFactory() {
287 										   public URLStreamHandler createURLStreamHandler(String protocol) {
288 											   if("vfszip".equals(protocol) || "vfsfile".equals(protocol) )
289 												   return new URLStreamHandler() {
290 													   protected URLConnection openConnection(URL u)
291 															   throws IOException {
292 														   return null;
293 													   }
294 												   };
295 											   return null;
296 										   }
297 									   });
298 
299 		URL jarUrl = defaultPar.toURL();
300 		ArchiveDescriptor descriptor = StandardArchiveDescriptorFactory.INSTANCE.buildArchiveDescriptor( jarUrl );
301 		assertEquals( JarFileBasedArchiveDescriptor.class.getName(), descriptor.getClass().getName() );
302 
303 		jarUrl  = explodedPar.toURL();
304 		descriptor = StandardArchiveDescriptorFactory.INSTANCE.buildArchiveDescriptor( jarUrl );
305 		assertEquals( ExplodedArchiveDescriptor.class.getName(), descriptor.getClass().getName() );
306 
307 		jarUrl  = new URL( defaultPar.toURL().toExternalForm().replace( "file:", "vfszip:" ) );
308 		descriptor = StandardArchiveDescriptorFactory.INSTANCE.buildArchiveDescriptor( jarUrl );
309 		assertEquals( JarFileBasedArchiveDescriptor.class.getName(), descriptor.getClass().getName());
310 
311 		jarUrl  = new URL( explodedPar.toURL().toExternalForm().replace( "file:", "vfsfile:" ) );
312 		descriptor = StandardArchiveDescriptorFactory.INSTANCE.buildArchiveDescriptor( jarUrl );
313 		assertEquals( ExplodedArchiveDescriptor.class.getName(), descriptor.getClass().getName() );
314 	}

362 	@Test
363 	@TestForIssue(jiraKey = "HHH-7835")
364 	public void testGetBytesFromInputStream() throws Exception {
365 		File file = buildLargeJar();
366 
367 		long start = System.currentTimeMillis();
368 		InputStream stream = new BufferedInputStream(
369 				new FileInputStream( file ) );
370 		int oldLength = getBytesFromInputStream( stream ).length;
371 		stream.close();
372 		long oldTime = System.currentTimeMillis() - start;
373 
374 		start = System.currentTimeMillis();
375 		stream = new BufferedInputStream( new FileInputStream( file ) );
376 		int newLength = ArchiveHelper.getBytesFromInputStream( stream ).length;
377 		stream.close();
378 		long newTime = System.currentTimeMillis() - start;
379 
380 		assertEquals( oldLength, newLength );
381 		assertTrue( oldTime > newTime );
382 	}

384 	// This is the old getBytesFromInputStream from JarVisitorFactory before
385 	// it was changed by HHH-7835. Use it as a regression test.
386 	private byte[] getBytesFromInputStream(InputStream inputStream) throws IOException {
387 		int size;
388 
389 		byte[] entryBytes = new byte[0];
390 		for ( ;; ) {
391 			byte[] tmpByte = new byte[4096];
392 			size = inputStream.read( tmpByte );
393 			if ( size == -1 )
394 				break;
395 			byte[] current = new byte[entryBytes.length + size];
396 			System.arraycopy( entryBytes, 0, current, 0, entryBytes.length );
397 			System.arraycopy( tmpByte, 0, current, entryBytes.length, size );
398 			entryBytes = current;
399 		}
400 		return entryBytes;
401 	}
