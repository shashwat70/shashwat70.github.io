7   package org.hibernate.id.enhanced;
8   
9   import java.io.Serializable;
10  import java.util.Map;
11  import java.util.concurrent.ConcurrentHashMap;
12  
13  import org.hibernate.HibernateException;
14  import org.hibernate.id.IntegralDataTypeHolder;
15  
16  import org.jboss.logging.Logger;
17  
18  /**
19   * Optimizer which applies a 'hilo' algorithm in memory to achieve
20   * optimization.
21   * <p/>
22   * A 'hilo' algorithm is simply a means for a single value stored in the
23   * database to represent a "bucket" of possible, contiguous values.  The
24   * database value identifies which particular bucket we are on.
25   * <p/>
26   * This database value must be paired with another value that defines the
27   * size of the bucket; the number of possible values available.
28   * The {@link #getIncrementSize() incrementSize} serves this purpose.  The
29   * naming here is meant more for consistency in that this value serves the
30   * same purpose as the increment supplied to the {@link PooledOptimizer}.
31   * <p/>
32   * The general algorithms used to determine the bucket are:<ol>
33   * <li>{@code upperLimit = (databaseValue * incrementSize) + 1}</li>
34   * <li>{@code lowerLimit = upperLimit - incrementSize}</li>
35   * </ol>
36   * As an example, consider a case with incrementSize of 20.  Initially the
37   * database holds 1:<ol>
38   * <li>{@code upperLimit = (1 * 20) + 1 = 21}</li>
39   * <li>{@code lowerLimit = 21 - 20 = 1}</li>
40   * </ol>
41   * From there we increment the value from lowerLimit until we reach the
42   * upperLimit, at which point we would define a new bucket.  The database
43   * now contains 2, though incrementSize remains unchanged:<ol>
44   * <li>{@code upperLimit = (2 * 20) + 1 = 41}</li>
45   * <li>{@code lowerLimit = 41 - 20 = 21}</li>
46   * </ol>
47   * And so on...
48   * <p/>
49   * Note, 'value' always (after init) holds the next value to return
50   *
51   * @author Steve Ebersole
52   */
53  public class HiLoOptimizer extends AbstractOptimizer {
54  	private static final Logger log = Logger.getLogger( HiLoOptimizer.class );
55  
56  	private static class GenerationState {
57  		private IntegralDataTypeHolder lastSourceValue;
58  		private IntegralDataTypeHolder upperLimit;
59  		private IntegralDataTypeHolder value;
60  	}
61  
62  
63  	/**
64  	 * Constructs a HiLoOptimizer
65  	 *
66  	 * @param returnClass The Java type of the values to be generated
67  	 * @param incrementSize The increment size.
68  	 */
69  	public HiLoOptimizer(Class returnClass, int incrementSize) {
70  		super( returnClass, incrementSize );
71  		if ( incrementSize < 1 ) {
72  			throw new HibernateException( "increment size cannot be less than 1" );
73  		}
74  		if ( log.isTraceEnabled() ) {
75  			log.tracev( "Creating hilo optimizer with [incrementSize={0}; returnClass={1}]", incrementSize, returnClass.getName() );
76  		}
77  	}
78  
79  	@Override
80  	public synchronized Serializable generate(AccessCallback callback) {
81  		final GenerationState generationState = locateGenerationState( callback.getTenantIdentifier() );
82  
83  		if ( generationState.lastSourceValue == null ) {
84  			// first call, so initialize ourselves.  we need to read the database
85  			// value and set up the 'bucket' boundaries
86  			generationState.lastSourceValue = callback.getNextValue();
87  			while ( generationState.lastSourceValue.lt( 1 ) ) {
88  				generationState.lastSourceValue = callback.getNextValue();
89  			}
90  			// upperLimit defines the upper end of the bucket values
91  			generationState.upperLimit = generationState.lastSourceValue.copy().multiplyBy( incrementSize ).increment();
92  			// initialize value to the low end of the bucket
93  			generationState.value = generationState.upperLimit.copy().subtract( incrementSize );
94  		}
95  		else if ( ! generationState.upperLimit.gt( generationState.value ) ) {
96  			generationState.lastSourceValue = callback.getNextValue();
97  			generationState.upperLimit = generationState.lastSourceValue.copy().multiplyBy( incrementSize ).increment();
98  			generationState.value = generationState.upperLimit.copy().subtract( incrementSize );
99  		}
100 		return generationState.value.makeValueThenIncrement();
101 	}
102 
103 	private GenerationState noTenantState;
104 	private Map<String,GenerationState> tenantSpecificState;
105 
106 	private GenerationState locateGenerationState(String tenantIdentifier) {
107 		if ( tenantIdentifier == null ) {
108 			if ( noTenantState == null ) {
109 				noTenantState = new GenerationState();
110 			}
111 			return noTenantState;
112 		}
113 		else {
114 			GenerationState state;
115 			if ( tenantSpecificState == null ) {
116 				tenantSpecificState = new ConcurrentHashMap<String, GenerationState>();
117 				state = new GenerationState();
118 				tenantSpecificState.put( tenantIdentifier, state );
119 			}
120 			else {
121 				state = tenantSpecificState.get( tenantIdentifier );
122 				if ( state == null ) {
123 					state = new GenerationState();
124 					tenantSpecificState.put( tenantIdentifier, state );
125 				}
126 			}
127 			return state;
128 		}
129 	}
130 
131 	private GenerationState noTenantGenerationState() {
132 		if ( noTenantState == null ) {
133 			throw new IllegalStateException( "Could not locate previous generation state for no-tenant" );
134 		}
135 		return noTenantState;
136 	}
137 
138 	@Override
139 	public synchronized IntegralDataTypeHolder getLastSourceValue() {
140 		return noTenantGenerationState().lastSourceValue;
141 	}
142 
143 	@Override
144 	public boolean applyIncrementSizeToSourceValues() {
145 		return false;
146 	}
147 
148 	/**
149 	 * Getter for property 'lastValue'.
150 	 * <p/>
151 	 * Exposure intended for testing purposes.
152 	 *
153 	 * @return Value for property 'lastValue'.
154 	 */
155 	public synchronized IntegralDataTypeHolder getLastValue() {
156 		return noTenantGenerationState().value.copy().decrement();
157 	}
158 
159 	/**
160 	 * Getter for property 'upperLimit'.
161 	 * <p/>
162 	 * Exposure intended for testing purposes.
163 	 *
164 	 * @return Value for property 'upperLimit'.
165 	 */
166 	public synchronized IntegralDataTypeHolder getHiValue() {
167 		return noTenantGenerationState().upperLimit;
168 	}
169 }
