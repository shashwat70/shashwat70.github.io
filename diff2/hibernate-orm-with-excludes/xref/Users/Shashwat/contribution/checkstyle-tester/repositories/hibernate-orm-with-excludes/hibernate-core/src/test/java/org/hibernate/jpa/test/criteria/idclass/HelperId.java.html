7   package org.hibernate.jpa.test.criteria.idclass;
8   
9   import java.io.Serializable;
10  import javax.persistence.Column;
11  import javax.persistence.Embeddable;
12  import javax.persistence.Id;
13  
14  /**
15   * @author Gail Badner
16   */
17  
18  @Embeddable
19  public class HelperId implements Serializable {
20  	private static final long serialVersionUID = 9122480802791185646L;
21  	
22  	@Id
23  	@Column( name = "NAME", length = 12 )
24  	private String name;
25  	public String getName( ) { return this.name; }
26  	public void setName( String value ) { this.name = value; }
27  	
28  	@Id
29  	@Column( name = "HELPER_TYPE", length = 4 )
30  	private String type;
31  	public String getType( ) { return this.type; }
32  	public void setType( String value ) { this.type = value; }
33  	
34  	@Override
35  	public boolean equals( Object obj ) {
36  		if( obj == null ) return false;
37  		if( !( obj instanceof HelperId ) ) return false;
38  		
39  		HelperId id = ( HelperId )obj;
40  		if( this.getName() == null || id.getName() == null || this.getType() == null || id.getType() == null ) return false;
41  		
42  		return this.toString( ).equals( id.toString( ) );
43  	}
44  	
45  	@Override
46  	public int hashCode( ) {
47  		return this.toString( ).hashCode( );
48  	}
49  	
50  	@Override
51  	public String toString( ) {
52  		StringBuilder buf = new StringBuilder( "[id:" );
53  		buf.append( ( this.getName() == null ) ? "null" : this.getName( ) );
54  		buf.append( ";type:" );
55  		buf.append( ( this.getType() == null ) ? "null" : this.getType() );
56  		buf.append( "]" );
57  		
58  		return buf.toString( );
59  	}
60  }
