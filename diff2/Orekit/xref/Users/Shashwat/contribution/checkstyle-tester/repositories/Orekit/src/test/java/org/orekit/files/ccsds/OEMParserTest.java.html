65          Assert.assertEquals("1996-062A", file.getEphemeridesBlocks().get(0).getMetaData().getObjectID());

68          Assert.assertEquals(62, file.getEphemeridesBlocks().get(0).getMetaData().getLaunchNumber());

75                              file.getEphemeridesBlocks().get(0).getStopTime());
76          Assert.assertEquals(new AbsoluteDate(1996, 12, 18, 12, 10, 0.331, TimeScalesFactory.getUTC()),
77                              file.getEphemeridesBlocks().get(0).getUseableStartTime());
78          Assert.assertEquals(new AbsoluteDate(1996, 12, 28, 21, 23, 0.331, TimeScalesFactory.getUTC()),
79                              file.getEphemeridesBlocks().get(0).getUseableStopTime());
80          Assert.assertEquals("HERMITE", file.getEphemeridesBlocks().get(0).getInterpolationMethod());
81          Assert.assertEquals(7, file.getEphemeridesBlocks().get(0).getInterpolationDegree());
82          ArrayList<String> ephemeridesDataLinesComment = new ArrayList<String>();

109             1.798098699846038e-07, 3.540310904497689e-07

111         final double[] column4 = {

149     @Test
150     public void testParseOEM1OrbitFile() throws OrekitException, IOException {
151  
152         final String ex = "/ccsds/OEMExample.txt";
153         final InputStream inEntry = getClass().getResourceAsStream(ex);
154         final OEMParser parser = new OEMParser().withMu(CelestialBodyFactory.getEarth().getGM());
155         final OEMFile file = parser.parse(inEntry);
156         Assert.assertEquals(TimeSystem.UTC, file.getTimeSystem());
157         Assert.assertEquals("MARS GLOBAL SURVEYOR", file.getEphemeridesBlocks().get(0).getMetaData().getObjectName());
158         Assert.assertEquals("1996-062A", file.getEphemeridesBlocks().get(0).getMetaData().getObjectID());
159 
160         Assert.assertEquals(1, file.getSatelliteCount());
161         Assert.assertEquals(true, file.containsSatellite("1996-062A"));
162         Assert.assertEquals(false, file.containsSatellite("MARS GLOBAL SURVEYOR"));
163         Assert.assertEquals(1, file.getSatellites().size());
164         Assert.assertEquals("1996-062A", file.getSatellites().iterator().next().getSatelliteId());
165         Assert.assertEquals(new AbsoluteDate("1996-12-18T12:00:00.331", TimeScalesFactory.getUTC()), file.getEpoch());
166 
167         final List<SatelliteTimeCoordinate> coordinates = file.getSatelliteCoordinates("1996-062A");
168         Assert.assertEquals(13, coordinates.size());
169 
170         final List<EphemeridesDataLine> dataLines = new ArrayList<EphemeridesDataLine>();
171         for (EphemeridesBlock block : file.getEphemeridesBlocks()) {
172             for (EphemeridesDataLine dataLine : block.getEphemeridesDataLines()) {
173                 dataLines.add(dataLine);
174             }
175         }
176 
177         int index = 0;
178         for (SatelliteTimeCoordinate coord : file.getSatelliteCoordinates("1996-062A")) {
179             final EphemeridesDataLine dataLine = dataLines.get(index++);
180             final CartesianOrbit orbit = dataLine.getOrbit();
181             Assert.assertEquals(orbit.getDate(), coord.getDate());
182             Assert.assertEquals(orbit.getDate(), coord.getEpoch());
183             Assert.assertTrue(coord.getEpoch().durationFrom(AbsoluteDate.MODIFIED_JULIAN_EPOCH) > 4.3e9);
184             coord.setEpoch(AbsoluteDate.MODIFIED_JULIAN_EPOCH);
185             Assert.assertEquals(0.0, coord.getEpoch().durationFrom(AbsoluteDate.MODIFIED_JULIAN_EPOCH), 1.0e-10);
186             Assert.assertEquals(orbit.getPVCoordinates(), coord.getCoordinate());
187             Assert.assertTrue(coord.getCoordinate().getPosition().getNorm() > 2.5e6);
188             coord.setCoordinate(PVCoordinates.ZERO);
189             Assert.assertEquals(0.0, coord.getCoordinate().getPosition().getNorm(), 1.0e-10);
190             Assert.assertEquals(0.0, coord.getClockCorrection(), 1.0e-15);
191             coord.setClockCorrection(1.0);
192             Assert.assertEquals(1.0, coord.getClockCorrection(), 1.0e-15);
193             Assert.assertEquals(0.0, coord.getClockRateChange(), 1.0e-15);
194             coord.setClockRateChange(1.0);
195             Assert.assertEquals(1.0, coord.getClockRateChange(), 1.0e-15);
196         }
197     }

199     @Test
200     public void testParseOEM2() 
201             throws OrekitException, URISyntaxException {
202 
203         final String name = getClass().getResource("/ccsds/OEMExample2.txt").toURI().getPath();
204         OEMParser parser = new OEMParser().
205                 withConventions(IERSConventions.IERS_2010).
206                 withSimpleEOP(true).
207                 withMu(CelestialBodyFactory.getMars().getGM()).
208                 withInternationalDesignator(1996, 2, "A").
209                 withMissionReferenceDate(new AbsoluteDate("1996-12-17T00:00:00.000",
210                                                           TimeScalesFactory.getUTC()));
211 
212         final OEMFile file = parser.parse(name);
213         final List<String> headerComment = new ArrayList<String>();
214         headerComment.add("comment");
215         Assert.assertEquals(headerComment, file.getHeaderComment());
216         final List<String> metadataComment = new ArrayList<String>();
217         metadataComment.add("comment 1");
218         metadataComment.add("comment 2");
219         Assert.assertEquals(metadataComment, file.getEphemeridesBlocks().get(0).getMetaData().getComment());
220         Assert.assertEquals("TOD/2010 simple EOP", file.getCoordinateSystem());
221         List<EphemeridesBlock> blocks = file.getEphemeridesBlocks();
222         Assert.assertEquals(2, blocks.size());
223         Assert.assertTrue(blocks.get(0).getHasRefFrameEpoch());
224         Assert.assertEquals(129600.331,
225                             blocks.get(0).getStartTime().durationFrom(file.getMissionReferenceDate()),
226                             1.0e-15);
227         Assert.assertEquals(941347.267,
228                             blocks.get(1).getStartTime().durationFrom(file.getMissionReferenceDate()),
229                             1.0e-15);
230 
231     }
