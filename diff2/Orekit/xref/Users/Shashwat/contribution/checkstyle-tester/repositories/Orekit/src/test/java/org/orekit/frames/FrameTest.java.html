112     @Test
113     public void testDepthAndAncestor() throws OrekitException{
114         Random random = new Random(0x01f8d3b944123044l);
115         Frame root = Frame.getRoot();
116 
117         Frame f1 = new Frame(root, randomTransform(random), "f1");
118         Frame f2 = new Frame(f1,   randomTransform(random), "f2");
119         Frame f3 = new Frame(f1,   randomTransform(random), "f3");
120         Frame f4 = new Frame(f2,   randomTransform(random), "f4");
121         Frame f5 = new Frame(f3,   randomTransform(random), "f5");
122         Frame f6 = new Frame(f5,   randomTransform(random), "f6");
123 
124         Assert.assertEquals(0, root.getDepth());
125         Assert.assertEquals(1, f1.getDepth());
126         Assert.assertEquals(2, f2.getDepth());
127         Assert.assertEquals(2, f3.getDepth());
128         Assert.assertEquals(3, f4.getDepth());
129         Assert.assertEquals(3, f5.getDepth());
130         Assert.assertEquals(4, f6.getDepth());
131 
132         Assert.assertTrue(root == f1.getAncestor(1));
133         Assert.assertTrue(root == f6.getAncestor(4));
134         Assert.assertTrue(f1   == f6.getAncestor(3));
135         Assert.assertTrue(f3   == f6.getAncestor(2));
136         Assert.assertTrue(f5   == f6.getAncestor(1));
137         Assert.assertTrue(f6   == f6.getAncestor(0));
138 
139         try {
140             f6.getAncestor(5);
141             Assert.fail("an exception should have been triggered");
142         } catch (IllegalArgumentException iae) {
143             // expected behavior
144         } catch (Exception e) {
145             Assert.fail("wrong exception caught: " + e.getClass().getName());
146         }

150     @Test
151     public void testIsChildOf() throws OrekitException{
152         Random random = new Random(0xb7d1a155e726da78l);
153         Frame eme2000 = FramesFactory.getEME2000();
154 
155         Frame f1 = new Frame(eme2000, randomTransform(random), "f1");
156         Frame f2 = new Frame(f1     , randomTransform(random), "f2");
157         Frame f4 = new Frame(f2     , randomTransform(random), "f4");
158         Frame f5 = new Frame(f4     , randomTransform(random), "f5");
159         Frame f6 = new Frame(eme2000, randomTransform(random), "f6");
160         Frame f7 = new Frame(f6     , randomTransform(random), "f7");
161         Frame f8 = new Frame(f6     , randomTransform(random), "f8");
162         Frame f9 = new Frame(f7     , randomTransform(random), "f9");
163 
164         // check if the root frame can be an ancestor of another frame
165         Assert.assertEquals(false, eme2000.isChildOf(f5));
166 
167         // check if a frame which belongs to the same branch than the 2nd frame is a branch of it
168         Assert.assertEquals(true, f5.isChildOf(f1));
169 
170         // check if a random frame is the child of the root frame
171         Assert.assertEquals(true, f9.isChildOf(eme2000));
172 
173         // check that a frame is not its own child
174         Assert.assertEquals(false, f4.isChildOf(f4));
175 
176         // check if a frame which belongs to a different branch than the 2nd frame can be a child for it
177         Assert.assertEquals(false, f9.isChildOf(f5));
178 
179         // check if the root frame is not a child of itself
180         Assert.assertEquals(false, eme2000.isChildOf(eme2000));
181 
182         Assert.assertEquals(false, f9.isChildOf(f8));
183 
184     }

186     @Test
187     public void testH0m9() throws OrekitException {
188         AbsoluteDate h0         = new AbsoluteDate("2010-07-01T10:42:09", TimeScalesFactory.getUTC());
189         Frame itrf              = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
190         Frame rotatingPadFrame  = new TopocentricFrame(new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
191                                                                             Constants.WGS84_EARTH_FLATTENING,
192                                                                             itrf),
193                                                        new GeodeticPoint(FastMath.toRadians(5.0),
194                                                                                               FastMath.toRadians(-100.0),
195                                                                                               0.0),
196                                                        "launch pad");
197 
198         // create a new inertially oriented frame that is aligned with ITRF at h0 - 9 seconds
199         AbsoluteDate h0M9       = h0.shiftedBy(-9.0);
200         Frame eme2000           = FramesFactory.getEME2000();
201         Frame frozenLaunchFrame = rotatingPadFrame.getFrozenFrame(eme2000, h0M9, "launch frame");
202 
203         // check velocity module is unchanged
204         Vector3D pEme2000 = new Vector3D(-29536113.0, 30329259.0, -100125.0);
205         Vector3D vEme2000 = new Vector3D(-2194.0, -2141.0, -8.0);
206         PVCoordinates pvEme2000 = new PVCoordinates(pEme2000, vEme2000);
207         PVCoordinates pvH0m9 = eme2000.getTransformTo(frozenLaunchFrame, h0M9).transformPVCoordinates(pvEme2000);
208         Assert.assertEquals(vEme2000.getNorm(), pvH0m9.getVelocity().getNorm(), 1.0e-6);
209 
210         // this frame is fixed with respect to EME2000 but rotates with respect to the non-frozen one
211         // the following loop should have a fixed angle a1 and an evolving angle a2
212         double minA1 = Double.POSITIVE_INFINITY;
213         double maxA1 = Double.NEGATIVE_INFINITY;
214         double minA2 = Double.POSITIVE_INFINITY;
215         double maxA2 = Double.NEGATIVE_INFINITY;
216         double dt;
217         for (dt = 0; dt < 86164; dt += 300.0) {
218             AbsoluteDate date = h0M9.shiftedBy(dt);
219             double a1 = frozenLaunchFrame.getTransformTo(eme2000,          date).getRotation().getAngle();
220             double a2 = frozenLaunchFrame.getTransformTo(rotatingPadFrame, date).getRotation().getAngle();
221             minA1 = FastMath.min(minA1, a1);
222             maxA1 = FastMath.max(maxA1, a1);
223             minA2 = FastMath.min(minA2, a2);
224             maxA2 = FastMath.max(maxA2, a2);
225         }
226         Assert.assertEquals(0, maxA1 - minA1, 1.0e-12);
227         Assert.assertEquals(FastMath.PI, maxA2 - minA2, 0.01);
228 
229     }

231     private Transform randomTransform(Random random) {
232         Transform transform = Transform.IDENTITY;
233         for (int i = random.nextInt(10); i > 0; --i) {
234             if (random.nextBoolean()) {
235                 Vector3D u = new Vector3D(random.nextDouble() * 1000.0,
236                                           random.nextDouble() * 1000.0,
237                                           random.nextDouble() * 1000.0);
238                 transform = new Transform(transform.getDate(), transform, new Transform(transform.getDate(), u));
239             } else {
240                 double q0 = random.nextDouble() * 2 - 1;
241                 double q1 = random.nextDouble() * 2 - 1;
242                 double q2 = random.nextDouble() * 2 - 1;
243                 double q3 = random.nextDouble() * 2 - 1;
244                 double q  = FastMath.sqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
245                 Rotation r = new Rotation(q0 / q, q1 / q, q2 / q, q3 / q, false);
246                 transform = new Transform(transform.getDate(), transform, new Transform(transform.getDate(), r));
247             }
248         }
249         return transform;
250     }

252     private void checkNoTransform(Transform transform, Random random) {
253         for (int i = 0; i < 100; ++i) {
254             Vector3D a = new Vector3D(random.nextDouble(),
255                                       random.nextDouble(),
256                                       random.nextDouble());
257             Vector3D b = transform.transformVector(a);
258             Assert.assertEquals(0, a.subtract(b).getNorm(), 1.0e-10);
259             Vector3D c = transform.transformPosition(a);
260             Assert.assertEquals(0, a.subtract(c).getNorm(), 1.0e-10);
261         }
262     }
