24  /*
25   * @test
26   * @bug 6832374 7052898
27   * @summary Test bad signatures get a GenericSignatureFormatError thrown.
28   * @author Joseph D. Darcy
29   * @modules java.base/sun.reflect.generics.parser
30   */
31  
32  import java.lang.reflect.*;
33  import sun.reflect.generics.parser.SignatureParser;
34  
35  public class TestBadSignatures {
36      public static void main(String[] args) {
37          String[] badSignatures = {
38              // Missing ":" after first type bound
39              "<T:Lfoo/tools/nsc/symtab/Names;Lfoo/tools/nsc/symtab/Symbols;",
40  
41              // Arrays improperly indicated for exception information
42              "<E:Ljava/lang/Exception;>(TE;[Ljava/lang/RuntimeException;)V^[TE;",
43          };
44  
45          for(String badSig : badSignatures) {
46              try {
47                  SignatureParser.make().parseMethodSig(badSig);
48                  throw new RuntimeException("Expected GenericSignatureFormatError for " +
49                                             badSig);
50              } catch(GenericSignatureFormatError gsfe) {
51                  System.out.println(gsfe.toString()); // Expected
52              }
53          }
54      }
55  }
