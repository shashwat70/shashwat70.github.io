57  public class StatementsActions extends SourceActions {
58  	public StatementsActions(Source source) {
59  		super(source);
60  	}
61  	
62  	public Node createBlock(List<Node> statements) {
63  		Block block = new Block();
64  		if (statements != null) for (Node s : statements) {
65  			if (s != null) block.rawContents().addToEnd(s);
66  		}
67  		
68  		return posify(block);
69  	}
70  	
71  	public Node createEmptyStatement() {
72  		return posify(new EmptyStatement());
73  	}
74  	
75  	public Node createLabelledStatement(List<Node> labelNames, Node statement) {
76  		Node current = statement;
77  		if (labelNames != null) {
78  			labelNames = Lists.newArrayList(labelNames);
79  			Collections.reverse(labelNames);
80  			for (Node n : labelNames) {
81  				if (n != null) {
82  					Position pos = current == null ? null : new Position(n.getPosition().getStart(), current.getPosition().getEnd());
83  					current = new LabelledStatement().astLabel(createIdentifierIfNeeded(n, currentPos())).rawStatement(current);
84  					current.setPosition(pos);
85  				}
86  			}
87  		}
88  		return current;
89  	}

91  	public Node createIfStatement(Node condition, Node statement, Node elseStatement) {
92  		return posify(new If().rawCondition(condition).rawStatement(statement).rawElseStatement(elseStatement));

95  	public Node createAssertStatement(Node assertion, Node message) {
96  		return posify(new Assert().rawAssertion(assertion).rawMessage(message));
97  	}

100 		return posify(new Switch().rawCondition(condition).rawBody(body));

108 		Default node = new Default();

118 		return posify(new DoWhile().rawCondition(condition).rawStatement(statement));

121 	public Node createStatementExpressionList(Node head, List<Node> tail) {
122 		TemporaryNode.StatementExpressionList result = new TemporaryNode.StatementExpressionList();
123 		if (head != null) result.expressions.add(head);
124 		if (tail != null) for (Node n : tail) if (n != null) result.expressions.add(n);
125 		return posify(result);
126 	}
127 	
128 	public Node createBasicFor(Node init, Node condition, Node update, Node statement) {
129 		For result = new For().rawCondition(condition).rawStatement(statement);
130 		List<Node> updates;
131 		
132 		if (update instanceof TemporaryNode.StatementExpressionList) {
133 			updates = ((TemporaryNode.StatementExpressionList)update).expressions;
134 		} else {
135 			updates = Collections.singletonList(update);
136 		}
137 		
138 		if (init instanceof TemporaryNode.StatementExpressionList) {
139 			for (Node n : ((TemporaryNode.StatementExpressionList)init).expressions) result.rawExpressionInits().addToEnd(n);
140 		} else {
141 			result.rawVariableDeclaration(init);
142 		}
143 		
144 		for (Node n : updates) if (n != null) result.rawUpdates().addToEnd(n);
145 		return posify(result);
146 	}
147 	
148 	public Node createEnhancedFor(
149 			org.parboiled.Node<Node> modifiers, Node type,
150 			org.parboiled.Node<Node> varDefEntry, Node iterable, Node statement) {
151 		
152 		VariableDefinition decl = new VariableDefinition().rawTypeReference(type).rawVariables()
153 				.addToEnd(varDefEntry.getValue());
154 		positionSpan(decl, modifiers, varDefEntry);
155 		decl.astModifiers(createModifiersIfNeeded(modifiers.getValue(), decl.getPosition().getStart()));
156 		return posify(new ForEach().rawVariable(decl).rawIterable(iterable).rawStatement(statement));
157 	}
158 	
159 	public Node createBreak(Node label) {
160 		Break b = new Break();
161 		if (label != null) b.astLabel(createIdentifierIfNeeded(label, currentPos()));
162 		return posify(b);
163 	}
164 	
165 	public Node createContinue(Node label) {
166 		Continue c = new Continue();
167 		if (label != null) c.astLabel(createIdentifierIfNeeded(label, currentPos()));
168 		return posify(c);
169 	}
170 	
171 	public Node createReturn(Node value) {
172 		return posify(new Return().rawValue(value));
173 	}
174 	
175 	public Node createThrow(Node throwable) {
176 		return posify(new Throw().rawThrowable(throwable));
177 	}
178 	
179 	public Node createSynchronizedStatement(Node lock, Node body) {
180 		return posify(new Synchronized().rawLock(lock).rawBody(body));
181 	}
182 	
183 	public Node createCatch(Node modifiers, Node type, Node varName, Node body) {
184 		VariableDefinitionEntry varNameEntry = new VariableDefinitionEntry().astName(createIdentifierIfNeeded(varName, currentPos()));
185 		if (varName != null) varNameEntry.setPosition(varName.getPosition());
186 		VariableDefinition decl = new VariableDefinition().rawTypeReference(type).rawVariables().addToEnd(
187 				varNameEntry);
188 		if (type != null && varName != null) decl.setPosition(new Position(type.getPosition().getStart(), varName.getPosition().getEnd()));
189 		if (modifiers != null) decl.astModifiers(createModifiersIfNeeded(modifiers, currentPos()));
190 		return posify(new Catch().rawExceptionDeclaration(decl).rawBody(body));
191 	}
192 	
193 	public Node createTryStatement(Node body, List<Node> catches, Node finallyBody) {
194 		Try result = new Try().rawBody(body).rawFinally(finallyBody);
195 		if (catches != null) for (Node c : catches) if (c != null) result.rawCatches().addToEnd(c);
196 		return posify(result);
197 	}
198 	
199 	public Node addLocalVariableModifiers(Node variableDefinition, Node modifiers) {
200 		if (modifiers != null && variableDefinition instanceof VariableDefinition) {
201 			((VariableDefinition)variableDefinition).astModifiers(createModifiersIfNeeded(modifiers, currentPos()));
202 		}
203 		
204 		return posify(variableDefinition);
205 	}
206 	
207 	public Node createAlternateConstructorInvocation(Node typeArguments, Node arguments) {
208 		AlternateConstructorInvocation result = new AlternateConstructorInvocation();
209 		
210 		if (typeArguments instanceof TemporaryNode.TypeArguments) {
211 			for (Node arg : ((TemporaryNode.TypeArguments)typeArguments).arguments) {
212 				result.rawConstructorTypeArguments().addToEnd(arg);
213 			}
214 		}
215 		
216 		if (arguments instanceof TemporaryNode.MethodArguments) {
217 			for (Node arg : ((TemporaryNode.MethodArguments)arguments).arguments) {
218 				result.rawArguments().addToEnd(arg);
219 			}
220 		}
221 		return posify(result);
222 	}
223 	
224 	public Node createSuperConstructorInvocation(org.parboiled.Node<Node> dot, Node qualifier, Node typeArguments, Node arguments) {
225 		SuperConstructorInvocation result = new SuperConstructorInvocation().rawQualifier(qualifier);
226 		
227 		if (typeArguments instanceof TemporaryNode.TypeArguments) {
228 			for (Node arg : ((TemporaryNode.TypeArguments)typeArguments).arguments) {
229 				result.rawConstructorTypeArguments().addToEnd(arg);
230 			}
231 		}
232 		
233 		if (arguments instanceof TemporaryNode.MethodArguments) {
234 			for (Node arg : ((TemporaryNode.MethodArguments)arguments).arguments) {
235 				result.rawArguments().addToEnd(arg);
236 			}
237 		}
238 		if (dot != null) source.registerStructure(result, dot);
239 		return posify(result);
240 	}
241 	
242 	public Node createExpressionStatement(Node expression) {
243 		return posify(new ExpressionStatement().rawExpression(expression));
244 	}
245 	
246 	public Node createVariableDeclaration(Node definition) {
247 		return posify(new VariableDeclaration().rawDefinition(definition));
248 	}
