41  //todo [anistor] command forwarding breaks the rule that we have only one originator for a command. this opens now the possibility to have two threads processing incoming remote commands for the same TX
42  /**
43   * This interceptor has two tasks:
44   * <ol>
45   *    <li>If the command's topology id is higher than the current topology id,
46   *    wait for the node to receive transaction data for the new topology id.</li>
47   *    <li>If the topology id changed during a command's execution, forward the command to the new owners.</li>
48   * </ol>
49   *
50   * Note that we don't keep track of old cache topologies (yet), so we actually forward the command to all the owners
51   * -- not just the ones added in the new topology. Transactional commands are idempotent, so this is fine.
52   *
53   * In non-transactional mode, the semantic of these tasks changes:
54   * <ol>
55   *    <li>We don't have transaction data, so the interceptor only waits for the new topology to be installed.</li>
56   *    <li>Instead of forwarding commands from any owner, we retry the command on the originator.</li>
57   * </ol>
58   *
59   * @author anistor@redhat.com
60   * @since 5.2
61   */
62  public class StateTransferInterceptor extends BaseStateTransferInterceptor {

153    @Override
154    public Object visitGetAllCommand(InvocationContext ctx, GetAllCommand command) throws Throwable {
155       if (isLocalOnly(ctx, command)) {
156          return invokeNextInterceptor(ctx, command);
157       }
158       CacheTopology beginTopology = stateTransferManager.getCacheTopology();
159       command.setConsistentHashAndAddress(beginTopology.getReadConsistentHash(),
160             transport.getAddress());
161       updateTopologyId(command);
162       try {
163          return invokeNextInterceptor(ctx, command);
164       } catch (CacheException e) {
165          Throwable ce = e;
166          while (ce instanceof RemoteException) {
167             ce = ce.getCause();
168          }
169          if (!(ce instanceof OutdatedTopologyException) && !(ce instanceof SuspectException))
170             throw e;
171 
172          // We increment the topology id so that updateTopologyIdAndWaitForTransactionData waits for the next topology.
173          // Without this, we could retry the command too fast and we could get the OutdatedTopologyException again.
174          if (trace) log.tracef("Retrying command because of topology change, current topology is %d: %s", currentTopologyId(), command);
175          int newTopologyId = Math.max(currentTopologyId(), command.getTopologyId() + 1);
176          command.setTopologyId(newTopologyId);
177          waitForTopology(newTopologyId);
178 
179          return visitGetAllCommand(ctx, command);
180       }
181    }

183    /**
184     * Special processing required for transaction commands.
185     *
186     */
187    private Object handleTxCommand(TxInvocationContext ctx, TransactionBoundaryCommand command) throws Throwable {
188       // For local commands we may not have a GlobalTransaction yet
189       Address origin = ctx.isOriginLocal() ? ctx.getOrigin() : ctx.getGlobalTransaction().getAddress();
190       if (trace) log.tracef("handleTxCommand for command %s, origin %s", command, origin);
191 
192       if (isLocalOnly(ctx, command)) {
193          return invokeNextInterceptor(ctx, command);
194       }
195       updateTopologyId(command);
196 
197       int retryTopologyId = -1;
198       Object localResult = null;
199       try {
200          localResult = invokeNextInterceptor(ctx, command);
201       } catch (OutdatedTopologyException e) {
202          // This can only happen on the originator
203          retryTopologyId = Math.max(currentTopologyId(), command.getTopologyId() + 1);
204       }
205 
206       // We need to forward the command to the new owners, if the command was asynchronous
207       boolean async = isTxCommandAsync(command);
208       if (async) {
209          stateTransferManager.forwardCommandIfNeeded(command, getAffectedKeys(ctx, command), origin, false);
210          return null;
211       }
212 
213       if (ctx.isOriginLocal()) {
214          // On the originator, we only retry if we got an OutdatedTopologyException
215          // Which could be caused either by an owner leaving or by an owner having a newer topology
216          // No need to retry just because we have a new topology on the originator, all entries were wrapped anyway
217          if (retryTopologyId > 0) {
218             // Only the originator can retry the command
219             command.setTopologyId(retryTopologyId);
220             waitForTransactionData(retryTopologyId);
221 
222             log.tracef("Retrying command %s for topology %d", command, retryTopologyId);
223             localResult = handleTxCommand(ctx, command);
224          }
225       } else {
226          if (currentTopologyId() > command.getTopologyId()) {
227             // Signal the originator to retry
228             localResult = UnsureResponse.INSTANCE;
229          }
230       }
231 
232       return localResult;
233    }
234 
235    private boolean isTxCommandAsync(TransactionBoundaryCommand command) {
236       boolean async = false;
237       if (command instanceof CommitCommand || command instanceof RollbackCommand) {
238          async = !cacheConfiguration.transaction().syncCommitPhase();
239       } else if (command instanceof PrepareCommand) {
240          async = !cacheConfiguration.clustering().cacheMode().isSynchronous();
241       }
242       return async;
243    }

245    protected Object handleWriteCommand(InvocationContext ctx, WriteCommand command) throws Throwable {
246       if (ctx.isInTxScope()) {
247          return handleTxWriteCommand(ctx, command);
248       } else {
249          return handleNonTxWriteCommand(ctx, command);
250       }
251    }

253    private Object handleTxWriteCommand(InvocationContext ctx, WriteCommand command) throws Throwable {
254       Address origin = ctx.getOrigin();
255       if (trace) log.tracef("handleTxWriteCommand for command %s, origin %s", command, origin);
256 
257       if (isLocalOnly(ctx, command)) {
258          return invokeNextInterceptor(ctx, command);
259       }
260       updateTopologyId(command);
261 
262       int retryTopologyId = -1;
263       Object localResult = null;
264       try {
265          localResult = invokeNextInterceptor(ctx, command);
266       } catch (OutdatedTopologyException e) {
267          // This can only happen on the originator
268          retryTopologyId = Math.max(currentTopologyId(), command.getTopologyId() + 1);
269       }
270 
271       if (ctx.isOriginLocal()) {
272          // On the originator, we only retry if we got an OutdatedTopologyException
273          // Which could be caused either by an owner leaving or by an owner having a newer topology
274          // No need to retry just because we have a new topology on the originator, all entries were wrapped anyway
275          if (retryTopologyId > 0) {
276             // Only the originator can retry the command
277             command.setTopologyId(retryTopologyId);
278             waitForTransactionData(retryTopologyId);
279 
280             log.tracef("Retrying command %s for topology %d", command, retryTopologyId);
281             localResult = handleTxWriteCommand(ctx, command);
282          }
283       } else {
284          if (currentTopologyId() > command.getTopologyId()) {
285             // Signal the originator to retry
286             return UnsureResponse.INSTANCE;
287          }
288       }
289 
290       // No need to forward tx write commands.
291       // Ancillary LockControlCommands will be forwarded by handleTxCommand on the target nodes.
292       return localResult;
293    }

295    /**
296     * For non-tx write commands, we retry the command locally if the topology changed, instead of forwarding to the
297     * new owners like we do for tx commands. But we only retry on the originator, and only if the command doesn't have
298     * the {@code CACHE_MODE_LOCAL} flag.
299     */
300    private Object handleNonTxWriteCommand(InvocationContext ctx, WriteCommand command) throws Throwable {
301       if (trace) log.tracef("handleNonTxWriteCommand for command %s, topology id %d", command, command.getTopologyId());
302 
303       if (isLocalOnly(ctx, command)) {
304          return invokeNextInterceptor(ctx, command);
305       }
306 
307       updateTopologyId(command);
308 
309       // Only catch OutdatedTopologyExceptions on the originator
310       if (!ctx.isOriginLocal()) {
311          return invokeNextInterceptor(ctx, command);
312       }
313 
314       int commandTopologyId = command.getTopologyId();
315       Object localResult;
316       try {
317          localResult = invokeNextInterceptor(ctx, command);
318          return localResult;
319       } catch (CacheException e) {
320          Throwable ce = e;
321          while (ce instanceof RemoteException) {
322             ce = ce.getCause();
323          }
324          if (!(ce instanceof OutdatedTopologyException) && !(ce instanceof SuspectException))
325             throw e;
326 
327          // We increment the topology id so that updateTopologyIdAndWaitForTransactionData waits for the next topology.
328          // Without this, we could retry the command too fast and we could get the OutdatedTopologyException again.
329          int currentTopologyId = currentTopologyId();
330          if (trace) log.tracef("Retrying command because of topology change, current topology is %d: %s",
331                  currentTopologyId, command);
332          int newTopologyId = Math.max(currentTopologyId, commandTopologyId + 1);
333          command.setTopologyId(newTopologyId);
334          waitForTransactionData(newTopologyId);
335 
336          command.setFlags(Flag.COMMAND_RETRY);
337          localResult = handleNonTxWriteCommand(ctx, command);
338       }
339 
340       // We retry the command every time the topology changes, either in NonTxConcurrentDistributionInterceptor or in
341       // EntryWrappingInterceptor. So we don't need to forward the command again here (without holding a lock).
342       // stateTransferManager.forwardCommandIfNeeded(command, command.getAffectedKeys(), ctx.getOrigin(), false);
343       return localResult;
344    }

346    @Override
347    protected Object handleDefault(InvocationContext ctx, VisitableCommand command) throws Throwable {
348       if (command instanceof TopologyAffectedCommand) {
349          return handleTopologyAffectedCommand(ctx, command, ctx.getOrigin());
350       } else {
351          return invokeNextInterceptor(ctx, command);
352       }
353    }
354 
355    private Object handleTopologyAffectedCommand(InvocationContext ctx, VisitableCommand command,
356                                                 Address origin) throws Throwable {
357       if (trace) log.tracef("handleTopologyAffectedCommand for command %s, origin %s", command, origin);
358 
359       if (isLocalOnly(ctx, command)) {
360          return invokeNextInterceptor(ctx, command);
361       }
362       updateTopologyId((TopologyAffectedCommand) command);
363 
364       return invokeNextInterceptor(ctx, command);
365    }
366 
367    private boolean isLocalOnly(InvocationContext ctx, VisitableCommand command) {
368       boolean cacheModeLocal = false;
369       if (command instanceof FlagAffectedCommand) {
370          cacheModeLocal = ((FlagAffectedCommand)command).hasFlag(Flag.CACHE_MODE_LOCAL);
371       }
372       return cacheModeLocal;
373    }
374 
375    @SuppressWarnings("unchecked")
376    private Set<Object> getAffectedKeys(InvocationContext ctx, VisitableCommand command) {
377       Set<Object> affectedKeys = null;
378       try {
379          affectedKeys = (Set<Object>) command.acceptVisitor(ctx, affectedKeysVisitor);
380       } catch (Throwable throwable) {
381          // impossible to reach this
382       }
383       if (affectedKeys == null) {
384          affectedKeys = InfinispanCollections.emptySet();
385       }
386       return affectedKeys;
387    }
