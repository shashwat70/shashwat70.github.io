1   package org.infinispan.tx;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.configuration.cache.CacheMode;
5   import org.infinispan.configuration.cache.ConfigurationBuilder;
6   import org.infinispan.configuration.global.GlobalConfiguration;
7   import org.infinispan.configuration.global.GlobalConfigurationBuilder;
8   import org.infinispan.manager.DefaultCacheManager;
9   import org.infinispan.test.AbstractCacheTest;
10  import org.infinispan.test.TestingUtil;
11  import org.infinispan.test.fwk.TestCacheManagerFactory;
12  import org.infinispan.transaction.TransactionMode;
13  import org.testng.annotations.AfterMethod;
14  import org.testng.annotations.Test;
15  
16  import static org.junit.Assert.assertEquals;
17  import static org.junit.Assert.assertFalse;
18  
19  @Test (groups = "functional", testName = "tx.DefaultEnlistmentModeTest")
20  public class DefaultEnlistmentModeTest extends AbstractCacheTest {
21  
22     private DefaultCacheManager dcm;
23  
24     @AfterMethod
25     protected void destroyCacheManager() {
26        TestingUtil.killCacheManagers(dcm);
27     }
28  
29     public void testDefaultEnlistment() {
30        ConfigurationBuilder builder = getLocalBuilder();
31        builder.transaction().transactionMode(TransactionMode.TRANSACTIONAL);
32        dcm = new DefaultCacheManager(getGlobalConfig(), builder.build());
33        Cache<Object,Object> cache = dcm.getCache();
34        assertFalse(cache.getCacheConfiguration().transaction().useSynchronization());
35        assertFalse(cache.getCacheConfiguration().transaction().recovery().enabled());
36        cache.put("k", "v");
37        assertEquals("v", cache.get("k"));
38     }
39  
40     public void testXAEnlistment() {
41        ConfigurationBuilder builder = getLocalBuilder();
42        builder.transaction()
43              .transactionMode(TransactionMode.TRANSACTIONAL)
44              .useSynchronization(false);
45        dcm = new DefaultCacheManager(getGlobalConfig(), builder.build());
46        Cache<Object,Object> cache = dcm.getCache();
47        assertFalse(cache.getCacheConfiguration().transaction().useSynchronization());
48        assertFalse(cache.getCacheConfiguration().transaction().recovery().enabled());
49        cache.put("k", "v");
50        assertEquals("v", cache.get("k"));
51     }
52  
53     public void testXAEnlistmentNoRecovery() {
54        ConfigurationBuilder builder = getLocalBuilder();
55        builder.transaction()
56              .transactionMode(TransactionMode.TRANSACTIONAL)
57              .useSynchronization(false)
58              .recovery().disable();
59        dcm = new DefaultCacheManager(getGlobalConfig(), builder.build());
60        Cache<Object,Object> cache = dcm.getCache();
61        assertFalse(cache.getCacheConfiguration().transaction().useSynchronization());
62        assertFalse(cache.getCacheConfiguration().transaction().recovery().enabled());
63     }
64  
65     private ConfigurationBuilder getLocalBuilder() {
66        ConfigurationBuilder builder = TestCacheManagerFactory.getDefaultCacheConfiguration(true);
67        builder.clustering().cacheMode(CacheMode.LOCAL);
68        return builder;
69     }
70  
71     private GlobalConfiguration getGlobalConfig() {
72        return new GlobalConfigurationBuilder().globalJmxStatistics().allowDuplicateDomains(true).build();
73     }
74  }
