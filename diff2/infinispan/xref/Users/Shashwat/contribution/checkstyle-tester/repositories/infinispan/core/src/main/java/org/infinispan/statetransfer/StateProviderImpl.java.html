106    public void onTopologyUpdate(CacheTopology cacheTopology, boolean isRebalance) {
107       // Cancel outbound state transfers for destinations that are no longer members in new topology
108       // If the rebalance was cancelled, stop every outbound transfer. This will prevent "leaking" transfers
109       // from one rebalance to the next.
110       boolean stateTransferInProgress = cacheTopology.getPendingCH() != null;
111       Set<Address> members = new HashSet<Address>(cacheTopology.getWriteConsistentHash().getMembers());
112       synchronized (transfersByDestination) {
113          for (Iterator<Address> it = transfersByDestination.keySet().iterator(); it.hasNext(); ) {
114             Address destination = it.next();
115             if (!members.contains(destination)) {
116                List<OutboundTransferTask> transfers = transfersByDestination.get(destination);
117                it.remove();
118                for (OutboundTransferTask outboundTransfer : transfers) {
119                   outboundTransfer.cancel();
120                }
121             }
122          }
123       }
124 
125       //todo [anistor] must cancel transfers for all segments that we no longer own
126    }

133    @Stop(priority = 20)
134    @Override
135    public void stop() {
136       if (trace) {
137          log.tracef("Shutting down StateProvider of cache %s on node %s", cacheName, rpcManager.getAddress());
138       }
139       // cancel all outbound transfers
140       try {
141          synchronized (transfersByDestination) {
142             for (Iterator<List<OutboundTransferTask>> it = transfersByDestination.values().iterator(); it.hasNext(); ) {
143                List<OutboundTransferTask> transfers = it.next();
144                it.remove();
145                for (OutboundTransferTask outboundTransfer : transfers) {
146                   outboundTransfer.cancel();
147                }
148             }
149          }
150       } catch (Throwable t) {
151          log.errorf(t, "Failed to stop StateProvider of cache %s on node %s", cacheName, rpcManager.getAddress());
152       }
153    }

155    public List<TransactionInfo> getTransactionsForSegments(Address destination, int requestTopologyId, Set<Integer> segments) throws InterruptedException {
156       if (trace) {
157          log.tracef("Received request for transactions from node %s for segments %s of cache %s with topology id %d", destination, segments, cacheName, requestTopologyId);
158       }
159 
160       final CacheTopology cacheTopology = getCacheTopology(requestTopologyId, destination, true);
161       final ConsistentHash readCh = cacheTopology.getReadConsistentHash();
162 
163       Set<Integer> ownedSegments = readCh.getSegmentsForOwner(rpcManager.getAddress());
164       if (!ownedSegments.containsAll(segments)) {
165          segments.removeAll(ownedSegments);
166          throw new IllegalArgumentException("Segments " + segments + " are not owned by " + rpcManager.getAddress());
167       }
168 
169       List<TransactionInfo> transactions = new ArrayList<TransactionInfo>();
170       //we migrate locks only if the cache is transactional and distributed
171       if (configuration.transaction().transactionMode().isTransactional()) {
172          collectTransactionsToTransfer(destination, transactions, transactionTable.getRemoteTransactions(), segments, cacheTopology);
173          collectTransactionsToTransfer(destination, transactions, transactionTable.getLocalTransactions(), segments, cacheTopology);
174          if (trace) {
175             log.tracef("Found %d transaction(s) to transfer", transactions.size());
176          }
177       }
178       return transactions;
179    }

186    private CacheTopology getCacheTopology(int requestTopologyId, Address destination, boolean isReqForTransactions) throws InterruptedException {
187       CacheTopology cacheTopology = stateConsumer.getCacheTopology();
188       int currentTopologyId = cacheTopology != null ? cacheTopology.getTopologyId() : -1;
189       if (requestTopologyId < currentTopologyId) {
190          if (isReqForTransactions)
191             log.debugf("Transactions were requested by node %s with topology %d, older than the local topology (%d)",
192                   destination, requestTopologyId, currentTopologyId);
193          else
194             log.debugf("Segments were requested by node %s with topology %d, older than the local topology (%d)",
195                   destination, requestTopologyId, currentTopologyId);
196       } else if (requestTopologyId > currentTopologyId) {
197          if (trace) {
198             log.tracef("%s were requested by node %s with topology %d, greater than the local " +
199                   "topology (%d). Waiting for topology %d to be installed locally.", isReqForTransactions ? "Transactions" : "Segments", destination,
200                   requestTopologyId, currentTopologyId, requestTopologyId);
201          }
202          stateTransferLock.waitForTopology(requestTopologyId, timeout, TimeUnit.MILLISECONDS);
203          cacheTopology = stateConsumer.getCacheTopology();
204       }
205       return cacheTopology;
206    }

208    private void collectTransactionsToTransfer(Address destination,
209                                               List<TransactionInfo> transactionsToTransfer,
210                                               Collection<? extends CacheTransaction> transactions,
211                                               Set<Integer> segments, CacheTopology cacheTopology) {
212       int topologyId = cacheTopology.getTopologyId();
213       List<Address> members = cacheTopology.getMembers();
214       ConsistentHash readCh = cacheTopology.getReadConsistentHash();
215 
216       // no need to filter out state transfer generated transactions because there should not be any such transactions running for any of the requested segments
217       for (CacheTransaction tx : transactions) {
218          // Skip transactions whose originators left. The topology id check is needed for joiners.
219          // Also skip transactions that originates after state transfer starts.
220          if (tx.getTopologyId() == topologyId || !members.contains(tx.getGlobalTransaction().getAddress())) {
221             if (trace) log.tracef("Skipping transaction %s as it was started in the current topology or by a leaver", tx);
222             continue;
223          }
224 
225          // transfer only locked keys that belong to requested segments
226          Set<Object> filteredLockedKeys = new HashSet<Object>();
227          Set<Object> lockedKeys = tx.getLockedKeys();
228          synchronized (lockedKeys) {
229             for (Object key : lockedKeys) {
230                if (segments.contains(readCh.getSegment(key))) {
231                   filteredLockedKeys.add(key);
232                }
233             }
234          }
235          Set<Object> backupLockedKeys = tx.getBackupLockedKeys();
236          synchronized (backupLockedKeys) {
237             for (Object key : backupLockedKeys) {
238                if (segments.contains(readCh.getSegment(key))) {
239                   filteredLockedKeys.add(key);
240                }
241             }
242          }
243          if (filteredLockedKeys.isEmpty()) {
244             if (trace) log.tracef("Skipping transaction %s because the state requestor %s doesn't own any key",
245                     tx, destination);
246             continue;
247          }
248          if (trace) log.tracef("Sending transaction %s to new owner %s", tx, destination);
249          List<WriteCommand> txModifications = tx.getModifications();
250          WriteCommand[] modifications = null;
251          if (!txModifications.isEmpty()) {
252             modifications = txModifications.toArray(new WriteCommand[txModifications.size()]);
253          }
254 
255          // If a key affected by a local transaction has a new owner, we must add the new owner to the transaction's
256          // affected nodes set, so that the it receives the commit/rollback command. See ISPN-3389.
257          if(tx instanceof LocalTransaction) {
258             LocalTransaction localTx = (LocalTransaction) tx;
259             localTx.locksAcquired(Collections.singleton(destination));
260             if (trace) log.tracef("Adding affected node %s to transferred transaction %s (keys %s)", destination,
261                   tx.getGlobalTransaction(), filteredLockedKeys);
262          }
263          transactionsToTransfer.add(new TransactionInfo(tx.getGlobalTransaction(), tx.getTopologyId(),
264                modifications, filteredLockedKeys));
265       }
266    }

268    @Override
269    public void startOutboundTransfer(Address destination, int requestTopologyId, Set<Integer> segments)
270          throws InterruptedException {
271       if (trace) {
272          log.tracef("Starting outbound transfer of segments %s to node %s with topology id %d for cache %s", segments,
273                destination, requestTopologyId, cacheName);
274       }
275 
276       final CacheTopology cacheTopology = getCacheTopology(requestTopologyId, destination, false);
277 
278       // the destination node must already have an InboundTransferTask waiting for these segments
279       OutboundTransferTask outboundTransfer = new OutboundTransferTask(destination, segments, chunkSize, requestTopologyId,
280             cacheTopology.getReadConsistentHash(), this, dataContainer, persistenceManager, rpcManager, commandsFactory, entryFactory, timeout, cacheName);
281       addTransfer(outboundTransfer);
282       outboundTransfer.execute(executorService);
283    }

285    private void addTransfer(OutboundTransferTask transferTask) {
286       if (trace) {
287          log.tracef("Adding outbound transfer of segments %s to %s", transferTask.getSegments(), transferTask.getDestination());
288       }
289       synchronized (transfersByDestination) {
290          List<OutboundTransferTask> transfers = transfersByDestination.get(transferTask.getDestination());
291          if (transfers == null) {
292             transfers = new ArrayList<OutboundTransferTask>();
293             transfersByDestination.put(transferTask.getDestination(), transfers);
294          }
295          transfers.add(transferTask);
296       }
297    }

299    @Override
300    public void cancelOutboundTransfer(Address destination, int topologyId, Set<Integer> segments) {
301       if (trace) {
302          log.tracef("Cancelling outbound transfer of segments %s to node %s with topology id %d for cache %s", segments, destination, topologyId, cacheName);
303       }
304       // get the outbound transfers for this address and given segments and cancel the transfers
305       synchronized (transfersByDestination) {
306          List<OutboundTransferTask> transferTasks = transfersByDestination.get(destination);
307          if (transferTasks != null) {
308             // get an array copy of the collection to avoid ConcurrentModificationException if the entire task gets cancelled and removeTransfer(transferTask) is called
309             OutboundTransferTask[] taskListCopy = transferTasks.toArray(new OutboundTransferTask[transferTasks.size()]);
310             for (OutboundTransferTask transferTask : taskListCopy) {
311                if (transferTask.getTopologyId() == topologyId) {
312                   transferTask.cancelSegments(segments); //this can potentially result in a call to removeTransfer(transferTask)
313                }
314             }
315          }
316       }
317    }
318 
319    private void removeTransfer(OutboundTransferTask transferTask) {
320       synchronized (transfersByDestination) {
321          List<OutboundTransferTask> transferTasks = transfersByDestination.get(transferTask.getDestination());
322          if (transferTasks != null) {
323             transferTasks.remove(transferTask);
324             if (transferTasks.isEmpty()) {
325                transfersByDestination.remove(transferTask.getDestination());
326             }
327          }
328       }
329    }

331    void onTaskCompletion(OutboundTransferTask transferTask) {
332       if (trace) {
333          log.tracef("Removing %s outbound transfer of segments %s to %s for cache %s",
334                transferTask.isCancelled() ? "cancelled" : "completed", transferTask.getSegments(), transferTask.getDestination(), cacheName);
335       }
336 
337       removeTransfer(transferTask);
338    }
