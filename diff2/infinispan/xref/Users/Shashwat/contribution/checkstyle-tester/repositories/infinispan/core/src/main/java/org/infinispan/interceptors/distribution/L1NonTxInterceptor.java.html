65     /**
66      *  This map holds all the current write synchronizers registered for a given key.  This map is only added to when an
67      * operation is invoked that would cause a remote get to occur (which is controlled by whether or not the
68      * {@link L1NonTxInterceptor#skipL1Lookup(org.infinispan.commands.LocalFlagAffectedCommand, Object)} method returns
69      * true.  This map <b>MUST</b> have the value inserted removed in a finally block after the remote get is done to
70      * prevent reference leaks.
71      * <p>
72      * Having a value in this map allows for other concurrent operations that require a remote get to not have to
73      * actually perform a remote get as the first thread is doing this.  So in this case any subsequent operations
74      * wanting the remote value can just call the
75      * {@link org.infinispan.interceptors.distribution.L1WriteSynchronizer#get()} method or one of it's overridden
76      * methods.  Note the way to tell if another thread is performing the remote get is to use the
77      * {@link ConcurrentMap#putIfAbsent(Object, Object)} method and check if the return value is null or not.
78      * <p>
79      * Having a value in this map allows for a concurrent write or L1 invalidation to try to stop the synchronizer from
80      * updating the L1 value by invoking it's
81      * {@link org.infinispan.interceptors.distribution.L1WriteSynchronizer#trySkipL1Update()} method.  If this method
82      * returns false, then the write or L1 invalidation <b>MUST</b> wait for the synchronizer to complete before
83      * continuing to ensure it is able to remove the newly cached L1 value as it is now invalid.  This waiting should be
84      * done by calling {@link org.infinispan.interceptors.distribution.L1WriteSynchronizer#get()} method or one of it's
85      * overridden methods.  Failure to wait for the update to occur could cause a L1 data inconsistency as the
86      * invalidation may not invalidate the new value.
87      */
88     private final ConcurrentMap<Object, L1WriteSynchronizer> concurrentWrites = CollectionFactory.makeConcurrentMap();

99        this.stateTransferLock = stateTransferLock;
100       this.commandsFactory = commandsFactory;
101    }

117    private Object visitDataReadCommand(InvocationContext ctx, AbstractDataCommand command, boolean isEntry) throws Throwable {
118       return performCommandWithL1WriteIfAble(ctx, command, isEntry, false, true);
119    }
120 
121    protected Object performCommandWithL1WriteIfAble(InvocationContext ctx, DataCommand command, boolean isEntry,
122                                                 boolean shouldAlwaysRunNextInterceptor, boolean registerL1) throws Throwable {
123       Object returnValue;
124       if (ctx.isOriginLocal()) {
125          Object key = command.getKey();
126          // If the command isn't going to return a remote value - just pass it down the interceptor chain
127          if (skipL1Lookup(command, key)) {
128             returnValue = invokeNextInterceptor(ctx, command);
129          } else {
130             returnValue = performL1Lookup(ctx, shouldAlwaysRunNextInterceptor, key, command, isEntry);
131          }
132       } else {
133          // If this is a remote command, and we found a value in our cache
134          // we store it so that we can later invalidate it
135          if (registerL1) {
136             l1Manager.addRequestor(command.getKey(), ctx.getOrigin());
137          }
138          returnValue = invokeNextInterceptor(ctx, command);
139       }
140       return returnValue;
141    }
142 
143    protected Object performL1Lookup(InvocationContext ctx, boolean runInterceptorOnConflict, Object key,
144                                      DataCommand command, boolean isEntry) throws Throwable {
145       // Most times the putIfAbsent will be successful, so not doing a get first
146       L1WriteSynchronizer l1WriteSync = new L1WriteSynchronizer(dataContainer, l1Lifespan, stateTransferLock,
147                                                                 cdl);
148       L1WriteSynchronizer presentSync = concurrentWrites.putIfAbsent(key, l1WriteSync);
149 
150       // If the sync was null that means we are the first to register for the given key.  If not that means there is
151       // a concurrent request that also wants to do a remote get for the key.  If there was another thread requesting
152       // the key we should wait until they get the value instead of doing another remote get.
153       if (presentSync == null) {
154          try {
155             // Note this is the same synchronizer we just created that is registered with the L1Manager
156             l1Manager.registerL1WriteSynchronizer(key, l1WriteSync);
157             Object returnValue;
158             try {
159                returnValue = invokeNextInterceptor(ctx, command);
160             }
161             finally {
162                l1Manager.unregisterL1WriteSynchronizer(key, l1WriteSync);
163             }
164             return returnValue;
165          }
166          catch (Throwable t) {
167             l1WriteSync.retrievalEncounteredException(t);
168             throw t;
169          }
170          finally {
171             concurrentWrites.remove(key);
172          }
173       } else {
174          if (trace) {
175             log.tracef("Found current request for key %s, waiting for their invocation's response", key);
176          }
177          try {
178             Object returnValue;
179             try {
180                returnValue = presentSync.get(replicationTimeout, TimeUnit.MILLISECONDS);
181                // Write commands could have different values so we always want to run them after we know the remote
182                // value is retrieved.  Gets however only need the return value so we don't need to run the additional
183                // interceptors
184                if (runInterceptorOnConflict) {
185                   returnValue = invokeNextInterceptor(ctx, command);
186                } else if (!isEntry && returnValue instanceof InternalCacheEntry) {
187                   returnValue = ((InternalCacheEntry)returnValue).getValue();
188                }
189             } catch (TimeoutException e) {
190                // This should never be required since the status is always set in a try catch above - but IBM doesn't...
191                log.warnf("Synchronizer didn't return in %s milliseconds - running command normally!", replicationTimeout);
192                // Always run next interceptor if a timeout occurs
193                returnValue = invokeNextInterceptor(ctx, command);
194             }
195             return returnValue;
196          }
197          catch (ExecutionException e) {
198             throw e.getCause();
199          }
200       }
201    }

203    protected boolean skipL1Lookup(LocalFlagAffectedCommand command, Object key) {
204       return command.hasFlag(Flag.CACHE_MODE_LOCAL) || command.hasFlag(Flag.SKIP_REMOTE_LOOKUP)
205             || command.hasFlag(Flag.IGNORE_RETURN_VALUES) || cdl.localNodeIsOwner(key)
206             || dataContainer.containsKey(key);
207    }

224    @Override
225    public Object visitPutMapCommand(InvocationContext ctx, PutMapCommand command) throws Throwable {
226       Future<Object> invalidationFuture = null;
227       Set<Object> keys = command.getMap().keySet();
228       Set<Object> toInvalidate = new HashSet<Object>(keys.size());
229       for (Object k : keys) {
230          if (cdl.localNodeIsOwner(k)) {
231             toInvalidate.add(k);
232          }
233       }
234       if (!toInvalidate.isEmpty()) {
235          invalidationFuture = l1Manager.flushCache(toInvalidate, ctx.getOrigin(), true);
236       }
237 
238       Object result = invokeNextInterceptor(ctx, command);
239       processInvalidationResult(ctx, command, invalidationFuture);
240       //we also need to remove from L1 the keys that are not ours
241       for (Object o : command.getAffectedKeys()) {
242          if (!cdl.localNodeIsOwner(o)) {
243             removeFromL1(ctx, o);
244          }
245       }
246       return result;
247    }

249    @Override
250    public Object visitInvalidateL1Command(InvocationContext ctx, InvalidateL1Command invalidateL1Command) throws Throwable {
251       for (Object key : invalidateL1Command.getKeys()) {
252          abortL1UpdateOrWait(key);
253          // If our invalidation was sent when the value wasn't yet cached but is still being requested the context
254          // may not have the value - if so we need to add it then now that we know we waited for the get response
255          // to complete
256          if (ctx.lookupEntry(key) == null) {
257             entryFactory.wrapEntryForRemove(ctx, key, true, true, false);
258          }
259       }
260       return super.visitInvalidateL1Command(ctx, invalidateL1Command);
261    }
262 
263    private void abortL1UpdateOrWait(Object key) throws InterruptedException {
264       L1WriteSynchronizer sync = concurrentWrites.remove(key);
265       if (sync != null) {
266          if (sync.trySkipL1Update()) {
267             if (trace) {
268                log.tracef("Aborted possible L1 update due to concurrent invalidation for key %s", key);
269             }
270          } else {
271             if (trace) {
272                log.tracef("L1 invalidation found a pending update for key %s - need to block until finished", key);
273             }
274             // We have to wait for the pending L1 update to complete before we can properly invalidate.  Any additional
275             // gets that come in after this invalidation we ignore for now.
276             boolean success;
277             try {
278                sync.get();
279                success = true;
280             }
281             catch (ExecutionException e) {
282                // We don't care what the L1 update exception was
283                success = false;
284             }
285             if (trace) {
286                log.tracef("Pending L1 update completed successfully: %b - L1 invalidation can occur for key %s", success, key);
287             }
288          }
289       }
290    }

292    private Object handleDataWriteCommand(InvocationContext ctx, DataWriteCommand command, boolean assumeOriginKeptEntryInL1) throws Throwable {
293       if (command.hasFlag(Flag.CACHE_MODE_LOCAL)) {
294          if (trace) {
295             log.tracef("local mode forced, suppressing L1 calls.");
296          }
297          return invokeNextInterceptor(ctx, command);
298       }
299       Future<Object> l1InvalidationFuture = invalidateL1(ctx, command, assumeOriginKeptEntryInL1);
300       Object returnValue = invokeNextInterceptor(ctx, command);
301       processInvalidationResult(ctx, command, l1InvalidationFuture);
302       removeFromLocalL1(ctx, command);
303       return returnValue;
304    }
305 
306    private void removeFromLocalL1(InvocationContext ctx, DataWriteCommand command) throws Throwable {
307       if (ctx.isOriginLocal() && !cdl.localNodeIsOwner(command.getKey())) {
308          removeFromL1(ctx, command.getKey());
309       } else if (trace) {
310          log.trace("Allowing entry to commit as local node is owner");
311       }
312    }
313 
314    private void removeFromL1(InvocationContext ctx, Object key) throws Throwable {
315       if (trace) {
316          log.tracef("Removing entry from L1 for key %s", key);
317       }
318       abortL1UpdateOrWait(key);
319       ctx.removeLookedUpEntry(key);
320       entryFactory.wrapEntryForRemove(ctx, key, true, true, false);
321 
322       InvalidateCommand command = commandsFactory.buildInvalidateFromL1Command(null, Collections.singleton(key));
323       invokeNextInterceptor(ctx, command);
324    }
325 
326    private void processInvalidationResult(InvocationContext ctx, FlagAffectedCommand command, Future<Object> l1InvalidationFuture) throws InterruptedException, ExecutionException {
327       if (l1InvalidationFuture != null) {
328          if (isSynchronous(command)) {
329             l1InvalidationFuture.get();
330          }
331       }
332    }
333 
334    private Future<Object> invalidateL1(InvocationContext ctx, DataWriteCommand command, boolean assumeOriginKeptEntryInL1) {
335       Future<Object> l1InvalidationFuture = null;
336       if (cdl.localNodeIsOwner(command.getKey())) {
337          l1InvalidationFuture = l1Manager.flushCache(Collections.singletonList(command.getKey()), ctx.getOrigin(), assumeOriginKeptEntryInL1);
338       } else if (trace) {
339          log.tracef("Not invalidating key '%s' as local node(%s) is not owner", command.getKey(), rpcManager.getAddress());
340       }
341       return l1InvalidationFuture;
342    }
