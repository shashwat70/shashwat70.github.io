1   package org.infinispan.client.hotrod;

3   import org.infinispan.client.hotrod.test.MultiHotRodServersTest;
4   import org.infinispan.configuration.cache.CacheMode;
5   import org.infinispan.configuration.cache.ConfigurationBuilder;
6   import org.testng.annotations.Test;
7   
8   import java.util.HashMap;
9   import java.util.Map;
10  
11  import static org.testng.AssertJUnit.assertEquals;
12  
13  @Test(testName = "client.hotrod.GetAllCompatDistTest", groups = "functional")
14  public class GetAllCompatDistTest extends MultiHotRodServersTest {
15  
16     @Override
17     protected void createCacheManagers() throws Throwable {
18        createHotRodServers(2, getCacheConfiguration());
19     }
20  
21     private ConfigurationBuilder getCacheConfiguration() {
22        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
23        builder.clustering().hash().numOwners(1).compatibility().enabled(true);
24        return builder;
25     }
26  
27     public void testGetAllWithCompatibility() {
28        RemoteCache<String, String> cache = client(0).getCache();
29        HashMap<String, String> cachedValues = new HashMap<>();
30        for(int i=0; i<100; i++){
31           String key = String.format("key-%d", i);
32           String value = String.format("value-%d", i);
33           cache.put(key, value);
34           cachedValues.put(key, value);
35        }
36  
37        Map<String, String> values = cache.getAll(cachedValues.keySet());
38        assertEquals(cachedValues.size(), values.size());
39        for(String key : values.keySet()){
40           assertEquals(cachedValues.get(key), values.get(key));
41        }
42     }
