137 	@Test
138 	public void testIsKnownAtCompileTime() throws Exception {
139 		// contract: one can ask whether an expression is known at compile time
140 		Launcher launcher = new Launcher();
141 		CtExpression el = launcher.getFactory().Code().createCodeSnippetExpression("(0+1)*3").compile();
142 		assertTrue(EvalHelper.isKnownAtCompileTime(el));
143 
144 		CtClass<?> type = build("spoon.test.eval.testclasses", "ToEvaluate");
145 		assertEquals("ToEvaluate", type.getSimpleName());
146 		CtExpression<?> foo = ((CtReturn)type.getMethodsByName("foo").get(0).getBody().getStatement(0)).getReturnedExpression();
147 		assertFalse(EvalHelper.isKnownAtCompileTime(foo));
148 
149 		CtExpression<?> foo2 = ((CtReturn)type.getMethodsByName("foo2").get(0).getBody().getStatement(0)).getReturnedExpression();
150 		assertTrue(EvalHelper.isKnownAtCompileTime(foo2));
151 
152 		CtExpression<?> foo3 = ((CtReturn)type.getMethodsByName("foo3").get(0).getBody().getStatement(0)).getReturnedExpression();
153 		assertTrue(EvalHelper.isKnownAtCompileTime(foo3));
154 
155 	}

157 	@Test
158 	public void testVisitorPartialEvaluator_binary() {
159 		Launcher launcher = new Launcher();
160 
161 		{ // binary operator
162 			CtCodeElement el = launcher.getFactory().Code().createCodeSnippetExpression("0+1").compile();
163 			VisitorPartialEvaluator eval = new VisitorPartialEvaluator();
164 			CtElement elnew = eval.evaluate(el);
165 			assertEquals("1", elnew.toString());
166 		}
167 
168 		{ // binary operator
169 			CtCodeElement el = launcher.getFactory().Code().createCodeSnippetExpression("(0+1)*3").compile();
170 			VisitorPartialEvaluator eval = new VisitorPartialEvaluator();
171 			CtElement elnew = eval.evaluate(el);
172 			assertEquals("3", elnew.toString());
173 		}
174 
175 		{ // binary operator
176 			CtCodeElement el = launcher.getFactory().Code().createCodeSnippetExpression("(0+1)*3>0").compile();
177 			VisitorPartialEvaluator eval = new VisitorPartialEvaluator();
178 			CtElement elnew = eval.evaluate(el);
179 			assertEquals("true", elnew.toString());
180 		}
181 
182 		{ // binary operator
183 			CtCodeElement el = launcher.getFactory().Code().createCodeSnippetExpression("(0+3-1)*3<=0").compile();
184 			VisitorPartialEvaluator eval = new VisitorPartialEvaluator();
185 			CtElement elnew = eval.evaluate(el);
186 			assertEquals("false", elnew.toString());
187 		}

216 	@Test
217 	public void testconvertElementToRuntimeObject() {
218 		// contract: getCorrespondingRuntimeObject works well for all kinds of expression
219 		Launcher launcher = new Launcher();
220 		launcher.addInputResource("src/test/java/spoon/test/eval/testclasses/Foo.java");
221 		launcher.buildModel();
222 		CtType<?> foo = launcher.getFactory().Type().get((Class<?>) Foo.class);
223 
224 		// also works for non-enum fields with partial evaluation embedded in convertElementToRuntimeObject
225 		assertEquals(false, EvalHelper.convertElementToRuntimeObject(foo.getField("b6").getDefaultExpression()));
226 
227 		// impossible with no partial evaluation
228 		try {
229 			assertEquals(false, EvalHelper.getCorrespondingRuntimeObject(foo.getField("b6").getDefaultExpression()));
230 			fail();
231 		} catch (SpoonException expected) {}
232 
233 		// also works for static runtime fields
234 		assertEquals(Integer.MAX_VALUE, EvalHelper.convertElementToRuntimeObject(foo.getField("i1").getDefaultExpression()));
235 		assertEquals(Integer.MAX_VALUE, EvalHelper.getCorrespondingRuntimeObject(foo.getField("i1").getDefaultExpression()));
236 		assertEquals(File.pathSeparator, EvalHelper.convertElementToRuntimeObject(foo.getField("str1").getDefaultExpression()));
237 		assertEquals(File.pathSeparator, EvalHelper.getCorrespondingRuntimeObject(foo.getField("str1").getDefaultExpression()));
