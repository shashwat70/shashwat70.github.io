84  	@Test
85  	public void testCheckModelConsistency() {
86  		//1) search for all variable declarations with name isTestFieldName(name)==true
87  		//2) check that each of them is using different identification value
88  		class Context {
89  			Map<Integer, CtElement> unique = new HashMap<>();
90  			int maxKey = 0;
91  			void checkKey(int key, CtElement ele) {
92  				CtElement ambiguous = unique.put(key, ele);
93  				if(ambiguous!=null) {
94  					fail("Two variables [" + ambiguous.toString() + " in " + getParentMethodName(ambiguous) + "," + ele.toString() + " in " + getParentMethodName(ele) + "] has same value");
95  				}
96  				maxKey = Math.max(maxKey, key);
97  			}
98  		}
99  		Context context = new Context();
100 
101 		modelClass.filterChildren((CtElement e)->{
102 			if (e instanceof CtVariable) {
103 				CtVariable<?> var = (CtVariable<?>) e;
104 				if(isTestFieldName(var.getSimpleName())==false) {
105 					return false;
106 				}
107 				//check only these variables whose name is isTestFieldName(name)==true
108 				Integer val = getLiteralValue(var);
109 				context.checkKey(val, var);
110 			}
111 			return false;
112 		}).list();
113 		assertFalse(context.unique.isEmpty());
114 		assertEquals("Only these keys were found: "+context.unique.keySet(), context.maxKey, context.unique.size());
115 		assertEquals("AllLocalVars#maxValue must be equal to maximum value number ", (int)getLiteralValue((CtVariable)modelClass.filterChildren(new NamedElementFilter<>(CtVariable.class,"maxValue")).first()), context.maxKey);
116 	}

186 	@Test
187 	public void testVariableScopeFunction() {
188 		//visits all the CtVariable elements whose name is "field" and search for all elements in their scopes
189 		//Comparing with the result found by basic functions
190 		List list = modelClass.filterChildren((CtVariable<?> var)->{
191 			if("field".equals(var.getSimpleName())) {
192 				if(var instanceof CtField) {
193 					//field scope is not supported
194 					return false;
195 				}
196 				CtElement[] real = var.map(new VariableScopeFunction()).list().toArray(new CtElement[0]);
197 				if(var instanceof CtLocalVariable) {
198 					assertArrayEquals(var.map(new LocalVariableScopeFunction()).list().toArray(new CtElement[0]), real);
199 				} else if(var instanceof CtField) {
200 					assertArrayEquals(var.map(new FieldScopeFunction()).list().toArray(new CtElement[0]), real);
201 				} else if(var instanceof CtParameter) {
202 					assertArrayEquals(var.map(new ParameterScopeFunction()).list().toArray(new CtElement[0]), real);
203 				} else if(var instanceof CtCatchVariable) {
204 					assertArrayEquals(var.map(new CatchVariableScopeFunction()).list().toArray(new CtElement[0]), real);
205 				} else {
206 					fail("Unexpected variable of type "+var.getClass().getName());
207 				}
208 				return true;
209 			}
210 			return false;
211 		}).list();
212 		assertFalse(list.isEmpty());
213 	}

227 	private void checkVariableAccess(CtVariable<?> var, int value, CtConsumableFunction<?> query) {
228 		class Context {
229 			int realCount = 0;
230 			int expectedCount = 0;
231 			Set<String> unique = new HashSet<>();
232 		}
233 		try {
234 			Context context = new Context();
235 			//use provided reference returning function to found all occurrences of the variable
236 			var.map(query).forEach((CtVariableReference<?> fr)->{
237 				//check that all the found field references has expected right hand expression
238 				assertEquals(value, getVariableReferenceValue(fr));
239 				//count number of found references
240 				context.realCount++;
241 			});
242 			//use filterChildren to scan all field references in model and count the number of field references which has same value => expectedCount
243 			modelClass.filterChildren(new TypeFilter<>(CtVariableReference.class))
244 			.forEach((CtVariableReference varRef)->{
245 				if(isTestFieldName(varRef.getSimpleName())==false) {
246 					return;
247 				}
248 				int refValue = getVariableReferenceValue(varRef);
249 				if(refValue<0) {
250 					fail("Variable reference has no value:\n"+varRef);
251 				}
252 				if(refValue==value) {
253 					context.expectedCount++;
254 				}
255 			});
256 			//check that both scans found same number of references
257 			assertEquals("Number of references to field="+value+" does not match", context.expectedCount, context.realCount);
258 			
259 		} catch (Throwable e) {
260 			e.printStackTrace();
261 			throw new AssertionError("Test failed on " + getParentMethodName(var), e);
262 		}
263 	}

265 	private String getParentMethodName(CtElement ele) {
266 		CtMethod parentMethod = ele.getParent(CtMethod.class);
267 		CtMethod m;
268 		while(parentMethod!=null && (m=parentMethod.getParent(CtMethod.class))!=null) {
269 			parentMethod = m;
270 		}
271 		if(parentMethod!=null) {
272 			return parentMethod.getParent(CtType.class).getSimpleName()+"#"+parentMethod.getSimpleName();
273 		} else {
274 			return ele.getParent(CtType.class).getSimpleName()+"#annonymous block";
275 		}
276 	}

278 	private int getVariableReferenceValue(CtVariableReference<?> fr) {
279 		CtBinaryOperator binOp = fr.getParent(CtBinaryOperator.class);
280 		if(binOp==null) {
281 			return getCommentValue(fr);
282 		}
283 		return getLiteralValue(binOp.getRightHandOperand());
284 	}

286 	private Integer getLiteralValue(CtVariable<?> var) {
287 		CtExpression<?> exp = var.getDefaultExpression();
288 		if(exp!=null) {
289 			try {
290 				return getLiteralValue(exp);
291 			} catch (ClassCastException e) {
292 				
293 			}
294 		}
295 		if (var instanceof CtParameter) {
296 			CtParameter param = (CtParameter) var;
297 			CtExecutable<?> l_exec = param.getParent(CtExecutable.class);
298 			int l_argIdx = l_exec.getParameters().indexOf(param);
299 			assertTrue(l_argIdx>=0);
300 			if (l_exec instanceof CtLambda) {
301 				CtLambda<?> lambda = (CtLambda<?>) l_exec;
302 				CtLocalVariable<?> lamVar = (CtLocalVariable)lambda.getParent();
303 				CtLocalVariableReference<?> lamVarRef = lamVar.getParent().filterChildren((CtLocalVariableReference ref)->ref.getSimpleName().equals(lamVar.getSimpleName())).first();
304 				CtAbstractInvocation inv = lamVarRef.getParent(CtAbstractInvocation.class);
305 				return getLiteralValue((CtExpression<?>)inv.getArguments().get(l_argIdx));
306 			} else {
307 				CtExecutableReference<?> l_execRef = l_exec.getReference();
308 				List<CtAbstractInvocation<?>> list = l_exec.getFactory().Package().getRootPackage().filterChildren((CtAbstractInvocation inv)->{
309 					return inv.getExecutable().getExecutableDeclaration()==l_exec;
310 				}).list();
311 				CtAbstractInvocation inv = list.get(0);
312 				Integer firstValue = getLiteralValue((CtExpression<?>)inv.getArguments().get(l_argIdx));
313 				//check that all found method invocations are using same key
314 				list.forEach(inv2->{
315 					assertEquals(firstValue, getLiteralValue(inv2.getArguments().get(l_argIdx)));
316 				});
317 				return firstValue;
318 			}
319 		}
320 		return getCommentValue(var);
321 	}
322 
323 	private int getCommentValue(CtElement e) {
324 		while(true) {
325 			if(e==null) {
326 				return -1;
327 			}
328 			if(e.getComments().isEmpty()==false) {
329 				break;
330 			}
331 			e = e.getParent();
332 		}
333 		if(e.getComments().size()==1) {
334 			String l_c = e.getComments().get(0).getContent();
335 			return Integer.parseInt(l_c);
336 		}
337 		return -1;
338 	}
339 
340 	private Integer getLiteralValue(CtExpression<?> exp) {
341 		return ((CtLiteral<Integer>) exp).getValue();
342 	}

344 	private SourcePosition getPosition(CtElement e) {
345 		SourcePosition sp = e.getPosition();
346 		while(sp instanceof NoSourcePosition) {
347 			e = e.getParent();
348 			if(e==null) {
349 				break;
350 			}
351 			sp = e.getPosition();
352 		}
353 		return sp;
354 	}
