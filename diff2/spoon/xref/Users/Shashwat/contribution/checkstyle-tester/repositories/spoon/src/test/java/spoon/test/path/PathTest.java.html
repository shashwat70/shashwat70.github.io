76  	private void equals(CtPath path, CtElement... elements) {
77  		Collection<CtElement> result = path.evaluateOn(factory.Package().getRootPackage());
78  		assertEquals(elements.length, result.size());
79  		assertArrayEquals(elements, result.toArray(new CtElement[0]));
80  	}

82  	private void equalsSet(CtPath path, Set<? extends CtElement> elements) {
83  		Collection<CtElement> result = path.evaluateOn(factory.Package().getRootPackage());
84  		assertEquals(elements.size(), result.size());
85  		assertTrue(result.containsAll(elements));
86  	}

269 	@Test
270 	public void testRoles() {
271 		// get the then statement
272 		CtType<?> fooClass = factory.Package().get("spoon.test.path.testclasses").getType("Foo");
273 		CtPath path = new CtPathStringBuilder().fromString(".**/CtIf#else");
274 		CtMethod<Object> method = fooClass.getMethod("foo");
275 		CtStatement expected = ((CtIf) method.getBody()
276 				.getStatement(2)).getElseStatement();
277 		equals(path,
278 				expected
279 		);
280 
281 		// now we get the absolute path
282 		CtPath absPath = path.evaluateOn(factory.getModel().getRootPackage()).get(0).getPath();
283 		assertEquals("#subPackage[name=spoon]#subPackage[name=test]#subPackage[name=path]#subPackage[name=testclasses]#containedType[name=Foo]#method[signature=foo()]#body#statement[index=2]#else", absPath.toString());
284 		
285 
286 		// contract: subpath enables to have relative path
287 		CtPath subPath = absPath.relativePath(fooClass);
288 		assertEquals("#method[signature=foo()]#body#statement[index=2]#else", subPath.toString());
289 		assertSame(expected, subPath.evaluateOn(fooClass).get(0));
290 
291 		CtPath subPath2 = absPath.relativePath(method);
292 		assertEquals("#body#statement[index=2]#else", subPath2.toString());
293 		assertSame(expected, subPath2.evaluateOn(method).get(0));
294 
295 		equals(new CtPathStringBuilder().fromString(".**#else"),
296 				expected
297 		);
298 	}
