40  	@Test
41  	public void testCreateTypeRef() {
42  		Launcher launcher = new Launcher();
43  		CtTypeReference<Object> ctTypeReference = launcher.getFactory().Code().createCtTypeReference(short.class);
44  		assertEquals("short", ctTypeReference.getSimpleName());
45  		assertEquals("short", ctTypeReference.getQualifiedName());
46  
47  		ctTypeReference = launcher.getFactory().Code().createCtTypeReference(Object.class);
48  		assertEquals("Object", ctTypeReference.getSimpleName());
49  		assertEquals("java.lang.Object", ctTypeReference.getQualifiedName());
50  
51  		ctTypeReference = launcher.getFactory().Code().createCtTypeReference(null);
52  		assertNull(ctTypeReference);
53  
54  		ctTypeReference = launcher.getFactory().Code().createCtTypeReference(CtJavaDoc.CommentType.class);
55  		assertEquals("CommentType", ctTypeReference.getSimpleName());
56  		assertEquals("spoon.reflect.code.CtComment$CommentType", ctTypeReference.getQualifiedName());
57  	}
58  
59  	@Test
60  	public void reflectionAPI() {
61  		// Spoon can be used as reflection API
62  		CtType s = new TypeFactory().get(String.class);
63  		assertEquals("String", s.getSimpleName());
64  		assertEquals("java.lang.String", s.getQualifiedName());
65  		assertEquals(3, s.getSuperInterfaces().size());
66  		assertEquals(2, s.getMethodsByName("toLowerCase").size());
67  	}
68  
69  	@Test
70  	public void reflectionAPIWithTypeParameter() {
71  		// check the creation of the ComparableExpressionBase with the reflexion API
72  		CtType<Object> ctType = new Launcher().getFactory().Type().get(ComparableExpressionBase.class);
73  		assertEquals("ComparableExpressionBase", ctType.getSimpleName());
74  		CtMethod<?> method = ctType.getMethodsByName("castToNum").get(0);
75  		assertEquals("A extends java.lang.Number & java.lang.Comparable<? super A>", method.getFormalCtTypeParameters().get(0).toString());
76  	}
77  
78  	@Test
79  	public void testGetClassInAnInterface() throws Exception {
80  		final CtType<Cooking> cook = ModelUtils.buildClass(Cooking.class);
81  
82  		assertNotNull(cook.getFactory().Type().get("spoon.test.factory.testclasses3.Cooking$Tacos"));
83  		assertNotNull(cook.getFactory().Class().get("spoon.test.factory.testclasses3.Cooking$Tacos"));
84  		assertNotNull(cook.getFactory().Type().get(Cooking.Tacos.class));
85  		assertNotNull(cook.getFactory().Class().get(Cooking.Tacos.class));
86  
87  		final CtType<Prepare> prepare = ModelUtils.buildClass(Prepare.class);
88  
89  		assertNotNull(prepare.getFactory().Type().get("spoon.test.factory.testclasses3.Prepare$Tacos"));
90  		assertNotNull(prepare.getFactory().Interface().get("spoon.test.factory.testclasses3.Prepare$Tacos"));
91  		assertNotNull(prepare.getFactory().Type().get(Prepare.Pozole.class));
92  		assertNotNull(prepare.getFactory().Interface().get(Prepare.Pozole.class));
93  	}

95  	@Test
96  	public void testGetClassWithDollarAndNestedClass() {
97  		//Classes with name containing $ without being nested classes can contain nested classes
98  		Factory factory = ModelUtils.build(new File("./src/test/resources/dollar-and-nested-classes"));
99  		CtType<?> poorName = factory.Type().get("$Poor$Name");
100 		CtType<?> poorNameChoice = factory.Type().get("$Poor$Name$Choice");
101 		assertNotNull(poorName);
102 		assertNotNull(poorNameChoice);
103 		assertEquals(poorNameChoice,poorName.getMethodsByName("lookingForTroubles").get(0).getType().getTypeDeclaration());
104 	}
