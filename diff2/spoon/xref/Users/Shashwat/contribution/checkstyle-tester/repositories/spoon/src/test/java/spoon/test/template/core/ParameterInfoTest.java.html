157 		Consumer<ParameterInfo> check = (namedParam) ->
158 		{//adding value into container, which already contains a empty list, creates a new container with List which contains that value
159 			ImmutableMap empty = new ImmutableMapImpl().putValue("year", Collections.emptyList());
160 			
161 			ImmutableMap val = namedParam.addValueAs(empty, 2018);
162 			//adding same value - adds the second value again
163 			ImmutableMap val2 = namedParam.addValueAs(val, 2018);
164 			//adding null value - adds nothing. Same container is returned
165 			assertSame(val2, namedParam.addValueAs(val2, null));
166 
167 			//empty is still empty
168 			assertEquals(map().put("year", Collections.emptyList()), empty.asMap());
169 			assertNotNull(val);
170 			assertEquals(map().put("year", Arrays.asList(2018)), val.asMap());
171 			assertNotNull(val2);
172 			assertEquals(map().put("year", Arrays.asList(2018, 2018)), val2.asMap());
173 		};
174 		//contract: it behaves like this when container kind is defined as LIST
175 		check.accept(new MapParameterInfo("year").setContainerKind(ContainerKind.LIST));
176 		//contract: it behaves like this even when container kind is not defined, so it is automatically detected from the existing parameter value
177 		check.accept(new MapParameterInfo("year"));
178 		//contract: it behaves like this when ListAccessor + NamedAccessor is used
179 		check.accept(new ListParameterInfo(new MapParameterInfo("year")));
180 		//contract: it behaves like this when ListAccessor + NamedAccessor with defined container is used with 
181 		check.accept(new ListParameterInfo(new MapParameterInfo("year").setContainerKind(ContainerKind.LIST)));

402 	private Map.Entry<String, Object> entry(String key, Object value) {
403 		return new Map.Entry<String, Object>() {
404 			@Override
405 			public Object setValue(Object value) {
406 				throw new RuntimeException();
407 			}
408 			
409 			@Override
410 			public Object getValue() {
411 				return value;
412 			}
413 			
414 			@Override
415 			public String getKey() {
416 				return key;
417 			}
418 		};
419 	}
