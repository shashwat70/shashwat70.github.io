17  package spoon.test.intercession;

19  import org.junit.Test;
20  import org.junit.runner.RunWith;

22  import spoon.Launcher;

27  import spoon.support.DefaultCoreFactory;
28  import spoon.support.StandardEnvironment;
29  
30  import java.lang.reflect.Method;
31  import java.util.ArrayList;
32  import java.util.Collection;
33  import java.util.List;
34  
35  import static spoon.test.parent.ContractOnSettersParametrizedTest.createCompatibleObject;
36  
37  // contract: one can call all setters with null as parameter (no problem with parent)
38  @RunWith(Parameterized.class)
39  public class OneCanCallSetterWithNullParameterizedTest {
40  
41  	@Parameterized.Parameters(name = "{1}")
42  	public static Collection<Object[]> data() {
43  		final Launcher launcher = new Launcher();
44  		final Factory factory = launcher.getFactory();
45  		launcher.getEnvironment().setNoClasspath(true);
46  		// all metamodel interfaces.
47  		launcher.addInputResource("./src/main/java/spoon/reflect/code");
48  		launcher.addInputResource("./src/main/java/spoon/reflect/declaration");
49  		launcher.addInputResource("./src/main/java/spoon/reflect/reference");
50  		launcher.buildModel();
51  
52  		final List<Object[]> values = new ArrayList<>();
53  		new IntercessionScanner(launcher.getFactory()) {
54  			@Override
55  			protected boolean isToBeProcessed(CtMethod<?> candidate) {
56  				return (candidate.getSimpleName().startsWith("set") //
57  						|| candidate.getSimpleName().startsWith("add")) //
58  						&& takeSetterForCtElement(candidate); //
59  			}
60  
61  			@Override
62  			protected void process(CtMethod<?> element) {
63  				values.add(new Object[] { createCompatibleObject(element.getDeclaringType().getReference()), element.getReference().getActualMethod() });
64  			}
65  		}.scan(launcher.getModel().getRootPackage());
66  		return values;
67  	}
68  
69  	@Parameterized.Parameter(0)
70  	public Object instance;
71  
72  	@Parameterized.Parameter(1)
73  	public Method toTest;
74  
75  	@Test
76  	public void testContract() throws Throwable {
77  		Factory factory = new FactoryImpl(new DefaultCoreFactory(),new StandardEnvironment());
78  		Object element = instance;
79  		if (element instanceof FactoryAccessor) {
80  			((FactoryAccessor) element).setFactory(factory);
81  		}
82  		// we invoke the setter
83  		toTest.invoke(element, new Object[] { null });
84  	}
85  }
