85      public static void main(String[] args) throws Exception {
86          if (!winStr.equals(new String(winStr.getBytes("MS932"), "MS932_0213")))
87              throw new RuntimeException("MS932_0213 failed on special codepoints!");

106             throw new RuntimeException("SJIS_0213 failed on mixed!");
107 
108         if (!(Arrays.equals(mixedCompStr.getBytes("SJIS_0213"), mixedCompBytes)) ||
109             !mixedCompStr.equals(new String(mixedCompBytes, "SJIS_0213")))
110             throw new RuntimeException("SJIS_0213 failed on mixedComp!");
111 
112         if (!Arrays.equals(new String(unmappableChars).getBytes("SJIS_0213"), unmappableBytes) ||
113             !Arrays.equals(new String(unmappableChars).getBytes("MS932_0213"), unmappableBytes))
114             throw new RuntimeException("SJIS/MS932_0213 failed on unmappable encoding!");
115     }
