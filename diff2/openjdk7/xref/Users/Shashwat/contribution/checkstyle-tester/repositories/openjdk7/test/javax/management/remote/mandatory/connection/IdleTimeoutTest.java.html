55      public static void main(String[] args) throws Exception {
56          boolean ok = true;
57          List protos;
58          if (args.length > 0)
59              protos = Arrays.asList(args);
60          else {
61              protos =
62                  new ArrayList(Arrays.asList(new String[] {"rmi", "iiop"}));
63              try {
64                  Class.forName("javax.management.remote.jmxmp." +
65                                "JMXMPConnectorServer");
66                  protos.add("jmxmp");
67              } catch (ClassNotFoundException e) {
68                  // OK: Optional JMXMP support is not present
69              }
70          }
71          for (Iterator it = protos.iterator(); it.hasNext(); ) {
72              String proto = (String) it.next();
73              int liCount;
74              if (proto.equals("jmxmp")) liCount=1;
75              else liCount=2;
76              if (test(proto,4,liCount))
77                  System.out.println("Test for protocol " + proto + " passed");
78              else {
79                  System.out.println("Test for protocol " + proto + " FAILED");
80                  ok = false;
81              }
82          }
83          if (!ok) {
84              System.out.println("SOME TESTS FAILED");
85              System.exit(1);
86          }
87      }

89      private static long getIdleTimeout(MBeanServer mbs, JMXServiceURL url)
90          throws Exception {
91          JMXConnectorServer server =
92              JMXConnectorServerFactory.newJMXConnectorServer(url, null, mbs);
93          server.start();
94          try {
95              url = server.getAddress();
96  
97              // Force initialization (class loading, JIT, etc...)
98              //
99              JMXConnector client = JMXConnectorFactory.connect(url);
100             try {
101                 String connId = client.getConnectionId();
102                 MBeanServerConnection conn = client.getMBeanServerConnection();
103             } finally {
104                 client.close();
105             }
106 
107             // Do the time measurement
108             //
109             final long firstTime = System.currentTimeMillis();
110             final long endtime;
111             client = JMXConnectorFactory.connect(url);
112             try {
113                 String connId = client.getConnectionId();
114                 MBeanServerConnection conn = client.getMBeanServerConnection();
115                 endtime = System.currentTimeMillis();
116             } finally {
117                 client.close();
118             }
119 
120             // multipled by 10 for a slow machine, plus 1500 for a fast one.
121             return 10*(endtime - firstTime) + 1500;
122         } finally {
123             server.stop();
124         }
125     }

127     private static class NotificationCounter
128         implements NotificationListener {
129         private final int[]  listenerCount;
130         private final String listenerName;
131         NotificationCounter(int[] counter, String name) {
132             listenerCount=counter;
133             listenerName=name;
134         }
135 
136         public void handleNotification(Notification n,
137                                        Object h) {
138             MBeanServerNotification mbsn =
139                 (MBeanServerNotification) n;
140             System.out.println(listenerName + " got notification: "
141                                + mbsn.getMBeanName());
142             synchronized (listenerCount) {
143                 listenerCount[0]++;
144                 listenerCount.notify();
145             }
146         }
147         public String toString() {
148             return listenerName;
149         }
150     }

152     private static boolean test(String proto,int opCount,int liCount)
153         throws Exception {
154         System.out.println("Idle timeout test for protocol " + proto);
155         ObjectName delegateName =
156             ObjectName.getInstance("JMImplementation:" +
157                                    "type=MBeanServerDelegate");
158         MBeanServer mbs = MBeanServerFactory.createMBeanServer();
159         JMXServiceURL url = new JMXServiceURL("service:jmx:" + proto + "://");
160 
161         final long timeout = getIdleTimeout(mbs,url);
162         System.out.println("Timeout for " + proto + " is: " +
163                            timeout + " ms");
164 
165         Map idleMap = new HashMap();
166         idleMap.put(EnvHelp.SERVER_CONNECTION_TIMEOUT, new Long(timeout));
167         JMXConnectorServer server =
168             JMXConnectorServerFactory.newJMXConnectorServer(url,idleMap,mbs);
169 
170         final int[] listenerCount = new int[1];
171         final NotificationListener countListeners[] =
172             new NotificationListener[liCount];
173         int i;
174         for (i=0; i<countListeners.length; i++) {
175             countListeners[i] =
176                 new NotificationCounter(listenerCount,"Listener"+i);
177         }
178 
179         server.start();
180         try {
181             url = server.getAddress();
182             final long firstTime = System.currentTimeMillis();
183             JMXConnector client = JMXConnectorFactory.connect(url);
184             long elapsed, startIdle=0;
185             try {
186                 String connId = client.getConnectionId();
187                 MBeanServerConnection conn =
188                     client.getMBeanServerConnection();
189                 elapsed   = System.currentTimeMillis() - firstTime;
190                 System.out.println("Idle Time: " + elapsed + "ms");
191 
192                 for (i=0; i<countListeners.length; i++) {
193                     System.out.println("add " + countListeners[i] +
194                                        ": starting at " + elapsed + "ms");
195                     conn.addNotificationListener(delegateName,
196                                                  countListeners[i],
197                                                  null,null);
198                 }
199 
200                 System.out.println("connId=" + connId);
201                 for (i = 0; i < opCount; i++) {
202                     elapsed   = System.currentTimeMillis() - firstTime;
203                     System.out.println("Operation[" + (i+1)
204                                        +"]: starting at " +
205                                        elapsed + "ms");
206                     final String name = "d:type=mlet,instance=" + i;
207                     mbs.createMBean("javax.management.loading.MLet",
208                                     new ObjectName(name));
209                     if (i == (opCount-1))
210                         startIdle = System.currentTimeMillis();
211                     Thread.sleep(2);
212                 }
213 
214                 // Wait for notifs to arrive before doing removeNListener
215                 long startTime = System.currentTimeMillis();
216                 long deadline = startTime + 10000;
217 
218                 System.out.println("Waiting for notifs: starting at " +
219                                    (startTime - firstTime) + "ms");
220 
221                 final int expectedCount = opCount*countListeners.length;
222                 while (System.currentTimeMillis() < deadline) {
223                     synchronized (listenerCount) {
224                         if (listenerCount[0] >= expectedCount)
225                             break;
226                         listenerCount.wait();
227                     }
228                 }
229 
230                 long elapsedWait = System.currentTimeMillis() - startTime;
231                 System.out.println("Waited " + elapsedWait +
232                                    "ms for notifs to arrive");
233 
234                 if (listenerCount[0] != expectedCount) {
235                     System.out.println("Did not get expected " +
236                                        expectedCount + " notifications: "
237                                        + listenerCount[0]);
238                     return false;
239                 }
240 
241                 elapsed   = System.currentTimeMillis() - firstTime;
242 
243                 System.out.println("idle time since last operation: " +
244                                    (elapsed + firstTime - startIdle) + "ms");
245                 System.out.println("Requesting conn id at: " +
246                                    elapsed + "ms");
247                 final String cid = client.getConnectionId();
248 
249                 elapsed   = System.currentTimeMillis() - firstTime;
250                 System.out.println("Got conn id <" + cid + "> at: " +
251                                    elapsed + "ms");
252 
253                 if (!connId.equals(cid)) {
254                     System.out.println("Client id changed: <" + connId +
255                                        "> -> <" + cid +
256                                        ">");
257                     return false;
258                 }
259 
260                 List ids = Arrays.asList(server.getConnectionIds());
261                 if (!ids.contains(connId)) {
262                     System.out.println("Server ids don't contain our id: " +
263                                        ids + " - " + connId);
264                     return false;
265                 }
266 
267                 for (i=0;i<countListeners.length;i++) {
268                     System.out.println("Removing notification listener: " +
269                                        countListeners[i]);
270                     conn.removeNotificationListener(delegateName,
271                                                     countListeners[i]);
272                 }
273 
274                 System.out.println("Waiting for id list to drop ours");
275                 deadline = System.currentTimeMillis() + timeout*2 + 10000;
276                 while (true) {
277                     ids = Arrays.asList(server.getConnectionIds());
278                     if (!ids.contains(connId)
279                         || System.currentTimeMillis() >= deadline)
280                         break;
281                     Thread.sleep(500);
282                 }
283                 if (ids.contains(connId)) {
284                     System.out.println("Client id still in list after " +
285                                        "deadline: " + ids);
286                     return false;
287                 }
288 
289                 conn.getDefaultDomain();
290                 if (connId.equals(client.getConnectionId())) {
291                     System.out.println("Client id did not change: <" + connId +
292                                        ">: idle timeout did not happen?");
293                     return false;
294                 } else {
295                     System.out.println("Client id changed as expected: <" +
296                                        connId + "> -> <" +
297                                        client.getConnectionId() + ">");
298                 }
299             } finally {
300                 client.close();
301                 System.out.println("Connection id list on server after " +
302                                    "client close: " +
303                                    Arrays.asList(server.getConnectionIds()));
304             }
305         } finally {
306             server.stop();
307         }
308         System.out.println("*** ------------------------------------------");
309         System.out.println("*** Test passed for " + proto);
310         System.out.println("*** ------------------------------------------");
311         return true;
312     }
