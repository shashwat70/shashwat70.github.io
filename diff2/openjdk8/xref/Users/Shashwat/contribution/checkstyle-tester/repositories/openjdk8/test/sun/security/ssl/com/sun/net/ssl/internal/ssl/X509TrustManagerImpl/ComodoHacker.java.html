255     private static X509TrustManager getTrustManager() throws Exception {
256         // generate certificate from cert string
257         CertificateFactory cf = CertificateFactory.getInstance("X.509");
258 
259         // create a key store
260         KeyStore ks = KeyStore.getInstance("JKS");
261         ks.load(null, null);
262 
263         // import the trusted cert
264         try (ByteArrayInputStream is =
265                 new ByteArrayInputStream(trustedCertStr.getBytes())) {
266             Certificate trustedCert = cf.generateCertificate(is);
267             ks.setCertificateEntry("RSA Export Signer", trustedCert);
268         }
269 
270         // create the trust manager
271         TrustManagerFactory tmf = TrustManagerFactory.getInstance(tmAlgorithm);
272         tmf.init(ks);
273 
274         return (X509TrustManager)tmf.getTrustManagers()[0];
275     }
