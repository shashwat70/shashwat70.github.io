25  package sun.awt.windows;
26  
27  import java.awt.*;
28  import java.awt.peer.*;
29  
30  final class WLabelPeer extends WComponentPeer implements LabelPeer {
31  
32      // ComponentPeer overrides
33  
34      public Dimension getMinimumSize() {
35          FontMetrics fm = getFontMetrics(((Label)target).getFont());
36          String label = ((Label)target).getText();
37          if (label == null)
38              label = "";
39          return new Dimension(fm.stringWidth(label) + 14, fm.getHeight() + 8);
40      }
41  
42      native void lazyPaint();
43      synchronized void start() {
44          super.start();
45          // if need then paint label
46          lazyPaint();
47      }
48      // LabelPeer implementation
49  
50      public boolean shouldClearRectBeforePaint() {
51          return false;
52      }
53  
54      public native void setText(String label);
55      public native void setAlignment(int alignment);
56  
57      // Toolkit & peer internals
58  
59      WLabelPeer(Label target) {
60          super(target);
61      }
62  
63      native void create(WComponentPeer parent);
64  
65      void initialize() {
66          Label   l = (Label)target;
67  
68          String  txt = l.getText();
69          if (txt != null) {
70              setText(txt);
71          }
72  
73          int align = l.getAlignment();
74          if (align != Label.LEFT) {
75              setAlignment(align);
76          }
77  
78          Color bg = ((Component)target).getBackground();
79          if (bg != null) {
80              setBackground(bg);
81          }
82  
83          super.initialize();
84      }
85  }
