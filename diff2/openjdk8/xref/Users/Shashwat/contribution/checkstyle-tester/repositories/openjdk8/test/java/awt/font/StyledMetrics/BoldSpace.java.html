24  /*
25    @test
26    @bug 6686365 7017637
27    @summary Confirm that styling does not affect metrics of zero advance glyphs
28  */
29  
30  import java.awt.*;
31  import java.awt.image.*;
32  
33  public class BoldSpace {
34      public static void main(String[] s) {
35          BufferedImage bi = new BufferedImage(100, 100, BufferedImage.TYPE_INT_ARGB);
36  
37          String errMsg = "ZWJ Space char should have 0 advance\n";
38  
39          Graphics g = bi.getGraphics();
40  
41          // It turns out that some fonts inexplicably treat this as
42          // a standard character. In this 14 pt font, if we see an advance
43          // that's clearly bigger than we'd have introduced in bolding we'll
44          // not error out this test, presuming that its a consequence of
45          // the actual font data. A Linux font 'TLwg Type Bold' is the case
46          // in point.
47          int errorMargin = 4;
48          g.setFont(new Font("monospaced", Font.BOLD, 14));
49          //g.setFont(new Font("Lucida Sans Regular", Font.BOLD, 14));
50          FontMetrics fm = g.getFontMetrics();
51          System.out.println("Bold: " + fm.charWidth('\u200b'));
52          int cwid = fm.charWidth('\u200b');
53          if (cwid > 0 && cwid < errorMargin) {
54              throw new RuntimeException(errMsg);
55          }
56  
57          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
58                                   RenderingHints.VALUE_TEXT_ANTIALIAS_LCD_HRGB);
59          fm = g.getFontMetrics();
60          System.out.println("Bold + LCD: "+fm.charWidth('\u200b'));
61          cwid = fm.charWidth('\u200b');
62          if (cwid > 0 && cwid < errorMargin) {
63              throw new RuntimeException(errMsg);
64          }
65  
66  
67          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
68                                   RenderingHints.VALUE_FRACTIONALMETRICS_ON);
69          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
70                                   RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
71          fm = g.getFontMetrics();
72          System.out.println("Bold FM OFF + AA: " + fm.charWidth('\u200b'));
73          cwid = fm.charWidth('\u200b');
74          if (cwid > 0 && cwid < errorMargin) {
75              throw new RuntimeException(errMsg);
76          }
77  
78          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
79                                   RenderingHints.VALUE_FRACTIONALMETRICS_OFF);
80          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
81                                   RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
82          fm = g.getFontMetrics();
83          System.out.println("Bold FM ON + AA: " + fm.charWidth('\u200b'));
84          cwid = fm.charWidth('\u200b');
85          if (cwid > 0 && cwid < errorMargin) {
86              throw new RuntimeException(errMsg);
87          }
88  
89          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
90                                   RenderingHints.VALUE_FRACTIONALMETRICS_ON);
91          ((Graphics2D)g).setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
92                                   RenderingHints.VALUE_TEXT_ANTIALIAS_OFF);
93          fm = g.getFontMetrics();
94          System.out.println("Bold FM ON + nonAA: " + fm.charWidth('\u200b'));
95          cwid = fm.charWidth('\u200b');
96          if (cwid > 0 && cwid < errorMargin) {
97              throw new RuntimeException(errMsg);
98          }
99  
100         System.out.println("All printed values should be 0 to PASS");
101     }
102 }
