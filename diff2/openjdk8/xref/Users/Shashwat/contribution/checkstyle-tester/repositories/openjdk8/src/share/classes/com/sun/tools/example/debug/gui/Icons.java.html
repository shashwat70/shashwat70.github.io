201     static private byte[] wordsToBytes(int[] wordArray) {
202         byte[] bytes = new byte[wordArray.length * 4];
203         int inx = bytes.length;
204         for (int i = wordArray.length-1; i >= 0; --i) {
205             int word = wordArray[i];
206             for (int j = 0; j < 4; ++j) {
207                 bytes[--inx] = (byte)(word & 0xff);
208                 word = word >>> 8;
209             }
210         }
211         return bytes;
212     }
